1use candle_core::{DType, Device, Result, Tensor};
2use image::{DynamicImage, GenericImageView};

4/// Output is (c, h, w)
5pub(crate) fn image_to_pixels(image: &DynamicImage, device: &Device) -> Result<Tensor> {
6    let (w, h) = image.dimensions();
7    let n_channels = n_channels(image);
8    let data = match n_channels {
9        1 => image.to_luma8().into_raw(),
10        2 => image.to_luma_alpha8().into_raw(),
11        3 => image.to_rgb8().into_raw(),
12        4 => image.to_rgba8().into_raw(),
13        _ => candle_core::bail!("Unsupported channel count {n_channels}"),
14    };
15    let data = Tensor::from_vec(data, (h as usize, w as usize, n_channels), device)?;
16    data.permute((2, 0, 1))?.to_dtype(DType::F32)
17}

19pub(crate) fn n_channels(image: &DynamicImage) -> usize {
