350    pub async fn build(self) -> anyhow::Result<Model> {
351        let config = NormalSpecificConfig {
352            prompt_chunksize: self.prompt_chunksize,
353            topology: self.topology,
354            organization: self.organization,
355            write_uqff: self.write_uqff,
356            from_uqff: self.from_uqff,
357            imatrix: self.imatrix,
358            calibration_file: self.calibration_file,
359            hf_cache_path: self.hf_cache_path,
360        };
361
362        if self.with_logging {
363            initialize_logging();
364        }
365
366        let loader = NormalLoaderBuilder::new(
367            config,
368            self.chat_template,
369            self.tokenizer_json,
370            Some(self.model_id),
371            self.no_kv_cache,
372            self.jinja_explicit,
373        )
374        .build(self.loader_type)?;
375
376        // Load, into a Pipeline
377        let pipeline = loader.load_model_from_hf(
378            self.hf_revision,
379            self.token_source,
380            &self.dtype,
381            &self.device.unwrap_or(best_device(self.force_cpu).unwrap()),
382            !self.with_logging,
383            self.device_mapping
384                .unwrap_or(DeviceMapSetting::Auto(AutoDeviceMapParams::default_text())),
385            self.isq,
386            self.paged_attn_cfg,
387        )?;
388
389        let scheduler_method = match self.paged_attn_cfg {
390            Some(_) => {
391                let config = pipeline
392                    .lock()
393                    .await
394                    .get_metadata()
395                    .cache_config
396                    .as_ref()
397                    .cloned();
398
399                if let Some(config) = config {
400                    SchedulerConfig::PagedAttentionMeta {
401                        max_num_seqs: self.max_num_seqs,
402                        config,
403                    }
404                } else {
405                    SchedulerConfig::DefaultScheduler {
406                        method: DefaultSchedulerMethod::Fixed(self.max_num_seqs.try_into()?),
407                    }
408                }
409            }
410            None => SchedulerConfig::DefaultScheduler {
411                method: DefaultSchedulerMethod::Fixed(self.max_num_seqs.try_into()?),
412            },
413        };
414
415        let mut runner = MistralRsBuilder::new(
416            pipeline,
417            scheduler_method,
418            self.throughput_logging,
419            self.search_bert_model,
420        );
421        if let Some(cb) = self.search_callback.clone() {
422            runner = runner.with_search_callback(cb);
423        }
424        for (name, cb) in &self.tool_callbacks {
425            runner = runner.with_tool_callback(name.clone(), cb.clone());
426        }
427        for (name, callback_with_tool) in &self.tool_callbacks_with_tools {
428            runner = runner.with_tool_callback_and_tool(
429                name.clone(),
430                callback_with_tool.callback.clone(),
431                callback_with_tool.tool.clone(),
432            );
433        }
434        if let Some(mcp_config) = self.mcp_client_config {
435            runner = runner.with_mcp_client(mcp_config);
436        }
437        runner = runner
438            .with_no_kv_cache(self.no_kv_cache)
439            .with_no_prefix_cache(self.prefix_cache_n.is_none());
440
441        if let Some(n) = self.prefix_cache_n {
442            runner = runner.with_prefix_cache_n(n)
443        }
444
445        Ok(Model::new(runner.build().await))
446    }

450/// Configure a UQFF text model with the various parameters for loading, running, and other inference behaviors.
451/// This wraps and implements `DerefMut` for the TextModelBuilder, so users should take care to not call UQFF-related methods.
452pub struct UqffTextModelBuilder(TextModelBuilder);
453
454impl UqffTextModelBuilder {
455    /// A few defaults are applied here:
456    /// - MoQE ISQ organization
457    /// - Token source is from the cache (.cache/huggingface/token)
458    /// - Maximum number of sequences running is 32
459    /// - Number of sequences to hold in prefix cache is 16.
460    /// - Automatic device mapping with model defaults according to `AutoDeviceMapParams`
461    pub fn new(model_id: impl ToString, uqff_file: Vec<PathBuf>) -> Self {
462        let mut inner = TextModelBuilder::new(model_id);
463        inner = inner.from_uqff(uqff_file);
464        Self(inner)
465    }
466
467    pub async fn build(self) -> anyhow::Result<Model> {
468        self.0.build().await
469    }
470
471    /// This wraps the VisionModelBuilder, so users should take care to not call UQFF-related methods.
472    pub fn into_inner(self) -> TextModelBuilder {
473        self.0
474    }
475}
