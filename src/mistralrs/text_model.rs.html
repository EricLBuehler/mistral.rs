388/// Configure a UQFF text model with the various parameters for loading, running, and other inference behaviors.
389/// This wraps and implements `DerefMut` for the TextModelBuilder, so users should take care to not call UQFF-related methods.
390pub struct UqffTextModelBuilder(TextModelBuilder);
391
392impl UqffTextModelBuilder {
393    /// A few defaults are applied here:
394    /// - MoQE ISQ organization
395    /// - Token source is from the cache (.cache/huggingface/token)
396    /// - Maximum number of sequences running is 32
397    /// - Number of sequences to hold in prefix cache is 16.
398    /// - Automatic device mapping with model defaults according to `AutoDeviceMapParams`
399    pub fn new(model_id: impl ToString, uqff_file: Vec<PathBuf>) -> Self {
400        let mut inner = TextModelBuilder::new(model_id);
401        inner.from_uqff = Some(uqff_file);
402        Self(inner)
403    }
404
405    pub async fn build(self) -> anyhow::Result<Model> {
406        self.0.build().await
407    }
408
409    /// This wraps the VisionModelBuilder, so users should take care to not call UQFF-related methods.
410    pub fn into_inner(self) -> TextModelBuilder {
411        self.0
412    }
413}
