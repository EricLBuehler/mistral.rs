75impl SearchResult {
76    pub fn cap_content_len(self, tokenizer: &Tokenizer, size: usize) -> Result<Self> {
77        let tokenized_content = tokenizer
78            .encode_fast(self.content, false)
79            .map_err(anyhow::Error::msg)?;
80        let ids = tokenized_content.get_ids();
81        let content = tokenizer
82            .decode(&ids[..size.min(ids.len())], false)
83            .map_err(anyhow::Error::msg)?;
84
85        Ok(Self {
86            title: self.title,
87            description: self.description,
88            url: self.url,
89            content,
90        })
91    }
92}

121pub fn get_search_tools(web_search_options: &WebSearchOptions) -> Result<Vec<Tool>> {
122    let search_tool = {
123        let parameters: HashMap<String, Value> = serde_json::from_value(json!({
124            "type": "object",
125            "properties": {
126                "query": {
127                    "type": "string",
128                    "description": "A query for web searching.",
129                },
130            },
131            "required": ["query"],
132        }))?;
133
134        let location_details = match &web_search_options.user_location {
135            Some(WebSearchUserLocation::Approximate { approximate }) => {
136                format!(
137                    "\nThe user's location is: {}, {}, {}, {}.",
138                    approximate.city, approximate.region, approximate.country, approximate.timezone
139                )
140            }
141            None => "".to_string(),
142        };
143
144        Tool {
145            tp: ToolType::Function,
146            function: Function {
147                description: Some(format!("{SEARCH_DESCRIPTION}{location_details}")),
148                name: SEARCH_TOOL_NAME.to_string(),
149                parameters: Some(parameters),
150            },
151        }
152    };
153
154    let extract_tool = {
155        let parameters: HashMap<String, Value> = serde_json::from_value(json!({
156            "type": "object",
157            "properties": {
158                "url": {
159                    "type": "string",
160                    "description": "A URL to extract the content of the website from.",
161                },
162            },
163            "required": ["url"],
164        }))?;
165
166        Tool {
167            tp: ToolType::Function,
168            function: Function {
169                description: Some(EXTRACT_DESCRIPTION.to_string()),
170                name: EXTRACT_TOOL_NAME.to_string(),
171                parameters: Some(parameters),
172            },
173        }
174    };
175
176    Ok(vec![search_tool, extract_tool])
177}

179pub fn run_search_tool(params: &SearchFunctionParameters) -> Result<Vec<SearchResult>> {
180    let client = reqwest::blocking::Client::new();
181
182    let encoded_query = urlencoding::encode(&params.query);
183    let url = format!("https://html.duckduckgo.com/html/?q={}", encoded_query);
184
185    let user_agent = format!("mistralrs/{APP_VERSION} ({OS}; {ARCH}; {FAMILY})");
186    let response = client.get(&url).header("User-Agent", &user_agent).send()?;
187
188    // Check the response status
189    if !response.status().is_success() {
190        anyhow::bail!("Failed to fetch search results: {}", response.status())
191    }
192
193    let html = response.text()?;
194
195    let document = Html::parse_document(&html);
196
197    let result_selector = Selector::parse(".result").unwrap();
198    let title_selector = Selector::parse(".result__title").unwrap();
199    let snippet_selector = Selector::parse(".result__snippet").unwrap();
200    let url_selector = Selector::parse(".result__url").unwrap();
201
202    // Phase 1: collect title, description, and url serially into a Vec of tuples
203    let partials: Vec<(String, String, String)> = document
204        .select(&result_selector)
205        .filter_map(|element| {
206            let title = element
207                .select(&title_selector)
208                .next()
209                .map(|e| e.text().collect::<String>().trim().to_string())
210                .unwrap_or_default();
211            let description = element
212                .select(&snippet_selector)
213                .next()
214                .map(|e| e.text().collect::<String>().trim().to_string())
215                .unwrap_or_default();
216            let mut url = element
217                .select(&url_selector)
218                .next()
219                .map(|e| e.text().collect::<String>().trim().to_string())
220                .unwrap_or_default();
221            if title.is_empty() || description.is_empty() || url.is_empty() {
222                return None;
223            }
224            if !url.starts_with("http") {
225                url = format!("https://{}", url);
226            }
227            Some((title, description, url))
228        })
229        .collect();
230
231    // Phase 2: fetch content in parallel using Rayon
232    let client = Arc::new(client);
233    let results: Vec<SearchResult> = partials
234        .into_par_iter()
235        .filter_map(|(title, description, url)| {
236            let content = match client.get(&url).header("User-Agent", &user_agent).send() {
237                Ok(response) => {
238                    let html = response.text().ok()?;
239                    config::with_decorator(PlainDecorator::new())
240                        .do_decorate()
241                        .string_from_read(html.as_bytes(), 80)
242                        .ok()?
243                }
244                Err(_) => return None,
245            };
246            Some(SearchResult {
247                title,
248                description,
249                url,
250                content,
251            })
252        })
253        .collect();
254
255    Ok(results)
256}

258pub fn run_extract_tool(params: &ExtractFunctionParameters) -> Result<ExtractResult> {
259    let client = reqwest::blocking::Client::new();
260
261    let user_agent = format!("mistralrs/{APP_VERSION} ({OS}; {ARCH}; {FAMILY})");
262
263    let content = match client
264        .get(&params.url)
265        .header("User-Agent", &user_agent)
266        .send()
267    {
268        Ok(response) => response.text().ok().and_then(|html| {
269            config::with_decorator(PlainDecorator::new())
270                .do_decorate()
271                .string_from_read(html.as_bytes(), 80)
272                .ok()
273        }),
274        Err(_) => None,
275    };
276    Ok(ExtractResult {
277        url: params.url.clone(),
278        content: content.unwrap_or("ERROR: failed to extract content".to_string()),
279    })
