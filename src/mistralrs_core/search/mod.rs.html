47impl SearchResult {
48    pub fn cap_content_len(self, tokenizer: &Tokenizer, size: usize) -> Result<Self> {
49        let tokenized_content = tokenizer
50            .encode_fast(self.content, false)
51            .map_err(anyhow::Error::msg)?;
52        let ids = tokenized_content.get_ids();
53        let content = tokenizer
54            .decode(&ids[..size.min(ids.len())], false)
55            .map_err(anyhow::Error::msg)?;
56
57        Ok(Self {
58            title: self.title,
59            description: self.description,
60            url: self.url,
61            content,
62        })
63    }
64}

71pub fn get_search_tool(web_search_options: &WebSearchOptions) -> Result<Tool> {
72    let parameters: HashMap<String, Value> = serde_json::from_value(json!({
73        "type": "object",
74        "properties": {
75            "query": {
76                "type": "string",
77                "description": "A query for web searching. This can also be a URL.",
78            },
79        },
80        "required": ["query"],
81    }))?;
82
83    let location_details = match &web_search_options.user_location {
84        Some(WebSearchUserLocation::Approximate { approximate }) => {
85            format!(
86                "\nThe user's location is: {}, {}, {}, {}.",
87                approximate.city, approximate.region, approximate.country, approximate.timezone
88            )
89        }
90        None => "".to_string(),
91    };
92
93    Ok(Tool {
94        tp: ToolType::Function,
95        function: Function {
96            description: Some(format!("{DESCRIPTION}{location_details}")),
97            name: SEARCH_TOOL_NAME.to_string(),
98            parameters: Some(parameters),
99        },
100    })
101}

103pub fn run_search_tool(params: &SearchFunctionParameters) -> Result<Vec<SearchResult>> {
104    let client = reqwest::blocking::Client::new();
105
106    let encoded_query = urlencoding::encode(&params.query);
107    let url = format!("https://html.duckduckgo.com/html/?q={}", encoded_query);
108
109    let user_agent = format!("mistralrs/{APP_VERSION} ({OS}; {ARCH}; {FAMILY})");
110    let response = client.get(&url).header("User-Agent", &user_agent).send()?;
111
112    // Check the response status
113    if !response.status().is_success() {
114        anyhow::bail!("Failed to fetch search results: {}", response.status())
115    }
116
117    let html = response.text()?;
118
119    let document = Html::parse_document(&html);
120
121    let result_selector = Selector::parse(".result").unwrap();
122    let title_selector = Selector::parse(".result__title").unwrap();
123    let snippet_selector = Selector::parse(".result__snippet").unwrap();
124    let url_selector = Selector::parse(".result__url").unwrap();
125
126    // Phase 1: collect title, description, and url serially into a Vec of tuples
127    let partials: Vec<(String, String, String)> = document
128        .select(&result_selector)
129        .filter_map(|element| {
130            let title = element
131                .select(&title_selector)
132                .next()
133                .map(|e| e.text().collect::<String>().trim().to_string())
134                .unwrap_or_default();
135            let description = element
136                .select(&snippet_selector)
137                .next()
138                .map(|e| e.text().collect::<String>().trim().to_string())
139                .unwrap_or_default();
140            let mut url = element
141                .select(&url_selector)
142                .next()
143                .map(|e| e.text().collect::<String>().trim().to_string())
144                .unwrap_or_default();
145            if title.is_empty() || description.is_empty() || url.is_empty() {
146                return None;
147            }
148            if !url.starts_with("http") {
149                url = format!("https://{}", url);
150            }
151            Some((title, description, url))
152        })
153        .collect();
154
155    // Phase 2: fetch content in parallel using Rayon
156    let client = Arc::new(client);
157    let results: Vec<SearchResult> = partials
158        .into_par_iter()
159        .filter_map(|(title, description, url)| {
160            let content = match client.get(&url).header("User-Agent", &user_agent).send() {
161                Ok(response) => {
162                    let html = response.text().ok()?;
163                    config::with_decorator(PlainDecorator::new())
164                        .do_decorate()
165                        .string_from_read(html.as_bytes(), 80)
166                        .ok()?
167                }
168                Err(_) => return None,
169            };
170            Some(SearchResult {
171                title,
172                description,
173                url,
174                content,
175            })
176        })
177        .collect();
178
179    Ok(results)
180}
