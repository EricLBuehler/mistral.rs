46impl SearchPipeline {
47    pub fn new(model: SearchEmbeddingModel, runner_device: &Device) -> anyhow::Result<Self> {
48        let model_id = model.hf_model_id().to_string();
49
50        once_log_info(format!("Loading embedding model ({model_id})."));
51
52        let loader = EmbeddingLoaderBuilder::new(
53            EmbeddingSpecificConfig::default(),
54            None,
55            Some(model_id.clone()),
56        )
57        .build(None);
58
59        let pipeline = loader.load_model_from_hf(
60            None,
61            TokenSource::CacheToken,
62            &ModelDType::Auto,
63            runner_device,
64            true,
65            DeviceMapSetting::Auto(AutoDeviceMapParams::default_text()),
66            None,
67            None,
68        )?;
69
70        let guard = get_mut_arcmutex!(pipeline);
71        let tokenizer = guard
72            .tokenizer()
73            .with_context(|| "Embedding model did not expose a tokenizer")?
74            .clone();
75        let device = guard.device();
76        let max_seq_len = guard.get_metadata().max_seq_len;
77        drop(guard);
78
79        Ok(Self {
80            model: pipeline,
81            tokenizer,
82            device,
83            has_causal_attention: false,
84            max_seq_len,
85        })
86    }
87
88    fn embed(&mut self, prompts: &[String]) -> anyhow::Result<Vec<Vec<f32>>> {
89        if prompts.is_empty() {
90            return Ok(Vec::new());
91        }
92
93        use std::collections::BTreeMap;
94        let mut by_len: BTreeMap<usize, Vec<(usize, Vec<u32>)>> = BTreeMap::new();
95        for (idx, prompt) in prompts.iter().enumerate() {
96            let encoding = self
97                .tokenizer
98                .encode(prompt.as_str(), true)
99                .map_err(E::msg)?;
100            let ids = encoding.get_ids().to_vec();
101            by_len.entry(ids.len()).or_default().push((idx, ids));
102        }
103
104        let mut outputs = vec![Vec::new(); prompts.len()];
105        for (_, sequences) in by_len {
106            for chunk_entries in sequences.chunks(EMBEDDING_BATCH) {
107                let slices: Vec<&[u32]> = chunk_entries
108                    .iter()
109                    .map(|(_, ids)| ids.as_slice())
110                    .collect();
111                let chunk =
112                    make_prompt_chunk(0, slices, &self.device, None, self.has_causal_attention)?;
113                let inputs = Box::new(ModelInputs {
114                    input_ids: chunk.input,
115                    flash_meta: chunk.flash_meta,
116                });
117                let mut pipeline = get_mut_arcmutex!(self.model);
118                let ForwardInputsResult::Embeddings { embeddings } =
119                    pipeline.forward_inputs(inputs, false)?
120                else {
121                    anyhow::bail!("Embedding pipeline returned non-embedding output");
122                };
123                drop(pipeline);
124                let vecs = embeddings
125                    .to_dtype(DType::F32)?
126                    .to_device(&Device::Cpu)?
127                    .to_vec2::<f32>()?;
128                for ((idx, _), embedding) in chunk_entries.iter().zip(vecs.into_iter()) {
129                    outputs[*idx] = embedding;
130                }
131            }
132        }
133
134        Ok(outputs)
135    }
136}

138impl SearchPipeline {
139    fn chunk_document(&self, sanitized_title: &str, text: &str) -> Result<Vec<DocumentChunk>> {
140        let trimmed = text.trim();
141        if trimmed.is_empty() {
142            return Ok(Vec::new());
143        }
144
145        let prefix_prompt = format_document_prompt(sanitized_title, "");
146        let prefix_tokens = self
147            .tokenizer
148            .encode(prefix_prompt.as_str(), true)
149            .map_err(E::msg)?
150            .len();
151        if self.max_seq_len <= prefix_tokens {
152            return Ok(vec![DocumentChunk {
153                prompt: format_document_prompt(sanitized_title, trimmed),
154                content: trimmed.to_string(),
155                token_len: self.tokenizer.encode(trimmed, false).map_err(E::msg)?.len(),
156            }]);
157        }
158
159        let budget = (self.max_seq_len - prefix_tokens).max(1);
160        let encoding = self.tokenizer.encode(trimmed, false).map_err(E::msg)?;
161        let token_count = encoding.len();
162        if token_count == 0 {
163            return Ok(Vec::new());
164        }
165
166        let offsets = encoding.get_offsets();
167        let mut chunks = Vec::new();
168        let mut start_idx = 0;
169
170        while start_idx < token_count {
171            let mut end_idx = (start_idx + budget).min(token_count);
172            let mut accepted: Option<(DocumentChunk, usize)> = None;
173
174            while end_idx > start_idx {
175                let range = offsets[start_idx].0..offsets[end_idx - 1].1;
176                let chunk_slice = trimmed[range.clone()].trim();
177                if chunk_slice.is_empty() {
178                    end_idx -= 1;
179                    continue;
180                }
181
182                let candidate = format_document_prompt(sanitized_title, chunk_slice);
183                let candidate_len = self
184                    .tokenizer
185                    .encode(candidate.as_str(), true)
186                    .map_err(E::msg)?
187                    .len();
188
189                if candidate_len <= self.max_seq_len {
190                    let token_len = end_idx - start_idx;
191                    accepted = Some((
192                        DocumentChunk {
193                            prompt: candidate,
194                            content: chunk_slice.to_string(),
195                            token_len,
196                        },
197                        end_idx,
198                    ));
199                    break;
200                }
201                end_idx -= 1;
202            }
203
204            if let Some((chunk, next_idx)) = accepted {
205                chunks.push(chunk);
206                start_idx = next_idx;
207                continue;
208            }
209
210            let single_range = offsets[start_idx].0..offsets[start_idx].1;
211            let chunk_slice = trimmed[single_range].trim();
212            if !chunk_slice.is_empty() {
213                chunks.push(DocumentChunk {
214                    prompt: format_document_prompt(sanitized_title, chunk_slice),
215                    content: chunk_slice.to_string(),
216                    token_len: 1,
217                });
218            }
219            start_idx += 1;
220        }
221
222        Ok(chunks)
223    }
224}

226fn sanitize_title(title: &str) -> String {
227    let trimmed = title.trim();
228    if trimmed.is_empty() {
229        "none".to_string()
230    } else {
231        trimmed.to_string()
232    }

243fn cosine_similarity(a: &[f32], b: &[f32]) -> f32 {
244    let dot = a.iter().zip(b).map(|(x, y)| x * y).sum::<f32>();
245    let norm_a = a.iter().map(|x| x * x).sum::<f32>().sqrt();
246    let norm_b = b.iter().map(|x| x * x).sum::<f32>().sqrt();
247    if norm_a == 0.0 || norm_b == 0.0 {
248        0.0
249    } else {
250        dot / (norm_a * norm_b)
251    }
252}

254pub fn rank_document_chunks(
255    query: &str,
256    results: &[SearchResult],
257    pipeline: &mut SearchPipeline,
258) -> Result<Vec<ScoredChunk>> {
259    if results.is_empty() {
260        return Ok(Vec::new());
261    }
262
263    let query_embedding = pipeline
264        .embed(&[format_query_prompt(query)])?
265        .into_iter()
266        .next()
267        .context("Failed to generate embedding for search query")?;
268
269    let mut bindings: Vec<(usize, DocumentChunk)> = Vec::new();
270    for (result_index, result) in results.iter().enumerate() {
271        let title = sanitize_title(&result.title);
272        bindings.extend(
273            pipeline
274                .chunk_document(&title, &result.content)?
275                .into_iter()
276                .map(|chunk| (result_index, chunk)),
277        );
278    }
279
280    if bindings.is_empty() {
281        return Ok(Vec::new());
282    }
283
284    let prompts: Vec<String> = bindings
285        .iter()
286        .map(|(_, chunk)| chunk.prompt.clone())
287        .collect();
288
289    let chunk_embeddings = pipeline.embed(&prompts)?;
290    let mut scored = Vec::with_capacity(bindings.len());
291    for ((result_index, chunk), embedding) in bindings.into_iter().zip(chunk_embeddings.into_iter())
292    {
293        let score = cosine_similarity(&query_embedding, &embedding);
294        scored.push(ScoredChunk {
295            result_index,
296            content: chunk.content,
297            token_len: chunk.token_len,
298            score,
299        });
300    }
301
302    scored.sort_by(|a, b| b.score.partial_cmp(&a.score).unwrap_or(Ordering::Less));
303    Ok(scored)
304}
