14fn normalize_l2(v: &Tensor) -> Result<Tensor> {
15    Ok(v.broadcast_div(&v.sqr()?.sum_keepdim(1)?.sqrt()?)?)
16}

18/// Get the indexes of requests most similar to the query. In decreasing order
19pub fn compute_most_similar(
20    device: &Device,
21    query: &str,
22    results: Vec<&SearchResult>,
23    BertPipeline { model, tokenizer }: &mut BertPipeline,
24) -> Result<Vec<usize>> {
25    let normalize_embeddings = false;
26
27    tokenizer.with_padding(Some(PaddingParams {
28        strategy: PaddingStrategy::BatchLongest,
29        ..Default::default()
30    }));
31
32    let mut mean_similarities = Vec::new();
33    for result in results {
34        let mean_content_similarity = {
35            let content = &result.content;
36            let chunks = content
37                .chars()
38                .chunks(4096)
39                .into_iter()
40                .map(|chunk| chunk.collect::<String>())
41                .collect::<Vec<_>>();
42            let sentences = [vec![query.to_string()], chunks].concat();
43            #[cfg(feature = "metal")]
44            let similarities = objc::rc::autoreleasepool(|| -> Result<Vec<f32>> {
45                compute_similarities(model, tokenizer, device, sentences, normalize_embeddings)
46            })?;
47            #[cfg(not(feature = "metal"))]
48            let similarities =
49                compute_similarities(model, tokenizer, device, sentences, normalize_embeddings)?;
50            similarities.iter().sum::<f32>() / similarities.len() as f32
51        };
52
53        let title_similarity = {
54            let title = &result.title;
55            let sentences = vec![query.to_string(), title.to_string()];
56            #[cfg(feature = "metal")]
57            let similarities = objc::rc::autoreleasepool(|| -> Result<Vec<f32>> {
58                compute_similarities(model, tokenizer, device, sentences, normalize_embeddings)
59            })?;
60            #[cfg(not(feature = "metal"))]
61            let similarities =
62                compute_similarities(model, tokenizer, device, sentences, normalize_embeddings)?;
63            similarities.iter().sum::<f32>() / similarities.len() as f32
64        };
65        mean_similarities.push(title_similarity * 2. + mean_content_similarity);
66    }
67
68    let mut indexed: Vec<(usize, f32)> = mean_similarities.iter().cloned().enumerate().collect();
69    indexed.sort_by(|a, b| b.1.partial_cmp(&a.1).unwrap_or(Ordering::Less));
70    let ordered_indexes: Vec<usize> = indexed.into_iter().map(|(i, _)| i).collect();
71
72    Ok(ordered_indexes)
73}

75fn compute_similarities(
76    model: &BertModel,
77    tokenizer: &Tokenizer,
78    device: &Device,
79    sentences: Vec<String>,
80    normalize_embeddings: bool,
81) -> Result<Vec<f32>> {
82    let n_sentences = sentences.len();
83    let sentences_batched = sentences
84        .iter()
85        .map(|s| InputSequence::Raw(Cow::from(s)))
86        .collect::<Vec<_>>();
87
88    let tokens = tokenizer
89        .encode_batch(sentences_batched.to_vec(), true)
90        .map_err(E::msg)?;
91    let mut embeddings_all = Vec::new();
92    for tokens in tokens.chunks(2) {
93        let token_ids = tokens
94            .iter()
95            .map(|tokens| {
96                let tokens = tokens.get_ids().to_vec();
97                Ok(Tensor::new(tokens.as_slice(), device)?)
98            })
99            .collect::<Result<Vec<_>>>()?;
100        let attention_mask = tokens
101            .iter()
102            .map(|tokens| {
103                let tokens = tokens.get_attention_mask().to_vec();
104                Ok(Tensor::new(tokens.as_slice(), device)?)
105            })
106            .collect::<Result<Vec<_>>>()?;
107
108        let token_ids = Tensor::stack(&token_ids, 0)?;
109        let attention_mask = Tensor::stack(&attention_mask, 0)?;
110        let token_type_ids = token_ids.zeros_like()?;
111
112        let embeddings = model
113            .forward(&token_ids, &token_type_ids, Some(&attention_mask))?
114            .to_dtype(DType::F32)?;
115        embeddings_all.push(embeddings)
116    }
117    let embeddings = Tensor::cat(&embeddings_all, 0)?;
118    drop(embeddings_all);
119
120    // Apply some avg-pooling by taking the mean embedding value for all tokens (including padding)
121    let (_n_sentence, n_tokens, _hidden_size) = embeddings.dims3()?;
122    let embeddings = (embeddings.sum(1)? / (n_tokens as f64))?;
123    let embeddings = if normalize_embeddings {
124        normalize_l2(&embeddings)?
125    } else {
126        embeddings
127    };
128
129    let query_embedding = embeddings.get(0)?;
130    let mut similarities = vec![];
131    for j in 1..n_sentences {
132        let e_j = embeddings.get(j)?;
133        let sum_ij = (&query_embedding * &e_j)?.sum_all()?.to_scalar::<f32>()?;
134        let sum_i2 = (&query_embedding * &query_embedding)?
135            .sum_all()?
136            .to_scalar::<f32>()?;
137        let sum_j2 = (&e_j * &e_j)?.sum_all()?.to_scalar::<f32>()?;
138        let cosine_similarity = sum_ij / (sum_i2 * sum_j2).sqrt();
139        similarities.push(cosine_similarity)
140    }
141
142    Ok(similarities)
143}
