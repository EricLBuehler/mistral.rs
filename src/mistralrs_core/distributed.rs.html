45#[allow(clippy::too_many_arguments)]
46pub(crate) fn prepare_distributed_mapper<
47    'a,
48    T: DeviceMappedModelLoader + IsqModelLoader + ?Sized,
49>(
50    dtype: DType,
51    device: &Device,
52    load_device: &Device,
53    available_devices: &[Device],
54    config: &str,
55    loading_isq: bool,
56    from_uqff: bool,
57    organization: IsqOrganization,
58    model: &T,
59    paths: &dyn ModelPaths,
60) -> anyhow::Result<(Box<dyn DeviceMapper + Send + Sync>, ShardedVarBuilder<'a>)> {
61    #[cfg(not(feature = "nccl"))]
62    tracing::warn!(
63        "NCCL support was included in the build, be sure to build with `--features nccl`."
64    );
65
66    // NCCL case!
67
68    let local_world_size = available_devices.len();
69    let global_world_size = if let Ok(x) = std::env::var("MISTRALRS_MN_GLOBAL_WORLD_SIZE") {
70        usize::from_str(&x).context("MISTRALRS_MN_GLOBAL_WORLD_SIZE")?
71    } else {
72        mistralrs_quant::distributed::get_global_tp_size_from_devices()?
73    };
74
75    let use_multi_node = std::env::var("MISTRALRS_MN_GLOBAL_WORLD_SIZE").is_ok();
76    if use_multi_node {
77        info!("MISTRALRS_MN_GLOBAL_WORLD_SIZE is set, entering multi-node.");
78    }
79
80    if global_world_size < local_world_size || global_world_size % local_world_size != 0 {
81        anyhow::bail!("Global world size {global_world_size} must both be at least and divide the local world size {local_world_size}");
82    }
83
84    info!("Local tensor parallel world size is {local_world_size}");
85    info!("Global tensor parallel world size is {global_world_size}");
86
87    // TP uses parallel pipelines.
88    let name = ipc_name()?;
89    let mut id;
90    let local_rank = if let Ok(payload) = env::var(IS_DAEMON_FLAG) {
91        let payload: WorkerTransferData = serde_json::from_str(&payload)?;
92        let WorkerTransferData::Init {
93            id: new_id,
94            worker_rank,
95        } = payload;
96        id = mistralrs_quant::Id::uninit(new_id.0);
97
98        let mut stream = LocalStream::connect(name)?;
99        stream.write_all(b"ready\n")?;
100        worker_rank + 1
101    } else {
102        id = mistralrs_quant::Id::new();
103        let num_workers = mistralrs_quant::distributed::get_global_tp_size_from_devices()? - 1;
104        let mut children = Vec::new();
105        for worker_rank in 0..num_workers {
106            let exe_path = env::current_exe().expect("Failed to get current exe");
107
108            let args: Vec<String> = env::args().collect();
109
110            let mut cmd = Command::new(exe_path);
111            cmd.args(&args[1..]);
112
113            let data = WorkerTransferData::Init {
114                id: BigCCharArray(*id.internal()),
115                worker_rank,
116            };
117
118            cmd.env(IS_DAEMON_FLAG, serde_json::to_string(&data)?);
119
120            cmd.stdout(std::process::Stdio::null());
121            cmd.stderr(std::process::Stdio::null());
122            cmd.stdin(std::process::Stdio::null());
123
124            children.push(cmd.spawn().expect("Failed to spawn process"));
125        }
126
127        let listener = ListenerOptions::new().name(name).create_sync()?;
128        let mut ready_count = 0;
129
130        while ready_count < num_workers {
131            let stream = listener.accept()?;
132            let mut reader = BufReader::new(stream);
133            let mut message = String::new();
134            reader.read_line(&mut message)?;
135            if message.trim() == "ready" {
136                ready_count += 1;
137            }
138        }
139        info!("All workers have received the ids!");
140
141        0
142    };
143
144    if use_multi_node {
145        if let Ok(n_nodes) = env::var("MISTRALRS_MN_HEAD_NUM_WORKERS") {
146            let n_nodes = usize::from_str(&n_nodes).context("MISTRALRS_MN_HEAD_NUM_WORKERS")?;
147            info!("Head node managing {n_nodes} workers.");
148            let Ok(port) = env::var("MISTRALRS_MN_HEAD_PORT") else {
149                anyhow::bail!("Got MISTRALRS_MN_HEAD_NUM_WORKERS, expected MISTRALRS_MN_HEAD_PORT");
150            };
151            info!("Head node initializing connection on {port}.");
152            let server = mistralrs_quant::Server::new(
153                &format!("0.0.0.0:{port}"),
154                n_nodes,
155                local_world_size,
156            )?;
157
158            server.broadcast_id(&id)?;
159        } else if let Ok(addr) = env::var("MISTRALRS_MN_WORKER_SERVER_ADDR") {
160            info!("Worker node connecting to {addr}.");
161            let client = mistralrs_quant::Client::new(addr.parse()?, local_world_size)?;
162
163            id = client.receive_id()?;
164        }
165    }
166
167    let rank_offset = if env::var("MISTRALRS_MN_WORKER_SERVER_ADDR").is_ok() {
168        let Ok(node_id) = env::var("MISTRALRS_MN_WORKER_ID") else {
169            anyhow::bail!("Got MISTRALRS_MN_WORKER_SERVER_ADDR, expected MISTRALRS_MN_WORKER_ID");
170        };
171        let node_id = usize::from_str(&node_id).context("MISTRALRS_MN_WORKER_ID")?;
172        info!("Worker ID is {node_id}.");
173        (node_id + 1) * local_world_size
174    } else {
175        0
176    };
177
178    // They each block on each other
179    // https://docs.nvidia.com/deeplearning/nccl/user-guide/docs/api/comms.html?ncclcomminitrank#ncclcomminitrank
180    let comm = mistralrs_quant::Comm::from_device(
181        id,
182        device,
183        local_rank + rank_offset,
184        global_world_size,
185    )?;
186
187    let make_dummy_regexes = if loading_isq && from_uqff {
188        // Dummy weights for the layers which will be overwritten...
189        Some(std::sync::Arc::new(
190            if matches!(organization, IsqOrganization::MoeExpertsOnly) {
191                model.isq_layer_regexes_moqe(config)?
192            } else {
193                model.isq_layer_regexes(config)?
194            },
195        ))
196    } else {
197        None
198    };
199
200    let sharded_vb = unsafe {
201        ShardedSafeTensors::sharded(
202            paths.get_weight_filenames(),
203            dtype,
204            load_device,
205            make_dummy_regexes,
206        )?
207    };
208
209    info!("Loading all ranks.");
210    // The mapper is specific to this pipeline
211    let mapper = DeviceMapSetting::Nccl {
212        nm_device: available_devices[0].clone(),
213        comm: Arc::new(comm),
214    }
215    .into_mapper(model.num_layers(config)?, device, None)?;
216
217    let sharded_vb = if !loading_isq {
218        sharded_vb.clone().set_device(device.clone())
219    } else {
220        sharded_vb.clone()
221    };
222
223    Ok((mapper, sharded_vb))
224}
