33struct MLP {
34    gate_proj: Arc<dyn LinearLayerLike + Send + Sync>,
35    up_proj: Arc<dyn LinearLayerLike + Send + Sync>,
36    down_proj: Arc<dyn LinearLayerLike + Send + Sync>,
37    act_fn: Activation,
38}
39
40impl MLP {
41    #[allow(clippy::too_many_arguments)]
42    fn new(
43        cfg: &Config,
44        vb: ShardedVarBuilder,
45        lora_config: &[((String, String), LoraConfig)],
46        count: &mut usize,
47        ord: &Ordering,
48        mapper: &dyn DeviceMapper,
49        layer_idx: usize,
50        loading_isq: bool,
51        preload_adapters: &Option<HashMap<String, (ShardedVarBuilder, LoraConfig)>>,
52    ) -> Result<Self> {
53        let hidden_sz = cfg.hidden_size;
54        let intermediate_sz = cfg.intermediate_size;
55        let gate_proj = linear_no_bias(
56            hidden_sz,
57            intermediate_sz,
58            mapper.set_device(layer_idx, vb.pp("gate_proj"), loading_isq),
59            mapper.set_device(layer_idx, vb.pp("gate_proj"), false),
60            lora_config,
61            count,
62            ord,
63            preload_adapters,
64        )?;
65        let up_proj = linear_no_bias(
66            hidden_sz,
67            intermediate_sz,
68            mapper.set_device(layer_idx, vb.pp("up_proj"), loading_isq),
69            mapper.set_device(layer_idx, vb.pp("up_proj"), false),
70            lora_config,
71            count,
72            ord,
73            preload_adapters,
74        )?;
75        let down_proj = linear_no_bias(
76            intermediate_sz,
77            hidden_sz,
78            mapper.set_device(layer_idx, vb.pp("down_proj"), loading_isq),
79            mapper.set_device(layer_idx, vb.pp("down_proj"), false),
80            lora_config,
81            count,
82            ord,
83            preload_adapters,
84        )?;
85        Ok(Self {
86            gate_proj,
87            up_proj,
88            down_proj,
89            act_fn: cfg.hidden_act,
90        })
91    }
92
93    fn forward(
94        &self,
95        xs: &Tensor,
96        scalings: Option<Tensor>,
97        global_scaling_weight: f64,
98        is_scaling_pass: Option<f64>,
99    ) -> Result<Tensor> {
100        let original_dtype = xs.dtype();
101        let mut xs = xs.clone();
102        if let Some(t) = self.gate_proj.quantized_act_type() {
103            xs = xs.to_dtype(t)?;
104        }
105        let lhs = self
106            .gate_proj
107            .lora_forward(
108                &xs,
109                scalings.clone(),
110                global_scaling_weight,
111                is_scaling_pass,
112            )?
113            .apply(&self.act_fn)?;
114        let rhs = self.up_proj.lora_forward(
115            &xs,
116            scalings.clone(),
117            global_scaling_weight,
118            is_scaling_pass,
119        )?;
120        let mut res = self.down_proj.lora_forward(
121            &(lhs * rhs)?,
122            scalings,
123            global_scaling_weight,
124            is_scaling_pass,
125        )?;
126        if self.gate_proj.quantized_act_type().is_some() {
127            res = res.to_dtype(original_dtype)?;
128        }
129        Ok(res)
130    }
131}

146impl Attention {
147    #[allow(clippy::too_many_arguments)]
148    fn new(
149        rotary_emb: Arc<RotaryEmbedding>,
150        cfg: &Config,
151        vb: ShardedVarBuilder,
152        lora_config: &[((String, String), LoraConfig)],
153        count: &mut usize,
154        ord: &Ordering,
155        mapper: &dyn DeviceMapper,

161        let num_heads = cfg.num_attention_heads;
162        let num_kv_heads = cfg.num_key_value_heads;
163        let head_dim = cfg.head_dim();
164        let q_proj = linear_no_bias(

166            num_heads * head_dim,

168            mapper.set_device(layer_idx, vb.pp("q_proj"), false),
169            lora_config,
170            count,
171            ord,
172            preload_adapters,
173        )?;
174        let k_proj = linear_no_bias(
175            hidden_sz,
176            num_kv_heads * head_dim,
177            mapper.set_device(layer_idx, vb.pp("k_proj"), loading_isq),
178            mapper.set_device(layer_idx, vb.pp("k_proj"), false),
179            lora_config,
180            count,
181            ord,
182            preload_adapters,
183        )?;
184        let v_proj = linear_no_bias(
185            hidden_sz,
186            num_kv_heads * head_dim,
187            mapper.set_device(layer_idx, vb.pp("v_proj"), loading_isq),
188            mapper.set_device(layer_idx, vb.pp("v_proj"), false),
189            lora_config,
190            count,
191            ord,
192            preload_adapters,
193        )?;
194        let o_proj = linear_no_bias(
195            num_heads * head_dim,
196            hidden_sz,
197            mapper.set_device(layer_idx, vb.pp("o_proj"), loading_isq),
198            mapper.set_device(layer_idx, vb.pp("o_proj"), false),
199            lora_config,
200            count,
201            ord,
202            preload_adapters,
203        )?;
204        Ok(Self {
205            q_proj,
206            k_proj,
207            v_proj,
208            o_proj,
209            num_heads,
210            num_kv_heads,
211            head_dim,
212            rotary_emb,
213            sliding_window: cfg.sliding_window,
214            sdpa_params: SdpaParams {
215                n_kv_groups: num_heads / num_kv_heads,
216                softcap: None,
217                softmax_scale: 1.0 / (head_dim as f32).sqrt(),
218                sliding_window: cfg.sliding_window,
219            },
220        })
221    }

223    #[allow(clippy::too_many_arguments)]
224    fn forward(
225        &self,
226        xs: &Tensor,
227        attention_mask: Option<&Tensor>,
228        seqlen_offsets: &[usize],
229        kv_cache: &mut Option<(Tensor, Tensor)>,
230        scalings: Option<Tensor>,
231        global_scaling_weight: f64,
232        is_scaling_pass: Option<f64>,
233        flash_params: &FlashParams,
234    ) -> Result<Tensor> {
235        let (b_sz, q_len, _) = xs.dims3()?;
236
237        let original_dtype = xs.dtype();
238        let mut xs = xs.clone();
239        if let Some(t) = self.q_proj.quantized_act_type() {
240            xs = xs.to_dtype(t)?;
241        }
242        let mut q = self.q_proj.lora_forward(
243            &xs,
244            scalings.clone(),
245            global_scaling_weight,
246            is_scaling_pass,
247        )?;
248        let mut k = self.k_proj.lora_forward(
249            &xs,
250            scalings.clone(),
251            global_scaling_weight,
252            is_scaling_pass,
253        )?;
254        let mut v = self.v_proj.lora_forward(
255            &xs,
256            scalings.clone(),
257            global_scaling_weight,
258            is_scaling_pass,
259        )?;
260        if self.q_proj.quantized_act_type().is_some() {
261            q = q.to_dtype(original_dtype)?;
262            k = k.to_dtype(original_dtype)?;
263            v = v.to_dtype(original_dtype)?;
264        }
265
266        let (q, k, v) = if q_len != 1 {
267            let q = q
268                .reshape((b_sz, q_len, self.num_heads, self.head_dim))?
269                .transpose(1, 2)?;
270            let k = k
271                .reshape((b_sz, q_len, self.num_kv_heads, self.head_dim))?
272                .transpose(1, 2)?;
273            let v = v
274                .reshape((b_sz, q_len, self.num_kv_heads, self.head_dim))?
275                .transpose(1, 2)?;
276            (q, k, v)
277        } else {
278            let q = q.reshape((b_sz, self.num_heads, q_len, self.head_dim))?;
279            let k = k.reshape((b_sz, self.num_kv_heads, q_len, self.head_dim))?;
280            let v = v.reshape((b_sz, self.num_kv_heads, q_len, self.head_dim))?;
281            (q, k, v)
282        };
283
284        let (q, k) = self.rotary_emb.forward(&q, &k, seqlen_offsets)?;
285
286        let (k, v, attn_mask) = Cache::update_kv_cache_sliding_window(
287            kv_cache,
288            k,
289            v,
290            attention_mask,
291            self.sliding_window,
292        )?;
293
294        let mut attn_output = Sdpa.run_attention(
295            &q,
296            &k,
297            &v,
298            attn_mask.as_ref(),
299            Some(flash_params),
300            &self.sdpa_params,
301        )?;
302
303        if let Some(t) = self.q_proj.quantized_act_type() {
304            attn_output = attn_output.to_dtype(t)?;
305        }
306        let mut res = self.o_proj.lora_forward(
307            &attn_output.transpose(1, 2)?.reshape((b_sz, q_len, ()))?,
308            scalings.clone(),
309            global_scaling_weight,
310            is_scaling_pass,
311        )?;
312        if self.q_proj.quantized_act_type().is_some() {
313            res = res.to_dtype(original_dtype)?;
314        }
315        Ok(res)
316    }
317}

319struct DecoderLayer {
320    self_attn: Attention,
321    mlp: MLP,
322    input_layernorm: RmsNorm,
323    post_attention_layernorm: RmsNorm,
324}
325
326impl DecoderLayer {
327    #[allow(clippy::too_many_arguments)]
328    fn new(
329        rotary_emb: Arc<RotaryEmbedding>,
330        cfg: &Config,
331        vb: ShardedVarBuilder,
332        lora_config: &[((String, String), LoraConfig)],
333        count: &mut usize,
334        ord: &Ordering,
335        mapper: &dyn DeviceMapper,
336        layer_idx: usize,
337        loading_isq: bool,
338        preload_adapters: &Option<HashMap<String, (ShardedVarBuilder, LoraConfig)>>,
339    ) -> Result<Self> {
340        let self_attn = Attention::new(
341            rotary_emb,
342            cfg,
343            vb.pp("self_attn"),
344            lora_config,
345            count,
346            ord,
347            mapper,
348            layer_idx,
349            loading_isq,
350            preload_adapters,
351        )?;
352        let mlp = MLP::new(
353            cfg,
354            vb.pp("mlp"),
355            lora_config,
356            count,
357            ord,
358            mapper,
359            layer_idx,
360            loading_isq,
361            preload_adapters,
362        )?;
363        let input_layernorm = RmsNorm::new(
364            cfg.hidden_size,
365            cfg.rms_norm_eps,
366            mapper.set_device(layer_idx, vb.pp("input_layernorm"), false),
367        )?;
368        let post_attention_layernorm = RmsNorm::new(
369            cfg.hidden_size,
370            cfg.rms_norm_eps,
371            mapper.set_device(layer_idx, vb.pp("post_attention_layernorm"), false),
372        )?;
373        Ok(Self {
374            self_attn,
375            mlp,
376            input_layernorm,
377            post_attention_layernorm,
378        })
379    }
380
381    #[allow(clippy::too_many_arguments)]
382    fn forward(
383        &self,
384        xs: &Tensor,
385        attention_mask: Option<&Tensor>,
386        seqlen_offsets: &[usize],
387        kv_cache: &mut Option<(Tensor, Tensor)>,
388        scalings: Option<Tensor>,
389        global_scaling_weight: f64,
390        is_scaling_pass: Option<f64>,
391        flash_params: &FlashParams,
392    ) -> Result<Tensor> {
393        let residual = xs;
394        let xs = self.input_layernorm.forward(xs)?;
395        let xs = self.self_attn.forward(
396            &xs,
397            attention_mask,
398            seqlen_offsets,
399            kv_cache,
400            scalings.clone(),
401            global_scaling_weight,
402            is_scaling_pass,
403            flash_params,
404        )?;
405        let xs = (xs + residual)?;
406        let residual = &xs;
407        let xs = self.mlp.forward(
408            &xs.apply(&self.post_attention_layernorm)?,
409            scalings,
410            global_scaling_weight,
411            is_scaling_pass,
412        )?;
413        residual + xs
414    }
415}

432impl XLoraModel {
433    #[allow(clippy::too_many_arguments)]
434    pub fn new(
435        cfg: &Config,
436        vb: ShardedVarBuilder,
437        lora_config: &[((String, String), LoraConfig)],
438        xlora_config: Option<XLoraConfig>,
439        xlora_ordering: Ordering,
440        is_gptx: bool,
441        normal_loading_metadata: NormalLoadingMetadata,
442        preload_adapters: &Option<HashMap<String, (ShardedVarBuilder, LoraConfig)>>,
443    ) -> Result<Self> {
444        if let Some(ref quant_cfg) = &cfg.quantization_config {
445            tracing::info!(
446                "Using {} quantization: {}.",
447                quant_cfg.name(),
448                quant_cfg.get_bits_name(&vb)
449            );
450        }
451        let mapper = normal_loading_metadata.mapper;
452
453        let vb_m = vb.pp("model");
454        let embed_tokens = layers::embedding(
455            cfg.vocab_size,
456            cfg.hidden_size,
457            mapper.set_nm_device(vb_m.pp("embed_tokens"), false),
458            &cfg.quantization_config,
459        )?;
460        let head_dim = cfg.head_dim();
461        let mut layers = Vec::with_capacity(cfg.num_hidden_layers);
462        let vb_l = vb_m.pp("layers");
463        let mut ropes = HashMap::new();
464        for layer_idx in 0..cfg.num_hidden_layers {
465            let device = mapper
466                .device_for(layer_idx, false)
467                .unwrap_or(&normal_loading_metadata.real_device);
468            ropes.insert(
469                device.location(),
470                Arc::new(RotaryEmbedding::new(
471                    cfg.rope_theta as f32,
472                    head_dim,
473                    cfg.max_position_embeddings,
474                    device,
475                    is_gptx,
476                    vb_m.dtype(),
477                )?),
478            );
479        }
480        let mut count = 0;
481        for layer_idx in NiceProgressBar::<_, 'b'>(
482            0..cfg.num_hidden_layers,
483            "Loading repeating layers",
484            &normal_loading_metadata.multi_progress,
485        ) {
486            let device = mapper
487                .device_for(layer_idx, false)
488                .unwrap_or(&normal_loading_metadata.real_device);
489            let rotary_emb = ropes
490                .get(&device.location())
491                .expect("No RoPE for device location!")
492                .clone();
493            let layer = DecoderLayer::new(
494                rotary_emb.clone(),
495                cfg,
496                vb_l.pp(layer_idx),
497                lora_config,
498                &mut count,
499                &xlora_ordering,
500                &*mapper,
501                layer_idx,
502                normal_loading_metadata.loading_isq,
503                preload_adapters,
504            )?;
505            layers.push(layer)
506        }
507        if xlora_config.is_none() && preload_adapters.is_none() {
508            // We are now a LoRA model so we must merge the weights
509            info!("Merging LoRA adapters.");
510            for layer in layers.iter_mut().tqdm() {
511                Arc::get_mut(&mut layer.self_attn.k_proj)
512                    .unwrap()
513                    .merge_weights()?;
514                Arc::get_mut(&mut layer.self_attn.o_proj)
515                    .unwrap()
516                    .merge_weights()?;
517                Arc::get_mut(&mut layer.self_attn.q_proj)
518                    .unwrap()
519                    .merge_weights()?;
520                Arc::get_mut(&mut layer.self_attn.v_proj)
521                    .unwrap()
522                    .merge_weights()?;
523
524                Arc::get_mut(&mut layer.mlp.down_proj)
525                    .unwrap()
526                    .merge_weights()?;
527                Arc::get_mut(&mut layer.mlp.gate_proj)
528                    .unwrap()
529                    .merge_weights()?;
530                Arc::get_mut(&mut layer.mlp.up_proj)
531                    .unwrap()
532                    .merge_weights()?;
533            }
534        }
535        let norm = RmsNorm::new(
536            cfg.hidden_size,
537            cfg.rms_norm_eps,
538            mapper.set_nm_device(vb_m.pp("norm"), false),
539        )?;
540        let lm_head = linear_no_bias(
541            cfg.hidden_size,
542            cfg.vocab_size,
543            mapper.set_nm_device(vb.pp("lm_head"), normal_loading_metadata.loading_isq),
544            mapper.set_nm_device(vb.pp("lm_head"), false),
545            lora_config,
546            &mut count,
547            &xlora_ordering,
548            preload_adapters,
549        )?;
550        if xlora_config.is_some() && lm_head.is_lora() {
551            // This is why we can pass dummy values (..., None, 1.0, None)?
552            candle_core::bail!("Got an adapter `lm_head` layer, this is unsupported with X-LoRA.");
553        }
554        Ok(Self {
555            embed_tokens,
556            layers,
557            norm,
558            lm_head,
559            sliding_window: cfg.sliding_window,
560            device: normal_loading_metadata.real_device,
561            dtype: vb.dtype(),
562            cache: EitherCache::Full(Cache::new(cfg.num_hidden_layers, true)),
563            max_seq_len: cfg.max_position_embeddings,
564            xlora_classifier: xlora_config.map(|xlora_config| {
565                XLoraClassifier::new(xlora_config, count, lora_config.len(), vb, false).unwrap()
566            }),
567            mapper,
568            cfg: ModelConfigMetadata {
569                max_seq_len: cfg.max_position_embeddings,
570                num_layers: cfg.num_hidden_layers,
571                hidden_size: cfg.hidden_size,
572                num_kv_heads: cfg.num_key_value_heads,
573                num_attn_heads: cfg.num_attention_heads,
574                sliding_window: cfg.sliding_window,
575                k_head_dim: cfg.head_dim(),
576                v_head_dim: cfg.head_dim(),
577            },
578        })
579    }

581    #[allow(clippy::too_many_arguments)]
582    fn inner_forward(
583        &self,
584        input_ids: &Tensor,
585        seqlen_offsets: &[usize],
586        scalings: Option<Tensor>,
587        is_full_pass: bool,
588        no_kv_cache: bool,
589        is_scaling_pass: Option<f64>,
590        flash_params: &FlashParams,
591    ) -> Result<Tensor> {
592        let mut cache = if is_full_pass {
593            if no_kv_cache {
594                let mut new_cache = Vec::new();
595                for _ in 0..self.cache.full().xlora_lock().len() {
596                    new_cache.push(None);
597                }
598
599                self.cache.full().xlora_lock().clone_from(&new_cache);
600            }
601            self.cache.full().xlora_lock()
602        } else {
603            self.cache.full().lock()
604        };
605        let mut xs = self.embed_tokens.forward(input_ids)?;
606        let attention_mask = CausalMasker.make_sliding_window_causal_mask_matrix(
607            input_ids,
608            &*cache,
609            self.sliding_window,
610            xs.dtype(),
611            self.cfg.num_attn_heads,
612        )?;
613        for (i, layer) in self.layers.iter().enumerate() {
614            xs = self.mapper.map(xs, i)?;
615            xs = layer.forward(
616                &xs,
617                attention_mask
618                    .as_ref()
619                    .map(|m| m.to_device(xs.device()).unwrap())
620                    .as_ref(),
621                seqlen_offsets,
622                &mut cache[i],
623                scalings.clone(),
624                self.xlora_classifier
625                    .as_ref()
626                    .map(|classifier| classifier.get_global_scaling_weight())
627                    .unwrap_or(1.0),
628                is_scaling_pass,
629                flash_params,
630            )?
631        }
632        let xs = xs.to_device(&self.device)?;
633        xs.apply(&self.norm)
634    }

636    #[allow(clippy::too_many_arguments)]
637    pub fn forward(
638        &self,
639        input_ids: &Tensor,
640        input_ids_full: &Tensor,
641        seqlen_offsets: &[usize],
642        seqlen_offsets_full: &[usize],
643        no_kv_cache: bool,
644        non_granular_state: &Option<NonGranularState>,
645        context_lens: Vec<(usize, usize)>,
646        flash_params: &FlashParams,
647        flash_params_full: &FlashParams,
648    ) -> Result<Tensor> {
649        if self.xlora_classifier.is_some() {
650            let scalings = self.get_scalings(
651                input_ids,
652                input_ids_full,
653                seqlen_offsets,
654                seqlen_offsets_full,
655                no_kv_cache,
656                non_granular_state,
657                &vec![usize::MAX; context_lens.len()],
658                flash_params,
659                flash_params_full,
660            )?;
661
662            if no_kv_cache {
663                let mut res = self
664                    .inner_forward(
665                        input_ids_full,
666                        seqlen_offsets_full,
667                        Some(scalings),
668                        true,
669                        no_kv_cache,
670                        None,
671                        flash_params_full,
672                    )?
673                    .contiguous()?;
674                if let Some(t) = self.lm_head.quantized_act_type() {
675                    res = res.to_dtype(t)?;
676                }
677                extract_logits(
678                    &self.lm_head.lora_forward(&res, None, 1.0, None)?,
679                    context_lens,
680                )
681            } else {
682                // is_full_pass=true is ok because no_kv_cache=false
683                let mut res = self
684                    .inner_forward(
685                        input_ids,
686                        seqlen_offsets,
687                        Some(scalings),
688                        true,
689                        no_kv_cache,
690                        None,
691                        flash_params,
692                    )?
693                    .contiguous()?;
694                if let Some(t) = self.lm_head.quantized_act_type() {
695                    res = res.to_dtype(t)?;
696                }
697                extract_logits(
698                    &self.lm_head.lora_forward(&res, None, 1.0, None)?,
699                    context_lens,
700                )
701            }
702        } else {
703            let mut res = self
704                .inner_forward(
705                    input_ids,
706                    seqlen_offsets,
707                    None,
708                    false,
709                    no_kv_cache,
710                    None,
711                    flash_params,
712                )?
713                .contiguous()?;
714            if let Some(t) = self.lm_head.quantized_act_type() {
715                res = res.to_dtype(t)?;
716            }
717            extract_logits(
718                &self.lm_head.lora_forward(&res, None, 1.0, None)?,
719                context_lens,
720            )
721        }
722    }
723}

725impl IsqModel for XLoraModel {
726    fn get_layers(
727        &mut self,
728    ) -> (
729        Vec<(&mut Arc<dyn QuantMethod>, Option<usize>)>,
730        &dyn DeviceMapper,
731    ) {
732        let mut tensors = Vec::new();
733        tensors.push((Arc::get_mut(&mut self.lm_head).unwrap().quant_inner(), None));
734        for (i, layer) in self.layers.iter_mut().enumerate() {
735            tensors.push((
736                Arc::get_mut(&mut layer.self_attn.q_proj)
737                    .unwrap()
738                    .quant_inner(),
739                Some(i),
740            ));
741            tensors.push((
742                Arc::get_mut(&mut layer.self_attn.k_proj)
743                    .unwrap()
744                    .quant_inner(),
745                Some(i),
746            ));
747            tensors.push((
748                Arc::get_mut(&mut layer.self_attn.v_proj)
749                    .unwrap()
750                    .quant_inner(),
751                Some(i),
752            ));
753            tensors.push((
754                Arc::get_mut(&mut layer.self_attn.o_proj)
755                    .unwrap()
756                    .quant_inner(),
757                Some(i),
758            ));
759            tensors.push((
760                Arc::get_mut(&mut layer.mlp.gate_proj)
761                    .unwrap()
762                    .quant_inner(),
763                Some(i),
764            ));
765            tensors.push((
766                Arc::get_mut(&mut layer.mlp.up_proj).unwrap().quant_inner(),
767                Some(i),
768            ));
769            tensors.push((
770                Arc::get_mut(&mut layer.mlp.down_proj)
771                    .unwrap()
772                    .quant_inner(),
773                Some(i),
774            ));
775        }
776        (tensors, &*self.mapper)
777    }
778
779    fn residual_tensors(&self) -> Vec<(String, Tensor)> {
780        panic!("Cannot generate UQFF for an adapter model.")
781    }
782}
