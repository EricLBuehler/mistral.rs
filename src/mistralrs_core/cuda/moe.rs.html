1use candle_core::{Result, Tensor};
2
3#[cfg(feature = "cuda")]
4pub fn moe_gemm(
5    input: &Tensor,
6    weights: &Tensor,
7    topk_weights: &Option<Tensor>,
8    sorted_token_ids: &Tensor,
9    experts_ids: &Tensor,
10    topk: usize,
11    is_prefill: bool,
12) -> Result<Tensor> {
13    use candle::cuda_backend::cudarc::driver::DevicePtr;
14    use candle_core as candle;
15    use candle_core::DType;
16    use half::{bf16, f16};
17
18    fn cuda_fwd<
19        T: candle::cuda_backend::CudaDType + candle::cuda_backend::cudarc::driver::DeviceRepr,
20    >(
21        input: &Tensor,
22        weights: &Tensor,
23        topk_weights: &Option<Tensor>,
24        sorted_token_ids: &Tensor,
25        experts_ids: &Tensor,
26        topk: usize,
27        is_prefill: bool,
28    ) -> Result<Tensor> {
29        let (mut size_m, size_k1) = input.dims2()?;
30        if topk_weights.is_none() {
31            size_m *= topk;
32        }
33        let (num_experts, size_n, size_k) = weights.dims3()?;
34        assert!(
35            size_k == size_k1,
36            "input {:?} and weight {:?} last dim mismatch!",
37            size_k1,
38            size_k
39        );
40        let dev = input.device().as_cuda_device()?;
41        let data_type = match input.dtype() {
42            DType::F16 => 0,
43            DType::BF16 => 1,
44            _ => {
45                candle_core::bail!("moe_gemm_wmma only accept f16/bf16 inputs!")
46            }
47        };
48
49        let (input, _) = input.storage_and_layout();
50        let input = match &*input {
51            candle::Storage::Cuda(c) => c.as_cuda_slice::<T>()?,
52            _ => candle::bail!("input must be a cuda tensor"),
53        };
54
55        let (weights, _) = weights.storage_and_layout();
56        let weights = match &*weights {
57            candle::Storage::Cuda(c) => c.as_cuda_slice::<T>()?,
58            _ => candle::bail!("weight must be a cuda tensor"),
59        };
60
61        let (sorted_token_ids, _) = sorted_token_ids.storage_and_layout();
62        let sorted_token_ids = match &*sorted_token_ids {
63            candle::Storage::Cuda(c) => c.as_cuda_slice::<u32>()?,
64            _ => candle::bail!("sorted_token_ids must be a cuda tensor"),
65        };
66
67        let (experts_ids, _) = experts_ids.storage_and_layout();
68        let experts_ids = match &*experts_ids {
69            candle::Storage::Cuda(c) => c.as_cuda_slice::<u32>()?,
70            _ => candle::bail!("experts_ids must be a cuda tensor"),
71        };
72
73        let topk_weights_ptr = if let Some(topk_weights) = &topk_weights {
74            let (topk_weights, _) = topk_weights.storage_and_layout();
75            let topk_weights = match &*topk_weights {
76                candle::Storage::Cuda(c) => c.as_cuda_slice::<f32>()?,
77                _ => candle::bail!("topk_weights must be a cuda tensor"),
78            };
79            let topk_w_ptr = topk_weights.device_ptr(topk_weights.stream()).0 as *const f32;
80            topk_w_ptr
81        } else {
82            std::ptr::null() as *const f32
83        };
84
85        let output = unsafe { dev.alloc::<T>(size_m * size_n) }?;
86
87        let stream = dev.cuda_stream().cu_stream() as i64;
88        use core::ffi::c_void;
89
90        let moe_func = if is_prefill {
91            crate::cuda::ffi::moe_gemm_wmma
92        } else {
93            crate::cuda::ffi::moe_gemm
94        };
95        unsafe {
96            moe_func(
97                input.device_ptr(input.stream()).0 as *const c_void, // [size_m, size_k]
98                weights.device_ptr(weights.stream()).0 as *const c_void, // [num_experts, size_n, size_k]
99                sorted_token_ids.device_ptr(sorted_token_ids.stream()).0 as *const i32,
100                experts_ids.device_ptr(experts_ids.stream()).0 as *const i32,
101                topk_weights_ptr,
102                output.device_ptr(output.stream()).0 as *mut c_void, // [size_m, size_n]
103                num_experts as i32,
104                topk as i32,
105                size_m as i32,
106                size_n as i32,
107                size_k as i32,
108                data_type as i32, // 0=float16, 1=bf16 (for input/output)
109                stream as i64,
110            );
111        }
112
113        let output = candle::CudaStorage::wrap_cuda_slice(output, dev.clone());
114        let output = Tensor::from((candle::Storage::Cuda(output), (size_m, size_n)));
115
116        Ok(output)
117    }
118
119    match input.dtype() {
120        DType::F16 => cuda_fwd::<f16>(
121            input,
122            weights,
123            topk_weights,
124            sorted_token_ids,
125            experts_ids,
126            topk,
127            is_prefill,
128        ),
129        DType::BF16 => cuda_fwd::<bf16>(
130            input,
131            weights,
132            topk_weights,
133            sorted_token_ids,
134            experts_ids,
135            topk,
136            is_prefill,
137        ),
138        _ => {
139            candle_core::bail!("moe_gemm only accept f16/bf16 inputs!")
140        }
141    }
142}
