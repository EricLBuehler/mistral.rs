151/// Customizable logits processor.
152///
153/// # Example
154/// ```rust
155/// use std::{sync::Arc, ops::Mul};
156/// use mistralrs_core::CustomLogitsProcessor;
157/// use candle_core::{Result, Tensor};
158///
159/// struct ThresholdLogitsProcessor;
160/// impl CustomLogitsProcessor for ThresholdLogitsProcessor {
161///     fn apply(&self, logits: &Tensor, _context: &[u32]) -> Result<Tensor> {
162///         // Mask is 1 for true, 0 for false.
163///         let mask = logits.ge(0.5)?;
164///         logits.broadcast_mul(&mask.to_dtype(logits.dtype())?)
165///     }
166/// }
167/// let processor1: Arc<dyn CustomLogitsProcessor> = Arc::new(|logits: &Tensor, _context: &[u32]| logits * 1.23);
168/// let processor2: Arc<dyn CustomLogitsProcessor> = Arc::new(ThresholdLogitsProcessor);
169/// ```
170pub trait CustomLogitsProcessor: Send + Sync {
171    /// Logits and sequence context (prompt and generated tokens), returning modified tokens.
172    fn apply(&self, logits: &Tensor, context: &[u32]) -> Result<Tensor>;
173}
174
175impl<T: Fn(&Tensor, &[u32]) -> Result<Tensor> + Send + Sync> CustomLogitsProcessor for T {
176    fn apply(&self, logits: &Tensor, context: &[u32]) -> Result<Tensor> {
177        self(logits, context)
178    }
179}

260    fn get_top_logprobs(&self, probs: &[f32], argsort_indices: &[u32]) -> Result<Vec<TopLogprob>> {
261        let mut argsort_indices_sorted = argsort_indices.to_vec();
262        // Sort by descending prob
263        argsort_indices_sorted.sort_by(|a, b| {
264            probs[*b as usize]
265                .partial_cmp(&probs[*a as usize])
266                .expect("No ordering.")
267        });
268        // These are where the top n are
269        let top_n_toks_range = 0..self.top_n_logprobs;
270        // The top n's values
271        let top_n_logprobs = argsort_indices_sorted[top_n_toks_range.clone()]
272            .iter()
273            .map(|x| probs[*x as usize].log(10.0))
274            .collect::<Vec<_>>();
275        // Find where they actually are in the logits
276        let mut top_n_toks = Vec::new();
277        for val in top_n_toks_range {
278            top_n_toks.push(argsort_indices[val]);
279        }
280
281        if let Some(tokenizer) = &self.tokenizer {
282            let mut bytes = Vec::new();
283            for tok in &top_n_toks {
284                bytes.push(
285                    tokenizer
286                        .decode(&[{ *tok }], false)
287                        .map_err(|x| Error::Msg(x.to_string()))?,
288                );
289            }
290
291            Ok(zip(bytes, zip(top_n_toks, top_n_logprobs))
292                .map(|(bytes, (token, logprob))| TopLogprob {
293                    token,
294                    logprob,
295                    bytes: Some(bytes),
296                })
297                .collect::<Vec<_>>())
298        } else {
299            Ok(zip(top_n_toks, top_n_logprobs)
300                .map(|(token, logprob)| TopLogprob {
301                    token,
302                    logprob,
303                    bytes: None,
304                })
305                .collect::<Vec<_>>())
306        }
307    }

309    fn sample_argmax(&self, logits: Tensor, return_logprobs: bool) -> Result<Logprobs> {
310        let next_token = logits.argmax(D::Minus1)?.to_scalar::<u32>()?;
311
312        let probs: Vec<f32> = logits.to_vec1()?;
313
314        let argsort_indices = (0..probs.len() as u32).collect::<Vec<_>>();
315        let logprob = probs[next_token as usize].log(10.0);
316
317        let top_logprobs = if return_logprobs {
318            Some(self.get_top_logprobs(&probs, &argsort_indices)?)
319        } else {
320            None
321        };
322
323        let bytes = if let Some(tokenizer) = &self.tokenizer {
324            Some(
325                tokenizer
326                    .decode(&[next_token], false)
327                    .map_err(|x| Error::Msg(x.to_string()))?,
328            )
329        } else {
330            None
331        };
332
333        Ok(Logprobs {
334            token: next_token,
335            logprob,
336            top_logprobs,
337            bytes,
338        })
339    }

341    fn sample_speculative_top_kp_min_p(
342        &self,
343        logits: Tensor,
344        return_logprobs: bool,
345        top_k: i64,
346        top_p: f32,
347        min_p: f32,
348    ) -> Result<Logprobs> {
349        let mut probs: Vec<f32> = logits.to_vec1()?;
350        let argsort_indices: Vec<u32> = logits.arg_sort_last_dim(false)?.to_vec1()?;
351
352        if top_k > 0 {
353            // Clamp smaller probabilities to zero.
354            for (index, val) in argsort_indices.iter().enumerate() {
355                if index >= top_k as usize {
356                    probs[*val as usize] = 0.0;
357                }
358            }
359        }
360
361        // TOP P
362
363        // top-p sampling (or "nucleus sampling") samples from the smallest set of
364        // tokens that exceed probability top_p. This way we never sample tokens that
365        // have very low probabilities and are less likely to go "off the rails".
366
367        // Clamp smaller probabilities to zero.
368        let mut cumsum = 0.;
369        for index in &argsort_indices {
370            if cumsum >= top_p {
371                probs[*index as usize] = 0.0;
372            } else {
373                cumsum += probs[*index as usize];
374            }
375        }
376
377        let max_p = probs[argsort_indices[0] as usize];
378
379        // MIN P
380
381        // min-p sampling samples from the tokens whose prob are greater than
382        // (max prob of token in dist) * min_p
383
384        // Clamp smaller probabilities to zero.
385        for index in &argsort_indices {
386            if max_p * min_p >= probs[*index as usize] {
387                probs[*index as usize] = 0.0;
388            }
389        }
390
391        let logits = Tensor::from_slice(&probs, logits.shape(), &Device::Cpu)?;
392
393        let next_token = argmax_sample_last_dim(&logits)?.to_scalar::<u32>()?;
394
395        let logprob = probs[next_token as usize].log(10.0);
396
397        let top_logprobs = if return_logprobs {
398            Some(self.get_top_logprobs(&probs, &argsort_indices)?)
399        } else {
400            None
401        };
402
403        let bytes = if let Some(tokenizer) = &self.tokenizer {
404            Some(
405                tokenizer
406                    .decode(&[next_token], false)
407                    .map_err(|x| Error::Msg(x.to_string()))?,
408            )
409        } else {
410            None
411        };
412
413        Ok(Logprobs {
414            token: next_token,
415            logprob,
416            top_logprobs,
417            bytes,
418        })
419    }

421    fn sample_multinomial(
422        &self,
423        probs: &mut Vec<f32>,
424        argsort_indices: Vec<u32>,
425        return_logprobs: bool,
426        rng: Arc<Mutex<Isaac64Rng>>,
427    ) -> Result<Logprobs> {
428        let distr = WeightedIndex::new(&*probs).map_err(Error::wrap)?;
429
430        let mut mut_ref_rng = &mut *rng.lock().expect("could not lock rng mutex");
431        let next_token = distr.sample(&mut mut_ref_rng); // "Find the first item which has a weight *higher* than the chosen weight."
432        let logprob = probs[next_token].log(10.0);
433
434        let top_logprobs = if return_logprobs {
435            Some(self.get_top_logprobs(probs, &argsort_indices)?)
436        } else {
437            None
438        };
439
440        let bytes = if let Some(tokenizer) = &self.tokenizer {
441            Some(
442                tokenizer
443                    .decode(&[next_token.try_into().unwrap()], false)
444                    .map_err(|x| Error::Msg(x.to_string()))?,
445            )
446        } else {
447            None
448        };
449
450        Ok(Logprobs {
451            token: next_token as u32,
452            logprob,
453            top_logprobs,
454            bytes,
455        })
456    }

458    #[allow(clippy::too_many_arguments)]
459    fn sample_top_kp_min_p(
460        &self,
461        probs: &mut Vec<f32>,
462        top_k: i64,
463        top_p: f32,
464        min_p: f32,
465        return_logprobs: bool,
466        rng: Arc<Mutex<Isaac64Rng>>,
467    ) -> Result<Logprobs> {
468        // Prepare argsort indices vector for top-p and min-p by sorting on CPU
469        let n = probs.len();
470        let mut idxs: Vec<usize> = (0..n).collect();
471        // Sort idxs by descending prob
472        idxs.sort_unstable_by(|&i, &j| probs[j].partial_cmp(&probs[i]).unwrap_or(Ordering::Equal));
473        // Top-k: zero out probabilities below the k-th largest
474        if top_k > 0 {
475            let k = top_k as usize;
476            if k <= n {
477                // threshold is the k-th largest probability
478                let threshold = probs[idxs[k - 1]];
479                for p in probs.iter_mut() {
480                    if *p < threshold {
481                        *p = 0.0;
482                    }
483                }
484            }
485        }
486
487        if top_p <= 0.0 || top_p >= 1.0 {
488            let argsort_indices = idxs.iter().map(|&i| i as u32).collect();
489            return self.sample_multinomial(probs, argsort_indices, return_logprobs, rng);
490        }
491
492        // TOP P
493
494        // top-p sampling (or "nucleus sampling") samples from the smallest set of
495        // tokens that exceed probability top_p. This way we never sample tokens that
496        // have very low probabilities and are less likely to go "off the rails".
497
498        // Clamp smaller probabilities to zero.
499        let mut cumsum = 0.;
500        // top-p filtering: zero out after cumulative probability exceeds top_p
501        for &i in &idxs {
502            if cumsum >= top_p {
503                probs[i] = 0.0;
504            } else {
505                cumsum += probs[i];
506            }
507        }
508
509        if min_p <= 0.0 || min_p >= 1.0 {
510            let argsort_indices = idxs.iter().map(|&i| i as u32).collect();
511            return self.sample_multinomial(probs, argsort_indices, return_logprobs, rng);
512        }
513
514        // min-p filtering based on max probability
515        let max_p = probs[idxs[0]];
516
517        // MIN P
518
519        // min-p sampling samples from the tokens whose prob are greater than
520        // (max prob of token in dist) * min_p
521
522        // Clamp smaller probabilities to zero.
523        for &i in &idxs {
524            if max_p * min_p >= probs[i] {
525                probs[i] = 0.0;
526            }
527        }
528
529        // Sample with clamped probabilities.
530        let argsort_indices = idxs.into_iter().map(|i| i as u32).collect();
531        self.sample_multinomial(probs, argsort_indices, return_logprobs, rng)
532    }

534    fn apply_penalties(&self, mut logits: Vec<f32>, context: &[u32]) -> Result<Tensor> {
535        if context.is_empty() {
536            candle_core::bail!("Penalty context is empty, this should not happen.");
537        }
538
539        // Dry penalty
540        self.apply_dry_penalty(&mut logits, context)?;
541
542        // Frequency and Presence penalty
543        self.apply_freq_presc_penalty(&mut logits, context)?;
544
545        let vocab_size = logits.len();
546        Tensor::from_vec(logits, vocab_size, &Device::Cpu)
547    }

549    fn apply_freq_presc_penalty(&self, logits: &mut [f32], context: &[u32]) -> Result<()> {
550        if self.frequency_penalty.is_some() || self.presence_penalty.is_some() {
551            let frequency_penalty = self.frequency_penalty.unwrap_or(0.);
552            let presence_penalty = self.presence_penalty.unwrap_or(0.);
553
554            //mu[j] -> mu[j] - c[j] * alpha_frequency - float(c[j] > 0) * alpha_presence
555
556            let mut counts = vec![0.0f32; logits.len()];
557            for ctx in context.iter() {
558                // Llama 3.2 uses a hack triggering this error... we wouldn't want a weight on it anyway
559                if *ctx as usize >= logits.len() {
560                    continue;
561                }
562                counts[*ctx as usize] += 1.0;
563            }
564
565            for (token_id, logit) in logits.iter_mut().enumerate() {
566                let count = counts[token_id];
567                *logit = *logit
568                    - count * frequency_penalty
569                    - if count > 0.0 { 1. } else { 0. } * presence_penalty;
570            }
571        }
572        Ok(())
573    }

575    fn apply_dry_penalty(&self, logits: &mut [f32], context: &[u32]) -> Result<()> {
576        if let Some(ref params) = self.dry_params {
577            if params.multiplier == 0. {
578                return Ok(());
579            }
580
581            let match_indices = context
582                .par_iter()
583                .enumerate()
584                .take(context.len() - 1)
585                .filter(|(_i, x)| *context.last().unwrap() == **x)
586                .map(|(i, _)| i)
587                .collect::<Vec<_>>();
588
589            let mut match_lengths = HashMap::new();
590
591            for i in match_indices {
592                let next_token = context[i + 1];
593
594                if params.sequence_breakers.contains(&next_token) {
595                    continue;
596                }
597
598                let mut match_length = 1;
599
600                // Limit match length to avoid quadratic runtime and potential DoS with adversarial inputs.
601                while match_length < 50 {
602                    if match_length > i {
603                        // Start of input
604                        break;
605                    }
606
607                    let j = i - match_length;
608
609                    let prev_tok = context[context.len() - (match_length + 1)];
610                    if context[j] != prev_tok {
611                        // Start of match reached
612                        break;
613                    }
614
615                    if params.sequence_breakers.contains(&prev_tok) {
616                        // Seq breaking tok reached
617                        break;
618                    }
619
620                    match_length += 1;
621                }
622
623                #[allow(clippy::map_entry)]
624                if match_lengths.contains_key(&next_token) {
625                    match_lengths.insert(next_token, match_length.max(match_lengths[&next_token]));
626                } else {
627                    match_lengths.insert(next_token, match_length);
628                }
629            }
630
631            // Actually apply penalties
632            for (tok, match_len) in match_lengths {
633                if match_len >= params.allowed_length {
634                    // Llama 3.2 uses a hack triggering this error... we wouldn't want a weight on it anyway
635                    if tok as usize >= logits.len() {
636                        continue;
637                    }
638                    let penalty = params.multiplier
639                        * params.base.powf((match_len - params.allowed_length) as f32);
640                    logits[tok as usize] -= penalty;
641                }
642            }
643        }
644        Ok(())
645    }

647    /// Sample the provided tokens.
648    ///
649    /// If the temperature is `None`, argmax sampling is used. Otherwise, the selected sampling is used.
650    /// With `top-p` sampling, if the `top-p` value is `<= 0.0` or `>= 1.0`, multinomial sampling is used.
651    pub fn sample(
652        &self,
653        logits: Tensor,
654        context: &[u32],
655        return_logprobs: bool,
656        rng: Arc<Mutex<Isaac64Rng>>,
657        sample_speculative: bool,
658    ) -> Result<Logprobs> {
659        let logits = logits.to_vec1()?;
660        let mut logits = self.apply_penalties(logits, context)?;
661        for processor in &self.logits_processors {
662            logits = processor.apply(&logits, context)?;
663        }
664        let next_token = if sample_speculative {
665            match self.temperature {
666                None => self.sample_speculative_top_kp_min_p(
667                    logits,
668                    return_logprobs,
669                    self.top_k,
670                    self.top_p as f32,
671                    self.min_p as f32,
672                )?,
673                Some(temperature) => {
674                    let logits = (&logits / temperature)?;
675                    let probs = candle_nn::ops::softmax_last_dim(&logits)?;
676
677                    self.sample_speculative_top_kp_min_p(
678                        probs,
679                        return_logprobs,
680                        self.top_k,
681                        self.top_p as f32,
682                        self.min_p as f32,
683                    )?
684                }
685            }
686        } else {
687            match self.temperature {
688                None => self.sample_argmax(logits, return_logprobs)?,
689                Some(temperature) => {
690                    let logits = (&logits / temperature)?;
691                    let logits = candle_nn::ops::softmax_last_dim(&logits)?;
692                    let mut probs: Vec<f32> = logits.to_vec1()?;
693
694                    self.sample_top_kp_min_p(
695                        &mut probs,
696                        self.top_k,
697                        self.top_p as f32,
698                        self.min_p as f32,
699                        return_logprobs,
700                        rng,
701                    )?
702                }
703            }
704        };
705        Ok(next_token)
706    }
707}
