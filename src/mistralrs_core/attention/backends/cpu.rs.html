42/// Elevate the thread QoS so macOS prefers running it on Performance (P) cores.
43unsafe fn set_thread_affinity() {
44    // USER_INTERACTIVE has the highest scheduling priority that user code
45    // can request and is most likely to be scheduled on P‑cores.
46    use libc::{pthread_set_qos_class_self_np, qos_class_t::QOS_CLASS_USER_INTERACTIVE};
47    // The second argument is a relative priority within the QoS class (0 = default).
48    pthread_set_qos_class_self_np(QOS_CLASS_USER_INTERACTIVE, 0);
49}

57/// Rayon pool used by the flash‑attention CPU kernels, with a per‑thread
58/// start handler that applies our affinity hint exactly once.
59static FLASH_ATTN_POOL: LazyLock<ThreadPool> = LazyLock::new(|| {
60    rayon::ThreadPoolBuilder::new()
61        .start_handler(|_| unsafe {
62            set_thread_affinity();
63        })
64        .build()
65        .expect("Failed to build custom Rayon thread‑pool for flash‑attention")

68const DOT_CHUNK: usize = 4;

70/// Size (in KV positions) processed by each inner‑tile job.
71const TILE_KV: usize = 16;

73/// Unrolled dot‑product that works purely in `f32`.
74#[inline]
75fn vec_dot_f32<T: UpcastF32 + Copy>(a: &[T], b: &[T]) -> f32 {
76    let mut sum = 0f32;
77    let chunks = a.len() / DOT_CHUNK;
78    for i in 0..chunks {
79        let i_chunk = i * DOT_CHUNK;
80        sum += a[i_chunk].to_f32() * b[i_chunk].to_f32()
81            + a[i_chunk + 1].to_f32() * b[i_chunk + 1].to_f32()
82            + a[i_chunk + 2].to_f32() * b[i_chunk + 2].to_f32()
83            + a[i_chunk + 3].to_f32() * b[i_chunk + 3].to_f32();
84    }
85    for i in (chunks * DOT_CHUNK)..a.len() {
86        sum += a[i].to_f32() * b[i].to_f32();
87    }
88    sum
89}

140pub fn run_flash_attn_cpu<T>(
141    q: &Tensor,
142    k: &Tensor,
143    v: &Tensor,
144    mask: Option<&Tensor>,
145    sdpa_params: &SdpaParams,
146) -> Result<Tensor>
147where
148    T: WithDType + Sum + num_traits::real::Real + DowncastF32 + UpcastF32,
149{
150    // Inline CPU slice extraction for q, k, v, and optional mask
151    let (q_guard, q_layout) = q.storage_and_layout();
152    let q_data: &[T] = if let Storage::Cpu(cpu) = &*q_guard {
153        let data = cpu.as_slice::<T>().context("Expected CPU storage for q")?;
154        &data[q_layout.start_offset()..]
155    } else {
156        return Err(candle_core::Error::Msg("Expected CPU storage for q".into()));
157    };
158    let (k_guard, k_layout) = k.storage_and_layout();
159    let k_data: &[T] = if let Storage::Cpu(cpu) = &*k_guard {
160        let data = cpu.as_slice::<T>().context("Expected CPU storage for k")?;
161        &data[k_layout.start_offset()..]
162    } else {
163        return Err(candle_core::Error::Msg("Expected CPU storage for k".into()));
164    };
165    let (v_guard, v_layout) = v.storage_and_layout();
166    let v_data: &[T] = if let Storage::Cpu(cpu) = &*v_guard {
167        let data = cpu.as_slice::<T>().context("Expected CPU storage for v")?;
168        &data[v_layout.start_offset()..]
169    } else {
170        return Err(candle_core::Error::Msg("Expected CPU storage for v".into()));
171    };
172    let mask_guard = mask.map(|mask| mask.storage_and_layout().0);
173    let mask_data: Option<&[T]> = if let Some(mask_guard) = &mask_guard {
174        let mask = mask.as_ref().unwrap();
175
176        if let Storage::Cpu(cpu) = &**mask_guard {
177            let data = cpu
178                .as_slice::<T>()
179                .context("Expected CPU storage for mask")?;
180            Some(&data[mask.layout().start_offset()..])
181        } else {
182            return Err(candle_core::Error::Msg(
183                "Expected CPU storage for mask".into(),
184            ));
185        }
186    } else {
187        None
188    };
189    // q_guard, k_guard, v_guard, and m_guard (if any) are kept in scope to hold storage alive
190
191    let q_stride = q.stride();
192    let k_stride = k.stride();
193    let v_stride = v.stride();
194
195    // Fast path for decode: q_len == 1
196    if q.shape().dims()[1] == 1 {
197        return flash_attn_cpu_single_q(
198            q_data,
199            k_data,
200            v_data,
201            mask_data,
202            q.shape().dims(),
203            k.shape().dims(),
204            v.shape().dims(),
205            q_stride,
206            k_stride,
207            v_stride,
208            sdpa_params.softmax_scale,
209            0.0,
210            sdpa_params.softcap.unwrap_or(0.0),
211        );
212    }
213
214    flash_attn_cpu(
215        q_data,
216        k_data,
217        v_data,
218        mask_data,
219        q.shape().dims(),
220        k.shape().dims(),
221        v.shape().dims(),
222        q_stride,
223        k_stride,
224        v_stride,
225        sdpa_params.softmax_scale,
226        0.0,
227        sdpa_params.softcap.unwrap_or(0.0),
228    )
229}

231/// Optimised path for the common decode case: q_len == 1 but kv_len ≫ 1.
232/// All intermediate maths is done in `f32`; we down‑cast to `T` exactly once
233/// when writing the output buffer.
234#[allow(clippy::too_many_arguments)]
235fn flash_attn_cpu_single_q<T>(
236    q_data: &[T],
237    k_data: &[T],
238    v_data: &[T],
239    mask_vec: Option<&[T]>,
240    qshape: &[usize],
241    kshape: &[usize],
242    vshape: &[usize],
243    qstride: &[usize],
244    kstride: &[usize],
245    vstride: &[usize],
246    scale: f32,
247    max_bias: f32,
248    logit_softcap: f32,
249) -> Result<Tensor>
250where
251    T: WithDType + Copy + UpcastF32 + DowncastF32,
252{
253    let (b, _qlen, h, d) = (qshape[0], qshape[1], qshape[2], qshape[3]);
254    let kv_len = kshape[1];
255    let kv_tiles = kv_len.div_ceil(TILE_KV);
256    let dv = d;
257
258    // --- head tiling factors and softmax helper -----------------------
259    let k_h = kshape[2];
260    let v_h = vshape[2];
261    let rk2 = h / k_h;
262    let rv2 = h / v_h;
263    let n2 = 2_usize.pow((h as f32).log2().ceil() as u32);
264
265    // final output
266    let mut out = vec![T::cast(0.0); b * h * dv];
267    // scratch: one tile  ×  heads ×  dv  (flattened)
268    let mut scratch = vec![0f32; kv_tiles * h * dv];
269    let mut scratch_s = vec![0f32; kv_tiles * h]; // softmax S per tile
270    let mut scratch_m = vec![f32::NEG_INFINITY; kv_tiles * h];
271
272    // Store as plain usize so the capture is Sync (raw *mut T is not Sync)
273    let scratch_ptr = scratch.as_mut_ptr() as usize;
274    let scratch_s_ptr = scratch_s.as_mut_ptr() as usize;
275    let scratch_m_ptr = scratch_m.as_mut_ptr() as usize;
276
277    FLASH_ATTN_POOL.install(|| {
278        (0..kv_tiles).into_par_iter().for_each(|tile_idx| {
279            let start = tile_idx * TILE_KV;
280            let end = (start + TILE_KV).min(kv_len);
281
282            for b_i in 0..b {
283                for h_i in 0..h {
284                    // Per‑head slope for ALiBi / NTK bias.
285                    let slope = if max_bias > 0.0 {
286                        2.0f32.powf(-max_bias * ((h_i + 1) as f32) / n2 as f32)
287                    } else {
288                        1.0
289                    };
290                    let k_head = h_i / rk2;
291                    let v_head = h_i / rv2;
292
293                    // pointers into scratch
294                    let base_vk = ((tile_idx * h + h_i) * dv) as isize;
295                    let base_s = (tile_idx * h + h_i) as isize;
296
297                    let q_base = b_i * qstride[0] + h_i * qstride[2];
298                    let q_row = &q_data[q_base..q_base + d];
299
300                    // Re‑materialise the pointer from the captured usize.
301                    let vkq: &mut [f32] = unsafe {
302                        std::slice::from_raw_parts_mut(
303                            (scratch_ptr as *mut f32).add(base_vk as usize),
304                            dv,
305                        )
306                    };
307                    let mut s = 0f32;
308                    let mut m = f32::NEG_INFINITY;
309
310                    for kv_pos in start..end {
311                        #[cfg(any(target_arch = "x86_64", target_arch = "aarch64"))]
312                        {
313                            // One‑step look‑ahead prefetch of the next K and V rows.
314                            let next_pos = kv_pos + 1;
315                            if next_pos < end {
316                                let next_k_base =
317                                    b_i * kstride[0] + next_pos * kstride[1] + k_head * kstride[2];
318                                let next_v_base =
319                                    b_i * vstride[0] + next_pos * vstride[1] + v_head * vstride[2];
320                                unsafe {
321                                    prefetch(k_data.as_ptr().add(next_k_base) as *const u8);
322                                    prefetch(v_data.as_ptr().add(next_v_base) as *const u8);
323                                }
324                            }
325                        }
326                        // ---------- dot(Q,K) ----------
327                        let k_base = b_i * kstride[0] + kv_pos * kstride[1] + k_head * kstride[2];
328                        let k_row = &k_data[k_base..k_base + d];
329
330                        let mut s_val = vec_dot_f32(q_row, k_row) * scale;
331
332                        if logit_softcap != 0.0 {
333                            s_val = logit_softcap * (s_val / logit_softcap).tanh();
334                        }
335                        if let Some(mv_vec) = mask_vec {
336                            s_val += slope * mv_vec[(b_i * kv_len) + kv_pos].to_f32();
337                        }
338
339                        // -------- tile-local softmax ----------
340                        let m_old = m;
341                        let mut ms = 1.0;
342                        let mut vs = 1.0;
343                        if s_val > m {
344                            m = s_val;
345                            ms = (m_old - m).exp();
346                            for v in vkq.iter_mut() {
347                                *v *= ms;
348                            }
349                        } else {
350                            vs = (s_val - m).exp();
351                        }
352
353                        // -------- add V ----------
354                        let v_base = b_i * vstride[0] + kv_pos * vstride[1] + v_head * vstride[2];
355                        for d_i in 0..dv {
356                            vkq[d_i] += v_data[v_base + d_i].to_f32() * vs;
357                        }
358
359                        s = s * ms + vs;
360                    }
361
362                    unsafe {
363                        *((scratch_s_ptr as *mut f32).add(base_s as usize)) = s;
364                        *((scratch_m_ptr as *mut f32).add(base_s as usize)) = m;
365                    }
366                }
367            }
368        });
369    });
370
371    // -------- serial reduction over tiles (per head) --------
372    for b_i in 0..b {
373        for h_i in 0..h {
374            let out_off = (b_i * h + h_i) * dv;
375
376            // start with tile 0
377            let mut vkq = vec![0f32; dv];
378            vkq.copy_from_slice(&scratch[(h_i * dv)..(h_i * dv + dv)]);
379            let mut s = scratch_s[h_i];
380            let mut m = scratch_m[h_i];
381
382            for tile_idx in 1..kv_tiles {
383                let base = (tile_idx * h + h_i) * dv;
384                let base_s = tile_idx * h + h_i;
385
386                let m_b = scratch_m[base_s];
387                let s_b = scratch_s[base_s];
388                let vkq_b = &scratch[base..base + dv];
389
390                if m >= m_b {
391                    let factor = (m_b - m).exp();
392                    for (v, v_b) in vkq.iter_mut().zip(vkq_b) {
393                        *v += v_b * factor;
394                    }
395                    s += s_b * factor;
396                } else {
397                    let factor = (m - m_b).exp();
398                    for (v, v_a) in vkq.iter_mut().zip(vkq_b) {
399                        *v = v_a + *v * factor;
400                    }
401                    s = s_b + s * factor;
402                    m = m_b;
403                }
404            }
405
406            let inv_s = 1.0 / s;
407            for (o, v) in out[out_off..][..dv].iter_mut().zip(vkq.iter()) {
408                *o = T::cast(*v * inv_s);
409            }
410        }
411    }
412
413    Tensor::from_vec(out, (b, h, 1, dv), &Device::Cpu)
414}

416/// Main forward flash-attention CPU routine.
417/// Shapes follow Candle convention: (B, S, H, D)
418#[allow(clippy::too_many_arguments)]
419fn flash_attn_cpu<T>(
420    q_data: &[T],
421    k_data: &[T],
422    v_data: &[T],
423    mask_vec: Option<&[T]>,
424    qshape: &[usize],
425    kshape: &[usize],
426    vshape: &[usize],
427    qstride: &[usize],
428    kstride: &[usize],
429    vstride: &[usize],
430    scale: f32,
431    max_bias: f32,
432    logit_softcap: f32,
433) -> Result<Tensor>
434where
435    T: WithDType + Copy + UpcastF32 + DowncastF32,
436{
437    let (b, q_len, h, d) = (qshape[0], qshape[1], qshape[2], qshape[3]);
438    let kv_len = kshape[1];
439    let k_h = kshape[2];
440    let v_h = vshape[2];
441    let rk2 = h / k_h;
442    let rv2 = h / v_h;
443    let dv = d;
444
445    let n2 = 2_usize.pow((h as f32).log2().ceil() as u32);
446
447    let mut out = vec![T::cast(0.0); b * q_len * h * dv];
448
449    assert_eq!(qstride[3], 1, "q must have contiguous rows");
450    assert_eq!(kstride[3], 1, "k must have contiguous rows");
451    assert_eq!(vstride[3], 1, "v must have contiguous rows");
452
453    FLASH_ATTN_POOL.install(|| {
454        out.par_chunks_mut(dv)
455            .with_min_len(64)
456            .enumerate()
457            .for_each(|(row_idx, out_chunk)| {
458                // Decode flat index back to (batch, head, q_pos)
459                let rows_per_batch = h * q_len;
460                let b_i = row_idx / rows_per_batch;
461                let rem = row_idx % rows_per_batch;
462                let h_i = rem / q_len;
463                let q_pos = rem % q_len;
464
465                let slope = if max_bias > 0.0 {
466                    2.0f32.powf(-max_bias * ((h_i + 1) as f32) / n2 as f32)
467                } else {
468                    1.0
469                };
470
471                let k_head = h_i / rk2;
472                let v_head = h_i / rv2;
473
474                let mut vkq = vec![0f32; dv];
475                let mut s = 0.0f32;
476                let mut m = f32::NEG_INFINITY;
477
478                // Q row
479                let q_base = b_i * qstride[0] + q_pos * qstride[1] + h_i * qstride[2];
480                let q_row = &q_data[q_base..q_base + d];
481
482                for kv_pos in 0..kv_len {
483                    // -------- ping‑pong prefetch of next KV row --------
484                    #[cfg(any(target_arch = "x86_64", target_arch = "aarch64"))]
485                    {
486                        let next_pos = kv_pos + 1;
487                        if next_pos < kv_len {
488                            let next_k_base =
489                                b_i * kstride[0] + next_pos * kstride[1] + k_head * kstride[2];
490                            let next_v_base =
491                                b_i * vstride[0] + next_pos * vstride[1] + v_head * vstride[2];
492                            unsafe {
493                                prefetch(k_data.as_ptr().add(next_k_base) as *const u8);
494                                prefetch(v_data.as_ptr().add(next_v_base) as *const u8);
495                            }
496                        }
497                    }
498                    // ----------------------------------------------------
499                    // Mask (optional)
500                    let mv = if let Some(mv_vec) = mask_vec {
501                        slope * mv_vec[((b_i * q_len + q_pos) * kv_len) + kv_pos].to_f32()
502                    } else {
503                        0.0
504                    };
505                    if mv == f32::NEG_INFINITY {
506                        continue;
507                    }
508
509                    // K row
510                    let k_base = b_i * kstride[0] + kv_pos * kstride[1] + k_head * kstride[2];
511                    let k_row = &k_data[k_base..k_base + d];
512
513                    // dot(Q, K)
514                    let mut s_val = vec_dot_f32(q_row, k_row);
515                    let mut scale_applied = scale;
516                    if logit_softcap != 0.0 {
517                        scale_applied /= logit_softcap;
518                    }
519                    s_val *= scale_applied;
520                    if logit_softcap != 0.0 {
521                        s_val = logit_softcap * s_val.tanh();
522                    }
523                    s_val += mv;
524
525                    // online softmax
526                    let m_old = m;
527                    let mut ms = 1.0f32;
528                    let mut vs = 1.0f32;
529                    if s_val > m {
530                        m = s_val;
531                        ms = (m_old - m).exp();
532                        for v in vkq.iter_mut() {
533                            *v *= ms;
534                        }
535                    } else {
536                        vs = (s_val - m).exp();
537                    }
538
539                    // V row (strided) as f32
540                    let v_base = b_i * vstride[0] + kv_pos * vstride[1] + v_head * vstride[2];
541                    for d_i in 0..dv {
542                        vkq[d_i] += v_data[v_base + d_i].to_f32() * vs;
543                    }
544
545                    s = s * ms + vs;
546                }
547
548                // final normalisation + downcast
549                let inv_s = 1.0 / s;
550                for (o, v) in out_chunk.iter_mut().zip(vkq.iter()) {
551                    *o = T::cast(*v * inv_s);
552                }
553            });
554    });
555
556    let out_shape = (b, h, q_len, dv);
557    Tensor::from_vec(out, out_shape, &Device::Cpu)
558}
