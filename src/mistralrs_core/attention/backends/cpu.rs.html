18#[cfg(target_arch = "x86_64")]
19#[inline(always)]
20unsafe fn prefetch(addr: *const u8) {
21    _mm_prefetch(addr as *const i8, core::arch::x86_64::_MM_HINT_T0);

25#[inline(always)]
26unsafe fn prefetch(addr: *const u8) {
27    core::arch::asm!(
28        "prfm pldl1keep, [{0}]",
29        in(reg) addr,
30        options(readonly, nostack, preserves_flags)
31    );
32}

42/// Elevate the thread QoS so macOS prefers running it on Performance (P) cores.
43unsafe fn set_thread_affinity() {
44    // USER_INTERACTIVE has the highest scheduling priority that user code
45    // can request and is most likely to be scheduled on P‑cores.
46    use libc::{pthread_set_qos_class_self_np, qos_class_t::QOS_CLASS_USER_INTERACTIVE};
47    // The second argument is a relative priority within the QoS class (0 = default).
48    pthread_set_qos_class_self_np(QOS_CLASS_USER_INTERACTIVE, 0);
49}

57/// Rayon pool used by the flash‑attention CPU kernels, with a per‑thread
58/// start handler that applies our affinity hint exactly once.
59static FLASH_ATTN_POOL: LazyLock<ThreadPool> = LazyLock::new(|| {
60    rayon::ThreadPoolBuilder::new()
61        .start_handler(|_| unsafe {
62            set_thread_affinity();
63        })
64        .build()
65        .expect("Failed to build custom Rayon thread‑pool for flash‑attention")
66});

68const DOT_CHUNK: usize = 4;
69
70/// Size (in KV positions) processed by each inner‑tile job.
71const TILE_KV: usize = 16;

73/// Unrolled dot‑product that works purely in `f32`.
74#[inline]
75fn vec_dot_f32<T: UpcastF32 + Copy>(a: &[T], b: &[T]) -> f32 {
76    let mut sum = 0f32;
77    let chunks = a.len() / DOT_CHUNK;
78    for i in 0..chunks {
79        let i_chunk = i * DOT_CHUNK;
80        sum += a[i_chunk].to_f32() * b[i_chunk].to_f32()
81            + a[i_chunk + 1].to_f32() * b[i_chunk + 1].to_f32()
82            + a[i_chunk + 2].to_f32() * b[i_chunk + 2].to_f32()
83            + a[i_chunk + 3].to_f32() * b[i_chunk + 3].to_f32();
84    }
85    for i in (chunks * DOT_CHUNK)..a.len() {
86        sum += a[i].to_f32() * b[i].to_f32();
87    }
88    sum
89}

140pub fn run_flash_attn_cpu<T>(
141    q: &Tensor,
142    k: &Tensor,
143    v: &Tensor,
144    mask: Option<&Tensor>,
145    sdpa_params: &SdpaParams,
146) -> Result<Tensor>
147where
148    T: WithDType + Sum + num_traits::real::Real + DowncastF32 + UpcastF32,
149{
150    // Inline CPU slice extraction for q, k, v, and optional mask
151    let (q_guard, q_layout) = q.storage_and_layout();
152    let q_data: &[T] = if let Storage::Cpu(cpu) = &*q_guard {
153        let data = cpu.as_slice::<T>().context("Expected CPU storage for q")?;
154        &data[q_layout.start_offset()..]
155    } else {
156        return Err(candle_core::Error::Msg("Expected CPU storage for q".into()));
157    };
158    let (k_guard, k_layout) = k.storage_and_layout();
159    let k_data: &[T] = if let Storage::Cpu(cpu) = &*k_guard {
160        let data = cpu.as_slice::<T>().context("Expected CPU storage for k")?;
161        &data[k_layout.start_offset()..]
162    } else {
163        return Err(candle_core::Error::Msg("Expected CPU storage for k".into()));
164    };
165    let (v_guard, v_layout) = v.storage_and_layout();
166    let v_data: &[T] = if let Storage::Cpu(cpu) = &*v_guard {
167        let data = cpu.as_slice::<T>().context("Expected CPU storage for v")?;
168        &data[v_layout.start_offset()..]
169    } else {
170        return Err(candle_core::Error::Msg("Expected CPU storage for v".into()));
171    };
172    let mask_guard = mask.map(|mask| mask.storage_and_layout().0);
173    let mask_data: Option<&[T]> = if let Some(mask_guard) = &mask_guard {
174        let mask = mask.as_ref().unwrap();
175
176        if let Storage::Cpu(cpu) = &**mask_guard {
177            let data = cpu
178                .as_slice::<T>()
179                .context("Expected CPU storage for mask")?;
180            Some(&data[mask.layout().start_offset()..])
181        } else {
182            return Err(candle_core::Error::Msg(
183                "Expected CPU storage for mask".into(),
184            ));
185        }
186    } else {
187        None
188    };
189    // q_guard, k_guard, v_guard, and m_guard (if any) are kept in scope to hold storage alive
190
191    let q_stride = q.stride();
192    let k_stride = k.stride();
193    let v_stride = v.stride();
194
195    // Fast path for decode: q_len == 1
196    if q.shape().dims()[1] == 1 {
197        return flash_attn_cpu_single_q(
198            q_data,
199            k_data,
200            v_data,
201            mask_data,
202            q.shape().dims(),
203            k.shape().dims(),
204            v.shape().dims(),
205            q_stride,
206            k_stride,
207            v_stride,
208            sdpa_params.softmax_scale,
209            0.0,
210            sdpa_params.softcap.unwrap_or(0.0),
211        );
212    }
213
214    flash_attn_cpu(
215        q_data,
216        k_data,
217        v_data,
218        mask_data,
219        q.shape().dims(),
220        k.shape().dims(),
221        v.shape().dims(),
222        q_stride,
223        k_stride,
224        v_stride,
225        sdpa_params.softmax_scale,
226        0.0,
227        sdpa_params.softcap.unwrap_or(0.0),
228    )
229}

231/// Optimised path for the common decode case: q_len == 1 but kv_len ≫ 1.
232/// All intermediate maths is done in `f32`; we down‑cast to `T` exactly once
233/// when writing the output buffer.
234#[allow(clippy::too_many_arguments)]
235fn flash_attn_cpu_single_q<T>(
236    q_data: &[T],
237    k_data: &[T],
238    v_data: &[T],
239    mask_vec: Option<&[T]>,
240    qshape: &[usize],
241    kshape: &[usize],
242    vshape: &[usize],
243    qstride: &[usize],
244    kstride: &[usize],
245    vstride: &[usize],
246    scale: f32,
247    max_bias: f32,
248    logit_softcap: f32,
249) -> Result<Tensor>
250where
251    T: WithDType + Copy + UpcastF32 + DowncastF32,
252{
253    // Shapes: (B, 1, H, D)
254    let (b, _q_len, h, d) = (qshape[0], qshape[1], qshape[2], qshape[3]);
255    let kv_len = kshape[1];
256    let k_h = kshape[2];
257    let v_h = vshape[2];
258    let rk2 = h / k_h;
259    let rv2 = h / v_h;
260    let dv = d;
261
262    let n2 = 2_usize.pow((h as f32).log2().ceil() as u32);
263
264    // Final output buffer (B, H, 1, D) – already typed as `T`
265    let mut out = vec![T::cast(0.0); b * h * dv];
266
267    // Split the KV axis into cache‑friendly tiles.
268    let kv_tiles = kv_len.div_ceil(TILE_KV);
269
270    assert_eq!(qstride[3], 1, "q must have contiguous rows");
271    assert_eq!(kstride[3], 1, "k must have contiguous rows");
272    assert_eq!(vstride[3], 1, "v must have contiguous rows");
273
274    FLASH_ATTN_POOL.install(|| {
275        out.par_chunks_mut(dv)
276            .with_min_len(64)
277            .enumerate()
278            .for_each(|(row_idx, out_chunk)| {
279                let b_i = row_idx / h;
280                let h_i = row_idx % h;
281
282                let slope = if max_bias > 0.0 {
283                    2.0f32.powf(-max_bias * ((h_i + 1) as f32) / n2 as f32)
284                } else {
285                    1.0
286                };
287
288                let k_head = h_i / rk2;
289                let v_head = h_i / rv2;
290
291                // ---------------- map‑reduce over KV tiles -----------------
292                let (vkq, s_tot, _m_tot) = (0..kv_tiles)
293                    .into_par_iter()
294                    .map(|tile_idx| {
295                        let start = tile_idx * TILE_KV;
296                        let end = (start + TILE_KV).min(kv_len);
297
298                        let mut vkq = vec![0f32; dv];
299                        let mut s = 0.0f32;
300                        let mut m = f32::NEG_INFINITY;
301
302                        // Q row is already contiguous in memory for q_len == 1
303                        let q_base = b_i * qstride[0] + h_i * qstride[2];
304                        let q_row = &q_data[q_base..q_base + d];
305
306                        for kv_pos in start..end {
307                            // -------- ping‑pong prefetch of next KV row --------
308                            #[cfg(any(target_arch = "x86_64", target_arch = "aarch64"))]
309                            {
310                                let next_pos = kv_pos + 1;
311                                if next_pos < end {
312                                    let next_k_base = b_i * kstride[0]
313                                        + next_pos * kstride[1]
314                                        + k_head * kstride[2];
315                                    let next_v_base = b_i * vstride[0]
316                                        + next_pos * vstride[1]
317                                        + v_head * vstride[2];
318                                    unsafe {
319                                        prefetch(k_data.as_ptr().add(next_k_base) as *const u8);
320                                        prefetch(v_data.as_ptr().add(next_v_base) as *const u8);
321                                    }
322                                }
323                            }
324                            // ----------------------------------------------------
325                            // ---- mask -------------------------------------------------------
326                            let mv = if let Some(mv_vec) = mask_vec {
327                                slope * mv_vec[(b_i * kv_len) + kv_pos].to_f32()
328                            } else {
329                                0.0
330                            };
331                            if mv == f32::NEG_INFINITY {
332                                continue;
333                            }
334
335                            // ---- K row -------------------------------------------------------
336                            let k_base =
337                                b_i * kstride[0] + kv_pos * kstride[1] + k_head * kstride[2];
338                            let k_row = &k_data[k_base..k_base + d];
339
340                            // dot(Q, K)
341                            let mut s_val = vec_dot_f32(q_row, k_row);
342
343                            let mut scale_applied = scale;
344                            if logit_softcap != 0.0 {
345                                scale_applied /= logit_softcap;
346                            }
347                            s_val *= scale_applied;
348                            if logit_softcap != 0.0 {
349                                s_val = logit_softcap * s_val.tanh();
350                            }
351                            s_val += mv;
352
353                            // ---- tile‑local online soft‑max -------------------------------
354                            let m_old = m;
355                            let mut ms = 1.0f32;
356                            let mut vs = 1.0f32;
357                            if s_val > m {
358                                m = s_val;
359                                ms = (m_old - m).exp();
360                                for v in vkq.iter_mut() {
361                                    *v *= ms;
362                                }
363                            } else {
364                                vs = (s_val - m).exp();
365                            }
366
367                            // ---- V row -------------------------------------------------------
368                            let v_base =
369                                b_i * vstride[0] + kv_pos * vstride[1] + v_head * vstride[2];
370                            for d_i in 0..dv {
371                                vkq[d_i] += v_data[v_base + d_i].to_f32() * vs;
372                            }
373
374                            s = s * ms + vs;
375                        }
376
377                        (vkq, s, m)
378                    })
379                    // --------------- reduce two tiles --------------------------------------
380                    .reduce(
381                        || (vec![0f32; dv], 0.0f32, f32::NEG_INFINITY),
382                        |mut a, b| {
383                            let (ref mut vkq_a, mut s_a, m_a) = a;
384                            let (vkq_b, s_b, m_b) = b;
385                            if m_a >= m_b {
386                                let factor = (m_b - m_a).exp();
387                                for (va, vb) in vkq_a.iter_mut().zip(vkq_b) {
388                                    *va += vb * factor;
389                                }
390                                s_a += s_b * factor;
391                                (vkq_a.clone(), s_a, m_a)
392                            } else {
393                                let factor = (m_a - m_b).exp();
394                                let mut vkq_new = vkq_b;
395                                for (vb, va) in vkq_new.iter_mut().zip(vkq_a) {
396                                    *vb += *va * factor;
397                                }
398                                (vkq_new, s_b + s_a * factor, m_b)
399                            }
400                        },
401                    );
402
403                // ---------------- final normalisation + down‑cast -------------
404                let inv_s = 1.0 / s_tot;
405                for (o, v) in out_chunk.iter_mut().zip(vkq.iter()) {
406                    *o = T::cast(*v * inv_s);
407                }
408            });
409    });
410
411    let out_shape = (b, h, 1usize, dv);
412    Tensor::from_vec(out, out_shape, &Device::Cpu)
413}

415/// Main forward flash-attention CPU routine.
416/// Shapes follow Candle convention: (B, S, H, D)
417#[allow(clippy::too_many_arguments)]
418fn flash_attn_cpu<T>(
419    q_data: &[T],
420    k_data: &[T],
421    v_data: &[T],
422    mask_vec: Option<&[T]>,
423    qshape: &[usize],
424    kshape: &[usize],
425    vshape: &[usize],
426    qstride: &[usize],
427    kstride: &[usize],
428    vstride: &[usize],
429    scale: f32,
430    max_bias: f32,
431    logit_softcap: f32,
432) -> Result<Tensor>
433where
434    T: WithDType + Copy + UpcastF32 + DowncastF32,
435{
436    let (b, q_len, h, d) = (qshape[0], qshape[1], qshape[2], qshape[3]);
437    let kv_len = kshape[1];
438    let k_h = kshape[2];
439    let v_h = vshape[2];
440    let rk2 = h / k_h;
441    let rv2 = h / v_h;
442    let dv = d;
443
444    let n2 = 2_usize.pow((h as f32).log2().ceil() as u32);
445
446    let mut out = vec![T::cast(0.0); b * q_len * h * dv];
447
448    assert_eq!(qstride[3], 1, "q must have contiguous rows");
449    assert_eq!(kstride[3], 1, "k must have contiguous rows");
450    assert_eq!(vstride[3], 1, "v must have contiguous rows");
451
452    FLASH_ATTN_POOL.install(|| {
453        out.par_chunks_mut(dv)
454            .with_min_len(64)
455            .enumerate()
456            .for_each(|(row_idx, out_chunk)| {
457                // Decode flat index back to (batch, head, q_pos)
458                let rows_per_batch = h * q_len;
459                let b_i = row_idx / rows_per_batch;
460                let rem = row_idx % rows_per_batch;
461                let h_i = rem / q_len;
462                let q_pos = rem % q_len;
463
464                let slope = if max_bias > 0.0 {
465                    2.0f32.powf(-max_bias * ((h_i + 1) as f32) / n2 as f32)
466                } else {
467                    1.0
468                };
469
470                let k_head = h_i / rk2;
471                let v_head = h_i / rv2;
472
473                let mut vkq = vec![0f32; dv];
474                let mut s = 0.0f32;
475                let mut m = f32::NEG_INFINITY;
476
477                // Q row
478                let q_base = b_i * qstride[0] + q_pos * qstride[1] + h_i * qstride[2];
479                let q_row = &q_data[q_base..q_base + d];
480
481                for kv_pos in 0..kv_len {
482                    // -------- ping‑pong prefetch of next KV row --------
483                    #[cfg(any(target_arch = "x86_64", target_arch = "aarch64"))]
484                    {
485                        let next_pos = kv_pos + 1;
486                        if next_pos < kv_len {
487                            let next_k_base =
488                                b_i * kstride[0] + next_pos * kstride[1] + k_head * kstride[2];
489                            let next_v_base =
490                                b_i * vstride[0] + next_pos * vstride[1] + v_head * vstride[2];
491                            unsafe {
492                                prefetch(k_data.as_ptr().add(next_k_base) as *const u8);
493                                prefetch(v_data.as_ptr().add(next_v_base) as *const u8);
494                            }
495                        }
496                    }
497                    // ----------------------------------------------------
498                    // Mask (optional)
499                    let mv = if let Some(mv_vec) = mask_vec {
500                        slope * mv_vec[((b_i * q_len + q_pos) * kv_len) + kv_pos].to_f32()
501                    } else {
502                        0.0
503                    };
504                    if mv == f32::NEG_INFINITY {
505                        continue;
506                    }
507
508                    // K row
509                    let k_base = b_i * kstride[0] + kv_pos * kstride[1] + k_head * kstride[2];
510                    let k_row = &k_data[k_base..k_base + d];
511
512                    // dot(Q, K)
513                    let mut s_val = vec_dot_f32(q_row, k_row);
514                    let mut scale_applied = scale;
515                    if logit_softcap != 0.0 {
516                        scale_applied /= logit_softcap;
517                    }
518                    s_val *= scale_applied;
519                    if logit_softcap != 0.0 {
520                        s_val = logit_softcap * s_val.tanh();
521                    }
522                    s_val += mv;
523
524                    // online softmax
525                    let m_old = m;
526                    let mut ms = 1.0f32;
527                    let mut vs = 1.0f32;
528                    if s_val > m {
529                        m = s_val;
530                        ms = (m_old - m).exp();
531                        for v in vkq.iter_mut() {
532                            *v *= ms;
533                        }
534                    } else {
535                        vs = (s_val - m).exp();
536                    }
537
538                    // V row (strided) as f32
539                    let v_base = b_i * vstride[0] + kv_pos * vstride[1] + v_head * vstride[2];
540                    for d_i in 0..dv {
541                        vkq[d_i] += v_data[v_base + d_i].to_f32() * vs;
542                    }
543
544                    s = s * ms + vs;
545                }
546
547                // final normalisation + downcast
548                let inv_s = 1.0 / s;
549                for (o, v) in out_chunk.iter_mut().zip(vkq.iter()) {
550                    *o = T::cast(*v * inv_s);
551                }
552            });
553    });
554
555    let out_shape = (b, h, q_len, dv);
556    Tensor::from_vec(out, out_shape, &Device::Cpu)
557}
