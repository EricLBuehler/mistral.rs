14/// Elevate the thread QoS so macOS prefers running it on Performance (P) cores.
15unsafe fn set_thread_affinity() {
16    // USER_INTERACTIVE has the highest scheduling priority that user code
17    // can request and is most likely to be scheduled on P‑cores.
18    use libc::{pthread_set_qos_class_self_np, qos_class_t::QOS_CLASS_USER_INTERACTIVE};
19    // The second argument is a relative priority within the QoS class (0 = default).
20    pthread_set_qos_class_self_np(QOS_CLASS_USER_INTERACTIVE, 0);
21}

29/// Rayon pool used by the flash‑attention CPU kernels, with a per‑thread
30/// start handler that applies our affinity hint exactly once.
31static FLASH_ATTN_POOL: LazyLock<ThreadPool> = LazyLock::new(|| {
32    rayon::ThreadPoolBuilder::new()
33        .start_handler(|_| unsafe {
34            set_thread_affinity();
35        })
36        .build()
37        .expect("Failed to build custom Rayon thread‑pool for flash‑attention")
38});

40const DOT_CHUNK: usize = 4;
41
42/// Size (in KV positions) processed by each inner‑tile job.
43const TILE_KV: usize = 16;
44
45#[inline]
46fn vec_dot<T: WithDType + Sum + Copy + std::ops::Mul<Output = T>>(a: &[T], b: &[T]) -> T {
47    let mut sum = T::zero();
48    let chunks = a.len() / DOT_CHUNK;
49
50    for i in 0..chunks {
51        let i_chunk = i * DOT_CHUNK;
52        sum = sum
53            + a[i_chunk] * b[i_chunk]
54            + a[i_chunk + 1] * b[i_chunk + 1]
55            + a[i_chunk + 2] * b[i_chunk + 2]
56            + a[i_chunk + 3] * b[i_chunk + 3];
57    }
58
59    for i in (chunks * DOT_CHUNK)..a.len() {
60        sum += a[i] * b[i];
61    }
62    sum
63}

65pub fn run_flash_attn_cpu<T>(
66    q: &Tensor,
67    k: &Tensor,
68    v: &Tensor,
69    mask: Option<&Tensor>,
70    sdpa_params: &SdpaParams,
71) -> Result<Tensor>
72where
73    T: WithDType + Sum + num_traits::real::Real,
74{
75    // Inline CPU slice extraction for q, k, v, and optional mask
76    let (q_guard, q_layout) = q.storage_and_layout();
77    let q_data: &[T] = if let Storage::Cpu(cpu) = &*q_guard {
78        let data = cpu.as_slice::<T>().context("Expected CPU storage for q")?;
79        &data[q_layout.start_offset()..]
80    } else {
81        return Err(candle_core::Error::Msg("Expected CPU storage for q".into()));
82    };
83    let (k_guard, k_layout) = k.storage_and_layout();
84    let k_data: &[T] = if let Storage::Cpu(cpu) = &*k_guard {
85        let data = cpu.as_slice::<T>().context("Expected CPU storage for k")?;
86        &data[k_layout.start_offset()..]
87    } else {
88        return Err(candle_core::Error::Msg("Expected CPU storage for k".into()));
89    };
90    let (v_guard, v_layout) = v.storage_and_layout();
91    let v_data: &[T] = if let Storage::Cpu(cpu) = &*v_guard {
92        let data = cpu.as_slice::<T>().context("Expected CPU storage for v")?;
93        &data[v_layout.start_offset()..]
94    } else {
95        return Err(candle_core::Error::Msg("Expected CPU storage for v".into()));
96    };
97    let mask_guard = mask.map(|mask| mask.storage_and_layout().0);
98    let mask_data: Option<&[T]> = if let Some(mask_guard) = &mask_guard {
99        let mask = mask.as_ref().unwrap();
100
101        if let Storage::Cpu(cpu) = &**mask_guard {
102            let data = cpu
103                .as_slice::<T>()
104                .context("Expected CPU storage for mask")?;
105            Some(&data[mask.layout().start_offset()..])
106        } else {
107            return Err(candle_core::Error::Msg(
108                "Expected CPU storage for mask".into(),
109            ));
110        }
111    } else {
112        None
113    };
114    // q_guard, k_guard, v_guard, and m_guard (if any) are kept in scope to hold storage alive
115
116    let q_stride = q.stride();
117    let k_stride = k.stride();
118    let v_stride = v.stride();
119
120    // Fast path for decode: q_len == 1
121    if q.shape().dims()[1] == 1 {
122        return flash_attn_cpu_single_q(
123            q_data,
124            k_data,
125            v_data,
126            mask_data,
127            q.shape().dims(),
128            k.shape().dims(),
129            v.shape().dims(),
130            q_stride,
131            k_stride,
132            v_stride,
133            sdpa_params.softmax_scale,
134            0.0,
135            sdpa_params.softcap.unwrap_or(0.0),
136        );
137    }
138
139    flash_attn_cpu(
140        q_data,
141        k_data,
142        v_data,
143        mask_data,
144        q.shape().dims(),
145        k.shape().dims(),
146        v.shape().dims(),
147        q_stride,
148        k_stride,
149        v_stride,
150        sdpa_params.softmax_scale,
151        0.0,
152        sdpa_params.softcap.unwrap_or(0.0),
153    )
154}

156/// Optimised path for the common decode case: q_len == 1 but kv_len ≫ 1.
157/// We drop the inner q‑position loop and parallelise over `(batch, head)`.
158#[allow(clippy::too_many_arguments)]
159fn flash_attn_cpu_single_q<T: WithDType + Sum + num_traits::real::Real>(
160    q_data: &[T],
161    k_data: &[T],
162    v_data: &[T],
163    mask_vec: Option<&[T]>,
164    qshape: &[usize],
165    kshape: &[usize],
166    vshape: &[usize],
167    qstride: &[usize],
168    kstride: &[usize],
169    vstride: &[usize],
170    scale: f32,
171    max_bias: f32,
172    logit_softcap: f32,
173) -> Result<Tensor> {
174    // Shapes: (B, 1, H, D)
175    let (b, _q_len, h, d) = (
176        qshape[0], qshape[1], // == 1
177        qshape[2], qshape[3],
178    );
179    let kv_len = kshape[1];
180    let k_h = kshape[2];
181    let v_h = vshape[2];
182    let rk2 = h / k_h;
183    let rv2 = h / v_h;
184    let dv = d;
185
186    let n2 = 2_usize.pow((h as f32).log2().ceil() as u32);
187
188    // Output buffer: (B, H, 1, D)
189    let mut out = vec![0f32; b * h * dv];
190
191    // Expose a second dimension of work: split the KV axis into tiles that
192    // fit in the last‑level cache and let Rayon schedule them.
193    let kv_tiles = kv_len.div_ceil(TILE_KV);
194
195    // SAFETY: `par_chunks_mut` hands out non‑overlapping &mut slices, so no two
196    // threads write the same output area.
197    FLASH_ATTN_POOL.install(|| {
198        out.par_chunks_mut(dv)
199            .with_min_len(64)
200            .enumerate()
201            .for_each(|(row_idx, out_chunk)| {
202                let b_i = row_idx / h;
203                let h_i = row_idx % h;
204
205                // ALiBi positional bias (standard formula)
206                let slope = if max_bias > 0.0 {
207                    2.0f32.powf(-max_bias * ((h_i + 1) as f32) / n2 as f32)
208                } else {
209                    1.0
210                };
211
212                // For grouped‑KV we collapse multiple query heads into the same K/V head.
213                let k_head = h_i / rk2;
214                let v_head = h_i / rv2;
215
216                // ------------------------------------------------------------------
217                // Nested parallelism: each KV tile is mapped independently, then we
218                // reduce the partial results with the correct soft‑max algebra.
219                // ------------------------------------------------------------------
220                let (vkq, s_tot, _m_tot) = (0..kv_tiles)
221                    .into_par_iter()
222                    .map(|tile_idx| {
223                        // ---- per‑tile scratch -------------------------------------------------
224                        let start = tile_idx * TILE_KV;
225                        let end = (start + TILE_KV).min(kv_len);
226
227                        let mut vkq = vec![0f32; dv];
228                        let mut s = 0.0f32;
229                        let mut m = f32::NEG_INFINITY;
230
231                        // ---------------- single‑Q row (already contiguous) -------------------
232                        let q_base =
233                            b_i * qstride[0] /*batch*/ + h_i * qstride[2] /*head*/;
234                        let q_row = &q_data[q_base..q_base + d];
235
236                        // ---------------- iterate over this KV slice --------------------------
237                        for kv_pos in start..end {
238                            // Mask
239                            let mv = if let Some(mv_vec) = mask_vec {
240                                let mval = mv_vec[(b_i * kv_len) + kv_pos];
241                                slope * mval.to_f64() as f32
242                            } else {
243                                0.0
244                            };
245                            if mv == f32::NEG_INFINITY {
246                                continue;
247                            }
248
249                            // K row
250                            let k_base =
251                                b_i * kstride[0] + kv_pos * kstride[1] + k_head * kstride[2];
252                            let k_row = &k_data[k_base..k_base + d];
253
254                            // dot(Q, K)
255                            let mut s_val = vec_dot::<T>(q_row, k_row).to_f64() as f32;
256
257                            let mut scale_applied = scale;
258                            if logit_softcap != 0.0 {
259                                scale_applied /= logit_softcap;
260                            }
261                            s_val *= scale_applied;
262                            if logit_softcap != 0.0 {
263                                s_val = logit_softcap * s_val.tanh();
264                            }
265                            s_val += mv;
266
267                            // Tile‑local online softmax ------------------------------------------
268                            let m_old = m;
269                            let mut ms = 1.0f32;
270                            let mut vs = 1.0f32;
271                            if s_val > m {
272                                m = s_val;
273                                ms = (m_old - m).exp();
274                                for v in vkq.iter_mut() {
275                                    *v *= ms;
276                                }
277                            } else {
278                                vs = (s_val - m).exp();
279                            }
280
281                            // V row
282                            let v_base =
283                                b_i * vstride[0] + kv_pos * vstride[1] + v_head * vstride[2];
284                            for d_i in 0..dv {
285                                vkq[d_i] += v_data[v_base + d_i * vstride[3]].to_f64() as f32 * vs;
286                            }
287
288                            s = s * ms + vs;
289                        }
290
291                        // Return per‑tile accumulator + softmax stats
292                        (vkq, s, m)
293                    })
294                    // -------- reduce two tiles -----------------------------------------------
295                    .reduce(
296                        || (vec![0f32; dv], 0.0f32, f32::NEG_INFINITY),
297                        |mut a, b| {
298                            let (ref mut vkq_a, mut s_a, m_a) = a;
299                            let (vkq_b, s_b, m_b) = b;
300                            if m_a >= m_b {
301                                let factor = (m_b - m_a).exp();
302                                for (va, vb) in vkq_a.iter_mut().zip(vkq_b) {
303                                    *va += vb * factor;
304                                }
305                                s_a += s_b * factor;
306                                (vkq_a.clone(), s_a, m_a)
307                            } else {
308                                let factor = (m_a - m_b).exp();
309                                let mut vkq_new = vkq_b;
310                                for (vb, va) in vkq_new.iter_mut().zip(vkq_a) {
311                                    *vb += *va * factor;
312                                }
313                                (vkq_new, s_b + s_a * factor, m_b)
314                            }
315                        },
316                    );
317
318                // ---------------- final normalisation ---------------------------------------
319                let inv_s = 1.0 / s_tot;
320                for v in out_chunk.iter_mut().zip(vkq.iter()) {
321                    *v.0 = *v.1 * inv_s;
322                }
323            });
324    });
325
326    let out_shape = (b, h, 1usize, dv);
327    Tensor::from_vec(out, out_shape, &Device::Cpu)
328}

330/// Main forward flash-attention CPU routine.
331/// Shapes follow Candle convention: (B, S, H, D)
332#[allow(clippy::too_many_arguments)]
333fn flash_attn_cpu<T: WithDType + Sum + num_traits::real::Real>(
334    q_data: &[T],
335    k_data: &[T],
336    v_data: &[T],
337    mask_vec: Option<&[T]>,
338    qshape: &[usize],
339    kshape: &[usize],
340    vshape: &[usize],
341    qstride: &[usize],
342    kstride: &[usize],
343    vstride: &[usize],
344    scale: f32,
345    max_bias: f32,
346    logit_softcap: f32,
347) -> Result<Tensor> {
348    let (b, q_len, h, d) = (qshape[0], qshape[1], qshape[2], qshape[3]);
349    let kv_len = kshape[1];
350    // --- Head broadcasting factors ----------------------------------------------------
351    // Allows K and V to have fewer heads than Q (grouped‑KV); the ratio is an
352    // integer factor.  rk2 = #Q‑heads / #K‑heads,  rv2 = #Q‑heads / #V‑heads.
353    let k_h = kshape[2];
354    let v_h = vshape[2];
355    let rk2 = h / k_h; // must divide exactly; panic otherwise
356    let rv2 = h / v_h;
357    let dv = d; // value dim = key dim in this kernel
358
359    // Precompute value for ALiBi slope calculation
360    let n2 = 2_usize.pow((h as f32).log2().ceil() as u32);
361
362    let mut out = vec![0f32; b * q_len * h * dv];
363
364    // ------------------------------------------------------------------
365    // Rayon‑parallel version: each (b_i, h_i, q_pos) row is independent.
366    // ------------------------------------------------------------------
367
368    let _rows = b * h * q_len; // total independent work items
369
370    // SAFETY: `par_chunks_mut` hands out non‑overlapping &mut [f32] slices,
371    // so no two threads can write the same output area.
372    FLASH_ATTN_POOL.install(|| {
373        out.par_chunks_mut(dv)
374            .with_min_len(64)
375            .enumerate()
376            .for_each(|(row_idx, out_chunk)| {
377                // Decode flat index back to (batch, head, q_pos)
378                let rows_per_batch = h * q_len;
379                let b_i = row_idx / rows_per_batch;
380                let rem = row_idx % rows_per_batch;
381                let h_i = rem / q_len;
382                let q_pos = rem % q_len;
383
384                let slope = if max_bias > 0.0 {
385                    2.0f32.powf(-max_bias * ((h_i + 1) as f32) / n2 as f32)
386                } else {
387                    1.0
388                };
389
390                // For grouped‑KV we collapse multiple query heads into the same K/V head.
391                let k_head = h_i / rk2;
392                let v_head = h_i / rv2;
393
394                // Buffers local to this row
395                let mut vkq = vec![0f32; dv];
396                let mut s = 0.0f32;
397                let mut m = f32::NEG_INFINITY;
398
399                // Allocate q_row and k_row once per row
400                let mut q_row: Vec<T> = Vec::with_capacity(d);
401                let mut k_row: Vec<T> = Vec::with_capacity(d);
402
403                // ------------------- gather Q (strided) --------------------
404                let q_base = b_i * qstride[0] + q_pos * qstride[1] + h_i * qstride[2];
405                q_row.clear();
406                for di in 0..d {
407                    q_row.push(q_data[q_base + di * qstride[3]]);
408                }
409
410                // ---------------- iterate over keys/values -----------------
411                for kv_pos in 0..kv_len {
412                    // Mask (optional)
413                    let mv = if let Some(mv_vec) = mask_vec {
414                        let mval = mv_vec[((b_i * q_len + q_pos) * kv_len) + kv_pos];
415                        slope * mval.to_f64() as f32
416                    } else {
417                        0.0
418                    };
419                    if mv == f32::NEG_INFINITY {
420                        continue;
421                    }
422
423                    // K row (strided)
424                    let k_base = b_i * kstride[0] + kv_pos * kstride[1] + k_head * kstride[2];
425                    k_row.clear();
426                    for di in 0..d {
427                        k_row.push(k_data[k_base + di * kstride[3]]);
428                    }
429
430                    // dot(Q, K)
431                    let mut s_val = vec_dot::<T>(&q_row, &k_row);
432                    let mut scale_applied = scale;
433                    if logit_softcap != 0.0 {
434                        scale_applied /= logit_softcap;
435                    }
436                    s_val *= T::from_f64(scale_applied as f64);
437                    if logit_softcap != 0.0 {
438                        s_val = T::from_f64(logit_softcap as f64 * s_val.to_f64().tanh());
439                    }
440                    s_val += T::from_f64(mv as f64);
441
442                    // online softmax
443                    let m_old = m;
444                    let mut ms = 1.0f32;
445                    let mut vs = 1.0f32;
446                    if s_val.to_f64() as f32 > m {
447                        m = s_val.to_f64() as f32;
448                        ms = (m_old - m).exp();
449                        for v in vkq.iter_mut() {
450                            *v *= ms;
451                        }
452                    } else {
453                        vs = (s_val.to_f64() as f32 - m).exp();
454                    }
455
456                    // V row (strided)
457                    let v_base = b_i * vstride[0] + kv_pos * vstride[1] + v_head * vstride[2];
458                    for d_i in 0..dv {
459                        vkq[d_i] += v_data[v_base + d_i * vstride[3]].to_f64() as f32 * vs;
460                    }
461
462                    s = s * ms + vs;
463                }
464
465                // ------------------- normalise & write out ------------------
466                let inv_s = 1.0 / s;
467                for v in vkq.iter_mut() {
468                    *v *= inv_s;
469                }
470                out_chunk.copy_from_slice(&vkq);
471            });
472    });
473
474    // Build output tensor with shape (B, H, S, D) to match standard (permute 0,2,1,3)
475    let out_shape = (b, h, q_len, dv);
476    Tensor::from_vec(out, out_shape, &Device::Cpu)
477}
