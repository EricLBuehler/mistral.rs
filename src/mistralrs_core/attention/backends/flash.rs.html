5#[cfg(feature = "flash-attn")]
6pub(crate) fn flash_attn(
7    q: &Tensor,
8    k: &Tensor,
9    v: &Tensor,
10    flash_params: Option<&crate::pipeline::text_models_inputs_processor::FlashParams>,
11    sdpa_params: &SdpaParams,
12) -> Result<Tensor> {
13    let (_b_sz, _n_attn_heads, seq_len, _head_dim) = q.dims4()?;
14    let window_size_left = sdpa_params.sliding_window;
15    let default_causal = seq_len > 1;
16
17    if let Some(params) = flash_params {
18        let window_size_right = if params.causal { Some(0) } else { None };
19        let qshape = q.shape();
20        let q = q.flatten_to(1)?;
21        let k = k.flatten_to(1)?;
22        let v = v.flatten_to(1)?;
23
24        let cumulative_seqlens_q = &params.cumulative_seqlens_q[&q.device().location()];
25        let cumulative_seqlens_k = &params.cumulative_seqlens_k[&q.device().location()];
26
27        if let Some(softcap) = sdpa_params.softcap {
28            candle_flash_attn::flash_attn_varlen_alibi_windowed_softcap(
29                &q,
30                &k,
31                &v,
32                None,
33                cumulative_seqlens_q,
34                cumulative_seqlens_k,
35                params.max_q as usize,
36                params.max_k as usize,
37                sdpa_params.softmax_scale,
38                window_size_left,
39                window_size_right,
40                softcap,
41            )?
42            .reshape(qshape)
43        } else {
44            candle_flash_attn::flash_attn_varlen_windowed(
45                &q,
46                &k,
47                &v,
48                cumulative_seqlens_q,
49                cumulative_seqlens_k,
50                params.max_q as usize,
51                params.max_k as usize,
52                sdpa_params.softmax_scale,
53                window_size_left,
54                window_size_right,
55            )?
56            .reshape(qshape)
57        }
58    } else {
59        let window_size_right = if default_causal { Some(0) } else { None };
60        if let Some(softcap) = sdpa_params.softcap {
61            candle_flash_attn::flash_attn_alibi_windowed_softcap(
62                q,
63                k,
64                v,
65                None,
66                sdpa_params.softmax_scale,
67                window_size_left,
68                window_size_right,
69                softcap,
70            )
71        } else {
72            candle_flash_attn::flash_attn_windowed(
73                q,
74                k,
75                v,
76                sdpa_params.softmax_scale,
77                window_size_left,
78                window_size_right,
79            )
80        }
81    }
82}
