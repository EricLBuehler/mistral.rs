12/// Not *really* sure why this is necessary but it is.
13pub(crate) fn maybe_synchronize(device: &Device) -> Result<()> {
14    // If less that 4 GB available, synchronize
15    if MemoryUsage.get_memory_available(device)? < 4 * 1024 * (1024 * 1024) {
16        device.synchronize()?;
17    }
18    Ok(())
19}

21/// Computes softmax(QK^T*sqrt(d_k))V
22pub(crate) fn naive_sdpa(
23    q: &Tensor,
24    k: &Tensor,
25    v: &Tensor,
26    mask: Option<&Tensor>,
27    sdpa_params: &SdpaParams,
28) -> Result<Tensor> {
29    if q.device().is_cpu() {
30        let q = q.transpose(1, 2)?;
31        let k = k.transpose(1, 2)?;
32        let v = v.transpose(1, 2)?;
33
34        match q.dtype() {
35            DType::F32 => cpu::run_flash_attn_cpu::<f32>(&q, &k, &v, mask, sdpa_params),
36            DType::F16 => cpu::run_flash_attn_cpu::<half::f16>(&q, &k, &v, mask, sdpa_params),
37            DType::BF16 => cpu::run_flash_attn_cpu::<half::bf16>(&q, &k, &v, mask, sdpa_params),
38            _ => Err(candle_core::Error::Msg("Unsupported data type".into())),
39        }
40    } else {
41        maybe_synchronize(q.device())?;
42
43        if let Some(mask) = mask {
44            let mut att = MatMul.matmul_affine_mul(q, &k.t()?, sdpa_params.softmax_scale.into())?;
45            if let Some(softcap) = sdpa_params.softcap {
46                att = (att / softcap as f64)?;
47                att = att.tanh()?;
48                att = (att * softcap as f64)?;
49            }
50
51            att = att.broadcast_add(mask)?;
52            att = candle_nn::ops::softmax_last_dim(&att)?;
53
54            MatMul.matmul(&att, v)
55        } else {
56            let mut att = MatMul.matmul_affine_mul(q, &k.t()?, sdpa_params.softmax_scale.into())?;
57            if let Some(softcap) = sdpa_params.softcap {
58                att = (att / softcap as f64)?;
59                att = att.tanh()?;
60                att = (att * softcap as f64)?;
61            }
62
63            att = candle_nn::ops::softmax_last_dim(&att)?;
64            MatMul.matmul(&att, v)
65        }
66    }
