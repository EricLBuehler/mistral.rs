10/// Not *really* sure why this is necessary but it is.
11pub(crate) fn maybe_synchronize(device: &Device) -> Result<()> {
12    // If less that 4 GB available, synchronize
13    if MemoryUsage.get_memory_available(device)? < 4 * 1024 * (1024 * 1024) {
14        device.synchronize()?;
15    }
16    Ok(())
17}

19/// Computes softmax(QK^T*sqrt(d_k))V
20pub(crate) fn naive_sdpa(
21    q: &Tensor,
22    k: &Tensor,
23    v: &Tensor,
24    mask: Option<&Tensor>,
25    sdpa_params: &SdpaParams,
26) -> Result<Tensor> {
27    maybe_synchronize(q.device())?;
28
29    // Use chunked attention with a closure that captures the necessary parameters
30    chunked_attention(q, k, v, mask, |q_chunk, k, v, mask_chunk| {
31        let mut att =
32            MatMul.matmul_affine_mul(q_chunk, &k.t()?, sdpa_params.softmax_scale.into())?;
33
34        if let Some(softcap) = sdpa_params.softcap {
35            att = (att / softcap as f64)?;
36            att = att.tanh()?;
37            att = (att * softcap as f64)?;
38        }
39
40        if let Some(mask) = mask_chunk {
41            att = att.broadcast_add(mask)?;
42        }
43
44        att = candle_nn::ops::softmax_last_dim(&att)?;
45        MatMul.matmul(&att, v)
46    })
47}
