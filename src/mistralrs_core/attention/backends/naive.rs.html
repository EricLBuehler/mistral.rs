17static METAL_VERSION_CACHE: AtomicUsize = AtomicUsize::new(usize::MAX);

19fn supports_attn_softmax() -> Result<bool> {
20    #[cfg(feature = "metal")]
21    {
22        use std::sync::atomic::Ordering;
23        let cache = METAL_VERSION_CACHE.load(Ordering::Relaxed);
24
25        let version = if cache != usize::MAX {
26            cache
27        } else {
28            // echo "__METAL_VERSION__" | xcrun -sdk macosx metal -E -x metal -P -
29
30            use std::process::{Command, Stdio};
31
32            // Create the `echo` command and pipe its output into `xcrun`
33            let mut echo = Command::new("echo")
34                .arg("__METAL_VERSION__")
35                .stdout(Stdio::piped())
36                .spawn()
37                .expect("Failed to start echo command");
38
39            echo.wait()?;
40
41            // Run the `xcrun` command, taking input from the `echo` command's output
42            let output = Command::new("xcrun")
43                .arg("-sdk")
44                .arg("macosx")
45                .arg("metal")
46                .arg("-E")
47                .arg("-x")
48                .arg("metal")
49                .arg("-P")
50                .arg("-")
51                .stdin(echo.stdout.unwrap())
52                .output()
53                .expect("Failed to run xcrun command");
54
55            // Handle the output
56            if output.status.success() {
57                let version = String::from_utf8_lossy(&output.stdout)
58                    .split('\n')
59                    .nth(1)
60                    .unwrap()
61                    .trim()
62                    .to_string()
63                    .parse::<usize>()
64                    .unwrap();
65                METAL_VERSION_CACHE.store(version, Ordering::Relaxed);
66                version
67            } else {
68                let stderr = String::from_utf8_lossy(&output.stderr);
69                panic!("Error:\n{}", stderr);
70            }
71        };
72        // Attn softmax is only supported for metal >= 310
73        Ok(version >= 310)
74    }
75
76    #[cfg(not(feature = "metal"))]
77    Ok(true)
78}

80/// Not *really* sure why this is necessary but it is.
81pub(crate) fn maybe_synchronize(device: &Device) -> Result<()> {
82    // If less that 4 GB available, synchronize
83    if MemoryUsage.get_memory_available(device)? < 4 * 1024 * (1024 * 1024) {
84        device.synchronize()?;
85    }
86    Ok(())
87}

89/// Computes softmax(QK^T*sqrt(d_k))V
90pub(crate) fn naive_sdpa(
91    q: &Tensor,
92    k: &Tensor,
93    v: &Tensor,
94    mask: Option<&Tensor>,
95    sdpa_params: &SdpaParams,
96) -> Result<Tensor> {
97    if q.device().is_cpu() {
98        let q = q.transpose(1, 2)?;
99        let k = k.transpose(1, 2)?;
100        let v = v.transpose(1, 2)?;
101
102        match q.dtype() {
103            DType::F32 => cpu::run_flash_attn_cpu::<f32>(&q, &k, &v, mask, sdpa_params),
104            DType::F16 => cpu::run_flash_attn_cpu::<half::f16>(&q, &k, &v, mask, sdpa_params),
105            DType::BF16 => cpu::run_flash_attn_cpu::<half::bf16>(&q, &k, &v, mask, sdpa_params),
106            _ => Err(candle_core::Error::Msg("Unsupported data type".into())),
107        }
108    } else {
109        maybe_synchronize(q.device())?;
110
111        // Use faster softmax if mask is rank 2 or it's rank 3
112        if mask.is_some_and(|mask| mask.rank() == 2 || mask.rank() == 3) && supports_attn_softmax()?
113        {
114            let mask = match mask {
115                Some(mask) if mask.rank() == 3 || mask.rank() == 2 => mask.clone(),
116                _ => candle_core::bail!("unsupported mask {mask:?}"),
117            };
118
119            let mut att = MatMul.matmul(q, &k.t()?)?;
120
121            candle_nn::ops::inplace_attn_softmax_last_dim(
122                &mut att,
123                &mask.contiguous()?,
124                sdpa_params.softmax_scale / sdpa_params.softcap.unwrap_or(1.0),
125            )?;
126
127            if let Some(softcap) = sdpa_params.softcap {
128                att = (att.tanh()? * softcap as f64)?;
129            }
130
131            MatMul.matmul(&att, v)
132        } else if let Some(mask) = mask {
133            let mut att = MatMul.matmul_affine_mul(q, &k.t()?, sdpa_params.softmax_scale.into())?;
134            if let Some(softcap) = sdpa_params.softcap {
135                att = (att / softcap as f64)?;
136                att = att.tanh()?;
137                att = (att * softcap as f64)?;
138            }
139
140            att = att.broadcast_add(mask)?;
141            candle_nn::ops::inplace_softmax_last_dim(&mut att)?;
142
143            MatMul.matmul(&att, v)
144        } else {
145            let mut att = MatMul.matmul_affine_mul(q, &k.t()?, sdpa_params.softmax_scale.into())?;
146            if let Some(softcap) = sdpa_params.softcap {
147                att = (att / softcap as f64)?;
148                att = att.tanh()?;
149                att = (att * softcap as f64)?;
150            }
151
152            candle_nn::ops::inplace_softmax_last_dim(&mut att)?;
153            MatMul.matmul(&att, v)
154        }
155    }
156}
