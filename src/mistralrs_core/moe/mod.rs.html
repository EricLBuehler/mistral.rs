31pub struct MoEConfig {
32    pub num_experts: usize,
33    pub num_experts_per_tok: usize,
34    pub hidden_size: usize,
35    pub moe_intermediate_size: usize,
36    pub norm_topk_prob: bool,
37}
38impl FusedMoe {
39    pub fn new(
40        cfg: &MoEConfig,
41        vb: ShardedVarBuilder,
42        layer_device: Device,
43        comm: &Arc<mistralrs_quant::Comm>,
44    ) -> Result<Self> {
45        let num_experts = cfg.num_experts;
46
47        let gate = linear_no_bias(
48            cfg.hidden_size,
49            num_experts,
50            vb.pp("gate").set_device(layer_device.clone()),
51        )?;
52
53        let experts_vb = vb.pp("experts").set_device(layer_device);
54        let mut gate_up_experts = Vec::with_capacity(num_experts);
55        let mut down_experts = Vec::with_capacity(num_experts);
56
57        //pack experts
58        for i in 0..num_experts {
59            let experts_vb = experts_vb.pp(format!("{}", i).as_str());
60            let (gate_up_expert, down_expert) = {
61                // n x k format
62                let gate_expert = experts_vb.pp("gate_proj").get_with_hints(
63                    (cfg.moe_intermediate_size, cfg.hidden_size),
64                    "weight",
65                    shard(0, comm.rank(), comm.world_size()),
66                )?;
67                let up_expert = experts_vb.pp("up_proj").get_with_hints(
68                    (cfg.moe_intermediate_size, cfg.hidden_size),
69                    "weight",
70                    shard(0, comm.rank(), comm.world_size()),
71                )?;
72                let down_expert = experts_vb.pp("down_proj").get_with_hints(
73                    (cfg.hidden_size, cfg.moe_intermediate_size),
74                    "weight",
75                    shard(1, comm.rank(), comm.world_size()),
76                )?;
77                //pack gate_proj and up_proj
78                let gate_up_expert = Tensor::cat(&[&gate_expert, &up_expert], 0)?;
79
80                (gate_up_expert, down_expert)
81            };
82
83            gate_up_experts.push(gate_up_expert);
84            down_experts.push(down_expert);
85        }
86
87        let gate_up_w = Tensor::stack(&gate_up_experts, 0)?;
88        let down_w = Tensor::stack(&down_experts, 0)?;
89        let world_size = comm.world_size();
90        let w_size_n = gate_up_w.dim(1)? / 2;
91
92        Ok(Self {
93            gate,
94            gate_up_w,
95            down_w,
96            w_size_n,
97            act: candle_nn::Activation::Silu,
98            norm_topk_prob: cfg.norm_topk_prob,
99            num_experts_per_tok: cfg.num_experts_per_tok,
100            all_reduce: SumAllReduce::new(comm),
101            world_size,
102            dtype: vb.dtype(),
103        })
104    }
105
106    pub fn forward(&self, xs: &Tensor, is_prefill: bool) -> Result<Tensor> {
107        let (b_size, seq_len, hidden_dim) = xs.dims3()?;
108        let xs = xs.reshape(((), hidden_dim))?;
109        let (num_tokens, hidden_dim) = xs.dims2()?;
110        let router_logits = self.gate.forward(&xs)?;
111
112        let routing_weights =
113            candle_nn::ops::softmax_last_dim(&router_logits.to_dtype(DType::F32)?)?;
114
115        let topk_ids = routing_weights
116            .arg_sort_last_dim(false)?
117            .narrow(D::Minus1, 0, self.num_experts_per_tok)?
118            .contiguous()?;
119
120        let mut topk_weights = routing_weights.gather(&topk_ids, D::Minus1)?;
121
122        if self.norm_topk_prob {
123            topk_weights = topk_weights.broadcast_div(&topk_weights.sum_keepdim(D::Minus1)?)?;
124        }
125
126        let (expert_ids, sorted_token_ids) = if is_prefill {
127            #[cfg(feature = "cuda")]
128            {
129                use crate::ops::ArgSortOp;
130                topk_ids.flatten_all()?.sort(true)?
131            }
132            #[cfg(not(feature = "cuda"))]
133            topk_ids.flatten_all()?.sort_last_dim(true)?
134        } else {
135            topk_ids.flatten_all()?.sort_last_dim(true)?
136        };
137
138        //out (M, top_k, N)
139        let gate_up = moe::moe_gemm(
140            &xs,
141            &self.gate_up_w,
142            &None,
143            &sorted_token_ids,
144            &expert_ids,
145            self.num_experts_per_tok,
146            is_prefill,
147        )?;
148
149        let gate = gate_up
150            .narrow(candle_core::D::Minus1, 0, self.w_size_n)?
151            .contiguous()?;
152        let up = gate_up
153            .narrow(candle_core::D::Minus1, self.w_size_n, self.w_size_n)?
154            .contiguous()?;
155
156        //(M * top_k, N // 2)
157        let down_inputs = (up * gate.apply(&self.act)?)?.reshape(((), self.w_size_n))?;
158
159        //view(M, top_k, K) -> sum -> (M, K)
160        let mut ys = moe::moe_gemm(
161            &down_inputs,
162            &self.down_w,
163            &Some(topk_weights),
164            &sorted_token_ids,
165            &expert_ids,
166            self.num_experts_per_tok,
167            is_prefill,
168        )?
169        .reshape((num_tokens, (), hidden_dim))?
170        .sum(D::Minus2)?;
171
172        if self.world_size > 1 {
173            ys = self.all_reduce.sum_all_reduce(&ys)?;
174        }
175        ys.reshape((b_size, seq_len, hidden_dim))
176    }
177}
