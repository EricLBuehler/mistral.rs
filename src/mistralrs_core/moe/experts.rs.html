1//! Unified MoE experts layer supporting multiple backends and weight formats.
2//!
3//! This module provides `MoEExperts`, a flexible experts layer that:
4//! - Does NOT carry the gate (router) - gate is external
5//! - Supports both per-expert and stacked weight formats
6//! - Handles backend selection (fused/fast/slow)
7//! - Manages tensor parallelism with all-reduce
8
9use candle_core::{DType, Device, Result, Tensor, D};
10use mistralrs_quant::{
11    FusedExperts, MatMul, PackedExperts, QuantMethod, QuantizedConfig, ShardedVarBuilder,
12    SumAllReduce,
13};
14use std::sync::Arc;

16use crate::cuda::moe;
17use crate::layers::Activation;
18use crate::moe::shard;

20/// Configuration for MoEExperts
21pub struct MoEExpertsConfig {
22    pub num_experts: usize,
23    pub num_experts_per_tok: usize,
24    pub hidden_size: usize,
25    pub moe_intermediate_size: usize,
26}

28/// Backend selection for MoE experts
29pub enum MoEExpertsBackend {
30    /// Use fused CUDA kernels with raw tensors (fastest for CUDA unquantized)
31    Fused,
32    /// Use gather-based implementation (good for Metal, ISQ)
33    Fast,
34    /// Use loop-based implementation (fallback for quantized)
35    Slow,
36}
37
38impl MoEExpertsBackend {
39    /// Determine the best backend based on device and quantization settings
40    pub fn select(
41        device: &Device,
42        loading_isq: bool,
43        quantization_config: &Option<QuantizedConfig>,
44    ) -> Self {
45        let use_fast = device.is_metal()
46            || (device.is_cuda() && (loading_isq || quantization_config.is_some()));
47
48        if use_fast {
49            Self::Fast
50        } else if quantization_config.is_none() && !loading_isq && device.is_cuda() {
51            Self::Fused
52        } else {
53            Self::Slow
54        }
55    }
56}

58/// Internal representation of fused expert weights for CUDA kernels
59struct FusedExpertsWeights {
60    /// gate_up weights: [E, N, K] for standard, [E, K, N] for stacked
61    gate_up_w: Tensor,
62    /// down weights: [E, N, K] for standard, [E, K, N] for stacked
63    down_w: Tensor,
64    /// Size of intermediate dimension (after sharding)
65    w_size_n: usize,
66    /// Whether weights are in stacked format [E, K, N]
67    stacked_format: bool,
68}
69
70/// Internal representation for gather-based experts (Metal/ISQ)
71struct FastExpertsWeights {
72    fused_gate_proj: Arc<dyn QuantMethod>,
73    fused_up_proj: Arc<dyn QuantMethod>,
74    fused_down_proj: Arc<dyn QuantMethod>,
75}
76
77/// Internal representation for loop-based experts (quantized fallback)
78struct SlowExpertsWeights {
79    experts: PackedExperts,
80}

82/// MoE experts layer without gate
83///
84/// This struct encapsulates the expert weights and forward logic,
85/// but does NOT include the routing gate. The caller is responsible
86/// for computing routing weights and topk indices.
87pub struct MoEExperts {
88    backend: MoEExpertsBackendImpl,
89    act: Activation,
90    num_experts_per_tok: usize,
91    all_reduce: SumAllReduce,
92    world_size: usize,
93}
94
95enum MoEExpertsBackendImpl {
96    Fused(FusedExpertsWeights),
97    Fast(FastExpertsWeights),
98    Slow(SlowExpertsWeights),
99}

102    /// Create MoEExperts with automatic backend selection
103    ///
104    /// Automatically detects weight format (stacked vs per-expert) and
105    /// selects the appropriate backend based on device and quantization.
106    pub fn new(
107        cfg: &MoEExpertsConfig,
108        vb: ShardedVarBuilder,
109        layer_device: Device,
110        comm: &Arc<mistralrs_quant::Comm>,
111        loading_isq: bool,
112        quantization_config: &Option<QuantizedConfig>,
113        act: Activation,
114    ) -> Result<Self> {
115        let backend = MoEExpertsBackend::select(&layer_device, loading_isq, quantization_config);
116        Self::new_with_backend(
117            cfg,
118            vb,
119            layer_device,
120            comm,
121            backend,
122            quantization_config,
123            act,
124        )
125    }

127    /// Create MoEExperts with explicit backend selection
128    pub fn new_with_backend(
129        cfg: &MoEExpertsConfig,
130        vb: ShardedVarBuilder,
131        layer_device: Device,
132        comm: &Arc<mistralrs_quant::Comm>,
133        backend: MoEExpertsBackend,
134        quantization_config: &Option<QuantizedConfig>,
135        act: Activation,
136    ) -> Result<Self> {
137        let experts_vb = vb.pp("experts").set_device(layer_device.clone());
138
139        // Detect format: stacked has "gate_up_proj", per-expert has "0.gate_proj"
140        let is_stacked = experts_vb.contains_tensor("gate_up_proj");
141
142        let backend_impl = match backend {
143            MoEExpertsBackend::Fused => {
144                if is_stacked {
145                    MoEExpertsBackendImpl::Fused(Self::load_fused_stacked(cfg, experts_vb, comm)?)
146                } else {
147                    MoEExpertsBackendImpl::Fused(Self::load_fused_standard(cfg, experts_vb, comm)?)
148                }
149            }
150            MoEExpertsBackend::Fast => {
151                if is_stacked {
152                    MoEExpertsBackendImpl::Fast(Self::load_fast_stacked(
153                        cfg,
154                        vb,
155                        quantization_config,
156                    )?)
157                } else {
158                    MoEExpertsBackendImpl::Fast(Self::load_fast_standard(
159                        cfg,
160                        vb,
161                        quantization_config,
162                    )?)
163                }
164            }
165            MoEExpertsBackend::Slow => MoEExpertsBackendImpl::Slow(Self::load_slow(
166                cfg,
167                experts_vb,
168                comm,
169                quantization_config,
170            )?),
171        };
172
173        Ok(Self {
174            backend: backend_impl,
175            act,
176            num_experts_per_tok: cfg.num_experts_per_tok,
177            all_reduce: SumAllReduce::new(comm),
178            world_size: comm.world_size(),
179        })
180    }

182    /// Load fused weights in standard per-expert format
183    fn load_fused_standard(
184        cfg: &MoEExpertsConfig,
185        experts_vb: ShardedVarBuilder,
186        comm: &Arc<mistralrs_quant::Comm>,
187    ) -> Result<FusedExpertsWeights> {
188        let num_experts = cfg.num_experts;
189        let mut gate_up_experts = Vec::with_capacity(num_experts);
190        let mut down_experts = Vec::with_capacity(num_experts);
191
192        for i in 0..num_experts {
193            let expert_vb = experts_vb.pp(i.to_string());
194            // n x k format
195            let gate_expert = expert_vb.pp("gate_proj").get_with_hints(
196                (cfg.moe_intermediate_size, cfg.hidden_size),
197                "weight",
198                shard(0, comm.rank(), comm.world_size()),
199            )?;
200            let up_expert = expert_vb.pp("up_proj").get_with_hints(
201                (cfg.moe_intermediate_size, cfg.hidden_size),
202                "weight",
203                shard(0, comm.rank(), comm.world_size()),
204            )?;
205            let down_expert = expert_vb.pp("down_proj").get_with_hints(
206                (cfg.hidden_size, cfg.moe_intermediate_size),
207                "weight",
208                shard(1, comm.rank(), comm.world_size()),
209            )?;
210            // Pack gate_proj and up_proj
211            let gate_up_expert = Tensor::cat(&[&gate_expert, &up_expert], 0)?;
212
213            gate_up_experts.push(gate_up_expert);
214            down_experts.push(down_expert);
215        }
216
217        let gate_up_w = Tensor::stack(&gate_up_experts, 0)?;
218        let down_w = Tensor::stack(&down_experts, 0)?;
219        let w_size_n = gate_up_w.dim(1)? / 2;
220
221        Ok(FusedExpertsWeights {
222            gate_up_w,
223            down_w,
224            w_size_n,
225            stacked_format: false,
226        })
227    }

229    /// Load fused weights in stacked format
230    fn load_fused_stacked(
231        cfg: &MoEExpertsConfig,
232        experts_vb: ShardedVarBuilder,
233        comm: &Arc<mistralrs_quant::Comm>,
234    ) -> Result<FusedExpertsWeights> {
235        let num_experts = cfg.num_experts;
236
237        // Load stacked gate_up_proj: [num_experts, hidden, inter*2]
238        let gate_up_w = experts_vb.get_with_hints(
239            (num_experts, cfg.hidden_size, cfg.moe_intermediate_size * 2),
240            "gate_up_proj",
241            shard(2, comm.rank(), comm.world_size()),
242        )?;
243
244        // Load stacked down_proj: [num_experts, inter, hidden]
245        let down_w = experts_vb.get_with_hints(
246            (num_experts, cfg.moe_intermediate_size, cfg.hidden_size),
247            "down_proj",
248            shard(1, comm.rank(), comm.world_size()),
249        )?;
250
251        let w_size_n = gate_up_w.dim(2)? / 2;
252
253        Ok(FusedExpertsWeights {
254            gate_up_w,
255            down_w,
256            w_size_n,
257            stacked_format: true,
258        })
259    }

332    /// Forward pass through experts
333    ///
334    /// # Arguments
335    /// * `xs` - Input tensor of shape [batch, seq_len, hidden_dim]
336    /// * `topk_weights` - Top-k routing weights of shape [num_tokens, num_experts_per_tok]
337    /// * `topk_ids` - Top-k expert indices of shape [num_tokens, num_experts_per_tok]
338    ///
339    /// # Returns
340    /// Output tensor of shape [batch, seq_len, hidden_dim]
341    pub fn forward(&self, xs: &Tensor, topk_weights: Tensor, topk_ids: &Tensor) -> Result<Tensor> {
342        let (b_size, seq_len, hidden_dim) = xs.dims3()?;
343        // Prefill = processing multiple tokens; Decode = single token generation
344        let is_prefill = seq_len > 1;
345
346        let mut ys = match &self.backend {
347            MoEExpertsBackendImpl::Fused(weights) => {
348                self.forward_fused(xs, &topk_weights, topk_ids, weights, is_prefill)?
349            }
350            MoEExpertsBackendImpl::Fast(weights) => {
351                self.forward_fast(xs, &topk_weights, topk_ids, weights)?
352            }
353            MoEExpertsBackendImpl::Slow(weights) => {
354                self.forward_slow(xs, &topk_weights, topk_ids, weights)?
355            }
356        };
357
358        // Apply all-reduce for tensor parallelism
359        if self.world_size > 1 {
360            ys = self.all_reduce.sum_all_reduce(&ys)?;
361        }
362
363        ys.reshape((b_size, seq_len, hidden_dim))
364    }

366    /// Fused CUDA kernel forward pass
367    fn forward_fused(
368        &self,
369        xs: &Tensor,
370        topk_weights: &Tensor,
371        topk_ids: &Tensor,
372        weights: &FusedExpertsWeights,
373        is_prefill: bool,
374    ) -> Result<Tensor> {
375        let (_b_size, _seq_len, hidden_dim) = xs.dims3()?;
376        let xs = xs.reshape(((), hidden_dim))?;
377        let (num_tokens, _) = xs.dims2()?;
378
379        // Sort tokens by expert for efficient processing
380        let (expert_ids, sorted_token_ids) = if is_prefill {
381            #[cfg(feature = "cuda")]
382            {
383                use crate::ops::ArgSortOp;
384                topk_ids.flatten_all()?.sort(true)?
385            }
386            #[cfg(not(feature = "cuda"))]
387            topk_ids.flatten_all()?.sort_last_dim(true)?
388        } else {
389            topk_ids.flatten_all()?.sort_last_dim(true)?
390        };
391
392        // First GEMM: gate_up projection
393        let gate_up = if weights.stacked_format {
394            moe::moe_gemm_transposed(
395                &xs,
396                &weights.gate_up_w,
397                &None,
398                &sorted_token_ids,
399                &expert_ids,
400                self.num_experts_per_tok,
401                is_prefill,
402            )?
403        } else {
404            moe::moe_gemm(
405                &xs,
406                &weights.gate_up_w,
407                &None,
408                &sorted_token_ids,
409                &expert_ids,
410                self.num_experts_per_tok,
411                is_prefill,
412            )?
413        };
414
415        // Split and apply activation
416        let gate = gate_up
417            .narrow(D::Minus1, 0, weights.w_size_n)?
418            .contiguous()?;
419        let up = gate_up
420            .narrow(D::Minus1, weights.w_size_n, weights.w_size_n)?
421            .contiguous()?;
422
423        let down_inputs = (up * gate.apply(&self.act)?)?.reshape(((), weights.w_size_n))?;
424
425        // Second GEMM: down projection with weight aggregation
426        let ys = if weights.stacked_format {
427            moe::moe_gemm_transposed(
428                &down_inputs,
429                &weights.down_w,
430                &Some(topk_weights.clone()),
431                &sorted_token_ids,
432                &expert_ids,
433                self.num_experts_per_tok,
434                is_prefill,
435            )?
436        } else {
437            moe::moe_gemm(
438                &down_inputs,
439                &weights.down_w,
440                &Some(topk_weights.clone()),
441                &sorted_token_ids,
442                &expert_ids,
443                self.num_experts_per_tok,
444                is_prefill,
445            )?
446        };
447
448        ys.reshape((num_tokens, (), hidden_dim))?.sum(D::Minus2)
449    }

451    /// Gather-based forward pass (Metal/ISQ)
452    fn forward_fast(
453        &self,
454        xs: &Tensor,
455        topk_weights: &Tensor,
456        topk_ids: &Tensor,
457        weights: &FastExpertsWeights,
458    ) -> Result<Tensor> {
459        let original_dtype = xs.dtype();
460        let (b_size, seq_len, hidden_dim) = xs.dims3()?;
461        let num_tokens = b_size * seq_len;
462
463        let xs_flat = xs.reshape((num_tokens, hidden_dim))?;
464
465        let ys = if xs.device().is_cuda() {
466            // CUDA path: use indexed_moe_forward compatible shapes
467            let xs = xs_flat.reshape((num_tokens, 1, hidden_dim))?;
468            let gate = weights
469                .fused_gate_proj
470                .gather_forward_autocast(&xs, topk_ids)?;
471            let up = weights
472                .fused_up_proj
473                .gather_forward_autocast(&xs, topk_ids)?;
474            weights
475                .fused_down_proj
476                .gather_forward_autocast(&(up * gate.apply(&self.act)?)?, topk_ids)?
477        } else {
478            // Metal path: use broadcast gather shapes
479            let xs = xs.reshape((b_size, seq_len, 1, 1, hidden_dim))?;
480            let indices = topk_ids.reshape((b_size, seq_len, self.num_experts_per_tok))?;
481            let gate = weights
482                .fused_gate_proj
483                .gather_forward_autocast(&xs, &indices)?;
484            let up = weights
485                .fused_up_proj
486                .gather_forward_autocast(&xs, &indices)?;
487            let xs = weights
488                .fused_down_proj
489                .gather_forward_autocast(&(up * gate.apply(&self.act)?)?, &indices)?;
490            xs.squeeze(D::Minus2)?
491                .reshape((num_tokens, self.num_experts_per_tok, hidden_dim))?
492        };
493
494        ys.to_dtype(DType::F32)?
495            .broadcast_mul(&topk_weights.unsqueeze(D::Minus1)?)?
496            .sum(D::Minus2)?
497            .to_dtype(original_dtype)
498    }

500    /// Loop-based forward pass (quantized fallback)
501    fn forward_slow(
502        &self,
503        xs: &Tensor,
504        topk_weights: &Tensor,
505        topk_ids: &Tensor,
506        weights: &SlowExpertsWeights,
507    ) -> Result<Tensor> {
508        let (b_size, seq_len, hidden_dim) = xs.dims3()?;
509        let xs = xs.reshape(((), hidden_dim))?;
510
511        let routing_weights = topk_weights.to_dtype(DType::F32)?.to_vec2::<f32>()?;
512        let experts_per_tok = topk_ids.to_vec2::<u32>()?;
513        let num_experts = weights.experts.gate_proj.len();
514
515        let mut top_x = vec![vec![]; num_experts];
516        let mut selected_experts = vec![vec![]; num_experts];
517
518        for (row_idx, (rw, expert_idxs)) in routing_weights
519            .iter()
520            .zip(experts_per_tok.iter())
521            .enumerate()
522        {
523            for (&rw, &expert_idx) in rw.iter().zip(expert_idxs.iter()) {
524                #[allow(clippy::cast_possible_truncation)]
525                top_x[expert_idx as usize].push(row_idx as u32);
526                selected_experts[expert_idx as usize].push(rw)
527            }
528        }
529
530        let mut ys = xs.zeros_like()?;
531        for expert_idx in 0..num_experts {
532            let top_x_expert = &top_x[expert_idx];
533            if top_x_expert.is_empty() {
534                continue;
535            }
536            let top_x_tensor = Tensor::new(top_x_expert.as_slice(), xs.device())?;
537            let selected_experts_tensor =
538                Tensor::new(selected_experts[expert_idx].as_slice(), xs.device())?
539                    .reshape(((), 1))?
540                    .to_dtype(xs.dtype())?;
541            let current_state = xs
542                .index_select(&top_x_tensor, 0)?
543                .reshape(((), hidden_dim))?;
544
545            // Forward through expert MLP
546            let original_dtype = current_state.dtype();
547            let mut expert_input = current_state.clone();
548            if let Some(t) = weights.experts.gate_proj[expert_idx].quantized_act_type() {
549                expert_input = expert_input.to_dtype(t)?;
550            }
551            let gate_out = MatMul
552                .qmethod_matmul(&expert_input, &*weights.experts.gate_proj[expert_idx])?
553                .apply(&self.act)?;
554            let up_out =
555                MatMul.qmethod_matmul(&expert_input, &*weights.experts.up_proj[expert_idx])?;
556            let mut current_hidden_states = MatMul.qmethod_matmul(
557                &(gate_out * up_out)?,
558                &*weights.experts.down_proj[expert_idx],
559            )?;
560            if weights.experts.gate_proj[expert_idx]
561                .quantized_act_type()
562                .is_some()
563            {
564                current_hidden_states = current_hidden_states.to_dtype(original_dtype)?;
565            }
566
567            let current_hidden_states =
568                current_hidden_states.broadcast_mul(&selected_experts_tensor)?;
569            ys = ys.index_add(&top_x_tensor, &current_hidden_states, 0)?;
570        }
571
572        ys.reshape((b_size * seq_len, hidden_dim))
573    }

575    /// Get mutable references to quantizable layers for ISQ
576    pub fn get_isq_layers(&mut self) -> Vec<&mut Arc<dyn QuantMethod>> {
577        match &mut self.backend {
578            MoEExpertsBackendImpl::Fused(_) => vec![],
579            MoEExpertsBackendImpl::Fast(weights) => {
580                vec![
581                    &mut weights.fused_gate_proj,
582                    &mut weights.fused_up_proj,
583                    &mut weights.fused_down_proj,
584                ]
585            }
586            MoEExpertsBackendImpl::Slow(weights) => {
587                let mut layers = Vec::new();
588                for (gate, (up, down)) in weights.experts.gate_proj.iter_mut().zip(
589                    weights
590                        .experts
591                        .up_proj
592                        .iter_mut()
593                        .zip(weights.experts.down_proj.iter_mut()),
594                ) {
595                    layers.push(gate);
596                    layers.push(up);
597                    layers.push(down);
598                }
599                layers
600            }
601        }
602    }
