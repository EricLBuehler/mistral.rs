95    rotary_emb_local: Arc<RotaryEmbedding>,

102impl Attention {
103    #[allow(clippy::too_many_arguments)]
104    fn new(
105        rotary_emb_global: Arc<Gemma3RotaryEmbedding>,
106        rotary_emb_local: Arc<RotaryEmbedding>,
107        cfg: &EmbeddingGemmaConfig,
108        layer_idx: usize,
109        mapper: &dyn DeviceMapper,
110        vb: ShardedVarBuilder,
111        comm: &Arc<mistralrs_quant::Comm>,
112    ) -> Result<Self> {
113        let hidden_sz = cfg.hidden_size;
114        let num_heads = cfg.num_attention_heads;
115        let num_kv_heads = cfg.num_key_value_heads;
116        let head_dim = cfg.head_dim;
117        let bias = cfg.attention_bias;
118        let q_proj = ColumnParallelLayer::new(
119            hidden_sz,
120            num_heads * head_dim,
121            &cfg.quantization_config,
122            bias,
123            comm,
124            vb.pp("q_proj"),
125        )?;
126        let kv_shard = mistralrs_quant::compute_kv_shard(
127            cfg.num_key_value_heads,
128            cfg.hidden_size / cfg.num_attention_heads,
129            comm,
130        );
131        let k_proj = ColumnParallelLayer::new_with_shard(
132            hidden_sz,
133            num_kv_heads * head_dim,
134            &cfg.quantization_config,
135            bias,
136            comm,
137            kv_shard,
138            vb.pp("k_proj"),
139        )?;
140        let v_proj = ColumnParallelLayer::new_with_shard(
141            hidden_sz,
142            num_kv_heads * head_dim,
143            &cfg.quantization_config,
144            bias,
145            comm,
146            kv_shard,
147            vb.pp("v_proj"),
148        )?;
149        let o_proj = RowParallelLayer::new(
150            num_heads * head_dim,
151            hidden_sz,
152            &cfg.quantization_config,
153            bias,
154            comm,
155            vb.pp("o_proj"),
156        )?;
157        let sliding_window = if is_sliding!(layer_idx, cfg) {
158            Some(cfg.sliding_window)
159        } else {
160            None
161        };
162
163        let q_norm = RmsNorm::new_gemma(
164            cfg.head_dim,
165            cfg.rms_norm_eps,
166            mapper.set_device(layer_idx, vb.pp("q_norm"), false),
167        )?;
168        let k_norm = RmsNorm::new_gemma(
169            cfg.head_dim,
170            cfg.rms_norm_eps,
171            mapper.set_device(layer_idx, vb.pp("k_norm"), false),
172        )?;
173        Ok(Self {
174            q_proj,
175            k_proj,
176            v_proj,
177            o_proj,
178            num_heads: num_heads / comm.world_size(),
179            num_kv_heads: (num_kv_heads / comm.world_size()).max(1),
180            head_dim,
181            rotary_emb_global,
182            rotary_emb_local,
183            use_sliding_window: sliding_window.is_some(),
184            sdpa_params: SdpaParams {
185                n_kv_groups: mistralrs_quant::compute_n_kv_groups(
186                    cfg.num_key_value_heads,
187                    cfg.num_attention_heads,
188                    comm,
189                ),
190                softcap: cfg.attn_logit_softcapping.map(|x| x as f32),
191                softmax_scale: 1.0 / (cfg.query_pre_attn_scalar as f32).sqrt(),
192                sliding_window,
193            },
194            q_norm,
195            k_norm,
196        })
197    }
198
199    #[allow(clippy::too_many_arguments)]
200    fn forward(
201        &self,
202        xs: &Tensor,
203        attention_mask: &Tensor,
204        sliding_attention_mask: &Tensor,
205        seqlen_offsets: &[usize],
206        flash_params: &FlashParams,
207    ) -> Result<Tensor> {
208        let (b_sz, q_len, _) = xs.dims3()?;
209
210        let original_dtype = xs.dtype();
211        let mut xs = xs.clone();
212        if let Some(t) = self.q_proj.quantized_act_type() {
213            xs = xs.to_dtype(t)?;
214        }
215        let mut q = MatMul.qmethod_matmul(&xs, &*self.q_proj)?;
216        let mut k = MatMul.qmethod_matmul(&xs, &*self.k_proj)?;
217        let mut v = MatMul.qmethod_matmul(&xs, &*self.v_proj)?;
218        if self.q_proj.quantized_act_type().is_some() {
219            q = q.to_dtype(original_dtype)?;
220            k = k.to_dtype(original_dtype)?;
221            v = v.to_dtype(original_dtype)?;
222        }
223
224        q = q
225            .reshape((b_sz, q_len, self.num_heads, self.head_dim))?
226            .transpose(1, 2)?;
227        k = k
228            .reshape((b_sz, q_len, self.num_kv_heads, self.head_dim))?
229            .transpose(1, 2)?;
230        v = v
231            .reshape((b_sz, q_len, self.num_kv_heads, self.head_dim))?
232            .transpose(1, 2)?;
233
234        q = q.apply(&self.q_norm)?;
235        k = k.apply(&self.k_norm)?;
236
237        (q, k) = match self.use_sliding_window {
238            true => self.rotary_emb_local.forward(&q, &k, seqlen_offsets)?,
239            false => self.rotary_emb_global.forward(&q, &k, seqlen_offsets)?,
240        };
241
242        let mask = if self.use_sliding_window {
243            sliding_attention_mask
244        } else {
245            attention_mask
246        };
247
248        let mut attn_output = Sdpa.run_attention(
249            &q,
250            &k,
251            &v,
252            Some(mask),
253            Some(flash_params),
254            &self.sdpa_params,
255        )?;
256
257        if let Some(t) = self.q_proj.quantized_act_type() {
258            attn_output = attn_output.to_dtype(t)?;
259        }
260        attn_output = attn_output.transpose(1, 2)?.reshape((b_sz, q_len, ()))?;
261        let mut res = MatMul.qmethod_matmul(&attn_output, &*self.o_proj)?;
262        if self.q_proj.quantized_act_type().is_some() {
263            res = res.to_dtype(original_dtype)?;
264        }
265        Ok(res)
266    }
267}

269struct DecoderLayer {
270    self_attn: Attention,
271    mlp: Box<dyn MlpLayer>,
272    input_layernorm: RmsNorm,
273    post_attention_layernorm: RmsNorm,
274    pre_feedforward_layernorm: RmsNorm,
275    post_feedforward_layernorm: RmsNorm,
276}
277
278impl DecoderLayer {
279    #[allow(clippy::too_many_arguments)]
280    fn new(
281        rotary_emb_global: Arc<Gemma3RotaryEmbedding>,
282        rotary_emb_local: Arc<RotaryEmbedding>,
283        cfg: &EmbeddingGemmaConfig,
284        vb: ShardedVarBuilder,
285        mapper: &dyn DeviceMapper,
286        layer_idx: usize,
287        loading_isq: bool,
288        comm: &Arc<mistralrs_quant::Comm>,
289    ) -> Result<Self> {
290        let self_attn = Attention::new(
291            rotary_emb_global,
292            rotary_emb_local,
293            cfg,
294            layer_idx,
295            mapper,
296            mapper.set_device(layer_idx, vb.pp("self_attn"), loading_isq),
297            comm,
298        )?;
299        let mlp = Mlp::new(
300            mapper.set_device(layer_idx, vb.pp("mlp"), loading_isq),
301            cfg.hidden_size,
302            cfg.intermediate_size,
303            &cfg.quantization_config,
304            cfg.hidden_activation,
305            comm,
306        )?;
307        let input_layernorm = RmsNorm::new_gemma(
308            cfg.hidden_size,
309            cfg.rms_norm_eps,
310            mapper.set_device(layer_idx, vb.pp("input_layernorm"), false),
311        )?;
312        let post_attention_layernorm = RmsNorm::new_gemma(
313            cfg.hidden_size,
314            cfg.rms_norm_eps,
315            mapper.set_device(layer_idx, vb.pp("post_attention_layernorm"), false),
316        )?;
317        let pre_feedforward_layernorm = RmsNorm::new_gemma(
318            cfg.hidden_size,
319            cfg.rms_norm_eps,
320            mapper.set_device(layer_idx, vb.pp("pre_feedforward_layernorm"), false),
321        )?;
322        let post_feedforward_layernorm = RmsNorm::new_gemma(
323            cfg.hidden_size,
324            cfg.rms_norm_eps,
325            mapper.set_device(layer_idx, vb.pp("post_feedforward_layernorm"), false),
326        )?;
327        Ok(Self {
328            self_attn,
329            mlp: Box::new(mlp),
330            input_layernorm,
331            post_attention_layernorm,
332            pre_feedforward_layernorm,
333            post_feedforward_layernorm,
334        })
335    }
336
337    #[allow(clippy::too_many_arguments)]
338    fn forward(
339        &self,
340        xs: &Tensor,
341        attention_mask: &Tensor,
342        sliding_attention_mask: &Tensor,
343        seqlen_offsets: &[usize],
344        flash_params: &FlashParams,
345    ) -> Result<Tensor> {
346        let residual = xs;
347        let xs = self.input_layernorm.forward(xs)?;
348        let xs = self
349            .self_attn
350            .forward(
351                &xs,
352                attention_mask,
353                sliding_attention_mask,
354                seqlen_offsets,
355                flash_params,
356            )?
357            .apply(&self.post_attention_layernorm)?;
358        let xs = (xs + residual)?;
359        let residual = &xs;
360        let xs = self
361            .mlp
362            .forward(&xs.apply(&self.pre_feedforward_layernorm)?)?
363            .apply(&self.post_feedforward_layernorm)?;
364        residual + xs
365    }
366}

368pub struct EmbeddingGemma {
369    embed_tokens: ScaledEmbedding,
370    layers: Vec<DecoderLayer>,
371    norm: RmsNorm,
372    device: Device,
373    mapper: Box<dyn DeviceMapper + Send + Sync>,
374    sliding_window: usize,
375    final_logit_softcapping: Option<f64>,
376}
377
378impl EmbeddingGemma {
379    pub fn new(
380        cfg: &EmbeddingGemmaConfig,
381        vb: ShardedVarBuilder,
382        is_gptx: bool,
383        normal_loading_metadata: NormalLoadingMetadata,
384        attention_mechanism: AttentionImplementation,
385    ) -> Result<Self> {
386        if let Some(ref quant_cfg) = &cfg.quantization_config {
387            tracing::info!(
388                "Using {} quantization: {}.",
389                quant_cfg.name(),
390                quant_cfg.get_bits_name(&vb)
391            );
392        }
393
394        if !matches!(attention_mechanism, AttentionImplementation::Eager) {
395            candle_core::bail!("Expected AttentionImplementation::Eager");
396        }
397
398        let mapper = normal_loading_metadata.mapper;
399
400        let embed_tokens = ScaledEmbedding::new(
401            (cfg.hidden_size as f64).sqrt(),
402            embedding(
403                cfg.vocab_size,
404                cfg.hidden_size,
405                mapper.set_nm_device(vb.pp("embed_tokens"), false),
406                &cfg.quantization_config,
407            )?,
408        );
409
410        let mut global_ropes = HashMap::new();
411        for layer_idx in 0..cfg.num_hidden_layers {
412            let device = mapper
413                .device_for(layer_idx, false)
414                .unwrap_or(&normal_loading_metadata.real_device);
415            global_ropes.insert(
416                device.location(),
417                Arc::new(Gemma3RotaryEmbedding::new_embedding_gemma(
418                    is_gptx,
419                    vb.dtype(),
420                    cfg,
421                    device,
422                )?),
423            );
424        }
425
426        let mut local_ropes = HashMap::new();
427        for layer_idx in 0..cfg.num_hidden_layers {
428            let device = mapper
429                .device_for(layer_idx, false)
430                .unwrap_or(&normal_loading_metadata.real_device);
431            local_ropes.insert(
432                device.location(),
433                Arc::new(RotaryEmbedding::new(
434                    cfg.rope_local_base_freq as f32,
435                    cfg.head_dim,
436                    cfg.max_position_embeddings,
437                    device,
438                    is_gptx,
439                    vb.dtype(),
440                )?),
441            );
442        }
443
444        let vb_l = vb.pp("layers");
445        let layers = NiceProgressBar::<_, 'b'>(
446            0..cfg.num_hidden_layers,
447            "Loading repeating layers",
448            &normal_loading_metadata.multi_progress,
449        )
450        .par_iter_if_isq(|layer_idx| {
451            let device = mapper
452                .device_for(layer_idx, false)
453                .unwrap_or(&normal_loading_metadata.real_device);
454            let rotary_emb_global = global_ropes
455                .get(&device.location())
456                .expect("No RoPE for device location!")
457                .clone();
458            let rotary_emb_local = local_ropes
459                .get(&device.location())
460                .expect("No RoPE for device location!")
461                .clone();
462            let comm = mapper.get_comm_for(layer_idx)?;
463            DecoderLayer::new(
464                rotary_emb_global,
465                rotary_emb_local,
466                cfg,
467                vb_l.pp(layer_idx),
468                &*mapper,
469                layer_idx,
470                normal_loading_metadata.loading_isq,
471                &comm,
472            )
473        })?;
474        let norm = RmsNorm::new_gemma(
475            cfg.hidden_size,
476            cfg.rms_norm_eps,
477            mapper.set_nm_device(vb.pp("norm"), false),
478        )?;
479
480        Ok(Self {
481            embed_tokens,
482            layers,
483            norm,
484            device: normal_loading_metadata.real_device,
485            sliding_window: cfg.sliding_window,
486            final_logit_softcapping: cfg.final_logit_softcapping,
487            mapper,
488        })
489    }
490
491    pub fn embed_tokens(&self, input_ids: &Tensor) -> Result<Tensor> {
492        self.embed_tokens.forward(input_ids)
493    }
494
495    pub fn forward_embeds(
496        &self,
497        input_ids: &Tensor,
498        mut xs: Tensor,
499        flash_params: &FlashParams,
500    ) -> Result<Tensor> {
501        let (bs, _seqlen) = input_ids.dims2()?;
502        let seqlen_offsets = vec![0; bs];
503
504        let attention_mask = BidirectionalMasker.make_mask(input_ids, xs.dtype())?;
505        let sliding_mask =
506            BidirectionalMasker.make_sliding_mask(input_ids, xs.dtype(), self.sliding_window)?;
507
508        for (i, layer) in self.layers.iter().enumerate() {
509            xs = self.mapper.map(xs, i)?;
510            xs = layer.forward(
511                &xs,
512                &attention_mask.to_device(xs.device())?,
513                &sliding_mask.to_device(xs.device())?,
514                &seqlen_offsets,
515                flash_params,
516            )?;
517        }
518        let xs = xs.to_device(&self.device)?;
519        let mut xs = xs.apply(&self.norm)?;
520
521        if let Some(final_logit_softcapping) = self.final_logit_softcapping {
522            xs = (xs / final_logit_softcapping)?;
523            xs = xs.tanh()?;
524            xs = (xs * final_logit_softcapping)?;
525        }
526
527        Ok(xs)
528    }
529}

531impl IsqModel for EmbeddingGemma {
532    fn get_layers(
533        &mut self,
534    ) -> (
535        Vec<(&mut Arc<dyn QuantMethod>, Option<usize>)>,
536        &dyn DeviceMapper,
537    ) {
538        let mut tensors = Vec::new();
539        for (i, layer) in self.layers.iter_mut().enumerate() {
540            tensors.push((&mut layer.self_attn.q_proj, Some(i)));
541            tensors.push((&mut layer.self_attn.k_proj, Some(i)));
542            tensors.push((&mut layer.self_attn.v_proj, Some(i)));
543            tensors.push((&mut layer.self_attn.o_proj, Some(i)));
544            tensors.extend(
545                layer
546                    .mlp
547                    .get_isq_layers()
548                    .into_iter()
549                    .map(|m| (m, Some(i)))
550                    .collect::<Vec<_>>(),
551            );
552        }
553        (tensors, &*self.mapper)
554    }
555
556    fn residual_tensors(&self) -> Vec<(String, Tensor)> {
557        let uvb = UnVarBuilder::new();
558
559        uvb.pp("embed_tokens").add(&self.embed_tokens);
560        uvb.pp("norm").add(&self.norm.undo_gemma().unwrap());
561
562        for (layer_idx, layer) in self.layers.iter().enumerate() {
563            let uvb_l = uvb.pp("layers").pp(layer_idx);
564            uvb_l
565                .pp("self_attn")
566                .pp("q_norm")
567                .add(&layer.self_attn.q_norm.undo_gemma().unwrap());
568            uvb_l
569                .pp("self_attn")
570                .pp("k_norm")
571                .add(&layer.self_attn.k_norm.undo_gemma().unwrap());
572            uvb_l
573                .pp("input_layernorm")
574                .add(&layer.input_layernorm.undo_gemma().unwrap());
575            uvb_l
576                .pp("post_attention_layernorm")
577                .add(&layer.post_attention_layernorm.undo_gemma().unwrap());
578            uvb_l
579                .pp("pre_feedforward_layernorm")
580                .add(&layer.pre_feedforward_layernorm.undo_gemma().unwrap());
581            uvb_l
582                .pp("post_feedforward_layernorm")
583                .add(&layer.post_feedforward_layernorm.undo_gemma().unwrap());
584        }
585
586        uvb.to_safetensors()
587    }

589    fn imatrix_names(&self) -> candle_core::Result<Vec<Option<String>>> {
590        // NOTE: dependant on the exact implementation in get_layers!
591        let mut names = Vec::new();
592        // lm_head
593        names.push(None);
594        for i in 0..self.layers.len() {
595            names.push(Some(format!("blk.{i}.attn_q.weight")));
596            names.push(Some(format!("blk.{i}.attn_k.weight")));
597            names.push(Some(format!("blk.{i}.attn_v.weight")));
598            names.push(Some(format!("blk.{i}.attn_output.weight")));
599            names.push(Some(format!("blk.{i}.ffn_gate.weight")));
600            names.push(Some(format!("blk.{i}.ffn_up.weight")));
601            names.push(Some(format!("blk.{i}.ffn_down.weight")));
602        }
603        Ok(names)
604    }
