26/// Bucket key: (sequence length, has_images && is_prompt, token_offset)
27/// We bucket sequences by these criteria to ensure all sequences in a batch have the same
28/// length, avoiding padding issues with flash attention varlen.
29type BucketKey = (usize, bool, usize);
30
31/// Allow sequences to wait for 64 scheduling passes before warning of deprivation.
32const WAITING_TIMEOUT: usize = 64;
33
34pub struct PagedAttentionSchedulerOutput {
35    /// Either ALL prompt or ALL completion.
36    pub scheduled: Vec<Arc<Mutex<Sequence>>>,
37    pub blocks_to_copy: HashMap<SrcBlockFrom, DstBlocksTo>,
38}

54    pub fn new(config: PagedAttentionSchedulerConfig, cache_config: CacheConfig) -> Self {
55        // Default to enabled, Engine::new will call set_prefix_caching_enabled
56        // based on the global no_prefix_cache flag
57        Self {
58            waiting: VecDeque::new(),
59            running: VecDeque::new(),
60            block_engine: Arc::new(tokio::sync::Mutex::new(BlockEngine::new(
61                cache_config.block_size,
62                cache_config.num_gpu_blocks,
63                true, // Default enabled, will be configured by Engine
64            ))),
65            block_size: cache_config.block_size,
66            config,
67            prefix_caching_enabled: true,
68        }

71    /// Set whether prefix caching is enabled. This also updates the block engine.
72    pub fn set_prefix_caching_enabled_sync(&mut self, enabled: bool) {
73        self.prefix_caching_enabled = enabled;
74        if enabled {
75            info!("Prefix caching enabled (block-level, PagedAttention). Expect higher multi-turn throughput for both text and multimodal.");
76        }
77        // Update the block engine - we need to block on the async mutex
78        // This is called once at startup, so blocking is acceptable
79        let block_engine = self.block_engine.clone();
80        tokio::task::block_in_place(|| {
81            let rt = tokio::runtime::Handle::current();
82            rt.block_on(async {
83                block_engine
84                    .lock()
85                    .await
86                    .set_prefix_caching_enabled(enabled);
87            });
88        });
89    }

91    /// Bucket sequences by (length, has_images && is_prompt, token_offset).
92    /// Returns the bucket with the shortest sequence length; sequences from other buckets
93    /// are preempted (blocks freed, state set to Waiting, added to waiting queue).
94    ///
95    /// This ensures all sequences in a batch have the same length, which is required for
96    /// correct flash attention varlen operation (avoiding soundness issues with padding).
97    ///
98    /// Also removes preempted sequences from self.running.
99    fn bucket_and_preempt_sequences(
100        &mut self,
101        sequences: VecDeque<Arc<Mutex<Sequence>>>,
102    ) -> VecDeque<Arc<Mutex<Sequence>>> {
103        if sequences.len() <= 1 {
104            return sequences;
105        }
106
107        let mut buckets: HashMap<BucketKey, VecDeque<Arc<Mutex<Sequence>>>> = HashMap::new();
108
109        for seq in sequences {
110            let seq_guard = get_mut_arcmutex!(seq);
111            let key: BucketKey = (
112                seq_guard.len(),
113                seq_guard.images().is_some() && seq_guard.is_prompt(),
114                seq_guard.token_offset(),
115            );
116            drop(seq_guard);
117
118            buckets.entry(key).or_default().push_back(seq);
119        }
120
121        if buckets.len() == 1 {
122            // All sequences are in the same bucket, return them all
123            return buckets.into_values().next().unwrap();
124        }
125
126        // Find the bucket with the shortest sequence length
127        let min_key = *buckets
128            .keys()
129            .min_by_key(|(len, _, _)| *len)
130            .expect("No sequence buckets");
131
132        let selected = buckets.remove(&min_key).unwrap();
133
134        // Collect IDs of sequences to preempt
135        let mut ids_to_preempt = Vec::new();
136
137        // Preempt sequences from other buckets (free blocks, set state to Waiting, add to waiting)
138        for (_, seqs) in buckets {
139            for seq in seqs.into_iter().rev() {
140                ids_to_preempt.push(get_mut_arcmutex!(seq).get_id());
141                self._preempt_by_recompute(seq);
142            }
143        }
144
145        // Remove preempted sequences from self.running
146        self.running
147            .retain(|seq| !ids_to_preempt.contains(&get_mut_arcmutex!(seq).get_id()));
148
149        selected
150    }

152    pub fn schedule(&mut self, logger: &IntervalLogger) -> PagedAttentionSchedulerOutput {
153        let mut scheduled: VecDeque<Arc<Mutex<Sequence>>> = VecDeque::new();
154        let mut for_waiting_again: VecDeque<Arc<Mutex<Sequence>>> = VecDeque::new();
155        let mut did_ignore = false;
156        while !self.waiting.is_empty() {
157            let seq = self.waiting.front().unwrap().clone();
158
159            if self.config.max_num_seqs == self.running.len() + 1 {
160                break;
161            }
162
163            let can_allocate =
164                get_mut_arcmutex!(self.block_engine).can_allocate(&mut *get_mut_arcmutex!(seq));
165            match can_allocate {
166                AllocStatus::Later { waitlisted_count } => {
167                    if waitlisted_count > WAITING_TIMEOUT {
168                        if let Some(seq_to_preempt) = self.running.pop_back() {
169                            self._preempt_by_recompute(seq_to_preempt);
170                            if !matches!(
171                                get_mut_arcmutex!(self.block_engine)
172                                    .can_allocate(&mut *get_mut_arcmutex!(seq)),
173                                AllocStatus::Ok
174                            ) {
175                                let id = *get_mut_arcmutex!(seq).id();
176                                let len = get_mut_arcmutex!(seq).get_toks().len();
177                                warn!(
178                                    "Sequence {id} with length of {len} tokens still exceeds KV cache size \
179                                     even after evicting another sequence.",
180                                );
181                                get_mut_arcmutex!(seq).set_state(SequenceState::FinishedIgnored);
182                                did_ignore = true;
183                            }
184                        } else {
185                            let id = *get_mut_arcmutex!(seq).id();
186                            let len = get_mut_arcmutex!(seq).get_toks().len();
187                            warn!(
188                                "Sequence {id} with length of {len} tokens is too long and exceeds KV cache size. \
189                                 To fix, increase the maximum sequence length for the KV cache, for example with \
190                                 `--max-seq-len`/ `max_seq_len` in automatic device mapping parameters.",
191                            );
192                            get_mut_arcmutex!(seq).set_state(SequenceState::FinishedIgnored);
193                            did_ignore = true;
194                        }
195                    } else {
196                        break;
197                    }
198                }
199                AllocStatus::Impossible => {
200                    let id = *get_mut_arcmutex!(seq).id();
201                    let len = get_mut_arcmutex!(seq).get_toks().len();
202                    warn!(
203                        "Sequence {id} with length of {len} tokens is too long and exceeds KV cache size. To fix, increase the maximum sequence length for the KV cache, for example with `--max-seq-len`/ `max_seq_len` in automatic device mapping parameters.",
204                    );
205                    get_mut_arcmutex!(seq).set_state(SequenceState::FinishedIgnored);
206                    did_ignore = true;
207                }
208                _ => {}
209            }
210
211            let new_seq_has_images = get_mut_arcmutex!(seq).has_images();
212            if !scheduled.is_empty()
213                && get_mut_arcmutex!(scheduled[0]).has_images() != new_seq_has_images
214            {
215                let seq = self.waiting.pop_front().unwrap();
216                for_waiting_again.push_back(seq.clone());
217                continue;
218            }
219            if !did_ignore {
220                get_mut_arcmutex!(seq).set_state(SequenceState::RunningPrompt);
221                let mut seq_handle = get_mut_arcmutex!(seq);
222                self._allocate(&mut seq_handle);
223                // Check for prefix cache hit and report to logger
224                let seq_id = seq_handle.get_id();
225                if get_mut_arcmutex!(self.block_engine).last_allocate_had_cache_hit(seq_id) > 0 {
226                    logger.add_prefix_cache_hit();
227                }
228            }
229
230            let seq = self.waiting.pop_front().unwrap();
231            self.running.push_back(seq.clone());
232            if !did_ignore {
233                scheduled.push_back(seq);
234            }
235        }
236        self.waiting.extend(for_waiting_again);
237
238        if !scheduled.is_empty() || did_ignore {
239            // Bucket scheduled prompts by sequence length to ensure all sequences in a batch
240            // have the same length (required for correct flash attention varlen operation).
241            let scheduled = self.bucket_and_preempt_sequences(scheduled);
242
243            logger.set_num_running(self.running.len());
244            logger.set_num_waiting(self.waiting.len());
245
246            return PagedAttentionSchedulerOutput {
247                scheduled: scheduled.into_iter().collect(),
248                blocks_to_copy: HashMap::new(),
249            };
250        }
251
252        let mut blocks_to_copy = HashMap::new();
253
254        // Reserve token slots for the running sequence groups, preempting the lowest (earliest) first.
255        // Preempt lowest priority sequences that are in the running queue, forming a
256        // new running queue that has the actually running sequences. Remember the preempted
257        // sequences, which will be put into the waiting or swapped out state depending on
258        // the preemption method (recompute or swap, respectively).
259
260        // Sorts by creation time, in descending order so that earliest are latest (first come first serve).
261        self.sort_running_by_priority_fcfs();
262
263        let mut running: VecDeque<Arc<Mutex<Sequence>>> = VecDeque::new();
264        while !self.running.is_empty() {
265            let seq = self.running.pop_front().unwrap();
266            let mut finished_with_break = false;
267            while !get_mut_arcmutex!(self.block_engine)
268                .can_append_token_to_seq(&*get_mut_arcmutex!(seq))
269            {
270                // If we cannot, now we need to preempt some seqs
271                if !self.running.is_empty() {
272                    // There is something to preempt.
273                    let seq_to_preempt = self.running.pop_back().unwrap();
274                    self._preempt(seq_to_preempt);
275                } else {
276                    // Nothing to preempt, preempt ourselves. Also, do not bother looking at anything else.
277                    self._preempt(seq.clone());
278                    finished_with_break = true;
279                    break;
280                }
281            }
282            if !finished_with_break {
283                {
284                    // If we need to, append physical blocks for a new token. We do not need to if there is enough space.
285                    // If we just got preempted, there is no reason to allocate
286                    let seq_handle = get_mut_arcmutex!(seq);
287                    self._append_token_slot_to_seq(&seq_handle, &mut blocks_to_copy);
288                }
289                let new_seq_has_images = get_mut_arcmutex!(seq).has_images();
290                // Only add it if has_images matches either current or there are none.
291                if running.is_empty()
292                    || get_mut_arcmutex!(running[0]).has_images() == new_seq_has_images
293                {
294                    running.push_back(seq);
295                } else {
296                    self.running.push_back(seq);
297                }
298            }
299        }
300        self.running = running;
301
302        // Bucket running completions by sequence length to ensure all sequences in a batch
303        // have the same length (required for correct flash attention varlen operation).
304        let running_for_bucket = std::mem::take(&mut self.running);
305        let bucketed = self.bucket_and_preempt_sequences(running_for_bucket);
306        self.running = bucketed;
307
308        self.running
309            .iter()
310            .for_each(|seq| get_mut_arcmutex!(seq).set_state(SequenceState::RunningCompletion));
311
312        if TERMINATE_ALL_NEXT_STEP.load(Ordering::SeqCst) {
313            self.running.iter().for_each(|seq| {
314                get_mut_arcmutex!(seq).set_state(SequenceState::Done(StopReason::Canceled))
315            });
316            TERMINATE_ALL_NEXT_STEP.store(false, Ordering::SeqCst);
317        }
318
319        logger.set_num_running(self.running.len());
320        logger.set_num_waiting(self.waiting.len());
321
322        PagedAttentionSchedulerOutput {
323            scheduled: self.running.clone().into_iter().collect(),
324            blocks_to_copy,
325        }
326    }

328    pub fn free_finished_sequence_groups(&mut self) {
329        let mut to_free: Vec<(usize, Vec<super::LogicalTokenBlock>)> = Vec::new();
330        self.running.retain(|seq| {
331            if get_mut_arcmutex!(seq).is_finished_paged_attn() {
332                let seq_guard = get_mut_arcmutex!(seq);
333                let id = seq_guard.get_id();
334                // Get logical blocks for caching (clone them since we're dropping the lock)
335                let logical_blocks = seq_guard.logical_token_blocks().to_vec();
336                drop(seq_guard);
337                to_free.push((id, logical_blocks));
338                false
339            } else {
340                true
341            }
342        });
343
344        for (id, logical_blocks) in to_free {
345            self._free_with_caching(id, Some(&logical_blocks));
346        }
347    }

350impl PagedAttentionScheduler {
351    #[allow(dead_code)]
352    fn remove_seq(&mut self, seq_id: usize) -> Arc<Mutex<Sequence>> {
353        // Remove it if it is in waiting
354        if let Some(idx) = self
355            .waiting
356            .iter()
357            .position(|other| get_mut_arcmutex!(other).get_id() == seq_id)
358        {
359            return self.waiting.remove(idx).unwrap();
360        };
361        // Remove it if it is in running
362        if let Some(idx) = self
363            .running
364            .iter()
365            .position(|other| get_mut_arcmutex!(other).get_id() == seq_id)
366        {
367            return self.running.remove(idx).unwrap();
368        };
369        panic!("Attempted to remove sequence id {seq_id} but it is not running or waiting.");
370    }
371
372    fn _append_token_slot_to_seq(
373        &mut self,
374        seq: &Sequence,
375        blocks_to_copy: &mut HashMap<usize, Vec<usize>>,
376    ) {
377        let op = get_mut_arcmutex!(self.block_engine).append_token_slot_to_seq(seq);
378        if let Some((src_block, dst_block)) = op {
379            if let std::collections::hash_map::Entry::Vacant(e) = blocks_to_copy.entry(src_block) {
380                e.insert(vec![dst_block]);
381            } else {
382                blocks_to_copy.get_mut(&src_block).unwrap().push(dst_block);
383            }
384        }
385    }
386
387    fn _abort_seq(&mut self, seq_id: usize) {
388        let removed = self.remove_seq(seq_id);
389        get_mut_arcmutex!(removed).set_state(SequenceState::FinishedAborted);
390        self._free(seq_id);
391    }
392
393    /// Preempt sequences by dropping their cache and recomputing later.
394    fn _preempt(&mut self, seq: Arc<Mutex<Sequence>>) {
395        self._preempt_by_recompute(seq)
396    }
397
398    fn _preempt_by_recompute(&mut self, seq: Arc<Mutex<Sequence>>) {
399        let seq_guard = get_mut_arcmutex!(seq);
400        seq_guard.set_state(SequenceState::Waiting);
401        let seq_id = seq_guard.get_id();
402        // Get logical blocks for proper cache ref release
403        let logical_blocks = seq_guard.logical_token_blocks().to_vec();
404        drop(seq_guard);
405        // Use free_with_caching to properly release prefix cache refs
406        // (but don't add blocks to cache since we're preempting)
407        self._free_for_preemption(seq_id, &logical_blocks);
408        self.waiting.push_front(seq);
409    }
410
411    fn _allocate(&mut self, seq: &mut Sequence) {
412        get_mut_arcmutex!(self.block_engine).allocate(seq)
413    }
414
415    fn _free(&mut self, seq_id: usize) {
416        get_mut_arcmutex!(self.block_engine).free_sequence(seq_id);
417    }
418
419    fn _free_for_preemption(&mut self, seq_id: usize, logical_blocks: &[super::LogicalTokenBlock]) {
420        get_mut_arcmutex!(self.block_engine).free_sequence_for_preemption(seq_id, logical_blocks);
421    }
422
423    fn _free_with_caching(
424        &mut self,
425        seq_id: usize,
426        logical_blocks: Option<&[super::LogicalTokenBlock]>,
427    ) {
428        get_mut_arcmutex!(self.block_engine).free_sequence_with_caching(seq_id, logical_blocks);
429    }
430
431    fn sort_running_by_priority_fcfs(&mut self) {
432        self.running
433            .make_contiguous()
434            .sort_by_key(|seq| get_mut_arcmutex!(seq).timestamp());
435        self.running.make_contiguous().reverse();
436    }
437}

439impl Scheduler for PagedAttentionScheduler {
440    fn add_seq(&mut self, seq: Sequence) {
441        self.waiting.push_back(Arc::new(Mutex::new(seq)));
442    }
443    fn schedule(&mut self, logger: &IntervalLogger) -> SchedulerOutput<'_> {
444        SchedulerOutput::PagedAttention {
445            output: self.schedule(logger),
446        }
447    }
448    fn waiting_len(&self) -> usize {
449        self.waiting.len()
450    }
451    fn running_len(&self) -> usize {
452        self.running.len()
453    }
454    fn block_tables(&self) -> Option<BlockTables> {
455        Some(get_mut_arcmutex!(self.block_engine).block_tables.clone())
456    }
457    fn block_size(&self) -> Option<usize> {
458        Some(self.block_size)
459    }
460    fn free_finished_sequence_groups(&mut self) {
461        self.free_finished_sequence_groups()
462    }
463    fn get_finished_mamba_indices(&self) -> Vec<usize> {
464        self.running
465            .iter()
466            .filter(|seq| get_mut_arcmutex!(seq).is_finished_paged_attn())
467            .filter_map(|seq| get_mut_arcmutex!(seq).mamba_state_idx())
468            .collect()
469    }
470    fn block_engine(&self) -> Option<Arc<tokio::sync::Mutex<BlockEngine>>> {
471        Some(self.block_engine.clone())
472    }
473    fn set_prefix_caching_enabled(&mut self, enabled: bool) {
474        self.set_prefix_caching_enabled_sync(enabled);
475    }
