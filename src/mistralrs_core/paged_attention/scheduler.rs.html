26/// Bucket key: (sequence length, has_images && is_prompt, token_offset)
27/// We bucket sequences by these criteria to ensure all sequences in a batch have the same
28/// length, avoiding padding issues with flash attention varlen.
29type BucketKey = (usize, bool, usize);
30
31/// Allow sequences to wait for 64 scheduling passes before warning of deprivation.
32const WAITING_TIMEOUT: usize = 64;
33
34pub struct PagedAttentionSchedulerOutput {
35    /// Either ALL prompt or ALL completion.
36    pub scheduled: Vec<Arc<Mutex<Sequence>>>,
37    pub blocks_to_copy: HashMap<SrcBlockFrom, DstBlocksTo>,
38}

52impl PagedAttentionScheduler {
53    pub fn new(config: PagedAttentionSchedulerConfig, cache_config: CacheConfig) -> Self {
54        Self {
55            waiting: VecDeque::new(),
56            running: VecDeque::new(),
57            config,
58            block_engine: Arc::new(tokio::sync::Mutex::new(BlockEngine::new(
59                cache_config.block_size,
60                cache_config.num_gpu_blocks,
61            ))),
62            block_size: cache_config.block_size,
63        }
64    }

66    /// Bucket sequences by (length, has_images && is_prompt, token_offset).
67    /// Returns the bucket with the shortest sequence length; sequences from other buckets
68    /// are preempted (blocks freed, state set to Waiting, added to waiting queue).
69    ///
70    /// This ensures all sequences in a batch have the same length, which is required for
71    /// correct flash attention varlen operation (avoiding soundness issues with padding).
72    ///
73    /// Also removes preempted sequences from self.running.
74    fn bucket_and_preempt_sequences(
75        &mut self,
76        sequences: VecDeque<Arc<Mutex<Sequence>>>,
77    ) -> VecDeque<Arc<Mutex<Sequence>>> {
78        if sequences.len() <= 1 {
79            return sequences;
80        }
81
82        let mut buckets: HashMap<BucketKey, VecDeque<Arc<Mutex<Sequence>>>> = HashMap::new();
83
84        for seq in sequences {
85            let seq_guard = get_mut_arcmutex!(seq);
86            let key: BucketKey = (
87                seq_guard.len(),
88                seq_guard.images().is_some() && seq_guard.is_prompt(),
89                seq_guard.token_offset(),
90            );
91            drop(seq_guard);
92
93            buckets.entry(key).or_default().push_back(seq);
94        }
95
96        if buckets.len() == 1 {
97            // All sequences are in the same bucket, return them all
98            return buckets.into_values().next().unwrap();
99        }
100
101        // Find the bucket with the shortest sequence length
102        let min_key = *buckets
103            .keys()
104            .min_by_key(|(len, _, _)| *len)
105            .expect("No sequence buckets");
106
107        let selected = buckets.remove(&min_key).unwrap();
108
109        // Collect IDs of sequences to preempt
110        let mut ids_to_preempt = Vec::new();
111
112        // Preempt sequences from other buckets (free blocks, set state to Waiting, add to waiting)
113        for (_, seqs) in buckets {
114            for seq in seqs.into_iter().rev() {
115                ids_to_preempt.push(get_mut_arcmutex!(seq).get_id());
116                self._preempt_by_recompute(seq);
117            }
118        }
119
120        // Remove preempted sequences from self.running
121        self.running
122            .retain(|seq| !ids_to_preempt.contains(&get_mut_arcmutex!(seq).get_id()));
123
124        selected
125    }

127    pub fn schedule(&mut self, logger: &IntervalLogger) -> PagedAttentionSchedulerOutput {
128        let mut scheduled: VecDeque<Arc<Mutex<Sequence>>> = VecDeque::new();
129        let mut for_waiting_again: VecDeque<Arc<Mutex<Sequence>>> = VecDeque::new();
130        let mut did_ignore = false;
131        while !self.waiting.is_empty() {
132            let seq = self.waiting.front().unwrap().clone();
133
134            if self.config.max_num_seqs == self.running.len() + 1 {
135                break;
136            }
137
138            let can_allocate =
139                get_mut_arcmutex!(self.block_engine).can_allocate(&mut *get_mut_arcmutex!(seq));
140            match can_allocate {
141                AllocStatus::Later { waitlisted_count } => {
142                    if waitlisted_count > WAITING_TIMEOUT {
143                        if let Some(seq_to_preempt) = self.running.pop_back() {
144                            self._preempt_by_recompute(seq_to_preempt);
145                            if !matches!(
146                                get_mut_arcmutex!(self.block_engine)
147                                    .can_allocate(&mut *get_mut_arcmutex!(seq)),
148                                AllocStatus::Ok
149                            ) {
150                                let id = *get_mut_arcmutex!(seq).id();
151                                let len = get_mut_arcmutex!(seq).get_toks().len();
152                                warn!(
153                                    "Sequence {id} with length of {len} tokens still exceeds KV cache size \
154                                     even after evicting another sequence.",
155                                );
156                                get_mut_arcmutex!(seq).set_state(SequenceState::FinishedIgnored);
157                                did_ignore = true;
158                            }
159                        } else {
160                            let id = *get_mut_arcmutex!(seq).id();
161                            let len = get_mut_arcmutex!(seq).get_toks().len();
162                            warn!(
163                                "Sequence {id} with length of {len} tokens is too long and exceeds KV cache size. \
164                                 To fix, increase the maximum sequence length for the KV cache, for example with \
165                                 `--max-seq-len`/ `max_seq_len` in automatic device mapping parameters.",
166                            );
167                            get_mut_arcmutex!(seq).set_state(SequenceState::FinishedIgnored);
168                            did_ignore = true;
169                        }
170                    } else {
171                        break;
172                    }
173                }
174                AllocStatus::Impossible => {
175                    let id = *get_mut_arcmutex!(seq).id();
176                    let len = get_mut_arcmutex!(seq).get_toks().len();
177                    warn!(
178                        "Sequence {id} with length of {len} tokens is too long and exceeds KV cache size. To fix, increase the maximum sequence length for the KV cache, for example with `--max-seq-len`/ `max_seq_len` in automatic device mapping parameters.",
179                    );
180                    get_mut_arcmutex!(seq).set_state(SequenceState::FinishedIgnored);
181                    did_ignore = true;
182                }
183                _ => {}
184            }
185
186            let new_seq_has_images = get_mut_arcmutex!(seq).has_images();
187            if !scheduled.is_empty()
188                && get_mut_arcmutex!(scheduled[0]).has_images() != new_seq_has_images
189            {
190                let seq = self.waiting.pop_front().unwrap();
191                for_waiting_again.push_back(seq.clone());
192                continue;
193            }
194            if !did_ignore {
195                get_mut_arcmutex!(seq).set_state(SequenceState::RunningPrompt);
196                let mut seq_handle = get_mut_arcmutex!(seq);
197                self._allocate(&mut seq_handle);
198            }
199
200            let seq = self.waiting.pop_front().unwrap();
201            self.running.push_back(seq.clone());
202            if !did_ignore {
203                scheduled.push_back(seq);
204            }
205        }
206        self.waiting.extend(for_waiting_again);
207
208        if !scheduled.is_empty() || did_ignore {
209            // Bucket scheduled prompts by sequence length to ensure all sequences in a batch
210            // have the same length (required for correct flash attention varlen operation).
211            let scheduled = self.bucket_and_preempt_sequences(scheduled);
212
213            logger.set_num_running(self.running.len());
214            logger.set_num_waiting(self.waiting.len());
215
216            return PagedAttentionSchedulerOutput {
217                scheduled: scheduled.into_iter().collect(),
218                blocks_to_copy: HashMap::new(),
219            };
220        }
221
222        let mut blocks_to_copy = HashMap::new();
223
224        // Reserve token slots for the running sequence groups, preempting the lowest (earliest) first.
225        // Preempt lowest priority sequences that are in the running queue, forming a
226        // new running queue that has the actually running sequences. Remember the preempted
227        // sequences, which will be put into the waiting or swapped out state depending on
228        // the preemption method (recompute or swap, respectively).
229
230        // Sorts by creation time, in descending order so that earliest are latest (first come first serve).
231        self.sort_running_by_priority_fcfs();
232
233        let mut running: VecDeque<Arc<Mutex<Sequence>>> = VecDeque::new();
234        while !self.running.is_empty() {
235            let seq = self.running.pop_front().unwrap();
236            let mut finished_with_break = false;
237            while !get_mut_arcmutex!(self.block_engine)
238                .can_append_token_to_seq(&*get_mut_arcmutex!(seq))
239            {
240                // If we cannot, now we need to preempt some seqs
241                if !self.running.is_empty() {
242                    // There is something to preempt.
243                    let seq_to_preempt = self.running.pop_back().unwrap();
244                    self._preempt(seq_to_preempt);
245                } else {
246                    // Nothing to preempt, preempt ourselves. Also, do not bother looking at anything else.
247                    self._preempt(seq.clone());
248                    finished_with_break = true;
249                    break;
250                }
251            }
252            if !finished_with_break {
253                {
254                    // If we need to, append physical blocks for a new token. We do not need to if there is enough space.
255                    // If we just got preempted, there is no reason to allocate
256                    let seq_handle = get_mut_arcmutex!(seq);
257                    self._append_token_slot_to_seq(&seq_handle, &mut blocks_to_copy);
258                }
259                let new_seq_has_images = get_mut_arcmutex!(seq).has_images();
260                // Only add it if has_images matches either current or there are none.
261                if running.is_empty()
262                    || get_mut_arcmutex!(running[0]).has_images() == new_seq_has_images
263                {
264                    running.push_back(seq);
265                } else {
266                    self.running.push_back(seq);
267                }
268            }
269        }
270        self.running = running;
271
272        // Bucket running completions by sequence length to ensure all sequences in a batch
273        // have the same length (required for correct flash attention varlen operation).
274        let running_for_bucket = std::mem::take(&mut self.running);
275        let bucketed = self.bucket_and_preempt_sequences(running_for_bucket);
276        self.running = bucketed;
277
278        self.running
279            .iter()
280            .for_each(|seq| get_mut_arcmutex!(seq).set_state(SequenceState::RunningCompletion));
281
282        if TERMINATE_ALL_NEXT_STEP.load(Ordering::SeqCst) {
283            self.running.iter().for_each(|seq| {
284                get_mut_arcmutex!(seq).set_state(SequenceState::Done(StopReason::Canceled))
285            });
286            TERMINATE_ALL_NEXT_STEP.store(false, Ordering::SeqCst);
287        }
288
289        logger.set_num_running(self.running.len());
290        logger.set_num_waiting(self.waiting.len());
291
292        PagedAttentionSchedulerOutput {
293            scheduled: self.running.clone().into_iter().collect(),
294            blocks_to_copy,
295        }
296    }

298    pub fn free_finished_sequence_groups(&mut self) {
299        let mut to_free_ids = Vec::new();
300        self.running.retain(|seq| {
301            if get_mut_arcmutex!(seq).is_finished_paged_attn() {
302                to_free_ids.push(get_mut_arcmutex!(seq).get_id());
303                false
304            } else {
305                true
306            }
307        });
308
309        for id in to_free_ids {
310            self._free(id);
311        }
312    }

315impl PagedAttentionScheduler {
316    #[allow(dead_code)]
317    fn remove_seq(&mut self, seq_id: usize) -> Arc<Mutex<Sequence>> {
318        // Remove it if it is in waiting
319        if let Some(idx) = self
320            .waiting
321            .iter()
322            .position(|other| get_mut_arcmutex!(other).get_id() == seq_id)
323        {
324            return self.waiting.remove(idx).unwrap();
325        };
326        // Remove it if it is in running
327        if let Some(idx) = self
328            .running
329            .iter()
330            .position(|other| get_mut_arcmutex!(other).get_id() == seq_id)
331        {
332            return self.running.remove(idx).unwrap();
333        };
334        panic!("Attempted to remove sequence id {seq_id} but it is not running or waiting.");
335    }
336
337    fn _append_token_slot_to_seq(
338        &mut self,
339        seq: &Sequence,
340        blocks_to_copy: &mut HashMap<usize, Vec<usize>>,
341    ) {
342        let op = get_mut_arcmutex!(self.block_engine).append_token_slot_to_seq(seq);
343        if let Some((src_block, dst_block)) = op {
344            if let std::collections::hash_map::Entry::Vacant(e) = blocks_to_copy.entry(src_block) {
345                e.insert(vec![dst_block]);
346            } else {
347                blocks_to_copy.get_mut(&src_block).unwrap().push(dst_block);
348            }
349        }
350    }
351
352    fn _abort_seq(&mut self, seq_id: usize) {
353        let removed = self.remove_seq(seq_id);
354        get_mut_arcmutex!(removed).set_state(SequenceState::FinishedAborted);
355        self._free(seq_id);
356    }
357
358    /// Preempt sequences by dropping their cache and recomputing later.
359    fn _preempt(&mut self, seq: Arc<Mutex<Sequence>>) {
360        self._preempt_by_recompute(seq)
361    }
362
363    fn _preempt_by_recompute(&mut self, seq: Arc<Mutex<Sequence>>) {
364        get_mut_arcmutex!(seq).set_state(SequenceState::Waiting);
365        self._free(get_mut_arcmutex!(seq).get_id());
366        self.waiting.push_front(seq);
367    }
368
369    fn _allocate(&mut self, seq: &mut Sequence) {
370        get_mut_arcmutex!(self.block_engine).allocate(seq)
371    }
372
373    fn _free(&mut self, seq_id: usize) {
374        get_mut_arcmutex!(self.block_engine).free_sequence(seq_id);
375    }
376
377    fn sort_running_by_priority_fcfs(&mut self) {
378        self.running
379            .make_contiguous()
380            .sort_by_key(|seq| get_mut_arcmutex!(seq).timestamp());
381        self.running.make_contiguous().reverse();
382    }
383}

385impl Scheduler for PagedAttentionScheduler {
386    fn add_seq(&mut self, seq: Sequence) {
387        self.waiting.push_back(Arc::new(Mutex::new(seq)));
388    }
389    fn schedule(&mut self, logger: &IntervalLogger) -> SchedulerOutput<'_> {
390        SchedulerOutput::PagedAttention {
391            output: self.schedule(logger),
392        }
393    }
394    fn waiting_len(&self) -> usize {
395        self.waiting.len()
396    }
397    fn running_len(&self) -> usize {
398        self.running.len()
399    }
400    fn block_tables(&self) -> Option<BlockTables> {
401        Some(get_mut_arcmutex!(self.block_engine).block_tables.clone())
402    }
403    fn block_size(&self) -> Option<usize> {
404        Some(self.block_size)
405    }
406    fn free_finished_sequence_groups(&mut self) {
407        self.free_finished_sequence_groups()
408    }
409    fn get_finished_mamba_indices(&self) -> Vec<usize> {
410        self.running
411            .iter()
412            .filter(|seq| get_mut_arcmutex!(seq).is_finished_paged_attn())
413            .filter_map(|seq| get_mut_arcmutex!(seq).mamba_state_idx())
414            .collect()
415    }
416    fn block_engine(&self) -> Option<Arc<tokio::sync::Mutex<BlockEngine>>> {
417        Some(self.block_engine.clone())
418    }
