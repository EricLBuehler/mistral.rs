14#[derive(Clone, Copy, Debug, Serialize, Deserialize, PartialEq, Default)]
15#[cfg_attr(feature = "pyo3_macros", pyo3::pyclass(eq, eq_int))]
16pub enum PagedCacheType {
17    #[default]
18    Auto,
19    F8E4M3,
20}
21
22impl PagedCacheType {
23    pub fn to_dtype(&self, act_dtype: DType) -> DType {
24        match self {
25            PagedCacheType::F8E4M3 => DType::F8E4M3,
26            PagedCacheType::Auto => act_dtype,
27        }
28    }
29}
30
31impl FromStr for PagedCacheType {
32    type Err = String;
33    fn from_str(s: &str) -> std::result::Result<Self, Self::Err> {
34        match s {
35            "auto" => Ok(Self::Auto),
36            "f8e4m3" => Ok(Self::F8E4M3),
37            other => Err(format!(
38                "Unexpected `PagedCacheType`, got `{other}` but expected `auto` and `f8e4m3`."
39            )),
40        }
41    }
42}

51pub type KVCache = (Tensor, Tensor);

53pub struct CacheEngine {
54    gpu_cache: Arc<Mutex<Vec<KVCache>>>,
55}
56
57impl CacheEngine {
58    pub fn new(
59        model_config: &dyn ModelConfigLike,
60        cache_config: &CacheConfig,
61        dtype: DType,
62        device: &Device,
63        layer_devices: Vec<Option<Device>>,
64    ) -> Result<Self> {
65        let dtype = cache_config.cache_type.to_dtype(dtype);
66        Ok(Self {
67            gpu_cache: Arc::new(Mutex::new(Self::allocate_gpu_cache(
68                model_config,
69                cache_config,
70                dtype,
71                device,
72                layer_devices,
73            )?)),
74        })
75    }
76
77    pub fn get_kv_cache(&self) -> MutexGuard<'_, Vec<KVCache>> {
78        // Use blocking lock instead of busy-wait spin loop to avoid CPU waste
79        // and potential thread starvation issues
80        self.gpu_cache.lock().expect("KV cache mutex was poisoned")
81    }
82
83    fn allocate_gpu_cache(
84        model_config: &dyn ModelConfigLike,
85        cache_config: &CacheConfig,
86        dtype: DType,
87        device: &Device,
88        layer_devices: Vec<Option<Device>>,
89    ) -> Result<Vec<KVCache>> {
90        let kv_cache_layout = model_config.kv_cache_layout();
91        let mut gpu_cache = Vec::new();
92
93        for device in layer_devices
94            .iter()
95            .take(model_config.num_layers())
96            .map(|x| x.as_ref().unwrap_or(device))
97        {
98            let (key_blocks, value_blocks) = match kv_cache_layout {
99                KvCacheLayout::Standard => {
100                    let key_block_shape = Self::calculate_key_block_shape(
101                        model_config,
102                        dtype,
103                        cache_config.block_size,
104                    );
105                    let value_block_shape =
106                        Self::calculate_value_block_shape(model_config, cache_config.block_size);
107                    #[allow(unused)]
108                    let key_blocks = if let Device::Metal(dev) = &device {
109                        #[cfg(feature = "metal")]
110                        {
111                            use candle_core::{MetalStorage, Shape, Storage};
112
113                            let elem_count = cache_config.num_gpu_blocks
114                                * key_block_shape.0
115                                * key_block_shape.1
116                                * key_block_shape.2
117                                * key_block_shape.3;
118                            let buffer = dev.new_private_buffer(elem_count, dtype, "k_cache")?;
119                            let storage = Storage::Metal(MetalStorage::new(
120                                buffer,
121                                dev.clone(),
122                                elem_count,
123                                dtype,
124                            ));
125                            Tensor::from((
126                                storage,
127                                Shape::from_dims(&[
128                                    cache_config.num_gpu_blocks,
129                                    key_block_shape.0,
130                                    key_block_shape.1,
131                                    key_block_shape.2,
132                                    key_block_shape.3,
133                                ]),
134                            ))
135                        }
136
137                        #[cfg(not(feature = "metal"))]
138                        {
139                            unreachable!()
140                        }
141                    } else {
142                        unsafe {
143                            Tensor::empty(
144                                (
145                                    cache_config.num_gpu_blocks,
146                                    key_block_shape.0,
147                                    key_block_shape.1,
148                                    key_block_shape.2,
149                                    key_block_shape.3,
150                                ),
151                                dtype,
152                                device,
153                            )?
154                        }
155                    };
156                    #[allow(unused)]
157                    let value_blocks = if let Device::Metal(dev) = &device {
158                        #[cfg(feature = "metal")]
159                        {
160                            use candle_core::{MetalStorage, Shape, Storage};
161
162                            let elem_count = cache_config.num_gpu_blocks
163                                * value_block_shape.0
164                                * value_block_shape.1
165                                * value_block_shape.2;
166                            let buffer = dev.new_private_buffer(elem_count, dtype, "v_cache")?;
167                            let storage = Storage::Metal(MetalStorage::new(
168                                buffer,
169                                dev.clone(),
170                                elem_count,
171                                dtype,
172                            ));
173                            Tensor::from((
174                                storage,
175                                Shape::from_dims(&[
176                                    cache_config.num_gpu_blocks,
177                                    value_block_shape.0,
178                                    value_block_shape.1,
179                                    value_block_shape.2,
180                                ]),
181                            ))
182                        }
183
184                        #[cfg(not(feature = "metal"))]
185                        {
186                            unreachable!()
187                        }
188                    } else {
189                        unsafe {
190                            Tensor::empty(
191                                (
192                                    cache_config.num_gpu_blocks,
193                                    value_block_shape.0,
194                                    value_block_shape.1,
195                                    value_block_shape.2,
196                                ),
197                                dtype,
198                                device,
199                            )?
200                        }
201                    };
202                    (key_blocks, value_blocks)
203                }
204                KvCacheLayout::Mla {
205                    kv_lora_rank,
206                    kpe_head_dim,
207                } => {
208                    #[allow(unused)]
209                    let key_blocks = if let Device::Metal(dev) = &device {
210                        #[cfg(feature = "metal")]
211                        {
212                            use candle_core::{MetalStorage, Shape, Storage};
213
214                            let elem_count = cache_config.num_gpu_blocks
215                                * cache_config.block_size
216                                * kv_lora_rank;
217                            let buffer = dev.new_private_buffer(elem_count, dtype, "k_cache")?;
218                            let storage = Storage::Metal(MetalStorage::new(
219                                buffer,
220                                dev.clone(),
221                                elem_count,
222                                dtype,
223                            ));
224                            Tensor::from((
225                                storage,
226                                Shape::from_dims(&[
227                                    cache_config.num_gpu_blocks,
228                                    cache_config.block_size,
229                                    kv_lora_rank,
230                                ]),
231                            ))
232                        }
233
234                        #[cfg(not(feature = "metal"))]
235                        {
236                            unreachable!()
237                        }
238                    } else {
239                        unsafe {
240                            Tensor::empty(
241                                (
242                                    cache_config.num_gpu_blocks,
243                                    cache_config.block_size,
244                                    kv_lora_rank,
245                                ),
246                                dtype,
247                                device,
248                            )?
249                        }
250                    };
251                    #[allow(unused)]
252                    let value_blocks = if let Device::Metal(dev) = &device {
253                        #[cfg(feature = "metal")]
254                        {
255                            use candle_core::{MetalStorage, Shape, Storage};
256
257                            let elem_count = cache_config.num_gpu_blocks
258                                * cache_config.block_size
259                                * kpe_head_dim;
260                            let buffer = dev.new_private_buffer(elem_count, dtype, "v_cache")?;
261                            let storage = Storage::Metal(MetalStorage::new(
262                                buffer,
263                                dev.clone(),
264                                elem_count,
265                                dtype,
266                            ));
267                            Tensor::from((
268                                storage,
269                                Shape::from_dims(&[
270                                    cache_config.num_gpu_blocks,
271                                    cache_config.block_size,
272                                    kpe_head_dim,
273                                ]),
274                            ))
275                        }
276
277                        #[cfg(not(feature = "metal"))]
278                        {
279                            unreachable!()
280                        }
281                    } else {
282                        unsafe {
283                            Tensor::empty(
284                                (
285                                    cache_config.num_gpu_blocks,
286                                    cache_config.block_size,
287                                    kpe_head_dim,
288                                ),
289                                dtype,
290                                device,
291                            )?
292                        }
293                    };
294                    (key_blocks, value_blocks)
295                }
296            };
297            gpu_cache.push((key_blocks, value_blocks));
298        }
299        Ok(gpu_cache)
300    }
301
302    fn calculate_key_block_shape(
303        model_config: &dyn ModelConfigLike,
304        dtype: DType,
305        block_size: usize,
306    ) -> (usize, usize, usize, usize) {
307        let element_size = dtype.size_in_bytes();
308        let x = 16 / element_size;
309        (
310            model_config.num_kv_heads(),
311            model_config.k_head_dim() / x,
312            block_size,
313            x,
314        )
315    }
316
317    fn calculate_value_block_shape(
318        model_config: &dyn ModelConfigLike,
319        block_size: usize,
320    ) -> (usize, usize, usize) {
321        (
322            model_config.num_kv_heads(),
323            model_config.v_head_dim(),
324            block_size,
325        )
326    }
327}

329impl CacheEngine {
330    pub fn execute_scheduler_ops(&self, blocks_to_copy: &HashMap<usize, Vec<usize>>) -> Result<()> {
331        if !blocks_to_copy.is_empty() {
332            self.copy(blocks_to_copy)?;
333        }
334        Ok(())
335    }
336
337    pub fn copy(&self, src_to_dst: &HashMap<usize, Vec<usize>>) -> Result<()> {
338        #[cfg(any(all(feature = "cuda", target_family = "unix"), feature = "metal"))]
339        {
340            let mut gpu_cache = self.get_kv_cache();
341            #[allow(clippy::map_identity)]
342            let caches: (Vec<&mut Tensor>, Vec<&mut Tensor>) =
343                gpu_cache.iter_mut().map(|(a, b)| (a, b)).unzip();
344            let (key_caches, value_caches) = caches;
345
346            // NOTE(EricLBuehler): This may synchronize the CPU and GPU
347            copy_blocks(key_caches, value_caches, src_to_dst)?;
348            Ok(())
349        }
350        #[cfg(not(any(all(feature = "cuda", target_family = "unix"), feature = "metal")))]
351        {
352            let _ = src_to_dst;
353            candle_core::bail!("Paged attention requires the CUDA or Metal feature flags.");
354        }
355    }
