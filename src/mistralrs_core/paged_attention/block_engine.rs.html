69/// Inner state for a single physical block.
70/// When this is dropped (last Arc reference gone), the block returns to the free list.
71struct BlockInner {
72    block_id: usize,
73    pool: Arc<BlockPoolInner>,
74}
75
76impl Drop for BlockInner {
77    fn drop(&mut self) {
78        // Automatically return block to free list when last reference is dropped
79        self.pool
80            .free_list
81            .lock()
82            .expect("BlockPool free_list lock poisoned")
83            .push(self.block_id);
84    }
85}
86
87/// Shared pool state accessed by all BlockRef instances.
88struct BlockPoolInner {
89    free_list: Mutex<Vec<usize>>,
90    #[allow(dead_code)]
91    block_size: usize,
92    #[allow(dead_code)]
93    num_blocks: usize,
94}

96/// A reference-counted handle to a physical block.
97///
98/// This works exactly like `Arc`:
99/// - Clone to share (automatic refcount increment)
100/// - Drop when done (automatic refcount decrement)
101/// - When the last reference is dropped, the block automatically returns to the free pool
102///
103/// This design makes refcount bugs impossible - there's no manual increment/decrement.
104#[derive(Clone)]
105pub struct BlockRef(Arc<BlockInner>);
106
107impl std::fmt::Debug for BlockRef {
108    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
109        f.debug_struct("BlockRef")
110            .field("block_id", &self.0.block_id)
111            .field("ref_count", &Arc::strong_count(&self.0))
112            .finish()
113    }
114}

151/// The block pool that manages physical block allocation.
152///
153/// Blocks are automatically returned to the pool when all references are dropped.
154pub struct BlockPool {
155    inner: Arc<BlockPoolInner>,
156}
157
158impl BlockPool {
159    /// Create a new block pool with the given number of blocks.
160    pub fn new(block_size: usize, num_blocks: usize) -> Self {
161        let free_list: Vec<usize> = (0..num_blocks).collect();
162        Self {
163            inner: Arc::new(BlockPoolInner {
164                free_list: Mutex::new(free_list),
165                block_size,
166                num_blocks,
167            }),
168        }
169    }
170
171    /// Allocate a block from the pool. Returns None if no blocks are available.
172    pub fn allocate(&self) -> Option<BlockRef> {
173        let block_id = self
174            .inner
175            .free_list
176            .lock()
177            .expect("BlockPool free_list lock poisoned")
178            .pop()?;
179
180        Some(BlockRef(Arc::new(BlockInner {
181            block_id,
182            pool: Arc::clone(&self.inner),
183        })))
184    }
185
186    /// Get the number of free blocks.
187    #[inline]
188    pub fn num_free(&self) -> usize {
189        self.inner
190            .free_list
191            .lock()
192            .expect("BlockPool free_list lock poisoned")
193            .len()
194    }
195
196    /// Get the block size.
197    #[inline]
198    #[allow(dead_code)]
199    pub fn block_size(&self) -> usize {
200        self.inner.block_size
201    }
202
203    /// Get the total number of blocks in the pool.
204    #[inline]
205    #[allow(dead_code)]
206    pub fn num_blocks(&self) -> usize {
207        self.inner.num_blocks
208    }
209}

224/// A BlockEngine maps each Sequence (identified by its SeqID), to physical token blocks.
225/// The physical token blocks may not match the logical token blocks because during
226/// scheduling, physical blocks are allocated to accommodate the new tokens generated.
227/// These new tokens will be added to the logical token block for each sequence.
228pub struct BlockEngine {
229    num_gpu_blocks: usize,
230    block_size: usize,
231    /// The block pool for allocation. Blocks are automatically returned when refs are dropped.
232    pool: BlockPool,
233    /// Maps sequence ID to its block table.
234    pub block_tables: HashMap<SeqID, BlockTable>,
235    /// Prefix cache for reusing KV cache blocks across requests with shared prefixes.
236    prefix_cacher: PrefixCacher,
237    /// Track number of cached blocks used per sequence (for releasing cache entries).
238    cached_blocks_per_seq: HashMap<SeqID, usize>,
239}

268    pub fn can_allocate(&mut self, seq: &mut impl BlockEngineSequence) -> AllocStatus {
269        let logical_blocks = seq.logical_token_blocks();
270        let num_required_blocks = logical_blocks.len();
271
272        // Check how many blocks we can get from prefix cache
273        let num_cached = if self.prefix_cacher.is_enabled() {
274            let (_, num_matched) = self.prefix_cacher.match_prefix(logical_blocks);
275            num_matched
276        } else {
277            0
278        };
279
280        // We only need to allocate blocks that aren't in the cache
281        let num_new_blocks_needed = num_required_blocks.saturating_sub(num_cached);
282        let num_free_gpu_blocks = self.pool.num_free();
283
284        // Also count evictable blocks from prefix cache as potentially available
285        let num_evictable = self.prefix_cacher.num_evictable_blocks();
286        let total_available = num_free_gpu_blocks + num_evictable;
287
288        if self.num_gpu_blocks < num_required_blocks {
289            AllocStatus::Impossible
290        } else if total_available < num_new_blocks_needed {
291            AllocStatus::Later {
292                waitlisted_count: seq.increment_waitlist_count(),
293            }
294        } else {
295            AllocStatus::Ok
296        }
297    }

299    pub fn allocate(&mut self, seq: &mut impl BlockEngineSequence) {
300        let num_blocks_needed = seq.logical_token_blocks().len();
301        let seq_id = seq.get_id();
302        let block_size = seq.block_size();
303
304        // If there are prefill physical blocks, use those here.
305        if let Some(physical_blocks_prefill) = seq.take_physical_blocks_prefill() {
306            let mut block_table = physical_blocks_prefill;
307            let n_extra_blocks = num_blocks_needed - block_table.len();
308            for _ in 0..n_extra_blocks {
309                block_table.push(self.allocate_block_with_eviction());
310            }
311            self.block_tables.insert(seq_id, block_table);
312            self.cached_blocks_per_seq.insert(seq_id, 0);
313            seq.set_prefix_cache_len(0);
314            return;
315        }
316
317        // Re-borrow logical_blocks after the mutable borrow above is done
318        let logical_blocks = seq.logical_token_blocks();
319
320        // Try to get blocks from prefix cache
321        let (cached_blocks, num_cached) = if self.prefix_cacher.is_enabled() {
322            self.prefix_cacher.match_prefix(logical_blocks)
323        } else {
324            (Vec::new(), 0)
325        };
326
327        let mut block_table = Vec::with_capacity(num_blocks_needed);
328
329        // Use cached blocks for the prefix (they're already cloned by match_prefix)
330        for (idx, block_ref) in cached_blocks {
331            // Extend block_table to the right size
332            while block_table.len() < idx {
333                block_table.push(self.allocate_block_with_eviction());
334            }
335            block_table.push(block_ref);
336        }
337
338        // Allocate new blocks for the rest
339        for _ in block_table.len()..num_blocks_needed {
340            block_table.push(self.allocate_block_with_eviction());
341        }
342
343        self.cached_blocks_per_seq.insert(seq_id, num_cached);
344        self.block_tables.insert(seq_id, block_table);
345
346        // Calculate number of cached tokens (full blocks only)
347        let cached_tokens = num_cached * block_size;
348        seq.set_prefix_cache_len(cached_tokens);
349    }

351    /// Check if the last allocate() call resulted in a prefix cache hit.
352    /// Returns the number of blocks that were reused from cache.
353    pub fn last_allocate_had_cache_hit(&self, seq_id: usize) -> usize {
354        self.cached_blocks_per_seq
355            .get(&seq_id)
356            .copied()
357            .unwrap_or(0)
358    }

360    /// Allocate a block, evicting from prefix cache if necessary.
361    fn allocate_block_with_eviction(&mut self) -> BlockRef {
362        // Try to allocate from free pool first
363        if let Some(block_ref) = self.pool.allocate() {
364            return block_ref;
365        }
366
367        // Need to evict from prefix cache
368        // Evicting drops the cache's reference, which may return blocks to the pool
369        self.prefix_cacher.evict_blocks(1);
370
371        // Now allocate
372        self.pool
373            .allocate()
374            .expect("Should have free blocks after eviction")
375    }

377    pub fn can_append_token_to_seq(&self, seq: &impl BlockEngineSequence) -> bool {
378        let free_blocks = self.pool.num_free();
379        let evictable = self.prefix_cacher.num_evictable_blocks();
380        seq.blocks_to_add_new_tok() <= free_blocks + evictable
381    }

383    /// Free a sequence's blocks and optionally cache them for prefix reuse.
384    /// If `logical_blocks` is provided and prefix caching is enabled, full blocks
385    /// will be added to the prefix cache for potential reuse by future requests.
386    pub fn free_sequence_with_caching(
387        &mut self,
388        id: usize,
389        logical_blocks: Option<&[LogicalTokenBlock]>,
390    ) {
391        if let Some(block_table) = self.block_tables.remove(&id) {
392            let num_cached = self.cached_blocks_per_seq.remove(&id).unwrap_or(0);
393
394            // Cache blocks for prefix reuse (skip already-cached blocks)
395            if let Some(logical_blocks) = logical_blocks {
396                if self.prefix_cacher.is_enabled() && block_table.len() == logical_blocks.len() {
397                    // Insert new blocks into cache (starting after cached blocks)
398                    // The cache will clone the BlockRefs, keeping them alive
399                    self.prefix_cacher
400                        .insert_blocks(logical_blocks, &block_table, num_cached);
401                }
402            }
403
404            // Release cached blocks' active user counts in the prefix cacher
405            if num_cached > 0 {
406                if let Some(logical_blocks) = logical_blocks {
407                    let cached_logical = &logical_blocks[..num_cached.min(logical_blocks.len())];
408                    self.prefix_cacher.release_blocks(cached_logical);
409                }
410            }
411
412            // The block_table is dropped here, which automatically decrements refcounts.
413            // Blocks that are only held by the cache will stay alive.
414            // Blocks that are not cached will return to the free pool.
415        }
416    }
417
418    pub fn free_sequence(&mut self, id: usize) {
419        // Free without caching (for aborted sequences or when we don't have logical blocks)
420        if let Some(_block_table) = self.block_tables.remove(&id) {
421            self.cached_blocks_per_seq.remove(&id);
422            // block_table is dropped here, automatically returning blocks to pool
423        }
424    }

426    /// Free a sequence's blocks during preemption.
427    /// This properly releases prefix cache active user counts.
428    pub fn free_sequence_for_preemption(
429        &mut self,
430        id: usize,
431        logical_blocks: &[LogicalTokenBlock],
432    ) {
433        if let Some(_block_table) = self.block_tables.remove(&id) {
434            let num_cached = self.cached_blocks_per_seq.remove(&id).unwrap_or(0);
435
436            // Release cached blocks' active user counts in the prefix cache
437            if num_cached > 0 && self.prefix_cacher.is_enabled() {
438                let cached_logical = &logical_blocks[..num_cached.min(logical_blocks.len())];
439                self.prefix_cacher.release_blocks(cached_logical);
440            }
441
442            // block_table is dropped here, automatically returning blocks to pool
443        }
444    }

446    /// Returns the COW mapping (src, dst).
447    /// COW is performed if there are multiple references to the last physical block.
448    pub fn append_token_slot_to_seq(
449        &mut self,
450        sequence: &impl BlockEngineSequence,
451    ) -> Option<(usize, usize)> {
452        let seq_id = sequence.get_id();
453        let blocks_to_add = sequence.blocks_to_add_new_tok();
454
455        // Check if table exists
456        if !self.block_tables.contains_key(&seq_id) {
457            return None;
458        }
459
460        match blocks_to_add {
461            1 => {
462                // Allocate first, then push to table
463                let new_block = self.allocate_block_with_eviction();
464                self.block_tables.get_mut(&seq_id).unwrap().push(new_block);
465                None
466            }
467            0 => {
468                // Get the last block
469                let table = self.block_tables.get(&seq_id).unwrap();
470                let last_block = table.last().unwrap();
471
472                if !last_block.is_shared() {
473                    None
474                } else {
475                    // We would be writing into shared, so COW.
476                    let old_number = last_block.block_id();
477
478                    // Allocate new block
479                    let new_block = self.allocate_block_with_eviction();
480                    let new_number = new_block.block_id();
481
482                    // Replace in table (old block ref is dropped, decrementing its refcount)
483                    let table = self.block_tables.get_mut(&seq_id).unwrap();
484                    *table.last_mut().unwrap() = new_block;
485
486                    Some((old_number, new_number))
487                }
488            }
489            _ => {
490                unreachable!()
491            }
492        }
493    }
