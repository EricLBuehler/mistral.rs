141struct Allocator<T> {
142    free_blocks: BlockTable,
143    _ghost: PhantomData<T>,
144}
145
146impl<T> Allocator<T> {
147    fn allocate(&mut self) -> Arc<PhysicalTokenBlock> {
148        let block = self.free_blocks.pop().unwrap();
149        block.deref_mut().refcount = 1;
150        block
151    }
152
153    fn free_block(&mut self, block: Arc<PhysicalTokenBlock>) {
154        if block.deref_mut().refcount == 0 {
155            panic!(
156                "PhysicalTokenBlock with id {} experienced a double free!",
157                block.deref_mut().block_id
158            );
159        }
160        block.deref_mut().refcount -= 1;
161        if block.deref_mut().refcount == 0 {
162            self.free_blocks.push(block);
163        }
164    }
165}

167impl Allocator<GPUAllocator> {
168    fn new(block_size: usize, num_blocks: usize) -> Self {
169        let mut free_blocks = Vec::new();
170        for id in 0..num_blocks {
171            free_blocks.push(Arc::new(PhysicalTokenBlock(Mutex::new(
172                _PhysicalTokenBlock {
173                    block_id: id,
174                    block_size,
175                    refcount: 0,
176                    is_gpu: true,
177                },
178            ))))
179        }
180        Allocator {
181            free_blocks,
182            _ghost: PhantomData,
183        }
184    }
185
186    fn get_num_free_blocks(&self) -> GPUAllocatorWrapper {
187        GPUAllocatorWrapper(self.free_blocks.len())
188    }

200/// A BlockEngine maps each Sequence (identified by its SeqID), to physical token blocks.
201/// The physical token blocks may not match the logical token blocks because during
202/// scheduling, physical blocks are allocated to accommodate the new tokens generated.
203/// These new tokens will be added to the logical token block for each sequence.
204pub struct BlockEngine {
205    num_gpu_blocks: usize,
206    block_size: usize,
207    gpu_allocator: Allocator<GPUAllocator>,
208    pub block_tables: HashMap<SeqID, BlockTable>,
209    /// Prefix cache for reusing KV cache blocks across requests with shared prefixes.
210    prefix_cacher: PrefixCacher,
211    /// Track number of cached blocks used per sequence (for freeing).
212    cached_blocks_per_seq: HashMap<SeqID, usize>,
213}
214
215pub type BlockTables = HashMap<usize, BlockTable>;

244    pub fn can_allocate(&mut self, seq: &mut impl BlockEngineSequence) -> AllocStatus {
245        let logical_blocks = seq.logical_token_blocks();
246        let num_required_blocks = logical_blocks.len();
247
248        // Check how many blocks we can get from prefix cache
249        let num_cached = if self.prefix_cacher.is_enabled() {
250            let (_, num_matched) = self.prefix_cacher.match_prefix(logical_blocks);
251            num_matched
252        } else {
253            0
254        };
255
256        // We only need to allocate blocks that aren't in the cache
257        let num_new_blocks_needed = num_required_blocks.saturating_sub(num_cached);
258        let num_free_gpu_blocks = self.gpu_allocator.get_num_free_blocks();
259
260        // Also count evictable blocks from prefix cache as potentially available
261        let num_evictable = self.prefix_cacher.num_evictable_blocks();
262        let total_available = *num_free_gpu_blocks + num_evictable;
263
264        if self.num_gpu_blocks < num_required_blocks {
265            AllocStatus::Impossible
266        } else if total_available < num_new_blocks_needed {
267            AllocStatus::Later {
268                waitlisted_count: seq.increment_waitlist_count(),
269            }
270        } else {
271            AllocStatus::Ok
272        }
273    }

275    pub fn allocate(&mut self, seq: &mut impl BlockEngineSequence) {
276        let num_blocks_needed = seq.logical_token_blocks().len();
277        let seq_id = seq.get_id();
278        let block_size = seq.block_size();
279
280        // If there are prefill physical blocks, use those here.
281        if let Some(physical_blocks_prefill) = seq.take_physical_blocks_prefill() {
282            let mut block_table = physical_blocks_prefill.clone();
283            let n_extra_blocks = num_blocks_needed - block_table.len();
284            for _ in 0..n_extra_blocks {
285                block_table.push(self.allocate_block_with_eviction());
286            }
287            self.block_tables.insert(seq_id, block_table);
288            self.cached_blocks_per_seq.insert(seq_id, 0);
289            seq.set_prefix_cache_len(0);
290            return;
291        }
292
293        // Re-borrow logical_blocks after the mutable borrow above is done
294        let logical_blocks = seq.logical_token_blocks();
295
296        // Try to get blocks from prefix cache
297        let (cached_blocks, num_cached) = if self.prefix_cacher.is_enabled() {
298            self.prefix_cacher.match_prefix(logical_blocks)
299        } else {
300            (Vec::new(), 0)
301        };
302
303        let mut block_table = Vec::with_capacity(num_blocks_needed);
304
305        // Use cached blocks for the prefix
306        for (idx, physical_block) in cached_blocks {
307            // Extend block_table to the right size
308            while block_table.len() < idx {
309                block_table.push(self.allocate_block_with_eviction());
310            }
311            // The cached block already has its refcount incremented by match_prefix
312            block_table.push(physical_block);
313        }
314
315        // Allocate new blocks for the rest
316        for _ in block_table.len()..num_blocks_needed {
317            block_table.push(self.allocate_block_with_eviction());
318        }
319
320        self.cached_blocks_per_seq.insert(seq_id, num_cached);
321        self.block_tables.insert(seq_id, block_table);
322
323        // Calculate number of cached tokens (full blocks only)
324        // num_cached is the number of full blocks that were cache hits
325        let cached_tokens = num_cached * block_size;
326        seq.set_prefix_cache_len(cached_tokens);
327    }

338    /// Allocate a block, evicting from prefix cache if necessary.
339    fn allocate_block_with_eviction(&mut self) -> Arc<PhysicalTokenBlock> {
340        // Try to allocate from free pool first
341        if *self.gpu_allocator.get_num_free_blocks() > 0 {
342            return self.gpu_allocator.allocate();
343        }
344
345        // Need to evict from prefix cache
346        let evicted = self.prefix_cacher.evict_blocks(1);
347        for block in evicted {
348            // Decrement refcount and return to free pool
349            block.deref_mut().decrement_refcount();
350            if block.deref_mut().refcount == 0 {
351                self.gpu_allocator.free_blocks.push(block);
352            }
353        }
354
355        // Now allocate
356        self.gpu_allocator.allocate()
357    }

359    pub fn can_append_token_to_seq(&self, seq: &impl BlockEngineSequence) -> bool {
360        let free_blocks = self.gpu_allocator.get_num_free_blocks();
361        let evictable = self.prefix_cacher.num_evictable_blocks();
362        // Physical blocks = logical blocks
363        seq.blocks_to_add_new_tok() <= *free_blocks + evictable
364    }

366    /// Free a sequence's blocks and optionally cache them for prefix reuse.
367    /// If `logical_blocks` is provided and prefix caching is enabled, full blocks
368    /// will be added to the prefix cache for potential reuse by future requests.
369    pub fn free_sequence_with_caching(
370        &mut self,
371        id: usize,
372        logical_blocks: Option<&[LogicalTokenBlock]>,
373    ) {
374        // Handle double free if run out of tokens
375        if let Some(block_table) = self.block_tables.remove(&id) {
376            let num_cached = self.cached_blocks_per_seq.remove(&id).unwrap_or(0);
377
378            // Cache blocks for prefix reuse (skip already-cached blocks)
379            if let Some(logical_blocks) = logical_blocks {
380                if self.prefix_cacher.is_enabled() && block_table.len() == logical_blocks.len() {
381                    // Insert new blocks into cache (starting after cached blocks)
382                    self.prefix_cacher
383                        .insert_blocks(logical_blocks, &block_table, num_cached);
384                }
385            }
386
387            // Release cached blocks' reference counts
388            if num_cached > 0 {
389                if let Some(logical_blocks) = logical_blocks {
390                    let cached_logical = &logical_blocks[..num_cached.min(logical_blocks.len())];
391                    self.prefix_cacher.release_blocks(cached_logical);
392                }
393            }
394
395            // Free non-cached blocks (or all if not caching)
396            for (idx, block) in block_table.iter().enumerate() {
397                // Skip blocks that were from cache and are now in cache
398                // (they have refcount > 1 due to cache holding a reference)
399                if idx < num_cached && self.prefix_cacher.is_enabled() {
400                    // This was a cached block - just decrement our reference
401                    self.gpu_allocator.free_block(block.clone());
402                } else if self.prefix_cacher.is_enabled() && logical_blocks.is_some() {
403                    // This block is being added to cache, so cache holds the ref
404                    // We don't free it to the allocator
405                } else {
406                    self.gpu_allocator.free_block(block.clone());
407                }
408            }
409        }
410    }

412    pub fn free_sequence(&mut self, id: usize) {
413        // Free without caching (for aborted sequences or when we don't have logical blocks)
414        if let Some(block_table) = self.block_tables.remove(&id) {
415            self.cached_blocks_per_seq.remove(&id);
416            // Free all blocks
417            for block in block_table.iter() {
418                self.gpu_allocator.free_block(block.clone());
419            }
420        }
421    }

423    /// Free a sequence's blocks during preemption.
424    /// This properly releases prefix cache refs so cached blocks can be evicted.
425    pub fn free_sequence_for_preemption(
426        &mut self,
427        id: usize,
428        logical_blocks: &[LogicalTokenBlock],
429    ) {
430        if let Some(block_table) = self.block_tables.remove(&id) {
431            let num_cached = self.cached_blocks_per_seq.remove(&id).unwrap_or(0);
432
433            // Release cached blocks' reference counts in the prefix cache
434            if num_cached > 0 && self.prefix_cacher.is_enabled() {
435                let cached_logical = &logical_blocks[..num_cached.min(logical_blocks.len())];
436                self.prefix_cacher.release_blocks(cached_logical);
437            }
438
439            // Free all blocks
440            for block in block_table.iter() {
441                self.gpu_allocator.free_block(block.clone());
442            }
443        }
444    }

446    // Returns the COW mapping (src, dst).
447    // COW is performed if there are multiple references to the last physical block.
448    pub fn append_token_slot_to_seq(
449        &mut self,
450        sequence: &impl BlockEngineSequence,
451    ) -> Option<(usize, usize)> {
452        let seq_id = sequence.get_id();
453        let blocks_to_add = sequence.blocks_to_add_new_tok();
454
455        // Check if table exists
456        if !self.block_tables.contains_key(&seq_id) {
457            return None;
458        }
459
460        match blocks_to_add {
461            1 => {
462                // Allocate first, then push to table
463                let new_block = self.allocate_block_with_eviction();
464                self.block_tables.get_mut(&seq_id).unwrap().push(new_block);
465                None
466            }
467            0 => {
468                // Get the last block's info first
469                let table = self.block_tables.get(&seq_id).unwrap();
470                let last_block = table.last().unwrap();
471                let is_gpu = last_block.deref_mut().is_gpu;
472                let refcount = last_block.deref_mut().refcount;
473
474                assert!(is_gpu);
475
476                if refcount == 1 {
477                    None
478                } else {
479                    // We would be writing into shared, so COW.
480                    let old_block = last_block.clone();
481                    let old_number = old_block.deref_mut().block_id;
482
483                    // Now allocate and mutate
484                    let new_block = self.allocate_block_with_eviction();
485                    let new_number = new_block.deref_mut().block_id;
486
487                    // Free old block
488                    self.gpu_allocator.free_block(old_block);
489
490                    // Replace in table
491                    let table = self.block_tables.get_mut(&seq_id).unwrap();
492                    *table.last_mut().unwrap() = new_block;
493
494                    Some((old_number, new_number))
495                }
496            }
497            _ => {
498                unreachable!()
499            }
500        }
501    }
