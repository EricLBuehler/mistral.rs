126impl GGUFLoaderBuilder {
127    /// Create a loader builder for a GGUF model. `tok_model_id` is the model ID where you can find a
128    /// `tokenizer_config.json` file. If the `chat_template` is specified, then it will be treated as a
129    /// path and used over remote files, removing all remote accesses.
130    pub fn new(
131        chat_template: Option<String>,
132        tok_model_id: Option<String>,
133        quantized_model_id: String,
134        quantized_filenames: Vec<String>,
135        config: GGUFSpecificConfig,
136        no_kv_cache: bool,
137        jinja_explicit: Option<String>,
138    ) -> Self {
139        let kind = ModelKind::GgufQuantized {
140            quant: QuantizationKind::Gguf,
141        };
142
143        Self {
144            chat_template,
145            model_id: tok_model_id,
146            kind,
147            quantized_filenames,
148            quantized_model_id,
149            config,
150            jinja_explicit,
151            no_kv_cache,
152            ..Default::default()
153        }
154    }
155
156    fn with_adapter(
157        mut self,
158        xlora_model_id: String,
159        xlora_order: Ordering,
160        no_kv_cache: bool,
161        tgt_non_granular_index: Option<usize>,
162    ) -> Self {
163        self.xlora_model_id = Some(xlora_model_id);
164        self.xlora_order = Some(xlora_order);
165        self.no_kv_cache = no_kv_cache;
166        self.tgt_non_granular_index = tgt_non_granular_index;
167        self.model_id = if let Some(id) = self.model_id {
168            Some(id)
169        } else {
170            info!(
171                "Using adapter base model ID: `{}`",
172                self.xlora_order.as_ref().unwrap().base_model_id
173            );
174            Some(self.xlora_order.as_ref().unwrap().base_model_id.clone())
175        };
176        self
177    }
178
179    pub fn with_xlora(
180        mut self,
181        xlora_model_id: String,
182        xlora_order: Ordering,
183        no_kv_cache: bool,
184        tgt_non_granular_index: Option<usize>,

192            tgt_non_granular_index,

198
199        self.with_adapter(lora_model_id, lora_order, false, None)

212            quantized_model_id: self.quantized_model_id,
213            config: self.config,
214            jinja_explicit: self.jinja_explicit,
215            lora_adapter_ids: None,

297    #[allow(clippy::type_complexity, clippy::too_many_arguments)]
298    fn load_model_from_path(
299        &self,
300        paths: &Box<dyn ModelPaths>,
301        dtype: &dyn TryIntoDType,
302        device: &Device,
303        silent: bool,
304        mut mapper: DeviceMapSetting,
305        in_situ_quant: Option<IsqType>,
306        mut paged_attn_config: Option<PagedAttentionConfig>,
307    ) -> Result<Arc<Mutex<dyn Pipeline + Send + Sync>>> {
308        let _progress_guard = ProgressScopeGuard::new(silent);
309        if in_situ_quant.is_some() {
310            anyhow::bail!(
311                "You are trying to in-situ quantize a GGUF model. This will not do anything."
312            );
313        }
314
315        info!("Prompt chunk size is {ATTENTION_CHUNK_SIZE}.");
316
317        let mut readers = Vec::new();
318        for filename in paths.get_weight_filenames() {
319            readers.push(std::fs::File::open(filename)?);
320        }
321        let mut readers = readers.iter_mut().collect::<Vec<_>>();
322
323        let model = Content::from_readers(&mut readers)?;
324        if !silent {
325            model.print_metadata()?;
326        }
327        let arch = model.arch();
328
329        // If auto, convert to Map
330        let num_layers = model.get_metadata()[&format!("{arch}.block_count")].to_u32()? as usize;
331        if let DeviceMapSetting::Auto(params) = mapper.clone() {
332            let devices = device_map::get_all_similar_devices(device)?;
333            // Initial dtype
334            let dtype = dtype.try_into_dtype(&devices.iter().collect::<Vec<_>>())?;
335
336            let model = GgufDeviceMapLoaderInner {
337                model: &model,
338                arch,
339            };
340
341            let layer_sizes_in_bytes =
342                model.layer_sizes_in_bytes("this is a dummy config!", dtype, 1, None)?;
343            let non_mapped_size_in_bytes =
344                model.non_mapped_size_in_bytes("this is a dummy config!", dtype, 1, None)?;
345            let total_model_size_in_bytes =
346                layer_sizes_in_bytes.iter().sum::<usize>() + non_mapped_size_in_bytes;
347
348            let new = model.get_device_layers(
349                "this is a dummy config!",
350                num_layers,
351                layer_sizes_in_bytes,
352                non_mapped_size_in_bytes,
353                total_model_size_in_bytes,
354                &devices,
355                dtype,
356                &params,
357                paged_attn_config.as_ref(),
358            )?;
359            mapper = DeviceMapSetting::Map(new);
360        }
361
362        #[cfg(feature = "cuda")]
363        if let Device::Cuda(dev) = &device {
364            unsafe { dev.disable_event_tracking() };
365        }
366
367        let use_nccl = mistralrs_quant::distributed::use_nccl();
368        let available_devices = if let Ok(payload) = env::var(distributed::IS_DAEMON_FLAG) {
369            let payload: WorkerTransferData = serde_json::from_str(&payload)?;
370            let WorkerTransferData::Init { id: _, worker_rank } = payload;
371            vec![candle_core::Device::new_cuda(worker_rank + 1)?]
372        } else if use_nccl {
373            vec![candle_core::Device::new_cuda(0)?]
374        } else {
375            device_map::get_all_similar_devices(device)?
376        };
377
378        let pipeline_mapper = mapper.into_mapper(
379            num_layers,
380            device,
381            self.config.topology.as_ref(),
382            &available_devices,
383        )?;
384        let mapper = mapper.into_mapper(
385            num_layers,
386            device,
387            self.config.topology.as_ref(),
388            &available_devices,
389        )?;
390        let mut layer_devices = Vec::new();
391        for layer in 0..num_layers {
392            let device = mapper.device_for(layer, false).cloned();
393            layer_devices.push(device);
394        }
395
396        // TODO: PagedAttention is not supported with CPU for now.
397        // This check is not really necessary because `get_device_layers` should prevent it.
398        let mapping_uses_cpu = mapper.get_unique_devices().iter().any(Device::is_cpu);
399        if mapping_uses_cpu {
400            warn!("Device mapping contains a mix of GPU and CPU. There is no CPU support for PagedAttention, disabling PagedAttention.");
401            paged_attn_config = None;
402        }
403
404        let GgufTokenizerConversion {
405            tokenizer,
406            bos,
407            eos,
408            unk,
409        } = if paths.get_tokenizer_filename().to_string_lossy().is_empty() {
410            convert_gguf_to_hf_tokenizer(&model)?
411        } else {
412            GgufTokenizerConversion {
413                tokenizer: get_tokenizer(paths.get_tokenizer_filename(), None)?,
414                bos: None,
415                eos: None,
416                unk: None,
417            }
418        };
419
420        // Only load gguf chat template if there is nothing else
421        let gguf_chat_template =
422            if paths.get_template_filename().is_none() && self.chat_template.is_none() {
423                get_gguf_chat_template(&model)?
424            } else {
425                None
426            };
427
428        let has_adapter = self.kind.is_adapted();
429        let is_xlora = self.kind.is_adapted_and(|a| a.is_x_lora());
430
431        let paged_attn_config = if matches!(self.kind, ModelKind::GgufAdapter { .. }) {
432            warn!("Adapter models do not currently support PagedAttention, running without");
433            None
434        } else {
435            paged_attn_config
436        };
437
438        let model_config_metadata: ContentConfig = (&model).into();
439        let internal_dtype = mapper.get_min_dtype(dtype)?;
440
441        let model_config = {
442            // Base config (quantization only):
443            let quant = ModelConfig::ParamsGGUF(
444                model,
445                (device, mapper).into(),
446                if paged_attn_config.is_some() {
447                    AttentionImplementation::PagedAttention
448                } else {
449                    AttentionImplementation::Eager
450                },
451                internal_dtype,
452            );
453
454            // With optional adapter config:
455            let mut adapter = None;
456            if has_adapter {
457                adapter.replace(ModelConfig::Adapter::try_new(
458                    paths, device, silent, is_xlora,
459                )?);
460            }
461
462            ModelConfig::ModelParams::new(quant, adapter)
463        };
464
465        // Config into model:
466        let model = match self.kind {
467            ModelKind::GgufQuantized { .. } => match arch {
468                GGUFArchitecture::Llama => Model::Llama(QLlama::try_from(model_config)?),
469                GGUFArchitecture::Phi2 => Model::Phi2(QPhi::try_from(model_config)?),
470                GGUFArchitecture::Phi3 => Model::Phi3(QPhi3::try_from(model_config)?),
471                GGUFArchitecture::Starcoder2 => {
472                    Model::Starcoder2(QStarcoder2::try_from(model_config)?)
473                }
474                GGUFArchitecture::Qwen2 => Model::Qwen(QQwen::try_from(model_config)?),
475                GGUFArchitecture::Qwen3 => Model::Qwen3(QQwen3::try_from(model_config)?),
476                GGUFArchitecture::Qwen3MoE => Model::Qwen3MoE(QQwen3MoE::try_from(model_config)?),
477                a => bail!("Unsupported architecture `{a:?}` for GGUF"),
478            },
479            ModelKind::GgufAdapter { adapter, .. } => match arch {
480                GGUFArchitecture::Llama => Model::XLoraLlama(XLoraQLlama::try_from(model_config)?),
481                GGUFArchitecture::Phi3 => Model::XLoraPhi3(XLoraQPhi3::try_from(model_config)?),
482                a => bail!(
483                    "Unsupported architecture `{a:?}` for GGUF {kind}",
484                    kind = adapter.pretty_name()
485                ),
486            },
487            _ => unreachable!(),
488        };
489
490        let (cache_config, cache_engine) = if let Some(paged_attn_config) = paged_attn_config {
491            let model_config: &dyn ModelConfigLike = &model_config_metadata;
492            let cache_config = calculate_cache_config(
493                paged_attn_config.mem_gpu,
494                paged_attn_config.block_size,
495                internal_dtype,
496                paged_attn_config.cache_type,
497                model_config,
498                device,
499                &layer_devices,
500                silent,
501            )?;
502            let cache_engine = CacheEngine::new(
503                model_config,
504                &cache_config,
505                internal_dtype,
506                device,
507                layer_devices,
508            )?;
509            (Some(cache_config), Some(cache_engine))
510        } else {
511            (None, None)
512        };
513
514        let gen_conf: Option<GenerationConfig> = paths
515            .get_gen_conf_filename()
516            .map(|f| serde_json::from_str(&fs::read_to_string(f).unwrap()).unwrap());
517        let chat_template_explicit = paths
518            .get_chat_template_explicit()
519            .as_ref()
520            .map(|x| x.to_string_lossy().to_string());
521        let mut chat_template = get_chat_template(
522            paths,
523            self.jinja_explicit.as_ref(),
524            chat_template_explicit.as_ref(),
525            self.chat_template.as_ref(),
526            gguf_chat_template,
527        );
528
529        let max_seq_len = match model {
530            Model::Llama(ref l) => l.max_seq_len,
531            Model::Phi2(ref p) => p.max_seq_len,
532            Model::XLoraLlama(ref xl) => xl.max_seq_len,
533            Model::Phi3(ref p) => p.max_seq_len,
534            Model::XLoraPhi3(ref p) => p.max_seq_len,
535            Model::Starcoder2(ref p) => p.max_seq_len,
536            Model::Qwen(ref p) => p.max_seq_len,
537            Model::Qwen3(ref p) => p.max_seq_len,
538            Model::Qwen3MoE(ref p) => p.max_seq_len,
539        };
540        let llg_factory = build_llg_factory(tokenizer.clone())?;
541        let num_hidden_layers = match model {
542            Model::Llama(ref model) => model.cache.normal().0.len(),
543            Model::Phi2(ref model) => model.cache.normal().0.len(),
544            Model::XLoraLlama(ref model) => model.cache.full().lock().len(),
545            Model::Phi3(ref model) => model.cache.normal().0.len(),
546            Model::XLoraPhi3(ref model) => model.cache.full().lock().len(),
547            Model::Starcoder2(ref model) => model.cache.normal().0.len(),
548            Model::Qwen(ref model) => model.cache.normal().0.len(),
549            Model::Qwen3(ref model) => model.cache.normal().0.len(),
550            Model::Qwen3MoE(ref model) => model.cache.normal().0.len(),
551        };
552
553        if chat_template.bos_token.is_none() {
554            if let Some(v) = bos {
555                chat_template.bos_token = Some(BeginEndUnkPadTok(Either::Left(v)));
556            }
557        }
558        if chat_template.eos_token.is_none() {
559            if let Some(v) = eos {
560                chat_template.eos_token = Some(BeginEndUnkPadTok(Either::Left(v)));
561            }
562        }
563        if chat_template.unk_token.is_none() {
564            if let Some(v) = unk {
565                chat_template.unk_token = Some(BeginEndUnkPadTok(Either::Left(v)));
566            }
567        }
568
569        let eos = calculate_eos_tokens(&chat_template, gen_conf, &tokenizer);
570        Ok(Arc::new(Mutex::new(GGUFPipeline {
571            model,
572            tokenizer: tokenizer.into(),
573            no_kv_cache: self.no_kv_cache,
574            chat_template: Arc::new(chat_template),
575            model_id: self
576                .model_id
577                .clone()
578                .unwrap_or(self.quantized_model_id.clone()),
579            non_granular_state: self.tgt_non_granular_index.map(|tgt_non_granular_index| {
580                NonGranularState {
581                    non_granular_index: Arc::new(Mutex::new(0)),
582                    tgt_non_granular_index,
583                }
584            }),
585            metadata: Arc::new(GeneralMetadata {
586                max_seq_len,
587                llg_factory: Some(llg_factory),
588                no_kv_cache: self.no_kv_cache,
589                no_prefix_cache: false,
590                num_hidden_layers,
591                eos_tok: eos,
592                kind: self.kind.clone(),
593                is_xlora,
594                activation_dtype: internal_dtype,
595                sliding_window: None,
596                cache_config,
597                cache_engine,
598                model_metadata: Some(Arc::new(model_config_metadata)),
599                modalities: Modalities {
600                    input: vec![SupportedModality::Text],
601                    output: vec![SupportedModality::Text],
602                },
603            }),
604            mapper: pipeline_mapper,
605        })))
606    }

722#[async_trait::async_trait]
723impl Pipeline for GGUFPipeline {
724    fn forward_inputs(
725        &mut self,
726        inputs: Box<dyn Any>,
727        return_raw_logits: bool,
728    ) -> Result<ForwardInputsResult, candle_core::Error> {
729        let ModelInputs {
730            input_ids,
731            input_ids_full,
732            seqlen_offsets,
733            seqlen_offsets_full,
734            context_lens,
735            position_ids: _, // NOTE(EricLBuehler): ignore, it is for phi3
736            paged_attn_meta,
737            flash_meta,
738            flash_meta_full,
739        } = *inputs.downcast().expect("Downcast failed.");
740        let metadata = self.get_metadata();
741        let paged_attn_meta = match (&metadata.cache_engine, &paged_attn_meta) {
742            (Some(engine), Some(meta)) => Some((engine.get_kv_cache().clone(), meta)),
743            (Some(_), None) => {
744                // This can happen if Rust-side user code is wrong
745                candle_core::bail!("Forward step expected a PagedAttention input metadata. This was not provided, please ensure that the scheduler config is correctly configured for PagedAttention.")
746            }
747            (None, Some(_)) => {
748                // This should never happen but we handle it anyway
749                candle_core::bail!("Forward step got a PagedAttention input metadata but there is no cache engine. Please raise an issue.")
750            }
751            (None, None) => None,
752        };
753        let logits = match self.model {
754            Model::Llama(ref model) => {
755                model.forward(&input_ids, &seqlen_offsets, context_lens, paged_attn_meta)?
756            }
757            Model::Phi2(ref model) => {
758                model.forward(&input_ids, &seqlen_offsets, context_lens, paged_attn_meta)?
759            }
760            Model::XLoraLlama(ref model) => model.forward(
761                &input_ids,
762                input_ids_full.as_ref().unwrap_or(&input_ids),
763                &seqlen_offsets,
764                seqlen_offsets_full.as_ref().unwrap_or(&seqlen_offsets),
765                self.no_kv_cache,
766                &self.non_granular_state,
767                context_lens,
768                &flash_meta,
769                flash_meta_full.as_ref().unwrap_or(&flash_meta),
770            )?,
771            Model::Phi3(ref model) => {
772                model.forward(&input_ids, &seqlen_offsets, paged_attn_meta)?
773            }
774            Model::XLoraPhi3(ref model) => model.forward(
775                &input_ids,
776                input_ids_full.as_ref().unwrap_or(&input_ids),
777                &seqlen_offsets,
778                seqlen_offsets_full.as_ref().unwrap_or(&seqlen_offsets),
779                self.no_kv_cache,
780                &self.non_granular_state,
781                context_lens,
782                &flash_meta,
783                flash_meta_full.as_ref().unwrap_or(&flash_meta),
784            )?,
785            Model::Starcoder2(ref model) => {
786                model.forward(&input_ids, &seqlen_offsets, paged_attn_meta)?
787            }
788            Model::Qwen(ref model) => {
789                model.forward(&input_ids, &seqlen_offsets, context_lens, paged_attn_meta)?
790            }
791            Model::Qwen3(ref model) => {
792                model.forward(&input_ids, &seqlen_offsets, context_lens, paged_attn_meta)?
793            }
794            Model::Qwen3MoE(ref model) => {
795                model.forward(&input_ids, &seqlen_offsets, context_lens, paged_attn_meta)?
796            }
797        };
798        if return_raw_logits {
799            Ok(ForwardInputsResult::RawLogits { logits })
800        } else {
801            Ok(ForwardInputsResult::CausalGeneration { logits })
802        }
803    }
804    async fn sample_causal_gen(
805        &self,
806        seqs: &mut [&mut Sequence],
807        logits: Vec<Tensor>,
808        prefix_cacher: &mut PrefixCacheManagerV2,
809        disable_eos_stop: bool,
810        rng: Arc<std::sync::Mutex<Isaac64Rng>>,
811    ) -> Result<(), candle_core::Error> {
812        sample_and_add_toks(self, seqs, logits, prefix_cacher, disable_eos_stop, rng).await
813    }
814    fn category(&self) -> ModelCategory {
815        ModelCategory::Text
816    }
817}
