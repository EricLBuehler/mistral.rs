16macro_rules! fixup_sentencepiece {
17    ($txt:expr) => {
18        $txt.to_string().replace("‚ñÅ", " ")
19    };
20    (Option $txt:expr) => {
21        match &$txt {
22            Some(txt) => Some(fixup_sentencepiece!(txt)),
23            None => None,
24        }
25    };
26}

28pub(crate) async fn finish_or_add_toks_to_seq(
29    this: &dyn Pipeline,
30    prefix_cacher: &mut PrefixCacheManagerV2,
31    seq: &mut Sequence,
32    logprobs: Logprobs,
33    eos_tok: Option<&[u32]>,
34    use_prefix_cacher: bool,
35) -> Result<()> {
36    let mut is_done = seq.is_done(logprobs.token, eos_tok, this.get_metadata().max_seq_len);
37    seq.add_token(
38        logprobs.clone(),
39        this.get_metadata()
40            .tok_env()
41            .ok_or(candle_core::Error::Msg(
42                "`finish_or_add_toks_to_seq` requires the pipeline to have a token trie"
43                    .to_string(),
44            ))?
45            .tok_trie()
46            .decode(&[logprobs.token]),
47        &is_done,
48    );
49
50    // If we can have a tool and we got a tool, stop the sequence early.
51    // Doesn't conflict with the logic below because it does the same thing anyway.
52    if let Some(ref t) = seq.tools {
53        if let Ok(Some(ref d)) = seq.peek_delta() {
54            let (_tool_use_still_possible, tool_use_is_done) =
55                t.prefix_could_be_tool(this, d.as_str())?;
56
57            if tool_use_is_done
58                && matches!(
59                    parse_text_tools(this, d, seq.tools.clone()),
60                    Ok((None, _tools))
61                )
62            {
63                seq.set_state(SequenceState::Done(StopReason::Eos));
64                is_done = Some(StopReason::Eos);
65            }
66        }
67    };
68
69    // Handle streaming requests
70    if seq.get_mut_group().is_streaming {
71        let mut tool_use_still_possible = false;
72        let mut tool_use_is_done = false;
73        if let Some(ref t) = seq.tools {
74            if let Ok(Some(ref d)) = seq.peek_delta() {
75                (tool_use_still_possible, tool_use_is_done) =
76                    t.prefix_could_be_tool(this, d.as_str())?;
77            }
78        };
79
80        // let send = seq.get_toks().len() % 2 == 0 || is_done.is_some();
81        let send = true;
82        // Send chunks when:
83        // 1. Tool call is not possible (!tool_use_still_possible) - normal streaming
84        // 2. Tool call is complete (tool_use_is_done) - send the tool call
85        // 3. Sequence is done (is_done.is_some()) - send buffered output as text since it wasn't a valid tool call
86        if !tool_use_still_possible || tool_use_is_done || is_done.is_some() {
87            if send {
88                let delta_result = seq.get_delta();
89                if let Some(delta) = crate::handle_seq_error_ok!(delta_result, seq.responder()) {
90                    if seq.get_mut_group().is_chat {
91                        // Check if we're in Harmony mode or think tag mode and use parsed content
92                        let (content_delta, reasoning_delta) = if seq.is_harmony_mode() {
93                            // In Harmony mode, use the parsed final content and reasoning
94                            let final_delta = seq.get_harmony_final_delta();
95                            let reasoning = seq.get_harmony_reasoning_delta();
96                            (final_delta, reasoning)
97                        } else if seq.is_think_tag_mode() {
98                            // In think tag mode, use the parsed content and reasoning
99                            let content = seq.get_think_tag_content_delta();
100                            let reasoning = seq.get_think_tag_reasoning_delta();
101                            (content, reasoning)
102                        } else {
103                            // Not in Harmony or think tag mode, use raw delta
104                            let (text_new, _) =
105                                parse_text_tools(this, delta.as_str(), seq.tools.clone())
106                                    .map_err(candle_core::Error::msg)?;
107                            (text_new.map(ToString::to_string), None)
108                        };
109
110                        // Detect tool calls
111                        let tool_calls = if seq.is_harmony_mode() {
112                            // In Harmony mode, only finalize tool calls when the sequence is done
113                            // (EOS token or stop string), not when we first detect a tool call.
114                            // This ensures tool call arguments are fully generated.
115                            if is_done.is_some() && seq.has_harmony_tool_calls() {
116                                // Sequence is done and has tool calls - finalize and send them
117                                is_done = Some(StopReason::ToolCalls);
118                                let harmony_tool_calls = seq.get_harmony_tool_calls();
119                                harmony_tool_calls
120                                    .into_iter()
121                                    .enumerate()
122                                    .map(|(i, tc)| ToolCallResponse {
123                                        index: i,
124                                        id: tc.id,
125                                        tp: ToolCallType::Function,
126                                        function: CalledFunction {
127                                            name: tc.name,
128                                            arguments: tc.arguments,
129                                        },
130                                    })
131                                    .collect()
132                            } else {
133                                vec![]
134                            }
135                        } else {
136                            // Not in Harmony mode - parse text for tool calls
137                            let (_, tool_calls) =
138                                parse_text_tools(this, delta.as_str(), seq.tools.clone())
139                                    .map_err(candle_core::Error::msg)?;
140                            if !tool_calls.is_empty() {
141                                is_done = Some(StopReason::ToolCalls);
142                            }
143                            tool_calls
144                        };
145
146                        seq.add_streaming_chunk_choice_to_group(crate::ChunkChoice {
147                            delta: crate::Delta {
148                                content: fixup_sentencepiece!(Option content_delta),
149                                role: "assistant".to_string(),
150                                tool_calls: Some(tool_calls).filter(|v| !v.is_empty()),
151                                reasoning_content: reasoning_delta,
152                            },
153                            index: seq.get_response_index(),
154                            finish_reason: is_done.map(|x| x.to_string()),
155                            logprobs: if seq.return_logprobs() {
156                                Some(crate::ResponseLogprob {
157                                    token: delta,
158                                    bytes: logprobs.bytes.clone().map(|b| b.into_bytes()),
159                                    logprob: logprobs.logprob,
160                                    top_logprobs: logprobs.top_logprobs.unwrap().clone(),
161                                })
162                            } else {
163                                None
164                            },
165                        });
166                    } else {
167                        seq.add_streaming_completion_chunk_choice_to_group(
168                            crate::CompletionChunkChoice {
169                                text: fixup_sentencepiece!(delta),
170                                index: seq.get_response_index(),
171                                finish_reason: is_done.map(|x| x.to_string()),
172                                logprobs: if seq.return_logprobs() {
173                                    Some(crate::ResponseLogprob {
174                                        token: delta,
175                                        bytes: logprobs.bytes.clone().map(|b| b.into_bytes()),
176                                        logprob: logprobs.logprob,
177                                        top_logprobs: logprobs.top_logprobs.unwrap().clone(),
178                                    })
179                                } else {
180                                    None
181                                },
182                            },
183                        );
184                    }
185                }
186            }
187
188            // Send usage on final chunk.
189            let usage_opt = if is_done.is_some() {
190                let usage = seq.get_mut_group().get_usage();
191                seq.get_mut_group().total_prompt_toks = 0;
192                seq.get_mut_group().total_toks = 0;
193                Some(usage)
194            } else {
195                None
196            };
197
198            if seq
199                .get_mut_group()
200                .maybe_send_streaming_response(seq, this.name().clone(), usage_opt)
201                .await
202                .is_err()
203            {
204                // If we can't send the response, cancel the sequence
205                seq.set_state(crate::sequence::SequenceState::Done(
206                    crate::sequence::StopReason::Canceled,
207                ));
208                this.reset_non_granular_state();
209            }
210        }
211
212        // Handle Done state regardless of tool detection - must be outside the tool_use check
213        // to ensure sequence completes even when tool detection thinks output might be a tool call
214        if let Some(reason) = is_done {
215            if use_prefix_cacher {
216                prefix_cacher.add_sequence(seq);
217                prefix_cacher.evict_caches()?;
218            }
219            seq.set_state(crate::sequence::SequenceState::Done(reason));
220            this.reset_non_granular_state();
221        }
222    } else if let Some(mut reason) = is_done {
223        /*
224        ***********************
225        Finish the sequence now
226        ***********************
227        */
228        {
229            seq.set_state(crate::sequence::SequenceState::Done(reason));
230            let (tokenizer, pipeline_name) = {
231                let pipeline_name = this.name();
232                let tokenizer = this.tokenizer();
233                (tokenizer, pipeline_name)
234            };
235
236            let logprobs = if seq.return_logprobs() {
237                let mut logprobs = Vec::new();
238                for logprob in seq.logprobs() {
239                    let resp_logprob = crate::ResponseLogprob {
240                        token: crate::handle_seq_error_ok!(
241                        tokenizer
242                        .as_ref()
243                        .ok_or(candle_core::Error::Msg(
244                            "`finish_or_add_toks_to_seq` requires the pipeline to have a tokenizer"
245                                .to_string(),
246                        ))?.decode(&[logprob.token], false),
247                        seq.responder()
248                    ),
249                        bytes: logprob.bytes.clone().map(|b| b.into_bytes()),
250                        logprob: logprob.logprob,
251                        top_logprobs: logprob.top_logprobs.clone().unwrap(),
252                    };
253                    logprobs.push(resp_logprob);
254                }
255                Some(logprobs)
256            } else {
257                None
258            };
259
260            // Signal EOS to Harmony parser if in Harmony mode
261            seq.harmony_process_eos();
262
263            // Finalize think tag parser if in think tag mode
264            seq.think_tag_finalize();
265
266            let text = match reason {
267                crate::sequence::StopReason::Length(_)
268                | crate::sequence::StopReason::ModelLength(_)
269                | crate::sequence::StopReason::Eos
270                | crate::sequence::StopReason::StopTok(_)
271                | crate::sequence::StopReason::Canceled
272                | crate::sequence::StopReason::ToolCalls => {
273                    String::from_utf8_lossy(seq.completion_bytes())
274                        .trim_start()
275                        .to_string()
276                }
277                crate::sequence::StopReason::StopString {
278                    completion_bytes_pos,
279                    ..
280                } => {
281                    let txt = String::from_utf8_lossy(seq.completion_bytes());
282                    txt[..completion_bytes_pos].trim_start().to_string()
283                }
284                crate::sequence::StopReason::GeneratedImage
285                | crate::sequence::StopReason::GeneratedSpeech => {
286                    candle_core::bail!("Stop reason was `GeneratedImage`.")
287                }
288            };
289
290            if seq.get_mut_group().is_chat {
291                // In Harmony or think tag mode, use parsed content and tool calls
292                let (text_new, tool_calls, reasoning_content) = if seq.is_harmony_mode() {
293                    let final_content = seq.get_harmony_final_content();
294                    let reasoning = seq.get_harmony_reasoning_content();
295
296                    // Get Harmony tool calls
297                    let harmony_tool_calls = seq.get_harmony_tool_calls();
298                    let tool_calls: Vec<ToolCallResponse> = harmony_tool_calls
299                        .into_iter()
300                        .enumerate()
301                        .map(|(i, tc)| ToolCallResponse {
302                            index: i,
303                            id: tc.id,
304                            tp: ToolCallType::Function,
305                            function: CalledFunction {
306                                name: tc.name,
307                                arguments: tc.arguments,
308                            },
309                        })
310                        .collect();
311
312                    (final_content, tool_calls, reasoning)
313                } else if seq.is_think_tag_mode() {
314                    // In think tag mode - finalize and get parsed content
315                    seq.think_tag_finalize();
316                    let final_content = seq.get_think_tag_content();
317                    let reasoning = seq.get_think_tag_reasoning_content();
318
319                    // Parse for tool calls in final content
320                    let (text_new, tool_calls) = if let Some(ref content) = final_content {
321                        parse_text_tools(this, content.as_str(), seq.tools.clone())
322                            .map_err(candle_core::Error::msg)?
323                    } else {
324                        (None, vec![])
325                    };
326                    (
327                        text_new.map(ToString::to_string).or(final_content),
328                        tool_calls,
329                        reasoning,
330                    )
331                } else {
332                    // Not in Harmony or think tag mode - parse text for tool calls
333                    let (text_new, tool_calls) =
334                        parse_text_tools(this, text.as_str(), seq.tools.clone())
335                            .map_err(candle_core::Error::msg)?;
336                    (text_new.map(ToString::to_string), tool_calls, None)
337                };
338
339                if !tool_calls.is_empty() {
340                    reason = StopReason::ToolCalls;
341                }
342
343                let choice = crate::Choice {
344                    finish_reason: fixup_sentencepiece!(reason),
345                    index: seq.get_response_index(),
346                    message: crate::ResponseMessage {
347                        content: text_new,
348                        role: "assistant".to_string(),
349                        tool_calls: Some(tool_calls).filter(|v| !v.is_empty()),
350                        reasoning_content,
351                    },
352                    logprobs: logprobs.map(|l| crate::Logprobs { content: Some(l) }),
353                };
354                seq.add_choice_to_group(choice);
355            } else {
356                let choice = crate::CompletionChoice {
357                    finish_reason: fixup_sentencepiece!(reason),
358                    index: seq.get_response_index(),
359                    text,
360                    logprobs: None,
361                };
362                seq.add_completion_choice_to_group(choice);
363            }
364
365            if use_prefix_cacher {
366                prefix_cacher.add_sequence(seq);
367                prefix_cacher.evict_caches()?;
368            }
369
370            let group = seq.get_mut_group();
371            if group.is_chat {
372                group
373                    .maybe_send_chat_done_response(
374                        crate::ChatCompletionResponse {
375                            id: seq.id().to_string(),
376                            choices: group.get_choices().to_vec(),
377                            created: seq.creation_time(),
378                            model: pipeline_name,
379                            system_fingerprint: crate::SYSTEM_FINGERPRINT.to_string(),
380                            object: "chat.completion".to_string(),
381                            usage: group.get_usage(),
382                        },
383                        seq.responder(),
384                    )
385                    .await
386                    .map_err(candle_core::Error::msg)?;
387            } else {
388                group
389                    .maybe_send_completion_done_response(
390                        crate::CompletionResponse {
391                            id: seq.id().to_string(),
392                            choices: group.get_completion_choices().to_vec(),
393                            created: seq.creation_time(),
394                            model: pipeline_name,
395                            system_fingerprint: crate::SYSTEM_FINGERPRINT.to_string(),
396                            object: "text_completion".to_string(),
397                            usage: group.get_usage(),
398                        },
399                        seq.responder(),
400                    )
401                    .await
402                    .map_err(candle_core::Error::msg)?;
403            }
404        }
405        this.reset_non_granular_state();
406    }
407
408    Ok(())
409}

411pub async fn sample_and_add_toks(
412    this: &dyn Pipeline,
413    seqs: &mut [&mut Sequence],
414    logits_seq: Vec<Tensor>,
415    prefix_cacher: &mut PrefixCacheManagerV2,
416    disable_eos_stop: bool,
417    rng: Arc<std::sync::Mutex<Isaac64Rng>>,
418) -> Result<()> {
419    let seqs_len = seqs.len();
420    debug_assert_eq!(logits_seq.len(), seqs_len);
421
422    let use_async_pool = seqs_len > 1;
423
424    let sampling_futures: Vec<_> = std::iter::zip(logits_seq, seqs.iter_mut())
425        .map(|(logits_per_seq, seq)| {
426            let return_logprobs = seq.return_logprobs();
427            sample_sequence(
428                logits_per_seq,
429                seq,
430                return_logprobs,
431                rng.clone(),
432                use_async_pool,
433                false,
434                use_async_pool,
435            )
436        })
437        .collect();
438    let sampled_vec = futures::future::join_all(sampling_futures).await;
439
440    for (sampled, seq) in std::iter::zip(sampled_vec, seqs.iter_mut()) {
441        let next_token = crate::handle_seq_error_stateaware_ok!(sampled, seq);
442
443        let metadata = this.get_metadata();
444        let eos_tok = if disable_eos_stop {
445            None
446        } else {
447            Some(&metadata.eos_tok[..])
448        };
449
450        finish_or_add_toks_to_seq(this, prefix_cacher, seq, next_token, eos_tok, true).await?;
451    }
452
453    Ok(())
454}

456/// Async sample optionally adding to trie.
457#[allow(clippy::too_many_arguments)]
458pub async fn sample_sequence(
459    logits: Tensor,
460    seq: &mut Sequence,
461    return_logprobs: bool,
462    rng: Arc<std::sync::Mutex<Isaac64Rng>>,
463    use_async_pool: bool,
464    sample_speculative: bool,
465    multiple_sequences: bool,
466) -> Result<Logprobs> {
467    let logits = logits.squeeze(0)?.squeeze(0)?.to_dtype(DType::F32)?;
468
469    let sampler = seq.sampler();
470    let ctx_clone = seq.get_toks().to_vec();
471    let rng_clone = rng.clone();
472    let logits_clone = logits.clone();
473    let first_lobprobs_response = if use_async_pool {
474        tokio_rayon::spawn(move || {
475            sampler.sample(
476                logits_clone,
477                &ctx_clone,
478                return_logprobs,
479                rng_clone,
480                sample_speculative,
481                multiple_sequences,
482            )
483        })
484        .await?
485    } else {
486        sampler.sample(
487            logits_clone,
488            &ctx_clone,
489            return_logprobs,
490            rng_clone,
491            sample_speculative,
492            multiple_sequences,
493        )?
494    };
495
496    let bias_if_not_allowed = match &mut seq.recognizer {
497        SequenceRecognizer::Llguidance(ref mut llg) => {
498            if !llg.is_stopped()
499                && llg
500                    .validate_tokens(&[first_lobprobs_response.token])
501                    .unwrap_or(0)
502                    == 1
503            {
504                None
505            } else {
506                let mask = llg.compute_mask_or_eos().map_err(candle_core::Error::msg)?;
507                if mask.is_allowed(first_lobprobs_response.token) {
508                    // shouldn't really happen, except for EOS
509                    None
510                } else {
511                    let mut acc = vec![-f32::INFINITY; logits.shape().dims1().unwrap()];
512                    mask.iter_set_entries(|idx| {
513                        if idx < acc.len() {
514                            acc[idx] = 0.0;
515                        }
516                    });
517
518                    Some(acc)
519                }
520            }
521        }
522        SequenceRecognizer::None => None,
523    };
524    let second_logprobs_response = match bias_if_not_allowed {
525        Some(acc) => {
526            let new_logits = (&logits + Tensor::from_slice(&acc, acc.len(), logits.device())?)?;
527
528            let ctx_clone = seq.get_toks().to_vec();
529            let rng_clone = rng.clone();
530            let sampler = seq.sampler();
531            if use_async_pool {
532                tokio_rayon::spawn(move || {
533                    sampler.sample(
534                        new_logits,
535                        &ctx_clone,
536                        return_logprobs,
537                        rng_clone,
538                        sample_speculative,
539                        multiple_sequences,
540                    )
541                })
542                .await?
543            } else {
544                sampler.sample(
545                    new_logits,
546                    &ctx_clone,
547                    return_logprobs,
548                    rng_clone,
549                    sample_speculative,
550                    multiple_sequences,
551                )?
552            }
553        }
554        None => first_lobprobs_response,
555    };
556
557    match seq.recognizer {
558        SequenceRecognizer::Llguidance(ref mut llg) => {
559            if !llg.is_stopped() {
560                llg.consume_token(second_logprobs_response.token)
561                    .map_err(candle_core::Error::msg)?;
562            }
563        }
564        SequenceRecognizer::None => {}
565    }
566
567    Ok(second_logprobs_response)
568}

570#[derive(Clone)]
571pub struct SpeculativeSample {
572    pub sample: Logprobs,
573}

575/// Async sample without modifying sequence (except for the constraint).
576pub async fn sample_target_sequence_speculative(
577    logits: Tensor,
578    seq: &mut Sequence,
579    return_logprobs: bool,
580    rng: Arc<std::sync::Mutex<Isaac64Rng>>,
581    draft_samples: &[SpeculativeSample],
582) -> Result<Vec<SpeculativeSample>> {
583    let n_toks = draft_samples.len();
584
585    // first, rollback the llg
586    match &mut seq.recognizer {
587        SequenceRecognizer::Llguidance(ref mut llg) => {
588            llg.rollback(n_toks).map_err(candle_core::Error::msg)?;
589        }
590        SequenceRecognizer::None => {}
591    }
592
593    let mut sampled = Vec::new();
594    for (chunk, draft) in logits
595        .chunk(n_toks, 1)?
596        .into_iter()
597        .zip(draft_samples.iter())
598    {
599        let sample = sample_sequence(
600            chunk,
601            seq,
602            return_logprobs,
603            rng.clone(),
604            true, // TODO(EricLBuehler): does this hurt perf?
605            true,
606            false,
607        )
608        .await?;
609        let sampled_token = sample.token;
610        sampled.push(SpeculativeSample { sample });
611        if sampled_token != draft.sample.token {
612            break;
613        }
614    }
615    Ok(sampled)
