69    fn _make_tensor_with_pad<D: WithDType>(
70        x: Vec<Vec<D>>,
71        max_len: usize,
72        pad: D,
73        device: &Device,
74    ) -> Result<Tensor> {
75        let mut padded_x = Vec::new();
76        for mut x_i in x {
77            assert!(x_i.len() <= max_len);
78            x_i.extend([pad].repeat(max_len - x_i.len()));
79            let shape = (x_i.len(),);
80            padded_x.push(Tensor::from_vec(x_i, shape, device)?);
81        }
82        Tensor::cat(&padded_x[..], 0).map_err(anyhow::Error::msg)
83    }

137    // chunk_offset_toks is the number of tokens by which the tokens are offset,
138    // chunk_offset_toks / prompt_chunksize = number of batches
139    #[allow(clippy::too_many_arguments)]
140    pub fn make_prompt_chunk<T: WithDType + Debug>(
141        chunk_offset_toks: usize,
142        toks: Vec<&[T]>,
143        seq_ids: &[usize],
144        device: &Device,
145        last_n_context_len: Option<(usize, usize)>,
146        return_raw_logits: bool,
147        mut paged_attn_metadata: Option<&mut PagedAttentionMeta>,
148        mapper: Option<&dyn DeviceMapper>,
149    ) -> Result<InputMetadata> {
150        let max_len = toks
151            .iter()
152            .map(|seq| seq.len())
153            .max()
154            .expect("No sequences");
155        let padding_tok = T::zero();
156        // Pad each sequence by the padding token to the max len.
157        let mut seqs_tensors = Vec::new();
158        let mut seqlen_offsets = Vec::new();
159        let mut context_lens = Vec::new();
160        let mut position_ids = Vec::new();
161        let mut slot_mappings = Vec::new();
162        let mut block_tables = Vec::new();
163        let mut paged_attn_context_lens = Vec::new();
164        let mut seqlens_q = vec![0];
165        let mut seqlens_k = vec![0];
166        for (seq_id, ctxt) in seq_ids.iter().zip(toks) {
167            let prompt_len = ctxt.len();
168            let offset = last_n_context_len.unwrap_or_default();
169            seqlen_offsets.push(offset.1 + chunk_offset_toks);
170
171            position_ids.push(ctxt.len() + chunk_offset_toks);
172            let mut ctxt = ctxt.to_vec();
173            ctxt.extend(std::iter::repeat_n(
174                padding_tok,
175                max_len.saturating_sub(ctxt.len()),
176            ));
177            // If we are returning raw logits, we want to not trim the logits at all.
178            if return_raw_logits {
179                if last_n_context_len.is_some() {
180                    anyhow::bail!("`return_raw_logits` is incompatible with `last_n_context_len`");
181                }
182
183                context_lens.push((0, ctxt.len()));
184            } else {
185                context_lens.push((
186                    ctxt.len()
187                        .saturating_sub(last_n_context_len.map(|(a, _)| a).unwrap_or(1)),
188                    last_n_context_len.map(|(a, _)| a).unwrap_or(1),
189                ));
190            }
191
192            seqlens_q.push(ctxt.len() as u32);
193            seqlens_k.push((ctxt.len() + chunk_offset_toks) as u32);
194
195            seqs_tensors.push(Tensor::new(ctxt, device).unwrap().unsqueeze(0).unwrap());
196
197            if let Some(paged_attn_metadata) = &mut paged_attn_metadata {
198                let block_engine = get_mut_arcmutex!(paged_attn_metadata.block_engine);
199                let table = block_engine.block_tables.get(seq_id);
200
201                if table.is_none() {
202                    // Will be None during profiling.
203                    slot_mappings.push([_PAD_SLOT_ID].repeat(prompt_len));
204                    continue;
205                }
206                let table = table
207                    .unwrap()
208                    .iter()
209                    .map(|block| block.deref_mut().block_id)
210                    .collect::<Vec<_>>();
211
212                let start_idx = if let Some(sliding_window) = paged_attn_metadata.sliding_window {
213                    prompt_len.saturating_sub(sliding_window)
214                } else {
215                    0
216                };
217
218                let mut slot_mapping = Vec::new();
219                let mut ctxt_len = Vec::new();
220                for i in chunk_offset_toks..prompt_len + chunk_offset_toks {
221                    if i < start_idx {
222                        // Pad [0,start_idx) with _PAD_TOKEN_ID
223                        slot_mapping.push(_PAD_SLOT_ID);
224                    }
225                    ctxt_len.push(i);
226
227                    let block_number = if i / paged_attn_metadata.block_size >= table.len() {
228                        panic!(
229                            "Block table is too small (prompt)! i={} block_size={} table_len={}",
230                            i,
231                            paged_attn_metadata.block_size,
232                            table.len()
233                        );
234                    } else {
235                        table.get(i / paged_attn_metadata.block_size).unwrap()
236                    };
237                    let block_offset = i % paged_attn_metadata.block_size;
238                    let slot = block_number * paged_attn_metadata.block_size + block_offset;
239                    slot_mapping.push(slot.try_into().unwrap());
240                    block_tables.push(table.clone());
241                }
242                slot_mappings.push(slot_mapping);
243                paged_attn_context_lens.push(ctxt_len);
244            }
245        }
246
247        let max_q = *seqlens_q.iter().max().unwrap();
248        let max_k = *seqlens_k.iter().max().unwrap();
249        let seqlens_q = Tensor::new(seqlens_q, device)?
250            .to_dtype(DType::F32)?
251            .cumsum(0)?
252            .to_dtype(DType::U32)?;
253        let seqlens_k = Tensor::new(seqlens_k, device)?
254            .to_dtype(DType::F32)?
255            .cumsum(0)?
256            .to_dtype(DType::U32)?;
257
258        let mut seqlens_q_map = HashMap::new();
259        let mut seqlens_k_map = HashMap::new();
260
261        let devices = mapper.unwrap().get_unique_devices();
262        for device in devices {
263            seqlens_q_map.insert(device.location(), seqlens_q.to_device(&device)?);
264            seqlens_k_map.insert(device.location(), seqlens_k.to_device(&device)?);
265        }
266
267        let input = Tensor::cat(&seqs_tensors, 0).unwrap();
268
269        let paged_attn_meta = if paged_attn_metadata.is_some() {
270            let max_slot_mapping_len = slot_mappings.iter().map(|x| x.len()).max().unwrap();
271            let slot_mappings =
272                _make_tensor_with_pad(slot_mappings, max_slot_mapping_len, _PAD_SLOT_ID, device)?;
273
274            let max_block_table_len = block_tables.iter().map(|x| x.len()).max().unwrap();
275            let block_tables = _make_tensor_with_pad(
276                block_tables
277                    .iter()
278                    .map(|x| x.iter().map(|x| *x as u32).collect::<Vec<_>>())
279                    .collect::<Vec<_>>(),
280                max_block_table_len,
281                0,
282                device,
283            )?;
284            let block_tables = block_tables.reshape(((), max_block_table_len))?;
285
286            let max_context_len = paged_attn_context_lens
287                .iter()
288                .map(|x| x.len())
289                .max()
290                .unwrap();
291
292            let context_lens = _make_tensor_with_pad(
293                paged_attn_context_lens
294                    .iter()
295                    .map(|x| x.iter().map(|x| *x as u32).collect::<Vec<_>>())
296                    .collect::<Vec<_>>(),
297                max_context_len,
298                0,
299                device,
300            )?
301            .reshape(((),))?;
302
303            // For device mapping, make a copy of each tensor for each device
304            let devices = mapper.unwrap().get_unique_devices();
305            let mut slot_mappings_map = HashMap::new();
306            let mut block_tables_map = HashMap::new();
307            let mut context_lens_map = HashMap::new();
308
309            for device in devices {
310                slot_mappings_map
311                    .insert(device.location(), slot_mappings.clone().to_device(&device)?);
312                block_tables_map
313                    .insert(device.location(), block_tables.clone().to_device(&device)?);
314                context_lens_map
315                    .insert(device.location(), context_lens.clone().to_device(&device)?);
316            }
317
318            Some(PagedAttentionInputMetadata {
319                slot_mappings: slot_mappings_map,
320                block_tables: Some(block_tables_map),
321                context_lens: Some(context_lens_map),
322                max_context_len: Some(max_context_len),
323                is_first_prompt_chunk: chunk_offset_toks == 0,
324            })
325        } else {
326            None
327        };
328
329        Ok(InputMetadata {
330            input,
331            positions: seqlen_offsets,
332            context_lens,
333            position_ids,
334            paged_attn_meta,
335            flash_meta: FlashParams {
336                max_k,
337                max_q,
338                cumulative_seqlens_k: seqlens_k_map,
339                cumulative_seqlens_q: seqlens_q_map,
340            },
341        })
342    }

344    fn make_completion_chunk<T: WithDType>(
345        toks: Vec<&[T]>,
346        input_seqs: &[&mut Sequence],
347        device: &Device,
348        mut paged_attn_metadata: Option<&mut PagedAttentionMeta>,
349        mapper: Option<&dyn DeviceMapper>,
350    ) -> Result<InputMetadata> {
351        // Pad each sequence by the padding token to the max len.
352        let mut seqs_tensors = Vec::new();
353        let mut seqlen_offsets = Vec::new();
354        let mut context_lens = Vec::new();
355        let mut position_ids = Vec::new();
356
357        let mut slot_mappings = Vec::new();
358        let mut block_tables = Vec::new();
359        let mut paged_attn_context_lens = Vec::new();
360        let mut seqlens_q = vec![0];
361        let mut seqlens_k = vec![0];
362        for (seq, ctxt) in input_seqs.iter().zip(toks) {
363            let start_pos = ctxt.len().saturating_sub(1);
364            let ctxt = ctxt[start_pos..].to_vec();
365            seqlen_offsets.push(start_pos);
366            context_lens.push((0, 1));
367            position_ids.push(seq.len());
368
369            seqlens_q.push(ctxt.len() as u32);
370            seqlens_k.push((ctxt.len() + start_pos) as u32);
371
372            seqs_tensors.push(Tensor::new(ctxt, device).unwrap().unsqueeze(0).unwrap());
373
374            if let Some(paged_attn_metadata) = &mut paged_attn_metadata {
375                let block_engine = get_mut_arcmutex!(paged_attn_metadata.block_engine);
376                let table = block_engine.block_tables.get(seq.id()).unwrap();
377
378                let table = table
379                    .iter()
380                    .map(|block| block.deref_mut().block_id)
381                    .collect::<Vec<_>>();
382
383                let block_pos = start_pos - seq.token_offset();
384                let block_number = if block_pos / paged_attn_metadata.block_size >= table.len() {
385                    panic!("Block table is too small (completion)! start_pos={} block_size={} table_len={}", block_pos, paged_attn_metadata.block_size, table.len());
386                } else {
387                    table
388                        .get(block_pos / paged_attn_metadata.block_size)
389                        .unwrap()
390                };
391                let block_offset = block_pos % paged_attn_metadata.block_size;
392                let slot = block_number * paged_attn_metadata.block_size + block_offset;
393                let slot = slot.try_into().unwrap();
394                slot_mappings.push(vec![slot]);
395
396                if let Some(sliding_window) = paged_attn_metadata.sliding_window {
397                    let sliding_window_blocks = sliding_window / paged_attn_metadata.block_size;
398                    let slide_idx = if table.len() > sliding_window_blocks {
399                        table.len() - sliding_window_blocks
400                    } else {
401                        0
402                    };
403                    block_tables.push(table.get(slide_idx..).unwrap().to_vec());
404                } else {
405                    block_tables.push(table);
406                }
407
408                let paged_attn_context_len =
409                    if let Some(sliding_window) = paged_attn_metadata.sliding_window {
410                        seq.len().min(sliding_window)
411                    } else {
412                        seq.len()
413                    };
414                paged_attn_context_lens.push(paged_attn_context_len);
415            }
416        }
417
418        let max_q = *seqlens_q.iter().max().unwrap();
419        let max_k = *seqlens_k.iter().max().unwrap();
420        let seqlens_q = Tensor::new(seqlens_q, device)?
421            .to_dtype(DType::F32)?
422            .cumsum(0)?
423            .to_dtype(DType::U32)?;
424        let seqlens_k = Tensor::new(seqlens_k, device)?
425            .to_dtype(DType::F32)?
426            .cumsum(0)?
427            .to_dtype(DType::U32)?;
428
429        let mut seqlens_q_map = HashMap::new();
430        let mut seqlens_k_map = HashMap::new();
431
432        let devices = mapper.unwrap().get_unique_devices();
433        for device in devices {
434            seqlens_q_map.insert(device.location(), seqlens_q.to_device(&device)?);
435            seqlens_k_map.insert(device.location(), seqlens_k.to_device(&device)?);
436        }
437
438        let paged_attn_meta = if paged_attn_metadata.is_some() {
439            let slot_mappings = _make_tensor_with_pad(slot_mappings, 1, _PAD_SLOT_ID, device)?;
440
441            let max_block_table_len = block_tables.iter().map(|x| x.len()).max().unwrap();
442
443            let block_tables = _make_tensor_with_pad(
444                block_tables
445                    .iter()
446                    .map(|x| x.iter().map(|x| *x as u32).collect::<Vec<_>>())
447                    .collect::<Vec<_>>(),
448                max_block_table_len,
449                0,
450                device,
451            )?;
452            let block_tables = block_tables.reshape(((), max_block_table_len))?;
453
454            let max_context_len = paged_attn_context_lens.iter().max().unwrap();
455
456            let context_lens = Tensor::from_vec(
457                paged_attn_context_lens
458                    .iter()
459                    .map(|x| *x as u32)
460                    .collect::<Vec<_>>(),
461                (paged_attn_context_lens.len(),),
462                device,
463            )?;
464
465            // For device mapping, make a copy of each tensor for each device
466            let devices = mapper.unwrap().get_unique_devices();
467            let mut slot_mappings_map = HashMap::new();
468            let mut block_tables_map = HashMap::new();
469            let mut context_lens_map = HashMap::new();
470
471            for device in devices {
472                slot_mappings_map
473                    .insert(device.location(), slot_mappings.clone().to_device(&device)?);
474                block_tables_map
475                    .insert(device.location(), block_tables.clone().to_device(&device)?);
476                context_lens_map
477                    .insert(device.location(), context_lens.clone().to_device(&device)?);
478            }
479
480            Some(PagedAttentionInputMetadata {
481                slot_mappings: slot_mappings_map,
482                block_tables: Some(block_tables_map),
483                context_lens: Some(context_lens_map),
484                max_context_len: Some(*max_context_len),
485                is_first_prompt_chunk: false,
486            })
487        } else {
488            None
489        };
490
491        Ok(InputMetadata {
492            input: Tensor::cat(&seqs_tensors, 0).unwrap(),
493            positions: seqlen_offsets,
494            context_lens,
495            position_ids,
496            paged_attn_meta,
497            flash_meta: FlashParams {
498                max_k,
499                max_q,
500                cumulative_seqlens_k: seqlens_k_map,
501                cumulative_seqlens_q: seqlens_q_map,
502            },
503        })
504    }

506    #[allow(clippy::too_many_arguments)]
507    pub(crate) fn get_prompt_input<T: WithDType + std::fmt::Debug>(
508        toks: Vec<&[T]>,
509        input_seqs: &[&mut Sequence],
510        device: &Device,
511        last_n_context_len: Option<(usize, usize)>,
512        return_raw_logits: bool,
513        paged_attn_metadata: Option<&mut PagedAttentionMeta>,
514        prompt_chunksize: Option<NonZeroUsize>,
515        mapper: Option<&dyn DeviceMapper>,
516    ) -> Box<dyn Iterator<Item = Result<InnerInputProcessorOutput>>> {
517        if let (Some(prompt_chunksize), true) = (prompt_chunksize, paged_attn_metadata.is_none()) {
518            let chunk_size = prompt_chunksize.get();
519            let offset = input_seqs[0].token_offset();
520            // Determine the maximum number of chunks across all sequences
521            let num_chunks = toks
522                .iter()
523                .map(|ctxt| ctxt.len().div_ceil(chunk_size))
524                .max()
525                .unwrap_or(0);
526
527            let mut outputs = Vec::with_capacity(num_chunks);
528            for chunk_idx in 0..num_chunks {
529                let mut slices = Vec::new();
530                let mut seq_ids = Vec::new();
531                let mut seq_indices = Vec::new();
532                for (seq_n, ctxt) in toks.iter().enumerate() {
533                    let start = chunk_idx * chunk_size;
534                    if start < ctxt.len() {
535                        let end = (start + chunk_size).min(ctxt.len());
536                        slices.push(&ctxt[start..end]);
537                        seq_indices.push(seq_n);
538                        seq_ids.push(*input_seqs[seq_n].id());
539                    }
540                }
541                let result = make_prompt_chunk(
542                    chunk_idx * chunk_size + offset,
543                    slices,
544                    &seq_ids,
545                    device,
546                    last_n_context_len,
547                    return_raw_logits,
548                    None,
549                    mapper,
550                )
551                .map(|inputs| InnerInputProcessorOutput {
552                    inputs,
553                    seq_indices,
554                });
555                outputs.push(result);
556            }
557            Box::new(outputs.into_iter())
558        } else {
559            let offset = input_seqs[0].token_offset();
560            Box::new(std::iter::once(
561                make_prompt_chunk(
562                    offset,
563                    toks,
564                    &input_seqs.iter().map(|s| *s.id()).collect::<Vec<_>>(),
565                    device,
566                    last_n_context_len,
567                    return_raw_logits,
568                    paged_attn_metadata,
569                    mapper,
570                )
571                .map(|inputs| InnerInputProcessorOutput {
572                    inputs,
573                    seq_indices: (0..input_seqs.len()).collect(),
574                }),
575            ))
576        }
577    }

579    #[allow(clippy::too_many_arguments)]
580    pub(crate) fn get_completion_input<T: WithDType + std::fmt::Debug>(
581        toks: Vec<&[T]>,
582        input_seqs: &[&mut Sequence],
583        device: &Device,
584        no_kv_cache: bool,
585        last_n_context_len: Option<(usize, usize)>,
586        return_raw_logits: bool,
587        paged_attn_metadata: Option<&mut PagedAttentionMeta>,
588        prompt_chunksize: Option<NonZeroUsize>,
589        mapper: Option<&dyn DeviceMapper>,
590    ) -> Box<dyn Iterator<Item = Result<InnerInputProcessorOutput>>> {
591        if no_kv_cache {
592            return get_prompt_input(
593                toks,
594                input_seqs,
595                device,
596                last_n_context_len,
597                return_raw_logits,
598                paged_attn_metadata,
599                prompt_chunksize,
600                mapper,
601            );
602        }
603
604        Box::new(std::iter::once(
605            make_completion_chunk(toks, input_seqs, device, paged_attn_metadata, mapper).map(
606                |inputs| InnerInputProcessorOutput {
607                    inputs,
608                    seq_indices: (0..input_seqs.len()).collect(),
609                },
610            ),
611        ))
612    }

629    impl InputsProcessor for TextInputsProcessor {
630        fn process_inputs(
631            &self,
632            _: Option<Arc<Tokenizer>>,
633            input_seqs: &mut [&mut Sequence],
634            is_prompt: bool,
635            is_xlora: bool,
636            device: &Device,
637            no_kv_cache: bool,
638            last_n_context_len: Option<(usize, usize)>,
639            return_raw_logits: bool,
640            _: Option<Arc<dyn Any>>,
641            mut paged_attn_metadata: Option<PagedAttentionMeta>,
642            prompt_chunksize: Option<NonZeroUsize>,
643            mapper: Option<&dyn DeviceMapper>,
644        ) -> Box<dyn Iterator<Item = Result<InputProcessorOutput>>> {
645            if is_xlora && !is_prompt {
646                Box::new(
647                    get_prompt_input(
648                        input_seqs
649                            .iter()
650                            .map(|seq| seq.get_toks())
651                            .collect::<Vec<_>>(),
652                        input_seqs,
653                        device,
654                        last_n_context_len,
655                        return_raw_logits,
656                        paged_attn_metadata.as_mut(),
657                        prompt_chunksize,
658                        mapper,
659                    )
660                    .zip(get_completion_input(
661                        input_seqs
662                            .iter()
663                            .map(|seq| seq.get_toks())
664                            .collect::<Vec<_>>(),
665                        input_seqs,
666                        device,
667                        no_kv_cache,
668                        last_n_context_len,
669                        return_raw_logits,
670                        paged_attn_metadata.as_mut(),
671                        prompt_chunksize,
672                        mapper,
673                    ))
674                    .map(|(prompt, completion)| {
675                        let InnerInputProcessorOutput {
676                            inputs:
677                                InputMetadata {
678                                    input: input_ids_full,
679                                    positions: seqlen_offsets_full,
680                                    context_lens: _,
681                                    position_ids,
682                                    paged_attn_meta: _,
683                                    flash_meta: flash_meta_full,
684                                },
685                            seq_indices,
686                        } = prompt?;
687                        let InnerInputProcessorOutput {
688                            inputs:
689                                InputMetadata {
690                                    input: input_ids,
691                                    positions: seqlen_offsets,
692                                    context_lens,
693                                    position_ids: _,
694                                    paged_attn_meta,
695                                    flash_meta,
696                                },
697                            seq_indices: _,
698                        } = completion?;
699                        let inputs: Box<dyn Any> = Box::new(ModelInputs {
700                            input_ids,
701                            input_ids_full: Some(input_ids_full),
702                            seqlen_offsets,
703                            seqlen_offsets_full: Some(seqlen_offsets_full),
704                            context_lens,
705                            position_ids,
706                            paged_attn_meta,
707                            flash_meta,
708                            flash_meta_full: Some(flash_meta_full),
709                        });
710                        Ok(InputProcessorOutput {
711                            inputs,
712                            seq_indices,
713                        })
714                    }),
715                )
716            } else if is_xlora && is_prompt {
717                Box::new(
718                    get_prompt_input(
719                        input_seqs
720                            .iter()
721                            .map(|seq| seq.get_toks())
722                            .collect::<Vec<_>>(),
723                        input_seqs,
724                        device,
725                        last_n_context_len,
726                        return_raw_logits,
727                        paged_attn_metadata.as_mut(),
728                        prompt_chunksize,
729                        mapper,
730                    )
731                    .map(|metadata| {
732                        let InnerInputProcessorOutput {
733                            inputs:
734                                InputMetadata {
735                                    input: input_ids,
736                                    positions: seqlen_offsets,
737                                    context_lens,
738                                    position_ids,
739                                    paged_attn_meta,
740                                    flash_meta,
741                                },
742                            seq_indices,
743                        } = metadata?;
744                        let inputs: Box<dyn Any> = Box::new(ModelInputs {
745                            input_ids: input_ids.clone(),
746                            input_ids_full: Some(input_ids),
747                            seqlen_offsets: seqlen_offsets.clone(),
748                            seqlen_offsets_full: Some(seqlen_offsets),
749                            context_lens,
750                            position_ids,
751                            paged_attn_meta,
752                            flash_meta: flash_meta.clone(),
753                            flash_meta_full: Some(flash_meta),
754                        });
755                        Ok(InputProcessorOutput {
756                            inputs,
757                            seq_indices,
758                        })
759                    }),
760                )
761            } else if is_prompt {
762                Box::new(
763                    get_prompt_input(
764                        input_seqs
765                            .iter()
766                            .map(|seq| seq.get_toks())
767                            .collect::<Vec<_>>(),
768                        input_seqs,
769                        device,
770                        last_n_context_len,
771                        return_raw_logits,
772                        paged_attn_metadata.as_mut(),
773                        prompt_chunksize,
774                        mapper,
775                    )
776                    .map(|metadata| {
777                        let InnerInputProcessorOutput {
778                            inputs:
779                                InputMetadata {
780                                    input: input_ids,
781                                    positions: seqlen_offsets,
782                                    context_lens,
783                                    position_ids,
784                                    paged_attn_meta,
785                                    flash_meta,
786                                },
787                            seq_indices,
788                        } = metadata?;
789                        let inputs: Box<dyn Any> = Box::new(ModelInputs {
790                            input_ids,
791                            input_ids_full: None,
792                            seqlen_offsets,
793                            seqlen_offsets_full: None,
794                            context_lens,
795                            position_ids,
796                            paged_attn_meta,
797                            flash_meta,
798                            flash_meta_full: None,
799                        });
800                        Ok(InputProcessorOutput {
801                            inputs,
802                            seq_indices,
803                        })
804                    }),
805                )
806            } else {
807                Box::new(
808                    get_completion_input(
809                        input_seqs
810                            .iter()
811                            .map(|seq| seq.get_toks())
812                            .collect::<Vec<_>>(),
813                        input_seqs,
814                        device,
815                        no_kv_cache,
816                        last_n_context_len,
817                        return_raw_logits,
818                        paged_attn_metadata.as_mut(),
819                        prompt_chunksize,
820                        mapper,
821                    )
822                    .map(|metadata| {
823                        let InnerInputProcessorOutput {
824                            inputs:
825                                InputMetadata {
826                                    input: input_ids,
827                                    positions: seqlen_offsets,
828                                    context_lens,
829                                    position_ids,
830                                    paged_attn_meta,
831                                    flash_meta,
832                                },
833                            seq_indices,
834                        } = metadata?;
835                        let inputs: Box<dyn Any> = Box::new(ModelInputs {
836                            input_ids,
837                            input_ids_full: None,
838                            seqlen_offsets,
839                            seqlen_offsets_full: None,
840                            context_lens,
841                            position_ids,
842                            paged_attn_meta,
843                            flash_meta,
844                            flash_meta_full: None,
845                        });
846                        Ok(InputProcessorOutput {
847                            inputs,
848                            seq_indices,
849                        })
850                    }),
851                )
852            }
853        }
