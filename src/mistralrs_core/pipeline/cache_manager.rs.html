29pub enum EitherCache {
30    Normal(Arc<Mutex<NormalCache>>),
31    Full(Cache),
32}
33
34impl EitherCache {
35    /// Panics otherwise!
36    pub fn full(&self) -> &Cache {
37        match self {
38            Self::Full(full) => full,
39            Self::Normal(_) => panic!("Got normal cache, expected full cache."),
40        }
41    }
42    /// Panics otherwise!
43    pub fn normal(&self) -> MutexGuard<'_, NormalCache> {
44        match self {
45            Self::Normal(normal) => normal.lock().unwrap(),
46            Self::Full(_) => panic!("Got full cache, expected normal cache."),
47        }
48    }
49}

51#[derive(Debug, Clone)]
52pub struct SingleCache {
53    // all_data is an option on a Tensor, this makes it possible to only create the actual tensor
54    // on the first call where the batch size is easily known.
55    // Also this makes it safe to clone a KvCache that has been reset (as in it will not share
56    // its internal state with the cloned instance).
57    pub all_data: Option<Tensor>,
58    pub dim: usize,
59    pub current_seq_len: usize,
60    pub capacity_seq_len: usize,
61    pub max_seq_len: usize,
62}
63
64impl SingleCache {
65    pub fn new(dim: usize, max_seq_len: usize, capacity_seq_len: usize) -> Self {
66        Self {
67            all_data: None,
68            dim,
69            current_seq_len: 0,
70            max_seq_len,
71            capacity_seq_len,
72        }
73    }
74
75    pub fn dim(&self) -> usize {
76        self.dim
77    }
78
79    pub fn current_seq_len(&self) -> usize {
80        self.current_seq_len
81    }
82
83    pub fn max_seq_len(&self) -> usize {
84        self.max_seq_len
85    }
86
87    pub fn all_data(&self) -> &Option<Tensor> {
88        &self.all_data
89    }
90
91    pub fn current_data(&self) -> Result<Option<Tensor>> {
92        let data = match self.all_data.as_ref() {
93            None => None,
94            Some(d) => Some(d.narrow(self.dim, 0, self.current_seq_len)?),
95        };
96        Ok(data)
97    }
98
99    pub fn reset(&mut self) {
100        self.current_seq_len = 0;
101        self.all_data = None;
102    }
103
104    pub fn set_len(&mut self, len: usize) -> candle_core::Result<()> {
105        self.current_seq_len = len;
106        Ok(())
107    }
108
109    pub fn append(&mut self, src: &Tensor) -> Result<()> {
110        let seq_len = src.dim(self.dim)?;
111        // This doesn't seem very idiomatic but because the creation can fail, it's tricky to use
112        // self.all_data.get_or_insert_with.
113        if self.all_data.is_none() {
114            let mut shape = src.dims().to_vec();
115            shape[self.dim] = self.capacity_seq_len;
116            let ad = Tensor::zeros(shape, src.dtype(), src.device())?;
117            self.all_data = Some(ad);
118        };
119
120        // Expand kv cache
121        if self.current_seq_len + seq_len > self.capacity_seq_len {
122            let diff = self.current_seq_len + seq_len - self.capacity_seq_len;
123            let n_blocks_needed = diff.div_ceil(NormalCache::CACHE_GROW_SIZE);
124            self.capacity_seq_len += n_blocks_needed * NormalCache::CACHE_GROW_SIZE;
125            if self.capacity_seq_len > self.max_seq_len {
126                candle_core::bail!(
127                    "kv-cache: requested capacity ({}) above max seq len ({})",
128                    self.capacity_seq_len,
129                    self.max_seq_len
130                )
131            }
132            let mut shape = src.dims().to_vec();
133            shape[self.dim] = self.capacity_seq_len;
134            let ad = Tensor::zeros(shape, src.dtype(), src.device())?;
135            ad.slice_set(self.all_data.as_ref().unwrap(), self.dim, 0)?;
136            self.all_data = Some(ad);
137        }
138
139        let ad = self.all_data.as_mut().unwrap();
140
141        ad.slice_set(src, self.dim, self.current_seq_len)?;
142        self.current_seq_len += seq_len;
143        Ok(())
144    }
145}

147#[derive(Debug, Clone)]
148pub struct RotatingCache {
149    pub all_data: Option<Tensor>,
150    pub dim: usize,
151    // `offset` is the current write index in the buffer
152    pub offset: usize,
153    // The total size of the sequence seen so far.
154    pub current_seq_len: usize,
155    // max_seq_len is the size of the rotating buffer, it is actually allowed for the full
156    // sequence to grow past this limit.
157    pub max_seq_len: usize,
158    pub capacity_seq_len: usize,
159}
160
161impl RotatingCache {
162    pub fn new(dim: usize, max_seq_len: usize, capacity_seq_len: usize) -> Self {
163        Self {
164            all_data: None,
165            dim,
166            offset: 0,
167            current_seq_len: 0,
168            max_seq_len,
169            capacity_seq_len,
170        }
171    }
172
173    pub fn offset(&self) -> usize {
174        self.offset
175    }
176
177    pub fn dim(&self) -> usize {
178        self.dim
179    }
180
181    pub fn current_seq_len(&self) -> usize {
182        self.current_seq_len
183    }
184
185    pub fn max_seq_len(&self) -> usize {
186        self.max_seq_len
187    }
188
189    pub fn all_data(&self) -> &Option<Tensor> {
190        &self.all_data
191    }
192
193    pub fn current_data(&self) -> Result<Option<Tensor>> {
194        let data = match self.all_data.as_ref() {
195            None => None,
196            Some(d) => {
197                if self.current_seq_len >= self.max_seq_len {
198                    Some(d.clone())
199                } else {
200                    Some(d.narrow(self.dim, 0, self.current_seq_len)?)
201                }
202            }
203        };
204        Ok(data)
205    }
206
207    pub fn reset(&mut self) {
208        self.offset = 0;
209        self.current_seq_len = 0;
210        self.all_data = None;
211    }
212
213    pub fn set_len(&mut self, len: usize) -> candle_core::Result<()> {
214        // If trying to roll it back past the boundary of max_seq_len, fail early.
215        if self.current_seq_len - len > self.max_seq_len {
216            candle_core::bail!(
217                "Rotating KV cache (usually for sliding window) tried to reset to len {len} while current is {} and max retained is {}",
218                self.current_seq_len,
219                self.max_seq_len
220            );
221        }
222        self.current_seq_len = len;
223        self.offset = len % self.max_seq_len;
224        Ok(())
225    }
226
227    pub fn append(&mut self, src: &Tensor) -> Result<Tensor> {
228        let seq_len = src.dim(self.dim)?;
229        // This doesn't seem very idiomatic but because the creation can fail, it's tricky to use
230        // self.all_data.get_or_insert_with.
231        if self.all_data.is_none() {
232            let mut shape = src.dims().to_vec();
233            shape[self.dim] = self.capacity_seq_len;
234            let ad = Tensor::zeros(shape, src.dtype(), src.device())?;
235            self.all_data = Some(ad)
236        };
237
238        // Expand kv cache, this case is a little more complex.
239        if (self.current_seq_len + seq_len > self.capacity_seq_len
240            && self.current_seq_len + seq_len < self.max_seq_len)
241            || self.current_seq_len == 0
242        {
243            let diff = self.current_seq_len + seq_len - self.capacity_seq_len;
244            let n_blocks_needed = diff.div_ceil(NormalCache::CACHE_GROW_SIZE);
245            self.capacity_seq_len += n_blocks_needed * NormalCache::CACHE_GROW_SIZE;
246            self.capacity_seq_len = self.capacity_seq_len.min(self.max_seq_len);
247            if self.capacity_seq_len > self.max_seq_len {
248                candle_core::bail!(
249                    "kv-cache: requested capacity ({}) above max seq len ({})",
250                    self.capacity_seq_len,
251                    self.max_seq_len
252                )
253            }
254            let mut shape = src.dims().to_vec();
255            shape[self.dim] = self.capacity_seq_len;
256            let ad = Tensor::zeros(shape, src.dtype(), src.device())?;
257            ad.slice_set(self.all_data.as_ref().unwrap(), self.dim, 0)?;
258            self.all_data = Some(ad);
259        }
260
261        let ad = self.all_data.as_mut().unwrap();
262
263        self.current_seq_len += seq_len;
264        if seq_len >= self.max_seq_len {
265            let to_copy = src
266                .narrow(self.dim, seq_len - self.max_seq_len, self.max_seq_len)?
267                .contiguous()?;
268            ad.slice_set(&to_copy, self.dim, 0)?;
269            self.offset = 0;
270            // Here we return `src` rather than `ad` so that all the past can be used.
271            Ok(src.clone())
272        } else {
273            let rem_len = self.max_seq_len - self.offset;
274            if seq_len <= rem_len {
275                ad.slice_set(&src.contiguous()?, self.dim, self.offset)?;
276                self.offset = (self.offset + seq_len) % self.max_seq_len;
277            } else {
278                // We have to make two copies here as we go over the boundary of the cache.
279                if rem_len > 0 {
280                    let src1 = src.narrow(self.dim, 0, rem_len)?.contiguous()?;
281                    ad.slice_set(&src1, self.dim, self.offset)?;
282                }
283                let src2 = src
284                    .narrow(self.dim, rem_len, seq_len - rem_len)?
285                    .contiguous()?;
286                ad.slice_set(&src2, self.dim, 0)?;
287                self.offset = seq_len - rem_len;
288            }
289            if self.current_seq_len >= self.max_seq_len {
290                Ok(ad.clone())
291            } else {
292                Ok(ad.narrow(self.dim, 0, self.current_seq_len)?)
293            }
294        }
295    }
296}

298#[derive(Debug, Clone)]
299pub enum KvCache {
300    Normal { k: SingleCache, v: SingleCache },
301    Rotating { k: RotatingCache, v: RotatingCache },
302}
303
304impl KvCache {
305    pub fn new_normal(dim: usize, max_seq_len: usize, capacity_seq_len: usize) -> Self {
306        let k = SingleCache::new(dim, max_seq_len, capacity_seq_len);
307        let v = SingleCache::new(dim, max_seq_len, capacity_seq_len);
308        Self::Normal { k, v }
309    }
310
311    pub fn new_rotating(dim: usize, sliding_window: usize, capacity_seq_len: usize) -> Self {
312        let k = RotatingCache::new(dim, sliding_window, capacity_seq_len);
313        let v = RotatingCache::new(dim, sliding_window, capacity_seq_len);
314        Self::Rotating { k, v }
315    }
316
317    pub fn k(&self) -> Result<Option<Tensor>> {
318        match self {
319            Self::Normal { k, .. } => k.current_data(),
320            Self::Rotating { k, .. } => k.current_data(),
321        }
322    }
323
324    pub fn v(&self) -> Result<Option<Tensor>> {
325        match self {
326            Self::Normal { v, .. } => v.current_data(),
327            Self::Rotating { v, .. } => v.current_data(),
328        }
329    }
330
331    pub fn append(&mut self, k: &Tensor, v: &Tensor) -> Result<(Tensor, Tensor)> {
332        let k = k.contiguous()?;
333        let v = v.contiguous()?;
334        let (out_k, out_v) = match self {
335            Self::Normal { k: kc, v: vc } => {
336                kc.append(&k)?;
337                vc.append(&v)?;
338                (kc.current_data()?, vc.current_data()?)
339            }
340            Self::Rotating { k: kc, v: vc } => {
341                let out_k = kc.append(&k)?;
342                let out_v = vc.append(&v)?;
343                (Some(out_k), Some(out_v))
344            }
345        };
346        let k = match out_k {
347            None => {
348                let mut shape = k.dims().to_vec();
349                match self {
350                    Self::Normal { k, .. } => shape[k.dim] = 0,
351                    Self::Rotating { k, .. } => shape[k.dim] = 0,
352                }
353                Tensor::zeros(shape, k.dtype(), k.device())?
354            }
355            Some(k) => k,
356        };
357        let v = match out_v {
358            None => {
359                let mut shape = v.dims().to_vec();
360                match self {
361                    Self::Normal { v, .. } => shape[v.dim] = 0,
362                    Self::Rotating { v, .. } => shape[v.dim] = 0,
363                }
364                Tensor::zeros(shape, v.dtype(), v.device())?
365            }
366            Some(v) => v,
367        };
368        Ok((k, v))
369    }
370
371    pub fn current_seq_len(&self) -> usize {
372        match self {
373            Self::Normal { k, .. } => k.current_seq_len(),
374            Self::Rotating { k, .. } => k.current_seq_len(),
375        }
376    }
377
378    pub fn reset(&mut self) {
379        match self {
380            Self::Normal { k, v } => {
381                k.reset();
382                v.reset();
383            }
384            Self::Rotating { k, v } => {
385                k.reset();
386                v.reset();
387            }
388        }
389    }
390
391    /// Returns Ok if the length reassignment was successful, otherwise returns Err.
392    pub fn set_len(&mut self, len: usize) -> candle_core::Result<()> {
393        match self {
394            Self::Normal { k, v } => {
395                k.set_len(len)?;
396                v.set_len(len)?;
397                Ok(())
398            }
399            Self::Rotating { k, v } => {
400                k.set_len(len)?;
401                v.set_len(len)?;
402                Ok(())
403            }
404        }
405    }
406
407    pub fn is_rotating(&self) -> bool {
408        matches!(self, Self::Rotating { .. })
409    }
410}

415#[derive(Debug)]
416pub enum NormalCacheType {
417    Normal { max_seq_len: usize },
418    SlidingWindow { window: usize },
419}
420
421impl NormalCache {
422    /// The number of tokens to grow the cache by
423    pub const CACHE_GROW_SIZE: usize = 512;
424
425    pub fn new(len: usize, max_seq_len: usize) -> Arc<Mutex<Self>> {
426        Arc::new(Mutex::new(Self(vec![
427            KvCache::new_normal(
428                2,
429                max_seq_len,
430                Self::CACHE_GROW_SIZE
431            );
432            len
433        ])))
434    }
435
436    pub fn new_sliding(
437        len: usize,
438        max_seq_len: usize,
439        sliding_window: Option<usize>,
440    ) -> Arc<Mutex<Self>> {
441        match sliding_window {
442            Some(sliding_window) => Arc::new(Mutex::new(Self(vec![
443                KvCache::new_rotating(
444                    2,
445                    sliding_window,
446                    Self::CACHE_GROW_SIZE
447                );
448                len
449            ]))),
450            None => Arc::new(Mutex::new(Self(vec![
451                KvCache::new_normal(
452                    2,
453                    max_seq_len,
454                    Self::CACHE_GROW_SIZE
455                );
456                len
457            ]))),
458        }
459    }
460
461    pub fn from_types(types: Vec<NormalCacheType>) -> Arc<Mutex<Self>> {
462        let mut caches = Vec::new();
463        for ty in types {
464            match ty {
465                NormalCacheType::Normal { max_seq_len } => {
466                    caches.push(KvCache::new_normal(2, max_seq_len, Self::CACHE_GROW_SIZE));
467                }
468                NormalCacheType::SlidingWindow { window } => {
469                    caches.push(KvCache::new_rotating(2, window, Self::CACHE_GROW_SIZE));
470                }
471            }
472        }
473        Arc::new(Mutex::new(Self(caches)))
474    }

479impl<T: CacheManagerMixin + MetadataMixin + ?Sized> CacheManager<T> for NormalCacheManager {
480    fn clone_in_cache(
481        &self,
482        pipeline: &T,
483        seqs: &mut [&mut crate::sequence::Sequence],
484        modify_draft_cache: bool,
485    ) {
486        let mut new_k_cache = Vec::new();
487        let mut new_v_cache = Vec::new();
488
489        for layer in 0..pipeline.get_metadata().num_hidden_layers {
490            // Preallocate combined k and v caches across all sequences, avoiding Tensor::cat copies
491            let batch_len = seqs.len();
492            // Use the first sequence as template
493            let (first_k, first_v) = {
494                let src_cache = if modify_draft_cache {
495                    seqs[0].normal_draft_cache()
496                } else {
497                    seqs[0].normal_cache()
498                };
499                let cache = src_cache.get(layer).unwrap().as_ref().unwrap();
500                match cache {
501                    KvCache::Normal { k, v } => {
502                        (k.all_data.clone().unwrap(), v.all_data.clone().unwrap())
503                    }
504                    KvCache::Rotating { k, v } => {
505                        (k.all_data.clone().unwrap(), v.all_data.clone().unwrap())
506                    }
507                }
508            };
509            // Build dims for batched cache
510            let mut dims_k = first_k.dims().to_vec();
511            let mut dims_v = first_v.dims().to_vec();
512            dims_k[0] *= batch_len;
513            dims_v[0] *= batch_len;
514            let batch_k = Tensor::zeros(dims_k.clone(), first_k.dtype(), first_k.device()).unwrap();
515            let batch_v = Tensor::zeros(dims_v.clone(), first_v.dtype(), first_v.device()).unwrap();
516            // Fill each sequence's cache slice
517            for (i, seq) in seqs.iter_mut().enumerate() {
518                let src_cache = if modify_draft_cache {
519                    seq.normal_draft_cache()
520                } else {
521                    seq.normal_cache()
522                };
523                let cache = src_cache.get(layer).unwrap().as_ref().unwrap();
524                let (src_k, src_v) = match cache {
525                    KvCache::Normal { k, v } => {
526                        (k.all_data.clone().unwrap(), v.all_data.clone().unwrap())
527                    }
528                    KvCache::Rotating { k, v } => {
529                        (k.all_data.clone().unwrap(), v.all_data.clone().unwrap())
530                    }
531                };
532                let offset = i * first_k.dims()[0];
533                batch_k.slice_set(&src_k, 0, offset).unwrap();
534                batch_v.slice_set(&src_v, 0, offset).unwrap();
535            }
536            new_k_cache.push(Some(batch_k));
537            new_v_cache.push(Some(batch_v));
538        }
539
540        let seq0_cache = if modify_draft_cache {
541            &*seqs[0].normal_draft_cache()
542        } else {
543            &*seqs[0].normal_cache()
544        };
545
546        let mut caches = Vec::new();
547        for (layer_idx, (k_cache, v_cache)) in new_k_cache.into_iter().zip(new_v_cache).enumerate()
548        {
549            // Use this for the various parameters. Assumes all seqs are from one model.
550            match seq0_cache[layer_idx].as_ref().unwrap() {
551                KvCache::Normal { k: old_k, .. } => {
552                    let template_cache_dim = old_k.dim;
553                    let template_cache_csl = old_k.current_seq_len;
554                    let template_cache_msl = old_k.max_seq_len;
555                    let template_cache_capsl = old_k.capacity_seq_len;
556
557                    caches.push(KvCache::Normal {
558                        k: SingleCache {
559                            all_data: k_cache.map(|x| x.contiguous().unwrap()),
560                            dim: template_cache_dim,
561                            current_seq_len: template_cache_csl,
562                            max_seq_len: template_cache_msl,
563                            capacity_seq_len: template_cache_capsl,
564                        },
565                        v: SingleCache {
566                            all_data: v_cache.map(|x| x.contiguous().unwrap()),
567                            dim: template_cache_dim,
568                            current_seq_len: template_cache_csl,
569                            max_seq_len: template_cache_msl,
570                            capacity_seq_len: template_cache_capsl,
571                        },
572                    });
573                }
574                KvCache::Rotating { k: old_k, .. } => {
575                    let template_cache_dim = old_k.dim;
576                    let template_cache_csl = old_k.current_seq_len;
577                    let template_cache_msl = old_k.max_seq_len;
578                    let template_cache_offset = old_k.offset;
579                    let template_cache_capsl = old_k.capacity_seq_len;
580
581                    caches.push(KvCache::Rotating {
582                        k: RotatingCache {
583                            all_data: k_cache.map(|x| x.contiguous().unwrap()),
584                            dim: template_cache_dim,
585                            current_seq_len: template_cache_csl,
586                            max_seq_len: template_cache_msl,
587                            offset: template_cache_offset,
588                            capacity_seq_len: template_cache_capsl,
589                        },
590                        v: RotatingCache {
591                            all_data: v_cache.map(|x| x.contiguous().unwrap()),
592                            dim: template_cache_dim,
593                            current_seq_len: template_cache_csl,
594                            max_seq_len: template_cache_msl,
595                            offset: template_cache_offset,
596                            capacity_seq_len: template_cache_capsl,
597                        },
598                    });
599                }
600            }
601        }
602        *pipeline.cache().normal() = NormalCache(caches);
603    }
604    fn clone_out_cache(&self, pipeline: &T, seqs: &mut [&mut Sequence], modify_draft_cache: bool) {
605        let all_cache = pipeline.cache().normal();
606        for layer in 0..pipeline.get_metadata().num_hidden_layers {
607            let cache = all_cache.0.get(layer).unwrap();
608            // This case for llama 3.2 vision cross attn
609            if cache.k().unwrap().is_none() {
610                continue;
611            }
612
613            let (k_cache, v_cache) = match cache {
614                KvCache::Normal { k, v } => {
615                    (k.all_data.clone().unwrap(), v.all_data.clone().unwrap())
616                }
617                KvCache::Rotating { k, v } => {
618                    (k.all_data.clone().unwrap(), v.all_data.clone().unwrap())
619                }
620            };
621
622            let k_caches = k_cache.chunk(seqs.len(), 0).unwrap();
623            debug_assert_eq!(k_caches.len(), seqs.len());
624            let v_caches = v_cache.chunk(seqs.len(), 0).unwrap();
625            debug_assert_eq!(v_caches.len(), seqs.len());
626
627            for (seq_i, seq) in seqs.iter_mut().enumerate() {
628                let output_cache = if modify_draft_cache {
629                    seq.normal_draft_cache()
630                } else {
631                    seq.normal_cache()
632                };
633                let seq_cache = &mut output_cache[layer];
634                let k = k_caches.get(seq_i).unwrap().clone();
635                let v = v_caches.get(seq_i).unwrap().clone();
636
637                match cache {
638                    KvCache::Normal {
639                        k: cache_k,
640                        v: cache_v,
641                    } => {
642                        *seq_cache = Some(KvCache::Normal {
643                            k: SingleCache {
644                                all_data: Some(k),
645                                dim: cache_k.dim,
646                                current_seq_len: cache_k.current_seq_len,
647                                max_seq_len: cache_k.max_seq_len,
648                                capacity_seq_len: cache_k.capacity_seq_len,
649                            },
650                            v: SingleCache {
651                                all_data: Some(v),
652                                dim: cache_v.dim,
653                                current_seq_len: cache_v.current_seq_len,
654                                max_seq_len: cache_v.max_seq_len,
655                                capacity_seq_len: cache_v.capacity_seq_len,
656                            },
657                        });
658                    }
659                    KvCache::Rotating {
660                        k: cache_k,
661                        v: cache_v,
662                    } => {
663                        *seq_cache = Some(KvCache::Rotating {
664                            k: RotatingCache {
665                                all_data: Some(k),
666                                dim: cache_k.dim,
667                                current_seq_len: cache_k.current_seq_len,
668                                max_seq_len: cache_k.max_seq_len,
669                                offset: cache_k.offset,
670                                capacity_seq_len: cache_k.capacity_seq_len,
671                            },
672                            v: RotatingCache {
673                                all_data: Some(v),
674                                dim: cache_v.dim,
675                                current_seq_len: cache_v.current_seq_len,
676                                max_seq_len: cache_v.max_seq_len,
677                                offset: cache_v.offset,
678                                capacity_seq_len: cache_v.capacity_seq_len,
679                            },
680                        });
681                    }
682                }
683            }
684        }
685    }
686    fn set_none_cache(
687        &self,
688        pipeline: &T,
689        seqs: &mut [&mut Sequence],
690        _modify_draft_cache: bool,
691        load_preallocated_cache: bool,
692    ) {
693        if seqs.iter().any(|seq| seq.preallocated_cache().is_none()) {
694            for layer in pipeline.cache().normal().0.iter_mut() {
695                layer.reset();
696            }
697            return;
698        }
699
700        let layer_devices = if let Some(device_mapper) = pipeline.device_mapper() {
701            let mut layer_devices = Vec::new();
702            for layer in 0..device_mapper.num_device_mapping_layers() {
703                let device = device_mapper.device_for(layer, false).cloned();
704                layer_devices.push(device.expect("Internal bug, layer out of range!"));
705            }
706            Some(layer_devices)
707        } else {
708            None
709        };
710
711        let old_caches = pipeline.cache().normal().0.clone();
712
713        for (layer_idx, layer) in pipeline.cache().normal().0.iter_mut().enumerate() {
714            if !load_preallocated_cache {
715                layer.reset();
716                continue;
717            }
718
719            let mut k_caches = Vec::new();
720            let mut v_caches = Vec::new();
721            for seq in seqs.iter_mut() {
722                let (mut k_preallocated_cache, mut v_preallocated_cache) =
723                    (*seq.preallocated_cache().as_ref().unwrap()).clone();
724                if let Some(layer_devices) = &layer_devices {
725                    let layer_dev = &layer_devices[layer_idx];
726                    k_preallocated_cache = k_preallocated_cache
727                        .to_device(layer_dev)
728                        .expect("Could not prepare cache");
729                    v_preallocated_cache = v_preallocated_cache
730                        .to_device(layer_dev)
731                        .expect("Could not prepare cache");
732                }
733                k_caches.push(k_preallocated_cache);
734                v_caches.push(v_preallocated_cache);
735            }
736            let k_cache = if k_caches.len() > 1 {
737                Tensor::cat(&k_caches, 0).unwrap()
738            } else {
739                k_caches[0].clone()
740            };
741            let v_cache = if v_caches.len() > 1 {
742                Tensor::cat(&v_caches, 0).unwrap()
743            } else {
744                v_caches[0].clone()
745            };
746
747            // Use this for the various parameters. Assumes all seqs are from one model.
748            match &old_caches[layer_idx] {
749                KvCache::Normal { k, .. } => {
750                    let template_cache_dim = k.dim;
751                    let template_cache_msl = k.max_seq_len;
752
753                    let cache = KvCache::Normal {
754                        k: SingleCache {
755                            all_data: Some(k_cache.zeros_like().unwrap()),
756                            dim: template_cache_dim,
757                            current_seq_len: 0,
758                            max_seq_len: template_cache_msl,
759                            capacity_seq_len: k_cache.dims()[template_cache_dim],
760                        },
761                        v: SingleCache {
762                            all_data: Some(v_cache.zeros_like().unwrap()),
763                            dim: template_cache_dim,
764                            current_seq_len: 0,
765                            max_seq_len: template_cache_msl,
766                            capacity_seq_len: k_cache.dims()[template_cache_dim],
767                        },
768                    };
769                    *layer = cache;
770                }
771                KvCache::Rotating { k, .. } => {
772                    let template_cache_dim = k.dim;
773                    let template_cache_msl = k.max_seq_len;
774
775                    // Rotating cache is not preallocated.
776                    let cache = KvCache::Rotating {
777                        k: RotatingCache {
778                            all_data: None,
779                            dim: template_cache_dim,
780                            current_seq_len: 0,
781                            max_seq_len: template_cache_msl,
782                            offset: 0,
783                            capacity_seq_len: 0,
784                        },
785                        v: RotatingCache {
786                            all_data: None,
787                            dim: template_cache_dim,
788                            current_seq_len: 0,
789                            max_seq_len: template_cache_msl,
790                            offset: 0,
791                            capacity_seq_len: 0,
792                        },
793                    };
794                    *layer = cache;
795                }
796            }
797        }
798    }
799}

809impl Cache {
810    pub(crate) fn new(len: usize, is_xlora: bool) -> Self {
811        Self {
812            cache: Arc::new(Mutex::new(vec![None; len])),
813            xlora_cache: if is_xlora {
814                Some(Arc::new(Mutex::new(vec![None; len])))
815            } else {
816                None
817            },
818            draft_cache: Arc::new(Mutex::new(vec![None; len])),
819            scalings_cache: if is_xlora {
820                Some(Arc::new(Mutex::new(None)))

822                None
823            },
824        }
825    }

828        get_mut_arcmutex!(self.cache)
829    }
830
831    pub(crate) fn draft_lock(&self) -> MutexGuard<'_, LayerCaches> {
832        get_mut_arcmutex!(self.draft_cache)
833    }

835    /// # Panics
836    /// If there is no xlora cache
837    pub(crate) fn xlora_lock(&self) -> MutexGuard<'_, LayerCaches> {
838        get_mut_arcmutex!(self.xlora_cache.as_ref().expect("No X-LoRA cache."))
839    }

842    /// If there is no xlora cache

844        get_mut_arcmutex!(self
845            .scalings_cache
846            .as_ref()
847            .expect("No X-LoRA scalings cache."))
848    }

850    pub(crate) fn is_xlora(&self) -> bool {
851        self.xlora_cache.is_some()
852    }
853
854    /// Update the KV cache and return (k,v)
855    pub(crate) fn update_kv_cache(
856        cache: &mut Option<(Tensor, Tensor)>,
857        k: Tensor,
858        v: Tensor,
859        slow_cat: bool,
860    ) -> Result<(Tensor, Tensor)> {
861        let (k, v) = match &*cache {
862            None => (k, v),
863            Some((k_cache, v_cache)) => {
864                if !slow_cat {
865                    let k = candle_nn::ops::kvconcat(k_cache, &k, 2)?.contiguous()?;
866                    let v = candle_nn::ops::kvconcat(v_cache, &v, 2)?.contiguous()?;
867                    (k, v)
868                } else {
869                    let k = Tensor::cat(&[k_cache, &k], 2)?.contiguous()?;
870                    let v = Tensor::cat(&[v_cache, &v], 2)?.contiguous()?;
871                    (k, v)
872                }
873            }
874        };
875        *cache = Some((k.clone(), v.clone()));
876        Ok((k.contiguous()?, v.contiguous()?))
877    }
878
879    /// Update the KV cache and return (k,v,attn_mask)
880    pub(crate) fn update_kv_cache_sliding_window(
881        cache: &mut Option<(Tensor, Tensor)>,
882        k: Tensor,
883        v: Tensor,
884        attention_mask: Option<&Tensor>,
885        sliding_window: Option<usize>,
886        slow_cat: bool,
887    ) -> Result<(Tensor, Tensor, Option<Tensor>)> {
888        let (k, v, attention_mask) = match cache.clone() {
889            None => (k, v, attention_mask.cloned()),
890            Some((mut prev_k, mut prev_v)) => {
891                let mut mask = attention_mask.cloned();
892                if let Some(sliding_window) = sliding_window {
893                    let kv_seq_len = prev_k.dim(2)?;
894                    if kv_seq_len > sliding_window {
895                        prev_k = prev_k.narrow(
896                            2,
897                            kv_seq_len - (sliding_window - 1),
898                            sliding_window - 1,
899                        )?;
900                        prev_v = prev_v.narrow(
901                            2,
902                            kv_seq_len - (sliding_window - 1),
903                            sliding_window - 1,
904                        )?;
905                        if let Some(ref mut mask) = mask {
906                            let mask_len = mask.dim(1)?;
907                            *mask = mask.narrow(
908                                1,
909                                mask_len - (sliding_window - 1),
910                                sliding_window - 1,
911                            )?;
912                            *mask = Tensor::cat(
913                                &[&*mask, &mask.narrow(1, mask_len - 1, 1)?.ones_like()?],
914                                D::Minus1,
915                            )?;
916                        }
917                    }
918                }
919                let (k, v) = if !slow_cat {
920                    let k = candle_nn::ops::kvconcat(&prev_k, &k, 2)?;
921                    let v = candle_nn::ops::kvconcat(&prev_v, &v, 2)?;
922                    (k, v)
923                } else {
924                    let k = Tensor::cat(&[prev_k, k], 2)?.contiguous()?;
925                    let v = Tensor::cat(&[prev_v, v], 2)?.contiguous()?;
926                    (k, v)
927                };
928                (k, v, mask)
929            }
930        };
931        *cache = Some((k.clone(), v.clone()));
932        Ok((k.contiguous()?, v.contiguous()?, attention_mask))
933    }
934}

938enum SeqCache {
939    Normal,
940    XLora,
941    Draft,
942}
943
944fn clone_in_cache(
945    num_hidden_layers: usize,
946    cache: &mut LayerCaches,
947    seqs: &mut [&mut crate::sequence::Sequence],
948    src: SeqCache,
949) {
950    let mut new_cache = Vec::new();
951    'outer: for layer in 0..num_hidden_layers {
952        let mut k_vec = Vec::new();
953        let mut v_vec = Vec::new();
954        for seq in &mut *seqs {
955            let src_cache = match src {
956                SeqCache::Normal => seq.cache(),
957                SeqCache::XLora => seq.xlora_cache(),
958                SeqCache::Draft => seq.draft_cache(),
959            };
960            let cache = src_cache.get(layer).unwrap();
961            // This case for llama 3.2 vision cross attn
962            if cache.is_none() {
963                new_cache.push(None);
964                continue 'outer;
965            }
966            let cache = cache
967                .as_ref()
968                .expect("Not handling completions in `clone_in_cache`.");
969            k_vec.push(cache.0.clone());
970            v_vec.push(cache.1.clone());
971        }
972        new_cache.push(Some((
973            if k_vec.len() > 1 {
974                Tensor::cat(&k_vec, 0).unwrap()
975            } else {
976                k_vec[0].clone()
977            },
978            if v_vec.len() > 1 {
979                Tensor::cat(&v_vec, 0).unwrap()
980            } else {
981                v_vec[0].clone()
982            },
983        )));
984    }
985    *cache = new_cache;
986}
987
988fn clone_out_cache(
989    num_hidden_layers: usize,
990    cache: &mut LayerCaches,
991    seqs: &mut [&mut crate::sequence::Sequence],
992    target: SeqCache,
993) {
994    for layer in 0..num_hidden_layers {
995        let cache = cache.get(layer).unwrap();
996        // This case for llama 3.2 vision cross attn
997        if cache.is_none() {
998            continue;
999        }
1000
1001        let k_cache = cache.as_ref().unwrap().0.clone();
1002        let v_cache = cache.as_ref().unwrap().1.clone();
1003
1004        let k_caches = k_cache.chunk(seqs.len(), 0).unwrap();
1005        debug_assert_eq!(k_caches.len(), seqs.len());
1006        let v_caches = v_cache.chunk(seqs.len(), 0).unwrap();
1007        debug_assert_eq!(v_caches.len(), seqs.len());
1008
1009        for (seq_i, seq) in seqs.iter_mut().enumerate() {
1010            let output_cache = match target {
1011                SeqCache::Normal => seq.cache(),
1012                SeqCache::XLora => seq.xlora_cache(),
1013                SeqCache::Draft => seq.draft_cache(),
1014            };
1015            let seq_cache = &mut output_cache[layer];
1016            let k = k_caches.get(seq_i).unwrap().clone();
1017            let v = v_caches.get(seq_i).unwrap().clone();
1018            *seq_cache = Some((k, v));
1019        }
1020    }
1021}

1023impl<T: CacheManagerMixin + MetadataMixin + ?Sized> CacheManager<T> for FullCacheManager {
1024    fn clone_in_cache(
1025        &self,
1026        pipeline: &T,
1027        seqs: &mut [&mut crate::sequence::Sequence],
1028        modify_draft_cache: bool,
1029    ) {
1030        if modify_draft_cache {
1031            clone_in_cache(
1032                pipeline.get_metadata().num_hidden_layers,
1033                &mut pipeline.cache().full().lock(),
1034                seqs,
1035                SeqCache::Draft,
1036            );
1037            return;
1038        }
1039        clone_in_cache(
1040            pipeline.get_metadata().num_hidden_layers,
1041            &mut pipeline.cache().full().lock(),
1042            seqs,
1043            SeqCache::Normal,
1044        );
1045        if pipeline.get_metadata().is_xlora && !pipeline.get_metadata().no_kv_cache {
1046            clone_in_cache(
1047                pipeline.get_metadata().num_hidden_layers,
1048                &mut pipeline.cache().full().xlora_lock(),
1049                seqs,
1050                SeqCache::XLora,
1051            );
1052        }
1053        if pipeline.get_metadata().is_xlora {
1054            pipeline
1055                .cache()
1056                .full()
1057                .get_scalings_cache()
1058                .clone_from(seqs[0].scaling_cache());
1059        }
1060    }
1061
1062    fn clone_out_cache(
1063        &self,
1064        pipeline: &T,
1065        seqs: &mut [&mut crate::sequence::Sequence],
1066        modify_draft_cache: bool,
1067    ) {
1068        if modify_draft_cache {
1069            clone_out_cache(
1070                pipeline.get_metadata().num_hidden_layers,
1071                &mut pipeline.cache().full().lock(),
1072                seqs,
1073                SeqCache::Draft,
1074            );
1075            return;
1076        }
1077        clone_out_cache(
1078            pipeline.get_metadata().num_hidden_layers,
1079            &mut pipeline.cache().full().lock(),
1080            seqs,
1081            SeqCache::Normal,
1082        );
1083        if pipeline.get_metadata().is_xlora && !pipeline.get_metadata().no_kv_cache {
1084            clone_out_cache(
1085                pipeline.get_metadata().num_hidden_layers,
1086                &mut pipeline.cache().full().xlora_lock(),
1087                seqs,
1088                SeqCache::XLora,
1089            );
1090        }
1091        if pipeline.get_metadata().is_xlora {
1092            seqs[0]
1093                .scaling_cache()
1094                .clone_from(&pipeline.cache().full().get_scalings_cache());
1095        }
1096    }
1097
1098    fn set_none_cache(
1099        &self,
1100        pipeline: &T,
1101        _seqs: &mut [&mut Sequence],
1102        modify_draft_cache: bool,
1103        _load_preallocated_cache: bool,
1104    ) {
1105        let mut new_cache = Vec::new();
1106        for _ in 0..pipeline.get_metadata().num_hidden_layers {
1107            new_cache.push(None);
1108        }
1109        pipeline.cache().full().lock().clone_from(&new_cache);
1110        if modify_draft_cache {
1111            pipeline.cache().full().draft_lock().clone_from(&new_cache);
1112        }
1113        if pipeline.cache().full().is_xlora() {
1114            *pipeline.cache().full().xlora_lock() = new_cache;
1115        }
1116    }
