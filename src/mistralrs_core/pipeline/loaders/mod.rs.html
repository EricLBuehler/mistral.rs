51/// `ModelPaths` abstracts the mechanism to get all necessary files for running a model. For
52/// example `LocalModelPaths` implements `ModelPaths` when all files are in the local file system.
53pub trait ModelPaths: AsAny + Debug + Send + Sync {
54    /// Model weights files (multiple files supported).
55    fn get_weight_filenames(&self) -> &[PathBuf];
56
57    /// Retrieve the [`PretrainedConfig`] file.
58    ///
59    /// [`PretrainedConfig`]: https://huggingface.co/docs/transformers/v4.40.2/en/main_classes/configuration#transformers.PretrainedConfig
60    fn get_config_filename(&self) -> &PathBuf;
61
62    /// A serialised [`tokenizers.Tokenizer`] HuggingFace object.
63    ///
64    /// [`tokenizers.Tokenizer`]: https://huggingface.co/docs/transformers/v4.40.2/en/main_classes/tokenizer
65    fn get_tokenizer_filename(&self) -> &PathBuf;
66
67    /// File where the content is expected to deserialize to [`ChatTemplate`].
68    ///
69    /// [`ChatTemplate`]: crate::ChatTemplate
70    fn get_template_filename(&self) -> &Option<PathBuf>;
71
72    /// Optional adapter files. `(String, PathBuf)` is of the form `(id name, path)`.
73    fn get_adapter_filenames(&self) -> &Option<Vec<(String, PathBuf)>>;
74
75    /// Configuration of optional adapters. `(String, String)` is of the form `(id name, name)`.
76    fn get_adapter_configs(&self) -> &Option<Vec<((String, String), LoraConfig)>>;
77
78    /// Filepath for the XLORA classifier
79    fn get_classifier_path(&self) -> &Option<PathBuf>;
80
81    /// `XLoraConfig` for the XLORA classifier
82    fn get_classifier_config(&self) -> &Option<XLoraConfig>;
83
84    /// Return the defined ordering of adapters and layers within the model.
85    fn get_ordering(&self) -> &Option<Ordering>;
86
87    /// Filepath for general model configuration.
88    fn get_gen_conf_filename(&self) -> Option<&PathBuf>;
89
90    /// Information for preloading LoRA adapters (adapter name, the weight file, and the config).
91    fn get_lora_preload_adapter_info(&self) -> &Option<HashMap<String, (PathBuf, LoraConfig)>>;
92
93    /// Get the preprocessor config (for the vision models). This is used to pre process images.
94    fn get_preprocessor_config(&self) -> &Option<PathBuf>;
95
96    /// Get the processor config (for the vision models). This is primarily used for the chat template.
97    fn get_processor_config(&self) -> &Option<PathBuf>;
98
99    /// Get the explicit chat template.
100    fn get_chat_template_explicit(&self) -> &Option<PathBuf>;
101}

205/// The source of the HF token.
206pub enum TokenSource {
207    Literal(String),
208    EnvVar(String),
209    Path(String),
210    CacheToken,
211    None,
212}
213
214impl FromStr for TokenSource {
215    type Err = String;
216
217    fn from_str(s: &str) -> Result<Self, Self::Err> {
218        let parts: Vec<&str> = s.splitn(2, ':').collect();
219        match parts[0] {
220            "literal" => parts
221                .get(1)
222                .map(|&value| TokenSource::Literal(value.to_string()))
223                .ok_or_else(|| "Expected a value for 'literal'".to_string()),
224            "env" => Ok(TokenSource::EnvVar(
225                parts
226                    .get(1)
227                    .unwrap_or(&"HUGGING_FACE_HUB_TOKEN")
228                    .to_string(),
229            )),
230            "path" => parts
231                .get(1)
232                .map(|&value| TokenSource::Path(value.to_string()))
233                .ok_or_else(|| "Expected a value for 'path'".to_string()),
234            "cache" => Ok(TokenSource::CacheToken),
235            "none" => Ok(TokenSource::None),
236            _ => Err("Invalid token source format".to_string()),
237        }
238    }
239}
240
241impl fmt::Display for TokenSource {
242    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
243        match self {
244            TokenSource::Literal(value) => write!(f, "literal:{}", value),
245            TokenSource::EnvVar(value) => write!(f, "env:{}", value),
246            TokenSource::Path(value) => write!(f, "path:{}", value),
247            TokenSource::CacheToken => write!(f, "cache"),
248            TokenSource::None => write!(f, "none"),
249        }
250    }
251}

317impl ModelKind {
318    // Quantized helpers:
319    pub fn is_quantized(&self) -> bool {
320        self.quantized_kind().iter().any(|q| q.is_some())
321    }
322
323    pub fn is_quantized_and(&self, mut f: impl FnMut(QuantizationKind) -> bool) -> bool {
324        self.quantized_kind().iter().any(|q| q.is_some_and(&mut f))
325    }
326
327    pub fn quantized_kind(&self) -> Vec<Option<QuantizationKind>> {
328        use ModelKind::*;
329
330        match self {
331            Normal | Adapter { .. } => vec![None],
332            GgufQuantized { quant } | GgufAdapter { quant, .. } => vec![Some(*quant)],
333            Speculative { target, draft } => {
334                let t = *target.clone();
335                let d = *draft.clone();
336
337                [t.quantized_kind(), d.quantized_kind()].concat()
338            }
339            AnyMoe { target } => target.quantized_kind(),
340        }
341    }
342
343    // Adapter helpers:
344    pub fn is_adapted(&self) -> bool {
345        self.adapted_kind().iter().any(|a| a.is_some())
346    }
347
348    pub fn is_adapted_and(&self, mut f: impl FnMut(AdapterKind) -> bool) -> bool {
349        self.adapted_kind().iter().any(|a| a.is_some_and(&mut f))
350    }
351
352    pub fn adapted_kind(&self) -> Vec<Option<AdapterKind>> {
353        use ModelKind::*;
354
355        match self {
356            Normal | GgufQuantized { .. } => vec![None],
357            Adapter { adapter } | GgufAdapter { adapter, .. } => vec![Some(*adapter)],
358            Speculative { target, draft } => {
359                let t = *target.clone();
360                let d = *draft.clone();
361
362                [t.adapted_kind(), d.adapted_kind()].concat()
363            }
364            AnyMoe { target } => target.adapted_kind(),
365        }
366    }

459fn calculate_key_block_shape(
460    model_config: &dyn ModelConfigLike,
461    dtype: DType,
462    block_size: usize,
463) -> (usize, usize, usize, usize) {
464    let element_size = dtype.size_in_bytes();
465    let x = 16 / element_size;
466    (
467        model_config.num_kv_heads(),
468        model_config.k_head_dim() / x,
469        block_size,
470        x,
471    )
472}
473
474fn calculate_value_block_shape(
475    model_config: &dyn ModelConfigLike,
476    block_size: usize,
477) -> (usize, usize, usize) {
478    (
479        model_config.num_kv_heads(),
480        model_config.v_head_dim(),
481        block_size,
482    )
483}

485pub trait DeviceMappedModelLoader {
486    /// Maximum activation size of non-mapped parts of this model.
487    /// Useful for the vision models which may prefer to keep the vison components on the GPU.
488    fn non_mapped_max_act_size_elems(
489        &self,
490        config: &str,
491        params: &AutoDeviceMapParams,
492    ) -> Result<usize>;
493    /// Maximum activation size of mapped parts of the model
494    fn mapped_max_act_size_elems(
495        &self,
496        config: &str,
497        params: &AutoDeviceMapParams,
498        prompt_chunksize: usize,
499    ) -> Result<usize>;
500    /// weight_pack_factor only applies to quantized weights.
501    fn non_mapped_size_in_bytes(
502        &self,
503        config: &str,
504        dtype: DType,
505        weight_pack_factor: usize,
506    ) -> Result<usize>;
507    /// weight_pack_factor only applies to quantized weights.
508    fn layer_sizes_in_bytes(
509        &self,
510        config: &str,
511        dtype: DType,
512        weight_pack_factor: usize,
513    ) -> Result<Vec<usize>>;
514    fn non_mapped_sub_models(&self) -> Option<Vec<NonMappedSubModel>> {
515        None
516    }
517    fn num_layers(&self, config: &str) -> Result<usize>;
518    fn model_config(&self, config: &str) -> Result<Box<dyn ModelConfigLike>>;
519
520    #[allow(clippy::too_many_arguments)]
521    fn get_device_layers(
522        &self,
523        config: &str,
524        num_layers: usize,
525        mut layer_sizes_in_bytes: Vec<usize>,
526        non_mapped_size_in_bytes: usize,
527        total_model_size_in_bytes: usize,
528        devices: &[Device],
529        dtype: DType,
530        params: &AutoDeviceMapParams,
531        prompt_chunksize: usize,
532        paged_attn_config: Option<&PagedAttentionConfig>,
533    ) -> Result<DeviceMapMetadata> {
534        let mapped_max_act_size_in_bytes =
535            self.mapped_max_act_size_elems(config, params, prompt_chunksize)?
536                * dtype.size_in_bytes();
537        let non_mapped_max_act_size_in_bytes =
538            self.non_mapped_max_act_size_elems(config, params)? * dtype.size_in_bytes();
539
540        let mut remaining_to_map = total_model_size_in_bytes;
541
542        let max_seq_len = match params {
543            AutoDeviceMapParams::Text { max_seq_len, .. }
544            | AutoDeviceMapParams::Vision { max_seq_len, .. } => *max_seq_len,
545        };
546        let max_batch_size = match params {
547            AutoDeviceMapParams::Text { max_batch_size, .. }
548            | AutoDeviceMapParams::Vision { max_batch_size, .. } => *max_batch_size,
549        };
550
551        let model_cfg = self.model_config(config)?;
552        let kv_cache_size_elems = match paged_attn_config {
553            Some(paged_attn_config) => {
554                let cache_config = calculate_cache_config(
555                    paged_attn_config.mem_gpu,
556                    paged_attn_config.mem_cpu,
557                    Some(
558                        paged_attn_config
559                            .block_size
560                            .unwrap_or(DEFAULT_PAGED_ATTENTION_BLOCK_SIZE),
561                    ),
562                    dtype,
563                    &*model_cfg,
564                    &devices[0],
565                    &devices.iter().map(|x| Some(x.clone())).collect::<Vec<_>>(),
566                    true,
567                )?;
568
569                let key_block_shape =
570                    calculate_key_block_shape(&*model_cfg, dtype, cache_config.block_size);
571                let key_block_size = cache_config.num_gpu_blocks
572                    * key_block_shape.0
573                    * key_block_shape.1
574                    * key_block_shape.2
575                    * key_block_shape.3;
576
577                let value_block_shape = calculate_value_block_shape(
578                    &*self.model_config(config)?,
579                    cache_config.block_size,
580                );
581                let value_block_size = cache_config.num_gpu_blocks
582                    * value_block_shape.0
583                    * value_block_shape.1
584                    * value_block_shape.2;
585
586                key_block_size + value_block_size
587            }
588            None => {
589                // Non-paged KV cache
590                let key_block_shape = [
591                    max_batch_size,
592                    model_cfg.num_kv_heads(),
593                    max_seq_len,
594                    model_cfg.k_head_dim(),
595                ];
596                let value_block_shape = [
597                    max_batch_size,
598                    model_cfg.num_kv_heads(),
599                    max_seq_len,
600                    model_cfg.v_head_dim(),
601                ];
602
603                key_block_shape.into_iter().product::<usize>()
604                    + value_block_shape.iter().product::<usize>()
605            }
606        };
607        let kv_cache_size_in_bytes = kv_cache_size_elems * dtype.size_in_bytes();
608
609        let mut per_layer_avail = Vec::new();
610        // Always add the CPU as fallback
611        for dev in [devices, &[Device::Cpu]].concat() {
612            let avail = MemoryUsage.get_memory_available(&dev)?;
613            per_layer_avail.push((avail, dev));
614        }
615        // Reverse so we don't use the cpu first!
616        per_layer_avail.reverse();
617
618        // Reverse layer sizes so we can pop
619        layer_sizes_in_bytes.reverse();
620
621        let mut device_layers = Vec::new();
622
623        info!("Using automatic device mapping parameters: {params}.");
624        if let Some(sub_models) = self.non_mapped_sub_models() {
625            let (_, last) = per_layer_avail.last().unwrap();
626            info!(
627                "The following sub-models will not be device mapped and will be loaded on {}: {}",
628                last.device_pretty_repr(),
629                sub_models.iter().map(|x| x.to_string()).join(", ")
630            );
631        }
632
633        let mut current_ordinal = 0;
634        let mut current_layer = 0;
635        let per_layer_avail_cpy = per_layer_avail.clone();
636        let mut mapping_includes_cpu = false;
637        while remaining_to_map > 0 && !per_layer_avail.is_empty() {
638            let (device_capacity, device) = per_layer_avail
639                .pop()
640                .context("No more devices to map to. The model does not fit on this system.")?;
641            // All usage of 90% of the memory as a maximum.
642            #[allow(clippy::cast_possible_truncation, clippy::cast_precision_loss)]
643            let device_capacity = (device_capacity as f64 * 0.90) as usize;
644
645            // Algorithm is to check the following:
646            // 1) (no mapping) if *everything* fits on the first dev (non mapped and mapped)
647            // 2) if the mapped activations plus remaining fits on the nth device
648            // 3) common case, iteratively find the optimal amount of layers to put on the nth device
649            //   - if this is the first dev: must hold the non-mapped act and non-mapped model
650            //   - otherwise, must hold the mapped act
651            #[allow(clippy::if_same_then_else)]
652            let layers_on_device = if current_ordinal == 0
653                && device_capacity
654                    >= remaining_to_map
655                        + non_mapped_max_act_size_in_bytes.max(mapped_max_act_size_in_bytes)
656                        + non_mapped_size_in_bytes
657                        + kv_cache_size_in_bytes * (num_layers - current_layer)
658            {
659                remaining_to_map = 0;
660
661                num_layers - current_layer
662            } else if current_ordinal != 0
663                && device_capacity
664                    >= remaining_to_map
665                        + mapped_max_act_size_in_bytes
666                        + kv_cache_size_in_bytes * (num_layers - current_layer)
667            {
668                remaining_to_map = 0;
669
670                num_layers - current_layer
671            } else {
672                // All devices need to account for the max mapped act size
673                let mut used_capacity = mapped_max_act_size_in_bytes;
674                let mut used_capacity_no_act = 0;
675                let mut layers_on_device = 0;
676
677                // Device w/ ordinal 0 carries the non-mapped things
678                if current_ordinal == 0 {
679                    // Ensure the activations are properly handled
680                    used_capacity = used_capacity.max(non_mapped_max_act_size_in_bytes);
681                    used_capacity += non_mapped_size_in_bytes;
682                    used_capacity_no_act += non_mapped_size_in_bytes;
683                }
684
685                while let Some(&last) = layer_sizes_in_bytes.last() {
686                    let delta = last + kv_cache_size_in_bytes;
687                    if used_capacity + delta > device_capacity {
688                        break;
689                    }
690                    let _ = layer_sizes_in_bytes.pop().unwrap();
691                    used_capacity += delta;
692                    used_capacity_no_act += delta;
693                    layers_on_device += 1;
694                }
695
696                // Do not reduce amount to map if this device can't fit any
697                // If the device cannot fit any layers, warn the user.
698                if layers_on_device > 0 {
699                    remaining_to_map = remaining_to_map.saturating_sub(used_capacity_no_act);
700                } else {
701                    warn!(
702                        "Device {} can fit 0 layers. Consider reducing auto map params from current: {params} (ex. reducing max seq len or max num images)",
703                        device.device_pretty_repr(),
704                    );
705                    current_ordinal += 1;
706                    continue;
707                }
708                layers_on_device
709            };
710
711            // CPU mappings are automatically handled by the traditional device mapper, we can just leave them out here.
712            if !device.is_cpu() {
713                device_layers.push(DeviceLayerMapMetadata {
714                    ordinal: current_ordinal,
715                    layers: layers_on_device,
716                });
717                current_ordinal += 1;
718            } else {
719                mapping_includes_cpu = true;
720            }
721
722            current_layer += layers_on_device;
723        }
724        if remaining_to_map > 0 {
725            anyhow::bail!(
726                "This model does not fit on the devices {:?}, and exceeds total capacity by {}MB. Auto device mapping params: {params}",
727                per_layer_avail_cpy
728                    .iter()
729                    .rev()
730                    .map(|(avail, dev)| format!(
731                        "{} (avail: {}MB)",
732                        dev.device_pretty_repr(),
733                        avail / (1024 * 1024),
734                    ))
735                    .collect::<Vec<_>>(),
736                b_to_mb!(remaining_to_map)
737            );
738        }
739
740        // TODO: PagedAttention is not supported with CPU for now.
741        // Recalculate without PagedAttention metadata.
742        if paged_attn_config.is_some_and(|_| mapping_includes_cpu) {
743            return self.get_device_layers(
744                config,
745                num_layers,
746                layer_sizes_in_bytes,
747                non_mapped_size_in_bytes,
748                total_model_size_in_bytes,
749                devices,
750                dtype,
751                params,
752                prompt_chunksize,
753                None,
754            );
755        }
756
757        Ok(DeviceMapMetadata::from_num_device_layers(device_layers))
758    }
759}

761/// The `Loader` trait abstracts the loading process. The primary entrypoint is the
762/// `load_model` method.
763///
764/// # Example
765/// ```no_run
766/// use mistralrs_core::{Loader, TokenSource, DeviceMapSetting, AutoDeviceMapParams, ModelDType};
767/// use candle_core::Device;
768///
769/// let loader: Box<dyn Loader> = todo!();
770/// let pipeline = loader.load_model_from_hf(
771///     None,
772///     TokenSource::CacheToken,
773///     &ModelDType::Auto,
774///     &Device::cuda_if_available(0).unwrap(),
775///     false,
776///     DeviceMapSetting::Auto(AutoDeviceMapParams::default_text()),
777///     None,
778///     None,
779/// ).unwrap();
780/// ```
781pub trait Loader: Send + Sync {
782    /// If `revision` is None, then it defaults to `main`.
783    /// If `dtype` is None, then it defaults to the model default (usually BF16).
784    /// If model is not found on HF, will attempt to resolve locally.
785    #[allow(clippy::type_complexity, clippy::too_many_arguments)]
786    fn load_model_from_hf(
787        &self,
788        revision: Option<String>,
789        token_source: TokenSource,
790        dtype: &dyn TryIntoDType,
791        device: &Device,
792        silent: bool,
793        mapper: DeviceMapSetting,
794        in_situ_quant: Option<IsqType>,
795        paged_attn_config: Option<PagedAttentionConfig>,
796    ) -> Result<Arc<Mutex<dyn Pipeline + Send + Sync>>>;
797
798    /// Load a model from the specified paths.
799    /// Also initializes `DEBUG`.
800    #[allow(
801        clippy::type_complexity,
802        clippy::too_many_arguments,
803        clippy::borrowed_box
804    )]
805    fn load_model_from_path(
806        &self,
807        paths: &Box<dyn ModelPaths>,
808        dtype: &dyn TryIntoDType,
809        device: &Device,
810        silent: bool,
811        mapper: DeviceMapSetting,
812        in_situ_quant: Option<IsqType>,
813        paged_attn_config: Option<PagedAttentionConfig>,
814    ) -> Result<Arc<Mutex<dyn Pipeline + Send + Sync>>>;
815
816    fn get_id(&self) -> String;
817    fn get_kind(&self) -> ModelKind;
818}
