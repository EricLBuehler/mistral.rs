64/// The architecture to load the vision model as.
65pub enum DiffusionLoaderType {
66    #[serde(rename = "flux")]
67    Flux,
68    #[serde(rename = "flux-offloaded")]
69    FluxOffloaded,
70}
71
72impl FromStr for DiffusionLoaderType {
73    type Err = String;
74    fn from_str(s: &str) -> Result<Self, Self::Err> {
75        match s {
76            "flux" => Ok(Self::Flux),
77            "flux-offloaded" => Ok(Self::FluxOffloaded),
78            a => Err(format!(
79                "Unknown architecture `{a}`. Possible architectures: `flux`."
80            )),
81        }
82    }
83}

126/// [`DiffusionLoader`] for a Flux Diffusion model.
127///
128/// [`DiffusionLoader`]: https://ericlbuehler.github.io/mistral.rs/mistralrs/struct.DiffusionLoader.html
129pub struct FluxLoader {
130    pub(crate) offload: bool,
131}
132
133impl DiffusionModelLoader for FluxLoader {
134    fn get_model_paths(&self, api: &ApiRepo, model_id: &Path) -> Result<Vec<PathBuf>> {
135        let regex = Regex::new(r"^flux\d+-(schnell|dev)\.safetensors$")?;
136        let flux_name = api_dir_list!(api, model_id)
137            .filter(|x| regex.is_match(x))
138            .nth(0)
139            .with_context(|| "Expected at least 1 .safetensors file matching the FLUX regex, please raise an issue.")?;
140        let flux_file = api_get_file!(api, &flux_name, model_id);
141        let ae_file = api_get_file!(api, "ae.safetensors", model_id);
142
143        // NOTE(EricLBuehler): disgusting way of doing this but the 0th path is the flux, 1 is ae
144        Ok(vec![flux_file, ae_file])
145    }
146    fn get_config_filenames(&self, api: &ApiRepo, model_id: &Path) -> Result<Vec<PathBuf>> {
147        let flux_file = api_get_file!(api, "transformer/config.json", model_id);
148        let ae_file = api_get_file!(api, "vae/config.json", model_id);
149
150        // NOTE(EricLBuehler): disgusting way of doing this but the 0th path is the flux, 1 is ae
151        Ok(vec![flux_file, ae_file])
152    }
153    fn force_cpu_vb(&self) -> Vec<bool> {
154        vec![self.offload, false]
155    }
156    fn load(
157        &self,
158        mut configs: Vec<String>,
159        mut vbs: Vec<ShardedVarBuilder>,
160        normal_loading_metadata: NormalLoadingMetadata,
161        _attention_mechanism: AttentionImplementation,
162        silent: bool,
163    ) -> Result<Box<dyn DiffusionModel + Send + Sync>> {
164        let (vae_cfg, vae_vb) = (configs.remove(1), vbs.remove(1));
165        let (flux_cfg, flux_vb) = (configs.remove(0), vbs.remove(0));
166
167        let vae_cfg: flux::autoencoder::Config = serde_json::from_str(&vae_cfg)?;
168        let flux_cfg: flux::model::Config = serde_json::from_str(&flux_cfg)?;
169
170        let flux_dtype = flux_vb.dtype();
171        if flux_dtype != vae_vb.dtype() {
172            anyhow::bail!(
173                "Expected VAE and FLUX model VBs to be the same dtype, got {:?} and {flux_dtype:?}",
174                vae_vb.dtype()
175            );
176        }
177
178        Ok(Box::new(FluxStepper::new(
179            FluxStepperConfig::default_for_guidance(flux_cfg.guidance_embeds),
180            (flux_vb, &flux_cfg),
181            (vae_vb, &vae_cfg),
182            flux_dtype,
183            &normal_loading_metadata.real_device,
184            silent,
185            self.offload,
186        )?))
187    }
188}
