87        &self,
88        config: &str,
89        vb: ShardedVarBuilder,
90        normal_loading_metadata: NormalLoadingMetadata,
91        attention_mechanism: AttentionImplementation,
92    ) -> Result<Box<dyn NormalModel + Send + Sync>>;
93    #[allow(clippy::too_many_arguments)]
94    fn load_xlora(
95        &self,
96        config: &str,
97        vb: ShardedVarBuilder,
98        lora_config: &[((String, String), LoraConfig)],
99        xlora_config: Option<XLoraConfig>,
100        xlora_ordering: Ordering,
101        normal_loading_metadata: NormalLoadingMetadata,
102        preload_adapters: &Option<HashMap<String, (ShardedVarBuilder, LoraConfig)>>,
103    ) -> Result<Box<dyn NormalModel + Send + Sync>>;
104    fn is_gptx(&self, config: &str) -> Result<bool>;
105    fn supports_paged_attention(&self, _config: &str) -> Result<bool> {
106        Ok(true)
107    }
108    fn get_config_repr(&self, config: &str) -> Result<Box<dyn Debug>>;
109    fn get_device_for_tensor(
110        &self,
111        config: &str,
112        _mapper: &dyn DeviceMapper,
113        loading_isq: bool,
114    ) -> Result<Arc<dyn Fn(String) -> DeviceForLoadTensor + Send + Sync + 'static>> {
115        if loading_isq {
116            Ok(Arc::new(|_| DeviceForLoadTensor::Base))
117        } else {
118            let re = Regex::new(r"\.layers\.(\d+)\.").unwrap();
119            let num_layers = self.model_config(config)?.num_layers();
120            let closure = move |name: String| {
121                if let Some(captures) = re.captures(&name) {
122                    captures
123                        .get(1)
124                        .and_then(|m| m.as_str().parse::<usize>().ok())
125                        .map(|l| l.min(num_layers))
126                        .map(DeviceForLoadTensor::Idx)
127                        .unwrap_or(DeviceForLoadTensor::Base)
128                } else {
129                    DeviceForLoadTensor::Base
130                }
131            };
132
133            Ok(Arc::new(closure))
134        }
135    }
136}

274            NormalLoaderType::Gemma => Ok(Box::new(GemmaLoader)),
275            NormalLoaderType::Llama => Ok(Box::new(LlamaLoader)),

277            NormalLoaderType::Phi2 => Ok(Box::new(Phi2Loader)),

459impl DeviceMappedModelLoader for MistralLoader {
460    fn mapped_max_act_size_elems(
461        &self,
462        config: &str,
463        params: &AutoDeviceMapParams,
464        prompt_chunksize: usize,
465    ) -> Result<usize> {
466        let AutoDeviceMapParams::Text {
467            max_seq_len: _,
468            max_batch_size,
469        } = params
470        else {
471            anyhow::bail!("Expected text AutoDeviceMapParams for this model!")
472        };
473
474        let cfg: crate::models::mistral::Config = serde_json::from_str(config)?;
475
476        Ok(max_batch_size * cfg.num_attention_heads * prompt_chunksize * prompt_chunksize)
477    }
478    fn non_mapped_max_act_size_elems(
479        &self,
480        _config: &str,
481        _params: &AutoDeviceMapParams,
482    ) -> Result<usize> {
483        Ok(0)
484    }
485
486    fn non_mapped_size_in_bytes(
487        &self,
488        config: &str,
489        dtype: DType,
490        weight_pack_factor: usize,
491    ) -> Result<usize> {
492        let cfg: crate::models::mistral::Config = serde_json::from_str(config)?;
493
494        let elems = {
495            let embed_tokens = cfg.hidden_size * cfg.vocab_size / weight_pack_factor;
496            // If embeddings are tied and no packing, reuse weights -> no separate lm_head needed
497            let lm_head = if !cfg.tie_word_embeddings || weight_pack_factor != 1 {
498                cfg.hidden_size * cfg.vocab_size / weight_pack_factor
499            } else {
500                0
501            };
502            let norm = cfg.hidden_size;
503            embed_tokens + lm_head + norm
504        };
505        Ok(elems * dtype.size_in_bytes())
506    }
507
508    fn layer_sizes_in_bytes(
509        &self,
510        config: &str,
511        dtype: DType,
512        weight_pack_factor: usize,
513    ) -> Result<Vec<usize>> {
514        let cfg: crate::models::mistral::Config = serde_json::from_str(config)?;
515
516        let per_layer_elems = {
517            let input_layernorm = cfg.hidden_size;
518            let post_attention_layernorm = cfg.hidden_size;
519
520            let size_in = cfg.hidden_size;
521            let size_q = (cfg.hidden_size / cfg.num_attention_heads) * cfg.num_attention_heads;
522            let size_kv = (cfg.hidden_size / cfg.num_attention_heads) * cfg.num_key_value_heads;
523            let q_proj = size_in * size_q / weight_pack_factor;
524            let k_proj = size_in * size_kv / weight_pack_factor;
525            let v_proj = size_in * size_kv / weight_pack_factor;
526            let o_proj = size_q * size_in / weight_pack_factor;
527
528            let h_size = cfg.hidden_size;
529            let i_size = cfg.intermediate_size;
530            let gate_proj = h_size * i_size / weight_pack_factor;
531            let up_proj = h_size * i_size / weight_pack_factor;
532            let down_proj = i_size * h_size / weight_pack_factor;
533
534            input_layernorm
535                + post_attention_layernorm
536                + q_proj
537                + k_proj
538                + v_proj
539                + o_proj
540                + gate_proj
541                + up_proj
542                + down_proj
543        };
544        Ok(vec![
545            per_layer_elems * dtype.size_in_bytes();
546            cfg.num_hidden_layers
547        ])
548    }
549
550    fn num_layers(&self, config: &str) -> Result<usize> {
551        let cfg: crate::models::mistral::Config = serde_json::from_str(config)?;
552        Ok(cfg.num_hidden_layers)
553    }
554
555    fn model_config(&self, config: &str) -> Result<Box<dyn ModelConfigLike>> {
556        let cfg: crate::models::mistral::Config = serde_json::from_str(config)?;
557
558        let cfg = ModelConfigMetadata {
559            max_seq_len: cfg.max_position_embeddings,
560            num_layers: cfg.num_hidden_layers,
561            hidden_size: cfg.hidden_size,
562            num_kv_heads: cfg.num_key_value_heads,
563            num_attn_heads: cfg.num_attention_heads,
564            sliding_window: cfg.sliding_window,
565            k_head_dim: cfg.head_dim(),
566            v_head_dim: cfg.head_dim(),
567        };
568
569        Ok(Box::new(cfg))
570    }
571}

2523        &self,

2525        dtype: DType,
2526        weight_pack_factor: usize,
2527    ) -> Result<Vec<usize>> {
2528        let cfg: crate::models::deepseek2::DeepSeekV2Config = serde_json::from_str(config)?;
2529        let mut per_layer_elems = Vec::new();
2530
2531        for layer_idx in 0..cfg.num_hidden_layers {
2532            let input_layernorm = cfg.hidden_size;
2533            let post_attention_layernorm = cfg.hidden_size;
2534
2535            let q_proj = match cfg.q_lora_rank {
2536                Some(lora_rank) => {
2537                    let a = cfg.hidden_size * lora_rank;
2538                    let norm = lora_rank;
2539                    let b = (cfg.num_attention_heads * cfg.q_head_dim()) * lora_rank;

2541                }
2542                None => (cfg.num_attention_heads * cfg.q_head_dim()) * cfg.hidden_size,
2543            };
2544            let kv_a_proj_with_mqa = cfg.hidden_size * (cfg.kv_lora_rank + cfg.qk_rope_head_dim)
2545                / weight_pack_factor
2546                + bias_if!(cfg.attention_bias, cfg.kv_lora_rank + cfg.qk_rope_head_dim);
2547            let kv_a_layernorm = cfg.kv_lora_rank;
2548            let kv_b_proj = cfg.kv_lora_rank
2549                * cfg.num_attention_heads
2550                * (cfg.q_head_dim() - cfg.qk_rope_head_dim + cfg.v_head_dim)
2551                / weight_pack_factor;
2552            let o_proj = cfg.num_attention_heads * cfg.v_head_dim * cfg.hidden_size
2553                / weight_pack_factor
2554                + bias_if!(cfg.attention_bias, cfg.hidden_size);
2555
2556            let moe_block = {
2557                let mut sum = 0;
2558                if cfg.n_routed_experts.is_some()
2559                    && layer_idx >= cfg.first_k_dense_replace
2560                    && layer_idx % cfg.moe_layer_freq == 0
2561                {
2562                    let h_size = cfg.hidden_size;
2563                    let gate_proj = h_size * cfg.moe_intermediate_size / weight_pack_factor
2564                        * cfg.n_routed_experts.unwrap();
2565                    let up_proj = h_size * cfg.moe_intermediate_size / weight_pack_factor
2566                        * cfg.n_routed_experts.unwrap();
2567                    let down_proj = cfg.moe_intermediate_size * h_size / weight_pack_factor
2568                        * cfg.n_routed_experts.unwrap();
2569                    let shared_experts = if let Some(n_shared_experts) = cfg.n_shared_experts {
2570                        let gate_proj = h_size * (cfg.intermediate_size * n_shared_experts)
2571                            / weight_pack_factor;
2572                        let up_proj = h_size * (cfg.intermediate_size * n_shared_experts)
2573                            / weight_pack_factor;
2574                        let down_proj = (cfg.intermediate_size * n_shared_experts) * h_size
2575                            / weight_pack_factor;
2576                        gate_proj + up_proj + down_proj
2577                    } else {
2578                        0
2579                    };
2580                    let gate_weight = cfg.n_routed_experts.unwrap() * cfg.hidden_size;
2581                    sum += gate_proj + up_proj + down_proj + shared_experts + gate_weight;
2582                } else {
2583                    let h_size = cfg.hidden_size;
2584                    let i_size = cfg.intermediate_size;
2585                    let gate_proj = h_size * i_size / weight_pack_factor;
2586                    let up_proj = h_size * i_size / weight_pack_factor;
2587                    let down_proj = i_size * h_size / weight_pack_factor;
2588                    sum += gate_proj + up_proj + down_proj;
2589                }
2590                sum
2591            };
2592
2593            per_layer_elems.push(
2594                input_layernorm
2595                    + post_attention_layernorm
2596                    + q_proj
2597                    + kv_a_layernorm
2598                    + kv_a_proj_with_mqa
2599                    + kv_b_proj
2600                    + o_proj
2601                    + moe_block,
2602            );
2603        }
2604
2605        Ok(per_layer_elems
2606            .into_iter()
2607            .map(|x| x * dtype.size_in_bytes())
2608            .collect())
2609    }

2902                    } else {

2904                    };
2905                    let gate_weight = cfg.n_routed_experts.unwrap() * cfg.hidden_size;
