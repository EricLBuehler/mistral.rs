34/// Parse ISQ value.
35///
36/// If the provided value is a valid integer (one of 2,3,4,5,6,8), the best quantization type will be chosen.
37/// Note that the fallback is always a Q/K quantization but on Metal 2,3,4,6,8 uses the fast AFQ.
38///
39/// One of:
40/// - `Q4_0`
41/// - `Q4_1`
42/// - `Q5_0`
43/// - `Q5_1`
44/// - `Q8_0`
45/// - `Q8_1`
46/// - `Q2K`
47/// - `Q3K`
48/// - `Q4K`
49/// - `Q5K`
50/// - `Q6K`
51/// - `Q8K`
52/// - `HQQ1`
53/// - `HQQ2`
54/// - `HQQ3`
55/// - `HQQ4`
56/// - `HQQ8`
57/// - `AFQ2`
58/// - `AFQ3`
59/// - `AFQ4`
60/// - `AFQ6`
61/// - `AFQ8`
62pub fn parse_isq_value(s: &str, device: Option<&Device>) -> Result<IsqType, String> {
63    let is_metal = device.map(|device| device.is_metal()).unwrap_or(false);
64    let tp = match s.to_lowercase().as_str() {
65        "2" if is_metal => IsqType::AFQ2,
66        "2" if !is_metal => IsqType::Q2K,
67        "3" if is_metal => IsqType::AFQ3,
68        "3" if !is_metal => IsqType::Q3K,
69        "4" if is_metal => IsqType::AFQ4,
70        "4" if !is_metal => IsqType::Q4K,
71        "5" => IsqType::Q5K,
72        "6" if is_metal => IsqType::AFQ6,
73        "6" if !is_metal => IsqType::Q6K,
74        "8" if is_metal => IsqType::AFQ8,
75        "8" if !is_metal => IsqType::Q8_0,
76        "q4_0" => IsqType::Q4_0,
77        "q4_1" => IsqType::Q4_1,
78        "q5_0" => IsqType::Q5_0,
79        "q5_1" => IsqType::Q5_1,
80        "q8_0" => IsqType::Q8_0,
81        "q8_1" => IsqType::Q8_1,
82        "q2k" => IsqType::Q2K,
83        "q3k" => IsqType::Q3K,
84        "q4k" => IsqType::Q4K,
85        "q5k" => IsqType::Q5K,
86        "q6k" => IsqType::Q6K,
87        "q8k" => IsqType::Q8K,
88        "hqq8" => IsqType::HQQ8,
89        "hqq4" => IsqType::HQQ4,
90        "fp8" => IsqType::F8E4M3,
91        "afq8" => IsqType::AFQ8,
92        "afq6" => IsqType::AFQ6,
93        "afq4" => IsqType::AFQ4,
94        "afq3" => IsqType::AFQ3,
95        "afq2" => IsqType::AFQ2,
96        // "hqq3" => IsqType::HQQ3,
97        // "hqq2" => IsqType::HQQ2,
98        // "hqq1" => IsqType::HQQ1,
99        _ => return Err(format!("ISQ type {s} unknown, choose one of `2`, `3`, `4`, `6`, `8`, `Q4_0`, `Q4_1`, `Q5_0`, `Q5_1`, `Q8_0`, `Q8_1`, `Q2K`, `Q3K`, `Q4K`, `Q5K`, `Q6K`, `Q8K`, `HQQ8`, `HQQ4`, `FP8`, `AFQ8`, `AFQ6`, `AFQ4`, `AFQ3`, `AFQ2`.")),
100    };
101    #[cfg(feature = "cuda")]
102    {
103        if !matches!(
104            tp,
105            IsqType::Q4_0
106                | IsqType::Q4_1
107                | IsqType::Q5_0
108                | IsqType::Q5_1
109                | IsqType::Q8_0
110                | IsqType::Q2K
111                | IsqType::Q3K
112                | IsqType::Q4K
113                | IsqType::Q5K
114                | IsqType::Q6K
115                | IsqType::HQQ8
116                | IsqType::HQQ4
117                | IsqType::F8E4M3 // | IsqType::HQQ3
118                                  // | IsqType::HQQ2
119                                  // | IsqType::HQQ1
120        ) {
121            return Err("ISQ type on CUDA must be one of `Q4_0`, `Q4_1`, `Q5_0`, `Q5_1`, `Q8_0`, `Q2K`, `Q3K`, `Q4K`, `Q5K`, `Q6K`, `HQQ8`, `HQQ4`, `FP8`".to_string());
122        }
123    }
124    Ok(tp)
125}

127#[derive(Clone, Debug, Copy, Default, Deserialize)]
128pub enum IsqOrganization {
129    #[default]
130    #[serde(rename = "default")]
131    Default,
132    /// Only quantize MoE experts, if applicable. The enables MoQE.
133    /// <https://arxiv.org/abs/2310.02410>
134    #[serde(rename = "moqe")]
135    MoeExpertsOnly,
136}
137
138impl FromStr for IsqOrganization {
139    type Err = String;
140    fn from_str(s: &str) -> Result<Self, Self::Err> {
141        match s {
142            "default" => Ok(Self::Default),
143            "moqe" => Ok(Self::MoeExpertsOnly),
144            other => Err(format!(
145                "Expected ISQ organization `default` or `moqe`, got `{other}`"
146            )),
147        }
148    }
149}

166pub trait IsqModel {
167    /// Corresponds to `IsqOrganization::Default`
168    #[allow(clippy::type_complexity)]

176    /// This is used for imatrix generation internally. Begin stats tracking.
177    fn begin_track_stats(&mut self) -> anyhow::Result<()> {
178        let layers = self
179            .get_layers()
180            .0
181            .into_iter()
182            .map(|(layer, _)| layer)
183            .collect::<Vec<_>>();
184        for layer in layers {
185            Arc::get_mut(layer).unwrap().begin_track_stats()?;
186        }
187        Ok(())
188    }
189
190    /// End stats tracking and return the imatrix data
191    fn extract_imatrix_data(&mut self) -> candle_core::Result<CollectedImatrixData> {
192        let layers = self
193            .get_layers()
194            .0
195            .into_iter()
196            .enumerate()
197            .map(|(i, (layer, _))| (i, layer))
198            .collect::<Vec<_>>();
199        let mut data = HashMap::new();
200        for (i, layer) in layers {
201            data.insert(i, Some(layer.end_track_stats()?.to_vec1::<f32>()?));
202        }
203        Ok(CollectedImatrixData(data))
204    }

207    /// https://arxiv.org/abs/2310.02410
208    #[allow(clippy::type_complexity)]
209    fn get_layers_moe_experts_only(
210        &mut self,
211    ) -> (
212        Vec<(&mut Arc<dyn QuantMethod>, Option<usize>)>,
213        &dyn DeviceMapper,
214    ) {
215        self.get_layers()
216    }

271    /// Quantize the model in-situ.
272    ///
273    /// This function will also create a UQFF file, or, if the model supports it (residual tensors are returned),
274    /// a full serialization is created.
275    #[allow(clippy::too_many_arguments)]
276    fn quantize(
277        &mut self,
278        dtype: Option<IsqType>,
279        device: Device,
280        topology: Option<&Topology>,
281        silent: bool,
282        imatrix_source: Option<ImatrixDataSource<'_>>,
283        organization: IsqOrganization,
284        write_artifacts: Option<&PathBuf>,
285        full_ser: UqffFullSer<'_>,
286        multi_progress: Arc<MultiProgress>,
287    ) -> candle_core::Result<()> {
288        {
289            let imatrix_to_weight = match imatrix_source {
290                Some(ImatrixDataSource::File(imatrix)) => {
291                    let ext = imatrix.extension().ok_or(candle_core::Error::msg(
292                        "Expected an extension for the imatrix source file.",
293                    ))?;
294                    if ext == "cimatrix" {
295                        info!(
296                            "Loading collected imatrix source file: `{}`",
297                            imatrix.display()
298                        );
299                        Some(CollectedImatrixData::load_imatrix(imatrix)?.0)
300                    } else if ext == "imatrix" {
301                        info!(
302                            "Loading GGUF-format imatrix source file: `{}`",
303                            imatrix.display()
304                        );
305                        let mut imatrix_data =
306                            quantized::imatrix_file::load_imatrix(imatrix.clone())?;
307                        let imatrix_mapping = self
308                            .imatrix_names()?
309                            .into_iter()
310                            .enumerate()
311                            .collect::<HashMap<_, _>>();
312
313                        let layer_to_weight = imatrix_mapping
314                            .into_iter()
315                            .map(|(i, name)| {
316                                if let Some(name) = name {
317                                    (i, Some(imatrix_data.remove(&name).unwrap()))
318                                } else {
319                                    (i, None)
320                                }
321                            })
322                            .collect::<HashMap<_, _>>();
323                        info!(
324                            "Quantizing with imatrix file `{}`, {} imatrix weights",
325                            imatrix.display(),
326                            layer_to_weight.iter().filter(|(_, x)| x.is_some()).count()
327                        );
328                        Some(layer_to_weight)
329                    } else {
330                        warn!("Imatrix source file extension is {ext:?}, expected .imatrix/.cimatrix. Assuming GGUF specification");
331                        info!(
332                            "Loading GGUF-format imatrix source file: `{}`",
333                            imatrix.display()
334                        );
335
336                        let mut imatrix_data =
337                            quantized::imatrix_file::load_imatrix(imatrix.clone())?;
338                        let imatrix_mapping = self
339                            .imatrix_names()?
340                            .into_iter()
341                            .enumerate()
342                            .collect::<HashMap<_, _>>();
343
344                        let layer_to_weight = imatrix_mapping
345                            .into_iter()
346                            .map(|(i, name)| {
347                                if let Some(name) = name {
348                                    (i, Some(imatrix_data.remove(&name).unwrap()))
349                                } else {
350                                    (i, None)
351                                }
352                            })
353                            .collect::<HashMap<_, _>>();
354                        info!(
355                            "Quantizing with imatrix file `{}`, {} imatrix weights",
356                            imatrix.display(),
357                            layer_to_weight.iter().filter(|(_, x)| x.is_some()).count()
358                        );
359                        Some(layer_to_weight)
360                    }
361                }
362                Some(ImatrixDataSource::Collected) => {
363                    let data = match organization {
364                        IsqOrganization::Default => self.extract_imatrix_data()?,
365                        IsqOrganization::MoeExpertsOnly => {
366                            self.extract_imatrix_data_moe_experts_only()?
367                        }
368                    };
369                    // Save the collected imatrix data so users can reuse it
370                    let count = data.0.iter().filter(|(_, x)| x.is_some()).count();
371                    let save_path = format!("collected-{count}.cimatrix");
372                    info!("Saving collected imatrix data to `{save_path}`");
373                    data.save_imatrix(save_path)?;
374                    info!("Quantizing with collected imatrix data, {count} imatrix weights");
375                    Some(data.0)
376                }
377                None => {
378                    // Dummy, just for zip
379                    None
380                }
381            };
382
383            let (mut tensors, mapper) = match organization {
384                IsqOrganization::Default => self.get_layers(),
385                IsqOrganization::MoeExpertsOnly => self.get_layers_moe_experts_only(),
386            };
387
388            let imatrix_to_weight: Vec<Option<Vec<f32>>> =
389                if let Some(mut imatrix_to_weight) = imatrix_to_weight {
390                    let ordered_keys = imatrix_to_weight
391                        .keys()
392                        .copied()
393                        .sorted()
394                        .collect::<Vec<_>>();
395                    ordered_keys
396                        .into_iter()
397                        .map(|layer| imatrix_to_weight.remove(&layer).unwrap())
398                        .collect()
399                } else {
400                    vec![None; tensors.len()]
401                };
402
403            let total_tensors = tensors.len();
404            let n_quantized = AtomicUsize::new(0);
405            if let Some(topology) = topology {
406                let mut dtypes = HashSet::new();
407                for layer in topology.0.iter().flatten() {
408                    if let LayerTopology {
409                        isq: Some(isq_dtype),
410                        device: _,
411                    } = layer
412                    {
413                        dtypes.insert(isq_dtype);
414                    }
415                }
416                info!("Applying in-situ quantization into {:?} to {total_tensors} tensors according to topology.", dtypes.into_iter().collect::<Vec<_>>());
417            } else {
418                info!("Applying in-situ quantization into {dtype:?} to {total_tensors} tensors.");
419            }
420            let bar = ProgressBar::new(total_tensors as u64);
421            bar.set_style(
422                ProgressStyle::default_bar()
423                    .template("[{elapsed_precise}] [{bar:40.cyan/blue}] {pos}/{len} ({eta})")
424                    .unwrap()
425                    .progress_chars("#>-"),
426            );
427            multi_progress.add(bar.clone());
428
429            let layers = topology.map(|x| {
430                x.0.iter()
431                    .filter_map(|topo| topo.as_ref().map(|x| (x.isq, x.device.clone())))
432                    .collect::<Vec<_>>()
433            });
434
435            let mut devices_and_dtypes = Vec::new();
436            for (_, layer_num) in &tensors {
437                let device = if let Some(ref layers) = layers {
438                    if let Some(layer) = layer_num {
439                        layers
440                            .get(*layer)
441                            .as_ref()
442                            .map(|x| x.1.clone())
443                            .unwrap_or(Some(device.clone()))
444                            .unwrap_or(device.clone())
445                    } else {
446                        device.clone()
447                    }
448                } else if let Some(layer_num) = layer_num {
449                    mapper
450                        .device_for(*layer_num, false)
451                        .cloned()
452                        .unwrap_or(device.clone())
453                } else {
454                    device.clone()
455                };
456                let dtype = if let Some(ref layers) = layers {
457                    if let Some(layer) = layer_num {
458                        layers.get(*layer).cloned().map(|x| x.0).unwrap_or(dtype)
459                    } else {
460                        dtype
461                    }
462                } else {
463                    dtype
464                };
465                devices_and_dtypes.push((device, dtype));
466            }
467
468            let t_start = Instant::now();
469
470            use rayon::iter::IntoParallelRefIterator;
471
472            // Get the MINIMUM of the max isq threads the quant method
473            let mut minimum_max_threads = {
474                let current_rayon_threads = rayon::current_num_threads();
475                if let Some(dtype) = dtype {
476                    dtype
477                        .get_max_isq_cpu_threads()
478                        .map(usize::from)
479                        .unwrap_or(current_rayon_threads)
480                } else {
481                    current_rayon_threads
482                }
483            };
484            if env::var("MISTRALRS_ISQ_SINGLETHREAD").is_ok() {
485                minimum_max_threads = 1;
486            }
487
488            if matches!(imatrix_source, Some(ImatrixDataSource::Collected)) {
489                // Collected imatrix means that the model is potentially on the gpu already
490                minimum_max_threads = 1;
491            }
492
493            info!("Applying ISQ on {minimum_max_threads} threads.");
494
495            let pool = rayon::ThreadPoolBuilder::new()
496                .num_threads(minimum_max_threads)
497                .build()
498                .map_err(candle_core::Error::msg)?;
499
500            let guard = QuantizeOntoGuard::new();
501
502            pool.install(|| {
503                use indicatif::ParallelProgressIterator;
504                use rayon::iter::{
505                    IndexedParallelIterator, IntoParallelRefMutIterator, ParallelIterator,
506                };
507                if silent {
508                    tensors
509                        .par_iter_mut()
510                        .zip(devices_and_dtypes)
511                        .zip(imatrix_to_weight)
512                        .for_each(|(((tensor, _), (device, dtype)), imatrix_weight)| {
513                            **tensor = tensor
514                                .clone()
515                                .apply_isq(
516                                    dtype,
517                                    device.clone(),
518                                    &n_quantized,
519                                    imatrix_weight,
520                                    guard.clone(),
521                                )
522                                .unwrap();
523                            device.synchronize().unwrap();
524                        });
525                } else {
526                    tensors
527                        .par_iter_mut()
528                        .zip(devices_and_dtypes)
529                        .zip(imatrix_to_weight)
530                        .progress_with(bar)
531                        .for_each(|(((tensor, _), (device, dtype)), imatrix_weight)| {
532                            **tensor = tensor
533                                .clone()
534                                .apply_isq(
535                                    dtype,
536                                    device.clone(),
537                                    &n_quantized,
538                                    imatrix_weight,
539                                    guard.clone(),
540                                )
541                                .unwrap();
542                            device.synchronize().unwrap();
543                        });
544                }
545            });
546
547            if let Some(serialized) = write_artifacts {
548                info!(
549                    "Serializing {total_tensors} ISQ tensors to `{}`.",
550                    serialized.display()
551                );
552
553                if serialized.extension().is_none_or(|ext| ext != "uqff") {
554                    candle_core::bail!("UQFF output path extension must be `.uqff`",);
555                }
556
557                let bar = ProgressBar::new(total_tensors as u64);
558                bar.set_style(
559                    ProgressStyle::default_bar()
560                        .template("[{elapsed_precise}] [{bar:40.red/magenta}] {pos}/{len} ({eta})")
561                        .unwrap()
562                        .progress_chars("#>-"),
563                );
564
565                #[cfg(not(feature = "metal"))]
566                let n_threads = 2;
567                #[cfg(feature = "metal")]
568                let n_threads = 1;
569
570                let pool = rayon::ThreadPoolBuilder::new()
571                    .num_threads(n_threads)
572                    .build()
573                    .map_err(candle_core::Error::msg)?;
574
575                let quantized_values = pool.install(|| {
576                    if silent {
577                        tensors
578                            .par_iter()
579                            .enumerate()
580                            .filter(|(_, (layer, _))| layer.isq_serde_supported())
581                            .map(|(i, (layer, _))| {
582                                Ok((
583                                    i.to_string(),
584                                    Tensor::new(Cow::into_owned(layer.serialize()?), &Device::Cpu)?,
585                                ))
586                            })
587                            .collect::<candle_core::Result<Vec<_>>>()
588                    } else {
589                        tensors
590                            .par_iter()
591                            .enumerate()
592                            .progress_with(bar)
593                            .filter(|(_, (layer, _))| layer.isq_serde_supported())
594                            .map(|(i, (layer, _))| {
595                                Ok((
596                                    i.to_string(),
597                                    Tensor::new(Cow::into_owned(layer.serialize()?), &Device::Cpu)?,
598                                ))
599                            })
600                            .collect::<candle_core::Result<Vec<_>>>()
601                    }
602                });
603                let quantized_values = quantized_values?;
604
605                let parent = serialized
606                    .parent()
607                    .context("Target UQFF path must have a filename!")?;
608
609                std::fs::create_dir_all(parent)?;
610
611                let file_stem = serialized
612                    .file_stem()
613                    .context("Target UQFF path must have a file stem!")?
614                    .to_string_lossy()
615                    .to_string();
616
617                let size_estimate_bytes = quantized_values
618                    .iter()
619                    .map(|(_, x)| x.elem_count() * x.dtype().size_in_bytes())
620                    .sum::<usize>();
621                let n_files = size_estimate_bytes.div_ceil(MAX_UQFF_SIZE_BYTES);
622
623                if n_files == 1 {
624                    info!("Writing to `{}`", serialized.display());
625                    safetensors::serialize_to_file(quantized_values, &None, serialized)?;
626                } else {
627                    let chunksize = quantized_values.len() / n_files;
628                    let quantized_values_chunks = quantized_values.into_iter().chunks(chunksize);
629                    for (i, chunk) in quantized_values_chunks.into_iter().enumerate() {
630                        let mut name = parent.to_path_buf();
631                        name.push(format!("{file_stem}-{i}.uqff"));
632                        info!("Writing shard {i} to `{}`", name.display());
633                        safetensors::serialize_to_file(chunk, &None, &name)?;
634                    }
635                }
636
637                let residual = match organization {
638                    IsqOrganization::Default => self.residual_tensors(),
639                    IsqOrganization::MoeExpertsOnly => self
640                        .residual_tensors_moe_experts_only()
641                        .unwrap_or(self.residual_tensors()),
642                };
643
644                let residual_out = parent.join(UQFF_RESIDUAL_SAFETENSORS);
645                let config_out = parent.join("config.json");
646                let tokenizer_out = parent.join("tokenizer.json");
647                let tokenizer_cfg_out = parent.join("tokenizer_config.json");
648                let gen_cfg_out = parent.join("generation_config.json");
649                let processor_out = parent.join("processor_config.json");
650                let preprocessor_out = parent.join("preprocessor_config.json");
651
652                info!(
653                    "Serializing {} residual tensors to `{}`.",
654                    residual.len(),
655                    residual_out.display()
656                );
657
658                safetensors::serialize_to_file(residual, &None, &residual_out)?;
659
660                let UqffFullSer {
661                    tokenizer,
662                    template_filename,
663                    generation_config,
664                    config,
665                    processor_filename,
666                    preprocessor_filename,
667                } = full_ser;
668
669                info!("Serializing configuration to `{}`.", config_out.display());
670
671                std::fs::write(config_out, config)?;
672
673                info!("Serializing tokenizer to `{}`.", tokenizer_out.display());
674
675                serde_json::to_writer_pretty(File::create(&tokenizer_out)?, tokenizer)
676                    .map_err(candle_core::Error::msg)?;
677
678                if let Some(template_filename) = template_filename {
679                    info!(
680                        "Serializing tokenizer config to `{}`.",
681                        tokenizer_cfg_out.display()
682                    );
683
684                    let template =
685                        std::fs::read(template_filename).map_err(candle_core::Error::msg)?;
686                    std::fs::write(&tokenizer_cfg_out, template)
687                        .map_err(candle_core::Error::msg)?;
688                }
689
690                if let Some(generation_config) = generation_config {
691                    info!(
692                        "Serializing generation config to `{}`.",
693                        gen_cfg_out.display()
694                    );
695
696                    let cfg = std::fs::read(generation_config).map_err(candle_core::Error::msg)?;
697                    std::fs::write(&gen_cfg_out, cfg).map_err(candle_core::Error::msg)?;
698                }
699
700                if let Some(processor_config) = processor_filename {
701                    info!(
702                        "Serializing processor config to `{}`.",
703                        processor_out.display()
704                    );
705
706                    let cfg = std::fs::read(processor_config).map_err(candle_core::Error::msg)?;
707                    std::fs::write(&processor_out, cfg).map_err(candle_core::Error::msg)?;
708                }
709
710                if let Some(preprocessor_config) = preprocessor_filename {
711                    info!(
712                        "Serializing preprocessor config to `{}`.",
713                        preprocessor_out.display()
714                    );
715
716                    let cfg =
717                        std::fs::read(preprocessor_config).map_err(candle_core::Error::msg)?;
718                    std::fs::write(&preprocessor_out, cfg).map_err(candle_core::Error::msg)?;
719                }
720            }
721            let delta = Instant::now().duration_since(t_start).as_secs_f32();
722            info!("Applied in-situ quantization into {dtype:?} to {n_quantized:?} tensors out of {total_tensors} total tensors. Took {delta:.2}s", );
723        }
724        Ok(())
725    }

727    fn load_from_artifacts(
728        &mut self,
729        device: Device,
730        topology: Option<&Topology>,
731        silent: bool,
732        artifacts: &[PathBuf],
733    ) -> candle_core::Result<()> {
734        let (tensors, mapper) = self.get_layers();
735        let total_tensors = tensors.len();
736
737        let layers = topology.map(|x| {
738            x.0.iter()
739                .filter_map(|topo| topo.as_ref().map(|x| (x.isq, x.device.clone())))
740                .collect::<Vec<_>>()
741        });
742
743        let mut devices = Vec::new();
744        let mut comms = Vec::new();
745        for (_, layer_num) in &tensors {
746            let device = if let Some(ref layers) = layers {
747                if let Some(layer) = layer_num {
748                    layers
749                        .get(*layer)
750                        .as_ref()
751                        .map(|x| x.1.clone())
752                        .unwrap_or(Some(device.clone()))
753                        .unwrap_or(device.clone())
754                } else {
755                    device.clone()
756                }
757            } else if let Some(layer_num) = layer_num {
758                mapper
759                    .device_for(*layer_num, false)
760                    .cloned()
761                    .unwrap_or(device.clone())
762            } else {
763                device.clone()
764            };
765            devices.push(device);
766            comms.push(mapper.get_comm_for(layer_num.unwrap_or(0))?)
767        }
768
769        let artifacts = unsafe { candle_core::safetensors::MmapedSafetensors::multi(artifacts)? };
770
771        let artifact_isqs = artifacts
772            .tensors()
773            .into_iter()
774            .map(|(name, tensor)| {
775                (
776                    name.parse::<usize>()
777                        .expect("Name should be parseable as usize"),
778                    tensor,
779                )
780            })
781            .collect::<HashMap<_, _>>();
782
783        if artifact_isqs.len() != total_tensors {
784            candle_core::bail!(
785                "Number of artifacts ({}) does not match the number of ISQ layers ({total_tensors})",
786                artifact_isqs.len(),
787            );
788        }
789
790        let bar = ProgressBar::new(total_tensors as u64);
791        bar.set_style(
792            ProgressStyle::default_bar()
793                .template("[{elapsed_precise}] [{bar:40.red/magenta}] {pos}/{len} ({eta})")
794                .unwrap()
795                .progress_chars("#>-"),
796        );
797
798        let t_start = Instant::now();
799
800        let guard = QuantizeOntoGuard::new();
801
802        if silent {
803            (0..tensors.len())
804                .into_par_iter()
805                .zip(tensors)
806                .map(|(i, (tensor, _))| {
807                    if let Some(artifact) = artifact_isqs.get(&i) {
808                        let artifact = artifact.data();
809
810                        let comm = comms[i].clone();
811                        let deserialized = match tensor.is_distributed() {
812                            Some(DistributedKind::ColumnParallel) => {
813                                ColumnParallelLayer::deserialize(
814                                    Cow::from(artifact),
815                                    &devices[i],
816                                    &comm,
817                                    guard.clone(),
818                                )?
819                            }
820                            Some(DistributedKind::RowParallel) => RowParallelLayer::deserialize(
821                                Cow::from(artifact),
822                                &devices[i],
823                                &comm,
824                                guard.clone(),
825                            )?,
826                            Some(DistributedKind::Replicated) => ReplicatedLayer::deserialize(
827                                Cow::from(artifact),
828                                &devices[i],
829                                &comm,
830                                guard.clone(),
831                            )?,
832                            None => {
833                                // NOTE(EricLBuehler): isq type is ALWAYS byte 4 (5th) of the tensor.
834                                let isq_type = artifact[mistralrs_quant::UQFF_QUANT_TYPE_OFFSET];
835                                match QuantizedSerdeType::try_from(isq_type as usize)? {
836                                    QuantizedSerdeType::Gguf => GgufMatMul::deserialize(
837                                        Cow::from(artifact),
838                                        &devices[i],
839                                        &comm,
840                                        guard.clone(),
841                                    )?,
842                                    QuantizedSerdeType::Unquant => UnquantLinear::deserialize(
843                                        Cow::from(artifact),
844                                        &devices[i],
845                                        &comm,
846                                        guard.clone(),
847                                    )?,
848                                    QuantizedSerdeType::Hqq => HqqLayer::deserialize(
849                                        Cow::from(artifact),
850                                        &devices[i],
851                                        &comm,
852                                        guard.clone(),
853                                    )?,
854                                    QuantizedSerdeType::Fp8 => FP8Linear::deserialize(
855                                        Cow::from(artifact),
856                                        &devices[i],
857                                        &comm,
858                                        guard.clone(),
859                                    )?,
860                                    QuantizedSerdeType::Afq => AfqLayer::deserialize(
861                                        Cow::from(artifact),
862                                        &devices[i],
863                                        &comm,
864                                        guard.clone(),
865                                    )?,
866                                }
867                            }
868                        };
869                        *tensor = deserialized;
870                    }
871                    Ok(())
872                })
873                .collect::<candle_core::Result<Vec<_>>>()?;
874        } else {
875            (0..tensors.len())
876                .into_par_iter()
877                .zip(tensors)
878                .progress_with(bar)
879                .map(|(i, (tensor, _))| {
880                    if let Some(artifact) = artifact_isqs.get(&i) {
881                        let artifact = artifact.data();
882
883                        let comm = comms[i].clone();
884                        let deserialized = match tensor.is_distributed() {
885                            Some(DistributedKind::ColumnParallel) => {
886                                ColumnParallelLayer::deserialize(
887                                    Cow::from(artifact),
888                                    &devices[i],
889                                    &comm,
890                                    guard.clone(),
891                                )?
892                            }
893                            Some(DistributedKind::RowParallel) => RowParallelLayer::deserialize(
894                                Cow::from(artifact),
895                                &devices[i],
896                                &comm,
897                                guard.clone(),
898                            )?,
899                            Some(DistributedKind::Replicated) => ReplicatedLayer::deserialize(
900                                Cow::from(artifact),
901                                &devices[i],
902                                &comm,
903                                guard.clone(),
904                            )?,
905                            None => {
906                                // NOTE(EricLBuehler): isq type is ALWAYS byte 4 (5th) of the tensor.
907                                let isq_type = artifact[mistralrs_quant::UQFF_QUANT_TYPE_OFFSET];
908                                match QuantizedSerdeType::try_from(isq_type as usize)? {
909                                    QuantizedSerdeType::Gguf => GgufMatMul::deserialize(
910                                        Cow::from(artifact),
911                                        &devices[i],
912                                        &comm,
913                                        guard.clone(),
914                                    )?,
915                                    QuantizedSerdeType::Unquant => UnquantLinear::deserialize(
916                                        Cow::from(artifact),
917                                        &devices[i],
918                                        &comm,
919                                        guard.clone(),
920                                    )?,
921                                    QuantizedSerdeType::Hqq => HqqLayer::deserialize(
922                                        Cow::from(artifact),
923                                        &devices[i],
924                                        &comm,
925                                        guard.clone(),
926                                    )?,
927                                    QuantizedSerdeType::Fp8 => FP8Linear::deserialize(
928                                        Cow::from(artifact),
929                                        &devices[i],
930                                        &comm,
931                                        guard.clone(),
932                                    )?,
933                                    QuantizedSerdeType::Afq => AfqLayer::deserialize(
934                                        Cow::from(artifact),
935                                        &devices[i],
936                                        &comm,
937                                        guard.clone(),
938                                    )?,
939                                }
940                            }
941                        };
942                        *tensor = deserialized;
943                    }
944                    Ok(())
945                })
946                .collect::<candle_core::Result<Vec<_>>>()?;
947        }
948
949        let delta = Instant::now().duration_since(t_start).as_secs_f32();
950        info!("Loaded in-situ quantization artifacts into {total_tensors} total tensors. Took {delta:.2}s", );
951
952        Ok(())
953    }
