79/// Helper function to extract token string from BeginEndUnkPadTok
80fn extract_token_string(token: &super::chat_template::BeginEndUnkPadTok) -> String {
81    match &token.0 {
82        Either::Left(lit) => lit.clone(),
83        Either::Right(added) => added.content.clone(),
84    }
85}

87/// Flatten a content field to extract only text from structured content
88fn flatten_content(content: MessageContent) -> MessageContent {
89    match content {
90        Either::Left(_) => content,
91        Either::Right(content_rows) => {
92            // Find the first "text" field in the content rows
93            content_rows
94                .into_iter()
95                .find_map(|content_row| {
96                    content_row
97                        .get("text")
98                        .and_then(|v| v.as_str())
99                        .map(|s| Either::Left(s.to_string()))
100                })
101                .unwrap_or(Either::Right(Vec::new()))
102        }
103    }
104}

106pub(crate) fn apply_chat_template(
107    pipeline: &dyn Pipeline,
108    messages: Vec<IndexMap<String, MessageContent>>,
109    add_generation_prompt: bool,
110    enable_thinking: Option<bool>,
111    reasoning_effort: Option<ReasoningEffort>,
112    action: MessagesAction,
113    tools: Vec<Tool>,
114) -> Result<String> {
115    let messages = match action {
116        MessagesAction::Keep => messages,
117        MessagesAction::FlattenOnlyText => {
118            // This is really only for image models. If they need to flatten it s.t. they only see
119            // the text, do that.
120            messages
121                .into_iter()
122                .map(|message| {
123                    message
124                        .into_iter()
125                        .map(|(key, value)| {
126                            let new_value = if key == "content" {
127                                flatten_content(value)
128                            } else {
129                                value
130                            };
131                            (key, new_value)
132                        })
133                        .collect()
134                })
135                .collect()
136        }
137    };
138
139    let chat_template = pipeline
140        .get_chat_template()
141        .with_context(|| "`apply_chat_template` expects the pipeline to have a chat template.")?;
142    let template = chat_template.chat_template.as_ref().unwrap();
143
144    let bos_tok = chat_template.bos_token.as_ref().map(extract_token_string);
145    let eos_tok = chat_template.eos_token.as_ref().map(extract_token_string);
146    let unk_tok = chat_template.unk_token.as_ref().map(extract_token_string);
147
148    apply_chat_template_to(
149        messages,
150        add_generation_prompt,
151        enable_thinking,
152        reasoning_effort,
153        template,
154        bos_tok,
155        eos_tok,
156        unk_tok,
157        tools,
158    )
159}
