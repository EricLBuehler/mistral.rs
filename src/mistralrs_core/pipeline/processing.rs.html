75pub(crate) fn apply_chat_template(
76    pipeline: &dyn Pipeline,
77    messages: Vec<IndexMap<String, MessageContent>>,
78    add_generation_prompt: bool,
79    enable_thinking: Option<bool>,
80    action: MessagesAction,
81    tools: Vec<Tool>,
82) -> Result<String> {
83    let messages = match action {
84        MessagesAction::Keep => messages,
85        MessagesAction::FlattenOnlyText => {
86            // This is really only for image models. If they need to flatten it s.t. they only see
87            // the text, do that.
88            let mut new_messages = Vec::new();
89            for message in messages {
90                let mut new_message = IndexMap::new();
91                for (k, v) in message {
92                    if k == "content" {
93                        match v {
94                            Either::Left(lv) => {
95                                new_message.insert(k, Either::Left(lv));
96                            }
97                            Either::Right(rv) => {
98                                'outer: for content_row in rv {
99                                    for (content_k, content_v) in content_row {
100                                        if content_k == "text" {
101                                            if let Some(content_str) = content_v.as_str() {
102                                                new_message.insert(
103                                                    k,
104                                                    Either::Left(content_str.to_string()),
105                                                );
106                                                break 'outer;
107                                            }
108                                        }
109                                    }
110                                }
111                            }
112                        }
113                    } else {
114                        new_message.insert(k, Either::Left(v.left().unwrap()));
115                    }
116                }
117                new_messages.push(new_message)
118            }
119            new_messages
120        }
121    };
122    let chat_template = pipeline
123        .get_chat_template()
124        .with_context(|| "`apply_chat_template` expects the pipeline to have a chat template.")?;
125    let template = chat_template.chat_template.as_ref().unwrap();
126    let bos_tok = if let Some(ref bos) = chat_template.bos_token {
127        match bos.0 {
128            Either::Left(ref lit) => Some(lit.to_string()),
129            Either::Right(ref added) => Some(added.content.to_string()),
130        }
131    } else {
132        None
133    };
134    let eos_tok = if let Some(ref eos) = chat_template.eos_token {
135        match eos.0 {
136            Either::Left(ref lit) => Some(lit.to_string()),
137            Either::Right(ref added) => Some(added.content.to_string()),
138        }
139    } else {
140        None
141    };
142    let unk_tok = if let Some(ref unk) = chat_template.unk_token {
143        match unk.0 {
144            Either::Left(ref lit) => Some(lit.to_string()),
145            Either::Right(ref added) => Some(added.content.to_string()),
146        }
147    } else {
148        None
149    };
150    apply_chat_template_to(
151        messages,
152        add_generation_prompt,
153        enable_thinking,
154        template,
155        bos_tok,
156        eos_tok,
157        unk_tok,
158        tools,
159    )
160}
