31/// Check if MLA decode should be used for single-token generation.
32///
33/// MLA decode is used when:
34/// - `MISTRALRS_NO_MLA` is not set to "1"
35/// - No attention mask (single-token decode)
36/// - Sequence length is 1
37/// - Paged attention is enabled
38/// - Running on CUDA
39/// - Paged KV indptr metadata is available
40#[cfg(all(feature = "cuda", target_family = "unix"))]
41pub fn should_use_mla_decode(
42    attention_mask: Option<&Tensor>,
43    seq_len: usize,
44    paged_attn_enabled: bool,
45    device: &Device,
46    metadata: &Option<((Tensor, Tensor), &PagedAttentionInputMetadata)>,
47) -> bool {
48    !is_mla_disabled()
49        && attention_mask.is_none()
50        && seq_len == 1
51        && paged_attn_enabled
52        && matches!(device, Device::Cuda(_))
53        && metadata
54            .as_ref()
55            .and_then(|(_, meta)| meta.paged_kv_indptr.as_ref())
56            .is_some()
57}

86/// MLA decode forward pass for single-token generation.
87///
88/// Uses FlashInfer MLA kernels for efficient decode with latent attention.
89///
90/// # Arguments
91/// * `q_nope` - Non-positional query tensor [bs, num_heads, seq_len, qk_nope_head_dim]
92/// * `q_pe` - Positional query tensor [bs, num_heads, seq_len, qk_rope_head_dim]
93/// * `ckv` - Compressed KV tensor [bs, seq_len, kv_lora_rank]
94/// * `k_pe` - Positional key tensor [bs, 1, seq_len, qk_rope_head_dim]
95/// * `metadata` - PagedAttention cache and metadata
96/// * `mla_weights` - Cached MLA weight matrices
97/// * `kv_b_proj` - KV B projection layer for weight computation
98/// * `sdpa_params` - SDPA parameters including softmax scale
99/// * `num_attention_heads` - Number of attention heads
100/// * `kv_lora_rank` - KV latent dimension
101/// * `qk_rope_head_dim` - Rope head dimension
102/// * `qk_nope_head_dim` - Non-positional head dimension
103/// * `v_head_dim` - Value head dimension
104/// * `bs` - Batch size
105/// * `seq_len` - Sequence length (should be 1)
106#[cfg(all(feature = "cuda", target_family = "unix"))]
107#[allow(clippy::too_many_arguments)]
108pub fn mla_decode_forward(
109    q_nope: &Tensor,
110    q_pe: &Tensor,
111    ckv: &Tensor,
112    k_pe: &Tensor,
113    metadata: &Option<((Tensor, Tensor), &PagedAttentionInputMetadata)>,
114    mla_weights: &MlaWeights,
115    kv_b_proj: &dyn mistralrs_quant::QuantMethod,
116    sdpa_params: &SdpaParams,
117    num_attention_heads: usize,
118    kv_lora_rank: usize,
119    qk_rope_head_dim: usize,
120    qk_nope_head_dim: usize,
121    v_head_dim: usize,
122    bs: usize,
123    seq_len: usize,
124) -> Result<Tensor> {
125    let ((key_cache, value_cache), input_metadata) = metadata
126        .as_ref()
127        .ok_or_else(|| candle_core::Error::msg("paged attention metadata missing"))?;
128    let device_location = q_nope.device().location();
129    let slot_mapping = input_metadata
130        .slot_mappings
131        .get(&device_location)
132        .ok_or_else(|| candle_core::Error::msg("slot mapping missing"))?;
133    let slot_mapping = if slot_mapping.dims().len() > 1 {
134        slot_mapping.flatten(0, slot_mapping.dims().len())?
135    } else {
136        slot_mapping.clone()
137    };
138    let paged_kv_indptr = input_metadata
139        .paged_kv_indptr
140        .as_ref()
141        .and_then(|m| m.get(&device_location))
142        .ok_or_else(|| candle_core::Error::msg("paged_kv_indptr missing"))?;
143    let paged_kv_indices = input_metadata
144        .paged_kv_indices
145        .as_ref()
146        .and_then(|m| m.get(&device_location))
147        .ok_or_else(|| candle_core::Error::msg("paged_kv_indices missing"))?;
148    let paged_kv_last_page_len = input_metadata
149        .paged_kv_last_page_len
150        .as_ref()
151        .and_then(|m| m.get(&device_location))
152        .ok_or_else(|| candle_core::Error::msg("paged_kv_last_page_len missing"))?;
153    let paged_kv_request_indices = input_metadata
154        .paged_kv_request_indices
155        .as_ref()
156        .and_then(|m| m.get(&device_location))
157        .ok_or_else(|| candle_core::Error::msg("paged_kv_request_indices missing"))?;
158    let paged_kv_tile_indices = input_metadata
159        .paged_kv_tile_indices
160        .as_ref()
161        .and_then(|m| m.get(&device_location))
162        .ok_or_else(|| candle_core::Error::msg("paged_kv_tile_indices missing"))?;
163    let paged_kv_o_indptr = input_metadata
164        .paged_kv_o_indptr
165        .as_ref()
166        .and_then(|m| m.get(&device_location))
167        .ok_or_else(|| candle_core::Error::msg("paged_kv_o_indptr missing"))?;
168    let paged_kv_chunk_size = input_metadata
169        .paged_kv_chunk_size
170        .as_ref()
171        .and_then(|m| m.get(&device_location))
172        .ok_or_else(|| candle_core::Error::msg("paged_kv_chunk_size missing"))?;
173
174    let ckv_flat = ckv.contiguous()?.reshape((bs * seq_len, kv_lora_rank))?;
175    let k_pe_flat = k_pe
176        .squeeze(1)?
177        .contiguous()?
178        .reshape((bs * seq_len, qk_rope_head_dim))?;
179
180    mistralrs_paged_attn::concat_and_cache_mla(
181        &ckv_flat,
182        &k_pe_flat,
183        key_cache,
184        value_cache,
185        &slot_mapping,
186    )?;
187
188    let q_nope = q_nope.squeeze(2)?.contiguous()?;
189    let q_pe = q_pe.squeeze(2)?.contiguous()?;
190    let (w_uk, w_uv_t) = mla_weights.get_or_compute(
191        kv_b_proj,
192        q_nope.device(),
193        num_attention_heads,
194        kv_lora_rank,
195        qk_nope_head_dim,
196        v_head_dim,
197    )?;
198    let ql_nope = q_nope
199        .unsqueeze(D::Minus2)?
200        .broadcast_matmul(&w_uk.unsqueeze(0)?)?
201        .squeeze(D::Minus2)?
202        .contiguous()?;
203
204    let attn_latent = mistralrs_paged_attn::flashinfer_mla_decode(
205        &ql_nope,
206        &q_pe,
207        key_cache,
208        value_cache,
209        paged_kv_indptr,
210        paged_kv_indices,
211        paged_kv_last_page_len,
212        paged_kv_request_indices,
213        paged_kv_tile_indices,
214        paged_kv_o_indptr,
215        paged_kv_chunk_size,
216        sdpa_params.softmax_scale,
217    )?;
218
219    attn_latent
220        .unsqueeze(D::Minus2)?
221        .broadcast_matmul(&w_uv_t.unsqueeze(0)?)?
222        .squeeze(D::Minus2)
223}

247/// MLA cache forward pass for prefill with prefix caching support.
248///
249/// Handles both fresh prefill and prefix-cached scenarios.
250///
251/// # Arguments
252/// * `q` - Query tensor [bs, num_heads, seq_len, head_dim]
253/// * `k` - Key tensor [bs, num_heads, seq_len, head_dim]
254/// * `v` - Value tensor [bs, num_heads, seq_len, v_head_dim]
255/// * `ckv` - Compressed KV tensor [bs, seq_len, kv_lora_rank]
256/// * `k_pe` - Positional key tensor [bs, 1, seq_len, qk_rope_head_dim]
257/// * `attention_mask` - Optional attention mask
258/// * `seqlen_offsets` - Prefix lengths for each sequence
259/// * `metadata` - PagedAttention cache and metadata
260/// * `flash_params` - Flash attention parameters
261/// * `kv_b_proj` - KV B projection layer
262/// * `sdpa_params` - SDPA parameters
263/// * `num_attention_heads` - Number of attention heads
264/// * `kv_lora_rank` - KV latent dimension
265/// * `qk_rope_head_dim` - Rope head dimension
266/// * `qk_nope_head_dim` - Non-positional head dimension
267/// * `v_head_dim` - Value head dimension
268/// * `bs` - Batch size
269/// * `seq_len` - Sequence length
270#[cfg(all(feature = "cuda", target_family = "unix"))]
271#[allow(clippy::too_many_arguments)]
272pub fn mla_cache_forward(
273    q: &Tensor,
274    k: &Tensor,
275    v: &Tensor,
276    ckv: &Tensor,
277    k_pe: &Tensor,
278    attention_mask: Option<&Tensor>,
279    seqlen_offsets: &[usize],
280    metadata: &Option<((Tensor, Tensor), &PagedAttentionInputMetadata)>,
281    flash_params: &FlashParams,
282    kv_b_proj: &dyn mistralrs_quant::QuantMethod,
283    sdpa_params: &SdpaParams,
284    num_attention_heads: usize,
285    kv_lora_rank: usize,
286    qk_rope_head_dim: usize,
287    qk_nope_head_dim: usize,
288    v_head_dim: usize,
289    bs: usize,
290    seq_len: usize,
291) -> Result<Tensor> {
292    let mut key_cache = None;
293    let mut value_cache = None;
294    let mut input_metadata = None;
295    if let Some(((key_cache_tensor, value_cache_tensor), meta)) = metadata.as_ref() {
296        key_cache = Some(key_cache_tensor);
297        value_cache = Some(value_cache_tensor);
298        input_metadata = Some(*meta);
299
300        let device_location = q.device().location();
301        let slot_mapping = meta
302            .slot_mappings
303            .get(&device_location)
304            .ok_or_else(|| candle_core::Error::msg("slot mapping missing"))?;
305        let slot_mapping = if slot_mapping.dims().len() > 1 {
306            slot_mapping.flatten(0, slot_mapping.dims().len())?
307        } else {
308            slot_mapping.clone()
309        };
310        let ckv_flat = ckv.contiguous()?.reshape((bs * seq_len, kv_lora_rank))?;
311        let k_pe_flat = k_pe
312            .squeeze(1)?
313            .contiguous()?
314            .reshape((bs * seq_len, qk_rope_head_dim))?;
315        mistralrs_paged_attn::concat_and_cache_mla(
316            &ckv_flat,
317            &k_pe_flat,
318            key_cache_tensor,
319            value_cache_tensor,
320            &slot_mapping,
321        )?;
322    }
323
324    let prefix_lens = seqlen_offsets;
325    let needs_prefix = prefix_lens.iter().any(|&len| len > 0);
326    if !needs_prefix && attention_mask.is_some() {
327        Sdpa.run_attention(q, k, v, attention_mask, Some(flash_params), sdpa_params)
328    } else {
329        let ((key_cache, value_cache), input_metadata) =
330            match (key_cache, value_cache, input_metadata) {
331                (Some(k), Some(v), Some(m)) => ((k, v), m),
332                _ => {
333                    return Sdpa.run_attention(
334                        q,
335                        k,
336                        v,
337                        attention_mask,
338                        Some(flash_params),
339                        sdpa_params,
340                    );
341                }
342            };
343
344        let device_location = q.device().location();
345        let slot_mapping = input_metadata
346            .slot_mappings
347            .get(&device_location)
348            .ok_or_else(|| candle_core::Error::msg("slot mapping missing"))?;
349        let slot_mapping_cpu = slot_mapping.to_device(&Device::Cpu)?;
350        let slot_mapping_cpu = if slot_mapping_cpu.dims().len() == 2 {
351            slot_mapping_cpu
352        } else {
353            slot_mapping_cpu.reshape((bs, seq_len))?
354        };
355        let slot_mapping_vec = slot_mapping_cpu.to_vec2::<i64>()?;
356        let cur_lens: Vec<usize> = slot_mapping_vec
357            .iter()
358            .map(|row| {
359                row.iter()
360                    .filter(|&&v| v != crate::paged_attention::_PAD_SLOT_ID)
361                    .count()
362            })
363            .collect();
364
365        let block_tables = input_metadata
366            .block_tables
367            .as_ref()
368            .and_then(|m| m.get(&device_location))
369            .ok_or_else(|| candle_core::Error::msg("block tables missing"))?;
370        let block_tables_cpu = block_tables.to_device(&Device::Cpu)?;
371        let (block_rows, block_stride) = block_tables_cpu.dims2()?;
372        let block_tables_vec = block_tables_cpu.to_vec2::<u32>()?;
373        let expected_rows: usize = cur_lens.iter().sum();
374        let mut block_tables_seq: Vec<i32> = Vec::with_capacity(bs * block_stride);
375        if block_rows == bs {
376            for row in block_tables_vec.iter().take(bs) {
377                block_tables_seq.extend(row.iter().map(|&v| v as i32));
378            }
379        } else if block_rows == expected_rows {
380            let mut offset = 0usize;
381            for len in cur_lens.iter() {
382                if offset < block_tables_vec.len() {
383                    block_tables_seq.extend(block_tables_vec[offset].iter().map(|&v| v as i32));
384                } else {
385                    block_tables_seq.extend(std::iter::repeat_n(0, block_stride));
386                }
387                offset = offset.saturating_add(*len);
388            }
389        } else {
390            candle_core::bail!(
391                "unexpected block_tables rows: got {block_rows}, expected {bs} or {expected_rows}"
392            );
393        }
394
395        let block_tables_seq =
396            Tensor::from_vec(block_tables_seq, (bs, block_stride), &Device::Cpu)?
397                .to_device(q.device())?;
398
399        let total_prefix_tokens: usize = prefix_lens.iter().sum();
400        let mut k_prefix = None;
401        let mut v_prefix = None;
402        let mut prefix_offsets = Vec::with_capacity(bs + 1);
403        prefix_offsets.push(0usize);
404        for len in prefix_lens {
405            let next = *prefix_offsets.last().unwrap() + *len;
406            prefix_offsets.push(next);
407        }
408
409        if total_prefix_tokens > 0 {
410            let mut cu_seq_lens = Vec::with_capacity(bs + 1);
411            cu_seq_lens.push(0i32);
412            let mut token_to_seq = Vec::with_capacity(total_prefix_tokens);
413            #[allow(clippy::cast_possible_truncation)]
414            for (seq_idx, len) in prefix_lens.iter().enumerate() {
415                let next = *cu_seq_lens.last().unwrap() + *len as i32;
416                cu_seq_lens.push(next);
417                token_to_seq.extend(std::iter::repeat_n(seq_idx as i32, *len));
418            }
419
420            let cu_seq_lens =
421                Tensor::from_vec(cu_seq_lens, (bs + 1,), &Device::Cpu)?.to_device(q.device())?;
422            let token_to_seq =
423                Tensor::from_vec(token_to_seq, (total_prefix_tokens,), &Device::Cpu)?
424                    .to_device(q.device())?;
425
426            let (ckv_prefix, kpe_prefix) = mistralrs_paged_attn::gather_mla_cache(
427                key_cache,
428                value_cache,
429                &block_tables_seq,
430                &cu_seq_lens,
431                &token_to_seq,
432            )?;
433
434            let mut kv_prefix = kv_b_proj.forward_autocast(&ckv_prefix)?;
435            kv_prefix = kv_prefix.reshape((
436                total_prefix_tokens,
437                num_attention_heads,
438                qk_nope_head_dim + v_head_dim,
439            ))?;
440            let kv_prefix_split = kv_prefix.split(&[qk_nope_head_dim, v_head_dim], D::Minus1)?;
441            let k_nope_prefix = kv_prefix_split[0].clone();
442            let v_prefix_full = kv_prefix_split[1].clone();
443            let kpe_prefix = kpe_prefix.unsqueeze(1)?.expand((
444                total_prefix_tokens,
445                num_attention_heads,
446                qk_rope_head_dim,
447            ))?;
448            let k_prefix_full = Tensor::cat(&[&k_nope_prefix, &kpe_prefix], D::Minus1)?;
449            k_prefix = Some(k_prefix_full);
450            v_prefix = Some(v_prefix_full);
451        }
452
453        let mut outputs = Vec::with_capacity(bs);
454        for (seq_idx, cur_len) in cur_lens.iter().enumerate() {
455            let cur_len = (*cur_len).min(seq_len);
456            if cur_len == 0 {
457                outputs.push(Tensor::zeros(
458                    (seq_len, num_attention_heads, v_head_dim),
459                    q.dtype(),
460                    q.device(),
461                )?);
462                continue;
463            }
464            let mut q_i = q.narrow(0, seq_idx, 1)?;
465            let mut k_i = k.narrow(0, seq_idx, 1)?;
466            let mut v_i = v.narrow(0, seq_idx, 1)?;
467            if cur_len < seq_len {
468                q_i = q_i.narrow(D::Minus2, 0, cur_len)?;
469                k_i = k_i.narrow(D::Minus2, 0, cur_len)?;
470                v_i = v_i.narrow(D::Minus2, 0, cur_len)?;
471            }
472
473            let prefix_len = prefix_lens[seq_idx];
474            let k_full = if let Some(k_prefix_full) = &k_prefix {
475                let start = prefix_offsets[seq_idx];
476                let k_prefix_i = if prefix_len > 0 {
477                    Some(k_prefix_full.narrow(0, start, prefix_len)?)
478                } else {
479                    None
480                };
481                let k_i = k_i.squeeze(0)?.contiguous()?;
482                if let Some(k_prefix_i) = k_prefix_i {
483                    let k_prefix_i = k_prefix_i.transpose(0, 1)?.contiguous()?;
484                    Tensor::cat(&[&k_prefix_i, &k_i], 1)?
485                } else {
486                    k_i
487                }
488            } else {
489                k_i.squeeze(0)?.contiguous()?
490            };
491
492            let v_full = if let Some(v_prefix_full) = &v_prefix {
493                let start = prefix_offsets[seq_idx];
494                let v_prefix_i = if prefix_len > 0 {
495                    Some(v_prefix_full.narrow(0, start, prefix_len)?)
496                } else {
497                    None
498                };
499                let v_i = v_i.squeeze(0)?.contiguous()?;
500                if let Some(v_prefix_i) = v_prefix_i {
501                    let v_prefix_i = v_prefix_i.transpose(0, 1)?.contiguous()?;
502                    Tensor::cat(&[&v_prefix_i, &v_i], 1)?
503                } else {
504                    v_i
505                }
506            } else {
507                v_i.squeeze(0)?.contiguous()?
508            };
509
510            let mask = if cur_len > 1 {
511                let offset = cur_len + prefix_len;
512                let mask: Vec<_> = (0..cur_len)
513                    .flat_map(|i| (0..offset).map(move |j| u8::from(j + cur_len > i + offset)))
514                    .collect();
515                let mask = Tensor::from_slice(&mask, (cur_len, offset), q.device())?
516                    .to_dtype(DType::U8)?;
517                let zero = Tensor::new(0.0f32, q.device())?;
518                let mask = crate::layers_masker::masked_fill(
519                    &zero.to_dtype(q.dtype())?.broadcast_as(mask.shape())?,
520                    &mask,
521                    f32::NEG_INFINITY,
522                )?;
523                Some(mask)
524            } else {
525                None
526            };
527
528            let attn_out_i = Sdpa.run_attention_noflash(
529                &q_i,
530                &k_full.unsqueeze(0)?,
531                &v_full.unsqueeze(0)?,
532                mask.as_ref(),
533                sdpa_params,
534            )?;
535            let mut attn_out_i = attn_out_i.squeeze(0)?.transpose(0, 1)?;
536            if cur_len < seq_len {
537                attn_out_i = attn_out_i.pad_with_zeros(D::Minus(3), 0, seq_len - cur_len)?;
538            }
539            outputs.push(attn_out_i);
540        }
541        Tensor::cat(&outputs, 0)
542    }
543}
