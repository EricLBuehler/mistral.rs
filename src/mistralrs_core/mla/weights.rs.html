12/// Cached MLA weight matrices for efficient decode operations.
13///
14/// Stores the precomputed w_uk and w_uv_t matrices extracted from kv_b_proj.
15/// These are computed lazily on first use and cached for subsequent calls.
16pub struct MlaWeights {
17    #[cfg(all(feature = "cuda", target_family = "unix"))]
18    weights: Option<Mutex<Option<(Tensor, Tensor)>>>,
19    #[cfg(not(all(feature = "cuda", target_family = "unix")))]
20    _phantom: std::marker::PhantomData<()>,
21}
22
23impl MlaWeights {
24    /// Create a new MlaWeights instance.
25    ///
26    /// If `paged_attn_enabled` is true and we're on CUDA, allocates the mutex for caching.
27    /// Otherwise, the weights are not cached (MLA decode won't be used).
28    #[cfg(all(feature = "cuda", target_family = "unix"))]
29    pub fn new(paged_attn_enabled: bool, device: Option<&Device>) -> Self {
30        let weights = if paged_attn_enabled {
31            if let Some(device) = device {
32                if matches!(device, Device::Cuda(_)) {
33                    Some(Mutex::new(None))
34                } else {
35                    None
36                }
37            } else {
38                // If no device is provided, assume we may need it
39                Some(Mutex::new(None))
40            }
41        } else {
42            None
43        };
44        Self { weights }
45    }
46
47    #[cfg(not(all(feature = "cuda", target_family = "unix")))]
48    pub fn new(_paged_attn_enabled: bool, _device: Option<&candle_core::Device>) -> Self {
49        Self {
50            _phantom: std::marker::PhantomData,
51        }
52    }
53
54    /// Compute MLA weights from the kv_b_proj layer.
55    ///
56    /// Extracts w_uk (for K nope projection) and w_uv_t (transposed V projection)
57    /// from the kv_b_proj weight matrix.
58    ///
59    /// # Arguments
60    /// * `kv_b_proj` - The kv_b_proj quantized layer
61    /// * `device` - Target device for the weights
62    /// * `num_attention_heads` - Number of attention heads
63    /// * `kv_lora_rank` - KV latent dimension
64    /// * `qk_nope_head_dim` - Non-positional head dimension
65    /// * `v_head_dim` - Value head dimension
66    #[cfg(all(feature = "cuda", target_family = "unix"))]
67    pub fn compute_weights(
68        kv_b_proj: &dyn QuantMethod,
69        device: &Device,
70        num_attention_heads: usize,
71        kv_lora_rank: usize,
72        qk_nope_head_dim: usize,
73        v_head_dim: usize,
74    ) -> Result<(Tensor, Tensor)> {
75        let mut w = kv_b_proj.dequantize_w()?;
76        if !w.device().same_device(device) {
77            w = w.to_device(device)?;
78        }
79        let (out_dim, in_dim) = w.dims2()?;
80        if in_dim != kv_lora_rank {
81            candle_core::bail!(
82                "kv_b_proj weight in_dim mismatch: expected {}, got {}",
83                kv_lora_rank,
84                in_dim
85            );
86        }
87        let per_head_dim = qk_nope_head_dim + v_head_dim;
88        if out_dim != num_attention_heads * per_head_dim {
89            candle_core::bail!(
90                "kv_b_proj weight out_dim mismatch: expected {}, got {}",
91                num_attention_heads * per_head_dim,
92                out_dim
93            );
94        }
95        let w = w.reshape((num_attention_heads, per_head_dim, kv_lora_rank))?;
96        let w_uk = w.narrow(D::Minus2, 0, qk_nope_head_dim)?.contiguous()?;
97        let w_uv = w
98            .narrow(D::Minus2, qk_nope_head_dim, v_head_dim)?
99            .contiguous()?;
100        let w_uv_t = w_uv.transpose(1, 2)?.contiguous()?;
101        Ok((w_uk, w_uv_t))
102    }
103
104    /// Get or compute the MLA weights.
105    ///
106    /// Returns cached weights if available, otherwise computes and caches them.
107    #[cfg(all(feature = "cuda", target_family = "unix"))]
108    pub fn get_or_compute(
109        &self,
110        kv_b_proj: &dyn QuantMethod,
111        device: &Device,
112        num_attention_heads: usize,
113        kv_lora_rank: usize,
114        qk_nope_head_dim: usize,
115        v_head_dim: usize,
116    ) -> Result<(Tensor, Tensor)> {
117        let Some(mla_weights) = &self.weights else {
118            candle_core::bail!("MLA weights are not initialized on this device");
119        };
120        let mut guard = mla_weights.lock().expect("MLA weights mutex was poisoned");
121        if let Some((w_uk, w_uv_t)) = guard.as_ref() {
122            return Ok((w_uk.clone(), w_uv_t.clone()));
123        }
124        let (w_uk, w_uv_t) = Self::compute_weights(
125            kv_b_proj,
126            device,
127            num_attention_heads,
128            kv_lora_rank,
129            qk_nope_head_dim,
130            v_head_dim,
131        )?;
132        *guard = Some((w_uk.clone(), w_uv_t.clone()));
133        Ok((w_uk, w_uv_t))
134    }
135
136    #[cfg(not(all(feature = "cuda", target_family = "unix")))]
137    #[allow(dead_code)]
138    pub fn get_or_compute(
139        &self,
140        _kv_b_proj: &dyn mistralrs_quant::QuantMethod,
141        _device: &candle_core::Device,
142        _num_attention_heads: usize,
143        _kv_lora_rank: usize,
144        _qk_nope_head_dim: usize,
145        _v_head_dim: usize,
146    ) -> candle_core::Result<(candle_core::Tensor, candle_core::Tensor)> {
147        candle_core::bail!("MLA weights require CUDA support")
148    }
149}
