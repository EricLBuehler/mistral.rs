31fn scaled_dot_product_attention(q: &Tensor, k: &Tensor, v: &Tensor) -> Result<Tensor> {
32    let dim = q.dim(D::Minus1)?;
33    let scale_factor = 1.0 / (dim as f64).sqrt();
34    let mut batch_dims = q.dims().to_vec();
35    batch_dims.pop();
36    batch_dims.pop();
37    let q = q.flatten_to(batch_dims.len() - 1)?;
38    let k = k.flatten_to(batch_dims.len() - 1)?;
39    let v = v.flatten_to(batch_dims.len() - 1)?;
40    let attn_weights = (MatMul.matmul(&q, &k.t()?)? * scale_factor)?;
41    let attn_scores = MatMul.matmul(&candle_nn::ops::softmax_last_dim(&attn_weights)?, &v)?;
42    batch_dims.push(attn_scores.dim(D::Minus2)?);
43    batch_dims.push(attn_scores.dim(D::Minus1)?);
44    attn_scores.reshape(batch_dims)
45}

47fn rope(pos: &Tensor, dim: usize, theta: usize) -> Result<Tensor> {
48    if dim % 2 == 1 {
49        candle_core::bail!("dim {dim} is odd")
50    }
51    let dev = pos.device();
52    let theta = theta as f64;
53    let inv_freq: Vec<_> = (0..dim)
54        .step_by(2)
55        .map(|i| 1f32 / theta.powf(i as f64 / dim as f64) as f32)
56        .collect();
57    let inv_freq_len = inv_freq.len();
58    let inv_freq = Tensor::from_vec(inv_freq, (1, 1, inv_freq_len), dev)?;
59    let inv_freq = inv_freq.to_dtype(pos.dtype())?;
60    let freqs = pos.unsqueeze(2)?.broadcast_mul(&inv_freq)?;
61    let cos = freqs.cos()?;
62    let sin = freqs.sin()?;
63    let out = Tensor::stack(&[&cos, &sin.neg()?, &sin, &cos], 3)?;
64    let (b, n, d, _ij) = out.dims4()?;
65    out.reshape((b, n, d, 2, 2))
66}

68fn apply_rope(x: &Tensor, freq_cis: &Tensor) -> Result<Tensor> {
69    let dims = x.dims();
70    let (b_sz, n_head, seq_len, n_embd) = x.dims4()?;
71    let x = x.reshape((b_sz, n_head, seq_len, n_embd / 2, 2))?;
72    let x0 = x.narrow(D::Minus1, 0, 1)?;
73    let x1 = x.narrow(D::Minus1, 1, 1)?;
74    let fr0 = freq_cis.get_on_dim(D::Minus1, 0)?;
75    let fr1 = freq_cis.get_on_dim(D::Minus1, 1)?;
76    (fr0.broadcast_mul(&x0)? + fr1.broadcast_mul(&x1)?)?.reshape(dims.to_vec())
77}

79fn attention(q: &Tensor, k: &Tensor, v: &Tensor, pe: &Tensor) -> Result<Tensor> {
80    let q = apply_rope(q, pe)?.contiguous()?;
81    let k = apply_rope(k, pe)?.contiguous()?;
82    let x = scaled_dot_product_attention(&q, &k, v)?;
83    x.transpose(1, 2)?.flatten_from(2)
84}

86fn timestep_embedding(t: &Tensor, dim: usize, dtype: DType) -> Result<Tensor> {
87    const TIME_FACTOR: f64 = 1000.;
88    const MAX_PERIOD: f64 = 10000.;
89    if dim % 2 == 1 {
90        candle_core::bail!("{dim} is odd")
91    }
92    let dev = t.device();
93    let half = dim / 2;
94    let t = (t * TIME_FACTOR)?;
95    let arange = Tensor::arange(0, half as u32, dev)?.to_dtype(candle_core::DType::F32)?;
96    let freqs = (arange * (-MAX_PERIOD.ln() / half as f64))?.exp()?;
97    let args = t
98        .unsqueeze(1)?
99        .to_dtype(candle_core::DType::F32)?
100        .broadcast_mul(&freqs.unsqueeze(0)?)?;
101    let emb = Tensor::cat(&[args.cos()?, args.sin()?], D::Minus1)?.to_dtype(dtype)?;
102    Ok(emb)
103}

123impl candle_core::Module for EmbedNd {
124    fn forward(&self, ids: &Tensor) -> Result<Tensor> {
125        let n_axes = ids.dim(D::Minus1)?;
126        let mut emb = Vec::with_capacity(n_axes);
127        for idx in 0..n_axes {
128            let r = rope(
129                &ids.get_on_dim(D::Minus1, idx)?,
130                self.axes_dim[idx],
131                self.theta,
132            )?;
133            emb.push(r)
134        }
135        let emb = Tensor::cat(&emb, 2)?;
136        emb.unsqueeze(1)
137    }

182struct ModulationOut {

188impl ModulationOut {

261#[derive(Debug, Clone)]
262pub struct SelfAttention {
263    qkv: Linear,
264    norm: QkNorm,
265    proj: Linear,
266    num_attention_heads: usize,
267}
268
269impl SelfAttention {
270    fn new(
271        dim: usize,
272        num_attention_heads: usize,
273        qkv_bias: bool,
274        vb: ShardedVarBuilder,
275    ) -> Result<Self> {
276        let head_dim = dim / num_attention_heads;
277        let qkv = layers::linear_b(dim, dim * 3, qkv_bias, vb.pp("qkv"))?;
278        let norm = QkNorm::new(head_dim, vb.pp("norm"))?;
279        let proj = layers::linear(dim, dim, vb.pp("proj"))?;
280        Ok(Self {
281            qkv,
282            norm,
283            proj,
284            num_attention_heads,
285        })
286    }
287
288    fn qkv(&self, xs: &Tensor) -> Result<(Tensor, Tensor, Tensor)> {
289        let qkv = xs.apply(&self.qkv)?;
290        let (b, l, _khd) = qkv.dims3()?;
291        let qkv = qkv.reshape((b, l, 3, self.num_attention_heads, ()))?;
292        let q = qkv.i((.., .., 0))?.transpose(1, 2)?;
293        let k = qkv.i((.., .., 1))?.transpose(1, 2)?;
294        let v = qkv.i((.., .., 2))?.transpose(1, 2)?;
295        let q = q.apply(&self.norm.query_norm)?;
296        let k = k.apply(&self.norm.key_norm)?;
297        Ok((q, k, v))
298    }
299
300    #[allow(unused)]
301    fn forward(&self, xs: &Tensor, pe: &Tensor) -> Result<Tensor> {
302        let (q, k, v) = self.qkv(xs)?;
303        attention(&q, &k, &v, pe)?.apply(&self.proj)
304    }
305
306    fn cast_to(&mut self, device: &Device) -> Result<()> {
307        self.qkv = Linear::new(
308            self.qkv.weight().to_device(device)?,
309            self.qkv.bias().map(|x| x.to_device(device).unwrap()),
310        );
311        self.proj = Linear::new(
312            self.proj.weight().to_device(device)?,
313            self.proj.bias().map(|x| x.to_device(device).unwrap()),
314        );
315        self.norm = QkNorm {
316            query_norm: RmsNorm::new(
317                self.norm
318                    .query_norm
319                    .clone()
320                    .into_inner()
321                    .weight()
322                    .to_device(device)?,
323                1e-6,
324            ),
325            key_norm: RmsNorm::new(
326                self.norm
327                    .key_norm
328                    .clone()
329                    .into_inner()
330                    .weight()
331                    .to_device(device)?,
332                1e-6,
333            ),
334        };
335        Ok(())
336    }
337}

371#[derive(Debug, Clone)]
372pub struct DoubleStreamBlock {
373    img_mod: Modulation2,
374    img_norm1: LayerNorm,
375    img_attn: SelfAttention,
376    img_norm2: LayerNorm,
377    img_mlp: Mlp,
378    txt_mod: Modulation2,
379    txt_norm1: LayerNorm,
380    txt_attn: SelfAttention,
381    txt_norm2: LayerNorm,
382    txt_mlp: Mlp,
383}
384
385impl DoubleStreamBlock {
386    fn new(cfg: &Config, vb: ShardedVarBuilder) -> Result<Self> {
387        let h_sz = HIDDEN_SIZE;
388        let mlp_sz = (h_sz as f64 * MLP_RATIO) as usize;
389        let img_mod = Modulation2::new(h_sz, vb.pp("img_mod"))?;
390        let img_norm1 = layer_norm(h_sz, vb.pp("img_norm1"))?;
391        let img_attn = SelfAttention::new(h_sz, cfg.num_attention_heads, true, vb.pp("img_attn"))?;
392        let img_norm2 = layer_norm(h_sz, vb.pp("img_norm2"))?;
393        let img_mlp = Mlp::new(h_sz, mlp_sz, vb.pp("img_mlp"))?;
394        let txt_mod = Modulation2::new(h_sz, vb.pp("txt_mod"))?;
395        let txt_norm1 = layer_norm(h_sz, vb.pp("txt_norm1"))?;
396        let txt_attn = SelfAttention::new(h_sz, cfg.num_attention_heads, true, vb.pp("txt_attn"))?;
397        let txt_norm2 = layer_norm(h_sz, vb.pp("txt_norm2"))?;
398        let txt_mlp = Mlp::new(h_sz, mlp_sz, vb.pp("txt_mlp"))?;
399        Ok(Self {
400            img_mod,
401            img_norm1,
402            img_attn,
403            img_norm2,
404            img_mlp,
405            txt_mod,
406            txt_norm1,
407            txt_attn,
408            txt_norm2,
409            txt_mlp,
410        })
411    }
412
413    fn forward(
414        &self,
415        img: &Tensor,
416        txt: &Tensor,
417        vec_: &Tensor,
418        pe: &Tensor,
419    ) -> Result<(Tensor, Tensor)> {
420        let (img_mod1, img_mod2) = self.img_mod.forward(vec_)?; // shift, scale, gate
421        let (txt_mod1, txt_mod2) = self.txt_mod.forward(vec_)?; // shift, scale, gate
422        let img_modulated = img.apply(&self.img_norm1)?;
423        let img_modulated = img_mod1.scale_shift(&img_modulated)?;
424        let (img_q, img_k, img_v) = self.img_attn.qkv(&img_modulated)?;
425
426        let txt_modulated = txt.apply(&self.txt_norm1)?;
427        let txt_modulated = txt_mod1.scale_shift(&txt_modulated)?;
428        let (txt_q, txt_k, txt_v) = self.txt_attn.qkv(&txt_modulated)?;
429
430        let q = Tensor::cat(&[txt_q, img_q], 2)?;
431        let k = Tensor::cat(&[txt_k, img_k], 2)?;
432        let v = Tensor::cat(&[txt_v, img_v], 2)?;
433
434        let attn = attention(&q, &k, &v, pe)?;
435        let txt_attn = attn.narrow(1, 0, txt.dim(1)?)?;
436        let img_attn = attn.narrow(1, txt.dim(1)?, attn.dim(1)? - txt.dim(1)?)?;
437
438        let img = (img + img_mod1.gate(&img_attn.apply(&self.img_attn.proj)?))?;
439        let img = (&img
440            + img_mod2.gate(
441                &img_mod2
442                    .scale_shift(&img.apply(&self.img_norm2)?)?
443                    .apply(&self.img_mlp)?,
444            )?)?;
445
446        let txt = (txt + txt_mod1.gate(&txt_attn.apply(&self.txt_attn.proj)?))?;
447        let txt = (&txt
448            + txt_mod2.gate(
449                &txt_mod2
450                    .scale_shift(&txt.apply(&self.txt_norm2)?)?
451                    .apply(&self.txt_mlp)?,
452            )?)?;
453
454        Ok((img, txt))
455    }
456
457    fn cast_to(&mut self, device: &Device) -> Result<()> {
458        self.img_mod.lin = Linear::new(
459            self.img_mod.lin.weight().to_device(device)?,
460            self.img_mod
461                .lin
462                .bias()
463                .map(|x| x.to_device(device).unwrap()),
464        );
465        self.img_norm1 = LayerNorm::new_no_bias(self.img_norm1.weight().to_device(device)?, 1e-6);
466        self.img_attn.cast_to(device)?;
467        self.img_norm2 = LayerNorm::new_no_bias(self.img_norm2.weight().to_device(device)?, 1e-6);
468        self.img_mlp.cast_to(device)?;
469
470        self.txt_mod.lin = Linear::new(
471            self.txt_mod.lin.weight().to_device(device)?,
472            self.txt_mod
473                .lin
474                .bias()
475                .map(|x| x.to_device(device).unwrap()),
476        );
477        self.txt_norm1 = LayerNorm::new_no_bias(self.txt_norm1.weight().to_device(device)?, 1e-6);
478        self.txt_attn.cast_to(device)?;
479        self.txt_norm2 = LayerNorm::new_no_bias(self.txt_norm2.weight().to_device(device)?, 1e-6);
480        self.txt_mlp.cast_to(device)?;
481
482        Ok(())
483    }
484}

486#[derive(Debug, Clone)]
487pub struct SingleStreamBlock {
488    linear1: Linear,
489    linear2: Linear,
490    norm: QkNorm,
491    pre_norm: LayerNorm,
492    modulation: Modulation1,
493    h_sz: usize,
494    mlp_sz: usize,
495    num_attention_heads: usize,
496}
497
498impl SingleStreamBlock {
499    fn new(cfg: &Config, vb: ShardedVarBuilder) -> Result<Self> {
500        let h_sz = HIDDEN_SIZE;
501        let mlp_sz = (h_sz as f64 * MLP_RATIO) as usize;
502        let head_dim = h_sz / cfg.num_attention_heads;
503        let linear1 = layers::linear(h_sz, h_sz * 3 + mlp_sz, vb.pp("linear1"))?;
504        let linear2 = layers::linear(h_sz + mlp_sz, h_sz, vb.pp("linear2"))?;
505        let norm = QkNorm::new(head_dim, vb.pp("norm"))?;
506        let pre_norm = layer_norm(h_sz, vb.pp("pre_norm"))?;
507        let modulation = Modulation1::new(h_sz, vb.pp("modulation"))?;
508        Ok(Self {
509            linear1,
510            linear2,
511            norm,
512            pre_norm,
513            modulation,
514            h_sz,
515            mlp_sz,
516            num_attention_heads: cfg.num_attention_heads,
517        })
518    }
519
520    fn forward(&self, xs: &Tensor, vec_: &Tensor, pe: &Tensor) -> Result<Tensor> {
521        let mod_ = self.modulation.forward(vec_)?;
522        let x_mod = mod_.scale_shift(&xs.apply(&self.pre_norm)?)?;
523        let x_mod = x_mod.apply(&self.linear1)?;
524        let qkv = x_mod.narrow(D::Minus1, 0, 3 * self.h_sz)?;
525        let (b, l, _khd) = qkv.dims3()?;
526        let qkv = qkv.reshape((b, l, 3, self.num_attention_heads, ()))?;
527        let q = qkv.i((.., .., 0))?.transpose(1, 2)?;
528        let k = qkv.i((.., .., 1))?.transpose(1, 2)?;
529        let v = qkv.i((.., .., 2))?.transpose(1, 2)?;
530        let mlp = x_mod.narrow(D::Minus1, 3 * self.h_sz, self.mlp_sz)?;
531        let q = q.apply(&self.norm.query_norm)?;
532        let k = k.apply(&self.norm.key_norm)?;
533        let attn = attention(&q, &k, &v, pe)?;
534        let output = Tensor::cat(&[attn, mlp.gelu()?], 2)?.apply(&self.linear2)?;
535        xs + mod_.gate(&output)
536    }
537
538    fn cast_to(&mut self, device: &Device) -> Result<()> {
539        self.linear1 = Linear::new(
540            self.linear1.weight().to_device(device)?,
541            self.linear1.bias().map(|x| x.to_device(device).unwrap()),
542        );
543        self.linear2 = Linear::new(
544            self.linear2.weight().to_device(device)?,
545            self.linear2.bias().map(|x| x.to_device(device).unwrap()),
546        );
547        self.norm = QkNorm {
548            query_norm: RmsNorm::new(
549                self.norm
550                    .query_norm
551                    .clone()
552                    .into_inner()
553                    .weight()
554                    .to_device(device)?,
555                1e-6,
556            ),
557            key_norm: RmsNorm::new(
558                self.norm
559                    .key_norm
560                    .clone()
561                    .into_inner()
562                    .weight()
563                    .to_device(device)?,
564                1e-6,
565            ),
566        };
567        self.pre_norm = LayerNorm::new_no_bias(self.pre_norm.weight().to_device(device)?, 1e-6);
568        self.modulation.lin = Linear::new(
569            self.modulation.lin.weight().to_device(device)?,
570            self.modulation
571                .lin
572                .bias()
573                .map(|x| x.to_device(device).unwrap()),
574        );
575        Ok(())
576    }

586impl LastLayer {
587    fn new(h_sz: usize, p_sz: usize, out_c: usize, vb: ShardedVarBuilder) -> Result<Self> {
588        let norm_final = layer_norm(h_sz, vb.pp("norm_final"))?;
589        let linear = layers::linear(h_sz, p_sz * p_sz * out_c, vb.pp("linear"))?;
590        let ada_ln_modulation = layers::linear(h_sz, 2 * h_sz, vb.pp("adaLN_modulation.1"))?;
591        Ok(Self {
592            norm_final,
593            linear,
594            ada_ln_modulation,
595        })
596    }
597
598    fn forward(&self, xs: &Tensor, vec: &Tensor) -> Result<Tensor> {
599        let chunks = vec.silu()?.apply(&self.ada_ln_modulation)?.chunk(2, 1)?;
600        let (shift, scale) = (&chunks[0], &chunks[1]);
601        let xs = xs
602            .apply(&self.norm_final)?
603            .broadcast_mul(&(scale.unsqueeze(1)? + 1.0)?)?
604            .broadcast_add(&shift.unsqueeze(1)?)?;
605        xs.apply(&self.linear)
606    }
607}
608
609#[derive(Debug, Clone)]
610pub struct Flux {
611    img_in: Linear,
612    txt_in: Linear,
613    time_in: MlpEmbedder,
614    vector_in: MlpEmbedder,
615    guidance_in: Option<MlpEmbedder>,
616    pe_embedder: EmbedNd,
617    double_blocks: Vec<DoubleStreamBlock>,
618    single_blocks: Vec<SingleStreamBlock>,
619    final_layer: LastLayer,
620    device: Device,
621    offloaded: bool,
622}
623
624impl Flux {
625    pub fn new(
626        cfg: &Config,
627        vb: ShardedVarBuilder,
628        device: Device,
629        offloaded: bool,
630    ) -> Result<Self> {
631        let img_in = layers::linear(
632            cfg.in_channels,
633            HIDDEN_SIZE,
634            vb.pp("img_in").set_device(device.clone()),
635        )?;
636        let txt_in = layers::linear(
637            cfg.joint_attention_dim,
638            HIDDEN_SIZE,
639            vb.pp("txt_in").set_device(device.clone()),
640        )?;
641        let mut double_blocks = Vec::with_capacity(cfg.num_layers);
642        let vb_d = vb.pp("double_blocks");
643        for idx in 0..cfg.num_layers {
644            let db = DoubleStreamBlock::new(cfg, vb_d.pp(idx))?;
645            double_blocks.push(db)
646        }
647        let mut single_blocks = Vec::with_capacity(cfg.num_single_layers);
648        let vb_s = vb.pp("single_blocks");
649        for idx in 0..cfg.num_single_layers {
650            let sb = SingleStreamBlock::new(cfg, vb_s.pp(idx))?;
651            single_blocks.push(sb)
652        }
653        let time_in = MlpEmbedder::new(
654            256,
655            HIDDEN_SIZE,
656            vb.pp("time_in").set_device(device.clone()),
657        )?;
658        let vector_in = MlpEmbedder::new(
659            cfg.pooled_projection_dim,
660            HIDDEN_SIZE,
661            vb.pp("vector_in").set_device(device.clone()),
662        )?;
663        let guidance_in = if cfg.guidance_embeds {
664            let mlp = MlpEmbedder::new(
665                256,
666                HIDDEN_SIZE,
667                vb.pp("guidance_in").set_device(device.clone()),
668            )?;
669            Some(mlp)
670        } else {
671            None
672        };
673        let final_layer = LastLayer::new(
674            HIDDEN_SIZE,
675            1,
676            cfg.in_channels,
677            vb.pp("final_layer").set_device(device.clone()),
678        )?;
679        let pe_dim = HIDDEN_SIZE / cfg.num_attention_heads;
680        let pe_embedder = EmbedNd::new(pe_dim, THETA, AXES_DIM.to_vec());
681        Ok(Self {
682            img_in,
683            txt_in,
684            time_in,
685            vector_in,
686            guidance_in,
687            pe_embedder,
688            double_blocks,
689            single_blocks,
690            final_layer,
691            device: device.clone(),
692            offloaded,
693        })
694    }
695
696    #[allow(clippy::too_many_arguments)]
697    pub fn forward(
698        &mut self,
699        img: &Tensor,
700        img_ids: &Tensor,
701        txt: &Tensor,
702        txt_ids: &Tensor,
703        timesteps: &Tensor,
704        y: &Tensor,
705        guidance: Option<&Tensor>,
706    ) -> Result<Tensor> {
707        if txt.rank() != 3 {
708            candle_core::bail!("unexpected shape for txt {:?}", txt.shape())
709        }
710        if img.rank() != 3 {
711            candle_core::bail!("unexpected shape for img {:?}", img.shape())
712        }
713        let dtype = img.dtype();
714        let pe = {
715            let ids = Tensor::cat(&[txt_ids, img_ids], 1)?;
716            ids.apply(&self.pe_embedder)?
717        };
718        let mut txt = txt.apply(&self.txt_in)?;
719        let mut img = img.apply(&self.img_in)?;
720        let vec_ = timestep_embedding(timesteps, 256, dtype)?.apply(&self.time_in)?;
721        let vec_ = match (self.guidance_in.as_ref(), guidance) {
722            (Some(g_in), Some(guidance)) => {
723                (vec_ + timestep_embedding(guidance, 256, dtype)?.apply(g_in))?
724            }
725            _ => vec_,
726        };
727        let vec_ = (vec_ + y.apply(&self.vector_in))?;
728
729        // Double blocks
730        for block in self.double_blocks.iter_mut() {
731            if self.offloaded {
732                block.cast_to(&self.device)?;
733            }
734            (img, txt) = block.forward(&img, &txt, &vec_, &pe)?;
735            if self.offloaded {
736                block.cast_to(&Device::Cpu)?;
737            }
738        }
739        // Single blocks
740        let mut img = Tensor::cat(&[&txt, &img], 1)?;
741        for block in self.single_blocks.iter_mut() {
742            if self.offloaded {
743                block.cast_to(&self.device)?;
744            }
745            img = block.forward(&img, &vec_, &pe)?;
746            if self.offloaded {
747                block.cast_to(&Device::Cpu)?;
748            }
749        }
750        let img = img.i((.., txt.dim(1)?..))?;
751        self.final_layer.forward(&img, &vec_)
752    }
753}
