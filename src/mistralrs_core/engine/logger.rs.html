3use std::sync::atomic::{AtomicBool, AtomicUsize, Ordering};
4use std::sync::Arc;
5use std::thread;
6use std::time::Duration;
7
8use tracing::info;
9
10pub struct IntervalLogger {
11    enable_logging: Arc<AtomicBool>,
12    prefix_cache_hits: Arc<AtomicUsize>,
13    tokens_processed: Arc<AtomicUsize>,
14    total_new_seqs: Arc<AtomicUsize>,
15    num_running: Arc<AtomicUsize>,
16    num_waiting: Arc<AtomicUsize>,
17}
18
19impl IntervalLogger {
20    /// Starts an interval logger. Call `begin_logging` to begin the logging process.
21    pub fn new(interval: Duration) -> Self {
22        let prefix_cache_hits = Arc::new(AtomicUsize::new(0));
23        let tokens_processed = Arc::new(AtomicUsize::new(0));
24        let total_new_seqs = Arc::new(AtomicUsize::new(0));
25        let enable_logging = Arc::new(AtomicBool::new(false));
26        let num_running = Arc::new(AtomicUsize::new(0));
27        let num_waiting = Arc::new(AtomicUsize::new(0));
28
29        let t_prefix_cache_hits = prefix_cache_hits.clone();
30        let t_tokens_processed = tokens_processed.clone();
31        let t_total_new_seqs = total_new_seqs.clone();
32        let t_enable_logging = enable_logging.clone();
33        let t_num_running = num_running.clone();
34        let t_num_waiting = num_waiting.clone();
35        thread::spawn(move || {
36            // Start the actual logging
37            loop {
38                thread::sleep(interval);
39                if !t_enable_logging.load(Ordering::Relaxed) {
40                    continue;
41                }
42
43                let total_new_seqs = t_total_new_seqs.load(Ordering::Relaxed);
44                let prefix_cache_hits = t_prefix_cache_hits.load(Ordering::Relaxed);
45                let tokens_processed = t_tokens_processed.swap(0, Ordering::Relaxed);
46                let num_running = t_num_running.load(Ordering::Relaxed);
47                let num_waiting = t_num_waiting.load(Ordering::Relaxed);
48
49                if total_new_seqs != 0 && tokens_processed != 0 {
50                    info!(
51                        "Throughput (T/s) {:.2}, Prefix cache hitrate {:.2}%, {num_running} running, {num_waiting} waiting",
52                        tokens_processed as f64 / interval.as_secs_f64(),
53                        100. * prefix_cache_hits as f64 / total_new_seqs as f64,
54                    );
55                }
56            }
57        });
58
59        Self {
60            prefix_cache_hits,
61            tokens_processed,
62            total_new_seqs,
63            enable_logging,
64            num_running,
65            num_waiting,
66        }
67    }
68
69    pub fn enable_logging(&self) {
70        self.enable_logging.store(true, Ordering::Relaxed);
71    }
72
73    pub fn add_tokens_processed(&self, num_tokens: usize) {
74        self.tokens_processed
75            .fetch_add(num_tokens, Ordering::Relaxed);
76    }
77
78    pub fn add_new_sequence(&self) {
79        self.total_new_seqs.fetch_add(1, Ordering::Relaxed);
80    }
81
82    pub fn add_prefix_cache_hit(&self) {
83        self.prefix_cache_hits.fetch_add(1, Ordering::Relaxed);
84    }
85
86    pub fn set_num_running(&self, running: usize) {
87        self.num_running.store(running, Ordering::Relaxed);
88    }
89
90    pub fn set_num_waiting(&self, waiting: usize) {
91        self.num_waiting.store(waiting, Ordering::Relaxed);
92    }
93}
