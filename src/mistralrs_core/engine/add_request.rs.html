32    pub async fn handle_request(self: Arc<Self>, request: Request) {
33        match request {
34            Request::ActivateAdapters(adapters) => {
35                match get_mut_arcmutex!(self.pipeline).activate_adapters(adapters) {
36                    Ok(n) => info!("Swapped adapters in {n} LoRA layers."),
37                    Err(e) => warn!("Adapter activation failed: {e:?}"),
38                }
39            }
40            Request::Normal(request) => {
41                if matches!(
42                    request.messages,
43                    RequestMessage::Chat { .. } | RequestMessage::VisionChat { .. }
44                ) && request.web_search_options.is_some()
45                    && !request.is_streaming
46                    && get_mut_arcmutex!(self.bert_pipeline).is_some()
47                {
48                    let Some(web_search_options) = request.web_search_options.clone() else {
49                        unreachable!()
50                    };
51                    let mut first_request = request.clone();
52                    // Actually add the search tool here
53                    first_request
54                        .tools
55                        .get_or_insert_with(Vec::new)
56                        .push(search::get_search_tool(&web_search_options).unwrap());
57
58                    let mut second_request = first_request.clone();
59                    first_request.web_search_options = None;
60                    second_request.web_search_options = None;
61
62                    let this = self.clone();
63                    let handle = tokio::spawn(async move {
64                        let (new_sender, mut first_receiver) = tokio::sync::mpsc::channel(1);
65                        second_request.response = new_sender;
66                        std::mem::swap(&mut first_request.response, &mut second_request.response);
67
68                        this.add_request(first_request).await;
69                        let ResponseOk::Done(done) =
70                            first_receiver.recv().await.unwrap().as_result().unwrap()
71                        else {
72                            unreachable!()
73                        };
74
75                        let tool_calls = match &done.choices[0].message.tool_calls {
76                            Some(tool_calls)
77                                if tool_calls.len() == 1
78                                    && tool_calls[0].function.name == search::SEARCH_TOOL_NAME =>
79                            {
80                                &tool_calls[0]
81                            }
82                            None => {
83                                second_request
84                                    .response
85                                    .send(Response::Done(done))
86                                    .await
87                                    .unwrap();
88                                return;
89                            }
90                            Some(_) => {
91                                second_request
92                                    .response
93                                    .send(Response::Done(done))
94                                    .await
95                                    .unwrap();
96                                return;
97                            }
98                        };
99
100                        let RequestMessage::Chat(messages) = &mut second_request.messages else {
101                            unreachable!()
102                        };
103
104                        // Add assistant call message
105                        {
106                            let mut message: IndexMap<String, MessageContent> = IndexMap::new();
107                            message
108                                .insert("role".to_string(), Either::Left("assistant".to_string()));
109                            message.insert(
110                                "content".to_string(),
111                                Either::Left(format!(
112                                    "{{\"name\":\"{}\",\"arguments\":\"{}\"}}",
113                                    tool_calls.function.name, tool_calls.function.arguments
114                                )),
115                            );
116                            messages.push(message);
117                        }
118                        let tool_call_params: SearchFunctionParameters =
119                            serde_json::from_str(&tool_calls.function.arguments).unwrap();
120
121                        // Add tool response
122                        {
123                            let tokenizer = get_mut_arcmutex!(this.pipeline)
124                                .tokenizer()
125                                .expect("A tokenizer is expected for non-diffusion models.");
126                            let mut results = search::run_search_tool(&tool_call_params)
127                                .unwrap()
128                                .into_iter()
129                                .map(|result| {
130                                    let len = {
131                                        let inp = InputSequence::Raw(Cow::from(&result.content));
132                                        tokenizer
133                                            .encode_fast(inp, false)
134                                            .map(|x| x.len())
135                                            .unwrap_or(usize::MAX)
136                                    };
137                                    (result, len)
138                                })
139                                .collect::<Vec<_>>();
140                            // Sort increasing by tokenized length, if it fails, put it at the end.
141                            results.sort_by_key(|(_, len)| *len);
142
143                            {
144                                let device = get_mut_arcmutex!(this.pipeline).device();
145
146                                let Some(bert_pipeline) =
147                                    &mut *get_mut_arcmutex!(this.bert_pipeline)
148                                else {
149                                    unreachable!()
150                                };
151
152                                let decreasing_indexes = search::rag::compute_most_similar(
153                                    &device,
154                                    &tool_call_params.query,
155                                    results.iter().map(|(res, _)| res).collect::<Vec<_>>(),
156                                    bert_pipeline,
157                                )
158                                .unwrap();
159
160                                // Rerank the results
161                                let mut results_old = Vec::new();
162                                std::mem::swap(&mut results_old, &mut results);
163                                for &index in &decreasing_indexes {
164                                    let mut current_result: (SearchResult, usize) =
165                                        Default::default();
166                                    std::mem::swap(&mut current_result, &mut results_old[index]);
167
168                                    results.push(current_result);
169                                }
170                            }
171
172                            // Manage context size by # of tokens. Apply default here.
173                            let max_results_budget_toks =
174                                match web_search_options.search_context_size.unwrap_or_default() {
175                                    SearchContextSize::High => 10000_usize,
176                                    SearchContextSize::Medium => 7500_usize,
177                                    SearchContextSize::Low => 3000_usize,
178                                };
179                            let mut used_results = Vec::new();
180                            let mut used_len = 0;
181                            for (item, len) in results {
182                                if used_len + len >= max_results_budget_toks {
183                                    break;
184                                }
185                                // So the info! below gets the correct value
186                                used_len += len;
187                                used_results.push(item);
188                            }
189
190                            let tool_result = serde_json::to_string(&used_results)
191                                .unwrap()
192                                .replace("\\n", "\n")
193                                .replace("\\\"", "\"")
194                                .replace("\\\\", "\\");
195                            info!("Web search executed, using {used_len} tokens of {} search results.", used_results.len());
196
197                            let mut message: IndexMap<String, MessageContent> = IndexMap::new();
198                            message.insert("role".to_string(), Either::Left("tool".to_string()));
199                            message.insert(
200                                "content".to_string(),
201                                Either::Left(format!("{{\"output\": \"{tool_result}\"}}")),
202                            );
203                            messages.push(message);
204                        }
205
206                        this.add_request(second_request).await;
207                    });
208                    get_mut_arcmutex!(self.handles).push(handle);
209                } else {
210                    self.add_request(request).await
211                }
212            }
213            Request::ReIsq(level) => {
214                if let Err(e) = get_mut_arcmutex!(self.pipeline).re_isq_model(level) {
215                    warn!("ISQ requantization failed: {e:?}");
216                }
217            }
218            Request::Tokenize(req) => self.tokenize_text(req).await,
219            Request::Detokenize(req) => self.detokenize_text(req).await,
220            Request::Terminate => (),
221            Request::TerminateAllSeqsNextStep => {
222                TERMINATE_ALL_NEXT_STEP.store(true, Ordering::SeqCst)
223            }
224        }
225    }

227    async fn add_request(&self, request: NormalRequest) {
228        let is_chat = matches!(
229            request.messages,
230            RequestMessage::Chat(_) | RequestMessage::VisionChat { .. }
231        );
232        let echo_prompt = matches!(
233            request.messages,
234            RequestMessage::Completion {
235                echo_prompt: true,
236                ..
237            }
238        );
239
240        let best_of = match request.messages {
241            RequestMessage::Completion { best_of, .. } => best_of,
242            RequestMessage::Chat(_)
243            | RequestMessage::CompletionTokens(_)
244            | RequestMessage::VisionChat { .. }
245            | RequestMessage::ImageGeneration { .. } => None,
246        };
247        if is_chat
248            && !get_mut_arcmutex!(self.pipeline)
249                .get_chat_template()
250                .as_ref()
251                .is_some_and(|ch_t| ch_t.has_chat_template())
252        {
253            request
254                    .response
255                    .send(Response::ValidationError(
256                        "Received messages for a model which does not have a chat template. Either use a different model or pass a single string as the prompt".into(),
257                    )).await.expect("Expected receiver.");
258            return;
259        }
260
261        let images = match request.messages {
262            RequestMessage::VisionChat {
263                ref images,
264                messages: _,
265            } => Some(images.clone()),
266            _ => None,
267        };
268
269        let matcher = Arc::new(handle_seq_error!(
270            ToolCallingMatcher::new(request.tool_choice.unwrap_or(ToolChoice::Auto),),
271            request.response
272        ));
273
274        let image_generation_format = match &request.messages {
275            RequestMessage::ImageGeneration { format, .. } => Some(*format),
276            _ => None,
277        };
278
279        let seq_step_type = match &request.messages {
280            RequestMessage::ImageGeneration { .. } => SeqStepType::OneShot,
281            _ => SeqStepType::PromptAndDecode,
282        };
283
284        let diffusion_params = match &request.messages {
285            RequestMessage::ImageGeneration {
286                generation_params, ..
287            } => Some(generation_params.clone()),
288            _ => None,
289        };
290
291        let (mut prompt_tokens, prompt_text) = match request.messages {
292            RequestMessage::Chat(messages)
293            | RequestMessage::VisionChat {
294                images: _,
295                messages,
296            } => {
297                let pipeline = &*get_mut_arcmutex!(self.pipeline);
298                let tools = request.tools.unwrap_or_default();
299                let template = pipeline
300                    .get_processor()
301                    .process(pipeline, messages, true, true, tools);
302                handle_seq_error!(template, request.response)
303            }
304            RequestMessage::Completion { text, .. } => {
305                let Some(tokenizer) = &get_mut_arcmutex!(self.pipeline).tokenizer() else {
306                    request
307                        .response
308                        .send(Response::ValidationError(
309                            "Completion requests require the pipeline to have a tokenizer".into(),
310                        ))
311                        .await
312                        .expect("Expected receiver.");
313                    return;
314                };
315                let prompt = tokenizer
316                    .encode_fast(text.clone(), true)
317                    .map_err(anyhow::Error::msg);
318                (
319                    handle_seq_error!(prompt, request.response)
320                        .get_ids()
321                        .to_vec(),
322                    text,
323                )
324            }
325            RequestMessage::ImageGeneration { prompt, .. } => (vec![u32::MAX], prompt),
326            RequestMessage::CompletionTokens(it) => {
327                let Some(tokenizer) = &get_mut_arcmutex!(self.pipeline).tokenizer() else {
328                    request
329                        .response
330                        .send(Response::ValidationError(
331                            "Completion requests w/ raw tokens require the pipeline to have a tokenizer".into(),
332                        ))
333                        .await
334                        .expect("Expected receiver.");
335                    return;
336                };
337                let prompt = tokenizer
338                    .decode(&it, false)
339                    .map_err(|e| anyhow::Error::msg(e.to_string()));
340                (it, handle_seq_error!(prompt, request.response))
341            }
342        };
343        if prompt_tokens.is_empty() {
344            request
345                .response
346                .send(Response::ValidationError(
347                    "Received an empty prompt.".into(),
348                ))
349                .await
350                .expect("Expected receiver.");
351            return;
352        }
353
354        if prompt_tokens.len() > get_mut_arcmutex!(self.pipeline).get_metadata().max_seq_len {
355            if !self.truncate_sequence {
356                request
357                    .response
358                    .send(Response::ValidationError(
359                        format!("Prompt sequence length is greater than {}, perhaps consider using `truncate_sequence`?", get_mut_arcmutex!(self.pipeline).get_metadata().max_seq_len).into(),
360                    )).await.expect("Expected receiver.");
361                return;
362            } else {
363                let prompt_len = prompt_tokens.len();
364                let max_len = get_mut_arcmutex!(self.pipeline).get_metadata().max_seq_len;
365                let currently_over = prompt_len - max_len;
366                let sampling_max = if let Some(sampling_max) = request.sampling_params.max_len {
367                    if currently_over + sampling_max >= prompt_len {
368                        10
369                    } else {
370                        sampling_max
371                    }
372                } else {
373                    10
374                };
375                prompt_tokens = prompt_tokens[(currently_over + sampling_max)..].to_vec();
376                warn!("Prompt for request {} was {} tokens over the model maximum length. The last {} tokens were truncated to make space for generation.", request.id, currently_over, prompt_len - prompt_tokens.len());
377            }
378        }
379        let prefill_cache = handle_seq_error!(
380            get_mut_arcmutex!(self.prefix_cacher).search_for_matching_cache(
381                &prompt_tokens,
382                images.as_ref().is_some_and(|x| !x.is_empty())
383            ),
384            request.response
385        );
386
387        let topk = request
388            .sampling_params
389            .top_k
390            .map(|x| x as i64)
391            .unwrap_or(-1);
392        let topp = request.sampling_params.top_p.unwrap_or(1.0);
393        let minp = request.sampling_params.min_p.unwrap_or(0.0);
394        let num_hidden_layers = get_mut_arcmutex!(self.pipeline)
395            .get_metadata()
396            .num_hidden_layers;
397
398        let (stop_toks, stop_strings) = match request.sampling_params.stop_toks {
399            None => (vec![], vec![]),
400            Some(StopTokens::Ids(ref i)) => {
401                let tok_env = {
402                    let pipeline = get_mut_arcmutex!(self.pipeline);
403                    pipeline.get_metadata().tok_env.clone()
404                };
405                for id in i {
406                    // We can't use ` ` (space) as a stop token because other tokens like ` moon` start with a space.
407                    if let Some(tok_env) = tok_env.as_ref() {
408                        let tok_trie = tok_env.tok_trie();
409                        if tok_trie.has_extensions(tok_trie.token(*id)) {
410                            request
411                                .response
412                                .send(Response::ValidationError(
413                                    format!("Stop token {:?} is also a prefix of other tokens and cannot be used as a stop token.", tok_trie.token_str(*id)).into(),
414                                ))
415                                .await .expect("Expected receiver.");
416                            return;
417                        }
418                    }
419                }
420
421                (i.clone(), vec![])
422            }
423            Some(StopTokens::Seqs(ref s)) => {
424                let mut stop_toks = Vec::new();
425                let mut stop_strings: Vec<String> = Vec::new();
426
427                let (tok_env, tokenizer) = {
428                    let pipeline = get_mut_arcmutex!(self.pipeline);
429                    let tok_env = pipeline.get_metadata().tok_env.clone();
430                    let tokenizer = pipeline.tokenizer();
431                    (tok_env, tokenizer)
432                };
433
434                for stop_txt in s {
435                    let Some(tokenizer) = &tokenizer else {
436                        request
437                            .response
438                            .send(Response::ValidationError(
439                                "Completion requests require the pipeline to have a tokenizer"
440                                    .into(),
441                            ))
442                            .await
443                            .expect("Expected receiver.");
444                        return;
445                    };
446                    let encoded = tokenizer.encode_fast(stop_txt.to_string(), true);
447                    let toks = handle_seq_error!(encoded, request.response)
448                        .get_ids()
449                        .to_vec();
450
451                    if toks.len() == 1 {
452                        if tok_env.as_ref().is_some_and(|tok_env| {
453                            let tok_trie = tok_env.tok_trie();
454                            tok_trie.has_extensions(tok_trie.token(toks[0]))
455                        }) {
456                            stop_strings.push(stop_txt.clone());
457                        } else {
458                            stop_toks.push(toks[0]);
459                        }
460                    } else {
461                        stop_strings.push(stop_txt.clone());
462                    }
463                }
464
465                (stop_toks, stop_strings)
466            }
467        };
468
469        let group = Arc::new(tokio::sync::Mutex::new(SequenceGroup::new(
470            request.sampling_params.n_choices,
471            request.is_streaming,
472            is_chat,
473            best_of,
474        )));
475
476        let tokenizer = get_mut_arcmutex!(self.pipeline).tokenizer();
477
478        let sampler = Sampler::new(
479            Some(request.sampling_params.temperature.unwrap_or(1.0)),
480            request.sampling_params.top_n_logprobs,
481            tokenizer,
482            request.sampling_params.frequency_penalty,
483            request.sampling_params.presence_penalty,
484            request.sampling_params.dry_params,
485            topk,
486            topp,
487            minp,
488            request.logits_processors.unwrap_or_default(),
489        );
490        let sampler = handle_seq_error!(sampler, request.response);
491
492        if request.sampling_params.n_choices == 0 {
493            request
494                .response
495                .send(Response::ValidationError(
496                    "Number of choices must be greater than 0.".into(),
497                ))
498                .await
499                .expect("Expected receiver.");
500            return;
501        }
502
503        // Add sequences
504        for response_index in 0..request.sampling_params.n_choices {
505            let trie = get_mut_arcmutex!(self.pipeline)
506                .get_metadata()
507                .tok_env
508                .clone();
509            let recognizer = match Self::build_sequence_recognizer(&trie, &request.constraint) {
510                Ok(recognizer) => recognizer,
511                Err(err) => {
512                    request
513                        .response
514                        .send(Response::ValidationError(
515                            format!("Invalid grammar. {}", err).into(),
516                        ))
517                        .await
518                        .expect("Expected receiver.");
519                    return;
520                }
521            };
522
523            let block_size = get_mut_arcmutex!(self.pipeline)
524                .get_metadata()
525                .cache_config
526                .clone()
527                .map(|conf| conf.block_size);
528
529            let eos_toks = get_mut_arcmutex!(self.pipeline)
530                .get_metadata()
531                .eos_tok
532                .clone();
533
534            let seq_preallocated_cache = if get_mut_arcmutex!(self.pipeline).do_preallocated_cache()
535            {
536                let metadata = get_mut_arcmutex!(self.pipeline).get_metadata();
537                let model_metadata = metadata
538                    .model_metadata
539                    .as_ref()
540                    .expect("If a model has a NormalCache it must have a model metadata");
541                let n_tokens = prompt_tokens.len();
542                let required_blocks = n_tokens.div_ceil(NormalCache::CACHE_GROW_SIZE);
543                let max_seq_len = required_blocks * NormalCache::CACHE_GROW_SIZE;
544                let k_shape = (
545                    1usize,
546                    model_metadata.num_kv_heads(),
547                    max_seq_len,
548                    model_metadata.k_head_dim(),
549                );
550                let v_shape = (
551                    1usize,
552                    model_metadata.num_kv_heads(),
553                    max_seq_len,
554                    model_metadata.v_head_dim(),
555                );
556                let dtype = get_mut_arcmutex!(self.pipeline)
557                    .get_metadata()
558                    .activation_dtype;
559
560                let k_seq_cache = {
561                    let k_seq_cache =
562                        Tensor::zeros(k_shape, dtype, &get_mut_arcmutex!(self.pipeline).device());
563                    match k_seq_cache {
564                        Ok(x) => x,
565                        Err(_) => {
566                            request
567                                .response
568                                .send(Response::InternalError(
569                                    "Failed to allocate preallocated KV cache."
570                                        .to_string()
571                                        .into(),
572                                ))
573                                .await
574                                .expect("Expected receiver.");
575                            return;
576                        }
577                    }
578                };
579                let v_seq_cache = if k_shape == v_shape {
580                    k_seq_cache.clone()
581                } else {
582                    let v_seq_cache =
583                        Tensor::zeros(v_shape, dtype, &get_mut_arcmutex!(self.pipeline).device());
584                    match v_seq_cache {
585                        Ok(x) => x,
586                        Err(_) => {
587                            request
588                                .response
589                                .send(Response::InternalError(
590                                    "Failed to allocate preallocated KV cache."
591                                        .to_string()
592                                        .into(),
593                                ))
594                                .await
595                                .expect("Expected receiver.");
596                            return;
597                        }
598                    }
599                };
600                Some((k_seq_cache, v_seq_cache))
601            } else {
602                None
603            };
604
605            let now = SystemTime::now()
606                .duration_since(UNIX_EPOCH)
607                .expect("Time travel has occurred!");
608            let seq = Sequence::new_waiting(
609                prompt_tokens.clone(),
610                prompt_text.clone(),
611                *get_mut_arcmutex!(self.id).deref(),
612                now.as_millis(),
613                num_hidden_layers,
614                request.response.clone(),
615                sampler.clone(),
616                stop_toks.clone(),
617                stop_strings.clone(),
618                request.sampling_params.max_len,
619                request.return_logprobs,
620                get_mut_arcmutex!(self.pipeline).get_metadata().is_xlora,
621                group.clone(),
622                response_index,
623                now.as_secs(),
624                recognizer,
625                request.suffix.clone(),
626                if echo_prompt {
627                    Some(prompt_text.clone())
628                } else {
629                    None
630                },
631                request.adapters.clone(),
632                images.clone(),
633                block_size,
634                Some(matcher.clone()),
635                image_generation_format,
636                seq_step_type,
637                diffusion_params.clone(),
638                seq_preallocated_cache,
639                request.return_raw_logits,
640                eos_toks,
641            );
642            self.logger.add_new_sequence();
643            let seq = if let Some(prefill_cache) = prefill_cache.clone() {
644                self.logger.add_prefix_cache_hit();
645
646                seq.prefill_v2(
647                    prefill_cache.normal,
648                    prefill_cache.toks,
649                    prefill_cache.offset,
650                )
651            } else {
652                seq
653            };
654            *get_mut_arcmutex!(self.id) += 1;
655            get_mut_arcmutex!(self.scheduler).add_seq(seq);
656        }
657    }

659    async fn tokenize_text(&self, request: TokenizationRequest) {
660        match request.text {
661            Either::Left(messages) => {
662                let pipeline = &*get_mut_arcmutex!(self.pipeline);
663                let tools = request.tools.unwrap_or_default();
664                let template = pipeline.get_processor().process(
665                    pipeline,
666                    messages,
667                    request.add_generation_prompt,
668                    request.add_special_tokens,
669                    tools,
670                );
671                let toks = match template {
672                    Ok((toks, _)) => toks,
673                    Err(e) => {
674                        request
675                            .response
676                            .send(Err(e))
677                            .await
678                            .expect("Expected receiver.");
679                        return;
680                    }
681                };
682                request
683                    .response
684                    .send(Ok(toks))
685                    .await
686                    .expect("Sender disconnected unexpectedly!");
687            }
688            Either::Right(text) => {
689                let pipeline = &*get_mut_arcmutex!(self.pipeline);
690                let tokenizer = pipeline.tokenizer();
691                let tokenizer = match tokenizer {
692                    Some(tokenizer) => tokenizer,
693                    None => {
694                        request
695                            .response
696                            .send(Err(anyhow::Error::msg(
697                                "Pipeline does not include a toksnizer.",
698                            )))
699                            .await
700                            .expect("Expected receiver.");
701                        return;
702                    }
703                };
704                let toks = tokenizer.encode_fast(text, request.add_special_tokens);
705                let toks = match toks {
706                    Ok(tokenizer) => tokenizer,
707                    Err(e) => {
708                        request
709                            .response
710                            .send(Err(anyhow::Error::msg(e)))
711                            .await
712                            .expect("Expected receiver.");
713                        return;
714                    }
715                };
716                request
717                    .response
718                    .send(Ok(toks.get_ids().to_vec()))
719                    .await
720                    .expect("Sender disconnected unexpectedly!");
721            }
722        };
723    }

725    async fn detokenize_text(&self, request: DetokenizationRequest) {
726        let pipeline = &*get_mut_arcmutex!(self.pipeline);
727        let tokenizer = pipeline.tokenizer();
728        let tokenizer = match tokenizer {
729            Some(tokenizer) => tokenizer,
730            None => {
731                request
732                    .response
733                    .send(Err(anyhow::Error::msg(
734                        "Pipeline does not include a toksnizer.",
735                    )))
736                    .await
737                    .expect("Expected receiver.");
738                return;
739            }
740        };
741        let txt = tokenizer.decode(&request.tokens, request.skip_special_tokens);
742        let txt = match txt {
743            Ok(tokenizer) => tokenizer,
744            Err(e) => {
745                request
746                    .response
747                    .send(Err(anyhow::Error::msg(e)))
748                    .await
749                    .expect("Expected receiver.");
750                return;
751            }
752        };
753        request
754            .response
755            .send(Ok(txt))
756            .await
757            .expect("Sender disconnected unexpectedly!");
758    }
