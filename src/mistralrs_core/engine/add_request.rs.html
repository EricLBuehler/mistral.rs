29    pub async fn handle_request(self: Arc<Self>, request: Request) {
30        match request {
31            Request::Normal(request) => {
32                if matches!(
33                    request.messages,
34                    RequestMessage::Chat { .. } | RequestMessage::VisionChat { .. }
35                ) && request.web_search_options.is_some()
36                    || !self.tool_callbacks.is_empty()
37                    || !self.tool_callbacks_with_tools.is_empty()
38                {
39                    search_request::search_request(self.clone(), *request).await;
40                } else {
41                    self.add_request(*request).await
42                }
43            }
44            Request::ReIsq(level) => {
45                if let Err(e) = get_mut_arcmutex!(self.pipeline).re_isq_model(level) {
46                    warn!("ISQ requantization failed: {e:?}");
47                }
48            }
49            Request::Tokenize(req) => self.tokenize_text(req).await,
50            Request::Detokenize(req) => self.detokenize_text(req).await,
51            Request::Terminate => (),
52            Request::TerminateAllSeqsNextStep => {
53                TERMINATE_ALL_NEXT_STEP.store(true, Ordering::SeqCst)
54            }
55        }
56    }

58    pub(super) async fn add_request(&self, request: NormalRequest) {
59        let is_chat = matches!(
60            request.messages,
61            RequestMessage::Chat { .. } | RequestMessage::VisionChat { .. }
62        );
63        let echo_prompt = matches!(
64            request.messages,
65            RequestMessage::Completion {
66                echo_prompt: true,
67                ..
68            }
69        );
70
71        let best_of = match request.messages {
72            RequestMessage::Completion { best_of, .. } => best_of,
73            RequestMessage::Chat { .. }
74            | RequestMessage::CompletionTokens(_)
75            | RequestMessage::VisionChat { .. }
76            | RequestMessage::ImageGeneration { .. }
77            | RequestMessage::SpeechGeneration { .. } => None,
78        };
79        if is_chat
80            && !get_mut_arcmutex!(self.pipeline)
81                .get_chat_template()
82                .as_ref()
83                .is_some_and(|ch_t| ch_t.has_chat_template())
84        {
85            request
86                    .response
87                    .send(Response::ValidationError(
88                        "Received messages for a model which does not have a chat template. Either use a different model or pass a single string as the prompt".into(),
89                    ))
90                    .await
91                    .unwrap_or_else(|_| warn!("Receiver disconnected"));
92            return;
93        }
94
95        // Verify the model's category matches the messages received.
96        match (
97            get_mut_arcmutex!(self.pipeline).category(),
98            &request.messages,
99        ) {
100            (
101                ModelCategory::Text | ModelCategory::Vision { .. },
102                RequestMessage::Chat { .. }
103                | RequestMessage::VisionChat { .. }
104                | RequestMessage::Completion { .. }
105                | RequestMessage::CompletionTokens(_),
106            ) => (),
107            (ModelCategory::Diffusion, RequestMessage::ImageGeneration { .. }) => (),
108            (ModelCategory::Speech, RequestMessage::SpeechGeneration { .. }) => (),
109            _ => {
110                request
111                    .response
112                    .send(Response::ValidationError(
113                        "Received a request incompatible for this model's category.".into(),
114                    ))
115                    .await
116                    .unwrap_or_else(|_| warn!("Receiver disconnected"));
117                return;
118            }
119        }
120
121        let images = match request.messages {
122            RequestMessage::VisionChat {
123                ref images,
124                messages: _,
125                enable_thinking: _,
126                audios: _,
127            } => Some(images.clone()),
128            _ => None,
129        };
130
131        let audios = match request.messages {
132            RequestMessage::VisionChat {
133                images: _,
134                messages: _,
135                enable_thinking: _,
136                ref audios,
137            } => Some(audios.clone()),
138            _ => None,
139        };
140
141        let matcher = Arc::new(handle_seq_error!(
142            ToolCallingMatcher::new(request.tool_choice.unwrap_or(ToolChoice::Auto),),
143            request.response
144        ));
145
146        let image_generation_format = match &request.messages {
147            RequestMessage::ImageGeneration { format, .. } => Some(*format),
148            _ => None,
149        };
150
151        let seq_step_type = match &request.messages {
152            RequestMessage::ImageGeneration { .. } | RequestMessage::SpeechGeneration { .. } => {
153                SeqStepType::OneShot
154            }
155            _ => SeqStepType::PromptAndDecode,
156        };
157
158        let diffusion_params = match &request.messages {
159            RequestMessage::ImageGeneration {
160                generation_params, ..
161            } => Some(generation_params.clone()),
162            _ => None,
163        };
164
165        let (mut prompt_tokens, prompt_text) = match request.messages {
166            RequestMessage::Chat {
167                messages,
168                enable_thinking,
169            }
170            | RequestMessage::VisionChat {
171                images: _,
172                audios: _,
173                messages,
174                enable_thinking,
175            } => {
176                let pipeline = &*get_mut_arcmutex!(self.pipeline);
177                let tools = request.tools.unwrap_or_default();
178                let template = pipeline.get_processor().process(
179                    pipeline,
180                    messages,
181                    true,
182                    true,
183                    enable_thinking,
184                    tools,
185                );
186                handle_seq_error!(template, request.response)
187            }
188            RequestMessage::Completion { text, .. } => {
189                let Some(tokenizer) = &get_mut_arcmutex!(self.pipeline).tokenizer() else {
190                    request
191                        .response
192                        .send(Response::ValidationError(
193                            "Completion requests require the pipeline to have a tokenizer".into(),
194                        ))
195                        .await
196                        .unwrap_or_else(|_| warn!("Receiver disconnected"));
197                    return;
198                };
199                let prompt = tokenizer
200                    .encode_fast(text.clone(), true)
201                    .map_err(anyhow::Error::msg);
202                (
203                    handle_seq_error!(prompt, request.response)
204                        .get_ids()
205                        .to_vec(),
206                    text,
207                )
208            }
209            RequestMessage::ImageGeneration { prompt, .. }
210            | RequestMessage::SpeechGeneration { prompt } => (vec![u32::MAX], prompt),
211            RequestMessage::CompletionTokens(it) => {
212                let Some(tokenizer) = &get_mut_arcmutex!(self.pipeline).tokenizer() else {
213                    request
214                        .response
215                        .send(Response::ValidationError(
216                            "Completion requests w/ raw tokens require the pipeline to have a tokenizer".into(),
217                        ))
218                        .await
219                        .unwrap_or_else(|_| warn!("Receiver disconnected"));
220                    return;
221                };
222                let prompt = tokenizer
223                    .decode(&it, false)
224                    .map_err(|e| anyhow::Error::msg(e.to_string()));
225                (it, handle_seq_error!(prompt, request.response))
226            }
227        };
228        if prompt_tokens.is_empty() {
229            request
230                .response
231                .send(Response::ValidationError(
232                    "Received an empty prompt.".into(),
233                ))
234                .await
235                .unwrap_or_else(|_| warn!("Receiver disconnected"));
236            return;
237        }
238
239        if prompt_tokens.len() > get_mut_arcmutex!(self.pipeline).get_metadata().max_seq_len {
240            if !self.truncate_sequence {
241                request
242                    .response
243                    .send(Response::ValidationError(
244                        format!("Prompt sequence length is greater than {}, perhaps consider using `truncate_sequence`?", get_mut_arcmutex!(self.pipeline).get_metadata().max_seq_len).into(),
245                    ))
246                    .await
247                    .unwrap_or_else(|_| warn!("Receiver disconnected"));
248                return;
249            } else {
250                let prompt_len = prompt_tokens.len();
251                let max_len = get_mut_arcmutex!(self.pipeline).get_metadata().max_seq_len;
252                let currently_over = prompt_len - max_len;
253
254                // Reserve space for generation tokens
255                // If user specified max_len (generation length), reserve that many tokens (capped to max_len)
256                // Otherwise, reserve just 1 token minimum to allow at least some generation
257                let sampling_max = if let Some(sampling_max) = request.sampling_params.max_len {
258                    sampling_max.min(max_len)
259                } else {
260                    1
261                };
262
263                // Calculate how many prompt tokens to keep: max_len - sampling_max
264                // This ensures we have room for generation
265                let tokens_to_keep = max_len.saturating_sub(sampling_max);
266
267                // Safely calculate slice start position - keep the end of the prompt
268                let slice_start = prompt_len.saturating_sub(tokens_to_keep);
269
270                prompt_tokens = prompt_tokens[slice_start..].to_vec();
271                warn!("Prompt for request {} was {} tokens over the model maximum length. The last {} tokens were truncated to make space for generation.", request.id, currently_over, prompt_len - prompt_tokens.len());
272            }
273        }
274
275        let topk = request
276            .sampling_params
277            .top_k
278            .map(|x| x as i64)
279            .unwrap_or(-1);
280        let topp = request.sampling_params.top_p.unwrap_or(1.0);
281        let minp = request.sampling_params.min_p.unwrap_or(0.0);
282        let num_hidden_layers = get_mut_arcmutex!(self.pipeline)
283            .get_metadata()
284            .num_hidden_layers;
285
286        let (stop_toks, stop_strings) = match request.sampling_params.stop_toks {
287            None => (vec![], vec![]),
288            Some(StopTokens::Ids(ref i)) => {
289                let tok_env = {
290                    let pipeline = get_mut_arcmutex!(self.pipeline);
291                    pipeline.get_metadata().tok_env()
292                };
293                for id in i {
294                    // We can't use ` ` (space) as a stop token because other tokens like ` moon` start with a space.
295                    if let Some(tok_env) = tok_env.as_ref() {
296                        let tok_trie = tok_env.tok_trie();
297                        if tok_trie.has_extensions(tok_trie.token(*id)) {
298                            request
299                                .response
300                                .send(Response::ValidationError(
301                                    format!("Stop token {:?} is also a prefix of other tokens and cannot be used as a stop token.", tok_trie.token_str(*id)).into(),
302                                ))
303                                .await
304                                .unwrap_or_else(|_| warn!("Receiver disconnected"));
305                            return;
306                        }
307                    }
308                }
309
310                (i.clone(), vec![])
311            }
312            Some(StopTokens::Seqs(ref s)) => {
313                let mut stop_toks = Vec::new();
314                let mut stop_strings: Vec<String> = Vec::new();
315
316                let (tok_env, tokenizer) = {
317                    let pipeline = get_mut_arcmutex!(self.pipeline);
318                    let tok_env = pipeline.get_metadata().tok_env();
319                    let tokenizer = pipeline.tokenizer();
320                    (tok_env, tokenizer)
321                };
322
323                for stop_txt in s {
324                    let Some(tokenizer) = &tokenizer else {
325                        request
326                            .response
327                            .send(Response::ValidationError(
328                                "Completion requests require the pipeline to have a tokenizer"
329                                    .into(),
330                            ))
331                            .await
332                            .unwrap_or_else(|_| warn!("Receiver disconnected"));
333                        return;
334                    };
335                    let encoded = tokenizer.encode_fast(stop_txt.to_string(), true);
336                    let toks = handle_seq_error!(encoded, request.response)
337                        .get_ids()
338                        .to_vec();
339
340                    if toks.len() == 1 {
341                        if tok_env.as_ref().is_some_and(|tok_env| {
342                            let tok_trie = tok_env.tok_trie();
343                            tok_trie.has_extensions(tok_trie.token(toks[0]))
344                        }) {
345                            stop_strings.push(stop_txt.clone());
346                        } else {
347                            stop_toks.push(toks[0]);
348                        }
349                    } else {
350                        stop_strings.push(stop_txt.clone());
351                    }
352                }
353
354                (stop_toks, stop_strings)
355            }
356        };
357
358        let group = Arc::new(tokio::sync::Mutex::new(SequenceGroup::new(
359            request.sampling_params.n_choices,
360            request.is_streaming,
361            is_chat,
362            best_of,
363        )));
364
365        let tokenizer = get_mut_arcmutex!(self.pipeline).tokenizer();
366
367        let sampler = Sampler::new(
368            Some(request.sampling_params.temperature.unwrap_or(1.0)),
369            request.sampling_params.top_n_logprobs,
370            tokenizer,
371            request.sampling_params.frequency_penalty,
372            request.sampling_params.presence_penalty,
373            request.sampling_params.repetition_penalty,
374            request.sampling_params.dry_params,
375            topk,
376            topp,
377            minp,
378            request.logits_processors.unwrap_or_default(),
379        );
380        let sampler = handle_seq_error!(sampler, request.response);
381
382        if request.sampling_params.n_choices == 0 {
383            request
384                .response
385                .send(Response::ValidationError(
386                    "Number of choices must be greater than 0.".into(),
387                ))
388                .await
389                .unwrap_or_else(|_| warn!("Receiver disconnected"));
390            return;
391        }
392
393        // Add sequences
394        for response_index in 0..request.sampling_params.n_choices {
395            let factory = get_mut_arcmutex!(self.pipeline)
396                .get_metadata()
397                .llg_factory
398                .clone();
399            let recognizer = match Self::build_sequence_recognizer(&factory, &request.constraint) {
400                Ok(recognizer) => recognizer,
401                Err(err) => {
402                    request
403                        .response
404                        .send(Response::ValidationError(
405                            format!("Invalid grammar. {err}").into(),
406                        ))
407                        .await
408                        .unwrap_or_else(|_| warn!("Receiver disconnected"));
409                    return;
410                }
411            };
412
413            let block_size = get_mut_arcmutex!(self.pipeline)
414                .get_metadata()
415                .cache_config
416                .clone()
417                .map(|conf| conf.block_size);
418
419            let eos_toks = get_mut_arcmutex!(self.pipeline)
420                .get_metadata()
421                .eos_tok
422                .clone();
423
424            let seq_preallocated_cache = if get_mut_arcmutex!(self.pipeline).do_preallocated_cache()
425            {
426                let metadata = get_mut_arcmutex!(self.pipeline).get_metadata();
427                let model_metadata = metadata
428                    .model_metadata
429                    .as_ref()
430                    .expect("If a model has a NormalCache it must have a model metadata");
431                let n_tokens = prompt_tokens.len();
432                let required_blocks = n_tokens.div_ceil(NormalCache::CACHE_GROW_SIZE);
433                let max_seq_len = required_blocks * NormalCache::CACHE_GROW_SIZE;
434                let k_shape = (
435                    1usize,
436                    model_metadata.num_kv_heads(),
437                    max_seq_len,
438                    model_metadata.k_head_dim(),
439                );
440                let v_shape = (
441                    1usize,
442                    model_metadata.num_kv_heads(),
443                    max_seq_len,
444                    model_metadata.v_head_dim(),
445                );
446                let dtype = get_mut_arcmutex!(self.pipeline)
447                    .get_metadata()
448                    .activation_dtype;
449
450                let k_seq_cache = {
451                    let k_seq_cache =
452                        Tensor::zeros(k_shape, dtype, &get_mut_arcmutex!(self.pipeline).device());
453                    match k_seq_cache {
454                        Ok(x) => x,
455                        Err(_) => {
456                            request
457                                .response
458                                .send(Response::InternalError(
459                                    "Failed to allocate preallocated KV cache."
460                                        .to_string()
461                                        .into(),
462                                ))
463                                .await
464                                .unwrap_or_else(|_| warn!("Receiver disconnected"));
465                            return;
466                        }
467                    }
468                };
469                let v_seq_cache = if k_shape == v_shape {
470                    k_seq_cache.clone()
471                } else {
472                    let v_seq_cache =
473                        Tensor::zeros(v_shape, dtype, &get_mut_arcmutex!(self.pipeline).device());
474                    match v_seq_cache {
475                        Ok(x) => x,
476                        Err(_) => {
477                            request
478                                .response
479                                .send(Response::InternalError(
480                                    "Failed to allocate preallocated KV cache."
481                                        .to_string()
482                                        .into(),
483                                ))
484                                .await
485                                .unwrap_or_else(|_| warn!("Receiver disconnected"));
486                            return;
487                        }
488                    }
489                };
490                Some((k_seq_cache, v_seq_cache))
491            } else {
492                None
493            };
494
495            let now = SystemTime::now()
496                .duration_since(UNIX_EPOCH)
497                .expect("Time travel has occurred!");
498            let mut seq = Sequence::new_waiting(
499                prompt_tokens.clone(),
500                prompt_text.clone(),
501                *get_mut_arcmutex!(self.id).deref(),
502                now.as_millis(),
503                num_hidden_layers,
504                request.response.clone(),
505                sampler.clone(),
506                stop_toks.clone(),
507                stop_strings.clone(),
508                request.sampling_params.max_len,
509                request.return_logprobs,
510                get_mut_arcmutex!(self.pipeline).get_metadata().is_xlora,
511                group.clone(),
512                response_index,
513                now.as_secs(),
514                recognizer,
515                request.suffix.clone(),
516                if echo_prompt {
517                    Some(prompt_text.clone())
518                } else {
519                    None
520                },
521                images.clone(),
522                audios.clone(),
523                block_size,
524                Some(matcher.clone()),
525                image_generation_format,
526                seq_step_type,
527                diffusion_params.clone(),
528                seq_preallocated_cache,
529                request.return_raw_logits,
530                eos_toks,
531            );
532
533            // Only "track" a new sequence if it is a traditional one
534            if matches!(seq_step_type, SeqStepType::PromptAndDecode) {
535                self.logger.add_new_sequence();
536            }
537
538            // Run the inputs processor to update the prompt for multimodal models.
539            if images.is_some() || audios.is_some() {
540                let pipeline = get_mut_arcmutex!(self.pipeline);
541                let _ = pipeline.get_processor().inputs_processor().process_inputs(
542                    pipeline.tokenizer(),
543                    &mut [&mut seq],
544                    true,
545                    pipeline.get_metadata().is_xlora,
546                    &pipeline.device(),
547                    pipeline.get_metadata().no_kv_cache,
548                    None,
549                    false,
550                    pipeline.get_input_processor_config(),
551                    None,
552                    pipeline.device_mapper(),
553                );
554            }
555
556            let prefill_cache = handle_seq_error!(
557                get_mut_arcmutex!(self.prefix_cacher).search_for_matching_cache(
558                    seq.get_toks(),
559                    seq.image_hashes(),
560                    seq.audio_hashes(),
561                ),
562                request.response
563            );
564
565            seq = match prefill_cache.clone() {
566                Some(MatchingCache::Normal {
567                    normal,
568                    images_to_keep,
569                    audios_to_keep,
570                    toks,
571                    offset,
572                }) => {
573                    self.logger.add_prefix_cache_hit();
574
575                    seq.keep_num_images(images_to_keep);
576                    seq.keep_num_audios(audios_to_keep);
577                    seq.prefill_v2_normal(normal, toks, offset)
578                }
579                Some(MatchingCache::Paged {
580                    logical_blocks,
581                    physical_blocks,
582                    images_to_keep,
583                    audios_to_keep,
584                    toks,
585                    offset,
586                }) => {
587                    self.logger.add_prefix_cache_hit();
588
589                    seq.keep_num_images(images_to_keep);
590                    seq.keep_num_audios(audios_to_keep);
591                    seq.prefill_v2_paged(logical_blocks, physical_blocks, toks, offset)
592                }
593                None => seq,
594            };
595
596            *get_mut_arcmutex!(self.id) += 1;
597            get_mut_arcmutex!(self.scheduler).add_seq(seq);
598        }
599    }

601    async fn tokenize_text(&self, request: TokenizationRequest) {
602        match request.text {
603            Either::Left(messages) => {
604                let pipeline = &*get_mut_arcmutex!(self.pipeline);
605                let tools = request.tools.unwrap_or_default();
606                let template = pipeline.get_processor().process(
607                    pipeline,
608                    messages,
609                    request.add_generation_prompt,
610                    request.add_special_tokens,
611                    request.enable_thinking,
612                    tools,
613                );
614                let toks = match template {
615                    Ok((toks, _)) => toks,
616                    Err(e) => {
617                        request
618                            .response
619                            .send(Err(e))
620                            .await
621                            .unwrap_or_else(|_| warn!("Receiver disconnected"));
622                        return;
623                    }
624                };
625                request
626                    .response
627                    .send(Ok(toks))
628                    .await
629                    .expect("Sender disconnected unexpectedly!");
630            }
631            Either::Right(text) => {
632                let pipeline = &*get_mut_arcmutex!(self.pipeline);
633                let tokenizer = pipeline.tokenizer();
634                let tokenizer = match tokenizer {
635                    Some(tokenizer) => tokenizer,
636                    None => {
637                        request
638                            .response
639                            .send(Err(anyhow::Error::msg(
640                                "Pipeline does not include a toksnizer.",
641                            )))
642                            .await
643                            .unwrap_or_else(|_| warn!("Receiver disconnected"));
644                        return;
645                    }
646                };
647                let toks = tokenizer.encode_fast(text, request.add_special_tokens);
648                let toks = match toks {
649                    Ok(tokenizer) => tokenizer,
650                    Err(e) => {
651                        request
652                            .response
653                            .send(Err(anyhow::Error::msg(e)))
654                            .await
655                            .unwrap_or_else(|_| warn!("Receiver disconnected"));
656                        return;
657                    }
658                };
659                request
660                    .response
661                    .send(Ok(toks.get_ids().to_vec()))
662                    .await
663                    .expect("Sender disconnected unexpectedly!");
664            }
665        };
666    }

668    async fn detokenize_text(&self, request: DetokenizationRequest) {
669        let pipeline = &*get_mut_arcmutex!(self.pipeline);
670        let tokenizer = pipeline.tokenizer();
671        let tokenizer = match tokenizer {
672            Some(tokenizer) => tokenizer,
673            None => {
674                request
675                    .response
676                    .send(Err(anyhow::Error::msg(
677                        "Pipeline does not include a toksnizer.",
678                    )))
679                    .await
680                    .unwrap_or_else(|_| warn!("Receiver disconnected"));
681                return;
682            }
683        };
684        let txt = tokenizer.decode(&request.tokens, request.skip_special_tokens);
685        let txt = match txt {
686            Ok(tokenizer) => tokenizer,
687            Err(e) => {
688                request
689                    .response
690                    .send(Err(anyhow::Error::msg(e)))
691                    .await
692                    .unwrap_or_else(|_| warn!("Receiver disconnected"));
693                return;
694            }
695        };
696        request
697            .response
698            .send(Ok(txt))
699            .await
700            .expect("Sender disconnected unexpectedly!");
701    }
