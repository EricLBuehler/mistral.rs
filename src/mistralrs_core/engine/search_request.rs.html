19async fn do_search(
20    this: Arc<Engine>,
21    mut second_request: NormalRequest,
22    tool_calls: &ToolCallResponse,
23    web_search_options: &WebSearchOptions,
24) -> NormalRequest {
25    let messages = match &mut second_request.messages {
26        RequestMessage::Chat { messages, .. } | RequestMessage::VisionChat { messages, .. } => {
27            messages
28        }
29        _ => unreachable!(),
30    };
31
32    // Add assistant call message
33    {
34        let mut message: IndexMap<String, MessageContent> = IndexMap::new();
35        message.insert("role".to_string(), Either::Left("assistant".to_string()));
36        message.insert(
37            "content".to_string(),
38            Either::Left(format!(
39                "{{\"name\":\"{}\",\"arguments\":\"{}\"}}",
40                tool_calls.function.name, tool_calls.function.arguments
41            )),
42        );
43        messages.push(message);
44    }
45    let tool_call_params: SearchFunctionParameters =
46        serde_json::from_str(&tool_calls.function.arguments).unwrap();
47    println!();
48    tracing::info!(
49        "Called search tool with query `{}`.",
50        tool_call_params.query
51    );
52
53    let start = Instant::now();
54    // Add tool response
55    {
56        let tokenizer = get_mut_arcmutex!(this.pipeline)
57            .tokenizer()
58            .expect("A tokenizer is expected for non-diffusion models.");
59
60        // Allow `info` and below; suppress `warn`
61        let subscriber = tracing_subscriber::fmt::Subscriber::builder()
62            .with_max_level(LevelFilter::INFO)
63            .finish();
64        let dispatch = Dispatch::new(subscriber);
65
66        // Manage context size by # of tokens. Apply default here.
67        let max_results_budget_toks =
68            match web_search_options.search_context_size.unwrap_or_default() {
69                SearchContextSize::High => 16384_usize,
70                SearchContextSize::Medium => 8192_usize,
71                SearchContextSize::Low => 4096_usize,
72            };
73        let mut results = tokio::task::block_in_place(|| {
74            tracing::dispatcher::with_default(&dispatch, || {
75                let base_results = if let Some(cb) = &this.search_callback {
76                    cb(&tool_call_params).unwrap()
77                } else {
78                    search::run_search_tool(&tool_call_params).unwrap()
79                };
80                base_results
81                    .into_iter()
82                    .map(|mut result| {
83                        result = result
84                            .cap_content_len(&tokenizer, max_results_budget_toks)
85                            .unwrap();
86                        let len = {
87                            let inp = InputSequence::Raw(Cow::from(&result.content));
88                            tokenizer
89                                .encode_fast(inp, false)
90                                .map(|x| x.len())
91                                .unwrap_or(usize::MAX)
92                        };
93                        (result, len)
94                    })
95                    .collect::<Vec<_>>()
96            })
97        });
98
99        // Sort increasing by tokenized length, if it fails, put it at the end.
100        results.sort_by_key(|(_, len)| *len);
101
102        {
103            // Determine ranking: use embedding model if available, otherwise fallback to BM25
104            let decreasing_indexes: Vec<usize> = if let Some(bert_pipeline) =
105                &mut *get_mut_arcmutex!(this.bert_pipeline)
106            {
107                // Semantic reranking with embeddings
108                let device = get_mut_arcmutex!(this.pipeline).device();
109                search::rag::compute_most_similar(
110                    &device,
111                    &tool_call_params.query,
112                    results.iter().map(|(res, _)| res).collect::<Vec<_>>(),
113                    bert_pipeline,
114                )
115                .unwrap()
116            } else {
117                tracing::warn!("No embedding model loaded; falling back to BM25 ranking for web search results.");
118
119                // Build an Embedder over the corpus, fitting to the entire set of documents.
120                //    - Language::English is chosen here
121                //    - This computes an in‑memory sparse embedding for each document.
122
123                let docs: Vec<String> =
124                    results.iter().map(|(res, _)| res.content.clone()).collect();
125                let doc_refs: Vec<&str> = docs.iter().map(|s| s.as_str()).collect();
126
127                let embedder: Embedder =
128                    EmbedderBuilder::with_fit_to_corpus(Language::English, &doc_refs).build();
129
130                // Initialize a Scorer keyed by usize (document index type).
131                let mut scorer = Scorer::<usize>::new();
132
133                // For each document, compute its embedding and upsert into the scorer.
134                for (i, doc_text) in docs.iter().enumerate() {
135                    let doc_embedding = embedder.embed(doc_text);
136                    scorer.upsert(&i, doc_embedding);
137                }
138
139                // Embed the query string into the same sparse embedding space.
140                let query_embedding = embedder.embed(&tool_call_params.query);
141
142                // Score all documents individually
143                let mut scored_docs: Vec<ScoredDocument<usize>> = docs
144                    .iter()
145                    .enumerate()
146                    .filter_map(|(i, _)| {
147                        scorer
148                            .score(&i, &query_embedding)
149                            .map(|score| ScoredDocument { id: i, score })
150                    })
151                    .collect();
152
153                // Sort the scored documents by descending `score` (f32).
154                scored_docs.sort_by(|a, b| {
155                    b.score
156                        .partial_cmp(&a.score)
157                        .unwrap_or(std::cmp::Ordering::Equal)
158                });
159
160                // Extract only the document indices (usize) in ranked order.
161                let decreasing_indexes: Vec<usize> =
162                    scored_docs.into_iter().map(|d| d.id).collect();
163
164                decreasing_indexes
165            };
166            // Reorder results according to ranking
167            let mut old = Vec::new();
168            std::mem::swap(&mut old, &mut results);
169            for &idx in &decreasing_indexes {
170                let mut item: (SearchResult, usize) = Default::default();
171                std::mem::swap(&mut item, &mut old[idx]);
172                results.push(item);
173            }
174        }
175
176        let mut used_results = Vec::new();
177        let mut used_len = 0;
178        for (item, len) in results {
179            if used_len + len > max_results_budget_toks {
180                break;
181            }
182            // So the info! below gets the correct value
183            used_len += len;
184            used_results.push(item);
185        }
186
187        let tool_result = serde_json::to_string(&used_results)
188            .unwrap()
189            .replace("\\n", "\n")
190            .replace("\\\"", "\"")
191            .replace("\\\\", "\\");
192        let end = Instant::now();
193        tracing::info!(
194            "Web search executed in {:.2}s, using {used_len} tokens of {} search results.",
195            (end - start).as_secs_f32(),
196            used_results.len()
197        );
198
199        let mut message: IndexMap<String, MessageContent> = IndexMap::new();
200        message.insert("role".to_string(), Either::Left("tool".to_string()));
201        message.insert(
202            "content".to_string(),
203            Either::Left(
204                // Format the tool output JSON and append the search tool description for context
205                format!(
206                    "{{\"output\": \"{}\"}}\n\n{}\n\n{}",
207                    tool_result,
208                    search::SEARCH_DESCRIPTION,
209                    search::EXTRACT_DESCRIPTION,
210                ),
211            ),
212        );
213        messages.push(message);
214    }
215
216    // Allow the assistant to invoke tools again on the next turn
217    second_request.tool_choice = Some(ToolChoice::Auto);
218    // Recursion is enabled here!
219    second_request.web_search_options = Some(web_search_options.clone());
220
221    second_request
222}

333async fn do_custom_tool(
334    this: Arc<Engine>,
335    mut second_request: NormalRequest,
336    tool_calls: &ToolCallResponse,
337) -> NormalRequest {
338    let messages = match &mut second_request.messages {
339        RequestMessage::Chat { messages, .. } | RequestMessage::VisionChat { messages, .. } => {
340            messages
341        }
342        _ => unreachable!(),
343    };
344
345    {
346        let mut message: IndexMap<String, MessageContent> = IndexMap::new();
347        message.insert("role".to_string(), Either::Left("assistant".to_string()));
348        message.insert(
349            "content".to_string(),
350            Either::Left(format!(
351                "{{\"name\":\"{}\",\"arguments\":\"{}\"}}",
352                tool_calls.function.name, tool_calls.function.arguments
353            )),
354        );
355        messages.push(message);
356    }
357
358    let result = if let Some(cb) = this.tool_callbacks.get(&tool_calls.function.name) {
359        tracing::info!("Called tool `{}`.", tool_calls.function.name);
360        cb(&tool_calls.function).unwrap_or_else(|e| {
361            tracing::error!(
362                "Error when calling tool `{}`: {e}",
363                tool_calls.function.name
364            );
365            format!("ERROR: {e}")
366        })
367    } else if let Some(callback_with_tool) = this
368        .tool_callbacks_with_tools
369        .get(&tool_calls.function.name)
370    {
371        tracing::info!("Called tool `{}`.", tool_calls.function.name);
372        (callback_with_tool.callback)(&tool_calls.function).unwrap_or_else(|e| {
373            tracing::error!(
374                "Error when calling tool `{}`: {e}",
375                tool_calls.function.name
376            );
377            format!("ERROR: {e}")
378        })
379    } else {
380        tracing::error!(
381            "Attempted to call tool `{}`, but it doesn't exist.",
382            tool_calls.function.name
383        );
384        format!("ERROR: no tool callback for {}", tool_calls.function.name)
385    };
386
387    {
388        let mut message: IndexMap<String, MessageContent> = IndexMap::new();
389        message.insert("role".to_string(), Either::Left("tool".to_string()));
390        message.insert("content".to_string(), Either::Left(result));
391        messages.push(message);
392    }
393
394    second_request.tool_choice = Some(ToolChoice::Auto);
395    second_request
396}

398/// Drive one or more web-search / extraction rounds without recursion.
399///
400/// Strategy:
401/// 1. Send a “probe” request that may call the search/extract tools.  
402/// 2. If such a tool is called, run it (`do_search` / `do_extraction`) to
403///    mutate the conversational context and build the next request.  
404/// 3. Repeat until no further tool call is made.
405/// 4. Forward every user-visible reply **except** the first, which is just the
406///    probe that discovers whether a tool call is needed.
407pub(super) async fn search_request(this: Arc<Engine>, request: NormalRequest) {
408    let web_search_options = request.web_search_options.clone();
409
410    // The sender that ultimately delivers data back to the caller.
411    let user_sender = request.response.clone();
412    let is_streaming = request.is_streaming;
413
414    // ---------------------------------------------------------------------
415    // Build the *first* request (the “probe”).
416    // ---------------------------------------------------------------------
417    let mut probe = request.clone();
418    if let Some(ref opts) = web_search_options {
419        probe
420            .tools
421            .get_or_insert_with(Vec::new)
422            .extend(search::get_search_tools(opts).unwrap());
423    }
424
425    // Add Tool definitions from tool callbacks with tools if they're not already present
426    if !this.tool_callbacks_with_tools.is_empty() {
427        let tools = probe.tools.get_or_insert_with(Vec::new);
428        let existing_tool_names: Vec<String> =
429            tools.iter().map(|t| t.function.name.clone()).collect();
430
431        for (name, callback_with_tool) in &this.tool_callbacks_with_tools {
432            if !existing_tool_names.contains(name) {
433                tools.push(callback_with_tool.tool.clone());
434            }
435        }
436    }
437
438    probe.tool_choice = Some(ToolChoice::Auto);
439    // Prevent accidental infinite recursion on the probe itself.
440    probe.web_search_options = None;
441
442    // The conversation context that the user *will* see.
443    let mut visible_req = probe.clone();
444    visible_req.response = user_sender.clone();
445
446    // We'll drive everything inside a single spawned task.
447    let this_clone = this.clone();
448    let handle = tokio::spawn(async move {
449        // `current` is what we actually dispatch each loop.
450        // The very first time that is the hidden probe.
451        let mut current = probe;
452
453        loop {
454            // Each dispatch gets its own one-shot channel so we can peek at
455            // the response before (optionally) forwarding it.
456            let (sender, mut receiver) = tokio::sync::mpsc::channel(1);
457            current.response = sender;
458
459            // Kick the request into the engine.
460            this_clone.add_request(current).await;
461
462            // ----------------------- NON-STREAMING ------------------------
463            if !is_streaming {
464                let done = match receiver.recv().await.unwrap().as_result().unwrap() {
465                    ResponseOk::Done(done) => done,
466                    other => {
467                        match other {
468                            ResponseOk::Chunk(res) => {
469                                user_sender.send(Response::Chunk(res)).await.unwrap()
470                            }
471                            ResponseOk::CompletionChunk(res) => user_sender
472                                .send(Response::CompletionChunk(res))
473                                .await
474                                .unwrap(),
475                            ResponseOk::Done(_) => unreachable!(),
476                            ResponseOk::CompletionDone(res) => user_sender
477                                .send(Response::CompletionDone(res))
478                                .await
479                                .unwrap(),
480                            ResponseOk::ImageGeneration(res) => user_sender
481                                .send(Response::ImageGeneration(res))
482                                .await
483                                .unwrap(),
484                            ResponseOk::Raw {
485                                logits_chunks,
486                                tokens,
487                            } => user_sender
488                                .send(Response::Raw {
489                                    logits_chunks,
490                                    tokens,
491                                })
492                                .await
493                                .unwrap(),
494                            ResponseOk::Speech {
495                                pcm,
496                                rate,
497                                channels,
498                            } => user_sender
499                                .send(Response::Speech {
500                                    pcm,
501                                    rate,
502                                    channels,
503                                })
504                                .await
505                                .unwrap(),
506                        };
507                        return;
508                    }
509                };
510
511                // Did the assistant ask to run a tool?
512                let tc_opt = match &done.choices[0].message.tool_calls {
513                    Some(calls) if calls.len() == 1 => Some(&calls[0]),
514                    _ => None,
515                };
516
517                // No tool call? We are finished.
518                if tc_opt.is_none() {
519                    user_sender
520                        .send(Response::Done(done.clone()))
521                        .await
522                        .unwrap();
523                    return;
524                }
525
526                // Tool requested → build the next turn.
527                let tc = tc_opt.unwrap();
528                let next_visible = if search::search_tool_called(&tc.function.name) {
529                    let web_search_options = web_search_options.as_ref().unwrap();
530                    if tc.function.name == search::SEARCH_TOOL_NAME {
531                        do_search(this_clone.clone(), visible_req, tc, web_search_options).await
532                    } else {
533                        do_extraction(this_clone.clone(), visible_req, tc, web_search_options).await
534                    }
535                } else {
536                    do_custom_tool(this_clone.clone(), visible_req, tc).await
537                };
538
539                // The fresh request becomes both the user-visible context and
540                // the next `current` we will dispatch.
541                visible_req = next_visible.clone();
542                visible_req.response = user_sender.clone();
543                current = visible_req.clone();
544            }
545            // ------------------------- STREAMING -------------------------
546            else {
547                // We need the *last* chunk to see whether a tool was called.
548                let mut last_choice = None;
549
550                while let Some(resp) = receiver.recv().await {
551                    match resp.as_result().unwrap() {
552                        ResponseOk::Chunk(chunk) => {
553                            // Forward every content‑bearing chunk immediately, but
554                            // *suppress* the ones that initiate a tool call. This ensures
555                            // the user sees the assistant’s streamed text from the very
556                            // first probe turn while still hiding the internal
557                            // search/extract trigger.
558                            let first_choice = &chunk.choices[0];
559                            if first_choice.delta.tool_calls.is_none() {
560                                user_sender
561                                    .send(Response::Chunk(chunk.clone()))
562                                    .await
563                                    .unwrap();
564                            }
565
566                            last_choice = Some(first_choice.clone());
567
568                            // Stop once the model marks completion.
569                            if last_choice
570                                .as_ref()
571                                .and_then(|c| c.finish_reason.as_ref())
572                                .is_some()
573                            {
574                                break;
575                            }
576                        }
577                        other => {
578                            match other {
579                                ResponseOk::Chunk(_) => unreachable!(),
580                                ResponseOk::CompletionChunk(res) => user_sender
581                                    .send(Response::CompletionChunk(res))
582                                    .await
583                                    .unwrap(),
584                                ResponseOk::Done(res) => {
585                                    user_sender.send(Response::Done(res)).await.unwrap()
586                                }
587                                ResponseOk::CompletionDone(res) => user_sender
588                                    .send(Response::CompletionDone(res))
589                                    .await
590                                    .unwrap(),
591                                ResponseOk::ImageGeneration(res) => user_sender
592                                    .send(Response::ImageGeneration(res))
593                                    .await
594                                    .unwrap(),
595                                ResponseOk::Raw {
596                                    logits_chunks,
597                                    tokens,
598                                } => user_sender
599                                    .send(Response::Raw {
600                                        logits_chunks,
601                                        tokens,
602                                    })
603                                    .await
604                                    .unwrap(),
605                                ResponseOk::Speech {
606                                    pcm,
607                                    rate,
608                                    channels,
609                                } => user_sender
610                                    .send(Response::Speech {
611                                        pcm,
612                                        rate,
613                                        channels,
614                                    })
615                                    .await
616                                    .unwrap(),
617                            };
618                            return;
619                        }
620                    }
621                }
622
623                let Some(choice) = last_choice else { break };
624
625                let tc_opt = match &choice.delta.tool_calls {
626                    Some(calls) if calls.len() == 1 => Some(&calls[0]),
627                    _ => None,
628                };
629
630                if tc_opt.is_none() {
631                    break; // No more tool calls → done.
632                }
633
634                let tc = tc_opt.unwrap();
635                let next_visible = if search::search_tool_called(&tc.function.name) {
636                    let web_search_options = web_search_options.as_ref().unwrap();
637                    if tc.function.name == search::SEARCH_TOOL_NAME {
638                        do_search(this_clone.clone(), visible_req, tc, web_search_options).await
639                    } else {
640                        do_extraction(this_clone.clone(), visible_req, tc, web_search_options).await
641                    }
642                } else {
643                    do_custom_tool(this_clone.clone(), visible_req, tc).await
644                };
645
646                visible_req = next_visible.clone();
647                visible_req.response = user_sender.clone();
648                current = visible_req.clone();
649            }
650        }
651    });
652
653    get_mut_arcmutex!(this.handles).push(handle);
654}
