18async fn do_search(
19    this: Arc<Engine>,
20    mut second_request: NormalRequest,
21    tool_calls: &ToolCallResponse,
22    web_search_options: &WebSearchOptions,
23) {
24    let messages = match &mut second_request.messages {
25        RequestMessage::Chat { messages, .. } | RequestMessage::VisionChat { messages, .. } => {
26            messages
27        }
28        _ => unreachable!(),
29    };
30
31    // Add assistant call message
32    {
33        let mut message: IndexMap<String, MessageContent> = IndexMap::new();
34        message.insert("role".to_string(), Either::Left("assistant".to_string()));
35        message.insert(
36            "content".to_string(),
37            Either::Left(format!(
38                "{{\"name\":\"{}\",\"arguments\":\"{}\"}}",
39                tool_calls.function.name, tool_calls.function.arguments
40            )),
41        );
42        messages.push(message);
43    }
44    let tool_call_params: SearchFunctionParameters =
45        serde_json::from_str(&tool_calls.function.arguments).unwrap();
46    println!();
47    tracing::info!(
48        "Called search tool with query `{}`.",
49        tool_call_params.query
50    );
51
52    let start = Instant::now();
53    // Add tool response
54    {
55        let tokenizer = get_mut_arcmutex!(this.pipeline)
56            .tokenizer()
57            .expect("A tokenizer is expected for non-diffusion models.");
58
59        // Allow `info` and below; suppress `warn`
60        let subscriber = tracing_subscriber::fmt::Subscriber::builder()
61            .with_max_level(LevelFilter::INFO)
62            .finish();
63        let dispatch = Dispatch::new(subscriber);
64
65        // Manage context size by # of tokens. Apply default here.
66        let max_results_budget_toks =
67            match web_search_options.search_context_size.unwrap_or_default() {
68                SearchContextSize::High => 16384_usize,
69                SearchContextSize::Medium => 8192_usize,
70                SearchContextSize::Low => 4096_usize,
71            };
72        let mut results = tracing::dispatcher::with_default(&dispatch, || {
73            search::run_search_tool(&tool_call_params)
74                .unwrap()
75                .into_iter()
76                .map(|mut result| {
77                    result = result
78                        .cap_content_len(&tokenizer, max_results_budget_toks)
79                        .unwrap();
80                    let len = {
81                        let inp = InputSequence::Raw(Cow::from(&result.content));
82                        tokenizer
83                            .encode_fast(inp, false)
84                            .map(|x| x.len())
85                            .unwrap_or(usize::MAX)
86                    };
87                    (result, len)
88                })
89                .collect::<Vec<_>>()
90        });
91
92        // Sort increasing by tokenized length, if it fails, put it at the end.
93        results.sort_by_key(|(_, len)| *len);
94
95        {
96            let device = get_mut_arcmutex!(this.pipeline).device();
97
98            let Some(bert_pipeline) = &mut *get_mut_arcmutex!(this.bert_pipeline) else {
99                unreachable!()
100            };
101
102            let decreasing_indexes = search::rag::compute_most_similar(
103                &device,
104                &tool_call_params.query,
105                results.iter().map(|(res, _)| res).collect::<Vec<_>>(),
106                bert_pipeline,
107            )
108            .unwrap();
109
110            // Rerank the results
111            let mut results_old = Vec::new();
112            std::mem::swap(&mut results_old, &mut results);
113            for &index in &decreasing_indexes {
114                let mut current_result: (SearchResult, usize) = Default::default();
115                std::mem::swap(&mut current_result, &mut results_old[index]);
116
117                results.push(current_result);
118            }
119        }
120
121        let mut used_results = Vec::new();
122        let mut used_len = 0;
123        for (item, len) in results {
124            if used_len + len > max_results_budget_toks {
125                break;
126            }
127            // So the info! below gets the correct value
128            used_len += len;
129            used_results.push(item);
130        }
131
132        let tool_result = serde_json::to_string(&used_results)
133            .unwrap()
134            .replace("\\n", "\n")
135            .replace("\\\"", "\"")
136            .replace("\\\\", "\\");
137        let end = Instant::now();
138        tracing::info!(
139            "Web search executed in {:.2}s, using {used_len} tokens of {} search results.",
140            (end - start).as_secs_f32(),
141            used_results.len()
142        );
143
144        let mut message: IndexMap<String, MessageContent> = IndexMap::new();
145        message.insert("role".to_string(), Either::Left("tool".to_string()));
146        message.insert(
147            "content".to_string(),
148            Either::Left(format!("{{\"output\": \"{tool_result}\"}}")),
149        );
150        messages.push(message);
151    }
152
153    // Recursion is enabled here!
154    second_request.web_search_options = Some(web_search_options.clone());
155
156    this.add_request(second_request).await;
157}

159/// The strategy is:
160/// - Send the first request to allow a tool call
161///     - If no, tool call, early return
162/// - Proceed to `do_search`
163///     1) Execute search
164///     2) Rank by relevance
165/// - Send final tool call, which is allowed to have web search for repeated queries.
166pub(super) async fn search_request(this: Arc<Engine>, request: NormalRequest) {
167    let Some(web_search_options) = request.web_search_options.clone() else {
168        unreachable!()
169    };
170    let mut first_request = request.clone();
171    // Actually add the search tool here
172    first_request
173        .tools
174        .get_or_insert_with(Vec::new)
175        .push(search::get_search_tool(&web_search_options).unwrap());
176
177    let mut second_request = first_request.clone();
178    first_request.web_search_options = None;
179    second_request.web_search_options = None;
180
181    let this_clone = this.clone();
182
183    if !request.is_streaming {
184        let handle = tokio::spawn(async move {
185            let (new_sender, mut first_receiver) = tokio::sync::mpsc::channel(1);
186            second_request.response = new_sender;
187            std::mem::swap(&mut first_request.response, &mut second_request.response);
188
189            this_clone.add_request(first_request).await;
190            let ResponseOk::Done(done) = first_receiver.recv().await.unwrap().as_result().unwrap()
191            else {
192                unreachable!()
193            };
194
195            let tool_calls = match &done.choices[0].message.tool_calls {
196                Some(tool_calls)
197                    if tool_calls.len() == 1
198                        && tool_calls[0].function.name == search::SEARCH_TOOL_NAME =>
199                {
200                    &tool_calls[0]
201                }
202                None => {
203                    second_request
204                        .response
205                        .send(Response::Done(done))
206                        .await
207                        .unwrap();
208                    return;
209                }
210                Some(_) => {
211                    second_request
212                        .response
213                        .send(Response::Done(done))
214                        .await
215                        .unwrap();
216                    return;
217                }
218            };
219
220            do_search(this_clone, second_request, tool_calls, &web_search_options).await;
221        });
222        get_mut_arcmutex!(this.handles).push(handle);
223    } else {
224        let handle = tokio::spawn(async move {
225            let (new_sender, mut first_receiver) = tokio::sync::mpsc::channel(1);
226            second_request.response = new_sender;
227            std::mem::swap(&mut first_request.response, &mut second_request.response);
228
229            this_clone.add_request(first_request).await;
230            let ResponseOk::Chunk(done) = first_receiver.recv().await.unwrap().as_result().unwrap()
231            else {
232                unreachable!()
233            };
234            second_request
235                .response
236                .send(Response::Chunk(done.clone()))
237                .await
238                .unwrap();
239
240            let mut choice = done.choices[0].clone();
241
242            while choice.finish_reason.is_none() {
243                let ResponseOk::Chunk(done) =
244                    first_receiver.recv().await.unwrap().as_result().unwrap()
245                else {
246                    unreachable!()
247                };
248                second_request
249                    .response
250                    .send(Response::Chunk(done.clone()))
251                    .await
252                    .unwrap();
253
254                choice = done.choices[0].clone();
255            }
256
257            let tool_calls = match &choice.delta.tool_calls {
258                Some(tool_calls)
259                    if tool_calls.len() == 1
260                        && tool_calls[0].function.name == search::SEARCH_TOOL_NAME =>
261                {
262                    &tool_calls[0]
263                }
264                None => {
265                    return;
266                }
267                Some(_) => {
268                    return;
269                }
270            };
271
272            do_search(this_clone, second_request, tool_calls, &web_search_options).await;
273        });
274        get_mut_arcmutex!(this.handles).push(handle);
275    }
