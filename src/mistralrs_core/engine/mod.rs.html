104    pub fn new(
105        rx: Receiver<Request>,
106        pipeline: Arc<Mutex<dyn Pipeline>>,
107        config: SchedulerConfig,
108        truncate_sequence: bool,
109        mut no_kv_cache: bool,
110        mut no_prefix_cache: bool,
111        prefix_cache_n: usize,
112        disable_eos_stop: bool,
113        throughput_logging_enabled: bool,
114        search_embedding_model: Option<BertEmbeddingModel>,
115    ) -> anyhow::Result<Self> {
116        no_kv_cache |= get_mut_arcmutex!(pipeline).get_metadata().no_kv_cache;
117
118        no_prefix_cache = matches!(config, SchedulerConfig::PagedAttentionMeta { .. })
119            || no_prefix_cache
120            || no_kv_cache;
121
122        let bert_pipeline = match search_embedding_model {
123            Some(search_embedding_model) => Some(BertPipeline::new(
124                search_embedding_model,
125                &get_mut_arcmutex!(pipeline).device(),
126            )?),
127            None => None,
128        };
129
130        Ok(Self {
131            rx: Arc::new(Mutex::new(rx)),
132            pipeline,
133            bert_pipeline: Arc::new(Mutex::new(bert_pipeline)),
134            scheduler: config.into_scheduler(),
135            id: Arc::new(Mutex::new(0)),
136            truncate_sequence,
137            no_kv_cache,
138            prefix_cacher: Arc::new(Mutex::new(PrefixCacheManagerV2::new(
139                prefix_cache_n,
140                no_prefix_cache,
141            ))),
142            is_debug: DEBUG.load(Ordering::Relaxed),
143            disable_eos_stop,
144            throughput_logging_enabled,
145            logger: IntervalLogger::new(Duration::from_secs(5)),
146            handles: Arc::new(Mutex::new(Vec::new())),
147        })
148    }

150    pub async fn run(self: Arc<Self>) {
151        if self.throughput_logging_enabled {
152            self.logger.enable_logging();
153        }
154
155        let rng = Arc::new(std::sync::Mutex::new(Isaac64Rng::seed_from_u64(SEED)));
156        let mut last_completion_ids: Vec<usize> = vec![];
157        'lp: loop {
158            if matches!(
159                ENGINE_INSTRUCTIONS
160                    .lock()
161                    .expect("`ENGINE_INSTRUCTIONS` was poisioned")
162                    .get(get_mut_arcmutex!(self.id).deref()),
163                Some(Some(EngineInstruction::Terminate))
164            ) {
165                self.replicate_request_to_daemons(&Request::Terminate);
166                break 'lp;
167            }
168
169            while let Ok(request) = get_mut_arcmutex!(self.rx).try_recv() {
170                self.replicate_request_to_daemons(&request);
171                if matches!(request, Request::Terminate) {
172                    break 'lp;
173                }
174                self.clone().handle_request(request).await;
175            }
176
177            if TERMINATE_ALL_NEXT_STEP.load(Ordering::SeqCst) {
178                self.replicate_request_to_daemons(&Request::TerminateAllSeqsNextStep);
179            }
180
181            let run_start = Instant::now();
182            let mut scheduler = get_mut_arcmutex!(self.scheduler);
183            let scheduled = scheduler.schedule();
184
185            match scheduled {
186                SchedulerOutput::DefaultScheduler {
187                    output: mut scheduled,
188                } => {
189                    if !scheduled.completion.is_empty() {
190                        let current_completion_ids: Vec<usize> =
191                            scheduled.completion.iter().map(|seq| *seq.id()).collect();
192                        let res = {
193                            let mut pipeline = get_mut_arcmutex!(self.pipeline);
194                            let pre_op = if !self.no_kv_cache
195                                && last_completion_ids != current_completion_ids
196                            {
197                                CacheInstruction::In(
198                                    scheduled.completion[0]
199                                        .get_adapters()
200                                        .map(AdapterInstruction::Activate)
201                                        .unwrap_or(AdapterInstruction::None),
202                                )
203                            } else {
204                                CacheInstruction::Nothing(
205                                    scheduled.completion[0]
206                                        .get_adapters()
207                                        .map(AdapterInstruction::Activate)
208                                        .unwrap_or(AdapterInstruction::None),
209                                )
210                            };
211                            let post_op = if !self.no_kv_cache {
212                                CacheInstruction::Out
213                            } else {
214                                CacheInstruction::Reset {
215                                    load_preallocated_cache: false,
216                                    reset_non_granular: false,
217                                    adapter_inst: AdapterInstruction::None,
218                                }
219                            };
220
221                            let return_raw_logits = scheduled.completion[0].return_raw_logits;
222                            assert!(
223                                scheduled
224                                    .completion
225                                    .iter()
226                                    .all(|seq| seq.return_raw_logits == return_raw_logits),
227                                "All sequences must either return raw logits, or not."
228                            );
229
230                            pipeline
231                                .step(
232                                    &mut scheduled.completion,
233                                    false,
234                                    return_raw_logits,
235                                    &mut *get_mut_arcmutex!(self.prefix_cacher),
236                                    self.disable_eos_stop,
237                                    rng.clone(),
238                                    CacheBackendMetadata::DefaultInstructions { pre_op, post_op },
239                                )
240                                .await
241                        };
242
243                        handle_pipeline_forward_error!(
244                            "completion step",
245                            res,
246                            &mut scheduled.completion,
247                            self.pipeline,
248                            'lp,
249                            self.prefix_cacher
250                        );
251
252                        self.logger.add_tokens_processed(scheduled.completion.len());
253
254                        last_completion_ids = current_completion_ids;
255                    }
256
257                    if !scheduled.prompt.is_empty() {
258                        let prompt_exec_time = {
259                            let mut pipeline = get_mut_arcmutex!(self.pipeline);
260
261                            // Run the prompt seqs
262                            let post_op = if !self.no_kv_cache {
263                                CacheInstruction::Out
264                            } else {
265                                CacheInstruction::Reset {
266                                    load_preallocated_cache: false,
267                                    reset_non_granular: false,
268                                    adapter_inst: AdapterInstruction::None,
269                                }
270                            };
271                            let adapter_inst = scheduled.prompt[0]
272                                .get_adapters()
273                                .map(AdapterInstruction::Activate)
274                                .unwrap_or(AdapterInstruction::None);
275
276                            let return_raw_logits = scheduled.prompt[0].return_raw_logits;
277                            assert!(
278                                scheduled
279                                    .prompt
280                                    .iter()
281                                    .all(|seq| seq.return_raw_logits == return_raw_logits),
282                                "All sequences must either return raw logits, or not."
283                            );
284
285                            // This comes from prefix caching
286                            // The invariant where all token offsets are the same is handled by the scheduler
287                            let pre_op = if scheduled.prompt[0].token_offset() != 0 {
288                                CacheInstruction::In(adapter_inst)
289                            } else {
290                                CacheInstruction::Reset {
291                                    load_preallocated_cache: true,
292                                    reset_non_granular: false,
293                                    adapter_inst,
294                                }
295                            };
296
297                            pipeline
298                                .step(
299                                    &mut scheduled.prompt,
300                                    true,
301                                    return_raw_logits,
302                                    &mut *get_mut_arcmutex!(self.prefix_cacher),
303                                    self.disable_eos_stop,
304                                    rng.clone(),
305                                    CacheBackendMetadata::DefaultInstructions { pre_op, post_op },
306                                )
307                                .await
308                        };
309
310                        let prompt_exec_time = handle_pipeline_forward_error!(
311                            "prompt step",
312                            prompt_exec_time,
313                            &mut scheduled.prompt,
314                            self.pipeline,
315                            'lp,
316                            self.prefix_cacher
317                        );
318
319                        let total_processed_tokens: usize = scheduled
320                            .prompt
321                            .iter()
322                            .map(|seq| seq.get_toks().len())
323                            .sum();
324                        self.logger.add_tokens_processed(total_processed_tokens);
325
326                        for seq in scheduled.prompt.iter_mut() {
327                            match seq.sequence_stepping_type() {
328                                SeqStepType::OneShot => {
329                                    seq.set_state(SequenceState::Done(StopReason::GeneratedImage))
330                                }
331                                SeqStepType::PromptAndDecode => {
332                                    seq.set_state(SequenceState::RunningCompletion)
333                                }
334                            }
335                            let now = SystemTime::now()
336                                .duration_since(UNIX_EPOCH)
337                                .expect("Time travel has occurred!")
338                                .as_millis();
339                            #[allow(clippy::cast_precision_loss)]
340                            let prompt_tok_per_sec =
341                                seq.len() as f32 / prompt_exec_time.as_secs_f32();
342                            seq.prompt_tok_per_sec = prompt_tok_per_sec;
343                            seq.prompt_timestamp = Some(now);
344                        }
345                        last_completion_ids = vec![];
346                    }
347
348                    if self.is_debug {
349                        let ms_from_last_run = run_start.elapsed().as_secs_f64();
350                        let total_len = scheduled.prompt.len() + scheduled.completion.len();
351                        if total_len > 0 {
352                            let prompt_lengths = scheduled
353                                .prompt
354                                .iter()
355                                .map(|seq| seq.len().to_string())
356                                .collect::<Vec<_>>()
357                                .join(", ");
358
359                            let completion_lengths = scheduled
360                                .completion
361                                .iter()
362                                .map(|seq| seq.len().to_string())
363                                .collect::<Vec<_>>()
364                                .join(", ");
365
366                            tracing::info!(
367                                "Prompt[{}] Completion[{}] - {}ms",
368                                prompt_lengths,
369                                completion_lengths,
370                                ms_from_last_run * 1000.,
371                            );
372                        }
373                    }
374                }
375                SchedulerOutput::PagedAttention { mut output } => {
376                    if !output.scheduled.is_empty() {
377                        let is_prompt = get_mut_arcmutex!(output.scheduled[0]).is_prompt();
378
379                        let mut guards = output
380                            .scheduled
381                            .iter_mut()
382                            .map(|seq| seq.lock().unwrap())
383                            .collect::<Vec<_>>();
384
385                        let mut guards_mut =
386                            guards.iter_mut().map(|seq| &mut **seq).collect::<Vec<_>>();
387
388                        let res = {
389                            let mut pipeline = get_mut_arcmutex!(self.pipeline);
390
391                            let block_size = scheduler.block_size().unwrap();
392
393                            let metadata = PagedAttentionMeta {
394                                block_size,
395                                sliding_window: pipeline.get_metadata().sliding_window,
396                                block_engine: scheduler.block_engine().unwrap(),
397                            };
398
399                            let return_raw_logits = guards_mut[0].return_raw_logits;
400                            assert!(
401                                guards_mut
402                                    .iter()
403                                    .all(|seq| seq.return_raw_logits == return_raw_logits),
404                                "All sequences must either return raw logits, or not."
405                            );
406
407                            pipeline
408                                .step(
409                                    &mut guards_mut,
410                                    is_prompt,
411                                    return_raw_logits,
412                                    &mut *get_mut_arcmutex!(self.prefix_cacher),
413                                    self.disable_eos_stop,
414                                    rng.clone(),
415                                    CacheBackendMetadata::PagedAttention {
416                                        metadata,
417                                        blocks_to_copy: output.blocks_to_copy,
418                                        blocks_to_swap_in: output.blocks_to_swap_in,
419                                        blocks_to_swap_out: output.blocks_to_swap_out,
420                                    },
421                                )
422                                .await
423                        };
424
425                        handle_pipeline_forward_error!(
426                            "step",
427                            res,
428                            &mut guards_mut,
429                            self.pipeline,
430                            'lp,
431                            self.prefix_cacher
432                        );
433
434                        let total_processed_tokens: usize = guards
435                            .iter()
436                            .map(|seq| {
437                                if seq.is_prompt() {
438                                    seq.get_toks().len()
439                                } else {
440                                    1
441                                }
442                            })
443                            .sum();
444                        self.logger.add_tokens_processed(total_processed_tokens);
445
446                        if self.is_debug {
447                            let ms_from_last_run = run_start.elapsed().as_secs_f64();
448                            let total_len = guards.len();
449                            if total_len > 0 {
450                                let lengths = guards
451                                    .iter()
452                                    .map(|seq| seq.len().to_string())
453                                    .collect::<Vec<_>>()
454                                    .join(", ");
455
456                                let (prompt_lengths, completion_lengths) = if is_prompt {
457                                    (lengths, "".to_string())
458                                } else {
459                                    ("".to_string(), lengths)
460                                };
461
462                                tracing::info!(
463                                    "Prompt[{}] Completion[{}] - {}ms",
464                                    prompt_lengths,
465                                    completion_lengths,
466                                    ms_from_last_run * 1000.,
467                                );
468                            }
469                        }
470
471                        if is_prompt {
472                            for mut seq in guards {
473                                let now = SystemTime::now()
474                                    .duration_since(UNIX_EPOCH)
475                                    .expect("Time travel has occurred!")
476                                    .as_millis();
477                                #[allow(clippy::cast_precision_loss)]
478                                let prompt_tok_per_sec =
479                                    seq.len() as f32 / (now - seq.timestamp()) as f32;
480                                seq.prompt_tok_per_sec = prompt_tok_per_sec * 1000.;
481                                seq.prompt_timestamp = Some(now);
482                            }
483                        }
484                    }
485                }
486            }
487
488            scheduler.free_finished_sequence_groups();
489        }
490    }

492    fn build_sequence_recognizer(
493        tok_env: &Option<TokEnv>,
494        constraint: &Constraint,
495    ) -> anyhow::Result<SequenceRecognizer> {

507    fn replicate_request_to_daemons(&self, request: &Request) {
508        if !distributed::is_daemon() && mistralrs_quant::distributed::use_nccl() {
509            let name = distributed::ipc_name().unwrap();
510            let num_workers =
511                mistralrs_quant::distributed::get_global_tp_size_from_devices().unwrap() - 1;
512            let listener = ListenerOptions::new().name(name).create_sync().unwrap();
513
514            for _ in 0..num_workers {
515                let stream = listener.accept().unwrap();
516                let mut writer = BufWriter::new(stream);
517                let req = format!("{}\n", serde_json::to_string(&request).unwrap());
518                writer.write_all(req.as_bytes()).unwrap();
519            }
520        };
521    }

523    async fn handle_request(self: Arc<Self>, request: Request) {
524        match request {
525            Request::ActivateAdapters(adapters) => {
526                match get_mut_arcmutex!(self.pipeline).activate_adapters(adapters) {
527                    Ok(n) => info!("Swapped adapters in {n} LoRA layers."),
528                    Err(e) => warn!("Adapter activation failed: {e:?}"),
529                }
530            }
531            Request::Normal(request) => {
532                if matches!(
533                    request.messages,
534                    RequestMessage::Chat { .. } | RequestMessage::VisionChat { .. }
535                ) && request.web_search_options.is_some()
536                    && !request.is_streaming
537                    && get_mut_arcmutex!(self.bert_pipeline).is_some()
538                {
539                    let Some(web_search_options) = request.web_search_options.clone() else {
540                        unreachable!()
541                    };
542                    let mut first_request = request.clone();
543                    // Actually add the search tool here
544                    first_request
545                        .tools
546                        .get_or_insert_with(Vec::new)
547                        .push(search::get_search_tool(&web_search_options).unwrap());
548
549                    let mut second_request = first_request.clone();
550                    first_request.web_search_options = None;
551                    second_request.web_search_options = None;
552
553                    let this = self.clone();
554                    let handle = tokio::spawn(async move {
555                        let (new_sender, mut first_receiver) = tokio::sync::mpsc::channel(1);
556                        second_request.response = new_sender;
557                        std::mem::swap(&mut first_request.response, &mut second_request.response);
558
559                        this.add_request(first_request).await;
560                        let ResponseOk::Done(done) =
561                            first_receiver.recv().await.unwrap().as_result().unwrap()
562                        else {
563                            unreachable!()
564                        };
565
566                        let tool_calls = match &done.choices[0].message.tool_calls {
567                            Some(tool_calls)
568                                if tool_calls.len() == 1
569                                    && tool_calls[0].function.name == search::SEARCH_TOOL_NAME =>
570                            {
571                                &tool_calls[0]
572                            }
573                            None => {
574                                second_request
575                                    .response
576                                    .send(Response::Done(done))
577                                    .await
578                                    .unwrap();
579                                return;
580                            }
581                            Some(_) => {
582                                second_request
583                                    .response
584                                    .send(Response::Done(done))
585                                    .await
586                                    .unwrap();
587                                return;
588                            }
589                        };
590
591                        let RequestMessage::Chat(messages) = &mut second_request.messages else {
592                            unreachable!()
593                        };
594
595                        // Add assistant call message
596                        {
597                            let mut message: IndexMap<String, MessageContent> = IndexMap::new();
598                            message
599                                .insert("role".to_string(), Either::Left("assistant".to_string()));
600                            message.insert(
601                                "content".to_string(),
602                                Either::Left(format!(
603                                    "{{\"name\":\"{}\",\"arguments\":\"{}\"}}",
604                                    tool_calls.function.name, tool_calls.function.arguments
605                                )),
606                            );
607                            messages.push(message);
608                        }
609                        let tool_call_params: SearchFunctionParameters =
610                            serde_json::from_str(&tool_calls.function.arguments).unwrap();
611
612                        // Add tool response
613                        {
614                            let tokenizer = get_mut_arcmutex!(this.pipeline)
615                                .tokenizer()
616                                .expect("A tokenizer is expected for non-diffusion models.");
617                            let mut results = search::run_search_tool(&tool_call_params)
618                                .unwrap()
619                                .into_iter()
620                                .map(|result| {
621                                    let len = {
622                                        let inp = InputSequence::Raw(Cow::from(&result.content));
623                                        tokenizer
624                                            .encode_fast(inp, false)
625                                            .map(|x| x.len())
626                                            .unwrap_or(usize::MAX)
627                                    };
628                                    (result, len)
629                                })
630                                .collect::<Vec<_>>();
631                            // Sort increasing by tokenized length, if it fails, put it at the end.
632                            results.sort_by_key(|(_, len)| *len);
633
634                            {
635                                let device = get_mut_arcmutex!(this.pipeline).device();
636
637                                let Some(bert_pipeline) =
638                                    &mut *get_mut_arcmutex!(this.bert_pipeline)
639                                else {
640                                    unreachable!()
641                                };
642
643                                let decreasing_indexes = search::rag::compute_most_similar(
644                                    &device,
645                                    &tool_call_params.query,
646                                    results.iter().map(|(res, _)| res).collect::<Vec<_>>(),
647                                    bert_pipeline,
648                                )
649                                .unwrap();
650
651                                // Rerank the results
652                                let mut results_old = Vec::new();
653                                std::mem::swap(&mut results_old, &mut results);
654                                for &index in &decreasing_indexes {
655                                    let mut current_result: (SearchResult, usize) =
656                                        Default::default();
657                                    std::mem::swap(&mut current_result, &mut results_old[index]);
658
659                                    results.push(current_result);
660                                }
661                            }
662
663                            // Manage context size by # of tokens. Apply default here.
664                            let max_results_budget_toks =
665                                match web_search_options.search_context_size.unwrap_or_default() {
666                                    SearchContextSize::High => 10000_usize,
667                                    SearchContextSize::Medium => 7500_usize,
668                                    SearchContextSize::Low => 3000_usize,
669                                };
670                            let mut used_results = Vec::new();
671                            let mut used_len = 0;
672                            for (item, len) in results {
673                                if used_len + len >= max_results_budget_toks {
674                                    break;
675                                }
676                                // So the info! below gets the correct value
677                                used_len += len;
678                                used_results.push(item);
679                            }
680
681                            let tool_result = serde_json::to_string(&used_results)
682                                .unwrap()
683                                .replace("\\n", "\n")
684                                .replace("\\\"", "\"")
685                                .replace("\\\\", "\\");
686                            info!("Web search executed, using {used_len} tokens of {} search results.", used_results.len());
687
688                            let mut message: IndexMap<String, MessageContent> = IndexMap::new();
689                            message.insert("role".to_string(), Either::Left("tool".to_string()));
690                            message.insert(
691                                "content".to_string(),
692                                Either::Left(format!("{{\"output\": \"{tool_result}\"}}")),
693                            );
694                            messages.push(message);
695                        }
696
697                        this.add_request(second_request).await;
698                    });
699                    get_mut_arcmutex!(self.handles).push(handle);
700                } else {
701                    self.add_request(request).await
702                }
703            }
704            Request::ReIsq(level) => {
705                if let Err(e) = get_mut_arcmutex!(self.pipeline).re_isq_model(level) {
706                    warn!("ISQ requantization failed: {e:?}");
707                }
708            }
709            Request::Tokenize(req) => self.tokenize_text(req).await,
710            Request::Detokenize(req) => self.detokenize_text(req).await,
711            Request::Terminate => (),
712            Request::TerminateAllSeqsNextStep => {
713                TERMINATE_ALL_NEXT_STEP.store(true, Ordering::SeqCst)
714            }
715        }
716    }

718    async fn add_request(&self, request: NormalRequest) {
719        let is_chat = matches!(
720            request.messages,
721            RequestMessage::Chat(_) | RequestMessage::VisionChat { .. }
722        );
723        let echo_prompt = matches!(
724            request.messages,
725            RequestMessage::Completion {
726                echo_prompt: true,
727                ..
728            }
729        );
730
731        let best_of = match request.messages {
732            RequestMessage::Completion { best_of, .. } => best_of,
733            RequestMessage::Chat(_)
734            | RequestMessage::CompletionTokens(_)
735            | RequestMessage::VisionChat { .. }
736            | RequestMessage::ImageGeneration { .. } => None,
737        };
738        if is_chat
739            && !get_mut_arcmutex!(self.pipeline)
740                .get_chat_template()
741                .as_ref()
742                .is_some_and(|ch_t| ch_t.has_chat_template())
743        {
744            request
745                    .response
746                    .send(Response::ValidationError(
747                        "Received messages for a model which does not have a chat template. Either use a different model or pass a single string as the prompt".into(),
748                    )).await.expect("Expected receiver.");
749            return;
750        }
751
752        let images = match request.messages {
753            RequestMessage::VisionChat {
754                ref images,
755                messages: _,
756            } => Some(images.clone()),
757            _ => None,
758        };
759
760        let matcher = Arc::new(handle_seq_error!(
761            ToolCallingMatcher::new(request.tool_choice.unwrap_or(ToolChoice::Auto),),
762            request.response
763        ));
764
765        let image_generation_format = match &request.messages {
766            RequestMessage::ImageGeneration { format, .. } => Some(*format),
767            _ => None,
768        };
769
770        let seq_step_type = match &request.messages {
771            RequestMessage::ImageGeneration { .. } => SeqStepType::OneShot,
772            _ => SeqStepType::PromptAndDecode,
773        };
774
775        let diffusion_params = match &request.messages {
776            RequestMessage::ImageGeneration {
777                generation_params, ..
778            } => Some(generation_params.clone()),
779            _ => None,
780        };
781
782        let (mut prompt_tokens, prompt_text) = match request.messages {
783            RequestMessage::Chat(messages)
784            | RequestMessage::VisionChat {
785                images: _,
786                messages,
787            } => {
788                let pipeline = &*get_mut_arcmutex!(self.pipeline);
789                let tools = request.tools.unwrap_or_default();
790                let template = pipeline
791                    .get_processor()
792                    .process(pipeline, messages, true, true, tools);
793                handle_seq_error!(template, request.response)
794            }
795            RequestMessage::Completion { text, .. } => {
796                let Some(tokenizer) = &get_mut_arcmutex!(self.pipeline).tokenizer() else {
797                    request
798                        .response
799                        .send(Response::ValidationError(
800                            "Completion requests require the pipeline to have a tokenizer".into(),
801                        ))
802                        .await
803                        .expect("Expected receiver.");
804                    return;
805                };
806                let prompt = tokenizer
807                    .encode_fast(text.clone(), true)
808                    .map_err(anyhow::Error::msg);
809                (
810                    handle_seq_error!(prompt, request.response)
811                        .get_ids()
812                        .to_vec(),
813                    text,
814                )
815            }
816            RequestMessage::ImageGeneration { prompt, .. } => (vec![u32::MAX], prompt),
817            RequestMessage::CompletionTokens(it) => {
818                let Some(tokenizer) = &get_mut_arcmutex!(self.pipeline).tokenizer() else {
819                    request
820                        .response
821                        .send(Response::ValidationError(
822                            "Completion requests w/ raw tokens require the pipeline to have a tokenizer".into(),
823                        ))
824                        .await
825                        .expect("Expected receiver.");
826                    return;
827                };
828                let prompt = tokenizer
829                    .decode(&it, false)
830                    .map_err(|e| anyhow::Error::msg(e.to_string()));
831                (it, handle_seq_error!(prompt, request.response))
832            }
833        };
834        if prompt_tokens.is_empty() {
835            request
836                .response
837                .send(Response::ValidationError(
838                    "Received an empty prompt.".into(),
839                ))
840                .await
841                .expect("Expected receiver.");
842            return;
843        }
844
845        if prompt_tokens.len() > get_mut_arcmutex!(self.pipeline).get_metadata().max_seq_len {
846            if !self.truncate_sequence {
847                request
848                    .response
849                    .send(Response::ValidationError(
850                        format!("Prompt sequence length is greater than {}, perhaps consider using `truncate_sequence`?", get_mut_arcmutex!(self.pipeline).get_metadata().max_seq_len).into(),
851                    )).await.expect("Expected receiver.");
852                return;
853            } else {
854                let prompt_len = prompt_tokens.len();
855                let max_len = get_mut_arcmutex!(self.pipeline).get_metadata().max_seq_len;
856                let currently_over = prompt_len - max_len;
857                let sampling_max = if let Some(sampling_max) = request.sampling_params.max_len {
858                    if currently_over + sampling_max >= prompt_len {
859                        10
860                    } else {
861                        sampling_max
862                    }
863                } else {
864                    10
865                };
866                prompt_tokens = prompt_tokens[(currently_over + sampling_max)..].to_vec();
867                warn!("Prompt for request {} was {} tokens over the model maximum length. The last {} tokens were truncated to make space for generation.", request.id, currently_over, prompt_len - prompt_tokens.len());
868            }
869        }
870        let prefill_cache = handle_seq_error!(
871            get_mut_arcmutex!(self.prefix_cacher).search_for_matching_cache(
872                &prompt_tokens,
873                images.as_ref().is_some_and(|x| !x.is_empty())
874            ),
875            request.response
876        );
877
878        let topk = request
879            .sampling_params
880            .top_k
881            .map(|x| x as i64)
882            .unwrap_or(-1);
883        let topp = request.sampling_params.top_p.unwrap_or(1.0);
884        let minp = request.sampling_params.min_p.unwrap_or(0.0);
885        let num_hidden_layers = get_mut_arcmutex!(self.pipeline)
886            .get_metadata()
887            .num_hidden_layers;
888
889        let (stop_toks, stop_strings) = match request.sampling_params.stop_toks {
890            None => (vec![], vec![]),
891            Some(StopTokens::Ids(ref i)) => {
892                let tok_env = {
893                    let pipeline = get_mut_arcmutex!(self.pipeline);
894                    pipeline.get_metadata().tok_env.clone()
895                };
896                for id in i {
897                    // We can't use ` ` (space) as a stop token because other tokens like ` moon` start with a space.
898                    if let Some(tok_env) = tok_env.as_ref() {
899                        let tok_trie = tok_env.tok_trie();
900                        if tok_trie.has_extensions(tok_trie.token(*id)) {
901                            request
902                                .response
903                                .send(Response::ValidationError(
904                                    format!("Stop token {:?} is also a prefix of other tokens and cannot be used as a stop token.", tok_trie.token_str(*id)).into(),
905                                ))
906                                .await .expect("Expected receiver.");
907                            return;
908                        }
909                    }
910                }
911
912                (i.clone(), vec![])
913            }
914            Some(StopTokens::Seqs(ref s)) => {
915                let mut stop_toks = Vec::new();
916                let mut stop_strings: Vec<String> = Vec::new();
917
918                let (tok_env, tokenizer) = {
919                    let pipeline = get_mut_arcmutex!(self.pipeline);
920                    let tok_env = pipeline.get_metadata().tok_env.clone();
921                    let tokenizer = pipeline.tokenizer();
922                    (tok_env, tokenizer)
923                };
924
925                for stop_txt in s {
926                    let Some(tokenizer) = &tokenizer else {
927                        request
928                            .response
929                            .send(Response::ValidationError(
930                                "Completion requests require the pipeline to have a tokenizer"
931                                    .into(),
932                            ))
933                            .await
934                            .expect("Expected receiver.");
935                        return;
936                    };
937                    let encoded = tokenizer.encode_fast(stop_txt.to_string(), true);
938                    let toks = handle_seq_error!(encoded, request.response)
939                        .get_ids()
940                        .to_vec();
941
942                    if toks.len() == 1 {
943                        if tok_env.as_ref().is_some_and(|tok_env| {
944                            let tok_trie = tok_env.tok_trie();
945                            tok_trie.has_extensions(tok_trie.token(toks[0]))
946                        }) {
947                            stop_strings.push(stop_txt.clone());
948                        } else {
949                            stop_toks.push(toks[0]);
950                        }
951                    } else {
952                        stop_strings.push(stop_txt.clone());
953                    }
954                }
955
956                (stop_toks, stop_strings)
957            }
958        };
959
960        let group = Arc::new(tokio::sync::Mutex::new(SequenceGroup::new(
961            request.sampling_params.n_choices,
962            request.is_streaming,
963            is_chat,
964            best_of,
965        )));
966
967        let tokenizer = get_mut_arcmutex!(self.pipeline).tokenizer();
968
969        let sampler = Sampler::new(
970            Some(request.sampling_params.temperature.unwrap_or(1.0)),
971            request.sampling_params.top_n_logprobs,
972            tokenizer,
973            request.sampling_params.frequency_penalty,
974            request.sampling_params.presence_penalty,
975            request.sampling_params.dry_params,
976            topk,
977            topp,
978            minp,
979            request.logits_processors.unwrap_or_default(),
980        );
981        let sampler = handle_seq_error!(sampler, request.response);
982
983        if request.sampling_params.n_choices == 0 {
984            request
985                .response
986                .send(Response::ValidationError(
987                    "Number of choices must be greater than 0.".into(),
988                ))
989                .await
990                .expect("Expected receiver.");
991            return;
992        }
993
994        // Add sequences
995        for response_index in 0..request.sampling_params.n_choices {
996            let trie = get_mut_arcmutex!(self.pipeline)
997                .get_metadata()
998                .tok_env
999                .clone();
1000            let recognizer = match Self::build_sequence_recognizer(&trie, &request.constraint) {
1001                Ok(recognizer) => recognizer,
1002                Err(err) => {
1003                    request
1004                        .response
1005                        .send(Response::ValidationError(
1006                            format!("Invalid grammar. {}", err).into(),
1007                        ))
1008                        .await
1009                        .expect("Expected receiver.");
1010                    return;
1011                }
1012            };
1013
1014            let block_size = get_mut_arcmutex!(self.pipeline)
1015                .get_metadata()
1016                .cache_config
1017                .clone()
1018                .map(|conf| conf.block_size);
1019
1020            let eos_toks = get_mut_arcmutex!(self.pipeline)
1021                .get_metadata()
1022                .eos_tok
1023                .clone();
1024
1025            let seq_preallocated_cache = if get_mut_arcmutex!(self.pipeline).do_preallocated_cache()
1026            {
1027                let metadata = get_mut_arcmutex!(self.pipeline).get_metadata();
1028                let model_metadata = metadata
1029                    .model_metadata
1030                    .as_ref()
1031                    .expect("If a model has a NormalCache it must have a model metadata");
1032                let n_tokens = prompt_tokens.len();
1033                let required_blocks = n_tokens.div_ceil(NormalCache::CACHE_GROW_SIZE);
1034                let max_seq_len = required_blocks * NormalCache::CACHE_GROW_SIZE;
1035                let k_shape = (
1036                    1usize,
1037                    model_metadata.num_kv_heads(),
1038                    max_seq_len,
1039                    model_metadata.k_head_dim(),
1040                );
1041                let v_shape = (
1042                    1usize,
1043                    model_metadata.num_kv_heads(),
1044                    max_seq_len,
1045                    model_metadata.v_head_dim(),
1046                );
1047                let dtype = get_mut_arcmutex!(self.pipeline)
1048                    .get_metadata()
1049                    .activation_dtype;
1050
1051                let k_seq_cache = {
1052                    let k_seq_cache =
1053                        Tensor::zeros(k_shape, dtype, &get_mut_arcmutex!(self.pipeline).device());
1054                    match k_seq_cache {
1055                        Ok(x) => x,
1056                        Err(_) => {
1057                            request
1058                                .response
1059                                .send(Response::InternalError(
1060                                    "Failed to allocate preallocated KV cache."
1061                                        .to_string()
1062                                        .into(),
1063                                ))
1064                                .await
1065                                .expect("Expected receiver.");
1066                            return;
1067                        }
1068                    }
1069                };
1070                let v_seq_cache = if k_shape == v_shape {
1071                    k_seq_cache.clone()
1072                } else {
1073                    let v_seq_cache =
1074                        Tensor::zeros(v_shape, dtype, &get_mut_arcmutex!(self.pipeline).device());
1075                    match v_seq_cache {
1076                        Ok(x) => x,
1077                        Err(_) => {
1078                            request
1079                                .response
1080                                .send(Response::InternalError(
1081                                    "Failed to allocate preallocated KV cache."
1082                                        .to_string()
1083                                        .into(),
1084                                ))
1085                                .await
1086                                .expect("Expected receiver.");
1087                            return;
1088                        }
1089                    }
1090                };
1091                Some((k_seq_cache, v_seq_cache))
1092            } else {
1093                None
1094            };
1095
1096            let now = SystemTime::now()
1097                .duration_since(UNIX_EPOCH)
1098                .expect("Time travel has occurred!");
1099            let seq = Sequence::new_waiting(
1100                prompt_tokens.clone(),
1101                prompt_text.clone(),
1102                *get_mut_arcmutex!(self.id).deref(),
1103                now.as_millis(),
1104                num_hidden_layers,
1105                request.response.clone(),
1106                sampler.clone(),
1107                stop_toks.clone(),
1108                stop_strings.clone(),
1109                request.sampling_params.max_len,
1110                request.return_logprobs,
1111                get_mut_arcmutex!(self.pipeline).get_metadata().is_xlora,
1112                group.clone(),
1113                response_index,
1114                now.as_secs(),
1115                recognizer,
1116                request.suffix.clone(),
1117                if echo_prompt {
1118                    Some(prompt_text.clone())
1119                } else {
1120                    None
1121                },
1122                request.adapters.clone(),
1123                images.clone(),
1124                block_size,
1125                Some(matcher.clone()),
1126                image_generation_format,
1127                seq_step_type,
1128                diffusion_params.clone(),
1129                seq_preallocated_cache,
1130                request.return_raw_logits,
1131                eos_toks,
1132            );
1133            self.logger.add_new_sequence();
1134            let seq = if let Some(prefill_cache) = prefill_cache.clone() {
1135                self.logger.add_prefix_cache_hit();
1136
1137                seq.prefill_v2(
1138                    prefill_cache.normal,
1139                    prefill_cache.toks,
1140                    prefill_cache.offset,
1141                )
1142            } else {
1143                seq
1144            };
1145            *get_mut_arcmutex!(self.id) += 1;
1146            get_mut_arcmutex!(self.scheduler).add_seq(seq);
1147        }
1148    }

1150    async fn tokenize_text(&self, request: TokenizationRequest) {
1151        match request.text {
1152            Either::Left(messages) => {
1153                let pipeline = &*get_mut_arcmutex!(self.pipeline);
1154                let tools = request.tools.unwrap_or_default();
1155                let template = pipeline.get_processor().process(
1156                    pipeline,
1157                    messages,
1158                    request.add_generation_prompt,
1159                    request.add_special_tokens,
1160                    tools,
1161                );
1162                let toks = match template {
1163                    Ok((toks, _)) => toks,
1164                    Err(e) => {
1165                        request
1166                            .response
1167                            .send(Err(e))
1168                            .await
1169                            .expect("Expected receiver.");
1170                        return;
1171                    }
1172                };
1173                request
1174                    .response
1175                    .send(Ok(toks))
1176                    .await
1177                    .expect("Sender disconnected unexpectedly!");
1178            }
1179            Either::Right(text) => {
1180                let pipeline = &*get_mut_arcmutex!(self.pipeline);
1181                let tokenizer = pipeline.tokenizer();
1182                let tokenizer = match tokenizer {
1183                    Some(tokenizer) => tokenizer,
1184                    None => {
1185                        request
1186                            .response
1187                            .send(Err(anyhow::Error::msg(
1188                                "Pipeline does not include a toksnizer.",
1189                            )))
1190                            .await
1191                            .expect("Expected receiver.");
1192                        return;
1193                    }
1194                };
1195                let toks = tokenizer.encode_fast(text, request.add_special_tokens);
1196                let toks = match toks {
1197                    Ok(tokenizer) => tokenizer,
1198                    Err(e) => {
1199                        request
1200                            .response
1201                            .send(Err(anyhow::Error::msg(e)))
1202                            .await
1203                            .expect("Expected receiver.");
1204                        return;
1205                    }
1206                };
1207                request
1208                    .response
1209                    .send(Ok(toks.get_ids().to_vec()))
1210                    .await
1211                    .expect("Sender disconnected unexpectedly!");
1212            }
1213        };
1214    }

1216    async fn detokenize_text(&self, request: DetokenizationRequest) {
1217        let pipeline = &*get_mut_arcmutex!(self.pipeline);
1218        let tokenizer = pipeline.tokenizer();
1219        let tokenizer = match tokenizer {
1220            Some(tokenizer) => tokenizer,
1221            None => {
1222                request
1223                    .response
1224                    .send(Err(anyhow::Error::msg(
1225                        "Pipeline does not include a toksnizer.",
1226                    )))
1227                    .await
1228                    .expect("Expected receiver.");
1229                return;
1230            }
1231        };
1232        let txt = tokenizer.decode(&request.tokens, request.skip_special_tokens);
1233        let txt = match txt {
1234            Ok(tokenizer) => tokenizer,
1235            Err(e) => {
1236                request
1237                    .response
1238                    .send(Err(anyhow::Error::msg(e)))
1239                    .await
1240                    .expect("Expected receiver.");
1241                return;
1242            }
1243        };
1244        request
1245            .response
1246            .send(Ok(txt))
1247            .await
1248            .expect("Sender disconnected unexpectedly!");
1249    }
