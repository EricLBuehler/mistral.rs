1use candle_core::{Result, Tensor};
2
3use super::NormalCache;
4
5#[derive(Debug, Clone)]
6pub struct RotatingCache {
7    pub all_data: Option<Tensor>,
8    pub dim: usize,
9    // `offset` is the current write index in the buffer
10    pub offset: usize,
11    // The total size of the sequence seen so far.
12    pub current_seq_len: usize,
13    // max_seq_len is the size of the rotating buffer, it is actually allowed for the full
14    // sequence to grow past this limit.
15    pub max_seq_len: usize,
16    pub capacity_seq_len: usize,
17}
18
19impl RotatingCache {
20    pub fn new(dim: usize, max_seq_len: usize, capacity_seq_len: usize) -> Self {
21        Self {
22            all_data: None,
23            dim,
24            offset: 0,
25            current_seq_len: 0,
26            max_seq_len,
27            capacity_seq_len,
28        }
29    }
30
31    pub fn offset(&self) -> usize {
32        self.offset
33    }
34
35    pub fn dim(&self) -> usize {
36        self.dim
37    }
38
39    pub fn current_seq_len(&self) -> usize {
40        self.current_seq_len
41    }
42
43    pub fn max_seq_len(&self) -> usize {
44        self.max_seq_len
45    }
46
47    pub fn all_data(&self) -> Option<&Tensor> {
48        self.all_data.as_ref()
49    }
50
51    pub fn current_data(&self) -> Result<Option<Tensor>> {
52        let data = match self.all_data.as_ref() {
53            None => None,
54            Some(d) => {
55                if self.current_seq_len >= self.max_seq_len {
56                    Some(d.clone())
57                } else {
58                    Some(d.narrow(self.dim, 0, self.current_seq_len)?)
59                }
60            }
61        };
62        Ok(data)
63    }
64
65    pub fn reset(&mut self) {
66        self.offset = 0;
67        self.current_seq_len = 0;
68        self.all_data = None;
69    }
70
71    pub fn try_set_len(&self, len: usize) -> candle_core::Result<()> {
72        // If trying to roll it back past the boundary of max_seq_len, fail early.
73        if self.current_seq_len.saturating_sub(len) > self.max_seq_len {
74            candle_core::bail!(
75                "Rotating KV cache (usually for sliding window) tried to reset to len {len} while current is {} and max retained is {}",
76                self.current_seq_len,
77                self.max_seq_len
78            );
79        }
80        Ok(())
81    }
82
83    pub fn set_len(&mut self, len: usize) -> candle_core::Result<()> {
84        self.try_set_len(len)?;
85        self.current_seq_len = len;
86        self.offset = len % self.max_seq_len;
87        Ok(())
88    }
89
90    pub fn append(&mut self, src: &Tensor) -> Result<Tensor> {
91        let seq_len = src.dim(self.dim)?;
92        // This doesn't seem very idiomatic but because the creation can fail, it's tricky to use
93        // self.all_data.get_or_insert_with.
94        if self.all_data.is_none() {
95            let mut shape = src.dims().to_vec();
96            shape[self.dim] = self.capacity_seq_len;
97            let ad = Tensor::zeros(shape, src.dtype(), src.device())?;
98            self.all_data = Some(ad)
99        };
100
101        // Expand kv cache, this case is a little more complex.
102        if self.current_seq_len + seq_len > self.capacity_seq_len || self.current_seq_len == 0 {
103            let diff = self.current_seq_len + seq_len.saturating_sub(self.capacity_seq_len);
104            let n_blocks_needed = diff.div_ceil(NormalCache::CACHE_GROW_SIZE);
105            self.capacity_seq_len += n_blocks_needed * NormalCache::CACHE_GROW_SIZE;
106            self.capacity_seq_len = self.capacity_seq_len.min(self.max_seq_len);
107            if self.capacity_seq_len > self.max_seq_len {
108                candle_core::bail!(
109                    "kv-cache: requested capacity ({}) above max seq len ({})",
110                    self.capacity_seq_len,
111                    self.max_seq_len
112                )
113            }
114            let mut shape = src.dims().to_vec();
115            shape[self.dim] = self.capacity_seq_len;
116            let ad = Tensor::zeros(shape, src.dtype(), src.device())?;
117            ad.slice_set(self.all_data.as_ref().unwrap(), self.dim, 0)?;
118            self.all_data = Some(ad);
119        }
120
121        let ad = self.all_data.as_mut().unwrap();
122
123        self.current_seq_len += seq_len;
124        if seq_len >= self.max_seq_len {
125            let to_copy = src
126                .narrow(self.dim, seq_len - self.max_seq_len, self.max_seq_len)?
127                .contiguous()?;
128            ad.slice_set(&to_copy, self.dim, 0)?;
129            self.offset = 0;
130            // Here we return `src` rather than `ad` so that all the past can be used.
131            Ok(src.clone())
132        } else {
133            let rem_len = self.max_seq_len - self.offset;
134            if seq_len <= rem_len {
135                ad.slice_set(&src.contiguous()?, self.dim, self.offset)?;
136                self.offset = (self.offset + seq_len) % self.max_seq_len;
137            } else {
138                // We have to make two copies here as we go over the boundary of the cache.
139                if rem_len > 0 {
140                    let src1 = src.narrow(self.dim, 0, rem_len)?.contiguous()?;
141                    ad.slice_set(&src1, self.dim, self.offset)?;
142                }
143                let src2 = src
144                    .narrow(self.dim, rem_len, seq_len - rem_len)?
145                    .contiguous()?;
146                ad.slice_set(&src2, self.dim, 0)?;
147                self.offset = seq_len - rem_len;
148            }
149            if self.current_seq_len >= self.max_seq_len {
150                Ok(ad.clone())
151            } else {
152                Ok(ad.narrow(self.dim, 0, self.current_seq_len)?)
153            }
154        }
155    }
156}
