36#[derive(Debug, Clone)]
37pub enum KvCache {
38    Normal { k: SingleCache, v: SingleCache },
39    Rotating { k: RotatingCache, v: RotatingCache },
40}
41
42impl KvCache {
43    pub fn new_normal(dim: usize, max_seq_len: usize, capacity_seq_len: usize) -> Self {
44        let k = SingleCache::new(dim, max_seq_len, capacity_seq_len);
45        let v = SingleCache::new(dim, max_seq_len, capacity_seq_len);
46        Self::Normal { k, v }
47    }
48
49    pub fn new_rotating(dim: usize, sliding_window: usize, capacity_seq_len: usize) -> Self {
50        let k = RotatingCache::new(dim, sliding_window, capacity_seq_len);
51        let v = RotatingCache::new(dim, sliding_window, capacity_seq_len);
52        Self::Rotating { k, v }
53    }
54
55    pub fn k(&self) -> Result<Option<Tensor>> {
56        match self {
57            Self::Normal { k, .. } => k.current_data(),
58            Self::Rotating { k, .. } => k.current_data(),
59        }
60    }
61
62    pub fn v(&self) -> Result<Option<Tensor>> {
63        match self {
64            Self::Normal { v, .. } => v.current_data(),
65            Self::Rotating { v, .. } => v.current_data(),
66        }
67    }
68
69    pub fn append(&mut self, k: &Tensor, v: &Tensor) -> Result<(Tensor, Tensor)> {
70        let k = k.contiguous()?;
71        let v = v.contiguous()?;
72        let (out_k, out_v) = match self {
73            Self::Normal { k: kc, v: vc } => {
74                kc.append(&k)?;
75                vc.append(&v)?;
76                (kc.current_data()?, vc.current_data()?)
77            }
78            Self::Rotating { k: kc, v: vc } => {
79                let out_k = kc.append(&k)?;
80                let out_v = vc.append(&v)?;
81                (Some(out_k), Some(out_v))
82            }
83        };
84        let k = match out_k {
85            None => {
86                let mut shape = k.dims().to_vec();
87                match self {
88                    Self::Normal { k, .. } => shape[k.dim] = 0,
89                    Self::Rotating { k, .. } => shape[k.dim] = 0,
90                }
91                Tensor::zeros(shape, k.dtype(), k.device())?
92            }
93            Some(k) => k,
94        };
95        let v = match out_v {
96            None => {
97                let mut shape = v.dims().to_vec();
98                match self {
99                    Self::Normal { v, .. } => shape[v.dim] = 0,
100                    Self::Rotating { v, .. } => shape[v.dim] = 0,
101                }
102                Tensor::zeros(shape, v.dtype(), v.device())?
103            }
104            Some(v) => v,
105        };
106        Ok((k, v))
107    }
108
109    pub fn current_seq_len(&self) -> usize {
110        match self {
111            Self::Normal { k, .. } => k.current_seq_len(),
112            Self::Rotating { k, .. } => k.current_seq_len(),
113        }
114    }
115
116    pub fn reset(&mut self) {
117        match self {
118            Self::Normal { k, v } => {
119                k.reset();
120                v.reset();
121            }
122            Self::Rotating { k, v } => {
123                k.reset();
124                v.reset();
125            }
126        }
127    }
128
129    /// Returns Ok if the length reassignment was successful, otherwise returns Err.
130    pub fn set_len(&mut self, len: usize) -> candle_core::Result<()> {
131        match self {
132            Self::Normal { k, v } => {
133                k.set_len(len)?;
134                v.set_len(len)?;
135                Ok(())
136            }
137            Self::Rotating { k, v } => {
138                k.set_len(len)?;
139                v.set_len(len)?;
140                Ok(())
141            }
142        }
143    }
144
145    pub fn is_rotating(&self) -> bool {
146        matches!(self, Self::Rotating { .. })
147    }
148}

153#[derive(Debug)]
154pub enum NormalCacheType {
155    Normal { max_seq_len: usize },
156    SlidingWindow { window: usize },
157}
158
159impl NormalCache {
160    /// The number of tokens to grow the cache by
161    pub const CACHE_GROW_SIZE: usize = 512;
162
163    pub fn new(len: usize, max_seq_len: usize) -> Arc<Mutex<Self>> {
164        Arc::new(Mutex::new(Self(vec![
165            KvCache::new_normal(
166                2,
167                max_seq_len,
168                Self::CACHE_GROW_SIZE
169            );
170            len
171        ])))
172    }
173
174    pub fn new_sliding(
175        len: usize,
176        max_seq_len: usize,
177        sliding_window: Option<usize>,
178    ) -> Arc<Mutex<Self>> {
179        match sliding_window {
180            Some(sliding_window) => Arc::new(Mutex::new(Self(vec![
181                KvCache::new_rotating(
182                    2,
183                    sliding_window,
184                    Self::CACHE_GROW_SIZE
185                );
186                len
187            ]))),
188            None => Arc::new(Mutex::new(Self(vec![
189                KvCache::new_normal(
190                    2,
191                    max_seq_len,
192                    Self::CACHE_GROW_SIZE
193                );
194                len
195            ]))),
196        }
197    }
198
199    pub fn from_types(types: Vec<NormalCacheType>) -> Arc<Mutex<Self>> {
200        let mut caches = Vec::new();
201        for ty in types {
202            match ty {
203                NormalCacheType::Normal { max_seq_len } => {
204                    caches.push(KvCache::new_normal(2, max_seq_len, Self::CACHE_GROW_SIZE));
205                }
206                NormalCacheType::SlidingWindow { window } => {
207                    caches.push(KvCache::new_rotating(2, window, Self::CACHE_GROW_SIZE));
208                }
209            }
210        }
211        Arc::new(Mutex::new(Self(caches)))
212    }

218    fn clone_in_cache(
219        &self,
220        pipeline: &T,
221        seqs: &mut [&mut crate::sequence::Sequence],
222        modify_draft_cache: bool,
223    ) {
224        let mut new_k_cache = Vec::new();
225        let mut new_v_cache = Vec::new();
226
227        for layer in 0..pipeline.get_metadata().num_hidden_layers {
228            // Preallocate combined k and v caches across all sequences, avoiding Tensor::cat copies
229            let batch_len = seqs.len();
230            // Use the first sequence as template
231            let (first_k, first_v) = {
232                let src_cache = if modify_draft_cache {
233                    seqs[0].normal_draft_cache()
234                } else {
235                    seqs[0].normal_cache()
236                };
237                let cache = src_cache.get(layer).unwrap().as_ref().unwrap();
238                match cache {
239                    KvCache::Normal { k, v } => {
240                        (k.all_data.clone().unwrap(), v.all_data.clone().unwrap())
241                    }
242                    KvCache::Rotating { k, v } => {
243                        (k.all_data.clone().unwrap(), v.all_data.clone().unwrap())
244                    }
245                }
246            };
247            // Build dims for batched cache
248            let mut dims_k = first_k.dims().to_vec();
249            let mut dims_v = first_v.dims().to_vec();
250            dims_k[0] *= batch_len;
251            dims_v[0] *= batch_len;
252            let batch_k = Tensor::zeros(dims_k.clone(), first_k.dtype(), first_k.device()).unwrap();
253            let batch_v = Tensor::zeros(dims_v.clone(), first_v.dtype(), first_v.device()).unwrap();
254            // Fill each sequence's cache slice
255            for (i, seq) in seqs.iter_mut().enumerate() {
256                let src_cache = if modify_draft_cache {
257                    seq.normal_draft_cache()
258                } else {
259                    seq.normal_cache()
260                };
261                let cache = src_cache.get(layer).unwrap().as_ref().unwrap();
262                let (src_k, src_v) = match cache {
263                    KvCache::Normal { k, v } => {
264                        (k.all_data.clone().unwrap(), v.all_data.clone().unwrap())
265                    }
266                    KvCache::Rotating { k, v } => {
267                        (k.all_data.clone().unwrap(), v.all_data.clone().unwrap())
268                    }
269                };
270                let offset = i * first_k.dims()[0];
271                batch_k.slice_set(&src_k, 0, offset).unwrap();
272                batch_v.slice_set(&src_v, 0, offset).unwrap();
273            }
274            new_k_cache.push(Some(batch_k));
275            new_v_cache.push(Some(batch_v));
276        }
277
278        let seq0_cache = if modify_draft_cache {
279            &*seqs[0].normal_draft_cache()
280        } else {
281            &*seqs[0].normal_cache()
282        };
283
284        let mut caches = Vec::new();
285        for (layer_idx, (k_cache, v_cache)) in new_k_cache.into_iter().zip(new_v_cache).enumerate()
286        {
287            // Use this for the various parameters. Assumes all seqs are from one model.
288            match seq0_cache[layer_idx].as_ref().unwrap() {
289                KvCache::Normal { k: old_k, .. } => {
290                    let template_cache_dim = old_k.dim;
291                    let template_cache_csl = old_k.current_seq_len;
292                    let template_cache_msl = old_k.max_seq_len;
293                    let template_cache_capsl = old_k.capacity_seq_len;
294
295                    caches.push(KvCache::Normal {
296                        k: SingleCache {
297                            all_data: k_cache.map(|x| x.contiguous().unwrap()),
298                            dim: template_cache_dim,
299                            current_seq_len: template_cache_csl,
300                            max_seq_len: template_cache_msl,
301                            capacity_seq_len: template_cache_capsl,
302                        },
303                        v: SingleCache {
304                            all_data: v_cache.map(|x| x.contiguous().unwrap()),
305                            dim: template_cache_dim,
306                            current_seq_len: template_cache_csl,
307                            max_seq_len: template_cache_msl,
308                            capacity_seq_len: template_cache_capsl,
309                        },
310                    });
311                }
312                KvCache::Rotating { k: old_k, .. } => {
313                    let template_cache_dim = old_k.dim;
314                    let template_cache_csl = old_k.current_seq_len;
315                    let template_cache_msl = old_k.max_seq_len;
316                    let template_cache_offset = old_k.offset;
317                    let template_cache_capsl = old_k.capacity_seq_len;
318
319                    caches.push(KvCache::Rotating {
320                        k: RotatingCache {
321                            all_data: k_cache.map(|x| x.contiguous().unwrap()),
322                            dim: template_cache_dim,
323                            current_seq_len: template_cache_csl,
324                            max_seq_len: template_cache_msl,
325                            offset: template_cache_offset,
326                            capacity_seq_len: template_cache_capsl,
327                        },
328                        v: RotatingCache {
329                            all_data: v_cache.map(|x| x.contiguous().unwrap()),
330                            dim: template_cache_dim,
331                            current_seq_len: template_cache_csl,
332                            max_seq_len: template_cache_msl,
333                            offset: template_cache_offset,
334                            capacity_seq_len: template_cache_capsl,
335                        },
336                    });
337                }
338            }
339        }
340        *pipeline.cache().normal() = NormalCache(caches);
341    }
342    fn clone_out_cache(&self, pipeline: &T, seqs: &mut [&mut Sequence], modify_draft_cache: bool) {
343        let all_cache = pipeline.cache().normal();
344        for layer in 0..pipeline.get_metadata().num_hidden_layers {
345            let cache = all_cache.0.get(layer).unwrap();
346            // This case for llama 3.2 vision cross attn
347            if cache.k().unwrap().is_none() {
348                continue;
349            }
350
351            let (k_cache, v_cache) = match cache {
352                KvCache::Normal { k, v } => {
353                    (k.all_data.clone().unwrap(), v.all_data.clone().unwrap())
354                }
355                KvCache::Rotating { k, v } => {
356                    (k.all_data.clone().unwrap(), v.all_data.clone().unwrap())
357                }
358            };
359
360            let k_caches = k_cache.chunk(seqs.len(), 0).unwrap();
361            debug_assert_eq!(k_caches.len(), seqs.len());
362            let v_caches = v_cache.chunk(seqs.len(), 0).unwrap();
363            debug_assert_eq!(v_caches.len(), seqs.len());
364
365            for (seq_i, seq) in seqs.iter_mut().enumerate() {
366                let output_cache = if modify_draft_cache {
367                    seq.normal_draft_cache()
368                } else {
369                    seq.normal_cache()
370                };
371                let seq_cache = &mut output_cache[layer];
372                let k = k_caches.get(seq_i).unwrap().clone();
373                let v = v_caches.get(seq_i).unwrap().clone();
374
375                match cache {
376                    KvCache::Normal {
377                        k: cache_k,
378                        v: cache_v,
379                    } => {
380                        *seq_cache = Some(KvCache::Normal {
381                            k: SingleCache {
382                                all_data: Some(k),
383                                dim: cache_k.dim,
384                                current_seq_len: cache_k.current_seq_len,
385                                max_seq_len: cache_k.max_seq_len,
386                                capacity_seq_len: cache_k.capacity_seq_len,
387                            },
388                            v: SingleCache {
389                                all_data: Some(v),
390                                dim: cache_v.dim,
391                                current_seq_len: cache_v.current_seq_len,
392                                max_seq_len: cache_v.max_seq_len,
393                                capacity_seq_len: cache_v.capacity_seq_len,
394                            },
395                        });
396                    }
397                    KvCache::Rotating {
398                        k: cache_k,
399                        v: cache_v,
400                    } => {
401                        *seq_cache = Some(KvCache::Rotating {
402                            k: RotatingCache {
403                                all_data: Some(k),
404                                dim: cache_k.dim,
405                                current_seq_len: cache_k.current_seq_len,
406                                max_seq_len: cache_k.max_seq_len,
407                                offset: cache_k.offset,
408                                capacity_seq_len: cache_k.capacity_seq_len,
409                            },
410                            v: RotatingCache {
411                                all_data: Some(v),
412                                dim: cache_v.dim,
413                                current_seq_len: cache_v.current_seq_len,
414                                max_seq_len: cache_v.max_seq_len,
415                                offset: cache_v.offset,
416                                capacity_seq_len: cache_v.capacity_seq_len,
417                            },
418                        });
419                    }
420                }
421            }
422        }
423    }
424    fn set_none_cache(
425        &self,
426        pipeline: &T,
427        seqs: &mut [&mut Sequence],
428        _modify_draft_cache: bool,
429        load_preallocated_cache: bool,
430    ) {
431        if seqs.iter().any(|seq| seq.preallocated_cache().is_none()) {
432            for layer in pipeline.cache().normal().0.iter_mut() {
433                layer.reset();
434            }
435            return;
436        }
437
438        let layer_devices = if let Some(device_mapper) = pipeline.device_mapper() {
439            let mut layer_devices = Vec::new();
440            for layer in 0..device_mapper.num_device_mapping_layers() {
441                let device = device_mapper.device_for(layer, false).cloned();
442                layer_devices.push(device.expect("Internal bug, layer out of range!"));
443            }
444            Some(layer_devices)
445        } else {
446            None
447        };
448
449        let old_caches = pipeline.cache().normal().0.clone();
450
451        for (layer_idx, layer) in pipeline.cache().normal().0.iter_mut().enumerate() {
452            if !load_preallocated_cache {
453                layer.reset();
454                continue;
455            }
456
457            let mut k_caches = Vec::new();
458            let mut v_caches = Vec::new();
459            for seq in seqs.iter_mut() {
460                let (mut k_preallocated_cache, mut v_preallocated_cache) =
461                    (*seq.preallocated_cache().as_ref().unwrap()).clone();
462                if let Some(layer_devices) = &layer_devices {
463                    let layer_dev = &layer_devices[layer_idx];
464                    k_preallocated_cache = k_preallocated_cache
465                        .to_device(layer_dev)
466                        .expect("Could not prepare cache");
467                    v_preallocated_cache = v_preallocated_cache
468                        .to_device(layer_dev)
469                        .expect("Could not prepare cache");
470                }
471                k_caches.push(k_preallocated_cache);
472                v_caches.push(v_preallocated_cache);
473            }
474            let k_cache = if k_caches.len() > 1 {
475                Tensor::cat(&k_caches, 0).unwrap()
476            } else {
477                k_caches[0].clone()
478            };
479            let v_cache = if v_caches.len() > 1 {
480                Tensor::cat(&v_caches, 0).unwrap()
481            } else {
482                v_caches[0].clone()
483            };
484
485            // Use this for the various parameters. Assumes all seqs are from one model.
486            match &old_caches[layer_idx] {
487                KvCache::Normal { k, .. } => {
488                    let template_cache_dim = k.dim;
489                    let template_cache_msl = k.max_seq_len;
490
491                    let cache = KvCache::Normal {
492                        k: SingleCache {
493                            all_data: Some(k_cache.zeros_like().unwrap()),
494                            dim: template_cache_dim,
495                            current_seq_len: 0,
496                            max_seq_len: template_cache_msl,
497                            capacity_seq_len: k_cache.dims()[template_cache_dim],
498                        },
499                        v: SingleCache {
500                            all_data: Some(v_cache.zeros_like().unwrap()),
501                            dim: template_cache_dim,
502                            current_seq_len: 0,
503                            max_seq_len: template_cache_msl,
504                            capacity_seq_len: k_cache.dims()[template_cache_dim],
505                        },
506                    };
507                    *layer = cache;
508                }
509                KvCache::Rotating { k, .. } => {
510                    let template_cache_dim = k.dim;
511                    let template_cache_msl = k.max_seq_len;
512
513                    // Rotating cache is not preallocated.
514                    let cache = KvCache::Rotating {
515                        k: RotatingCache {
516                            all_data: None,
517                            dim: template_cache_dim,
518                            current_seq_len: 0,
519                            max_seq_len: template_cache_msl,
520                            offset: 0,
521                            capacity_seq_len: 0,
522                        },
523                        v: RotatingCache {
524                            all_data: None,
525                            dim: template_cache_dim,
526                            current_seq_len: 0,
527                            max_seq_len: template_cache_msl,
528                            offset: 0,
529                            capacity_seq_len: 0,
530                        },
531                    };
532                    *layer = cache;
533                }
534            }
535        }
536    }
537}

592    /// Update the KV cache and return (k,v)
593    pub(crate) fn update_kv_cache(
594        cache: &mut Option<(Tensor, Tensor)>,
595        k: Tensor,
596        v: Tensor,
597        slow_cat: bool,
598    ) -> Result<(Tensor, Tensor)> {
599        let (k, v) = match &*cache {
600            None => (k, v),
601            Some((k_cache, v_cache)) => {
602                if !slow_cat {
603                    let k = candle_nn::ops::kvconcat(k_cache, &k, 2)?.contiguous()?;
604                    let v = candle_nn::ops::kvconcat(v_cache, &v, 2)?.contiguous()?;
605                    (k, v)
606                } else {
607                    let k = Tensor::cat(&[k_cache, &k], 2)?.contiguous()?;
608                    let v = Tensor::cat(&[v_cache, &v], 2)?.contiguous()?;
609                    (k, v)
610                }
611            }
612        };
613        *cache = Some((k.clone(), v.clone()));
614        Ok((k.contiguous()?, v.contiguous()?))
615    }

617    /// Update the KV cache and return (k,v,attn_mask)
618    pub(crate) fn update_kv_cache_sliding_window(
619        cache: &mut Option<(Tensor, Tensor)>,
620        k: Tensor,
621        v: Tensor,
622        attention_mask: Option<&Tensor>,
623        sliding_window: Option<usize>,
624        slow_cat: bool,
625    ) -> Result<(Tensor, Tensor, Option<Tensor>)> {
626        let (k, v, attention_mask) = match cache.clone() {
627            None => (k, v, attention_mask.cloned()),
628            Some((mut prev_k, mut prev_v)) => {
629                let mut mask = attention_mask.cloned();
630                if let Some(sliding_window) = sliding_window {
631                    let kv_seq_len = prev_k.dim(2)?;
632                    if kv_seq_len > sliding_window {
633                        prev_k = prev_k.narrow(
634                            2,
635                            kv_seq_len - (sliding_window - 1),
636                            sliding_window - 1,
637                        )?;
638                        prev_v = prev_v.narrow(
639                            2,
640                            kv_seq_len - (sliding_window - 1),
641                            sliding_window - 1,
642                        )?;
643                        if let Some(ref mut mask) = mask {
644                            let mask_len = mask.dim(1)?;
645                            *mask = mask.narrow(
646                                1,
647                                mask_len - (sliding_window - 1),
648                                sliding_window - 1,
649                            )?;
650                            *mask = Tensor::cat(
651                                &[&*mask, &mask.narrow(1, mask_len - 1, 1)?.ones_like()?],
652                                D::Minus1,
653                            )?;
654                        }
655                    }
656                }
657                let (k, v) = if !slow_cat {
658                    let k = candle_nn::ops::kvconcat(&prev_k, &k, 2)?;
659                    let v = candle_nn::ops::kvconcat(&prev_v, &v, 2)?;
660                    (k, v)
661                } else {
662                    let k = Tensor::cat(&[prev_k, k], 2)?.contiguous()?;
663                    let v = Tensor::cat(&[prev_v, v], 2)?.contiguous()?;
664                    (k, v)
665                };
666                (k, v, mask)
667            }
668        };
669        *cache = Some((k.clone(), v.clone()));
670        Ok((k.contiguous()?, v.contiguous()?, attention_mask))
671    }
672}
673
674pub struct FullCacheManager;
675
676enum SeqCache {
677    Normal,
678    XLora,
679    Draft,
680}
681
682fn clone_in_cache(
683    num_hidden_layers: usize,
684    cache: &mut LayerCaches,
685    seqs: &mut [&mut crate::sequence::Sequence],
686    src: SeqCache,
687) {
688    let mut new_cache = Vec::new();
689    'outer: for layer in 0..num_hidden_layers {
690        let mut k_vec = Vec::new();
691        let mut v_vec = Vec::new();
692        for seq in &mut *seqs {
693            let src_cache = match src {
694                SeqCache::Normal => seq.cache(),
695                SeqCache::XLora => seq.xlora_cache(),
696                SeqCache::Draft => seq.draft_cache(),
697            };
698            let cache = src_cache.get(layer).unwrap();
699            // This case for llama 3.2 vision cross attn
700            if cache.is_none() {
701                new_cache.push(None);
702                continue 'outer;
703            }
704            let cache = cache
705                .as_ref()
706                .expect("Not handling completions in `clone_in_cache`.");
707            k_vec.push(cache.0.clone());
708            v_vec.push(cache.1.clone());
709        }
710        new_cache.push(Some((
711            if k_vec.len() > 1 {
712                Tensor::cat(&k_vec, 0).unwrap()
713            } else {
714                k_vec[0].clone()
715            },
716            if v_vec.len() > 1 {
717                Tensor::cat(&v_vec, 0).unwrap()
718            } else {
719                v_vec[0].clone()
720            },
721        )));
722    }
723    *cache = new_cache;
724}
725
726fn clone_out_cache(
727    num_hidden_layers: usize,
728    cache: &mut LayerCaches,
729    seqs: &mut [&mut crate::sequence::Sequence],
730    target: SeqCache,
731) {
732    for layer in 0..num_hidden_layers {
733        let cache = cache.get(layer).unwrap();
734        // This case for llama 3.2 vision cross attn
735        if cache.is_none() {
736            continue;
737        }
738
739        let k_cache = cache.as_ref().unwrap().0.clone();
740        let v_cache = cache.as_ref().unwrap().1.clone();
741
742        let k_caches = k_cache.chunk(seqs.len(), 0).unwrap();
743        debug_assert_eq!(k_caches.len(), seqs.len());
744        let v_caches = v_cache.chunk(seqs.len(), 0).unwrap();
745        debug_assert_eq!(v_caches.len(), seqs.len());
746
747        for (seq_i, seq) in seqs.iter_mut().enumerate() {
748            let output_cache = match target {
749                SeqCache::Normal => seq.cache(),
750                SeqCache::XLora => seq.xlora_cache(),
751                SeqCache::Draft => seq.draft_cache(),
752            };
753            let seq_cache = &mut output_cache[layer];
754            let k = k_caches.get(seq_i).unwrap().clone();
755            let v = v_caches.get(seq_i).unwrap().clone();
756            *seq_cache = Some((k, v));
757        }
758    }
759}
760
761impl<T: CacheManagerMixin + MetadataMixin + ?Sized> CacheManager<T> for FullCacheManager {
762    fn clone_in_cache(
763        &self,
764        pipeline: &T,
765        seqs: &mut [&mut crate::sequence::Sequence],
766        modify_draft_cache: bool,
767    ) {
768        if modify_draft_cache {
769            clone_in_cache(
770                pipeline.get_metadata().num_hidden_layers,
771                &mut pipeline.cache().full().lock(),
772                seqs,
773                SeqCache::Draft,
774            );
775            return;
776        }
777        clone_in_cache(
778            pipeline.get_metadata().num_hidden_layers,
779            &mut pipeline.cache().full().lock(),
780            seqs,
781            SeqCache::Normal,
782        );
783        if pipeline.get_metadata().is_xlora && !pipeline.get_metadata().no_kv_cache {
784            clone_in_cache(
785                pipeline.get_metadata().num_hidden_layers,
786                &mut pipeline.cache().full().xlora_lock(),
787                seqs,
788                SeqCache::XLora,
789            );
790        }
791        if pipeline.get_metadata().is_xlora {
792            pipeline
793                .cache()
794                .full()
795                .get_scalings_cache()
796                .clone_from(seqs[0].scaling_cache());
797        }
798    }
799
800    fn clone_out_cache(
801        &self,
802        pipeline: &T,
803        seqs: &mut [&mut crate::sequence::Sequence],
804        modify_draft_cache: bool,
805    ) {
806        if modify_draft_cache {
807            clone_out_cache(
808                pipeline.get_metadata().num_hidden_layers,
809                &mut pipeline.cache().full().lock(),
810                seqs,
811                SeqCache::Draft,
812            );
813            return;
814        }
815        clone_out_cache(
816            pipeline.get_metadata().num_hidden_layers,
817            &mut pipeline.cache().full().lock(),
818            seqs,
819            SeqCache::Normal,
820        );
821        if pipeline.get_metadata().is_xlora && !pipeline.get_metadata().no_kv_cache {
822            clone_out_cache(
823                pipeline.get_metadata().num_hidden_layers,
824                &mut pipeline.cache().full().xlora_lock(),
825                seqs,
826                SeqCache::XLora,
827            );
828        }
829        if pipeline.get_metadata().is_xlora {
830            seqs[0]
831                .scaling_cache()
832                .clone_from(&pipeline.cache().full().get_scalings_cache());
833        }
834    }
835
836    fn set_none_cache(
837        &self,
838        pipeline: &T,
839        _seqs: &mut [&mut Sequence],
840        modify_draft_cache: bool,
841        _load_preallocated_cache: bool,
842    ) {
843        let mut new_cache = Vec::new();
844        for _ in 0..pipeline.get_metadata().num_hidden_layers {
845            new_cache.push(None);
846        }
847        pipeline.cache().full().lock().clone_from(&new_cache);
848        if modify_draft_cache {
849            pipeline.cache().full().draft_lock().clone_from(&new_cache);
850        }
851        if pipeline.cache().full().is_xlora() {
852            *pipeline.cache().full().xlora_lock() = new_cache;
853        }
854    }
855}
