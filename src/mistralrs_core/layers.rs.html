191impl Module for F32RmsNorm {
192    fn forward(&self, xs: &Tensor) -> Result<Tensor> {
193        let initial_type = xs.dtype();
194        let mut xs = xs.to_dtype(DType::F32)?;
195        let var = xs.powf(2.)?.mean_keepdim(D::Minus1)?;
196        xs = xs.broadcast_mul(&(&var + self.eps)?.recip()?.sqrt()?)?;
197        xs.to_dtype(initial_type)?.broadcast_mul(&self.w)
198    }

279pub struct PhiRopeConfig {
280    pub rope_scaling: Option<PhiRopeScalingConfig>,
281    pub max_position_embeddings: usize,
282    pub original_max_position_embeddings: usize,

288impl PhiRotaryEmbedding {
289    fn new_classic_scaled(
290        short_factor: &[f64],
291        long_factor: &[f64],
292        scaling_type: &ScaledRopeType,
293        cfg: &PhiRopeConfig,
294        dtype: DType,
295        dev: &Device,
296    ) -> Result<Self> {
297        let max_seq_len = cfg.max_position_embeddings;
298        let dim = (cfg.head_dim as f64 * cfg.partial_rotary_factor.unwrap_or(1.)) as usize;
299
300        // Calculate scale
301        let scale =
302            cfg.max_position_embeddings as f64 / cfg.original_max_position_embeddings as f64;
303        let scaling_factor = if scale <= 1.0 {
304            1.0
305        } else {
306            match scaling_type {
307                ScaledRopeType::Su => {
308                    (1.0 + scale.ln() / (cfg.original_max_position_embeddings as f64).ln()).sqrt()
309                }
310                ScaledRopeType::Yarn => 0.1 * scale.ln() + 1.0,
311                _ => candle_core::bail!("Expected either `su` or `yarn` RoPE"),
312            }
313        };
314
315        // Calculate inv freqs for short, long
316        let inv_freq_long = (0..dim)
317            .step_by(2)
318            .enumerate()
319            .map(|(k, i)| {
320                (1f64 / (long_factor[k] * cfg.rope_theta.powf(i as f64 / dim as f64))) as f32
321            })
322            .collect::<Vec<_>>();
323        let inv_freq_short = (0..dim)
324            .step_by(2)
325            .enumerate()
326            .map(|(k, i)| {
327                (1f64 / (short_factor[k] * cfg.rope_theta.powf(i as f64 / dim as f64))) as f32
328            })
329            .collect::<Vec<_>>();
330        let inv_freq_len = inv_freq_long.len();
331
332        let t = Tensor::arange(0u32, max_seq_len as u32, dev)?
333            .to_dtype(DType::F32)?
334            .reshape((max_seq_len, 1))?;
335
336        // Calculate sin,cos for long
337        let inv_freq_long = Tensor::from_vec(inv_freq_long, (1, inv_freq_len), dev)?;
338        let freqs_long = t.matmul(&inv_freq_long)?;
339        let long_sin = freqs_long.sin()?.mul(scaling_factor)?.to_dtype(dtype)?;
340        let long_cos = freqs_long.cos()?.mul(scaling_factor)?.to_dtype(dtype)?;
341
342        // Calculate sin,cos for short
343        let inv_freq_short =
344            Tensor::from_vec(inv_freq_short, (1, inv_freq_len), dev)?.to_dtype(DType::F32)?;
345        let freqs_short = t.matmul(&inv_freq_short)?;
346        let short_sin = freqs_short.sin()?.mul(scaling_factor)?.to_dtype(dtype)?;
347        let short_cos = freqs_short.cos()?.mul(scaling_factor)?.to_dtype(dtype)?;
348
349        Ok(Self {
350            short_cos,
351            short_sin,
352            long_cos: Some(long_cos),
353            long_sin: Some(long_sin),
354            original_max_position_embeddings: cfg.original_max_position_embeddings,
355        })
356    }

358    fn new_unscaled(cfg: &PhiRopeConfig, dtype: DType, dev: &Device) -> Result<Self> {
359        let max_seq_len = cfg.max_position_embeddings;
360        let dim = (cfg.head_dim as f64 * cfg.partial_rotary_factor.unwrap_or(1.)) as usize;
361
362        let inv_freq: Vec<_> = (0..dim)
363            .step_by(2)
364            .map(|i| 1f32 / cfg.rope_theta.powf(i as f64 / dim as f64) as f32)
365            .collect();
366        let inv_freq_len = inv_freq.len();
367        let inv_freq = Tensor::from_vec(inv_freq, (1, inv_freq_len), dev)?;
368        let t = Tensor::arange(0u32, max_seq_len as u32, dev)?
369            .to_dtype(DType::F32)?
370            .reshape((max_seq_len, 1))?;
371        let freqs = t.matmul(&inv_freq)?;
372        let sin = freqs.sin()?.to_dtype(dtype)?;
373        let cos = freqs.cos()?.to_dtype(dtype)?;
374        Ok(Self {
375            short_cos: cos,
376            short_sin: sin,
377            long_cos: None,
378            long_sin: None,
379            original_max_position_embeddings: cfg.original_max_position_embeddings,
380        })
381    }

383    #[allow(clippy::too_many_arguments)]
384    fn new_scaled(
385        short_factor: &[f64],
386        long_factor: &[f64],
387        scaling_type: &ScaledRopeType,
388        long_mscale: f64,
389        short_mscale: f64,
390        cfg: &PhiRopeConfig,
391        dtype: DType,
392        dev: &Device,
393    ) -> Result<Self> {
394        let max_seq_len = cfg.max_position_embeddings;
395        let dim = (cfg.head_dim as f64 * cfg.partial_rotary_factor.unwrap_or(1.)) as usize;
396
397        if !matches!(scaling_type, ScaledRopeType::Su) {
398            candle_core::bail!("Scaled Phi3 RoPE (non-classic scaled, with mscales) must have type `su`/`longrope`.");
399        }
400
401        if short_factor.len() != dim / 2 {
402            candle_core::bail!(
403                "Misaligned length {}, expected {} for `su`/`longrope` short rescale factors",
404                short_factor.len(),
405                dim / 2
406            );
407        }
408        if long_factor.len() != dim / 2 {
409            candle_core::bail!(
410                "Misaligned length {}, expected {} for `su`/`longrope` long rescale factors",
411                long_factor.len(),
412                dim / 2
413            );
414        }
415
416        // Short cos/sin
417        let inv_freq_short: Vec<_> = (0..dim)
418            .step_by(2)
419            .enumerate()
420            .map(|(k, i)| {
421                1f32 / (short_factor[k] * cfg.rope_theta.powf(i as f64 / dim as f64)) as f32
422            })
423            .collect();
424        let inv_freq_len_short = inv_freq_short.len();
425        let inv_freq_short = Tensor::from_vec(inv_freq_short, (1, inv_freq_len_short), dev)?;
426        let t_short = Tensor::arange(0u32, max_seq_len as u32, dev)?
427            .to_dtype(DType::F32)?
428            .reshape((max_seq_len, 1))?;
429        let freqs_short = t_short.matmul(&inv_freq_short)?;
430        let sin_short = (freqs_short.sin()?.to_dtype(dtype)? * short_mscale)?;
431        let cos_short = (freqs_short.cos()?.to_dtype(dtype)? * short_mscale)?;
432
433        // Long cos/sin
434        let inv_freq_long: Vec<_> = (0..dim)
435            .step_by(2)
436            .enumerate()
437            .map(|(k, i)| {
438                1f32 / (long_factor[k] * cfg.rope_theta.powf(i as f64 / dim as f64)) as f32
439            })
440            .collect();
441        let inv_freq_len_long = inv_freq_long.len();
442        let inv_freq_long = Tensor::from_vec(inv_freq_long, (1, inv_freq_len_long), dev)?;
443        let t_long = Tensor::arange(0u32, max_seq_len as u32, dev)?
444            .to_dtype(DType::F32)?
445            .reshape((max_seq_len, 1))?;
446        let freqs_long = t_long.matmul(&inv_freq_long)?;
447        let sin_long = (freqs_long.sin()?.to_dtype(dtype)? * long_mscale)?;
448        let cos_long = (freqs_long.cos()?.to_dtype(dtype)? * long_mscale)?;
449        Ok(Self {
450            short_cos: cos_short,
451            short_sin: sin_short,
452            long_cos: Some(cos_long),
453            long_sin: Some(sin_long),
454            original_max_position_embeddings: cfg.original_max_position_embeddings,
455        })
456    }

491    /// Returns (sin, cos) taking into account LongRope
492    fn get_long_or_short_sin_cos(&self, position_ids: &[usize]) -> (&Tensor, &Tensor) {
493        if self.long_cos.is_none() {
494            return (&self.short_sin, &self.short_cos);
495        }
496        let seq_len = position_ids.iter().max().unwrap() + 1;
497        if seq_len > self.original_max_position_embeddings {
498            (
499                self.long_sin.as_ref().unwrap(),
500                self.long_cos.as_ref().unwrap(),
501            )
502        } else {
503            (&self.short_sin, &self.short_cos)
504        }
505    }

507    pub fn forward(
508        &self,
509        q: &Tensor,
510        k: &Tensor,
511        seqlen_offsets: &[usize],
512        position_ids: &[usize],
513    ) -> Result<(Tensor, Tensor)> {
514        let (sin, cos) = self.get_long_or_short_sin_cos(position_ids);
515        let (_b_sz, _h, seq_len, _n_embd) = q.dims4()?;
516
517        let rot_dim = cos.dim(D::Minus1)? * 2;
518
519        // Case for Phi 3 / Phi 4 mini
520        if rot_dim != q.dim(D::Minus1)? {
521            let rot_dim = cos.dim(D::Minus1)? * 2;
522            let q_rot = q.narrow(D::Minus1, 0, rot_dim)?;
523            let q_pass = q.narrow(D::Minus1, rot_dim, q.dim(D::Minus1)? - rot_dim)?;
524            let k_rot = k.narrow(D::Minus1, 0, rot_dim)?;
525            let k_pass = k.narrow(D::Minus1, rot_dim, k.dim(D::Minus1)? - rot_dim)?;
526
527            let (q_rot, k_rot) = if seqlen_offsets.len() == 1 {
528                let cos = cos.narrow(0, seqlen_offsets[0], seq_len)?;
529                let sin = sin.narrow(0, seqlen_offsets[0], seq_len)?;
530                let q_embed = candle_nn::rotary_emb::rope(&q_rot.contiguous()?, &cos, &sin)?;
531                let k_embed = candle_nn::rotary_emb::rope(&k_rot.contiguous()?, &cos, &sin)?;
532                (q_embed, k_embed)
533            } else {
534                let mut q_embeds = Vec::new();
535                let mut k_embeds = Vec::new();
536                for (i, offset) in seqlen_offsets.iter().enumerate() {
537                    let cos = cos.narrow(0, *offset, seq_len)?;
538                    let sin = sin.narrow(0, *offset, seq_len)?;
539                    let q_embed = candle_nn::rotary_emb::rope(
540                        &q_rot.i(i)?.unsqueeze(0)?.contiguous()?,
541                        &cos,
542                        &sin,
543                    )?;
544                    let k_embed = candle_nn::rotary_emb::rope(
545                        &k_rot.i(i)?.unsqueeze(0)?.contiguous()?,
546                        &cos,
547                        &sin,
548                    )?;
549                    q_embeds.push(q_embed);
550                    k_embeds.push(k_embed);
551                }
552                let q_rot = Tensor::cat(&q_embeds, 0)?;
553                let k_rot = Tensor::cat(&k_embeds, 0)?;
554                (q_rot, k_rot)
555            };
556
557            Ok((
558                Tensor::cat(&[q_rot, q_pass], D::Minus1)?.contiguous()?,
559                Tensor::cat(&[k_rot, k_pass], D::Minus1)?.contiguous()?,
560            ))
561        } else if seqlen_offsets.len() == 1 {
562            let cos = cos.narrow(0, seqlen_offsets[0], seq_len)?;
563            let sin = sin.narrow(0, seqlen_offsets[0], seq_len)?;
564            let q_embed = candle_nn::rotary_emb::rope(&q.contiguous()?, &cos, &sin)?;
565            let k_embed = candle_nn::rotary_emb::rope(&k.contiguous()?, &cos, &sin)?;
566            Ok((q_embed, k_embed))
567        } else {
568            let mut q_embeds = Vec::new();
569            let mut k_embeds = Vec::new();
570            for (i, offset) in seqlen_offsets.iter().enumerate() {
571                let cos = cos.narrow(0, *offset, seq_len)?;
572                let sin = sin.narrow(0, *offset, seq_len)?;
573                let q_embed =
574                    candle_nn::rotary_emb::rope(&q.i(i)?.unsqueeze(0)?.contiguous()?, &cos, &sin)?;
575                let k_embed =
576                    candle_nn::rotary_emb::rope(&k.i(i)?.unsqueeze(0)?.contiguous()?, &cos, &sin)?;
577                q_embeds.push(q_embed);
578                k_embeds.push(k_embed);
579            }
580            Ok((Tensor::cat(&q_embeds, 0)?, Tensor::cat(&k_embeds, 0)?))
581        }
582    }

1024        dtype: DType,

1102        &self,
1103        q: &Tensor,
1104        k: &Tensor,
1105        seqlen_offsets: &[usize],
1106    ) -> Result<(Tensor, Tensor)> {
1107        let (_b_sz, _h, seq_len, _n_embd) = q.dims4()?;
1108
1109        if seqlen_offsets.len() == 1 {
1110            let cos = self.cos.narrow(0, seqlen_offsets[0], seq_len)?;
1111            let sin = self.sin.narrow(0, seqlen_offsets[0], seq_len)?;
1112            let q_embed = candle_nn::rotary_emb::rope_i(&q.contiguous()?, &cos, &sin)?;
1113            let k_embed = candle_nn::rotary_emb::rope_i(&k.contiguous()?, &cos, &sin)?;
1114            Ok((q_embed, k_embed))
1115        } else {
1116            let mut q_embeds = Vec::new();
1117            let mut k_embeds = Vec::new();
1118            for (i, offset) in seqlen_offsets.iter().enumerate() {
1119                let cos = self.cos.narrow(0, *offset, seq_len)?;
1120                let sin = self.sin.narrow(0, *offset, seq_len)?;
1121                let q_embed = candle_nn::rotary_emb::rope_i(
1122                    &q.i(i)?.unsqueeze(0)?.contiguous()?,
1123                    &cos,
1124                    &sin,
1125                )?;
1126                let k_embed = candle_nn::rotary_emb::rope_i(
1127                    &k.i(i)?.unsqueeze(0)?.contiguous()?,
1128                    &cos,
1129                    &sin,
1130                )?;
1131                q_embeds.push(q_embed);
1132                k_embeds.push(k_embed);
1133            }
1134            Ok((Tensor::cat(&q_embeds, 0)?, Tensor::cat(&k_embeds, 0)?))
1135        }
1136    }

1179        let freqs = t.matmul(&inv_freq)?;
1180        let sin = freqs.sin()?.to_dtype(dtype)?;
1181        let cos = freqs.cos()?.to_dtype(dtype)?;
1182        Ok(Self {
1183            short_cos: cos,
1184            short_sin: sin,
1185            long_cos: None,
1186            long_sin: None,
1187            original_max_position_embeddings: cfg.original_max_position_embeddings,
1188        })
1189    }

1191    #[allow(clippy::too_many_arguments)]
1192    fn new_longrope(
1193        short_factor: &[f64],
1194        long_factor: &[f64],
1195        cfg: &Phi4MMConfig,
1196        dtype: DType,
1197        dev: &Device,
1198    ) -> Result<Self> {
1199        let max_seq_len = cfg.max_position_embeddings;
1200        let dim = (cfg.head_dim() as f64 * cfg.partial_rotary_factor) as usize;
1201
1202        // Calculate scale
1203        let scale =
1204            cfg.max_position_embeddings as f64 / cfg.original_max_position_embeddings as f64;
1205        let scaling_factor = if scale <= 1.0 {
1206            1.0
1207        } else {
1208            (1.0 + scale.ln() / (cfg.original_max_position_embeddings as f64).ln()).sqrt()
1209        };
1210
1211        // Short cos/sin
1212        let inv_freq_short: Vec<_> = (0..dim)
1213            .step_by(2)
1214            .enumerate()
1215            .map(|(k, i)| {
1216                1f32 / (short_factor[k] * cfg.rope_theta.powf(i as f64 / dim as f64)) as f32
1217            })
1218            .collect();
1219        let inv_freq_len_short = inv_freq_short.len();
1220        let inv_freq_short = Tensor::from_vec(inv_freq_short, (1, inv_freq_len_short), dev)?;
1221        let t_short = Tensor::arange(0u32, max_seq_len as u32, dev)?
1222            .to_dtype(DType::F32)?
1223            .reshape((max_seq_len, 1))?;
1224        let freqs_short = t_short.matmul(&inv_freq_short)?;
1225        let sin_short = (freqs_short.sin()?.to_dtype(dtype)? * scaling_factor)?;
1226        let cos_short = (freqs_short.cos()?.to_dtype(dtype)? * scaling_factor)?;
1227
1228        // Long cos/sin
1229        let inv_freq_long: Vec<_> = (0..dim)
1230            .step_by(2)
1231            .enumerate()
1232            .map(|(k, i)| {
1233                1f32 / (long_factor[k] * cfg.rope_theta.powf(i as f64 / dim as f64)) as f32
1234            })
1235            .collect();
1236        let inv_freq_len_long = inv_freq_long.len();
1237        let inv_freq_long = Tensor::from_vec(inv_freq_long, (1, inv_freq_len_long), dev)?;
1238        let t_long = Tensor::arange(0u32, max_seq_len as u32, dev)?
1239            .to_dtype(DType::F32)?
1240            .reshape((max_seq_len, 1))?;
1241        let freqs_long = t_long.matmul(&inv_freq_long)?;
1242        let sin_long = (freqs_long.sin()?.to_dtype(dtype)? * scaling_factor)?;
1243        let cos_long = (freqs_long.cos()?.to_dtype(dtype)? * scaling_factor)?;
1244
1245        Ok(Self {
1246            short_cos: cos_short,
1247            short_sin: sin_short,
1248            long_cos: Some(cos_long),
1249            long_sin: Some(sin_long),
1250            original_max_position_embeddings: cfg.original_max_position_embeddings,
1251        })
1252    }

1266    /// Returns (sin, cos) taking into account LongRope
1267    fn get_long_or_short_sin_cos(&self, position_ids: &[usize]) -> (&Tensor, &Tensor) {
1268        if self.long_cos.is_none() {
1269            return (&self.short_sin, &self.short_cos);
1270        }
1271        let seq_len = position_ids.iter().max().unwrap() + 1;
1272        if seq_len > self.original_max_position_embeddings {
1273            (
1274                self.long_sin.as_ref().unwrap(),
1275                self.long_cos.as_ref().unwrap(),
1276            )
1277        } else {
1278            (&self.short_sin, &self.short_cos)
1279        }
1280    }

1282    pub fn forward(
1283        &self,
1284        q: &Tensor,
1285        k: &Tensor,
1286        seqlen_offsets: &[usize],
1287        position_ids: &[usize],
1288    ) -> Result<(Tensor, Tensor)> {
1289        let (_b_sz, _h, seq_len, _n_embd) = q.dims4()?;
1290        let (sin, cos) = self.get_long_or_short_sin_cos(position_ids);
1291
1292        let rot_dim = cos.dim(D::Minus1)? * 2;
1293        let q_rot = q.narrow(D::Minus1, 0, rot_dim)?;
1294        let q_pass = q.narrow(D::Minus1, rot_dim, q.dim(D::Minus1)? - rot_dim)?;
1295        let k_rot = k.narrow(D::Minus1, 0, rot_dim)?;
1296        let k_pass = k.narrow(D::Minus1, rot_dim, k.dim(D::Minus1)? - rot_dim)?;
1297
1298        let (q_rot, k_rot) = if seqlen_offsets.len() == 1 {
1299            let cos = cos.narrow(0, seqlen_offsets[0], seq_len)?;
1300            let sin = sin.narrow(0, seqlen_offsets[0], seq_len)?;
1301            let q_embed = candle_nn::rotary_emb::rope(&q_rot.contiguous()?, &cos, &sin)?;
1302            let k_embed = candle_nn::rotary_emb::rope(&k_rot.contiguous()?, &cos, &sin)?;
1303            (q_embed, k_embed)
1304        } else {
1305            let mut q_embeds = Vec::new();
1306            let mut k_embeds = Vec::new();
1307            for (i, offset) in seqlen_offsets.iter().enumerate() {
1308                let cos = cos.narrow(0, *offset, seq_len)?;
1309                let sin = sin.narrow(0, *offset, seq_len)?;
1310                let q_embed = candle_nn::rotary_emb::rope(
1311                    &q_rot.i(i)?.unsqueeze(0)?.contiguous()?,
1312                    &cos,
1313                    &sin,
1314                )?;
1315                let k_embed = candle_nn::rotary_emb::rope(
1316                    &k_rot.i(i)?.unsqueeze(0)?.contiguous()?,
1317                    &cos,
1318                    &sin,
1319                )?;
1320                q_embeds.push(q_embed);
1321                k_embeds.push(k_embed);
1322            }
1323            let q_rot = Tensor::cat(&q_embeds, 0)?;
1324            let k_rot = Tensor::cat(&k_embeds, 0)?;
1325            (q_rot, k_rot)
1326        };
1327
1328        Ok((
1329            Tensor::cat(&[q_rot, q_pass], D::Minus1)?.contiguous()?,
1330            Tensor::cat(&[k_rot, k_pass], D::Minus1)?.contiguous()?,
1331        ))
1332    }
1333}

1573        &self,

1626            let q_embed = rope(&q.contiguous()?, &cos, &sin)?;
1627            let k_embed = rope(&k.contiguous()?, &cos, &sin)?;

1629        } else {
1630            let mut q_embeds = Vec::new();
1631            let mut k_embeds = Vec::new();
1632            for (i, offset) in seqlen_offsets.iter().enumerate() {
1633                let cos = self.cos.narrow(0, *offset, seq_len)?;

1790    where
1791        Self: Sized;
1792    fn any(&self) -> Result<bool>;
1793}

1795impl TensorInfExtend for Tensor {
1796    fn is_inf(&self) -> Result<Self> {
1797        self.broadcast_eq(&Tensor::new(f32::INFINITY, self.device())?.to_dtype(self.dtype())?)
1798    }
1799
1800    fn any(&self) -> Result<bool> {
1801        let sum = self.sum_all()?;
1802        match self.dtype() {
1803            DType::U8 => Ok(sum.to_scalar::<u8>()? == 0),
1804            DType::U32 => Ok(sum.to_scalar::<u32>()? == 0),
1805            DType::I16 => Ok(sum.to_scalar::<i16>()? == 0),
1806            DType::I32 => Ok(sum.to_scalar::<i32>()? == 0),
1807            DType::I64 => Ok(sum.to_scalar::<i64>()? == 0),
1808            DType::F16 => Ok(sum.to_scalar::<half::f16>()? == half::f16::from_f32_const(0.)),
1809            DType::BF16 => Ok(sum.to_scalar::<half::bf16>()? == half::bf16::from_f32_const(0.)),
1810            DType::F32 => Ok(sum.to_scalar::<f32>()? == 0.),
1811            DType::F64 => Ok(sum.to_scalar::<f64>()? == 0.),
1812            DType::F8E4M3 => Ok(sum.to_scalar::<F8E4M3>()? == F8E4M3::ZERO),
1813        }
1814    }

1817pub fn clamp_for_f16(xs: &Tensor) -> Result<Tensor> {
1818    let mut max = match xs.dtype() {
1819        DType::U8 => u8::MAX as f32 - 1000.,
1820        DType::U32 => u32::MAX as f32 - 1000.,
1821        DType::I16 => i16::MAX as f32 - 1000.,
1822        DType::I32 => i32::MAX as f32 - 1000.,
1823        DType::I64 => i64::MAX as f32 - 1000.,
1824        DType::F16 => half::f16::MAX.to_f32_const() - 1000.,
1825        DType::BF16 => half::bf16::MAX.to_f32_const() - 1000.,
1826        DType::F32 => f32::MAX - 1000.,
1827        DType::F64 => f64::MAX as f32 - 1000.,
1828        DType::F8E4M3 => F8E4M3::MAX.to_f32() - 1000.,
1829    };
1830    if xs.is_inf()?.any()? {
1831        max -= 1000.;
1832    }

2057/// Applies 2D reflection padding to a tensor of shape (N, C, H, W).
2058///
2059/// The `padding` argument is a 4-tuple (pad_left, pad_right, pad_top, pad_bottom).
2060/// For left padding, it reflects the values from column 1 up to pad_left (in reverse order);
2061/// for right padding, it reflects from the second-to-last column backwards, and similarly for
2062/// vertical (height) padding.
2063pub struct ReflectionPad2d {
2064    padding: (usize, usize, usize, usize),
2065}
2066
2067impl ReflectionPad2d {
2068    pub fn new(padding: (usize, usize, usize, usize)) -> Self {
2069        Self { padding }
2070    }
2071}
2072
2073impl Module for ReflectionPad2d {
2074    fn forward(&self, xs: &Tensor) -> Result<Tensor> {
2075        let (pad_left, pad_right, pad_top, pad_bottom) = self.padding;
2076
2077        let (_n, _c, h, w) = xs.dims4()?;
2078
2079        // --- Horizontal Padding (along width, axis = 3) ---
2080        // For left padding, we reflect columns 1..=pad_left (in reverse order).
2081        let left_pad = if pad_left > 0 {
2082            // Create indices: [pad_left, pad_left-1, ..., 1]
2083            let indices: Vec<i64> = (1..=pad_left as i64).rev().collect();
2084            Some(xs.index_select(&Tensor::new(indices, &Device::Cpu)?, 3)?)
2085        } else {
2086            None
2087        };
2088
2089        // For right padding, we reflect from the right side (excluding the last column).
2090        let right_pad = if pad_right > 0 {
2091            // For pad_right == 2, generate indices: [w-2, w-3, ... , w-1-pad_right]
2092            let start = w as i64 - 2;
2093            let indices: Vec<i64> = (0..pad_right as i64).map(|i| start - i).collect();
2094            Some(xs.index_select(&Tensor::new(indices, &Device::Cpu)?, 3)?)
2095        } else {
2096            None
2097        };
2098
2099        // Concatenate horizontally (along width, dim=3)
2100        let x_padded_width = match (left_pad, right_pad) {
2101            (Some(l), Some(r)) => Tensor::cat(&[l, xs.clone(), r], 3)?,
2102            (Some(l), None) => Tensor::cat(&[l, xs.clone()], 3)?,
2103            (None, Some(r)) => Tensor::cat(&[xs.clone(), r], 3)?,
2104            (None, None) => xs.clone(),
2105        };
2106
2107        // --- Vertical Padding (along height, axis = 2) ---
2108        // For top padding, reflect rows 1..=pad_top (in reverse order)
2109        let top_pad = if pad_top > 0 {
2110            let indices: Vec<i64> = (1..=pad_top as i64).rev().collect();
2111            Some(x_padded_width.index_select(&Tensor::new(indices, &Device::Cpu)?, 2)?)
2112        } else {
2113            None
2114        };
2115
2116        // For bottom padding, reflect from the bottom (excluding the last row)
2117        let bottom_pad = if pad_bottom > 0 {
2118            let start = h as i64 - 2;
2119            let indices: Vec<i64> = (0..pad_bottom as i64).map(|i| start - i).collect();
2120            Some(x_padded_width.index_select(&Tensor::new(indices, &Device::Cpu)?, 2)?)
2121        } else {
2122            None
2123        };
2124
2125        // Concatenate vertically (along height, dim=2)
2126        let x_padded = match (top_pad, bottom_pad) {
2127            (Some(t), Some(b)) => Tensor::cat(&[t, x_padded_width, b], 2)?,
2128            (Some(t), None) => Tensor::cat(&[t, x_padded_width], 2)?,
2129            (None, Some(b)) => Tensor::cat(&[x_padded_width, b], 2)?,
2130            (None, None) => x_padded_width,
2131        };
2132
2133        Ok(x_padded)
2134    }
2135}
