28/// Specialized QLoRA for no bias
29impl QLoraLinear {
30    #[allow(clippy::too_many_arguments)]
31    pub fn new(
32        old: QMatMul,
33        linear_config: &LoraLinearConfig,
34        config: &[((String, String), LoraConfig)],
35        vb: &ShardedVarBuilder,
36        ordering: &Ordering,
37        prefix: String,
38        count: &mut usize,
39        preload_adapters: &Option<HashMap<String, (ShardedVarBuilder, LoraConfig)>>,
40    ) -> Result<Self> {
41        let target_modules = &config.first().map(|c| &c.1.target_modules);
42        for (_, cfg) in config {
43            if target_modules
44                .as_ref()
45                .is_some_and(|target_modules| &cfg.target_modules != *target_modules)
46            {
47                candle_core::bail!("Expected all target modules to be the same.");
48            }
49        }
50
51        let old: Arc<dyn QuantMethod> = match old {
52            QMatMul::QTensor(q) => Arc::new(GgufMatMul::new(QuantMethodConfig::Gguf {
53                q_weight: q,
54                b: None,
55            })?),
56            QMatMul::TensorF16(t) | QMatMul::Tensor(t) => Arc::new(UnquantLinear::new(
57                QuantMethodConfig::Unquantized(Linear::new(t, None)),
58            )?),
59        };
60
61        let module = prefix.split('.').next_back().unwrap();
62        if target_modules.is_some_and(|target_modules| !target_modules.contains(module)) {
63            return Ok(Self {
64                old,
65                a_adapters: Either::Left(vec![]),
66                b_adapters: Either::Left(vec![]),
67                scale_adapters: vec![],
68                layer_n: usize::MAX,
69                merged: false,
70                adapters: HashMap::default(),
71            });
72        }
73
74        *count += 1;
75
76        let mut a_adapters = Vec::with_capacity(config.len());
77        let mut b_adapters = Vec::with_capacity(config.len());
78        let mut scale_adapters = Vec::with_capacity(config.len());
79        let vb = vb.pp(prefix.clone());
80        let a_vb = vb.pp("lora_A".to_string());
81        let b_vb = vb.pp("lora_B".to_string());
82        let mut state = None;
83        let mut all_same = true;
84        let mut adapters = HashMap::new();
85        for ((name_id, adapter_name), cfg) in config.iter() {
86            let a_pp = a_vb.pp(name_id);
87            let b_pp = b_vb.pp(name_id);
88            let adapter = make_adapter(a_pp, b_pp, cfg, linear_config)?;
89            a_adapters.push(adapter.a.clone());
90            b_adapters.push(adapter.b.clone());
91            scale_adapters.push(adapter.scale);
92            if state.is_some_and(|x| {
93                x == (
94                    cfg.rank,
95                    linear_config.in_features,
96                    linear_config.out_features,
97                    cfg.alpha,
98                    cfg.dropout,
99                )
100            }) || state.is_none()
101            {
102                state = Some((
103                    cfg.rank,
104                    linear_config.in_features,
105                    linear_config.out_features,
106                    cfg.alpha,
107                    cfg.dropout,
108                ));
109            } else {
110                all_same = false;
111            }
112            adapters.insert(adapter_name.clone(), adapter);
113        }
114
115        if let Some(preload_adapters) = preload_adapters {
116            all_same = false;
117            for (name, (vb, cfg)) in preload_adapters {
118                let a_vb = vb.set_prefix(a_vb.prefix());
119                let b_vb = vb.set_prefix(b_vb.prefix());
120                let adapter = make_adapter(a_vb, b_vb, cfg, linear_config)?;
121                adapters.insert(name.clone(), adapter);
122            }
123        }
124
125        let layer = if let Some(ref layers) = ordering.layers {
126            *layers.get(&prefix).unwrap()
127        } else {
128            0
129        };
130
131        if all_same {
132            let a_adapters_stack = Tensor::cat(
133                &a_adapters
134                    .iter()
135                    .map(|x| x.weight().unsqueeze(0))
136                    .collect::<Result<Vec<_>>>()?,
137                0,
138            )?;
139            let b_adapters_stack = Tensor::cat(
140                &b_adapters
141                    .iter()
142                    .map(|x| x.weight().unsqueeze(0))
143                    .collect::<Result<Vec<_>>>()?,
144                0,
145            )?;
146            #[allow(clippy::cast_possible_truncation)]
147            let scale_adapters_t = Tensor::from_vec(
148                scale_adapters.iter().map(|&x| x as f32).collect::<Vec<_>>(),
149                (scale_adapters.len(), 1, 1),
150                a_adapters_stack.device(),
151            )?
152            .to_dtype(a_adapters_stack.dtype())?;
153            let a_adapters_stack = a_adapters_stack.broadcast_mul(&scale_adapters_t)?;
154            Ok(QLoraLinear {
155                old,
156                a_adapters: Either::Right((a_adapters_stack.clone(), a_adapters)),
157                b_adapters: Either::Right((b_adapters_stack.clone(), b_adapters)),
158                scale_adapters,
159                layer_n: layer,
160                merged: false,
161                adapters,
162            })
163        } else {
164            Ok(QLoraLinear {
165                old,
166                a_adapters: Either::Left(a_adapters),
167                b_adapters: Either::Left(b_adapters),
168                scale_adapters,
169                layer_n: layer,
170                merged: false,
171                adapters,
172            })
173        }
174    }
175}

177impl Merge for QLoraLinear {
178    fn get_delta_weight(&self, adapter: usize) -> Result<Tensor> {
179        match (&self.a_adapters, &self.b_adapters) {
180            (Either::Left(a), Either::Left(b)) | (Either::Right((_, a)), Either::Right((_, b))) => {
181                let w_a = a[adapter].weight();
182                let w_b = b[adapter].weight();
183
184                MatMul.matmul(w_b, w_a)? * self.scale_adapters[adapter]
185            }
186            _ => unreachable!("Both adapters must be Either::Left or Either::Right."),
187        }
188    }
189
190    fn merge_weights(&mut self) -> Result<()> {
191        let mut w_base_layer: Option<Tensor> = None;
192        for adapter in 0..self.scale_adapters.len() {
193            if let Some(w_base_layer) = &mut w_base_layer {
194                *w_base_layer = (&*w_base_layer + &self.get_delta_weight(adapter)?)?;
195            } else {
196                w_base_layer = Some(self.get_delta_weight(adapter)?)
197            }
198        }
199        self.old
200            .add_delta_w(w_base_layer.as_ref().expect("Found no adapters to merge."))?;
201        self.merged = true;
202        Ok(())
203    }
204}

206impl LinearLayerLike for QLoraLinear {
207    fn quant_inner(&mut self) -> &mut Arc<dyn QuantMethod> {
208        &mut self.old
209    }
210    fn bias(&self) -> Option<&Tensor> {
211        None
212    }
213    fn weight(&self) -> &Tensor {
214        unimplemented!()
215    }
216    fn quantized_act_type(&self) -> Option<DType> {
217        Some(DType::F32)
218    }
219    fn lora_forward(
220        &self,
221        input: &Tensor,
222        scalings: Option<Tensor>,
223        global_scaling_weight: f64,
224        is_scaling_pass: Option<f64>,
225    ) -> Result<Tensor> {
226        //No fan_in_fan_out so no weight.transpose(0,1)
227        let mut result = self.old.forward(input)?;
228        if self.merged {
229            return Ok(result);
230        }
231
232        if self
233            .a_adapters
234            .as_ref()
235            .left()
236            .is_some_and(|x| x.is_empty())
237            || (is_scaling_pass.is_some_and(|x| x == 0.))
238        {
239            return Ok(result);
240        }
241
242        let scalings =
243            scalings.map(|scalings| get_maybe_topk_scalings(scalings, self.layer_n).unwrap());
244        if self.a_adapters.is_left()
245            || scalings
246                .as_ref()
247                .is_some_and(|scalings| scalings.dims3().unwrap().1 != 1)
248        {
249            let a_adapters = if self.a_adapters.is_right() {
250                self.a_adapters.as_ref().unwrap_right().1.clone()
251            } else {
252                self.a_adapters.as_ref().unwrap_left().clone()
253            };
254            let b_adapters = if self.b_adapters.is_right() {
255                self.b_adapters.as_ref().unwrap_right().1.clone()
256            } else {
257                self.b_adapters.as_ref().unwrap_left().clone()
258            };
259            for (i, (adapter_a, (adapter_b, adapter_scale))) in
260                zip(a_adapters, zip(b_adapters, &self.scale_adapters)).enumerate()
261            {
262                let input_new = if let Some(scalings) = &scalings {
263                    apply_scalings_to_x(input.clone(), scalings, i)?
264                } else {
265                    input.clone()
266                };
267
268                let res = adapter_b
269                    .forward(&adapter_a.forward(&input_new)?)?
270                    .mul(*adapter_scale)?
271                    .mul(global_scaling_weight)?;
272                result = (result + res)?;
273            }
274            Ok(result)
275        } else {
276            let adapter_a = &self.a_adapters.as_ref().unwrap_right().0;
277            let adapter_b = &self.b_adapters.as_ref().unwrap_right().0;
278            let adapter_scales = &self.scale_adapters;
279            let n_adapters = adapter_scales.len();
280            let adapter_a = if let Some(scalings) = scalings.as_ref() {
281                let scalings = scalings
282                    .squeeze(0)?
283                    .squeeze(0)?
284                    .unsqueeze(1)?
285                    .unsqueeze(1)?;
286                adapter_a
287                    .broadcast_mul(&scalings)?
288                    .mul(global_scaling_weight)?
289            } else {
290                adapter_a.clone().mul(global_scaling_weight)?
291            };
292
293            let (b, s, h) = input.dims3()?;
294            let input = input.reshape((b * s, h))?;
295            let out = adapter_a.broadcast_matmul(&input.t()?)?;
296            let out = adapter_b.broadcast_matmul(&out)?;
297            let o_h = out.dims()[1];
298            let out = out.reshape((n_adapters, b, s, o_h))?;
299            let out = out.sum(0)?;
300            out + result
301        }
302    }
303    fn is_lora(&self) -> bool {
304        !self.adapters.is_empty()
305    }
