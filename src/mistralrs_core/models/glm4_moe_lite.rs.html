99impl QProj {
100    fn forward(&self, xs: &Tensor) -> Result<Tensor> {
101        match self {
102            Self::Lora { a, norm, b } => {
103                b.forward_autocast(&norm.forward(&a.forward_autocast(xs)?)?)

123impl Attention {
124    #[allow(clippy::too_many_arguments)]
125    fn new(
126        rotary_emb: Arc<DeepSeekV2RotaryEmbedding>,
127        cfg: &Glm4MoeLiteConfig,
128        vb: ShardedVarBuilder,
129        mapper: &dyn DeviceMapper,
130        layer_idx: usize,
131        loading_isq: bool,
132        paged_attn: Option<PagedAttention>,
133        comm: &Arc<mistralrs_quant::Comm>,
134    ) -> Result<Self> {
135        let q_head_dim = cfg.q_head_dim();
136
137        // GLM4MoeLite always uses LoRA for Q projection
138        let q = {
139            let a = ReplicatedLayer::new(
140                cfg.hidden_size,
141                cfg.q_lora_rank,
142                &cfg.quantization_config,
143                false,
144                mapper.set_device(layer_idx, vb.pp("q_a_proj"), loading_isq),
145            )?;
146            let norm = RmsNorm::new(
147                cfg.q_lora_rank,
148                cfg.rms_norm_eps,
149                mapper.set_device(layer_idx, vb.pp("q_a_layernorm"), false),
150            )?;
151            let b = ColumnParallelLayer::new(
152                cfg.q_lora_rank,
153                cfg.num_attention_heads * q_head_dim,
154                &cfg.quantization_config,
155                false,
156                comm,
157                mapper.set_device(layer_idx, vb.pp("q_b_proj"), loading_isq),
158            )?;
159            QProj::Lora { a, norm, b }
160        };
161
162        let kv_a_proj_with_mqa = ReplicatedLayer::new(
163            cfg.hidden_size,
164            cfg.kv_lora_rank + cfg.qk_rope_head_dim,
165            &cfg.quantization_config,
166            false,
167            mapper.set_device(layer_idx, vb.pp("kv_a_proj_with_mqa"), loading_isq),
168        )?;
169        let kv_a_layernorm = RmsNorm::new(
170            cfg.kv_lora_rank,
171            cfg.rms_norm_eps,
172            mapper.set_device(layer_idx, vb.pp("kv_a_layernorm"), false),
173        )?;
174        let kv_b_proj = ColumnParallelLayer::new(
175            cfg.kv_lora_rank,
176            cfg.num_attention_heads * (q_head_dim - cfg.qk_rope_head_dim + cfg.v_head_dim),
177            &cfg.quantization_config,
178            false,
179            comm,
180            mapper.set_device(layer_idx, vb.pp("kv_b_proj"), loading_isq),
181        )?;
182
183        let o_proj = RowParallelLayer::new(
184            cfg.num_attention_heads * cfg.v_head_dim,
185            cfg.hidden_size,
186            &cfg.quantization_config,
187            false,
188            comm,
189            mapper.set_device(layer_idx, vb.pp("o_proj"), loading_isq),
190        )?;
191
192        Ok(Self {
193            q,
194            kv_a_proj_with_mqa,
195            kv_a_layernorm,
196            kv_b_proj,
197            o_proj,
198            rotary_emb,
199            cfg: cfg.clone(),
200            q_head_dim,
201            paged_attn,
202            num_attention_heads: cfg.num_attention_heads / comm.world_size(),
203            sdpa_params: SdpaParams {
204                n_kv_groups: 1,
205                softcap: None,
206                softmax_scale: cfg.softmax_scale(),
207                sliding_window: None,
208            },
209        })
210    }
211
212    fn forward(
213        &self,
214        xs: &Tensor,
215        attention_mask: Option<&Tensor>,
216        seqlen_offsets: &[usize],
217        kv_cache: &mut KvCache,
218        metadata: Option<((Tensor, Tensor), &PagedAttentionInputMetadata)>,
219        flash_params: &FlashParams,
220    ) -> Result<Tensor> {
221        let (bs, seq_len, _) = xs.dims3()?;
222
223        let mut q = self.q.forward(xs)?;
224        q = q
225            .reshape((bs, seq_len, self.num_attention_heads, self.q_head_dim))?
226            .transpose(1, 2)?;
227        let q_split = q.split(
228            &[self.cfg.qk_nope_head_dim, self.cfg.qk_rope_head_dim],
229            D::Minus1,
230        )?;
231        let q_nope = q_split[0].clone();
232        let mut q_pe = q_split[1].clone();
233
234        let mut compressed_kv = self.kv_a_proj_with_mqa.forward_autocast(xs)?;
235        let ckv_split = compressed_kv.split(
236            &[self.cfg.kv_lora_rank, self.cfg.qk_rope_head_dim],
237            D::Minus1,
238        )?;
239        compressed_kv = ckv_split[0].clone();
240        let mut k_pe = ckv_split[1].clone();
241        k_pe = k_pe
242            .reshape((bs, seq_len, 1, self.cfg.qk_rope_head_dim))?
243            .transpose(1, 2)?;
244        let mut kv = self
245            .kv_b_proj
246            .forward_autocast(&self.kv_a_layernorm.forward(&compressed_kv)?)?;
247        kv = kv
248            .reshape((
249                bs,
250                seq_len,
251                self.num_attention_heads,
252                self.cfg.qk_nope_head_dim + self.cfg.v_head_dim,
253            ))?
254            .transpose(1, 2)?;
255
256        let kv_split = kv.split(&[self.cfg.qk_nope_head_dim, self.cfg.v_head_dim], D::Minus1)?;
257        let k_nope = kv_split[0].clone();
258        let mut v = kv_split[1].clone();
259
260        (q_pe, k_pe) = self.rotary_emb.forward(&q_pe, &k_pe, seqlen_offsets)?;
261
262        let q = Tensor::cat(&[&q_nope, &q_pe], D::Minus1)?.contiguous()?;
263        let mut k = Tensor::cat(
264            &[&k_nope, &k_pe.repeat((1, self.num_attention_heads, 1, 1))?],
265            D::Minus1,
266        )?
267        .contiguous()?;
268
269        let mut attn_out = match &self.paged_attn {
270            Some(paged_attn) => match metadata {
271                Some(((key_cache, value_cache), input_metadata)) => {
272                    let v = v
273                        .pad_with_zeros(D::Minus1, 0, self.q_head_dim - self.cfg.v_head_dim)?
274                        .contiguous()?;
275                    paged_attn
276                        .forward(
277                            &q,
278                            &k,
279                            &v,
280                            attention_mask,
281                            Some(key_cache),
282                            Some(value_cache),
283                            input_metadata,
284                            &self.sdpa_params,
285                            Some(flash_params),
286                        )?
287                        .narrow(D::Minus1, 0, self.cfg.v_head_dim)?
288                }
289                None => {
290                    // If we don't have metadata, we are most likely generating an imatrix so we don't want to populate that.
291                    // Generating the dummy metadata with the assumption that we are not generating text (only processing prompts).
292                    let input_metadata = PagedAttentionInputMetadata::dummy(q.device())?;
293                    // Sanity check.
294                    assert!(attention_mask.is_some());
295                    let v = v
296                        .pad_with_zeros(D::Minus1, 0, self.q_head_dim - self.cfg.v_head_dim)?
297                        .contiguous()?;
298                    paged_attn
299                        .forward(
300                            &q,
301                            &k,
302                            &v,
303                            attention_mask,
304                            None,
305                            None,
306                            &input_metadata,
307                            &self.sdpa_params,
308                            Some(flash_params),
309                        )?
310                        .narrow(D::Minus1, 0, self.cfg.v_head_dim)?
311                }
312            },
313            None => {
314                (k, v) = kv_cache.append(&k, &v)?;
315
316                Sdpa.run_attention(
317                    &q,
318                    &k,
319                    &v,
320                    attention_mask,
321                    Some(flash_params),
322                    &self.sdpa_params,
323                )?
324            }
325        };
326
327        attn_out = if attention_mask.is_some() {
328            attn_out.transpose(1, 2)?.reshape((bs, seq_len, ()))?
329        } else {
330            attn_out.reshape((bs, seq_len, ()))?
331        };
332
333        self.o_proj.forward_autocast(&attn_out)
334    }
335}

337struct Expert {
338    gate: Arc<dyn QuantMethod>,
339    up: Arc<dyn QuantMethod>,
340    down: Arc<dyn QuantMethod>,
341    act: Activation,
342}
343
344impl Expert {
345    fn new(
346        cfg: &Glm4MoeLiteConfig,
347        vb: ShardedVarBuilder,
348        hidden_size: Option<usize>,
349        intermediate_size: Option<usize>,
350    ) -> Result<Self> {
351        let hidden_size = hidden_size.unwrap_or(cfg.hidden_size);
352        let intermediate_size = intermediate_size.unwrap_or(cfg.intermediate_size);
353
354        Ok(Self {
355            gate: ReplicatedLayer::new(
356                hidden_size,
357                intermediate_size,
358                &cfg.quantization_config,
359                false,
360                vb.pp("gate_proj"),
361            )?,
362            up: ReplicatedLayer::new(
363                hidden_size,
364                intermediate_size,
365                &cfg.quantization_config,
366                false,
367                vb.pp("up_proj"),
368            )?,
369            down: ReplicatedLayer::new(
370                intermediate_size,
371                hidden_size,
372                &cfg.quantization_config,
373                false,
374                vb.pp("down_proj"),
375            )?,
376            act: cfg.hidden_act,
377        })
378    }
379
380    fn forward(&self, xs: &Tensor) -> Result<Tensor> {
381        let original_dtype = xs.dtype();
382        let mut xs = xs.clone();
383        if let Some(t) = self.gate.quantized_act_type() {
384            xs = xs.to_dtype(t)?;
385        }
386        let lhs = self.gate.forward(&xs)?;
387        let rhs = self.up.forward(&xs)?;
388        let mut res = self
389            .down
390            .forward(&crate::ops::mul_and_act(&lhs, &rhs, self.act)?)?;
391        if self.gate.quantized_act_type().is_some() {
392            res = res.to_dtype(original_dtype)?;
393        }
394        Ok(res)
395    }
396}

398struct MoeGate {
399    weight: Tensor,
400    cfg: Glm4MoeLiteConfig,
401    top_k: usize,
402    n_routed_experts: usize,
403    e_score_correction_bias: Tensor,
404}
405
406impl MoeGate {
407    fn new(
408        cfg: &Glm4MoeLiteConfig,
409        vb: ShardedVarBuilder,
410        n_routed_experts: usize,
411    ) -> Result<Self> {
412        let weight = vb.get((n_routed_experts, cfg.hidden_size), "weight")?;
413        // GLM4MoeLite uses NoAuxTc routing with e_score_correction_bias
414        let e_score_correction_bias = vb.get_with_hints_dtype(
415            n_routed_experts,
416            "e_score_correction_bias",
417            Default::default(),
418            DType::F32,
419        )?;
420        Ok(Self {
421            weight,
422            cfg: cfg.clone(),
423            top_k: cfg.num_experts_per_tok,
424            n_routed_experts,
425            e_score_correction_bias,
426        })
427    }
428
429    /// (topk_idx, topk_weight)
430    fn forward(&self, xs: &Tensor) -> Result<(Tensor, Tensor)> {
431        let (bs, seq_len, h) = xs.dims3()?;
432        // Compute gating score
433        let xs = xs.reshape(((), h))?;
434        let logits = xs
435            .to_dtype(DType::F32)?
436            .broadcast_matmul(&self.weight.t()?.to_dtype(DType::F32)?)?;
437        // GLM4MoeLite uses sigmoid scoring
438        let scores = candle_nn::ops::sigmoid(&logits)?;
439
440        // NoAuxTc routing with e_score_correction_bias
441        let scores_for_choice = scores
442            .reshape((bs * seq_len, ()))?
443            .broadcast_add(&self.e_score_correction_bias.unsqueeze(0)?)?;
444        // (n, n_group)
445        let group_scores = scores_for_choice
446            .reshape((bs * seq_len, self.cfg.n_group, ()))?
447            .topk(2)?
448            .values
449            .sum(D::Minus1)?;
450        // (n, topk_group)
451        let group_idx = group_scores.topk(self.cfg.topk_group)?.indices;
452        // (n, n_group)
453        let mut group_mask = group_scores.zeros_like()?;
454        // (n, n_group)
455        group_mask = group_mask.scatter_add(
456            &group_idx,
457            &group_idx.ones_like()?.to_dtype(group_mask.dtype())?,
458            1,
459        )?;
460        // (n, e)
461        let score_mask = group_mask
462            .unsqueeze(D::Minus1)?
463            .expand((
464                bs * seq_len,
465                self.cfg.n_group,
466                self.n_routed_experts / self.cfg.n_group,
467            ))?
468            .reshape((bs * seq_len, ()))?;
469        // (n, e)
470        // Invert the mask
471        let tmp_scores = scores_for_choice.broadcast_mul(&score_mask)?;
472        let topk_idx = tmp_scores.topk(self.top_k)?.indices;
473        let mut topk_weight = scores.gather(&topk_idx, 1)?;
474
475        // Normalize with sigmoid
476        let denominator = (topk_weight.sum_keepdim(D::Minus1)? + 1e-20)?;
477        topk_weight = topk_weight.broadcast_div(&denominator)?;
478
479        // Must multiply the scaling factor
480        topk_weight = (topk_weight * self.cfg.routed_scaling_factor)?;
481
482        Ok((topk_idx, topk_weight))
483    }
484}

486struct Moe {
487    experts: MoEExperts,
488    shared_experts: Option<Expert>,
489    gate: MoeGate,
490}
491
492impl Moe {
493    #[allow(clippy::too_many_arguments)]
494    fn new(
495        cfg: &Glm4MoeLiteConfig,
496        vb: ShardedVarBuilder,
497        mapper: &dyn DeviceMapper,
498        layer_idx: usize,
499        loading_isq: bool,
500        n_shared_experts: usize,
501        n_routed_experts: usize,
502        comm: &Arc<mistralrs_quant::Comm>,
503        real_device: Device,
504    ) -> Result<Self> {
505        let layer_device = mapper
506            .device_for(layer_idx, false)
507            .cloned()
508            .unwrap_or(real_device);
509
510        let moe_cfg = MoEExpertsConfig {
511            num_experts: n_routed_experts,
512            num_experts_per_tok: cfg.num_experts_per_tok,
513            hidden_size: cfg.hidden_size,
514            moe_intermediate_size: cfg.moe_intermediate_size,
515        };
516
517        // Use the optimized MoEExperts with automatic backend selection
518        let experts = MoEExperts::new(
519            &moe_cfg,
520            mapper.set_device(layer_idx, vb.clone(), loading_isq),
521            layer_device,
522            comm,
523            loading_isq,
524            &cfg.quantization_config,
525            cfg.hidden_act,
526        )?;
527
528        // Shared experts are handled separately
529        let shared_experts = if n_shared_experts > 0 {
530            Some(Expert::new(
531                cfg,
532                mapper.set_device(layer_idx, vb.pp("shared_experts"), loading_isq),
533                None,
534                Some(cfg.moe_intermediate_size),
535            )?)
536        } else {
537            None
538        };
539
540        let gate = MoeGate::new(
541            cfg,
542            mapper.set_device(layer_idx, vb.pp("gate"), false),
543            n_routed_experts,
544        )?;
545
546        Ok(Self {
547            experts,
548            shared_experts,
549            gate,
550        })
551    }
552
553    fn forward(&self, xs: &Tensor) -> Result<Tensor> {
554        let identity = xs.clone();
555        let (b_size, seq_len, hidden_dim) = xs.dims3()?;
556
557        // Get routing weights from custom gate (NoAuxTc with e_score_correction_bias)
558        let (topk_idx, topk_weight) = self.gate.forward(xs)?;
559
560        // Forward through routed experts using optimized MoEExperts
561        let mut y = self.experts.forward(xs, topk_weight, &topk_idx)?;
562        y = y.reshape((b_size, seq_len, hidden_dim))?;
563
564        // Add shared expert output
565        if let Some(ref shared_experts) = self.shared_experts {
566            y = (y + shared_experts.forward(&identity)?)?;
567        }
568
569        Ok(y)
570    }
571
572    fn get_isq_layers(&mut self) -> Vec<&mut Arc<dyn QuantMethod>> {
573        let mut layers = self.experts.get_isq_layers();
574        if let Some(ref mut shared) = self.shared_experts {
575            layers.push(&mut shared.gate);
576            layers.push(&mut shared.up);
577            layers.push(&mut shared.down);
578        }
579        layers
580    }
581}

604impl DecoderLayer {
605    #[allow(clippy::too_many_arguments)]
606    fn new(
607        rotary_emb: Arc<DeepSeekV2RotaryEmbedding>,
608        cfg: &Glm4MoeLiteConfig,
609        vb: ShardedVarBuilder,
610        mapper: &dyn DeviceMapper,
611        layer_idx: usize,
612        loading_isq: bool,
613        paged_attn: Option<PagedAttention>,
614        comm: &Arc<mistralrs_quant::Comm>,
615        real_device: Device,
616    ) -> Result<Self> {
617        let attn = Attention::new(
618            rotary_emb,
619            cfg,
620            vb.pp("self_attn"),
621            mapper,
622            layer_idx,
623            loading_isq,
624            paged_attn,
625            comm,
626        )?;
627        let input_layernorm = RmsNorm::new(
628            cfg.hidden_size,
629            cfg.rms_norm_eps,
630            mapper.set_device(layer_idx, vb.pp("input_layernorm"), false),
631        )?;
632        let post_attention_layernorm = RmsNorm::new(
633            cfg.hidden_size,
634            cfg.rms_norm_eps,
635            mapper.set_device(layer_idx, vb.pp("post_attention_layernorm"), false),
636        )?;
637        // Layer 0 uses dense MLP (first_k_dense_replace=1 by default), other layers use MoE
638        let moe_or_mlp =
639            if layer_idx >= cfg.first_k_dense_replace && layer_idx % cfg.moe_layer_freq == 0 {
640                MoeOrMlp::Moe(Box::new(Moe::new(
641                    cfg,
642                    vb.pp("mlp"),
643                    mapper,
644                    layer_idx,
645                    loading_isq,
646                    cfg.n_shared_experts,
647                    cfg.n_routed_experts,
648                    comm,
649                    real_device,
650                )?))
651            } else {
652                MoeOrMlp::Mlp(Mlp::new(
653                    mapper.set_device(layer_idx, vb.pp("mlp"), loading_isq),
654                    cfg.hidden_size,
655                    cfg.intermediate_size,
656                    &cfg.quantization_config,
657                    cfg.hidden_act,
658                    comm,
659                )?)
660            };
661
662        Ok(Self {
663            input_layernorm,
664            post_attention_layernorm,
665            attn,
666            moe_or_mlp,
667        })
668    }
669
670    fn forward(
671        &self,
672        xs: &Tensor,
673        attention_mask: Option<&Tensor>,
674        seqlen_offsets: &[usize],
675        kv_cache: &mut KvCache,
676        metadata: Option<((Tensor, Tensor), &PagedAttentionInputMetadata)>,
677        flash_params: &FlashParams,
678    ) -> Result<Tensor> {
679        let residual = xs;
680        let xs = self.input_layernorm.forward(xs)?;
681        let xs = self.attn.forward(
682            &xs,
683            attention_mask,
684            seqlen_offsets,
685            kv_cache,
686            metadata,
687            flash_params,
688        )?;
689        let xs = (xs + residual)?;
690        let residual = &xs;
691        let xs = self
692            .moe_or_mlp
693            .forward(&xs.apply(&self.post_attention_layernorm)?)?;
694        residual + xs
695    }
696}

698pub struct Glm4MoeLite {
699    lm_head: Arc<dyn QuantMethod>,
700    embed_tokens: Embedding,
701    norm: RmsNorm,
702    layers: Vec<DecoderLayer>,
703    cache: EitherCache,
704    device: Device,
705    max_seq_len: usize,
706    cfg: ModelConfigMetadata,
707    mapper: Box<dyn DeviceMapper + Send + Sync>,
708}
709
710impl Glm4MoeLite {
711    pub fn new(
712        cfg: &Glm4MoeLiteConfig,
713        vb: ShardedVarBuilder,
714        _is_gptx: bool,
715        normal_loading_metadata: NormalLoadingMetadata,
716        attention_mechanism: AttentionImplementation,
717    ) -> Result<Self> {
718        let vb_m = vb.pp("model");
719
720        let mapper = normal_loading_metadata.mapper;
721
722        let embed_tokens = embedding(
723            cfg.vocab_size,
724            cfg.hidden_size,
725            mapper.set_nm_device(vb_m.pp("embed_tokens"), false),
726            &cfg.quantization_config,
727        )?;
728        let lm_head = if !cfg.tie_word_embeddings {
729            ReplicatedLayer::new(
730                cfg.hidden_size,
731                cfg.vocab_size,
732                &cfg.quantization_config,
733                false,
734                mapper.set_nm_device(vb.pp("lm_head"), normal_loading_metadata.loading_isq),
735            )?
736        } else {
737            ReplicatedLayer::from_linear(candle_nn::Linear::new(
738                mapper.cast_nm_device(
739                    embed_tokens.embeddings(),
740                    normal_loading_metadata.loading_isq,
741                )?,
742                None,
743            ))?
744        };
745        let norm = RmsNorm::new(
746            cfg.hidden_size,
747            cfg.rms_norm_eps,
748            mapper.set_nm_device(vb_m.pp("norm"), false),
749        )?;
750
751        let mut ropes = HashMap::new();
752        let rope_cfg = DeepSeekV2RopeConfig {
753            rope_scaling: None,
754            max_position_embeddings: cfg.max_position_embeddings,
755            rope_theta: cfg.rope_theta,
756            qk_rope_head_dim: cfg.qk_rope_head_dim,
757        };
758        for i in 0..cfg.num_hidden_layers {
759            let device = mapper
760                .device_for(i, false)
761                .unwrap_or(&normal_loading_metadata.real_device);
762            ropes.insert(
763                device.location(),
764                Arc::new(DeepSeekV2RotaryEmbedding::new(
765                    &rope_cfg,
766                    vb.dtype(),
767                    device,
768                )?),
769            );
770        }
771
772        let vb_l = vb_m.pp("layers");
773        let layers: Vec<DecoderLayer> = NiceProgressBar::<_, 'b'>(
774            0..cfg.num_hidden_layers,
775            "Loading repeating layers",
776            &normal_loading_metadata.multi_progress,
777        )
778        .par_iter_if_isq(|layer_idx| {
779            let device = mapper
780                .device_for(layer_idx, false)
781                .unwrap_or(&normal_loading_metadata.real_device);
782            let rotary_emb = ropes
783                .get(&device.location())
784                .expect("No RoPE for device location!")
785                .clone();
786            let paged_attn = match &attention_mechanism {
787                AttentionImplementation::Eager => None,
788                AttentionImplementation::PagedAttention => Some(
789                    PagedAttention::new(cfg.v_head_dim, device, None)
790                        .expect("Failed to create PagedAttention"),
791                ),
792            };
793            let comm = mapper.get_comm_for(layer_idx)?;
794            DecoderLayer::new(
795                rotary_emb.clone(),
796                cfg,
797                vb_l.pp(layer_idx),
798                &*mapper,
799                layer_idx,
800                normal_loading_metadata.loading_isq,
801                paged_attn,
802                &comm,
803                normal_loading_metadata.real_device.clone(),
804            )
805        })?;
806
807        Ok(Self {
808            lm_head,
809            embed_tokens,
810            norm,
811            layers,
812            cache: EitherCache::Normal(NormalCache::new(
813                cfg.num_hidden_layers,
814                cfg.max_position_embeddings,
815            )),
816            device: normal_loading_metadata.real_device.clone(),
817            max_seq_len: cfg.max_position_embeddings,
818            cfg: ModelConfigMetadata {
819                max_seq_len: cfg.max_position_embeddings,
820                num_layers: cfg.num_hidden_layers,
821                hidden_size: cfg.hidden_size,
822                num_kv_heads: (cfg.num_attention_heads / mapper.get_comm_for(0)?.world_size())
823                    .max(1),
824                num_attn_heads: (cfg.num_attention_heads / mapper.get_comm_for(0)?.world_size())
825                    .max(1),
826                sliding_window: None,
827                k_head_dim: cfg.q_head_dim(),
828                v_head_dim: if matches!(
829                    attention_mechanism,
830                    AttentionImplementation::PagedAttention
831                ) {
832                    cfg.q_head_dim()
833                } else {
834                    cfg.v_head_dim
835                },
836            },
837            mapper,
838        })
839    }
840
841    #[allow(clippy::too_many_arguments)]
842    pub fn forward(
843        &self,
844        input_ids: &Tensor,
845        seqlen_offsets: &[usize],
846        context_lens: Vec<(usize, usize)>,
847        metadata: Option<(Vec<(Tensor, Tensor)>, &PagedAttentionInputMetadata)>,
848        flash_params: &FlashParams,
849    ) -> Result<Tensor> {
850        let mut xs = self.embed_tokens.forward(input_ids)?;
851        let cache = &mut self.cache.normal().0;
852        let attention_mask = CausalMasker.make_causal_mask_matrix(
853            input_ids,
854            metadata
855                .as_ref()
856                .map(|(_, _)| &seqlen_offsets as &dyn PastKvLenCache)
857                .unwrap_or(cache as &dyn PastKvLenCache),
858            xs.dtype(),
859            self.cfg.num_attn_heads,
860        )?;
861        // PagedAttention prompt chunking
862        let attention_mask = attention_mask.filter(|_| {
863            metadata
864                .as_ref()
865                .map(|(_, meta)| meta.is_first_prompt_chunk)
866                .unwrap_or(true)
867        });
868        for (i, layer) in self.layers.iter().enumerate() {
869            xs = self.mapper.map(xs, i)?;
870            xs = layer.forward(
871                &xs,
872                attention_mask
873                    .as_ref()
874                    .map(|m| m.to_device(xs.device()).unwrap())
875                    .as_ref(),
876                seqlen_offsets,
877                &mut cache[i],
878                metadata
879                    .as_ref()
880                    .map(|(kv_cache, metadata)| (kv_cache[i].clone(), *metadata)),
881                flash_params,
882            )?;
883        }
884        let xs = xs.to_device(&self.device)?;
885        let xs = xs.apply(&self.norm)?;
886        extract_logits(&self.lm_head.forward_autocast(&xs)?, context_lens)
887    }
888}

890impl IsqModel for Glm4MoeLite {
891    fn get_layers(
892        &mut self,
893    ) -> (
894        Vec<(&mut Arc<dyn QuantMethod>, Option<usize>)>,
895        &dyn DeviceMapper,
896    ) {
897        let mut tensors = Vec::new();
898        tensors.push((&mut self.lm_head, None));
899        for (i, layer) in self.layers.iter_mut().enumerate() {
900            match &mut layer.attn.q {
901                QProj::Lora { a, norm: _, b } => {
902                    tensors.push((a, Some(i)));
903                    tensors.push((b, Some(i)));
904                }
905            }
906            tensors.push((&mut layer.attn.kv_a_proj_with_mqa, Some(i)));
907            tensors.push((&mut layer.attn.kv_b_proj, Some(i)));
908            tensors.push((&mut layer.attn.o_proj, Some(i)));
909            match &mut layer.moe_or_mlp {
910                MoeOrMlp::Mlp(mlp) => {
911                    tensors.push((&mut mlp.gate, Some(i)));
912                    tensors.push((&mut mlp.up, Some(i)));
913                    tensors.push((&mut mlp.down, Some(i)));
914                }
915                MoeOrMlp::Moe(moe) => {
916                    for layer in moe.get_isq_layers() {
917                        tensors.push((layer, Some(i)));
918                    }
919                }
920            }
921        }
922        (tensors, &*self.mapper)
923    }
