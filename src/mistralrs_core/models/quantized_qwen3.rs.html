26struct Mlp {
27    feed_forward_w1: Arc<dyn QuantMethod>,
28    feed_forward_w2: Arc<dyn QuantMethod>,
29    feed_forward_w3: Arc<dyn QuantMethod>,
30}
31
32impl Mlp {
33    fn forward(&self, xs: &Tensor) -> Result<Tensor> {
34        let w1 = MatMul.qmethod_matmul(xs, &*self.feed_forward_w1)?;
35        let w3 = MatMul.qmethod_matmul(xs, &*self.feed_forward_w3)?;
36        let y = &(candle_nn::ops::silu(&w1)? * w3)?;
37        MatMul.qmethod_matmul(y, &*self.feed_forward_w2)
38    }
39}

41struct LayerWeights {
42    attention_wq: Arc<dyn QuantMethod>,
43    attention_wk: Arc<dyn QuantMethod>,
44    attention_wv: Arc<dyn QuantMethod>,
45    attention_wo: Arc<dyn QuantMethod>,
46    attention_norm: QRmsNorm,
47    q_norm: QRmsNorm,
48    k_norm: QRmsNorm,
49    mlp: Mlp,
50    ffn_norm: QRmsNorm,
51    n_head: usize,
52    n_kv_head: usize,
53    head_dim: usize,
54    rotary: Arc<RotaryEmbedding>,
55    paged_attn: Option<PagedAttention>,
56    sdpa_params: SdpaParams,
57    dtype: DType,
58}

60impl LayerWeights {
61    fn forward_attn(
62        &self,
63        x: &Tensor,
64        mask: Option<&Tensor>,
65        start_offsets: &[usize],
66        kv_cache: &mut KvCache,
67        metadata: Option<((Tensor, Tensor), &PagedAttentionInputMetadata)>,
68    ) -> Result<Tensor> {
69        let (b_sz, seq_len, _) = x.dims3()?;
70
71        let q = MatMul.qmethod_matmul(x, &*self.attention_wq)?;
72        let k = MatMul.qmethod_matmul(x, &*self.attention_wk)?;
73        let v = MatMul.qmethod_matmul(x, &*self.attention_wv)?;
74
75        let (q, k, v) = if seq_len != 1 {
76            let q = q
77                .reshape((b_sz, seq_len, self.n_head, self.head_dim))?
78                .transpose(1, 2)?;
79            let k = k
80                .reshape((b_sz, seq_len, self.n_kv_head, self.head_dim))?
81                .transpose(1, 2)?;
82            let v = v
83                .reshape((b_sz, seq_len, self.n_kv_head, self.head_dim))?
84                .transpose(1, 2)?;
85            (q, k, v)
86        } else {
87            let q = q.reshape((b_sz, self.n_head, seq_len, self.head_dim))?;
88            let k = k.reshape((b_sz, self.n_kv_head, seq_len, self.head_dim))?;
89            let v = v.reshape((b_sz, self.n_kv_head, seq_len, self.head_dim))?;
90            (q, k, v)
91        };
92
93        // Per-head RMSNorm in Qwen3
94        let q_flat = q.flatten(0, 2)?;
95        let k_flat = k.flatten(0, 2)?;
96        let q_flat = self.q_norm.forward(&q_flat)?;
97        let k_flat = self.k_norm.forward(&k_flat)?;
98        let q = q_flat.reshape((b_sz, self.n_head, seq_len, self.head_dim))?;
99        let k = k_flat.reshape((b_sz, self.n_kv_head, seq_len, self.head_dim))?;
100
101        let (q, k) = self.rotary.forward(&q, &k, start_offsets)?;
102
103        let (q, k, v) = (
104            q.to_dtype(self.dtype)?,
105            k.to_dtype(self.dtype)?,
106            v.to_dtype(self.dtype)?,
107        );
108
109        let y = match &self.paged_attn {
110            Some(paged_attn) => {
111                let ((key_cache, value_cache), input_metadata) = metadata.unwrap();
112                paged_attn.forward(
113                    &q,
114                    &k,
115                    &v,
116                    mask,
117                    Some(key_cache),
118                    Some(value_cache),
119                    input_metadata,
120                    &self.sdpa_params,
121                    None,
122                )?
123            }
124            None => {
125                let (k, v) = kv_cache.append(&k, &v)?;
126
127                Sdpa.run_attention(&q, &k, &v, mask, None, &self.sdpa_params)?
128            }
129        };
130
131        let y = if mask.is_some() {
132            y.transpose(1, 2)?.reshape((b_sz, seq_len, ()))?
133        } else {
134            y.reshape((b_sz, seq_len, ()))?
135        };
136
137        let y = MatMul.qmethod_matmul(&y.to_dtype(x.dtype())?, &*self.attention_wo)?;
138        Ok(y)
139    }
140}

154// qwen3 `llm` fields:
155// https://github.com/ggerganov/ggml/blob/master/docs/gguf.md#llm
156// NOTE: Types here do not match spec
157pub(crate) struct PropsGGUF {
158    pub head_count: usize,
159    pub head_count_kv: usize,
160    pub block_count: usize,
161    pub embedding_length: usize,
162    pub rms_norm_eps: f32,
163    pub max_seq_len: usize,
164    pub rope_freq_base: f32,
165    pub key_length: usize,
166    pub value_length: usize,
167}

169fn verify_qwen3_arch(
170    metadata: &HashMap<String, candle_core::quantized::gguf_file::Value>,
171) -> Result<String> {
172    use crate::utils::gguf_metadata::TryValueInto;
173    let actual_arch: String = metadata
174        .get("general.architecture")
175        .cloned()
176        .try_value_into()?;
177
178    if actual_arch != "qwen3" {
179        candle_core::bail!("Expected `qwen3` architecture, got `{actual_arch}`.");
180    }
181    Ok(actual_arch)
182}
183
184impl TryFrom<ContentMetadata<'_>> for PropsGGUF {
185    type Error = anyhow::Error;
186
187    fn try_from(c: ContentMetadata) -> std::result::Result<Self, Self::Error> {
188        let _ = verify_qwen3_arch(c.metadata)?;
189
190        let required = [
191            "attention.head_count",
192            "attention.head_count_kv",
193            "block_count",
194            "embedding_length",
195            "attention.layer_norm_rms_epsilon",
196        ];
197        c.has_required_keys(&required)?;
198
199        let embed_len = c.get_value::<u32>("embedding_length")? as usize;
200        let head_count = c.get_value::<u32>("attention.head_count")? as usize;
201
202        // NOTE: Values are not aligned with GGUFv3 types
203        // TODO: Normalize value types to spec
204        let props = Self {
205            head_count,
206            head_count_kv: c.get_value::<u32>("attention.head_count_kv")? as usize,
207            block_count: c.get_value::<u32>("block_count")? as usize,
208            embedding_length: embed_len,
209            rms_norm_eps: c.get_value("attention.layer_norm_rms_epsilon")?,
210            max_seq_len: c
211                .get_value::<u64>("context_length")
212                .ok()
213                .unwrap_or(DEFAULT_MAX_SEQ_LEN as u64) as usize,
214            rope_freq_base: c.get_value("rope.freq_base").ok().unwrap_or(10_000_f32),
215            key_length: c
216                .get_value::<u32>("attention.key_length")
217                .ok()
218                .map(|x| x as usize)
219                .unwrap_or(embed_len / head_count),
220            value_length: c
221                .get_value::<u32>("attention.value_length")
222                .ok()
223                .map(|x| x as usize)
224                .unwrap_or(embed_len / head_count),
225        };
226
227        Ok(props)
228    }
229}

231impl ModelConfig::FromGGUF for ModelWeights {
232    fn from_gguf<R: std::io::Seek + std::io::Read>(
233        mut ct: Content<'_, R>,
234        device: &Device,
235        mapper: Box<dyn DeviceMapper + Send + Sync>,
236        attention_mechanism: AttentionImplementation,
237        dtype: DType,
238    ) -> Result<Self> {
239        // Parameter extraction from metadata.
240        let meta = ct.get_metadata();
241        let actual_arch = verify_qwen3_arch(meta)?;
242
243        let metadata = ContentMetadata {
244            path_prefix: &actual_arch,
245            metadata: meta,
246        };
247        let PropsGGUF {
248            head_count,
249            head_count_kv,
250            block_count,
251            embedding_length,
252            rms_norm_eps,
253            max_seq_len,
254            rope_freq_base,
255            key_length,
256            value_length,
257        } = PropsGGUF::try_from(metadata).or_else(|err| candle_core::bail!("{err}"))?;
258
259        let qtok_embeddings = ct.tensor("token_embd.weight", device)?;
260        let tok_embeddings = qtok_embeddings.dequantize(device)?;
261        let norm = QRmsNorm::new(ct.tensor("output_norm.weight", device)?, rms_norm_eps)?;
262        let output = if !ct.has_tensor("output.weight") {
263            ct.tensor("token_embd.weight", device)?
264        } else {
265            ct.tensor("output.weight", device)?
266        };
267        let mut layers = Vec::with_capacity(block_count);
268
269        let head_dim = key_length;
270        if key_length != value_length {
271            candle_core::bail!(
272                "Expected key_length == value_length, got {key_length} != {value_length}"
273            );
274        }
275
276        let mut ropes = HashMap::new();
277        for layer_idx in 0..block_count {
278            let device = mapper.device_for(layer_idx, false).unwrap_or(device);
279            ropes.insert(
280                device.location(),
281                Arc::new(RotaryEmbedding::new(
282                    rope_freq_base,
283                    head_dim,
284                    max_seq_len,
285                    device,
286                    true,
287                    DType::F32,
288                )?),
289            );
290        }
291
292        for layer_idx in NiceProgressBar::<_, 'b'>(
293            0..block_count,
294            "Loading repeating layers",
295            &MultiProgress::new(),
296        ) {
297            let prefix = format!("blk.{layer_idx}");
298            let device = mapper.device_for(layer_idx, false).unwrap_or(device);
299            let rotary = ropes
300                .get(&device.location())
301                .expect("No RoPE for device location!")
302                .clone();
303
304            let attention_wq = ct.tensor(&format!("{prefix}.attn_q.weight"), device)?;
305            let attention_wk = ct.tensor(&format!("{prefix}.attn_k.weight"), device)?;
306            let attention_wv = ct.tensor(&format!("{prefix}.attn_v.weight"), device)?;
307            let attention_wo = ct.tensor(&format!("{prefix}.attn_output.weight"), device)?;
308
309            let feed_forward_w1 = ct.tensor(&format!("{prefix}.ffn_gate.weight"), device)?;
310            let feed_forward_w2 = ct.tensor(&format!("{prefix}.ffn_down.weight"), device)?;
311            let feed_forward_w3 = ct.tensor(&format!("{prefix}.ffn_up.weight"), device)?;
312            let mlp = Mlp {
313                feed_forward_w1: Arc::new(GgufMatMul::new(QuantMethodConfig::Gguf {
314                    q_weight: Arc::new(feed_forward_w1),
315                    b: None,
316                })?),
317                feed_forward_w2: Arc::new(GgufMatMul::new(QuantMethodConfig::Gguf {
318                    q_weight: Arc::new(feed_forward_w2),
319                    b: None,
320                })?),
321                feed_forward_w3: Arc::new(GgufMatMul::new(QuantMethodConfig::Gguf {
322                    q_weight: Arc::new(feed_forward_w3),
323                    b: None,
324                })?),
325            };
326
327            // Qwen3 always has q_norm and k_norm
328            let q_norm = QRmsNorm::new(
329                ct.tensor(&format!("{prefix}.attn_q_norm.weight"), device)?,
330                rms_norm_eps,
331            )?;
332            let k_norm = QRmsNorm::new(
333                ct.tensor(&format!("{prefix}.attn_k_norm.weight"), device)?,
334                rms_norm_eps,
335            )?;
336
337            let attention_norm = ct.tensor(&format!("{prefix}.attn_norm.weight"), device)?;
338            let ffn_norm = ct.tensor(&format!("{prefix}.ffn_norm.weight"), device)?;
339            let paged_attn = match &attention_mechanism {
340                AttentionImplementation::Eager => None,
341                AttentionImplementation::PagedAttention => {
342                    Some(PagedAttention::new(head_dim, device, None)?)
343                }
344            };
345            layers.push(LayerWeights {
346                attention_wq: Arc::new(GgufMatMul::new(QuantMethodConfig::Gguf {
347                    q_weight: Arc::new(attention_wq),
348                    b: None,
349                })?),
350                attention_wk: Arc::new(GgufMatMul::new(QuantMethodConfig::Gguf {
351                    q_weight: Arc::new(attention_wk),
352                    b: None,
353                })?),
354                attention_wv: Arc::new(GgufMatMul::new(QuantMethodConfig::Gguf {
355                    q_weight: Arc::new(attention_wv),
356                    b: None,
357                })?),
358                attention_wo: Arc::new(GgufMatMul::new(QuantMethodConfig::Gguf {
359                    q_weight: Arc::new(attention_wo),
360                    b: None,
361                })?),
362                attention_norm: QRmsNorm::new(attention_norm, rms_norm_eps)?,
363                q_norm,
364                k_norm,
365                mlp,
366                ffn_norm: QRmsNorm::new(ffn_norm, rms_norm_eps)?,
367                n_head: head_count,
368                n_kv_head: head_count_kv,
369                head_dim,
370                rotary: rotary.clone(),
371                paged_attn,
372                sdpa_params: SdpaParams {
373                    n_kv_groups: head_count / head_count_kv,
374                    softcap: None,
375                    softmax_scale: 1.0 / (head_dim as f32).sqrt(),
376                    sliding_window: None,
377                },
378                dtype,
379            })
380        }
381        Ok(Self {
382            tok_embeddings: Embedding::new(tok_embeddings, embedding_length),
383            layers,
384            norm,
385            output: Arc::new(GgufMatMul::new(QuantMethodConfig::Gguf {
386                q_weight: Arc::new(output),
387                b: None,
388            })?),
389            device: device.clone(),
390            cache: EitherCache::Normal(NormalCache::new(block_count, max_seq_len)),
391            max_seq_len,
392            mapper: Some(mapper),
393            dtype,
394        })
395    }
396}

398impl ModelWeights {
399    pub fn forward(
400        &self,
401        x: &Tensor,
402        start_offsets: &[usize],
403        context_lens: Vec<(usize, usize)>,
404        metadata: Option<(Vec<(Tensor, Tensor)>, &PagedAttentionInputMetadata)>,
405    ) -> Result<Tensor> {
406        let mut layer_in = self.tok_embeddings.forward(x)?;
407        let cache = &mut self.cache.normal().0;
408        let mask = CausalMasker.make_causal_mask_matrix(
409            x,
410            metadata
411                .as_ref()
412                .map(|(_, _)| &start_offsets as &dyn PastKvLenCache)
413                .unwrap_or(cache as &dyn PastKvLenCache),
414            self.dtype,
415            self.layers[0].n_head,
416        )?;
417        let mask = mask.filter(|_| {
418            metadata
419                .as_ref()
420                .map(|(_, meta)| meta.is_first_prompt_chunk)
421                .unwrap_or(true)
422        });
423        for (i, layer) in self.layers.iter().enumerate() {
424            if let Some(ref mapper) = self.mapper {
425                layer_in = mapper.map(layer_in, i)?;
426            }
427            let x = layer_in;
428            let residual = &x;
429            let x = layer.attention_norm.forward(&x)?;
430            let attn = layer.forward_attn(
431                &x,
432                mask.as_ref()
433                    .map(|m| m.to_device(x.device()).unwrap())
434                    .as_ref(),
435                start_offsets,
436                &mut cache[i],
437                metadata
438                    .as_ref()
439                    .map(|(kv_cache, metadata)| (kv_cache[i].clone(), *metadata)),
440            )?;
441            let x = (attn + residual)?;
442
443            // MLP
444            let residual = &x;
445            let x = layer.ffn_norm.forward(&x)?;
446            let x = layer.mlp.forward(&x)?;
447            let x = (x + residual)?;
448            layer_in = x;
449        }
450        let x = self.norm.forward(&layer_in)?;
451        extract_logits(
452            &MatMul.qmethod_matmul(&x.contiguous()?, &*self.output)?,
453            context_lens,
454        )
455    }
