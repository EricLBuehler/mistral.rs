53struct Attention {
54    q_proj: Arc<dyn QuantMethod>,
55    k_proj: Arc<dyn QuantMethod>,
56    v_proj: Arc<dyn QuantMethod>,
57    o_proj: Arc<dyn QuantMethod>,
58    num_heads: usize,
59    num_kv_heads: usize,
60    head_dim: usize,
61    rotary_emb: Arc<RotaryEmbedding>,
62    paged_attn: Option<PagedAttention>,
63    sdpa_params: SdpaParams,
64}
65
66impl Attention {
67    fn new(
68        rotary_emb: Arc<RotaryEmbedding>,
69        cfg: &Config,
70        vb: ShardedVarBuilder,
71        paged_attn: Option<PagedAttention>,
72        comm: &Arc<mistralrs_quant::Comm>,
73    ) -> Result<Self> {
74        let hidden_sz = cfg.hidden_size;
75        let num_heads = cfg.num_attention_heads;
76        let num_kv_heads = cfg.num_key_value_heads;
77        let head_dim = hidden_sz / num_heads;
78        let q_proj = ColumnParallelLayer::new(
79            hidden_sz,
80            num_heads * head_dim,
81            &cfg.quantization_config,
82            false,
83            comm,
84            vb.pp("q_proj"),
85        )?;
86        let kv_shard = mistralrs_quant::compute_kv_shard(
87            cfg.num_key_value_heads,
88            cfg.hidden_size / cfg.num_attention_heads,
89            comm,
90        );
91        let k_proj = ColumnParallelLayer::new_with_shard(
92            hidden_sz,
93            num_kv_heads * head_dim,
94            &cfg.quantization_config,
95            false,
96            comm,
97            kv_shard,
98            vb.pp("k_proj"),
99        )?;
100        let v_proj = ColumnParallelLayer::new_with_shard(
101            hidden_sz,
102            num_kv_heads * head_dim,
103            &cfg.quantization_config,
104            false,
105            comm,
106            kv_shard,
107            vb.pp("v_proj"),
108        )?;
109        let o_proj = RowParallelLayer::new(
110            num_heads * head_dim,
111            hidden_sz,
112            &cfg.quantization_config,
113            false,
114            comm,
115            vb.pp("o_proj"),
116        )?;
117        Ok(Self {
118            q_proj,
119            k_proj,
120            v_proj,
121            o_proj,
122            num_heads: num_heads / comm.world_size(),
123            num_kv_heads: (num_kv_heads / comm.world_size()).max(1),
124            head_dim,
125            rotary_emb,
126            paged_attn,
127            sdpa_params: SdpaParams {
128                n_kv_groups: mistralrs_quant::compute_n_kv_groups(
129                    cfg.num_key_value_heads,
130                    cfg.num_attention_heads,
131                    comm,
132                ),
133                softcap: None,
134                softmax_scale: 1.0 / (head_dim as f32).sqrt(),
135                sliding_window: cfg.sliding_window,
136            },
137        })
138    }
139
140    #[allow(clippy::too_many_arguments)]
141    fn forward(
142        &self,
143        xs: &Tensor,
144        attention_mask: Option<&Tensor>,
145        seqlen_offsets: &[usize],
146        kv_cache: &mut KvCache,
147        metadata: Option<((Tensor, Tensor), &PagedAttentionInputMetadata)>,
148        flash_params: &FlashParams,
149    ) -> Result<Tensor> {
150        let (b_sz, q_len, _) = xs.dims3()?;
151
152        let original_dtype = xs.dtype();
153        let mut xs = xs.clone();
154        if let Some(t) = self.q_proj.quantized_act_type() {
155            xs = xs.to_dtype(t)?;
156        }
157        let mut q = MatMul.qmethod_matmul(&xs, &*self.q_proj)?;
158        let mut k = MatMul.qmethod_matmul(&xs, &*self.k_proj)?;
159        let mut v = MatMul.qmethod_matmul(&xs, &*self.v_proj)?;
160        if self.q_proj.quantized_act_type().is_some() {
161            q = q.to_dtype(original_dtype)?;
162            k = k.to_dtype(original_dtype)?;
163            v = v.to_dtype(original_dtype)?;
164        }
165
166        let (q, k, v) = if q_len != 1 {
167            let q = q
168                .reshape((b_sz, q_len, self.num_heads, self.head_dim))?
169                .transpose(1, 2)?;
170            let k = k
171                .reshape((b_sz, q_len, self.num_kv_heads, self.head_dim))?
172                .transpose(1, 2)?;
173            let v = v
174                .reshape((b_sz, q_len, self.num_kv_heads, self.head_dim))?
175                .transpose(1, 2)?;
176            (q, k, v)
177        } else {
178            let q = q.reshape((b_sz, self.num_heads, q_len, self.head_dim))?;
179            let k = k.reshape((b_sz, self.num_kv_heads, q_len, self.head_dim))?;
180            let v = v.reshape((b_sz, self.num_kv_heads, q_len, self.head_dim))?;
181            (q, k, v)
182        };
183
184        let (q, k) = self.rotary_emb.forward(&q, &k, seqlen_offsets)?;
185
186        let mut attn_output = match &self.paged_attn {
187            Some(paged_attn) => match metadata {
188                Some(((key_cache, value_cache), input_metadata)) => paged_attn.forward(
189                    &q,
190                    &k,
191                    &v,
192                    attention_mask,
193                    Some(key_cache),
194                    Some(value_cache),
195                    input_metadata,
196                    &self.sdpa_params,
197                    Some(flash_params),
198                )?,
199                None => {
200                    // If we don't have metadata, we are most likely generating an imatrix so we don't want to populate that.
201                    // Generating the dummy metadata with the assumption that we are not generating text (only processing prompts).
202                    let input_metadata = PagedAttentionInputMetadata::dummy(q.device())?;
203                    // Sanity check.
204                    assert!(attention_mask.is_some());
205                    paged_attn.forward(
206                        &q,
207                        &k,
208                        &v,
209                        attention_mask,
210                        None,
211                        None,
212                        &input_metadata,
213                        &self.sdpa_params,
214                        Some(flash_params),
215                    )?
216                }
217            },
218            None => {
219                let (k, v) = kv_cache.append(&k, &v)?;
220
221                Sdpa.run_attention(
222                    &q,
223                    &k,
224                    &v,
225                    attention_mask,
226                    Some(flash_params),
227                    &self.sdpa_params,
228                )?
229            }
230        };
231
232        if let Some(t) = self.q_proj.quantized_act_type() {
233            attn_output = attn_output.to_dtype(t)?;
234        }
235        attn_output = if attention_mask.is_some() {
236            attn_output.transpose(1, 2)?.reshape((b_sz, q_len, ()))?
237        } else {
238            attn_output.reshape((b_sz, q_len, ()))?
239        };
240        let mut res = MatMul.qmethod_matmul(&attn_output, &*self.o_proj)?;
241        if self.q_proj.quantized_act_type().is_some() {
242            res = res.to_dtype(original_dtype)?;
243        }
244        Ok(res)
245    }
246}

249struct BlockSparseTop2MLP {
250    w1: Arc<dyn QuantMethod>,
251    w2: Arc<dyn QuantMethod>,
252    w3: Arc<dyn QuantMethod>,
253    act_fn: Activation,
254}
255
256impl BlockSparseTop2MLP {
257    fn new(cfg: &Config, vb: ShardedVarBuilder, comm: &Arc<mistralrs_quant::Comm>) -> Result<Self> {
258        let hidden_sz = cfg.hidden_size;
259        let intermediate_sz = cfg.intermediate_size;
260        let w1 = ColumnParallelLayer::new(
261            hidden_sz,
262            intermediate_sz,
263            &cfg.quantization_config,
264            false,
265            comm,
266            vb.pp("w1"),
267        )?;
268        let w2 = RowParallelLayer::new(
269            intermediate_sz,
270            hidden_sz,
271            &cfg.quantization_config,
272            false,
273            comm,
274            vb.pp("w2"),
275        )?;
276        let w3 = ColumnParallelLayer::new(
277            hidden_sz,
278            intermediate_sz,
279            &cfg.quantization_config,
280            false,
281            comm,
282            vb.pp("w3"),
283        )?;
284        Ok(Self {
285            w1,
286            w2,
287            w3,
288            act_fn: cfg.hidden_act,
289        })
290    }
291}
292
293impl Module for BlockSparseTop2MLP {
294    fn forward(&self, xs: &Tensor) -> Result<Tensor> {
295        let original_dtype = xs.dtype();
296        let mut xs = xs.clone();
297        if let Some(t) = self.w1.quantized_act_type() {
298            xs = xs.to_dtype(t)?;
299        }
300        let w1_out = MatMul.qmethod_matmul(&xs, &*self.w1)?;
301        let w3_out = MatMul.qmethod_matmul(&xs, &*self.w3)?;
302        let activated = crate::ops::mul_and_act(&w1_out, &w3_out, self.act_fn)?;
303        let mut res = MatMul.qmethod_matmul(&activated, &*self.w2)?;
304        if self.w1.quantized_act_type().is_some() {
305            res = res.to_dtype(original_dtype)?;
306        }
307        Ok(res)
308    }

312struct SparseMoeBlock {
313    gate: Arc<dyn QuantMethod>,
314    experts: Vec<BlockSparseTop2MLP>,
315    num_experts_per_tok: usize,
316}
317
318impl SparseMoeBlock {
319    fn new(cfg: &Config, vb: ShardedVarBuilder, comm: &Arc<mistralrs_quant::Comm>) -> Result<Self> {
320        let gate = mistralrs_quant::linear_no_bias(
321            cfg.hidden_size,
322            cfg.num_local_experts,
323            &cfg.quantization_config,
324            vb.pp("gate"),
325        )?;
326        let mut experts = Vec::with_capacity(cfg.num_local_experts);
327        let vb = vb.pp("experts");
328        for idx in 0..cfg.num_local_experts {
329            let expert = BlockSparseTop2MLP::new(cfg, vb.pp(idx), comm)?;
330            experts.push(expert)
331        }
332        Ok(SparseMoeBlock {
333            gate,
334            experts,
335            num_experts_per_tok: cfg.num_experts_per_tok,
336        })
337    }
338}
339
340impl Module for SparseMoeBlock {
341    fn forward(&self, xs: &Tensor) -> Result<Tensor> {
342        let (b_size, seq_len, hidden_dim) = xs.dims3()?;
343        let xs = xs.reshape(((), hidden_dim))?;
344
345        let original_dtype = xs.dtype();
346        let mut xs = xs.clone();
347        if let Some(t) = self.gate.quantized_act_type() {
348            xs = xs.to_dtype(t)?;
349        }
350        let mut router_logits = MatMul.qmethod_matmul(&xs, &*self.gate)?;
351        if self.gate.quantized_act_type().is_some() {
352            router_logits = router_logits.to_dtype(original_dtype)?;
353        }
354
355        let routing_weights = candle_nn::ops::softmax_last_dim(&router_logits)?;
356
357        // In order to extract topk, we extract the data from the tensor and manipulate it
358        // directly. Maybe we will want to use some custom ops instead at some point.
359        let routing_weights = routing_weights.to_dtype(DType::F32)?.to_vec2::<f32>()?;
360
361        // routing_weights, selected_experts = torch.topk(routing_weights, self.top_k, dim=-1)
362        // top_x contains the row indexes to evaluate for each expert.
363        let mut top_x = vec![vec![]; self.experts.len()];
364        let mut selected_rws = vec![vec![]; self.experts.len()];
365        for (row_idx, rw) in routing_weights.iter().enumerate() {
366            let mut dst = (0..rw.len() as u32).collect::<Vec<u32>>();
367            dst.sort_by(|&i, &j| rw[j as usize].total_cmp(&rw[i as usize]));
368            let mut sum_routing_weights = 0f32;
369            for &expert_idx in dst.iter().take(self.num_experts_per_tok) {
370                let expert_idx = expert_idx as usize;
371                let routing_weight = rw[expert_idx];
372                sum_routing_weights += routing_weight;
373                top_x[expert_idx].push(row_idx as u32);
374            }
375            for &expert_idx in dst.iter().take(self.num_experts_per_tok) {
376                let expert_idx = expert_idx as usize;
377                let routing_weight = rw[expert_idx];
378                selected_rws[expert_idx].push(routing_weight / sum_routing_weights)
379            }
380        }
381
382        // routing_weights /= routing_weights.sum(dim=-1, keepdim=True)
383        // expert_mask = torch.nn.functional.one_hot(selected_experts, num_classes=self.num_experts).permute(2, 1, 0)
384
385        let mut ys = xs.zeros_like()?;
386        for (expert_idx, expert_layer) in self.experts.iter().enumerate() {
387            let top_x = &top_x[expert_idx];
388            if top_x.is_empty() {
389                continue;
390            }
391            let top_x = Tensor::new(top_x.as_slice(), xs.device())?;
392            let selected_rws =
393                Tensor::new(selected_rws[expert_idx].as_slice(), xs.device())?.reshape(((), 1))?;
394            // Index the correct hidden states and compute the expert hidden state for
395            // the current expert. We need to make sure to multiply the output hidden
396            // states by `routing_weights` on the corresponding tokens (top-1 and top-2)
397            let current_state = xs.index_select(&top_x, 0)?.reshape(((), hidden_dim))?;
398            // current_hidden_states = expert_layer(current_state, routing_weights[top_x_list, idx_list, None])
399            let current_hidden_states = expert_layer.forward(&current_state)?;
400            let current_hidden_states = current_hidden_states.broadcast_mul(&selected_rws)?;
401            ys = ys.index_add(&top_x, &current_hidden_states, 0)?;
402        }
403
404        let ys = ys.reshape((b_size, seq_len, hidden_dim))?;
405        Ok(ys)
406    }
407}

409struct DecoderLayer {
410    self_attn: Attention,
411    block_sparse_moe: SparseMoeBlock,
412    input_layernorm: RmsNorm,
413    post_attention_layernorm: RmsNorm,
414}
415
416impl DecoderLayer {
417    #[allow(clippy::too_many_arguments)]
418    fn new(
419        rotary_emb: Arc<RotaryEmbedding>,
420        cfg: &Config,
421        vb: ShardedVarBuilder,
422        mapper: &dyn DeviceMapper,
423        layer_idx: usize,
424        loading_isq: bool,
425        paged_attn: Option<PagedAttention>,
426        comm: &Arc<mistralrs_quant::Comm>,
427    ) -> Result<Self> {
428        let self_attn = Attention::new(
429            rotary_emb,
430            cfg,
431            mapper.set_device(layer_idx, vb.pp("self_attn"), loading_isq),
432            paged_attn,
433            comm,
434        )?;
435        let block_sparse_moe = SparseMoeBlock::new(
436            cfg,
437            mapper.set_device(layer_idx, vb.pp("block_sparse_moe"), loading_isq),
438            comm,
439        )?;
440        let input_layernorm = RmsNorm::new(
441            cfg.hidden_size,
442            cfg.rms_norm_eps,
443            mapper.set_device(layer_idx, vb.pp("input_layernorm"), false),
444        )?;
445        let post_attention_layernorm = RmsNorm::new(
446            cfg.hidden_size,
447            cfg.rms_norm_eps,
448            mapper.set_device(layer_idx, vb.pp("post_attention_layernorm"), false),
449        )?;
450        Ok(Self {
451            self_attn,
452            block_sparse_moe,
453            input_layernorm,
454            post_attention_layernorm,
455        })
456    }
457
458    #[allow(clippy::too_many_arguments)]
459    fn forward(
460        &self,
461        xs: &Tensor,
462        attention_mask: Option<&Tensor>,
463        seqlen_offsets: &[usize],
464        kv_cache: &mut KvCache,
465        metadata: Option<((Tensor, Tensor), &PagedAttentionInputMetadata)>,
466        flash_params: &FlashParams,
467    ) -> Result<Tensor> {
468        let residual = xs;
469        let xs = self.input_layernorm.forward(xs)?;
470        let xs = self.self_attn.forward(
471            &xs,
472            attention_mask,
473            seqlen_offsets,
474            kv_cache,
475            metadata,
476            flash_params,
477        )?;
478        let xs = (xs + residual)?;
479        let residual = &xs;
480        let xs = xs
481            .apply(&self.post_attention_layernorm)?
482            .apply(&self.block_sparse_moe)?
483            .to_dtype(residual.dtype())?;
484        residual + xs
485    }
486}

488pub struct Model {
489    embed_tokens: candle_nn::Embedding,
490    layers: Vec<DecoderLayer>,
491    norm: RmsNorm,
492    lm_head: Arc<dyn QuantMethod>,
493    sliding_window: Option<usize>,
494    device: Device,
495    cache: EitherCache,
496    max_seq_len: usize,
497    mapper: Box<dyn DeviceMapper + Send + Sync>,
498    cfg: ModelConfigMetadata,
499}
500
501impl Model {
502    pub fn new(
503        cfg: &Config,
504        vb: ShardedVarBuilder,
505        is_gptx: bool,
506        normal_loading_metadata: NormalLoadingMetadata,
507        attention_mechanism: AttentionImplementation,
508    ) -> Result<Self> {
509        if let Some(ref quant_cfg) = &cfg.quantization_config {
510            tracing::info!(
511                "Using {} quantization: {}.",
512                quant_cfg.name(),
513                quant_cfg.get_bits_name(&vb)
514            );
515        }
516        let mapper = normal_loading_metadata.mapper;
517        let vb_m = vb.pp("model");
518
519        let embed_tokens = layers::embedding(
520            cfg.vocab_size,
521            cfg.hidden_size,
522            mapper.set_nm_device(vb_m.pp("embed_tokens"), false),
523            &cfg.quantization_config,
524        )?;
525        let head_dim = cfg.hidden_size / cfg.num_attention_heads;
526        let mut ropes = HashMap::new();
527        for layer_idx in 0..cfg.num_hidden_layers {
528            let device = mapper
529                .device_for(layer_idx, false)
530                .unwrap_or(&normal_loading_metadata.real_device);
531            ropes.insert(
532                device.location(),
533                Arc::new(RotaryEmbedding::new(
534                    cfg.rope_theta as f32,
535                    head_dim,
536                    cfg.max_position_embeddings,
537                    device,
538                    is_gptx,
539                    vb_m.dtype(),
540                )?),
541            );
542        }
543        let vb_l = vb_m.pp("layers");
544        let layers: Vec<DecoderLayer> = NiceProgressBar::<_, 'b'>(
545            0..cfg.num_hidden_layers,
546            "Loading repeating layers",
547            &normal_loading_metadata.multi_progress,
548        )
549        .par_iter_if_isq(|layer_idx| {
550            let device = mapper
551                .device_for(layer_idx, false)
552                .unwrap_or(&normal_loading_metadata.real_device);
553            let rotary_emb = ropes
554                .get(&device.location())
555                .expect("No RoPE for device location!")
556                .clone();
557            let paged_attn = match &attention_mechanism {
558                AttentionImplementation::Eager => None,
559                AttentionImplementation::PagedAttention => {
560                    Some(PagedAttention::new(head_dim, device, None)?)
561                }
562            };
563            let comm = mapper.get_comm_for(layer_idx)?;
564            DecoderLayer::new(
565                rotary_emb.clone(),
566                cfg,
567                vb_l.pp(layer_idx),
568                &*mapper,
569                layer_idx,
570                normal_loading_metadata.loading_isq,
571                paged_attn,
572                &comm,
573            )
574        })?;
575        let norm = RmsNorm::new(
576            cfg.hidden_size,
577            cfg.rms_norm_eps,
578            mapper.set_nm_device(vb_m.pp("norm"), false),
579        )?;
580        let lm_head = if !cfg.tie_word_embeddings {
581            ReplicatedLayer::new(
582                cfg.hidden_size,
583                cfg.vocab_size,
584                &cfg.quantization_config,
585                false,
586                mapper.set_nm_device(vb.pp("lm_head"), normal_loading_metadata.loading_isq),
587            )?
588        } else {
589            ReplicatedLayer::from_linear(candle_nn::Linear::new(
590                mapper.cast_nm_device(
591                    embed_tokens.embeddings(),
592                    normal_loading_metadata.loading_isq,
593                )?,
594                None,
595            ))?
596        };
597        Ok(Self {
598            embed_tokens,
599            layers,
600            norm,
601            lm_head,
602            sliding_window: cfg.sliding_window,
603            device: normal_loading_metadata.real_device,
604            cache: EitherCache::Normal(NormalCache::new_sliding(
605                cfg.num_hidden_layers,
606                cfg.max_position_embeddings,
607                cfg.sliding_window,
608            )),
609            max_seq_len: cfg.max_position_embeddings,
610            cfg: ModelConfigMetadata {
611                max_seq_len: cfg.max_position_embeddings,
612                num_layers: cfg.num_hidden_layers,
613                hidden_size: cfg.hidden_size,
614                num_attn_heads: cfg.num_attention_heads / mapper.get_comm_for(0)?.world_size(),
615                num_kv_heads: (cfg.num_key_value_heads / mapper.get_comm_for(0)?.world_size())
616                    .max(1),
617                sliding_window: cfg.sliding_window,
618                k_head_dim: cfg.hidden_size / cfg.num_attention_heads,
619                v_head_dim: cfg.hidden_size / cfg.num_attention_heads,
620            },
621            mapper,
622        })
623    }
624
625    pub fn forward(
626        &self,
627        input_ids: &Tensor,
628        seqlen_offsets: &[usize],
629        context_lens: Vec<(usize, usize)>,
630        metadata: Option<(Vec<(Tensor, Tensor)>, &PagedAttentionInputMetadata)>,
631        flash_params: &FlashParams,
632    ) -> Result<Tensor> {
633        let mut xs = self.embed_tokens.forward(input_ids)?;
634        let cache = &mut self.cache.normal().0;
635        let attention_mask = CausalMasker.make_sliding_window_causal_mask_matrix(
636            input_ids,
637            metadata
638                .as_ref()
639                .map(|(_, _)| &seqlen_offsets as &dyn PastKvLenCache)
640                .unwrap_or(cache as &dyn PastKvLenCache),
641            self.sliding_window,
642            xs.dtype(),
643            self.cfg.num_attn_heads,
644        )?;
645        // PagedAttention prompt chunking
646        let attention_mask = attention_mask.filter(|_| {
647            metadata
648                .as_ref()
649                .map(|(_, meta)| meta.is_first_prompt_chunk)
650                .unwrap_or(true)
651        });
652        for (i, layer) in self.layers.iter().enumerate() {
653            xs = self.mapper.map(xs, i)?;
654            xs = layer.forward(
655                &xs,
656                attention_mask
657                    .as_ref()
658                    .map(|m| m.to_device(xs.device()).unwrap())
659                    .as_ref(),
660                seqlen_offsets,
661                &mut cache[i],
662                metadata
663                    .as_ref()
664                    .map(|(kv_cache, metadata)| (kv_cache[i].clone(), *metadata)),
665                flash_params,
666            )?;
667        }
668        let xs = xs.to_device(&self.device)?;
669        let mut xs = xs.apply(&self.norm)?;
670        if let Some(t) = self.lm_head.quantized_act_type() {
671            xs = xs.to_dtype(t)?;
672        }
673        extract_logits(&MatMul.qmethod_matmul(&xs, &*self.lm_head)?, context_lens)
674    }
675}

677impl IsqModel for Model {
678    fn get_layers(
679        &mut self,
680    ) -> (
681        Vec<(&mut Arc<dyn QuantMethod>, Option<usize>)>,
682        &dyn DeviceMapper,
683    ) {
684        let mut tensors = Vec::new();
685        tensors.push((&mut self.lm_head, None));
686        for (i, layer) in self.layers.iter_mut().enumerate() {
687            tensors.push((&mut layer.self_attn.q_proj, Some(i)));
688            tensors.push((&mut layer.self_attn.k_proj, Some(i)));
689            tensors.push((&mut layer.self_attn.v_proj, Some(i)));
690            tensors.push((&mut layer.self_attn.o_proj, Some(i)));
691            tensors.push((&mut layer.block_sparse_moe.gate, Some(i)));
692            for expert in &mut layer.block_sparse_moe.experts {
693                tensors.push((&mut expert.w1, Some(i)));
694                tensors.push((&mut expert.w2, Some(i)));
695                tensors.push((&mut expert.w3, Some(i)));
696            }
697        }
698        (tensors, &*self.mapper)
699    }
700
701    fn residual_tensors(&self) -> Vec<(String, Tensor)> {
702        let uvb = UnVarBuilder::new();
703
704        let uvb_m = uvb.pp("model");
705        uvb_m.pp("embed_tokens").add(&self.embed_tokens);
706        uvb_m.pp("norm").add(&self.norm);
707
708        for (layer_idx, layer) in self.layers.iter().enumerate() {
709            let uvb_l = uvb_m.pp("layers").pp(layer_idx);
710            uvb_l.pp("input_layernorm").add(&layer.input_layernorm);
711            uvb_l
712                .pp("post_attention_layernorm")
713                .add(&layer.post_attention_layernorm);
714        }
715
716        uvb.to_safetensors()
717    }
