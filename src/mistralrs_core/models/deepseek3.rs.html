104impl DeepSeekV3Config {
105    pub(crate) fn q_head_dim(&self) -> usize {
106        self.qk_rope_head_dim + self.qk_nope_head_dim
107    }

109    fn softmax_scale(&self) -> f32 {
110        let mut softmax_scale = 1.0 / (self.q_head_dim() as f32).sqrt();
111        if let Some(DeepSeekV2RopeScaling::Yarn {
112            mscale_all_dim,
113            factor,
114            ..
115        }) = self.rope_scaling
116        {
117            let mscale = DeepSeekV2RotaryEmbedding::yarn_get_mscale(factor, mscale_all_dim);
118            softmax_scale = softmax_scale * mscale * mscale;
119        }
120        softmax_scale
121    }

133impl QProj {
134    fn forward(&self, xs: &Tensor) -> Result<Tensor> {
135        match self {
136            Self::Lora { a, norm, b } => {
137                b.forward_autocast(&norm.forward(&a.forward_autocast(xs)?)?)
138            }
139            Self::Plain(lin) => lin.forward_autocast(xs),
140        }
141    }

159impl Attention {
160    #[allow(clippy::too_many_arguments)]
161    fn new(
162        rotary_emb: Arc<DeepSeekV2RotaryEmbedding>,
163        cfg: &DeepSeekV3Config,
164        vb: ShardedVarBuilder,
165        mapper: &dyn DeviceMapper,
166        layer_idx: usize,
167        loading_isq: bool,
168        paged_attn: Option<PagedAttention>,
169        comm: &Arc<mistralrs_quant::Comm>,
170    ) -> Result<Self> {
171        let q_head_dim = cfg.q_head_dim();
172        let q = match cfg.q_lora_rank {
173            Some(lora_rank) => {
174                let a = ReplicatedLayer::new(
175                    cfg.hidden_size,
176                    lora_rank,
177                    &cfg.quantization_config,
178                    cfg.attention_bias,
179                    mapper.set_device(layer_idx, vb.pp("q_a_proj"), loading_isq),
180                )?;
181                let norm = RmsNorm::new(
182                    lora_rank,
183                    cfg.rms_norm_eps,
184                    mapper.set_device(layer_idx, vb.pp("q_a_layernorm"), false),
185                )?;
186                let b = ColumnParallelLayer::new(
187                    lora_rank,
188                    cfg.num_attention_heads * q_head_dim,
189                    &cfg.quantization_config,
190                    false,
191                    comm,
192                    mapper.set_device(layer_idx, vb.pp("q_b_proj"), loading_isq),
193                )?;
194                QProj::Lora { a, norm, b }
195            }
196            None => QProj::Plain(ColumnParallelLayer::new(
197                cfg.hidden_size,
198                cfg.num_attention_heads * q_head_dim,
199                &cfg.quantization_config,
200                false,
201                comm,
202                mapper.set_device(layer_idx, vb.pp("q_proj"), loading_isq),
203            )?),
204        };
205
206        let kv_a_proj_with_mqa = ReplicatedLayer::new(
207            cfg.hidden_size,
208            cfg.kv_lora_rank + cfg.qk_rope_head_dim,
209            &cfg.quantization_config,
210            cfg.attention_bias,
211            mapper.set_device(layer_idx, vb.pp("kv_a_proj_with_mqa"), loading_isq),
212        )?;
213        let kv_a_layernorm = RmsNorm::new(
214            cfg.kv_lora_rank,
215            cfg.rms_norm_eps,
216            mapper.set_device(layer_idx, vb.pp("kv_a_layernorm"), false),
217        )?;
218        let kv_b_proj = ColumnParallelLayer::new(
219            cfg.kv_lora_rank,
220            cfg.num_attention_heads * (q_head_dim - cfg.qk_rope_head_dim + cfg.v_head_dim),
221            &cfg.quantization_config,
222            false,
223            comm,
224            mapper.set_device(layer_idx, vb.pp("kv_b_proj"), loading_isq),
225        )?;
226
227        let o_proj = RowParallelLayer::new(
228            cfg.num_attention_heads * cfg.v_head_dim,
229            cfg.hidden_size,
230            &cfg.quantization_config,
231            cfg.attention_bias,
232            comm,
233            mapper.set_device(layer_idx, vb.pp("o_proj"), loading_isq),
234        )?;
235
236        let mla_weights = MlaWeights::new(
237            paged_attn.is_some(),
238            mapper.device_for(layer_idx, loading_isq),
239        );
240
241        Ok(Self {
242            q,
243            kv_a_proj_with_mqa,
244            kv_a_layernorm,
245            kv_b_proj,
246            o_proj,
247            rotary_emb,
248            cfg: cfg.clone(),
249            q_head_dim,
250            paged_attn,
251            num_attention_heads: cfg.num_attention_heads / comm.world_size(),
252            sdpa_params: SdpaParams {
253                n_kv_groups: 1,
254                softcap: None,
255                softmax_scale: cfg.softmax_scale(),
256                sliding_window: None,
257            },
258            mla_weights,
259        })
260    }

262    fn forward(
263        &self,
264        xs: &Tensor,
265        attention_mask: Option<&Tensor>,
266        seqlen_offsets: &[usize],
267        kv_cache: &mut KvCache,
268        metadata: Option<((Tensor, Tensor), &PagedAttentionInputMetadata)>,
269        flash_params: &FlashParams,
270    ) -> Result<Tensor> {
271        let (bs, seq_len, _) = xs.dims3()?;
272
273        let mut q = self.q.forward(xs)?;
274        q = q
275            .reshape((bs, seq_len, self.num_attention_heads, self.q_head_dim))?
276            .transpose(1, 2)?;
277        let q_split = q.split(
278            &[self.cfg.qk_nope_head_dim, self.cfg.qk_rope_head_dim],
279            D::Minus1,
280        )?;
281        let q_nope = q_split[0].clone();
282        let mut q_pe = q_split[1].clone();
283
284        let mut compressed_kv = self.kv_a_proj_with_mqa.forward_autocast(xs)?;
285        let ckv_split = compressed_kv.split(
286            &[self.cfg.kv_lora_rank, self.cfg.qk_rope_head_dim],
287            D::Minus1,
288        )?;
289        compressed_kv = ckv_split[0].clone();
290        let mut k_pe = ckv_split[1].clone();
291        k_pe = k_pe
292            .reshape((bs, seq_len, 1, self.cfg.qk_rope_head_dim))?
293            .transpose(1, 2)?;
294
295        let ckv = self.kv_a_layernorm.forward(&compressed_kv)?;
296
297        (q_pe, k_pe) = self.rotary_emb.forward(&q_pe, &k_pe, seqlen_offsets)?;
298
299        let use_mla_decode = should_use_mla_decode(
300            attention_mask,
301            seq_len,
302            self.paged_attn.is_some(),
303            q_nope.device(),
304            &metadata,
305        );
306
307        let mut attn_out = if use_mla_decode {
308            mla_decode_forward(
309                &q_nope,
310                &q_pe,
311                &ckv,
312                &k_pe,
313                &metadata,
314                &self.mla_weights,
315                self.kv_b_proj.as_ref(),
316                &self.sdpa_params,
317                self.num_attention_heads,
318                self.cfg.kv_lora_rank,
319                self.cfg.qk_rope_head_dim,
320                self.cfg.qk_nope_head_dim,
321                self.cfg.v_head_dim,
322                bs,
323                seq_len,
324            )?
325        } else {
326            let mut kv = self.kv_b_proj.forward_autocast(&ckv)?;
327            kv = kv
328                .reshape((
329                    bs,
330                    seq_len,
331                    self.num_attention_heads,
332                    self.cfg.qk_nope_head_dim + self.cfg.v_head_dim,
333                ))?
334                .transpose(1, 2)?;
335
336            let kv_split =
337                kv.split(&[self.cfg.qk_nope_head_dim, self.cfg.v_head_dim], D::Minus1)?;
338            let k_nope = kv_split[0].clone();
339            let mut v = kv_split[1].clone();
340
341            let q = Tensor::cat(&[&q_nope, &q_pe], D::Minus1)?.contiguous()?;
342            let mut k = Tensor::cat(
343                &[&k_nope, &k_pe.repeat((1, self.num_attention_heads, 1, 1))?],
344                D::Minus1,
345            )?
346            .contiguous()?;
347
348            let use_mla_cache = should_use_mla_cache(self.paged_attn.is_some(), q.device());
349
350            if use_mla_cache {
351                mla_cache_forward(
352                    &q,
353                    &k,
354                    &v,
355                    &ckv,
356                    &k_pe,
357                    attention_mask,
358                    seqlen_offsets,
359                    &metadata,
360                    flash_params,
361                    self.kv_b_proj.as_ref(),
362                    &self.sdpa_params,
363                    self.num_attention_heads,
364                    self.cfg.kv_lora_rank,
365                    self.cfg.qk_rope_head_dim,
366                    self.cfg.qk_nope_head_dim,
367                    self.cfg.v_head_dim,
368                    bs,
369                    seq_len,
370                )?
371            } else {
372                match &self.paged_attn {
373                    Some(paged_attn) => match metadata {
374                        Some(((key_cache, value_cache), input_metadata)) => {
375                            let v = v
376                                .pad_with_zeros(
377                                    D::Minus1,
378                                    0,
379                                    self.q_head_dim - self.cfg.v_head_dim,
380                                )?
381                                .contiguous()?;
382                            paged_attn
383                                .forward(
384                                    &q,
385                                    &k,
386                                    &v,
387                                    attention_mask,
388                                    Some(key_cache),
389                                    Some(value_cache),
390                                    input_metadata,
391                                    &self.sdpa_params,
392                                    Some(flash_params),
393                                )?
394                                .narrow(D::Minus1, 0, self.cfg.v_head_dim)?
395                        }
396                        None => {
397                            // If we don't have metadata, we are most likely generating an imatrix so we don't want to populate that.
398                            // Generating the dummy metadata with the assumption that we are not generating text (only processing prompts).
399                            let input_metadata = PagedAttentionInputMetadata::dummy(q.device())?;
400                            // Sanity check.
401                            assert!(attention_mask.is_some());
402                            let v = v
403                                .pad_with_zeros(
404                                    D::Minus1,
405                                    0,
406                                    self.q_head_dim - self.cfg.v_head_dim,
407                                )?
408                                .contiguous()?;
409                            paged_attn
410                                .forward(
411                                    &q,
412                                    &k,
413                                    &v,
414                                    attention_mask,
415                                    None,
416                                    None,
417                                    &input_metadata,
418                                    &self.sdpa_params,
419                                    Some(flash_params),
420                                )?
421                                .narrow(D::Minus1, 0, self.cfg.v_head_dim)?
422                        }
423                    },
424                    None => {
425                        (k, v) = kv_cache.append(&k, &v)?;
426
427                        Sdpa.run_attention(
428                            &q,
429                            &k,
430                            &v,
431                            attention_mask,
432                            Some(flash_params),
433                            &self.sdpa_params,
434                        )?
435                    }
436                }
437            }
438        };
439
440        attn_out = if attention_mask.is_some() {
441            attn_out.transpose(1, 2)?.reshape((bs, seq_len, ()))?
442        } else {
443            attn_out.reshape((bs, seq_len, ()))?
444        };
445
446        self.o_proj.forward_autocast(&attn_out)
447    }
448}

450struct MoeGate {
451    weight: Tensor,
452    cfg: DeepSeekV3Config,
453    top_k: usize,
454    n_routed_experts: usize,
455    e_score_correction_bias: Option<Tensor>,
456}
457
458impl MoeGate {
459    fn new(cfg: &DeepSeekV3Config, vb: ShardedVarBuilder, n_routed_experts: usize) -> Result<Self> {
460        let weight = vb.get((n_routed_experts, cfg.hidden_size), "weight")?;
461        let e_score_correction_bias = if matches!(cfg.topk_method, TopkMethod::NoAuxTc) {
462            Some(vb.get_with_hints_dtype(
463                n_routed_experts,
464                "e_score_correction_bias",
465                Default::default(),
466                DType::F32,
467            )?)
468        } else {
469            None
470        };
471        Ok(Self {
472            weight,
473            cfg: cfg.clone(),
474            top_k: cfg.num_experts_per_tok.unwrap(),
475            n_routed_experts,
476            e_score_correction_bias,
477        })
478    }
479
480    /// (topk_idx, topk_weight)
481    fn forward(&self, xs: &Tensor) -> Result<(Tensor, Tensor)> {
482        let (bs, seq_len, h) = xs.dims3()?;
483        // Compute gating score
484        let xs = xs.reshape(((), h))?;
485        let logits = xs
486            .to_dtype(DType::F32)?
487            .broadcast_matmul(&self.weight.t()?.to_dtype(DType::F32)?)?;
488        let scores = match self.cfg.scoring_func {
489            ScoringFunc::Softmax => candle_nn::ops::softmax_last_dim(&logits)?,
490            ScoringFunc::Sigmoid => candle_nn::ops::sigmoid(&logits)?,
491        };
492
493        // Select top-k experts
494        let (mut topk_weight, topk_idx) = match self.cfg.topk_method {
495            TopkMethod::Greedy => {
496                let TopKOutput { values, indices } = scores.topk_unsorted(self.top_k)?;
497                (values, indices)
498            }
499            TopkMethod::NoAuxTc => {
500                let Some(e_score_correction_bias) = &self.e_score_correction_bias else {
501                    candle_core::bail!("Expected e_score_correction_bias")
502                };
503                let scores_for_choice = scores
504                    .reshape((bs * seq_len, ()))?
505                    .broadcast_add(&e_score_correction_bias.unsqueeze(0)?)?;
506                // (n, n_group)
507                let group_scores = scores_for_choice
508                    .reshape((bs * seq_len, self.cfg.n_group, ()))?
509                    .topk(2)?
510                    .values
511                    .sum(D::Minus1)?;
512                // (n, topk_group)
513                let group_idx = group_scores.topk(self.cfg.topk_group)?.indices;
514                // (n, n_group)
515                let mut group_mask = group_scores.zeros_like()?;
516                // (n, n_group)
517                group_mask = group_mask.scatter_add(
518                    &group_idx,
519                    &group_idx.ones_like()?.to_dtype(group_mask.dtype())?,
520                    1,
521                )?;
522                // (n, e)
523                let score_mask = group_mask
524                    .unsqueeze(D::Minus1)?
525                    .expand((
526                        bs * seq_len,
527                        self.cfg.n_group,
528                        self.n_routed_experts / self.cfg.n_group,
529                    ))?
530                    .reshape((bs * seq_len, ()))?;
531                // (n, e)
532                // Invert the mask
533                let tmp_scores = scores_for_choice.broadcast_mul(&score_mask)?;
534                let topk_idx = tmp_scores.topk(self.top_k)?.indices;
535                (scores.gather(&topk_idx, 1)?, topk_idx)
536            }
537            TopkMethod::GroupLimitedGreedy => {
538                // (n, n_group)
539                let group_scores = scores
540                    .reshape((bs * seq_len, self.cfg.n_group, ()))?
541                    .max(D::Minus1)?;
542                // (n, topk_group)
543                let group_idx = group_scores.topk_unsorted(self.cfg.topk_group)?.indices;
544                // (n, n_group)
545                let mut group_mask = group_scores.zeros_like()?;
546                // (n, n_group)
547                group_mask = group_mask.scatter_add(
548                    &group_idx,
549                    &group_idx.ones_like()?.to_dtype(group_mask.dtype())?,
550                    1,
551                )?;
552                // (n, e)
553                let score_mask = group_mask
554                    .unsqueeze(D::Minus1)?
555                    .expand((
556                        bs * seq_len,
557                        self.cfg.n_group,
558                        self.n_routed_experts / self.cfg.n_group,
559                    ))?
560                    .reshape((bs * seq_len, ()))?;
561                // (n, e)
562                // Invert the mask
563                let tmp_scores = masked_fill(&score_mask, &(1. - &score_mask.ne(0.)?)?, 0.)?;
564                let TopKOutput { values, indices } = tmp_scores.topk_unsorted(self.top_k)?;
565                (values, indices)
566            }
567        };
568
569        if matches!(self.cfg.scoring_func, ScoringFunc::Sigmoid) {
570            let denmoninator = (topk_weight.sum_keepdim(D::Minus1)? + 1e-20)?;
571            topk_weight = topk_weight.broadcast_div(&denmoninator)?;
572        }
573
574        // Must multiply the scaling factor
575        topk_weight = (topk_weight * self.cfg.routed_scaling_factor)?;
576
577        Ok((topk_idx, topk_weight))
578    }
579}

617            comm,
618            loading_isq,
619            &cfg.quantization_config,
620            cfg.hidden_act,
621        )?;
622
623        let shared_experts = if let Some(n_shared_experts) = n_shared_experts {
624            let intermediate_size = cfg.moe_intermediate_size * n_shared_experts;
625            Some(Mlp::new(
626                mapper.set_device(layer_idx, vb.pp("shared_experts"), loading_isq),
627                cfg.hidden_size,
628                intermediate_size,
629                &cfg.quantization_config,
630                cfg.hidden_act,
631                comm,
632            )?)
633        } else {
634            None
635        };
636        let gate = MoeGate::new(
637            cfg,
638            mapper.set_device(layer_idx, vb.pp("gate"), false),
639            n_routed_experts,
640        )?;
641        Ok(Self {
642            experts,
643            shared_experts,
644            gate,
645        })
646    }
647
648    fn forward(&self, xs: &Tensor) -> Result<Tensor> {
649        let identity = xs.clone();
650        let (b_size, seq_len, hidden_dim) = xs.dims3()?;
651
652        // Get routing weights from gate
653        let (topk_idx, topk_weight) = self.gate.forward(xs)?;
654
655        // Forward through routed experts using optimized MoEExperts
656        let mut y = self.experts.forward(xs, topk_weight, &topk_idx)?;
657        y = y.reshape((b_size, seq_len, hidden_dim))?;
658
659        // Add shared expert output
660        if let Some(ref shared_experts) = self.shared_experts {
661            y = (y + shared_experts.forward(&identity)?)?;
662        }
663
664        Ok(y)
665    }
666
667    fn get_isq_layers(&mut self) -> Vec<&mut Arc<dyn QuantMethod>> {
668        let mut layers = self.experts.get_isq_layers();
669        if let Some(ref mut shared) = self.shared_experts {
670            layers.push(&mut shared.gate);
671            layers.push(&mut shared.up);
672            layers.push(&mut shared.down);
673        }
674        layers
675    }
676}

678enum MoeOrMlp {
679    Moe(Box<Moe>),
680    Mlp(Mlp),

683impl MoeOrMlp {
684    fn forward(&self, xs: &Tensor) -> Result<Tensor> {
685        match self {
686            Self::Mlp(mlp) => mlp.forward(xs),
687            Self::Moe(moe) => moe.forward(xs),
688        }
689    }

692struct DecoderLayer {
693    input_layernorm: RmsNorm,
694    post_attention_layernorm: RmsNorm,
695    attn: Attention,
696    moe_or_mlp: MoeOrMlp,
697}
698
699impl DecoderLayer {
700    #[allow(clippy::too_many_arguments)]
701    fn new(
702        rotary_emb: Arc<DeepSeekV2RotaryEmbedding>,
703        cfg: &DeepSeekV3Config,
704        vb: ShardedVarBuilder,
705        mapper: &dyn DeviceMapper,
706        layer_idx: usize,
707        loading_isq: bool,
708        paged_attn: Option<PagedAttention>,
709        comm: &Arc<mistralrs_quant::Comm>,
710        real_device: Device,
711    ) -> Result<Self> {
712        let attn = Attention::new(
713            rotary_emb,
714            cfg,
715            vb.pp("self_attn"),
716            mapper,
717            layer_idx,
718            loading_isq,
719            paged_attn,
720            comm,
721        )?;
722        let input_layernorm = RmsNorm::new(
723            cfg.hidden_size,
724            cfg.rms_norm_eps,
725            mapper.set_device(layer_idx, vb.pp("input_layernorm"), false),
726        )?;
727        let post_attention_layernorm = RmsNorm::new(
728            cfg.hidden_size,
729            cfg.rms_norm_eps,
730            mapper.set_device(layer_idx, vb.pp("post_attention_layernorm"), false),
731        )?;
732        let moe_or_mlp = if let Some(n_routed_experts) = cfg.n_routed_experts.filter(|_| {
733            layer_idx >= cfg.first_k_dense_replace && layer_idx % cfg.moe_layer_freq == 0
734        }) {
735            MoeOrMlp::Moe(Box::new(Moe::new(
736                cfg,
737                vb.pp("mlp"),
738                mapper,
739                layer_idx,
740                loading_isq,
741                cfg.n_shared_experts,
742                n_routed_experts,
743                comm,
744                real_device,
745            )?))
746        } else {
747            MoeOrMlp::Mlp(Mlp::new(
748                mapper.set_device(layer_idx, vb.pp("mlp"), loading_isq),
749                cfg.hidden_size,
750                cfg.intermediate_size,
751                &cfg.quantization_config,
752                cfg.hidden_act,
753                comm,
754            )?)
755        };
756
757        Ok(Self {
758            input_layernorm,
759            post_attention_layernorm,
760            attn,
761            moe_or_mlp,
762        })
763    }
764
765    fn forward(
766        &self,
767        xs: &Tensor,
768        attention_mask: Option<&Tensor>,
769        seqlen_offsets: &[usize],
770        kv_cache: &mut KvCache,
771        metadata: Option<((Tensor, Tensor), &PagedAttentionInputMetadata)>,
772        flash_params: &FlashParams,
773    ) -> Result<Tensor> {
774        let residual = xs;
775        let xs = self.input_layernorm.forward(xs)?;
776        let xs = self.attn.forward(
777            &xs,
778            attention_mask,
779            seqlen_offsets,
780            kv_cache,
781            metadata,
782            flash_params,
783        )?;
784        let xs = (xs + residual)?;
785        let residual = &xs;
786        let xs = self
787            .moe_or_mlp
788            .forward(&xs.apply(&self.post_attention_layernorm)?)?;
789        residual + xs
790    }
791}

806    pub fn new(
807        cfg: &DeepSeekV3Config,
808        vb: ShardedVarBuilder,
809        _is_gptx: bool,
810        normal_loading_metadata: NormalLoadingMetadata,
811        attention_mechanism: AttentionImplementation,
812    ) -> Result<Self> {

814
815        let mapper = normal_loading_metadata.mapper;
816

843            mapper.set_nm_device(vb_m.pp("norm"), false),

848            rope_scaling: cfg.rope_scaling.clone(),
849            max_position_embeddings: cfg.max_position_embeddings,

851            qk_rope_head_dim: cfg.qk_rope_head_dim,

859                Arc::new(DeepSeekV2RotaryEmbedding::new(

862                    device,

868        let layers: Vec<DecoderLayer> = NiceProgressBar::<_, 'b'>(
869            0..cfg.num_hidden_layers,
870            "Loading repeating layers",
871            &normal_loading_metadata.multi_progress,
872        )
873        .par_iter_if_isq(|layer_idx| {
874            let device = mapper
875                .device_for(layer_idx, false)
876                .unwrap_or(&normal_loading_metadata.real_device);

887            };
888            let comm = mapper.get_comm_for(layer_idx)?;
889            DecoderLayer::new(
890                rotary_emb.clone(),
891                cfg,
892                vb_l.pp(layer_idx),

894                layer_idx,

900        })?;
901
902        Ok(Self {
903            lm_head,
904            embed_tokens,
905            norm,
906            layers,
907            cache: EitherCache::Normal(NormalCache::new(
908                cfg.num_hidden_layers,
909                cfg.max_position_embeddings,
910            )),
911            device: normal_loading_metadata.real_device.clone(),
912            max_seq_len: cfg.max_position_embeddings,
913            cfg: ModelConfigMetadata {
914                max_seq_len: cfg.max_position_embeddings,
915                num_layers: cfg.num_hidden_layers,
916                hidden_size: cfg.hidden_size,
917                num_kv_heads: (cfg.num_attention_heads / mapper.get_comm_for(0)?.world_size())
918                    .max(1),
919                num_attn_heads: (cfg.num_attention_heads / mapper.get_comm_for(0)?.world_size())
920                    .max(1),
921                sliding_window: None,
922                k_head_dim: cfg.q_head_dim(),
923                v_head_dim: if matches!(
924                    attention_mechanism,
925                    AttentionImplementation::PagedAttention
926                ) {
927                    cfg.q_head_dim()
928                } else {
929                    cfg.v_head_dim
930                },
931                #[cfg(all(feature = "cuda", target_family = "unix"))]
932                kv_cache_layout: if matches!(
933                    attention_mechanism,
934                    AttentionImplementation::PagedAttention
935                ) {
936                    crate::paged_attention::KvCacheLayout::Mla {

939                    }
940                } else {
941                    crate::paged_attention::KvCacheLayout::Standard
942                },
943                #[cfg(not(all(feature = "cuda", target_family = "unix")))]
944                kv_cache_layout: crate::paged_attention::KvCacheLayout::Standard,
945            },

947        })

949
950    #[allow(clippy::too_many_arguments)]
951    pub fn forward(
952        &self,
953        input_ids: &Tensor,
954        seqlen_offsets: &[usize],
955        context_lens: Vec<(usize, usize)>,
956        metadata: Option<(Vec<(Tensor, Tensor)>, &PagedAttentionInputMetadata)>,
957        flash_params: &FlashParams,
958    ) -> Result<Tensor> {
959        let mut xs = self.embed_tokens.forward(input_ids)?;
960        let cache = &mut self.cache.normal().0;
961        let attention_mask = CausalMasker.make_causal_mask_matrix(
962            input_ids,
963            metadata
964                .as_ref()
965                .map(|(_, _)| &seqlen_offsets as &dyn PastKvLenCache)
966                .unwrap_or(cache as &dyn PastKvLenCache),
967            xs.dtype(),
968            self.cfg.num_attn_heads,
969        )?;
970        // PagedAttention prompt chunking
971        let attention_mask = attention_mask.filter(|_| {
972            metadata
973                .as_ref()
974                .map(|(_, meta)| meta.is_first_prompt_chunk)
975                .unwrap_or(true)
976        });
977        for (i, layer) in self.layers.iter().enumerate() {
978            xs = self.mapper.map(xs, i)?;
979            xs = layer.forward(
980                &xs,
981                attention_mask
982                    .as_ref()
983                    .map(|m| m.to_device(xs.device()).unwrap())
984                    .as_ref(),
985                seqlen_offsets,
986                &mut cache[i],
987                metadata
988                    .as_ref()
989                    .map(|(kv_cache, metadata)| (kv_cache[i].clone(), *metadata)),
990                flash_params,
991            )?;
992        }
993        let xs = xs.to_device(&self.device)?;
994        let xs = xs.apply(&self.norm)?;
995        extract_logits(&self.lm_head.forward_autocast(&xs)?, context_lens)
996    }
997}
