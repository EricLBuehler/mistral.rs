52impl Config {
53    pub fn hidden_act(&self) -> Result<Activation> {
54        match (self.hidden_act, self.hidden_activation) {
55            (None, Some(act)) | (Some(act), None) => Ok(act),
56            (Some(act), Some(_)) => {
57                // If both are set just use hidden_act
58                Ok(act)
59            }
60            (None, None) => candle_core::bail!("none of hidden_act and hidden_activation are set"),
61        }
62    }

64struct Attention {
65    q_proj: Arc<dyn QuantMethod>,
66    k_proj: Arc<dyn QuantMethod>,
67    v_proj: Arc<dyn QuantMethod>,
68    o_proj: Arc<dyn QuantMethod>,
69    num_heads: usize,
70    num_kv_heads: usize,
71    head_dim: usize,
72    rotary_emb: Arc<RotaryEmbedding>,
73    use_sliding_window: bool,
74    paged_attn: Option<PagedAttention>,
75    sdpa_params: SdpaParams,
76}

78impl Attention {
79    fn new(
80        rotary_emb: Arc<RotaryEmbedding>,
81        cfg: &Config,
82        layer_idx: usize,
83        vb: ShardedVarBuilder,
84        paged_attn: Option<PagedAttention>,
85        comm: &Arc<mistralrs_quant::Comm>,
86    ) -> Result<Self> {
87        let hidden_sz = cfg.hidden_size;
88        let num_heads = cfg.num_attention_heads;
89        let num_kv_heads = cfg.num_key_value_heads;
90        let head_dim = cfg.head_dim;
91        let bias = cfg.attention_bias;
92        let q_proj = ColumnParallelLayer::new(
93            hidden_sz,
94            num_heads * head_dim,
95            &cfg.quantization_config,
96            bias,
97            comm,
98            vb.pp("q_proj"),
99        )?;
100        let kv_shard = mistralrs_quant::compute_kv_shard(
101            cfg.num_key_value_heads,
102            cfg.hidden_size / cfg.num_attention_heads,
103            comm,
104        );
105        let k_proj = ColumnParallelLayer::new_with_shard(
106            hidden_sz,
107            num_kv_heads * head_dim,
108            &cfg.quantization_config,
109            bias,
110            comm,
111            kv_shard,
112            vb.pp("k_proj"),
113        )?;
114        let v_proj = ColumnParallelLayer::new_with_shard(
115            hidden_sz,
116            num_kv_heads * head_dim,
117            &cfg.quantization_config,
118            bias,
119            comm,
120            kv_shard,
121            vb.pp("v_proj"),
122        )?;
123        let o_proj = RowParallelLayer::new(
124            num_heads * head_dim,
125            hidden_sz,
126            &cfg.quantization_config,
127            bias,
128            comm,
129            vb.pp("o_proj"),
130        )?;
131        let sliding_window = if layer_idx % 2 == 0 {
132            // ^ Order is SWA, global, SWA
133            Some(cfg.sliding_window)
134        } else {
135            None
136        };
137        Ok(Self {
138            q_proj,
139            k_proj,
140            v_proj,
141            o_proj,
142            num_heads: num_heads / comm.world_size(),
143            num_kv_heads: (num_kv_heads / comm.world_size()).max(1),
144            head_dim,
145            rotary_emb,
146            use_sliding_window: layer_idx % 2 == 0, // Order is SWA, global, SWA
147            paged_attn,
148            sdpa_params: SdpaParams {
149                n_kv_groups: mistralrs_quant::compute_n_kv_groups(
150                    cfg.num_key_value_heads,
151                    cfg.num_attention_heads,
152                    comm,
153                ),
154                softcap: cfg.attn_logit_softcapping.map(|x| x as f32),
155                softmax_scale: 1.0 / (cfg.query_pre_attn_scalar as f32).sqrt(),
156                sliding_window,
157            },
158        })
159    }

161    #[allow(clippy::too_many_arguments)]
162    fn forward(
163        &self,
164        xs: &Tensor,
165        attention_mask: Option<&Tensor>,
166        sliding_attention_mask: Option<&Tensor>,
167        seqlen_offsets: &[usize],
168        kv_cache: &mut KvCache,
169        metadata: Option<((Tensor, Tensor), &PagedAttentionInputMetadata)>,
170        flash_params: &FlashParams,
171    ) -> Result<Tensor> {
172        let (b_sz, q_len, _) = xs.dims3()?;
173
174        let original_dtype = xs.dtype();
175        let mut xs = xs.clone();
176        if let Some(t) = self.q_proj.quantized_act_type() {
177            xs = xs.to_dtype(t)?;
178        }
179        let mut q = MatMul.qmethod_matmul(&xs, &*self.q_proj)?;
180        let mut k = MatMul.qmethod_matmul(&xs, &*self.k_proj)?;
181        let mut v = MatMul.qmethod_matmul(&xs, &*self.v_proj)?;
182        if self.q_proj.quantized_act_type().is_some() {
183            q = q.to_dtype(original_dtype)?;
184            k = k.to_dtype(original_dtype)?;
185            v = v.to_dtype(original_dtype)?;
186        }
187
188        let (q, k, v) = if q_len != 1 {
189            let q = q
190                .reshape((b_sz, q_len, self.num_heads, self.head_dim))?
191                .transpose(1, 2)?;
192            let k = k
193                .reshape((b_sz, q_len, self.num_kv_heads, self.head_dim))?
194                .transpose(1, 2)?;
195            let v = v
196                .reshape((b_sz, q_len, self.num_kv_heads, self.head_dim))?
197                .transpose(1, 2)?;
198            (q, k, v)
199        } else {
200            let q = q.reshape((b_sz, self.num_heads, q_len, self.head_dim))?;
201            let k = k.reshape((b_sz, self.num_kv_heads, q_len, self.head_dim))?;
202            let v = v.reshape((b_sz, self.num_kv_heads, q_len, self.head_dim))?;
203            (q, k, v)
204        };
205
206        let (q, k) = self.rotary_emb.forward(&q, &k, seqlen_offsets)?;
207
208        let mask = if self.use_sliding_window {
209            sliding_attention_mask
210        } else {
211            attention_mask
212        };
213
214        let mut attn_output = match &self.paged_attn {
215            Some(paged_attn) => match metadata {
216                Some(((key_cache, value_cache), input_metadata)) => paged_attn.forward(
217                    &q,
218                    &k,
219                    &v,
220                    attention_mask,
221                    Some(key_cache),
222                    Some(value_cache),
223                    input_metadata,
224                    &self.sdpa_params,
225                    Some(flash_params),
226                )?,
227                None => {
228                    // If we don't have metadata, we are most likely generating an imatrix so we don't want to populate that.
229                    // Generating the dummy metadata with the assumption that we are not generating text (only processing prompts).
230                    let input_metadata = PagedAttentionInputMetadata::dummy(q.device())?;
231                    // Sanity check.
232                    assert!(attention_mask.is_some());
233                    paged_attn.forward(
234                        &q,
235                        &k,
236                        &v,
237                        attention_mask,
238                        None,
239                        None,
240                        &input_metadata,
241                        &self.sdpa_params,
242                        Some(flash_params),
243                    )?
244                }
245            },
246            None => {
247                // self.sliding_window is None if !self.use_sliding_window
248                let (k, v) = kv_cache.append(&k, &v)?;
249
250                Sdpa.run_attention(&q, &k, &v, mask, Some(flash_params), &self.sdpa_params)?
251            }
252        };
253
254        if let Some(t) = self.q_proj.quantized_act_type() {
255            attn_output = attn_output.to_dtype(t)?;
256        }
257        attn_output = if attention_mask.is_some() {
258            attn_output.transpose(1, 2)?.reshape((b_sz, q_len, ()))?
259        } else {
260            attn_output.reshape((b_sz, q_len, ()))?
261        };
262        let mut res = MatMul.qmethod_matmul(&attn_output, &*self.o_proj)?;
263        if self.q_proj.quantized_act_type().is_some() {
264            res = res.to_dtype(original_dtype)?;
265        }
266        Ok(res)
267    }
268}

270struct DecoderLayer {
271    self_attn: Attention,
272    mlp: Box<dyn MlpLayer>,
273    input_layernorm: RmsNorm,
274    post_attention_layernorm: RmsNorm,
275    pre_feedforward_layernorm: RmsNorm,
276    post_feedforward_layernorm: RmsNorm,
277}
278
279impl DecoderLayer {
280    #[allow(clippy::too_many_arguments)]
281    fn new(
282        rotary_emb: Arc<RotaryEmbedding>,
283        cfg: &Config,
284        vb: ShardedVarBuilder,
285        mapper: &dyn DeviceMapper,
286        layer_idx: usize,
287        loading_isq: bool,
288        paged_attn: Option<PagedAttention>,
289        comm: &Arc<mistralrs_quant::Comm>,
290    ) -> Result<Self> {
291        let self_attn = Attention::new(
292            rotary_emb,
293            cfg,
294            layer_idx,
295            mapper.set_device(layer_idx, vb.pp("self_attn"), loading_isq),
296            paged_attn,
297            comm,
298        )?;
299        let mlp = Mlp::new(
300            mapper.set_device(layer_idx, vb.pp("mlp"), loading_isq),
301            cfg.hidden_size,
302            cfg.intermediate_size,
303            &cfg.quantization_config,
304            cfg.hidden_act()?,
305            comm,
306        )?;
307        let input_layernorm = RmsNorm::new_gemma(
308            cfg.hidden_size,
309            cfg.rms_norm_eps,
310            mapper.set_device(layer_idx, vb.pp("input_layernorm"), false),
311        )?;
312        let post_attention_layernorm = RmsNorm::new_gemma(
313            cfg.hidden_size,
314            cfg.rms_norm_eps,
315            mapper.set_device(layer_idx, vb.pp("post_attention_layernorm"), false),
316        )?;
317        let pre_feedforward_layernorm = RmsNorm::new_gemma(
318            cfg.hidden_size,
319            cfg.rms_norm_eps,
320            mapper.set_device(layer_idx, vb.pp("pre_feedforward_layernorm"), false),
321        )?;
322        let post_feedforward_layernorm = RmsNorm::new_gemma(
323            cfg.hidden_size,
324            cfg.rms_norm_eps,
325            mapper.set_device(layer_idx, vb.pp("post_feedforward_layernorm"), false),
326        )?;
327        Ok(Self {
328            self_attn,
329            mlp: Box::new(mlp),
330            input_layernorm,
331            post_attention_layernorm,
332            pre_feedforward_layernorm,
333            post_feedforward_layernorm,
334        })
335    }
336
337    #[allow(clippy::too_many_arguments)]
338    fn forward(
339        &self,
340        xs: &Tensor,
341        attention_mask: Option<&Tensor>,
342        sliding_attention_mask: Option<&Tensor>,
343        seqlen_offsets: &[usize],
344        kv_cache: &mut KvCache,
345        metadata: Option<((Tensor, Tensor), &PagedAttentionInputMetadata)>,
346        flash_params: &FlashParams,
347    ) -> Result<Tensor> {
348        let residual = xs;
349        let xs = self.input_layernorm.forward(xs)?;
350        let xs = self
351            .self_attn
352            .forward(
353                &xs,
354                attention_mask,
355                sliding_attention_mask,
356                seqlen_offsets,
357                kv_cache,
358                metadata,
359                flash_params,
360            )?
361            .apply(&self.post_attention_layernorm)?;
362        let xs = (xs + residual)?;
363        let residual = &xs;
364        let xs = self
365            .mlp
366            .forward(&xs.apply(&self.pre_feedforward_layernorm)?)?
367            .apply(&self.post_feedforward_layernorm)?;
368        residual + xs
369    }
370}

387impl Model {
388    pub fn new(
389        cfg: &Config,
390        vb: ShardedVarBuilder,
391        is_gptx: bool,
392        normal_loading_metadata: NormalLoadingMetadata,
393        attention_mechanism: AttentionImplementation,
394    ) -> Result<Self> {
395        if let Some(ref quant_cfg) = &cfg.quantization_config {
396            tracing::info!(
397                "Using {} quantization: {}.",
398                quant_cfg.name(),
399                quant_cfg.get_bits_name(&vb)
400            );
401        }
402        let mapper = normal_loading_metadata.mapper;
403
404        let vb_m = vb.pp("model");
405        let embed_tokens = embedding(
406            cfg.vocab_size,
407            cfg.hidden_size,
408            mapper.set_nm_device(vb_m.pp("embed_tokens"), false),
409            &cfg.quantization_config,
410        )?;
411        let mut ropes = HashMap::new();
412        for layer_idx in 0..cfg.num_hidden_layers {
413            let device = mapper
414                .device_for(layer_idx, false)
415                .unwrap_or(&normal_loading_metadata.real_device);
416            ropes.insert(
417                device.location(),
418                Arc::new(RotaryEmbedding::new(
419                    cfg.rope_theta as f32,
420                    cfg.head_dim,
421                    cfg.max_position_embeddings,
422                    device,
423                    is_gptx,
424                    vb_m.dtype(),
425                )?),
426            );
427        }
428        let vb_l = vb_m.pp("layers");
429        let layers: Vec<DecoderLayer> = NiceProgressBar::<_, 'b'>(
430            0..cfg.num_hidden_layers,
431            "Loading repeating layers",
432            &normal_loading_metadata.multi_progress,
433        )
434        .par_iter_if_isq(|layer_idx| {
435            let device = mapper
436                .device_for(layer_idx, false)
437                .unwrap_or(&normal_loading_metadata.real_device);
438            let rotary_emb = ropes
439                .get(&device.location())
440                .expect("No RoPE for device location!")
441                .clone();
442            let paged_attn = match &attention_mechanism {
443                AttentionImplementation::Eager => None,
444                AttentionImplementation::PagedAttention => {
445                    Some(PagedAttention::new(cfg.head_dim, device, None)?)
446                }
447            };
448            let comm = mapper.get_comm_for(layer_idx)?;
449            DecoderLayer::new(
450                rotary_emb.clone(),
451                cfg,
452                vb_l.pp(layer_idx),
453                &*mapper,
454                layer_idx,
455                normal_loading_metadata.loading_isq,
456                paged_attn,
457                &comm,
458            )
459        })?;
460        let norm = RmsNorm::new_gemma(
461            cfg.hidden_size,
462            cfg.rms_norm_eps,
463            mapper.set_nm_device(vb_m.pp("norm"), false),
464        )?;
465        let lm_head = mapper.cast_nm_device(
466            embed_tokens.embeddings(),
467            normal_loading_metadata.loading_isq,
468        )?;
469        Ok(Self {
470            embed_tokens,
471            layers,
472            norm,
473            lm_head: Arc::new(UnquantLinear::new(QuantMethodConfig::Unquantized(
474                Linear::new(lm_head, None),
475            ))?),
476            device: normal_loading_metadata.real_device,
477            hidden_size: cfg.hidden_size,
478            cache: EitherCache::Normal(NormalCache::new_sliding(
479                cfg.num_hidden_layers,
480                cfg.max_position_embeddings,
481                Some(cfg.sliding_window),
482            )),
483            max_seq_len: cfg.max_position_embeddings,
484            sliding_window: cfg.sliding_window,
485            final_logit_softcapping: cfg.final_logit_softcapping,
486            cfg: ModelConfigMetadata {
487                max_seq_len: cfg.max_position_embeddings,
488                num_layers: cfg.num_hidden_layers,
489                hidden_size: cfg.hidden_size,
490                num_attn_heads: cfg.num_attention_heads / mapper.get_comm_for(0)?.world_size(),
491                num_kv_heads: (cfg.num_key_value_heads / mapper.get_comm_for(0)?.world_size())
492                    .max(1),
493                sliding_window: None,
494                k_head_dim: cfg.head_dim,
495                v_head_dim: cfg.head_dim,
496            },
497            mapper,
498        })
499    }
500
501    pub fn forward(
502        &self,
503        input_ids: &Tensor,
504        seqlen_offsets: &[usize],
505        context_lens: Vec<(usize, usize)>,
506        metadata: Option<(Vec<(Tensor, Tensor)>, &PagedAttentionInputMetadata)>,
507        flash_params: &FlashParams,
508    ) -> Result<Tensor> {
509        let xs = self.embed_tokens.forward(input_ids)?;
510        let mut xs = (xs * (self.hidden_size as f64).sqrt())?;
511        let cache = &mut self.cache.normal().0;
512        let attention_mask = CausalMasker.make_causal_mask_matrix(
513            input_ids,
514            &*cache,
515            xs.dtype(),
516            self.cfg.num_attn_heads,
517        )?;
518        // PagedAttention prompt chunking
519        let attention_mask = attention_mask.filter(|_| {
520            metadata
521                .as_ref()
522                .map(|(_, meta)| meta.is_first_prompt_chunk)
523                .unwrap_or(true)
524        });
525        let sliding_attention_mask = CausalMasker.make_sliding_window_causal_mask_matrix(
526            input_ids,
527            &*cache,
528            Some(self.sliding_window),
529            xs.dtype(),
530            self.cfg.num_attn_heads,
531        )?;
532        // PagedAttention prompt chunking
533        let sliding_attention_mask = sliding_attention_mask.filter(|_| {
534            metadata
535                .as_ref()
536                .map(|(_, meta)| meta.is_first_prompt_chunk)
537                .unwrap_or(true)
538        });
539        for (i, layer) in self.layers.iter().enumerate() {
540            xs = self.mapper.map(xs, i)?;
541            xs = layer.forward(
542                &xs,
543                attention_mask
544                    .as_ref()
545                    .map(|m| m.to_device(xs.device()).unwrap())
546                    .as_ref(),
547                sliding_attention_mask
548                    .as_ref()
549                    .map(|m| m.to_device(xs.device()).unwrap())
550                    .as_ref(),
551                seqlen_offsets,
552                &mut cache[i],
553                metadata
554                    .as_ref()
555                    .map(|(kv_cache, metadata)| (kv_cache[i].clone(), *metadata)),
556                flash_params,
557            )?;
558        }
559        let xs = xs.to_device(&self.device)?;
560        let mut xs = xs.apply(&self.norm)?;
561        if let Some(t) = self.lm_head.quantized_act_type() {
562            xs = xs.to_dtype(t)?;
563        }
564
565        let mut xs = MatMul.qmethod_matmul(&xs, &*self.lm_head)?;
566
567        if let Some(final_logit_softcapping) = self.final_logit_softcapping {
568            xs = (xs / final_logit_softcapping)?;
569            xs = xs.tanh()?;
570            xs = (xs * final_logit_softcapping)?;
571        }
572
573        extract_logits(&xs, context_lens)
574    }
575}

577impl IsqModel for Model {
578    fn get_layers(
579        &mut self,
580    ) -> (
581        Vec<(&mut Arc<dyn QuantMethod>, Option<usize>)>,
582        &dyn DeviceMapper,
583    ) {
584        let mut tensors = Vec::new();
585        tensors.push((&mut self.lm_head, None));
586        for (i, layer) in self.layers.iter_mut().enumerate() {
587            tensors.push((&mut layer.self_attn.q_proj, Some(i)));
588            tensors.push((&mut layer.self_attn.k_proj, Some(i)));
589            tensors.push((&mut layer.self_attn.v_proj, Some(i)));
590            tensors.push((&mut layer.self_attn.o_proj, Some(i)));
591            tensors.extend(
592                layer
593                    .mlp
594                    .get_isq_layers()
595                    .into_iter()
596                    .map(|m| (m, Some(i)))
597                    .collect::<Vec<_>>(),
598            );
599        }
600        (tensors, &*self.mapper)
601    }
602
603    fn residual_tensors(&self) -> Vec<(String, Tensor)> {
604        let uvb = UnVarBuilder::new();
605
606        let uvb_m = uvb.pp("model");
607        uvb_m.pp("embed_tokens").add(&self.embed_tokens);
608        uvb_m.pp("norm").add(&self.norm.undo_gemma().unwrap());
609
610        for (layer_idx, layer) in self.layers.iter().enumerate() {
611            let uvb_l = uvb_m.pp("layers").pp(layer_idx);
612            uvb_l
613                .pp("input_layernorm")
614                .add(&layer.input_layernorm.undo_gemma().unwrap());
615            uvb_l
616                .pp("post_attention_layernorm")
617                .add(&layer.post_attention_layernorm.undo_gemma().unwrap());
618            uvb_l
619                .pp("pre_feedforward_layernorm")
620                .add(&layer.pre_feedforward_layernorm.undo_gemma().unwrap());
621            uvb_l
622                .pp("post_feedforward_layernorm")
623                .add(&layer.post_feedforward_layernorm.undo_gemma().unwrap());
624        }
625
626        uvb.to_safetensors()
627    }
628
629    fn imatrix_names(&self) -> candle_core::Result<Vec<Option<String>>> {
630        // NOTE: dependant on the exact implementation in get_layers!
631        let mut names = Vec::new();
632        // lm_head
633        names.push(None);
634        for i in 0..self.layers.len() {
635            names.push(Some(format!("blk.{i}.attn_q.weight")));
636            names.push(Some(format!("blk.{i}.attn_k.weight")));
637            names.push(Some(format!("blk.{i}.attn_v.weight")));

700impl AnyMoeBaseModelMixin for Model {
701    fn get_mlps(&self) -> Vec<&dyn MlpLayer> {
702        let mut mlps = Vec::new();
703        for layer in &self.layers {
704            mlps.push(&*layer.mlp);
705        }
706        mlps
707    }
708    fn get_mlps_mut(&mut self) -> Vec<&mut Box<dyn MlpLayer>> {
709        let mut mlps = Vec::new();
710        for layer in &mut self.layers {
711            mlps.push(&mut layer.mlp);
712        }
713        mlps
714    }
715    fn create_anymoe_layers(
716        &mut self,
717        additional_vbs: Vec<ShardedVarBuilder>,
718        config: AnyMoeConfig,
719        (prefix, mlp): (String, String),
720        mut layers: Vec<usize>,
721        expert_type: AnyMoeExpertType,
722        gate_vb: Option<ShardedVarBuilder>,
723    ) -> Result<()> {
724        let mut experts: Vec<Vec<Box<dyn MlpLayer>>> = Vec::new();
725        if layers.is_empty() {
726            layers = (0..self.layers.len()).collect::<Vec<_>>();
727        }
728        for _ in 0..layers.len() {
729            experts.push(Vec::new());
730        }
731        for vb in additional_vbs {
732            let vb = vb.pp(&prefix);
733            for (layer, row) in experts.iter_mut().enumerate() {
734                if !layers.contains(&layer) {
735                    continue;
736                }
737
738                let intermediate_size = self.layers[layer].mlp.get_params()[1];
739                let hidden_size = self.layers[layer].mlp.get_params()[0];
740                match expert_type {
741                    AnyMoeExpertType::FineTuned => {
742                        let (dtype, device) = self.layers[layer].mlp.dtype_device();
743                        row.push(Box::new(Mlp::replicate(
744                            self.layers[layer].mlp.get_params(),
745                            vb.pp(layer).pp(&mlp).set_dtype(dtype).set_device(device),
746                            self.layers[layer].mlp.hidden_act(),
747                            &self.mapper.get_comm_for(layer)?,
748                        )?));
749                    }
750                    AnyMoeExpertType::LoraAdapter {
751                        rank,
752                        alpha,
753                        ref target_modules,
754                    } => {
755                        let vb_mlp = vb.pp(layer).pp(&mlp);
756
757                        let gate_proj_delta = if target_modules.contains(&"gate_proj".to_string()) {
758                            Some(get_delta_from_lora_ab!(
759                                vb_mlp,
760                                rank,
761                                alpha,
762                                (hidden_size, intermediate_size),
763                                "gate_proj"
764                            ))
765                        } else {
766                            None
767                        };
768                        let up_proj_delta = if target_modules.contains(&"up_proj".to_string()) {
769                            Some(get_delta_from_lora_ab!(
770                                vb_mlp,
771                                rank,
772                                alpha,
773                                (hidden_size, intermediate_size),
774                                "up_proj"
775                            ))
776                        } else {
777                            None
778                        };
779                        let down_proj_delta = if target_modules.contains(&"down_proj".to_string()) {
780                            Some(get_delta_from_lora_ab!(
781                                vb_mlp,
782                                rank,
783                                alpha,
784                                (intermediate_size, hidden_size),
785                                "down_proj"
786                            ))
787                        } else {
788                            None
789                        };
790
791                        row.push(self.layers[layer].mlp.new_added_delta(vec![
792                            gate_proj_delta,
793                            up_proj_delta,
794                            down_proj_delta,
795                        ])?);
796                    }
797                }
798            }
799        }
800        for (layer, expert) in layers.into_iter().zip(experts) {
801            let mut experts_all = vec![self.layers[layer].mlp.clone()];
802            experts_all.extend(expert);
803            let (dtype, device) = self.layers[layer].mlp.dtype_device();
804            self.layers[layer].mlp = Box::new(MoeMlp::new(
805                experts_all,
806                config.clone(),
807                dtype,
808                &device,
809                layer,
810                gate_vb.as_ref(),
811            )?);
812        }
813        Ok(())
814    }
815    fn amoe_supported(&self) -> bool {
816        true
817    }
