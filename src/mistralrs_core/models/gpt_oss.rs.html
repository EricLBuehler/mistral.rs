111/// Custom SwiGLU activation: (up + 1) * gate * sigmoid(gate * alpha)
112/// With clamping: gate max=limit, up [-limit, limit]
113#[allow(dead_code)]
114fn gptoss_swiglu(gate: &Tensor, up: &Tensor, alpha: f32, limit: f32) -> Result<Tensor> {
115    #[cfg(feature = "cuda")]
116    if gate.device().is_cuda() {
117        return mistralrs_quant::gptoss_swiglu_fused(gate, up, alpha, limit);
118    }
119
120    let dtype = gate.dtype();
121    let limit_d = limit as f64;
122
123    let gate_clamped =
124        gate.minimum(&Tensor::full(limit_d as f32, gate.shape(), gate.device())?.to_dtype(dtype)?)?;
125    let up_clamped = up.clamp(-limit_d, limit_d)?;
126
127    let gate_scaled = (&gate_clamped * alpha as f64)?;
128    let sigmoid_val = candle_nn::ops::sigmoid(&gate_scaled)?;
129    let glu = (&gate_clamped * &sigmoid_val)?;
130
131    let up_plus_one = (&up_clamped + 1.0)?;
132    up_plus_one.mul(&glu)
133}

153    #[allow(clippy::too_many_arguments)]
154    fn new(
155        rotary_emb: GptOssRotaryEmbeddingVariant,
156        cfg: &Config,
157        vb: ShardedVarBuilder,
158        mapper: &dyn DeviceMapper,
159        layer_idx: usize,
160        loading_isq: bool,
161        paged_attn: Option<PagedAttention>,
162        comm: &Arc<mistralrs_quant::Comm>,
163    ) -> Result<Self> {
164        let hidden_sz = cfg.hidden_size;
165        let num_heads = cfg.num_attention_heads;
166        let num_kv_heads = cfg.num_key_value_heads;
167        let head_dim = cfg.head_dim();
168
169        let q_proj = ColumnParallelLayer::new(
170            hidden_sz,
171            num_heads * head_dim,
172            &None,
173            cfg.attention_bias,
174            comm,
175            mapper.set_device(layer_idx, vb.pp("q_proj"), loading_isq),
176        )?;
177        let kv_shard = mistralrs_quant::compute_kv_shard(
178            cfg.num_key_value_heads,
179            cfg.hidden_size / cfg.num_attention_heads,
180            comm,
181        );
182        let k_proj = ColumnParallelLayer::new_with_shard(
183            hidden_sz,
184            num_kv_heads * head_dim,
185            &None,
186            cfg.attention_bias,
187            comm,
188            kv_shard,
189            mapper.set_device(layer_idx, vb.pp("k_proj"), loading_isq),
190        )?;
191        let v_proj = ColumnParallelLayer::new_with_shard(
192            hidden_sz,
193            num_kv_heads * head_dim,
194            &None,
195            cfg.attention_bias,
196            comm,
197            kv_shard,
198            mapper.set_device(layer_idx, vb.pp("v_proj"), loading_isq),
199        )?;
200        let o_proj = RowParallelLayer::new(
201            num_heads * head_dim,
202            hidden_sz,
203            &None,
204            cfg.attention_bias,
205            comm,
206            mapper.set_device(layer_idx, vb.pp("o_proj"), loading_isq),
207        )?;
208
209        let sinks = mapper
210            .set_device(layer_idx, vb.clone(), false)
211            .get((num_heads,), "sinks")?;
212
213        let is_sliding = matches!(
214            cfg.layer_types.get(layer_idx),
215            Some(LayerType::SlidingAttention)
216        );
217        let sliding_window = if is_sliding { cfg.sliding_window } else { None };
218
219        Ok(Self {
220            q_proj,
221            k_proj,
222            v_proj,
223            o_proj,
224            sinks,
225            num_heads: num_heads / comm.world_size(),
226            num_kv_heads: (num_kv_heads / comm.world_size()).max(1),
227            head_dim,
228            rotary_emb,
229            paged_attn,
230            sdpa_params: SdpaParams {
231                n_kv_groups: mistralrs_quant::compute_n_kv_groups(
232                    cfg.num_key_value_heads,
233                    cfg.num_attention_heads,
234                    comm,
235                ),
236                softcap: None,
237                softmax_scale: 1.0 / (head_dim as f32).sqrt(),
238                sliding_window,
239            },
240            is_sliding,
241        })
242    }

244    /// GPT-OSS specific attention forward with sinks
245    /// The sinks are added to attention logits before softmax, then dropped after
246    #[allow(clippy::too_many_arguments)]
247    fn forward(
248        &self,
249        xs: &Tensor,
250        attention_mask: Option<&Tensor>,
251        seqlen_offsets: &[usize],
252        kv_cache: &mut KvCache,
253        metadata: Option<((Tensor, Tensor), &PagedAttentionInputMetadata)>,
254        flash_params: &FlashParams,
255        layer_idx: usize,
256    ) -> Result<Tensor> {
257        let (b_sz, q_len, _) = xs.dims3()?;
258
259        let original_dtype = xs.dtype();
260        let mut xs = xs.clone();
261        if let Some(t) = self.q_proj.quantized_act_type() {
262            xs = xs.to_dtype(t)?;
263        }
264        let mut q = MatMul.qmethod_matmul(&xs, &*self.q_proj)?;
265        let mut k = MatMul.qmethod_matmul(&xs, &*self.k_proj)?;
266        let mut v = MatMul.qmethod_matmul(&xs, &*self.v_proj)?;
267        if self.q_proj.quantized_act_type().is_some() {
268            q = q.to_dtype(original_dtype)?;
269            k = k.to_dtype(original_dtype)?;
270            v = v.to_dtype(original_dtype)?;
271        }
272
273        (q, k, v) = if q_len != 1 {
274            let q = q
275                .reshape((b_sz, q_len, self.num_heads, self.head_dim))?
276                .transpose(1, 2)?;
277            let k = k
278                .reshape((b_sz, q_len, self.num_kv_heads, self.head_dim))?
279                .transpose(1, 2)?;
280            let v = v
281                .reshape((b_sz, q_len, self.num_kv_heads, self.head_dim))?
282                .transpose(1, 2)?;
283            (q, k, v)
284        } else {
285            let q = q.reshape((b_sz, self.num_heads, q_len, self.head_dim))?;
286            let k = k.reshape((b_sz, self.num_kv_heads, q_len, self.head_dim))?;
287            let v = v.reshape((b_sz, self.num_kv_heads, q_len, self.head_dim))?;
288            (q, k, v)
289        };
290
291        (q, k) = self.rotary_emb.forward(&q, &k, seqlen_offsets)?;
292        let mut attn_output = match &self.paged_attn {
293            Some(paged_attn) => match metadata {
294                Some(((key_cache, value_cache), input_metadata)) => paged_attn.forward(
295                    &q,
296                    &k,
297                    &v,
298                    attention_mask,
299                    Some(key_cache),
300                    Some(value_cache),
301                    input_metadata,
302                    &self.sdpa_params,
303                    Some(flash_params),
304                )?,
305                None => {
306                    let input_metadata = PagedAttentionInputMetadata::dummy(q.device())?;
307                    assert!(attention_mask.is_some());
308                    paged_attn.forward(
309                        &q,
310                        &k,
311                        &v,
312                        attention_mask,
313                        None,
314                        None,
315                        &input_metadata,
316                        &self.sdpa_params,
317                        Some(flash_params),
318                    )?
319                }
320            },
321            None => {
322                let (k, v) = kv_cache.append(&k, &v)?;
323
324                self.attention_with_sinks(
325                    &q,
326                    &k,
327                    &v,
328                    attention_mask,
329                    seqlen_offsets,
330                    flash_params,
331                    layer_idx,
332                )?
333            }
334        };
335
336        if let Some(t) = self.q_proj.quantized_act_type() {
337            attn_output = attn_output.to_dtype(t)?;
338        }
339        attn_output = if attention_mask.is_some() {
340            attn_output.transpose(1, 2)?.reshape((b_sz, q_len, ()))?
341        } else {
342            attn_output.reshape((b_sz, q_len, ()))?
343        };
344        let mut res = MatMul.qmethod_matmul(&attn_output, &*self.o_proj)?;
345        if self.q_proj.quantized_act_type().is_some() {
346            res = res.to_dtype(original_dtype)?;
347        }
348        Ok(res)
349    }

351    /// Attention with sinks: sinks are added as extra logits, softmaxed, then dropped
352    #[allow(clippy::too_many_arguments)]
353    fn attention_with_sinks(
354        &self,
355        q: &Tensor,
356        k: &Tensor,
357        v: &Tensor,
358        attention_mask: Option<&Tensor>,
359        _seqlen_offsets: &[usize],
360        _flash_params: &FlashParams,
361        _layer_idx: usize,
362    ) -> Result<Tensor> {
363        let (b_sz, num_heads, q_len, _head_dim) = q.dims4()?;
364        let (_, _, k_len, _) = k.dims4()?;
365
366        let k_expanded;
367        let v_expanded;
368        let (k_ref, v_ref): (&Tensor, &Tensor) = if self.sdpa_params.n_kv_groups > 1 {
369            k_expanded = crate::layers::repeat_kv(k.clone(), self.sdpa_params.n_kv_groups)?;
370            v_expanded = crate::layers::repeat_kv(v.clone(), self.sdpa_params.n_kv_groups)?;
371            (&k_expanded, &v_expanded)
372        } else {
373            (k, v)
374        };
375
376        let attn_weights = (q.matmul(&k_ref.transpose(D::Minus2, D::Minus1)?)?
377            * self.sdpa_params.softmax_scale as f64)?;
378
379        #[cfg(feature = "cuda")]
380        let scores = if let Some(mask) = attention_mask {
381            let mask_last_dim = mask.dim(D::Minus1)?;
382            let causal_mask = if mask_last_dim > k_len {
383                mask.narrow(D::Minus1, 0, k_len)?
384                    .to_dtype(attn_weights.dtype())?
385            } else {
386                mask.to_dtype(attn_weights.dtype())?
387            };
388            let attn_weights = attn_weights.broadcast_add(&causal_mask)?;
389
390            let sinks_expanded = self
391                .sinks
392                .reshape((1, num_heads, 1, 1))?
393                .broadcast_as((b_sz, num_heads, q_len, 1))?
394                .to_dtype(attn_weights.dtype())?;
395            let combined_logits = Tensor::cat(&[&attn_weights, &sinks_expanded], D::Minus1)?;
396
397            let max_logits = combined_logits.max_keepdim(D::Minus1)?;
398            let combined_logits = combined_logits.broadcast_sub(&max_logits)?;
399            let probs = candle_nn::ops::softmax_last_dim(&combined_logits)?;
400
401            probs.narrow(D::Minus1, 0, k_len)?.contiguous()?
402        } else {
403            let sinks = self.sinks.to_dtype(attn_weights.dtype())?;
404            mistralrs_quant::softmax_with_sinks(&attn_weights, &sinks, None)?
405        };
406
407        #[cfg(not(feature = "cuda"))]
408        let scores = {
409            let attn_weights = if let Some(mask) = attention_mask {
410                attn_weights.broadcast_add(mask)?
411            } else {
412                attn_weights
413            };
414
415            let sinks_expanded = self
416                .sinks
417                .reshape((1, num_heads, 1, 1))?
418                .broadcast_as((b_sz, num_heads, q_len, 1))?
419                .to_dtype(attn_weights.dtype())?;
420
421            let combined_logits = Tensor::cat(&[&attn_weights, &sinks_expanded], D::Minus1)?;
422
423            let max_logits = combined_logits.max_keepdim(D::Minus1)?;
424            let combined_logits = combined_logits.broadcast_sub(&max_logits)?;
425
426            let probs = candle_nn::ops::softmax_last_dim(&combined_logits)?;
427
428            probs.narrow(D::Minus1, 0, k_len)?.contiguous()?
429        };
430
431        scores.matmul(v_ref)
432    }
433}

435struct GptOssMoE {
436    gate: Linear,
437    gate_up_proj: Arc<dyn QuantMethod>,
438    down_proj: Arc<dyn QuantMethod>,
439    #[allow(dead_code)]
440    num_experts: usize,
441    num_experts_per_tok: usize,
442    intermediate_size: usize,
443    alpha: f32,
444    limit: f32,
445}
446
447impl GptOssMoE {
448    fn new(cfg: &Config, vb: ShardedVarBuilder, layer_device: Device) -> Result<Self> {
449        let gate = layers::linear(
450            cfg.hidden_size,
451            cfg.num_local_experts,
452            vb.pp("router").set_device(layer_device.clone()),
453        )?;
454
455        let experts_vb = vb.pp("experts").set_device(layer_device);
456
457        let gate_up_proj = MXFP4Layer::packed_gptoss_linear(
458            cfg.num_local_experts,
459            cfg.hidden_size,
460            cfg.intermediate_size * 2,
461            true,
462            "gate_up_proj",
463            experts_vb.clone(),
464        )?;
465
466        let down_proj = MXFP4Layer::packed_gptoss_linear(
467            cfg.num_local_experts,
468            cfg.intermediate_size,
469            cfg.hidden_size,
470            true,
471            "down_proj",
472            experts_vb,
473        )?;
474
475        Ok(Self {
476            gate,
477            gate_up_proj,
478            down_proj,
479            num_experts: cfg.num_local_experts,
480            num_experts_per_tok: cfg.num_experts_per_tok,
481            intermediate_size: cfg.intermediate_size,
482            alpha: cfg.alpha,
483            limit: cfg.swiglu_limit,
484        })
485    }
486
487    fn forward(&self, xs: &Tensor, _layer_idx: usize) -> Result<Tensor> {
488        let (b_size, seq_len, hidden_dim) = xs.dims3()?;
489        let xs_flat = xs.reshape(((), hidden_dim))?;
490
491        let router_logits = self.gate.forward(&xs_flat)?;
492
493        #[cfg(feature = "cuda")]
494        let (topk_weights, topk_ids) = {
495            let result = crate::ops::cuda_topk_softmax(&router_logits, self.num_experts_per_tok)?;
496            (result.values, result.indices)
497        };
498        #[cfg(not(feature = "cuda"))]
499        let (topk_weights, topk_ids) = {
500            use crate::ops::TopKLastDimOp;
501            let router_f32 = router_logits.to_dtype(DType::F32)?;
502            let topk_result = router_f32.topk(self.num_experts_per_tok)?;
503            let topk_weights = candle_nn::ops::softmax_last_dim(&topk_result.values)?;
504            (topk_weights, topk_result.indices)
505        };
506
507        let gate_up = self.gate_up_proj.gather_forward(&xs_flat, &topk_ids)?;
508        let (num_tokens, topk_dim, _) = gate_up.dims3()?;
509
510        #[cfg(feature = "cuda")]
511        let activated = {
512            let gate_up_for_kernel =
513                gate_up.reshape((num_tokens * topk_dim, self.intermediate_size, 2))?;
514            let result = mistralrs_quant::gptoss_swiglu_interleaved(
515                &gate_up_for_kernel,
516                self.intermediate_size,
517                self.alpha,
518                self.limit,
519            )?;
520            result.reshape((num_tokens, topk_dim, self.intermediate_size))?
521        };
522
523        #[cfg(not(feature = "cuda"))]
524        let activated = {
525            let gate_up_reshaped =
526                gate_up.reshape((num_tokens, topk_dim, self.intermediate_size, 2))?;
527            let gate = gate_up_reshaped
528                .narrow(D::Minus1, 0, 1)?
529                .squeeze(D::Minus1)?;
530            let up = gate_up_reshaped
531                .narrow(D::Minus1, 1, 1)?
532                .squeeze(D::Minus1)?;
533            gptoss_swiglu(&gate, &up, self.alpha, self.limit)?
534        };
535
536        let expert_out = self.down_proj.gather_forward(&activated, &topk_ids)?;
537
538        let topk_weights = topk_weights
539            .to_dtype(expert_out.dtype())?
540            .unsqueeze(D::Minus1)?;
541        let weighted = expert_out.broadcast_mul(&topk_weights)?;
542        let output = weighted.sum(D::Minus2)?;
543
544        output.reshape((b_size, seq_len, hidden_dim))
545    }
546}

555impl DecoderLayer {
556    #[allow(clippy::too_many_arguments)]
557    fn new(
558        rotary_emb: GptOssRotaryEmbeddingVariant,
559        cfg: &Config,
560        vb: ShardedVarBuilder,
561        mapper: &dyn DeviceMapper,
562        layer_idx: usize,
563        loading_isq: bool,
564        paged_attn: Option<PagedAttention>,
565        real_device: Device,
566        comm: &Arc<mistralrs_quant::Comm>,
567    ) -> Result<Self> {
568        let self_attn = Attention::new(
569            rotary_emb,
570            cfg,
571            mapper.set_device(layer_idx, vb.pp("self_attn"), loading_isq),
572            mapper,
573            layer_idx,
574            loading_isq,
575            paged_attn,
576            comm,
577        )?;
578
579        let mlp = GptOssMoE::new(
580            cfg,
581            mapper.set_device(layer_idx, vb.pp("mlp"), false),
582            real_device,
583        )?;
584
585        let input_layernorm = RmsNorm::new(
586            cfg.hidden_size,
587            cfg.rms_norm_eps,
588            mapper.set_device(layer_idx, vb.pp("input_layernorm"), false),
589        )?;
590        let post_attention_layernorm = RmsNorm::new(
591            cfg.hidden_size,
592            cfg.rms_norm_eps,
593            mapper.set_device(layer_idx, vb.pp("post_attention_layernorm"), false),
594        )?;
595
596        Ok(Self {
597            self_attn,
598            mlp,
599            input_layernorm,
600            post_attention_layernorm,
601        })
602    }
603
604    #[allow(clippy::too_many_arguments)]
605    fn forward(
606        &self,
607        xs: &Tensor,
608        attention_mask: Option<&Tensor>,
609        seqlen_offsets: &[usize],
610        kv_cache: &mut KvCache,
611        metadata: Option<((Tensor, Tensor), &PagedAttentionInputMetadata)>,
612        flash_params: &FlashParams,
613        layer_idx: usize,
614    ) -> Result<Tensor> {
615        let residual = xs;
616        let xs = self.input_layernorm.forward(xs)?;
617        let xs = self.self_attn.forward(
618            &xs,
619            attention_mask,
620            seqlen_offsets,
621            kv_cache,
622            metadata,
623            flash_params,
624            layer_idx,
625        )?;
626        let xs = (residual + xs)?;
627
628        let residual = &xs;
629        let xs = self.post_attention_layernorm.forward(&xs)?;
630        let xs = self.mlp.forward(&xs, layer_idx)?;
631
632        residual + xs
633    }
634}

636pub struct Model {
637    embed_tokens: candle_nn::Embedding,
638    layers: Vec<DecoderLayer>,
639    norm: RmsNorm,
640    lm_head: Arc<dyn QuantMethod>,
641    device: Device,
642    cache: EitherCache,
643    max_seq_len: usize,
644    mapper: Box<dyn DeviceMapper + Send + Sync>,
645    #[allow(dead_code)]
646    cfg: Config,
647    cfg_metadata: ModelConfigMetadata,
648}
649
650impl Model {
651    #[allow(clippy::too_many_arguments)]
652    pub fn new(
653        cfg: &Config,
654        vb: ShardedVarBuilder,
655        is_gptx: bool,
656        normal_loading_metadata: NormalLoadingMetadata,
657        attention_mechanism: AttentionImplementation,
658    ) -> Result<Self> {
659        let vb_m = vb.pp("model");
660        let mapper = normal_loading_metadata.mapper;
661
662        let embed_tokens = embedding(
663            cfg.vocab_size,
664            cfg.hidden_size,
665            mapper.set_nm_device(vb_m.pp("embed_tokens"), false),
666            &None,
667        )?;
668
669        let mut ropes: HashMap<_, GptOssRotaryEmbeddingVariant> = HashMap::new();
670        for layer_idx in 0..cfg.num_hidden_layers {
671            let device = mapper
672                .device_for(layer_idx, false)
673                .unwrap_or(&normal_loading_metadata.real_device);
674
675            let rope = if let Some(rope_scaling) = &cfg.rope_scaling {
676                if rope_scaling.rope_type == "yarn" {
677                    GptOssRotaryEmbeddingVariant::Yarn(Arc::new(GptOssRotaryEmbedding::new(
678                        cfg.rope_theta,
679                        cfg.head_dim(),
680                        cfg.max_position_embeddings,
681                        rope_scaling.factor,
682                        rope_scaling.original_max_position_embeddings,
683                        rope_scaling.beta_fast,
684                        rope_scaling.beta_slow,
685                        rope_scaling.truncate,
686                        device,
687                        vb_m.dtype(),
688                    )?))
689                } else {
690                    GptOssRotaryEmbeddingVariant::Standard(Arc::new(RotaryEmbedding::new(
691                        cfg.rope_theta as f32,
692                        cfg.head_dim(),
693                        cfg.max_position_embeddings,
694                        device,
695                        is_gptx,
696                        vb_m.dtype(),
697                    )?))
698                }
699            } else {
700                GptOssRotaryEmbeddingVariant::Standard(Arc::new(RotaryEmbedding::new(
701                    cfg.rope_theta as f32,
702                    cfg.head_dim(),
703                    cfg.max_position_embeddings,
704                    device,
705                    is_gptx,
706                    vb_m.dtype(),
707                )?))
708            };
709            ropes.insert(device.location(), rope);
710        }
711
712        let vb_l = vb_m.pp("layers");
713
714        let layers: Vec<DecoderLayer> = NiceProgressBar::<_, 'b'>(
715            0..cfg.num_hidden_layers,
716            "Loading repeating layers",
717            &normal_loading_metadata.multi_progress,
718        )
719        .into_iter()
720        .map(|layer_idx| {
721            let device = mapper
722                .device_for(layer_idx, false)
723                .unwrap_or(&normal_loading_metadata.real_device)
724                .clone();
725
726            let rotary_emb = ropes
727                .get(&device.location())
728                .cloned()
729                .expect("No RoPE for device");
730
731            let paged_attn = match &attention_mechanism {
732                AttentionImplementation::Eager => None,
733                AttentionImplementation::PagedAttention => {
734                    Some(PagedAttention::new(cfg.head_dim(), &device, None)?)
735                }
736            };
737
738            let comm = mapper.get_comm_for(layer_idx)?;
739
740            DecoderLayer::new(
741                rotary_emb,
742                cfg,
743                vb_l.pp(layer_idx),
744                &*mapper,
745                layer_idx,
746                normal_loading_metadata.loading_isq,
747                paged_attn,
748                device,
749                &comm,
750            )
751        })
752        .collect::<Result<Vec<_>>>()?;
753
754        let norm = RmsNorm::new(
755            cfg.hidden_size,
756            cfg.rms_norm_eps,
757            mapper.set_nm_device(vb_m.pp("norm"), false),
758        )?;
759
760        let lm_head = if !cfg.tie_word_embeddings {
761            ReplicatedLayer::new(
762                cfg.hidden_size,
763                cfg.vocab_size,
764                &None,
765                false,
766                mapper.set_nm_device(vb.pp("lm_head"), normal_loading_metadata.loading_isq),
767            )?
768        } else {
769            ReplicatedLayer::from_linear(candle_nn::Linear::new(
770                mapper.cast_nm_device(
771                    embed_tokens.embeddings(),
772                    normal_loading_metadata.loading_isq,
773                )?,
774                None,
775            ))?
776        };
777
778        let head_dim = cfg.head_dim();
779        let cfg_metadata = ModelConfigMetadata {
780            max_seq_len: cfg.max_position_embeddings,
781            num_layers: cfg.num_hidden_layers,
782            hidden_size: cfg.hidden_size,
783            num_kv_heads: cfg.num_key_value_heads,
784            num_attn_heads: cfg.num_attention_heads,
785            sliding_window: cfg.sliding_window,
786            k_head_dim: head_dim,
787            v_head_dim: head_dim,
788            kv_cache_layout: crate::paged_attention::KvCacheLayout::Standard,
789        };
790
791        let cache_types: Vec<NormalCacheType> = (0..cfg.num_hidden_layers)
792            .map(|layer_idx| match cfg.layer_types.get(layer_idx) {
793                Some(LayerType::SlidingAttention) => NormalCacheType::SlidingWindow {
794                    window: cfg.sliding_window.unwrap_or(cfg.max_position_embeddings),
795                },
796                _ => NormalCacheType::Normal {
797                    max_seq_len: cfg.max_position_embeddings,
798                },
799            })
800            .collect();
801
802        Ok(Self {
803            embed_tokens,
804            layers,
805            norm,
806            lm_head,
807            device: normal_loading_metadata.real_device,
808            cache: EitherCache::Normal(NormalCache::from_types(cache_types)),
809            max_seq_len: cfg.max_position_embeddings,
810            mapper,
811            cfg: cfg.clone(),
812            cfg_metadata,
813        })
814    }
815
816    #[allow(clippy::too_many_arguments)]
817    fn inner_forward(
818        &self,
819        input_ids: &Tensor,
820        seqlen_offsets: &[usize],
821        context_lens: Vec<(usize, usize)>,
822        metadata: Option<(Vec<(Tensor, Tensor)>, &PagedAttentionInputMetadata)>,
823        flash_params: &FlashParams,
824    ) -> Result<Tensor> {
825        let mut xs = self.embed_tokens.forward(input_ids)?;
826        let cache = &mut self.cache.normal().0;
827
828        let (_b_sz, tgt_len) = input_ids.dims2()?;
829        let sliding_window = self.cfg.sliding_window;
830
831        let (causal_mask, sliding_mask) = if tgt_len == 1 {
832            (None, None)
833        } else {
834            let past_kv_len = cache[0].current_seq_len();
835            let kv_len = tgt_len + past_kv_len;
836
837            let q_pos = Tensor::arange(
838                past_kv_len as u32,
839                (past_kv_len + tgt_len) as u32,
840                input_ids.device(),
841            )?
842            .to_dtype(DType::I64)?
843            .reshape((tgt_len, 1))?;
844            let k_pos = Tensor::arange(0u32, kv_len as u32, input_ids.device())?
845                .to_dtype(DType::I64)?
846                .reshape((1, kv_len))?;
847
848            let causal_mask_bool = k_pos.broadcast_gt(&q_pos)?;
849            let zeros = Tensor::zeros((tgt_len, kv_len), xs.dtype(), input_ids.device())?;
850            let neg_inf = Tensor::full(f32::NEG_INFINITY, (tgt_len, kv_len), input_ids.device())?
851                .to_dtype(xs.dtype())?;
852            let causal_mask = Some(
853                causal_mask_bool
854                    .where_cond(&neg_inf, &zeros)?
855                    .unsqueeze(0)?
856                    .unsqueeze(0)?,
857            );
858
859            let sliding_mask = if let Some(window) = sliding_window {
860                let window_tensor = Tensor::full(window as i64, q_pos.shape(), input_ids.device())?;
861                let min_k = q_pos.sub(&window_tensor)?.maximum(0i64)?;
862
863                let too_far_left = k_pos.broadcast_lt(&min_k)?;
864                let too_far_right = k_pos.broadcast_gt(&q_pos)?;
865                let sliding_mask_bool = too_far_left
866                    .to_dtype(DType::U8)?
867                    .add(&too_far_right.to_dtype(DType::U8)?)?
868                    .gt(0u8)?;
869
870                Some(
871                    sliding_mask_bool
872                        .where_cond(&neg_inf, &zeros)?
873                        .unsqueeze(0)?
874                        .unsqueeze(0)?,
875                )
876            } else {
877                None
878            };
879
880            (causal_mask, sliding_mask)
881        };
882
883        let should_use_mask = metadata
884            .as_ref()
885            .map(|(_, meta)| meta.is_first_prompt_chunk)
886            .unwrap_or(true);
887        let causal_mask = if should_use_mask { causal_mask } else { None };
888        let sliding_mask = if should_use_mask { sliding_mask } else { None };
889
890        for (i, layer) in self.layers.iter().enumerate() {
891            xs = self.mapper.map(xs, i)?;
892
893            let layer_mask = if layer.self_attn.is_sliding {
894                sliding_mask.as_ref().or(causal_mask.as_ref())
895            } else {
896                causal_mask.as_ref()
897            };
898
899            xs = layer.forward(
900                &xs,
901                layer_mask
902                    .map(|m| m.to_device(xs.device()).unwrap())
903                    .as_ref(),
904                seqlen_offsets,
905                &mut cache[i],
906                metadata.as_ref().map(|(kv, m)| (kv[i].clone(), *m)),
907                flash_params,
908                i,
909            )?;
910        }
911
912        xs = xs.to_device(&self.device)?;
913        xs = self.norm.forward(&xs)?;
914
915        if let Some(t) = self.lm_head.quantized_act_type() {
916            xs = xs.to_dtype(t)?;
917        }
918        xs = MatMul.qmethod_matmul(&xs, &*self.lm_head)?;
919        extract_logits(&xs, context_lens)
920    }
921}

924    fn get_layers(
925        &mut self,
926    ) -> (
927        Vec<(&mut Arc<dyn QuantMethod>, Option<usize>)>,
928        &dyn DeviceMapper,
929    ) {
930        let mut layers = Vec::new();
931        for (i, layer) in self.layers.iter_mut().enumerate() {
932            layers.push((&mut layer.self_attn.q_proj, Some(i)));
933            layers.push((&mut layer.self_attn.k_proj, Some(i)));
934            layers.push((&mut layer.self_attn.v_proj, Some(i)));
935            layers.push((&mut layer.self_attn.o_proj, Some(i)));
936        }
937        layers.push((&mut self.lm_head, None));
938        (layers, &*self.mapper)
939    }
