96struct Attention {
97    q_proj: Arc<dyn QuantMethod>,
98    k_proj: Arc<dyn QuantMethod>,
99    v_proj: Arc<dyn QuantMethod>,
100    o_proj: Arc<dyn QuantMethod>,
101    num_heads: usize,
102    num_kv_heads: usize,
103    head_dim: usize,
104    rotary_emb: Arc<RotaryEmbedding>,
105    paged_attn: Option<PagedAttention>,
106    sdpa_params: SdpaParams,
107}

109impl Attention {
110    fn new(
111        rotary_emb: Arc<RotaryEmbedding>,
112        cfg: &Config,
113        vb: ShardedVarBuilder,
114        paged_attn: Option<PagedAttention>,
115        comm: &Arc<mistralrs_quant::Comm>,
116    ) -> Result<Self> {
117        let hidden_sz = cfg.hidden_size;
118        let num_heads = cfg.num_attention_heads;
119        let num_kv_heads = cfg.num_key_value_heads;
120        let head_dim = cfg.head_dim();
121        let q_proj = ColumnParallelLayer::new(
122            hidden_sz,
123            num_heads * head_dim,
124            &cfg.quantization_config,
125            false,
126            comm,
127            vb.pp("q_proj"),
128        )?;
129        let kv_shard = mistralrs_quant::compute_kv_shard(
130            cfg.num_key_value_heads,
131            cfg.hidden_size / cfg.num_attention_heads,
132            comm,
133        );
134        let k_proj = ColumnParallelLayer::new_with_shard(
135            hidden_sz,
136            num_kv_heads * head_dim,
137            &cfg.quantization_config,
138            false,
139            comm,
140            kv_shard,
141            vb.pp("k_proj"),
142        )?;
143        let v_proj = ColumnParallelLayer::new_with_shard(
144            hidden_sz,
145            num_kv_heads * head_dim,
146            &cfg.quantization_config,
147            false,
148            comm,
149            kv_shard,
150            vb.pp("v_proj"),
151        )?;
152        let o_proj = RowParallelLayer::new(
153            num_heads * head_dim,
154            hidden_sz,
155            &cfg.quantization_config,
156            false,
157            comm,
158            vb.pp("o_proj"),
159        )?;
160        Ok(Self {
161            q_proj,
162            k_proj,
163            v_proj,
164            o_proj,
165            num_heads: num_heads / comm.world_size(),
166            num_kv_heads: (num_kv_heads / comm.world_size()).max(1),
167            head_dim,
168            rotary_emb,
169            paged_attn,
170            sdpa_params: SdpaParams {
171                n_kv_groups: mistralrs_quant::compute_n_kv_groups(
172                    cfg.num_key_value_heads,
173                    cfg.num_attention_heads,
174                    comm,
175                ),
176                softcap: None,
177                softmax_scale: 1.0 / (head_dim as f32).sqrt(),
178                sliding_window: cfg.sliding_window,
179            },
180        })
181    }
182
183    #[allow(clippy::too_many_arguments)]
184    fn forward(
185        &self,
186        xs: &Tensor,
187        attention_mask: Option<&Tensor>,
188        seqlen_offsets: &[usize],
189        kv_cache: &mut KvCache,
190        metadata: Option<((Tensor, Tensor), &PagedAttentionInputMetadata)>,
191        flash_params: &FlashParams,
192    ) -> Result<Tensor> {
193        let (b_sz, q_len, _) = xs.dims3()?;
194
195        let original_dtype = xs.dtype();
196        let mut xs = xs.clone();
197        if let Some(t) = self.q_proj.quantized_act_type() {
198            xs = xs.to_dtype(t)?;
199        }
200        let mut q = MatMul.qmethod_matmul(&xs, &*self.q_proj)?;
201        let mut k = MatMul.qmethod_matmul(&xs, &*self.k_proj)?;
202        let mut v = MatMul.qmethod_matmul(&xs, &*self.v_proj)?;
203        if self.q_proj.quantized_act_type().is_some() {
204            q = q.to_dtype(original_dtype)?;
205            k = k.to_dtype(original_dtype)?;
206            v = v.to_dtype(original_dtype)?;
207        }
208
209        let (q, k, v) = if q_len != 1 {
210            let q = q
211                .reshape((b_sz, q_len, self.num_heads, self.head_dim))?
212                .transpose(1, 2)?;
213            let k = k
214                .reshape((b_sz, q_len, self.num_kv_heads, self.head_dim))?
215                .transpose(1, 2)?;
216            let v = v
217                .reshape((b_sz, q_len, self.num_kv_heads, self.head_dim))?
218                .transpose(1, 2)?;
219            (q, k, v)
220        } else {
221            let q = q.reshape((b_sz, self.num_heads, q_len, self.head_dim))?;
222            let k = k.reshape((b_sz, self.num_kv_heads, q_len, self.head_dim))?;
223            let v = v.reshape((b_sz, self.num_kv_heads, q_len, self.head_dim))?;
224            (q, k, v)
225        };
226
227        let (q, k) = self.rotary_emb.forward(&q, &k, seqlen_offsets)?;
228
229        let mut attn_output = match &self.paged_attn {
230            Some(paged_attn) => match metadata {
231                Some(((key_cache, value_cache), input_metadata)) => paged_attn.forward(
232                    &q,
233                    &k,
234                    &v,
235                    attention_mask,
236                    Some(key_cache),
237                    Some(value_cache),
238                    input_metadata,
239                    &self.sdpa_params,
240                    Some(flash_params),
241                )?,
242                None => {
243                    // If we don't have metadata, we are most likely generating an imatrix so we don't want to populate that.
244                    // Generating the dummy metadata with the assumption that we are not generating text (only processing prompts).
245                    let input_metadata = PagedAttentionInputMetadata::dummy(q.device())?;
246                    // Sanity check.
247                    assert!(attention_mask.is_some());
248                    paged_attn.forward(
249                        &q,
250                        &k,
251                        &v,
252                        attention_mask,
253                        None,
254                        None,
255                        &input_metadata,
256                        &self.sdpa_params,
257                        Some(flash_params),
258                    )?
259                }
260            },
261            None => {
262                let (k, v) = kv_cache.append(&k, &v)?;
263
264                Sdpa.run_attention(
265                    &q,
266                    &k,
267                    &v,
268                    attention_mask,
269                    Some(flash_params),
270                    &self.sdpa_params,
271                )?
272            }
273        };
274
275        if let Some(t) = self.q_proj.quantized_act_type() {
276            attn_output = attn_output.to_dtype(t)?;
277        }
278        attn_output = if attention_mask.is_some() {
279            attn_output.transpose(1, 2)?.reshape((b_sz, q_len, ()))?
280        } else {
281            attn_output.reshape((b_sz, q_len, ()))?
282        };
283        let mut res = MatMul.qmethod_matmul(&attn_output, &*self.o_proj)?;
284        if self.q_proj.quantized_act_type().is_some() {
285            res = res.to_dtype(original_dtype)?;
286        }
287        Ok(res)
288    }
289}

291struct DecoderLayer {
292    self_attn: Attention,
293    mlp: Box<dyn MlpLayer>,
294    input_layernorm: RmsNorm,
295    post_attention_layernorm: RmsNorm,
296}
297
298impl DecoderLayer {
299    #[allow(clippy::too_many_arguments)]
300    fn new(
301        rotary_emb: Arc<RotaryEmbedding>,
302        cfg: &Config,
303        vb: ShardedVarBuilder,
304        mapper: &dyn DeviceMapper,
305        layer_idx: usize,
306        loading_isq: bool,
307        paged_attn: Option<PagedAttention>,
308        comm: &Arc<mistralrs_quant::Comm>,
309    ) -> Result<Self> {
310        let self_attn = Attention::new(
311            rotary_emb,
312            cfg,
313            mapper.set_device(layer_idx, vb.pp("self_attn"), loading_isq),
314            paged_attn,
315            comm,
316        )?;
317        let mlp = Mlp::new(
318            mapper.set_device(layer_idx, vb.pp("mlp"), loading_isq),
319            cfg.hidden_size,
320            cfg.intermediate_size,
321            &cfg.quantization_config,
322            cfg.hidden_act,
323            comm,
324        )?;
325        let input_layernorm = RmsNorm::new(
326            cfg.hidden_size,
327            cfg.rms_norm_eps,
328            mapper.set_device(layer_idx, vb.pp("input_layernorm"), false),
329        )?;
330        let post_attention_layernorm = RmsNorm::new(
331            cfg.hidden_size,
332            cfg.rms_norm_eps,
333            mapper.set_device(layer_idx, vb.pp("post_attention_layernorm"), false),
334        )?;
335        Ok(Self {
336            self_attn,
337            mlp: Box::new(mlp),
338            input_layernorm,
339            post_attention_layernorm,
340        })
341    }
342
343    #[allow(clippy::too_many_arguments)]
344    fn forward(
345        &self,
346        xs: &Tensor,
347        attention_mask: Option<&Tensor>,
348        seqlen_offsets: &[usize],
349        kv_cache: &mut KvCache,
350        metadata: Option<((Tensor, Tensor), &PagedAttentionInputMetadata)>,
351        flash_params: &FlashParams,
352    ) -> Result<Tensor> {
353        let residual = xs;
354        let xs = self.input_layernorm.forward(xs)?;
355        let xs = self.self_attn.forward(
356            &xs,
357            attention_mask,
358            seqlen_offsets,
359            kv_cache,
360            metadata,
361            flash_params,
362        )?;
363        let xs = (xs + residual)?;
364        let residual = &xs;
365        let xs = self
366            .mlp
367            .forward(&xs.apply(&self.post_attention_layernorm)?)?;
368        residual + xs
369    }
370}

385impl Model {
386    pub fn new(
387        cfg: &Config,
388        vb: ShardedVarBuilder,
389        is_gptx: bool,
390        normal_loading_metadata: NormalLoadingMetadata,
391        attention_mechanism: AttentionImplementation,
392    ) -> Result<Self> {
393        let vb_m = vb.pp("model");
394        let vb_lm_head = vb.pp("lm_head");
395        Self::new_inner(
396            cfg,
397            vb_m,
398            vb_lm_head,
399            is_gptx,
400            normal_loading_metadata,
401            attention_mechanism,

404
405    pub fn new_inner(
406        cfg: &Config,
407        vb_m: ShardedVarBuilder,
408        vb_lm_head: ShardedVarBuilder,
409        is_gptx: bool,
410        normal_loading_metadata: NormalLoadingMetadata,
411        attention_mechanism: AttentionImplementation,
412    ) -> Result<Self> {
413        if let Some(ref quant_cfg) = &cfg.quantization_config {
414            tracing::info!(
415                "Using {} quantization: {}.",
416                quant_cfg.name(),
417                quant_cfg.get_bits_name(&vb_m)
418            );
419        }
420        let mapper = normal_loading_metadata.mapper;
421
422        let embed_tokens = embedding(
423            cfg.vocab_size,
424            cfg.hidden_size,
425            mapper.set_nm_device(vb_m.pp("embed_tokens"), false),
426            &cfg.quantization_config,
427        )?;
428
429        let head_dim = cfg.head_dim();
430        let mut ropes = HashMap::new();
431        for layer_idx in 0..cfg.num_hidden_layers {
432            let device = mapper
433                .device_for(layer_idx, false)
434                .unwrap_or(&normal_loading_metadata.real_device);
435            ropes.insert(
436                device.location(),
437                Arc::new(RotaryEmbedding::new(
438                    cfg.get_rope_theta() as f32,
439                    head_dim,
440                    cfg.max_position_embeddings,
441                    device,
442                    is_gptx,
443                    vb_m.dtype(),
444                )?),
445            );
446        }
447
448        let vb_l = vb_m.pp("layers");
449        let layers: Vec<DecoderLayer> = NiceProgressBar::<_, 'b'>(
450            0..cfg.num_hidden_layers,
451            "Loading repeating layers",
452            &normal_loading_metadata.multi_progress,
453        )
454        .par_iter_if_isq(|layer_idx| {
455            let device = mapper
456                .device_for(layer_idx, false)
457                .unwrap_or(&normal_loading_metadata.real_device);
458            let rotary_emb = ropes
459                .get(&device.location())
460                .expect("No RoPE for device location!")
461                .clone();
462            let paged_attn = match &attention_mechanism {
463                AttentionImplementation::Eager => None,
464                AttentionImplementation::PagedAttention => {
465                    Some(PagedAttention::new(head_dim, device, None)?)
466                }
467            };
468            let comm = mapper.get_comm_for(layer_idx)?;
469            DecoderLayer::new(
470                rotary_emb.clone(),
471                cfg,
472                vb_l.pp(layer_idx),
473                &*mapper,
474                layer_idx,
475                normal_loading_metadata.loading_isq,
476                paged_attn,
477                &comm,
478            )
479        })?;
480        let norm = RmsNorm::new(
481            cfg.hidden_size,
482            cfg.rms_norm_eps,
483            mapper.set_nm_device(vb_m.pp("norm"), false),
484        )?;
485        let lm_head = if !cfg.tie_word_embeddings {
486            ReplicatedLayer::new(
487                cfg.hidden_size,
488                cfg.vocab_size,
489                &cfg.quantization_config,
490                false,
491                mapper.set_nm_device(vb_lm_head, normal_loading_metadata.loading_isq),
492            )?
493        } else {
494            ReplicatedLayer::from_linear(candle_nn::Linear::new(
495                mapper.cast_nm_device(
496                    embed_tokens.embeddings(),
497                    normal_loading_metadata.loading_isq,
498                )?,
499                None,
500            ))?
501        };
502        Ok(Self {
503            embed_tokens,
504            layers,
505            norm,
506            lm_head,
507            sliding_window: cfg.sliding_window,
508            device: normal_loading_metadata.real_device,
509            cache: EitherCache::Normal(NormalCache::new_sliding(
510                cfg.num_hidden_layers,
511                cfg.max_position_embeddings,
512                cfg.sliding_window,
513            )),
514            max_seq_len: cfg.max_position_embeddings,
515            cfg: ModelConfigMetadata {
516                max_seq_len: cfg.max_position_embeddings,
517                num_layers: cfg.num_hidden_layers,
518                hidden_size: cfg.hidden_size,
519                num_kv_heads: (cfg.num_key_value_heads / mapper.get_comm_for(0)?.world_size())
520                    .max(1),
521                num_attn_heads: cfg.num_attention_heads / mapper.get_comm_for(0)?.world_size(),
522                sliding_window: cfg.sliding_window,
523                k_head_dim: cfg.head_dim(),
524                v_head_dim: cfg.head_dim(),
525                kv_cache_layout: crate::paged_attention::KvCacheLayout::Standard,
526            },
527            mapper,
528        })
529    }

531    pub fn get_input_embeddings(&self, input_ids: &Tensor) -> Result<Tensor> {
532        self.embed_tokens.forward(input_ids)
533    }

535    pub fn forward(
536        &self,
537        input_ids: &Tensor,
538        seqlen_offsets: &[usize],
539        context_lens: Vec<(usize, usize)>,
540        metadata: Option<(Vec<(Tensor, Tensor)>, &PagedAttentionInputMetadata)>,
541        flash_params: &FlashParams,
542    ) -> Result<Tensor> {
543        self.forward_embeds(
544            input_ids,
545            self.embed_tokens.forward(input_ids)?,
546            seqlen_offsets,
547            context_lens,
548            metadata,
549            flash_params,
550        )
551    }
552
553    #[allow(clippy::too_many_arguments)]
554    pub fn forward_embeds(
555        &self,
556        input_ids: &Tensor,
557        input_embeds: Tensor,
558        seqlen_offsets: &[usize],
559        context_lens: Vec<(usize, usize)>,
560        metadata: Option<(Vec<(Tensor, Tensor)>, &PagedAttentionInputMetadata)>,
561        flash_params: &FlashParams,
562    ) -> Result<Tensor> {
563        let mut xs = input_embeds;
564        let cache = &mut self.cache.normal().0;
565        let attention_mask = CausalMasker.make_sliding_window_causal_mask_matrix(
566            input_ids,
567            metadata
568                .as_ref()
569                .map(|(_, _)| &seqlen_offsets as &dyn PastKvLenCache)
570                .unwrap_or(cache as &dyn PastKvLenCache),
571            self.sliding_window,
572            xs.dtype(),
573            self.cfg.num_attn_heads,
574        )?;
575        // PagedAttention prompt chunking
576        let attention_mask = attention_mask.filter(|_| {
577            metadata
578                .as_ref()
579                .map(|(_, meta)| meta.is_first_prompt_chunk)
580                .unwrap_or(true)
581        });
582        for (i, layer) in self.layers.iter().enumerate() {
583            xs = self.mapper.map(xs, i)?;
584            xs = layer.forward(
585                &xs,
586                attention_mask
587                    .as_ref()
588                    .map(|m| m.to_device(xs.device()).unwrap())
589                    .as_ref(),
590                seqlen_offsets,
591                &mut cache[i],
592                metadata
593                    .as_ref()
594                    .map(|(kv_cache, metadata)| (kv_cache[i].clone(), *metadata)),
595                flash_params,
596            )?;
597        }
598        let xs = xs.to_device(&self.device)?;
599        let mut xs = xs.apply(&self.norm)?;
600        if let Some(t) = self.lm_head.quantized_act_type() {
601            xs = xs.to_dtype(t)?;
602        }
603        extract_logits(&MatMul.qmethod_matmul(&xs, &*self.lm_head)?, context_lens)
604    }
605}

607impl IsqModel for Model {
608    fn get_layers(
609        &mut self,
610    ) -> (
611        Vec<(&mut Arc<dyn QuantMethod>, Option<usize>)>,
612        &dyn DeviceMapper,
613    ) {
614        let mut tensors = Vec::new();
615        tensors.push((&mut self.lm_head, None));
616        for (i, layer) in self.layers.iter_mut().enumerate() {
617            tensors.push((&mut layer.self_attn.q_proj, Some(i)));
618            tensors.push((&mut layer.self_attn.k_proj, Some(i)));
619            tensors.push((&mut layer.self_attn.v_proj, Some(i)));
620            tensors.push((&mut layer.self_attn.o_proj, Some(i)));
621            tensors.extend(
622                layer
623                    .mlp
624                    .get_isq_layers()
625                    .into_iter()
626                    .map(|m| (m, Some(i)))
627                    .collect::<Vec<_>>(),
628            );
629        }
630        (tensors, &*self.mapper)
631    }
632
633    fn residual_tensors(&self) -> Vec<(String, Tensor)> {
634        let uvb = UnVarBuilder::new();
635
636        let uvb_m = uvb.pp("model");
637        uvb_m.pp("embed_tokens").add(&self.embed_tokens);
638        uvb_m.pp("norm").add(&self.norm);
639
640        for (layer_idx, layer) in self.layers.iter().enumerate() {
641            let uvb_l = uvb_m.pp("layers").pp(layer_idx);
642            uvb_l.pp("input_layernorm").add(&layer.input_layernorm);
643            uvb_l
644                .pp("post_attention_layernorm")
645                .add(&layer.post_attention_layernorm);
646        }
647
648        uvb.to_safetensors()
649    }

651    fn imatrix_names(&self) -> candle_core::Result<Vec<Option<String>>> {
652        // NOTE: dependant on the exact implementation in get_layers!
653        let mut names = Vec::new();
654        // lm_head
655        names.push(None);
656        for i in 0..self.layers.len() {
657            names.push(Some(format!("blk.{i}.attn_q.weight")));
658            names.push(Some(format!("blk.{i}.attn_k.weight")));
659            names.push(Some(format!("blk.{i}.attn_v.weight")));
660            names.push(Some(format!("blk.{i}.attn_output.weight")));
661            names.push(Some(format!("blk.{i}.ffn_gate.weight")));
662            names.push(Some(format!("blk.{i}.ffn_up.weight")));
663            names.push(Some(format!("blk.{i}.ffn_down.weight")));
664        }
665        Ok(names)
666    }

722impl AnyMoeBaseModelMixin for Model {
723    fn get_mlps(&self) -> Vec<&dyn MlpLayer> {
724        let mut mlps = Vec::new();
725        for layer in &self.layers {
726            mlps.push(&*layer.mlp);
727        }
728        mlps
729    }
730    fn get_mlps_mut(&mut self) -> Vec<&mut Box<dyn MlpLayer>> {
731        let mut mlps = Vec::new();
732        for layer in &mut self.layers {
733            mlps.push(&mut layer.mlp);
734        }
735        mlps
736    }
737    fn create_anymoe_layers(
738        &mut self,
739        additional_vbs: Vec<ShardedVarBuilder>,
740        config: AnyMoeConfig,
741        (prefix, mlp): (String, String),
742        mut layers: Vec<usize>,
743        expert_type: AnyMoeExpertType,
744        gate_vb: Option<ShardedVarBuilder>,
745    ) -> Result<()> {
746        let mut experts: Vec<Vec<Box<dyn MlpLayer>>> = Vec::new();
747        if layers.is_empty() {
748            layers = (0..self.layers.len()).collect::<Vec<_>>();
749        }
750        for _ in 0..layers.len() {
751            experts.push(Vec::new());
752        }
753        for vb in additional_vbs {
754            let vb = vb.pp(&prefix);
755            for (layer, row) in experts.iter_mut().enumerate() {
756                if !layers.contains(&layer) {
757                    continue;
758                }
759
760                let intermediate_size = self.layers[layer].mlp.get_params()[1];
761                let hidden_size = self.layers[layer].mlp.get_params()[0];
762                match expert_type {
763                    AnyMoeExpertType::FineTuned => {
764                        let (dtype, device) = self.layers[layer].mlp.dtype_device();
765                        row.push(Box::new(Mlp::replicate(
766                            self.layers[layer].mlp.get_params(),
767                            vb.pp(layer).pp(&mlp).set_dtype(dtype).set_device(device),
768                            self.layers[layer].mlp.hidden_act(),
769                            &self.mapper.get_comm_for(layer)?,
770                        )?));
771                    }
772                    AnyMoeExpertType::LoraAdapter {
773                        rank,
774                        alpha,
775                        ref target_modules,
776                    } => {
777                        let vb_mlp = vb.pp(layer).pp(&mlp);
778
779                        let gate_proj_delta = if target_modules.contains(&"gate_proj".to_string()) {
780                            Some(get_delta_from_lora_ab!(
781                                vb_mlp,
782                                rank,
783                                alpha,
784                                (hidden_size, intermediate_size),
785                                "gate_proj"
786                            ))
787                        } else {
788                            None
789                        };
790                        let up_proj_delta = if target_modules.contains(&"up_proj".to_string()) {
791                            Some(get_delta_from_lora_ab!(
792                                vb_mlp,
793                                rank,
794                                alpha,
795                                (hidden_size, intermediate_size),
796                                "up_proj"
797                            ))
798                        } else {
799                            None
800                        };
801                        let down_proj_delta = if target_modules.contains(&"down_proj".to_string()) {
802                            Some(get_delta_from_lora_ab!(
803                                vb_mlp,
804                                rank,
805                                alpha,
806                                (intermediate_size, hidden_size),
807                                "down_proj"
808                            ))
809                        } else {
810                            None
811                        };
812
813                        row.push(self.layers[layer].mlp.new_added_delta(vec![
814                            gate_proj_delta,
815                            up_proj_delta,
816                            down_proj_delta,
817                        ])?);
818                    }
819                }
820            }
821        }
822        for (layer, expert) in layers.into_iter().zip(experts) {
823            let mut experts_all = vec![self.layers[layer].mlp.clone()];
824            experts_all.extend(expert);
825            let (dtype, device) = self.layers[layer].mlp.dtype_device();
826            self.layers[layer].mlp = Box::new(MoeMlp::new(
827                experts_all,
828                config.clone(),
829                dtype,
830                &device,
831                layer,
832                gate_vb.as_ref(),
833            )?);
834        }
835        Ok(())
836    }
837    fn amoe_supported(&self) -> bool {
838        true
839    }
