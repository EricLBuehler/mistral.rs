68struct CausalSelfAttention {
69    q_proj: Arc<dyn QuantMethod>,
70    k_proj: Arc<dyn QuantMethod>,
71    v_proj: Arc<dyn QuantMethod>,
72    o_proj: Arc<dyn QuantMethod>,
73    num_attention_heads: usize,
74    num_key_value_heads: usize,
75    head_dim: usize,
76    rotary_emb: Option<Arc<SmolLm3RotaryEmbedding>>,
77    max_seq_len: usize,
78    paged_attn: Option<PagedAttention>,
79    sdpa_params: SdpaParams,
80}

82impl CausalSelfAttention {
83    #[allow(clippy::too_many_arguments)]
84    fn forward(
85        &self,
86        x: &Tensor,
87        attention_mask: &Option<Tensor>,
88        seqlen_offsets: &[usize],
89        kv_cache: &mut KvCache,
90        metadata: Option<((Tensor, Tensor), &PagedAttentionInputMetadata)>,
91        flash_params: &FlashParams,
92    ) -> Result<Tensor> {
93        let (b_sz, seq_len, _) = x.dims3()?;
94
95        let original_dtype = x.dtype();
96        let mut x = x.clone();
97        if let Some(t) = self.q_proj.quantized_act_type() {
98            x = x.to_dtype(t)?;
99        }
100        let mut q = MatMul.qmethod_matmul(&x, &*self.q_proj)?;
101        let mut k = MatMul.qmethod_matmul(&x, &*self.k_proj)?;
102        let mut v = MatMul.qmethod_matmul(&x, &*self.v_proj)?;
103        if self.q_proj.quantized_act_type().is_some() {
104            q = q.to_dtype(original_dtype)?;
105            k = k.to_dtype(original_dtype)?;
106            v = v.to_dtype(original_dtype)?;
107        }
108
109        (q, k, v) = if seq_len != 1 {
110            let q = q
111                .reshape((b_sz, seq_len, self.num_attention_heads, self.head_dim))?
112                .transpose(1, 2)?;
113            let k = k
114                .reshape((b_sz, seq_len, self.num_key_value_heads, self.head_dim))?
115                .transpose(1, 2)?;
116            let v = v
117                .reshape((b_sz, seq_len, self.num_key_value_heads, self.head_dim))?
118                .transpose(1, 2)?;
119            (q, k, v)
120        } else {
121            let q = q.reshape((b_sz, self.num_attention_heads, seq_len, self.head_dim))?;
122            let k = k.reshape((b_sz, self.num_key_value_heads, seq_len, self.head_dim))?;
123            let v = v.reshape((b_sz, self.num_key_value_heads, seq_len, self.head_dim))?;
124            (q, k, v)
125        };
126
127        if let Some(rotary_emb) = &self.rotary_emb {
128            (q, k) = rotary_emb.forward(&q, &k, seqlen_offsets)?;
129        }
130
131        let mut y = match &self.paged_attn {
132            Some(paged_attn) => match metadata {
133                Some(((key_cache, value_cache), input_metadata)) => paged_attn.forward(
134                    &q,
135                    &k,
136                    &v,
137                    attention_mask.clone().as_ref(),
138                    Some(key_cache),
139                    Some(value_cache),
140                    input_metadata,
141                    &self.sdpa_params,
142                    Some(flash_params),
143                )?,
144                None => {
145                    // If we don't have metadata, we are most likely generating an imatrix so we don't want to populate that.
146                    // Generating the dummy metadata with the assumption that we are not generating text (only processing prompts).
147                    let input_metadata = PagedAttentionInputMetadata::dummy(q.device())?;
148                    // Sanity check.
149                    assert!(attention_mask.is_some());
150                    paged_attn.forward(
151                        &q,
152                        &k,
153                        &v,
154                        attention_mask.clone().as_ref(),
155                        None,
156                        None,
157                        &input_metadata,
158                        &self.sdpa_params,
159                        Some(flash_params),
160                    )?
161                }
162            },
163            None => {
164                let (k, v) = kv_cache.append(&k, &v)?;
165
166                Sdpa.run_attention(
167                    &q,
168                    &k,
169                    &v,
170                    attention_mask.clone().as_ref(),
171                    Some(flash_params),
172                    &self.sdpa_params,
173                )?
174            }
175        };
176
177        if let Some(t) = self.q_proj.quantized_act_type() {
178            y = y.to_dtype(t)?;
179        }
180        y = if attention_mask.is_some() {
181            y.transpose(1, 2)?.reshape((b_sz, seq_len, ()))?
182        } else {
183            y.reshape((b_sz, seq_len, ()))?
184        };
185        let mut res = MatMul.qmethod_matmul(&y, &*self.o_proj)?;
186        if self.q_proj.quantized_act_type().is_some() {
187            res = res.to_dtype(original_dtype)?;
188        }
189        Ok(res)
190    }
191
192    fn load(
193        vb: ShardedVarBuilder,
194        cfg: &Config,
195        rope: Option<Arc<SmolLm3RotaryEmbedding>>,
196        paged_attn: Option<PagedAttention>,
197        comm: &Arc<mistralrs_quant::Comm>,
198    ) -> Result<Self> {
199        let size_in = cfg.hidden_size;
200        let size_q = (cfg.hidden_size / cfg.num_attention_heads) * cfg.num_attention_heads;
201        let size_kv = (cfg.hidden_size / cfg.num_attention_heads) * cfg.num_key_value_heads;
202        let q_proj = ColumnParallelLayer::new(
203            size_in,
204            size_q,
205            &cfg.quantization_config,
206            false,
207            comm,
208            vb.pp("q_proj"),
209        )?;
210        let kv_shard = mistralrs_quant::compute_kv_shard(
211            cfg.num_key_value_heads,
212            cfg.hidden_size / cfg.num_attention_heads,
213            comm,
214        );
215        let k_proj = ColumnParallelLayer::new_with_shard(
216            size_in,
217            size_kv,
218            &cfg.quantization_config,
219            false,
220            comm,
221            kv_shard,
222            vb.pp("k_proj"),
223        )?;
224        let v_proj = ColumnParallelLayer::new_with_shard(
225            size_in,
226            size_kv,
227            &cfg.quantization_config,
228            false,
229            comm,
230            kv_shard,
231            vb.pp("v_proj"),
232        )?;
233        let o_proj = RowParallelLayer::new(
234            size_q,
235            size_in,
236            &cfg.quantization_config,
237            false,
238            comm,
239            vb.pp("o_proj"),
240        )?;
241        Ok(Self {
242            q_proj,
243            k_proj,
244            v_proj,
245            o_proj,
246            num_attention_heads: cfg.num_attention_heads / comm.world_size(),
247            num_key_value_heads: (cfg.num_key_value_heads / comm.world_size()).max(1),
248            head_dim: cfg.hidden_size / cfg.num_attention_heads,
249            rotary_emb: rope,
250            max_seq_len: cfg.max_position_embeddings,
251            paged_attn,
252            sdpa_params: SdpaParams {
253                n_kv_groups: mistralrs_quant::compute_n_kv_groups(
254                    cfg.num_key_value_heads,
255                    cfg.num_attention_heads,
256                    comm,
257                ),
258                softcap: None,
259                softmax_scale: 1.0 / ((cfg.hidden_size / cfg.num_attention_heads) as f32).sqrt(),
260                sliding_window: None,
261            },
262        })
263    }

266struct Block {
267    rms_1: RmsNorm,
268    attn: CausalSelfAttention,
269    rms_2: RmsNorm,
270    mlp: Box<dyn MlpLayer>,
271}
272
273impl Block {
274    #[allow(clippy::too_many_arguments)]
275    fn forward(
276        &self,
277        x: &Tensor,
278        attention_mask: &Option<Tensor>,
279        seqlen_offsets: &[usize],
280        kv_cache: &mut KvCache,
281        metadata: Option<((Tensor, Tensor), &PagedAttentionInputMetadata)>,
282        flash_params: &FlashParams,
283    ) -> Result<Tensor> {
284        let residual = x;
285        let x = self.rms_1.forward(x)?;
286        let x = (self.attn.forward(
287            &x,
288            attention_mask,
289            seqlen_offsets,
290            kv_cache,
291            metadata,
292            flash_params,
293        )? + residual)?;
294        let residual = &x;
295        let x = (self.mlp.forward(&self.rms_2.forward(&x)?)? + residual)?;
296        Ok(x)
297    }
298
299    #[allow(clippy::too_many_arguments)]
300    fn load(
301        vb: ShardedVarBuilder,
302        cfg: &Config,
303        mapper: &dyn DeviceMapper,
304        layer_idx: usize,
305        loading_isq: bool,
306        rope: Option<Arc<SmolLm3RotaryEmbedding>>,
307        paged_attn: Option<PagedAttention>,
308        comm: &Arc<mistralrs_quant::Comm>,
309    ) -> Result<Self> {
310        let attn = CausalSelfAttention::load(
311            mapper.set_device(layer_idx, vb.pp("self_attn"), loading_isq),
312            cfg,
313            rope,
314            paged_attn,
315            comm,
316        )?;
317        let mlp = Mlp::new(
318            mapper.set_device(layer_idx, vb.pp("mlp"), loading_isq),
319            cfg.hidden_size,
320            cfg.intermediate_size,
321            &cfg.quantization_config,
322            cfg.hidden_act,
323            comm,
324        )?;
325        let rms_1 = RmsNorm::new(
326            cfg.hidden_size,
327            cfg.rms_norm_eps,
328            mapper.set_device(layer_idx, vb.pp("input_layernorm"), false),
329        )?;
330        let rms_2 = RmsNorm::new(
331            cfg.hidden_size,
332            cfg.rms_norm_eps,
333            mapper.set_device(layer_idx, vb.pp("post_attention_layernorm"), false),
334        )?;
335        Ok(Self {
336            rms_1,
337            attn,
338            rms_2,
339            mlp: Box::new(mlp),
340        })
341    }

362    ) -> Result<Self> {

364        let vb_lm_head = vb.pp("lm_head");
365        Self::new_inner(

374
375    pub fn new_inner(
376        cfg: &Config,
377        vb_m: ShardedVarBuilder,
378        vb_lm_head: ShardedVarBuilder,
379        is_gptx: bool,
380        normal_loading_metadata: NormalLoadingMetadata,
381        attention_mechanism: AttentionImplementation,
382    ) -> Result<Self> {
383        if let Some(ref quant_cfg) = &cfg.quantization_config {
384            tracing::info!(
385                "Using {} quantization: {}.",
386                quant_cfg.name(),
387                quant_cfg.get_bits_name(&vb_m)
388            );
389        }
390        let mapper = normal_loading_metadata.mapper;
391
392        let wte = embedding(
393            cfg.vocab_size,
394            cfg.hidden_size,
395            mapper.set_nm_device(vb_m.pp("embed_tokens"), false),
396            &cfg.quantization_config,
397        )?;
398        let lm_head = if !cfg.tie_word_embeddings {
399            ReplicatedLayer::new(
400                cfg.hidden_size,
401                cfg.vocab_size,
402                &cfg.quantization_config,
403                false,
404                mapper.set_nm_device(vb_lm_head, normal_loading_metadata.loading_isq),
405            )?
406        } else {
407            ReplicatedLayer::from_linear(candle_nn::Linear::new(
408                mapper.cast_nm_device(wte.embeddings(), normal_loading_metadata.loading_isq)?,
409                None,
410            ))?
411        };
412        let ln_f = RmsNorm::new(
413            cfg.hidden_size,
414            cfg.rms_norm_eps,
415            mapper.set_nm_device(vb_m.pp("norm"), false),
416        )?;
417        let head_dim = cfg.hidden_size / cfg.num_attention_heads;
418        let mut ropes = HashMap::new();
419        for i in 0..cfg.num_hidden_layers {
420            let device = mapper
421                .device_for(i, false)
422                .unwrap_or(&normal_loading_metadata.real_device);
423            ropes.insert(
424                device.location(),
425                Arc::new(SmolLm3RotaryEmbedding::new_llama3(
426                    vb_m.dtype(),
427                    cfg,
428                    device,
429                    is_gptx,
430                )?),
431            );
432        }
433        let blocks: Vec<_> = NiceProgressBar::<_, 'b'>(
434            0..cfg.num_hidden_layers,
435            "Loading repeating layers",
436            &normal_loading_metadata.multi_progress,
437        )
438        .par_iter_if_isq(|i| {
439            let device = mapper
440                .device_for(i, false)
441                .unwrap_or(&normal_loading_metadata.real_device);
442            let use_rope = cfg.no_rope_layers()[i];
443            let rotary_emb = if use_rope {
444                Some(
445                    ropes
446                        .get(&device.location())
447                        .expect("No RoPE for device location!")
448                        .clone(),
449                )
450            } else {
451                None
452            };
453            let paged_attn = match &attention_mechanism {
454                AttentionImplementation::Eager => None,
455                AttentionImplementation::PagedAttention => {
456                    Some(PagedAttention::new(head_dim, device, None)?)
457                }
458            };
459            let comm = mapper.get_comm_for(i)?;
460            Block::load(
461                vb_m.pp(format!("layers.{i}")),
462                cfg,
463                &*mapper,
464                i,
465                normal_loading_metadata.loading_isq,
466                rotary_emb,
467                paged_attn,
468                &comm,
469            )
470        })?;
471
472        Ok(Self {
473            wte,
474            blocks,
475            ln_f,
476            lm_head,
477            kv_cache: EitherCache::Normal(NormalCache::new(
478                cfg.num_hidden_layers,
479                cfg.max_position_embeddings,
480            )),
481            device: normal_loading_metadata.real_device,
482            cfg: ModelConfigMetadata {
483                max_seq_len: cfg.max_position_embeddings,
484                num_layers: cfg.num_hidden_layers,
485                hidden_size: cfg.hidden_size,
486                num_kv_heads: (cfg.num_key_value_heads / mapper.get_comm_for(0)?.world_size())
487                    .max(1),
488                num_attn_heads: cfg.num_attention_heads / mapper.get_comm_for(0)?.world_size(),
489                sliding_window: None,
490                k_head_dim: cfg.hidden_size / cfg.num_attention_heads,
491                v_head_dim: cfg.hidden_size / cfg.num_attention_heads,
492            },
493            mapper,
494        })
495    }

497    pub fn forward(
498        &self,
499        input_ids: &Tensor,
500        seqlen_offsets: &[usize],
501        context_lens: Vec<(usize, usize)>,
502        metadata: Option<(Vec<(Tensor, Tensor)>, &PagedAttentionInputMetadata)>,
503        flash_params: &FlashParams,
504    ) -> Result<Tensor> {
505        self.forward_embeds(
506            input_ids,
507            self.wte.forward(input_ids)?,
508            seqlen_offsets,
509            context_lens,
510            metadata,
511            flash_params,
512        )
513    }
514
515    #[allow(clippy::too_many_arguments)]
516    pub fn forward_embeds(
517        &self,
518        input_ids: &Tensor,
519        input_embeds: Tensor,
520        seqlen_offsets: &[usize],
521        context_lens: Vec<(usize, usize)>,
522        metadata: Option<(Vec<(Tensor, Tensor)>, &PagedAttentionInputMetadata)>,
523        flash_params: &FlashParams,
524    ) -> Result<Tensor> {
525        let mut x = input_embeds;
526        let cache = &mut self.kv_cache.normal().0;
527        let mask = CausalMasker.make_causal_mask_matrix(
528            input_ids,
529            metadata
530                .as_ref()
531                .map(|(_, _)| &seqlen_offsets as &dyn PastKvLenCache)
532                .unwrap_or(cache as &dyn PastKvLenCache),
533            x.dtype(),
534            self.blocks[0].attn.num_attention_heads,
535        )?;
536        // PagedAttention prompt chunking
537        let mask = mask.filter(|_| {
538            metadata
539                .as_ref()
540                .map(|(_, meta)| meta.is_first_prompt_chunk)
541                .unwrap_or(true)
542        });
543        for (block_idx, block) in self.blocks.iter().enumerate() {
544            x = self.mapper.map(x, block_idx)?;
545            x = block.forward(
546                &x,
547                &mask.clone().map(|m| m.to_device(x.device()).unwrap()),
548                seqlen_offsets,
549                &mut cache[block_idx],
550                metadata
551                    .as_ref()
552                    .map(|(kv_cache, metadata)| (kv_cache[block_idx].clone(), *metadata)),
553                flash_params,
554            )?;
555        }
556        let x = x.to_device(&self.device)?;
557        let mut x = self.ln_f.forward(&x)?;
558        if let Some(t) = self.lm_head.quantized_act_type() {
559            x = x.to_dtype(t)?;
560        }
561        let xs = MatMul.qmethod_matmul(&x, &*self.lm_head)?;
562        extract_logits(&xs, context_lens)
563    }
564
565    pub fn residual_tensors_m(&self, uvb_m: UnVarBuilder) -> Vec<(String, Tensor)> {
566        uvb_m.pp("embed_tokens").add(&self.wte);
567        uvb_m.pp("norm").add(&self.ln_f);
568
569        for (layer_idx, layer) in self.blocks.iter().enumerate() {
570            let uvb_l = uvb_m.pp("layers").pp(layer_idx);
571            uvb_l.pp("input_layernorm").add(&layer.rms_1);
572            uvb_l.pp("post_attention_layernorm").add(&layer.rms_2);
573        }
574
575        uvb_m.to_safetensors()
576    }

579impl IsqModel for SmolLm3 {
580    fn get_layers(
581        &mut self,
582    ) -> (
583        Vec<(&mut Arc<dyn QuantMethod>, Option<usize>)>,
584        &dyn DeviceMapper,
585    ) {
586        let mut tensors = Vec::new();
587        tensors.push((&mut self.lm_head, None));
588        for (i, layer) in self.blocks.iter_mut().enumerate() {
589            tensors.push((&mut layer.attn.q_proj, Some(i)));
590            tensors.push((&mut layer.attn.k_proj, Some(i)));
591            tensors.push((&mut layer.attn.v_proj, Some(i)));
592            tensors.push((&mut layer.attn.o_proj, Some(i)));
593            tensors.extend(
594                layer
595                    .mlp
596                    .get_isq_layers()
597                    .into_iter()
598                    .map(|m| (m, Some(i)))
599                    .collect::<Vec<_>>(),
600            );
601        }
602        (tensors, &*self.mapper)
603    }
604
605    fn residual_tensors(&self) -> Vec<(String, Tensor)> {
606        let uvb = UnVarBuilder::new();
607        self.residual_tensors_m(uvb.pp("model"))
608    }
609
610    fn imatrix_names(&self) -> candle_core::Result<Vec<Option<String>>> {
611        // NOTE: dependant on the exact implementation in get_layers!
612        let mut names = Vec::new();
613        // lm_head
614        names.push(None);
615        for i in 0..self.blocks.len() {
616            names.push(Some(format!("blk.{i}.attn_q.weight")));
617            names.push(Some(format!("blk.{i}.attn_k.weight")));
618            names.push(Some(format!("blk.{i}.attn_v.weight")));
619            names.push(Some(format!("blk.{i}.attn_output.weight")));
620            names.push(Some(format!("blk.{i}.ffn_gate.weight")));
621            names.push(Some(format!("blk.{i}.ffn_up.weight")));
622            names.push(Some(format!("blk.{i}.ffn_down.weight")));
623        }
624        Ok(names)
625    }

682    fn get_mlps(&self) -> Vec<&dyn MlpLayer> {

689    fn get_mlps_mut(&mut self) -> Vec<&mut Box<dyn MlpLayer>> {
690        let mut mlps = Vec::new();
691        for layer in &mut self.blocks {
692            mlps.push(&mut layer.mlp);

696    fn create_anymoe_layers(
697        &mut self,
698        additional_vbs: Vec<ShardedVarBuilder>,
699        config: AnyMoeConfig,
700        (prefix, mlp): (String, String),
701        mut layers: Vec<usize>,
702        expert_type: AnyMoeExpertType,
703        gate_vb: Option<ShardedVarBuilder>,
704    ) -> Result<()> {
705        let mut experts: Vec<Vec<Box<dyn MlpLayer>>> = Vec::new();
706        if layers.is_empty() {
707            layers = (0..self.blocks.len()).collect::<Vec<_>>();
708        }
709        for _ in 0..layers.len() {
710            experts.push(Vec::new());
711        }
712        for vb in additional_vbs {
713            let vb = vb.pp(&prefix);
714            for (layer, row) in experts.iter_mut().enumerate() {
715                if !layers.contains(&layer) {
716                    continue;
717                }
718
719                let intermediate_size = self.blocks[layer].mlp.get_params()[1];
720                let hidden_size = self.blocks[layer].mlp.get_params()[0];
721                match expert_type {
722                    AnyMoeExpertType::FineTuned => {
723                        let (dtype, device) = self.blocks[layer].mlp.dtype_device();
724                        row.push(Box::new(Mlp::replicate(
725                            self.blocks[layer].mlp.get_params(),
726                            vb.pp(layer).pp(&mlp).set_dtype(dtype).set_device(device),
727                            self.blocks[layer].mlp.hidden_act(),
728                            &self.mapper.get_comm_for(layer)?,
729                        )?));
730                    }
731                    AnyMoeExpertType::LoraAdapter {
732                        rank,
733                        alpha,
734                        ref target_modules,
735                    } => {
736                        let vb_mlp = vb.pp(layer).pp(&mlp);
737
738                        let c_fc1_delta = if target_modules.contains(&"c_fc1".to_string()) {
739                            Some(get_delta_from_lora_ab!(
740                                vb_mlp,
741                                rank,
742                                alpha,
743                                (hidden_size, intermediate_size),
744                                "c_fc1"
745                            ))
746                        } else {
747                            None
748                        };
749                        let c_fc2_delta = if target_modules.contains(&"c_fc2".to_string()) {
750                            Some(get_delta_from_lora_ab!(
751                                vb_mlp,
752                                rank,
753                                alpha,
754                                (hidden_size, intermediate_size),
755                                "c_fc2"
756                            ))
757                        } else {
758                            None
759                        };
760                        let c_proj_delta = if target_modules.contains(&"c_proj".to_string()) {
761                            Some(get_delta_from_lora_ab!(
762                                vb_mlp,
763                                rank,
764                                alpha,
765                                (intermediate_size, hidden_size),
766                                "c_proj"
767                            ))
768                        } else {
769                            None
770                        };
771
772                        row.push(self.blocks[layer].mlp.new_added_delta(vec![
773                            c_fc1_delta,
774                            c_fc2_delta,
775                            c_proj_delta,
776                        ])?);
777                    }
778                }
779            }
780        }
781        for (layer, expert) in layers.into_iter().zip(experts) {
782            let mut experts_all = vec![self.blocks[layer].mlp.clone()];
783            experts_all.extend(expert);
784            let (dtype, device) = self.blocks[layer].mlp.dtype_device();
785            self.blocks[layer].mlp = Box::new(MoeMlp::new(
786                experts_all,
787                config.clone(),
788                dtype,
789                &device,
790                layer,
791                gate_vb.as_ref(),
792            )?);
793        }
794        Ok(())
795    }
796    fn amoe_supported(&self) -> bool {
797        true
798    }
