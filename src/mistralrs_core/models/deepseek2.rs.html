106    fn softmax_scale(&self) -> f32 {
107        let mut softmax_scale = 1.0 / (self.q_head_dim() as f32).sqrt();
108        if let Some(DeepSeekV2RopeScaling::Yarn {
109            mscale_all_dim,
110            factor,

112        }) = self.rope_scaling
113        {
114            let mscale = DeepSeekV2RotaryEmbedding::yarn_get_mscale(factor, mscale_all_dim);
115            softmax_scale = softmax_scale * mscale * mscale;
116        }
117        softmax_scale

130impl QProj {
131    fn forward(&self, xs: &Tensor) -> Result<Tensor> {
132        match self {
133            Self::Lora { a, norm, b } => {
134                b.forward_autocast(&norm.forward(&a.forward_autocast(xs)?)?)
135            }
136            Self::Plain(lin) => lin.forward_autocast(xs),
137        }
138    }

155impl Attention {
156    #[allow(clippy::too_many_arguments)]
157    fn new(
158        rotary_emb: Arc<DeepSeekV2RotaryEmbedding>,
159        cfg: &DeepSeekV2Config,
160        vb: ShardedVarBuilder,
161        mapper: &dyn DeviceMapper,
162        layer_idx: usize,
163        loading_isq: bool,
164        paged_attn: Option<PagedAttention>,
165        comm: &Arc<mistralrs_quant::Comm>,
166    ) -> Result<Self> {
167        let q_head_dim = cfg.q_head_dim();
168        let q = match cfg.q_lora_rank {
169            Some(lora_rank) => {
170                let a = ReplicatedLayer::new(
171                    cfg.hidden_size,
172                    lora_rank,
173                    &cfg.quantization_config,
174                    cfg.attention_bias,
175                    mapper.set_device(layer_idx, vb.pp("q_a_proj"), loading_isq),
176                )?;

178                    lora_rank,
179                    cfg.rms_norm_eps,
180                    mapper.set_device(layer_idx, vb.pp("q_a_layernorm"), false),
181                )?;
182                let b = ColumnParallelLayer::new(
183                    lora_rank,

185                    &cfg.quantization_config,
186                    false,
187                    comm,
188                    mapper.set_device(layer_idx, vb.pp("q_b_proj"), loading_isq),
189                )?;
190                QProj::Lora { a, norm, b }
191            }
192            None => QProj::Plain(ColumnParallelLayer::new(
193                cfg.hidden_size,
194                cfg.num_attention_heads * q_head_dim,
195                &cfg.quantization_config,
196                false,
197                comm,
198                mapper.set_device(layer_idx, vb.pp("q_proj"), loading_isq),
199            )?),
200        };
201
202        let kv_a_proj_with_mqa = ReplicatedLayer::new(
203            cfg.hidden_size,
204            cfg.kv_lora_rank + cfg.qk_rope_head_dim,
205            &cfg.quantization_config,
206            cfg.attention_bias,
207            mapper.set_device(layer_idx, vb.pp("kv_a_proj_with_mqa"), loading_isq),
208        )?;
209        let kv_a_layernorm = RmsNorm::new(
210            cfg.kv_lora_rank,
211            cfg.rms_norm_eps,
212            mapper.set_device(layer_idx, vb.pp("kv_a_layernorm"), false),
213        )?;
214        let kv_b_proj = ColumnParallelLayer::new(
215            cfg.kv_lora_rank,
216            cfg.num_attention_heads * (q_head_dim - cfg.qk_rope_head_dim + cfg.v_head_dim),
217            &cfg.quantization_config,
218            false,
219            comm,
220            mapper.set_device(layer_idx, vb.pp("kv_b_proj"), loading_isq),
221        )?;
222
223        let o_proj = RowParallelLayer::new(
224            cfg.num_attention_heads * cfg.v_head_dim,
225            cfg.hidden_size,
226            &cfg.quantization_config,
227            cfg.attention_bias,
228            comm,
229            mapper.set_device(layer_idx, vb.pp("o_proj"), loading_isq),
230        )?;
231
232        Ok(Self {
233            q,
234            kv_a_proj_with_mqa,
235            kv_a_layernorm,
236            kv_b_proj,
237            o_proj,
238            rotary_emb,
239            cfg: cfg.clone(),
240            q_head_dim,
241            paged_attn,
242            num_attention_heads: cfg.num_attention_heads / comm.world_size(),
243            sdpa_params: SdpaParams {
244                n_kv_groups: 1,
245                softcap: None,
246                softmax_scale: cfg.softmax_scale(),
247                sliding_window: None,
248            },
249        })
250    }

252    fn forward(
253        &self,
254        xs: &Tensor,
255        attention_mask: Option<&Tensor>,
256        seqlen_offsets: &[usize],
257        kv_cache: &mut KvCache,
258        metadata: Option<((Tensor, Tensor), &PagedAttentionInputMetadata)>,
259        flash_params: &FlashParams,
260    ) -> Result<Tensor> {
261        let (bs, seq_len, _) = xs.dims3()?;
262
263        let mut q = self.q.forward(xs)?;
264        q = q
265            .reshape((bs, seq_len, self.num_attention_heads, self.q_head_dim))?
266            .transpose(1, 2)?;
267        let q_split = q.split(
268            &[self.cfg.qk_nope_head_dim, self.cfg.qk_rope_head_dim],
269            D::Minus1,
270        )?;
271        let q_nope = q_split[0].clone();
272        let mut q_pe = q_split[1].clone();
273
274        let mut compressed_kv = self.kv_a_proj_with_mqa.forward_autocast(xs)?;
275        let ckv_split = compressed_kv.split(
276            &[self.cfg.kv_lora_rank, self.cfg.qk_rope_head_dim],
277            D::Minus1,
278        )?;
279        compressed_kv = ckv_split[0].clone();
280        let mut k_pe = ckv_split[1].clone();
281        k_pe = k_pe
282            .reshape((bs, seq_len, 1, self.cfg.qk_rope_head_dim))?
283            .transpose(1, 2)?;
284        let mut kv = self
285            .kv_b_proj
286            .forward_autocast(&self.kv_a_layernorm.forward(&compressed_kv)?)?;
287        kv = kv
288            .reshape((
289                bs,
290                seq_len,
291                self.num_attention_heads,
292                self.cfg.qk_nope_head_dim + self.cfg.v_head_dim,
293            ))?
294            .transpose(1, 2)?;
295
296        let kv_split = kv.split(&[self.cfg.qk_nope_head_dim, self.cfg.v_head_dim], D::Minus1)?;
297        let k_nope = kv_split[0].clone();
298        let mut v = kv_split[1].clone();
299
300        (q_pe, k_pe) = self.rotary_emb.forward(&q_pe, &k_pe, seqlen_offsets)?;
301
302        let q = Tensor::cat(&[&q_nope, &q_pe], D::Minus1)?.contiguous()?;
303        let mut k = Tensor::cat(
304            &[&k_nope, &k_pe.repeat((1, self.num_attention_heads, 1, 1))?],
305            D::Minus1,
306        )?
307        .contiguous()?;
308
309        let mut attn_out = match &self.paged_attn {
310            Some(paged_attn) => match metadata {
311                Some(((key_cache, value_cache), input_metadata)) => {
312                    let v = v
313                        .pad_with_zeros(D::Minus1, 0, self.q_head_dim - self.cfg.v_head_dim)?
314                        .contiguous()?;
315                    paged_attn
316                        .forward(
317                            &q,
318                            &k,
319                            &v,
320                            attention_mask,
321                            Some(key_cache),
322                            Some(value_cache),
323                            input_metadata,
324                            &self.sdpa_params,
325                            Some(flash_params),
326                        )?
327                        .narrow(D::Minus1, 0, self.cfg.v_head_dim)?
328                }
329                None => {
330                    // If we don't have metadata, we are most likely generating an imatrix so we don't want to populate that.
331                    // Generating the dummy metadata with the assumption that we are not generating text (only processing prompts).
332                    let input_metadata = PagedAttentionInputMetadata::dummy(q.device())?;
333                    // Sanity check.
334                    assert!(attention_mask.is_some());
335                    let v = v
336                        .pad_with_zeros(D::Minus1, 0, self.q_head_dim - self.cfg.v_head_dim)?
337                        .contiguous()?;
338                    paged_attn
339                        .forward(
340                            &q,
341                            &k,
342                            &v,
343                            attention_mask,
344                            None,
345                            None,
346                            &input_metadata,
347                            &self.sdpa_params,
348                            Some(flash_params),
349                        )?
350                        .narrow(D::Minus1, 0, self.cfg.v_head_dim)?
351                }
352            },
353            None => {
354                (k, v) = kv_cache.append(&k, &v)?;
355
356                Sdpa.run_attention(
357                    &q,
358                    &k,
359                    &v,
360                    attention_mask,
361                    Some(flash_params),
362                    &self.sdpa_params,
363                )?
364            }
365        };
366
367        attn_out = if attention_mask.is_some() {
368            attn_out.transpose(1, 2)?.reshape((bs, seq_len, ()))?
369        } else {
370            attn_out.reshape((bs, seq_len, ()))?
371        };
372
373        self.o_proj.forward_autocast(&attn_out)
374    }
375}

377struct Expert {
378    gate: Arc<dyn QuantMethod>,
379    up: Arc<dyn QuantMethod>,
380    down: Arc<dyn QuantMethod>,
381    act: Activation,
382}
383
384impl Expert {
385    fn new(
386        cfg: &DeepSeekV2Config,
387        vb: ShardedVarBuilder,
388        hidden_size: Option<usize>,
389        intermediate_size: Option<usize>,
390    ) -> Result<Self> {
391        let hidden_size = hidden_size.unwrap_or(cfg.hidden_size);
392        let intermediate_size = intermediate_size.unwrap_or(cfg.intermediate_size);
393
394        Ok(Self {
395            gate: ReplicatedLayer::new(
396                hidden_size,
397                intermediate_size,
398                &cfg.quantization_config,
399                false,
400                vb.pp("gate_proj"),
401            )?,
402            up: ReplicatedLayer::new(
403                hidden_size,
404                intermediate_size,
405                &cfg.quantization_config,
406                false,
407                vb.pp("up_proj"),
408            )?,
409            down: ReplicatedLayer::new(
410                intermediate_size,
411                hidden_size,
412                &cfg.quantization_config,
413                false,
414                vb.pp("down_proj"),
415            )?,
416            act: cfg.hidden_act,
417        })
418    }
419
420    fn forward(&self, xs: &Tensor) -> Result<Tensor> {
421        let original_dtype = xs.dtype();
422        let mut xs = xs.clone();
423        if let Some(t) = self.gate.quantized_act_type() {
424            xs = xs.to_dtype(t)?;
425        }
426        let lhs = self.gate.forward(&xs)?;
427        let rhs = self.up.forward(&xs)?;
428        let mut res = self
429            .down
430            .forward(&crate::ops::mul_and_act(&lhs, &rhs, self.act)?)?;
431        if self.gate.quantized_act_type().is_some() {
432            res = res.to_dtype(original_dtype)?;
433        }
434        Ok(res)
435    }
436}

438struct MoeGate {
439    weight: Tensor,
440    cfg: DeepSeekV2Config,
441    top_k: usize,
442    n_routed_experts: usize,
443}
444
445impl MoeGate {
446    fn new(cfg: &DeepSeekV2Config, vb: ShardedVarBuilder, n_routed_experts: usize) -> Result<Self> {
447        let weight = vb.get((n_routed_experts, cfg.hidden_size), "weight")?;
448        Ok(Self {
449            weight,
450            cfg: cfg.clone(),
451            top_k: cfg.num_experts_per_tok.unwrap(),
452            n_routed_experts,
453        })
454    }
455
456    /// (topk_idx, topk_weight)
457    fn forward(&self, xs: &Tensor) -> Result<(Tensor, Tensor)> {
458        let (bs, seq_len, h) = xs.dims3()?;
459        // Compute gating score
460        let xs = xs.reshape(((), h))?;
461        let logits = xs
462            .to_dtype(DType::F32)?
463            .broadcast_matmul(&self.weight.t()?.to_dtype(DType::F32)?)?;
464        let scores = match self.cfg.scoring_func {
465            ScoringFunc::Softmax => candle_nn::ops::softmax_last_dim(&logits)?,
466        };
467
468        // Select top-k experts
469        let (mut topk_weight, topk_idx) = match self.cfg.topk_method {
470            TopkMethod::Greedy => {
471                let TopKOutput { values, indices } = scores.topk_unsorted(self.top_k)?;
472                (values, indices)
473            }
474            TopkMethod::GroupLimitedGreedy => {
475                // (n, n_group)
476                let group_scores = scores
477                    .reshape((bs * seq_len, self.cfg.n_group, ()))?
478                    .max(D::Minus1)?;
479                // (n, topk_group)
480                let group_idx = group_scores.topk_unsorted(self.cfg.topk_group)?.indices;
481                // (n, n_group)
482                let mut group_mask = group_scores.zeros_like()?;
483                // (n, n_group)
484                group_mask = group_mask.scatter_add(
485                    &group_idx,
486                    &group_idx.ones_like()?.to_dtype(group_mask.dtype())?,
487                    1,
488                )?;
489                // (n, e)
490                let score_mask = group_mask
491                    .unsqueeze(D::Minus1)?
492                    .expand((
493                        bs * seq_len,
494                        self.cfg.n_group,
495                        self.n_routed_experts / self.cfg.n_group,
496                    ))?
497                    .reshape((bs * seq_len, ()))?;
498                // (n, e)
499                // Invert the mask
500                let tmp_scores = masked_fill(&score_mask, &(1. - &score_mask.ne(0.)?)?, 0.)?;
501                let TopKOutput { values, indices } = tmp_scores.topk_unsorted(self.top_k)?;
502                (values, indices)
503            }
504        };
505
506        if self.top_k > 1 && self.cfg.norm_topk_prob {
507            let denmoninator = (topk_weight.sum_keepdim(D::Minus1)? + 1e-20)?;
508            topk_weight = (topk_weight / denmoninator)?;
509        } else {
510            topk_weight = (topk_weight * self.cfg.routed_scaling_factor)?;
511        }
512        Ok((topk_idx, topk_weight))
513    }
514}

526impl Moe {

530        vb: ShardedVarBuilder,
531        mapper: &dyn DeviceMapper,
532        layer_idx: usize,
533        loading_isq: bool,
534        n_shared_experts: Option<usize>,
535        n_routed_experts: usize,
536        comm: &Arc<mistralrs_quant::Comm>,
537    ) -> Result<Self> {
538        let mut experts = Vec::with_capacity(n_routed_experts);
539        let n_local_experts = n_routed_experts / comm.world_size();
540        let experts_start_idx = comm.rank() * n_local_experts;
541        let experts_end_idx = experts_start_idx + n_local_experts;
542        for i in 0..n_routed_experts {
543            if i >= experts_start_idx && i < experts_end_idx {
544                let vb_e = vb.pp("experts").pp(i);
545                experts.push(Some(Expert::new(
546                    cfg,
547                    mapper.set_device(layer_idx, vb_e, loading_isq),
548                    None,
549                    Some(cfg.moe_intermediate_size),
550                )?));
551            } else {
552                experts.push(None);
553            }
554        }
555        let shared_experts = if let Some(n_shared_experts) = n_shared_experts {
556            let intermediate_size = cfg.moe_intermediate_size * n_shared_experts;
557            Some(Mlp::new(
558                mapper.set_device(layer_idx, vb.pp("shared_experts"), loading_isq),
559                cfg.hidden_size,
560                intermediate_size,
561                &cfg.quantization_config,
562                cfg.hidden_act,
563                comm,
564            )?)
565        } else {
566            None
567        };
568        let gate = MoeGate::new(
569            cfg,
570            mapper.set_device(layer_idx, vb.pp("gate"), false),
571            n_routed_experts,
572        )?;
573        Ok(Self {
574            experts,
575            shared_experts,
576            gate,
577            all_reduce: SumAllReduce::new(comm),
578            experts_end_idx,
579            experts_start_idx,
580            world_size: comm.world_size(),
581        })
582    }

584    fn moe_infer(&self, xs: &Tensor, topk_ids: &Tensor, topk_weight: &Tensor) -> Result<Tensor> {
585        let mut y = xs.zeros_like()?;
586        let topk_weight = if topk_weight.dtype() != xs.dtype() {
587            topk_weight.to_dtype(xs.dtype())?
588        } else {
589            topk_weight.to_owned()
590        };
591        let unique_ids: HashSet<u32> =
592            HashSet::from_iter(topk_ids.to_device(&Device::Cpu)?.flatten_all()?.to_vec1()?);
593        for i in self.experts_start_idx..self.experts_end_idx {
594            if !unique_ids.contains(&(i as u32)) {
595                continue;
596            }
597            let idx_top = topk_ids.eq(i as f64)?.nonzero()?.t()?;
598            let idx = &idx_top.i(0)?.contiguous()?;
599            let top = &idx_top.i(1)?.contiguous()?;
600
601            let expert = self.experts[i]
602                .as_ref()
603                .context("Expert is not present for this rank.")?;
604
605            y = y.index_add(
606                idx,
607                &expert.forward(&xs.index_select(idx, 0)?)?.broadcast_mul(
608                    &topk_weight
609                        .index_select(idx, 0)?
610                        .gather(&top.unsqueeze(1)?, 1)?
611                        .squeeze(1)?
612                        .unsqueeze(D::Minus1)?,
613                )?,
614                0,
615            )?;
616        }
617
618        if self.world_size > 1 {
619            y = self.all_reduce.sum_all_reduce(&y)?;
620        }
621
622        Ok(y)
623    }
624
625    fn forward(&self, xs: &Tensor) -> Result<Tensor> {
626        let identity = xs.clone();
627        let orig_shape = xs.shape();
628        let (topk_idx, topk_weight) = self.gate.forward(xs)?;
629        let xs = xs.reshape(((), xs.dim(D::Minus1)?))?;
630
631        let mut y = self
632            .moe_infer(&xs, &topk_idx, &topk_weight)?
633            .reshape(orig_shape)?;
634        if let Some(ref shared_experts) = self.shared_experts {
635            y = (y + shared_experts.forward(&identity)?)?;
636        }
637        Ok(y)
638    }
639}
640
641enum MoeOrMlp {
642    Moe(Box<Moe>),
643    Mlp(Mlp),
644}
645
646impl MoeOrMlp {
647    fn forward(&self, xs: &Tensor) -> Result<Tensor> {
648        match self {
649            Self::Mlp(mlp) => mlp.forward(xs),
650            Self::Moe(moe) => moe.forward(xs),
651        }
652    }
653}

655struct DecoderLayer {
656    input_layernorm: RmsNorm,
657    post_attention_layernorm: RmsNorm,
658    attn: Attention,
659    moe_or_mlp: MoeOrMlp,
660}
661
662impl DecoderLayer {
663    #[allow(clippy::too_many_arguments)]
664    fn new(
665        rotary_emb: Arc<DeepSeekV2RotaryEmbedding>,
666        cfg: &DeepSeekV2Config,
667        vb: ShardedVarBuilder,
668        mapper: &dyn DeviceMapper,
669        layer_idx: usize,
670        loading_isq: bool,
671        paged_attn: Option<PagedAttention>,
672        comm: &Arc<mistralrs_quant::Comm>,
673    ) -> Result<Self> {
674        let attn = Attention::new(
675            rotary_emb,
676            cfg,
677            vb.pp("self_attn"),
678            mapper,
679            layer_idx,
680            loading_isq,
681            paged_attn,
682            comm,
683        )?;
684        let input_layernorm = RmsNorm::new(
685            cfg.hidden_size,
686            cfg.rms_norm_eps,
687            mapper.set_device(layer_idx, vb.pp("input_layernorm"), false),
688        )?;
689        let post_attention_layernorm = RmsNorm::new(
690            cfg.hidden_size,
691            cfg.rms_norm_eps,
692            mapper.set_device(layer_idx, vb.pp("post_attention_layernorm"), false),
693        )?;
694        let moe_or_mlp = if let Some(n_routed_experts) = cfg.n_routed_experts.filter(|_| {
695            layer_idx >= cfg.first_k_dense_replace && layer_idx % cfg.moe_layer_freq == 0
696        }) {
697            MoeOrMlp::Moe(Box::new(Moe::new(
698                cfg,
699                vb.pp("mlp"),
700                mapper,
701                layer_idx,
702                loading_isq,
703                cfg.n_shared_experts,
704                n_routed_experts,
705                comm,
706            )?))
707        } else {
708            MoeOrMlp::Mlp(Mlp::new(
709                mapper.set_device(layer_idx, vb.pp("mlp"), loading_isq),
710                cfg.hidden_size,
711                cfg.intermediate_size,
712                &cfg.quantization_config,
713                cfg.hidden_act,
714                comm,
715            )?)
716        };
717
718        Ok(Self {
719            input_layernorm,
720            post_attention_layernorm,
721            attn,
722            moe_or_mlp,
723        })
724    }
725
726    fn forward(
727        &self,
728        xs: &Tensor,
729        attention_mask: Option<&Tensor>,
730        seqlen_offsets: &[usize],
731        kv_cache: &mut KvCache,
732        metadata: Option<((Tensor, Tensor), &PagedAttentionInputMetadata)>,
733        flash_params: &FlashParams,
734    ) -> Result<Tensor> {
735        let residual = xs;
736        let xs = self.input_layernorm.forward(xs)?;
737        let xs = self.attn.forward(
738            &xs,
739            attention_mask,
740            seqlen_offsets,
741            kv_cache,
742            metadata,
743            flash_params,
744        )?;
745        let xs = (xs + residual)?;
746        let residual = &xs;
747        let xs = self
748            .moe_or_mlp
749            .forward(&xs.apply(&self.post_attention_layernorm)?)?;
750        residual + xs
751    }
752}

755    lm_head: Arc<dyn QuantMethod>,
756    embed_tokens: Embedding,
757    norm: RmsNorm,
758    layers: Vec<DecoderLayer>,

766impl DeepSeekV2 {
767    pub fn new(
768        cfg: &DeepSeekV2Config,
769        vb: ShardedVarBuilder,
770        _is_gptx: bool,
771        normal_loading_metadata: NormalLoadingMetadata,
772        attention_mechanism: AttentionImplementation,
773    ) -> Result<Self> {
774        let vb_m = vb.pp("model");
775
776        let mapper = normal_loading_metadata.mapper;
777
778        let embed_tokens = embedding(
779            cfg.vocab_size,
780            cfg.hidden_size,
781            mapper.set_nm_device(vb_m.pp("embed_tokens"), false),
782            &cfg.quantization_config,
783        )?;
784        let lm_head = if !cfg.tie_word_embeddings {
785            ReplicatedLayer::new(
786                cfg.hidden_size,
787                cfg.vocab_size,
788                &cfg.quantization_config,
789                false,
790                mapper.set_nm_device(vb.pp("lm_head"), normal_loading_metadata.loading_isq),
791            )?
792        } else {
793            ReplicatedLayer::from_linear(candle_nn::Linear::new(
794                mapper.cast_nm_device(
795                    embed_tokens.embeddings(),
796                    normal_loading_metadata.loading_isq,
797                )?,
798                None,
799            ))?
800        };
801        let norm = RmsNorm::new(
802            cfg.hidden_size,
803            cfg.rms_norm_eps,
804            mapper.set_nm_device(vb_m.pp("norm"), false),
805        )?;
806
807        let mut ropes = HashMap::new();
808        let rope_cfg = DeepSeekV2RopeConfig {
809            rope_scaling: cfg.rope_scaling.clone(),
810            max_position_embeddings: cfg.max_position_embeddings,
811            rope_theta: cfg.rope_theta,
812            qk_rope_head_dim: cfg.qk_rope_head_dim,
813        };
814        for i in 0..cfg.num_hidden_layers {
815            let device = mapper
816                .device_for(i, false)
817                .unwrap_or(&normal_loading_metadata.real_device);
818            ropes.insert(
819                device.location(),
820                Arc::new(DeepSeekV2RotaryEmbedding::new(
821                    &rope_cfg,
822                    vb.dtype(),
823                    device,
824                )?),
825            );
826        }
827
828        let vb_l = vb_m.pp("layers");
829        let layers: Vec<DecoderLayer> = NiceProgressBar::<_, 'b'>(
830            0..cfg.num_hidden_layers,
831            "Loading repeating layers",
832            &normal_loading_metadata.multi_progress,
833        )
834        .par_iter_if_isq(|layer_idx| {
835            let device = mapper
836                .device_for(layer_idx, false)
837                .unwrap_or(&normal_loading_metadata.real_device);
838            let rotary_emb = ropes
839                .get(&device.location())
840                .expect("No RoPE for device location!")
841                .clone();
842            let paged_attn = match &attention_mechanism {
843                AttentionImplementation::Eager => None,
844                AttentionImplementation::PagedAttention => Some(
845                    PagedAttention::new(cfg.v_head_dim, device, None)
846                        .expect("Failed to create PagedAttention"),
847                ),
848            };
849            let comm = mapper.get_comm_for(layer_idx)?;
850            DecoderLayer::new(
851                rotary_emb.clone(),
852                cfg,
853                vb_l.pp(layer_idx),
854                &*mapper,
855                layer_idx,
856                normal_loading_metadata.loading_isq,
857                paged_attn,
858                &comm,
859            )
860        })?;
861
862        Ok(Self {
863            lm_head,
864            embed_tokens,
865            norm,
866            layers,
867            cache: EitherCache::Normal(NormalCache::new(
868                cfg.num_hidden_layers,
869                cfg.max_position_embeddings,
870            )),
871            device: normal_loading_metadata.real_device.clone(),
872            max_seq_len: cfg.max_position_embeddings,
873            cfg: ModelConfigMetadata {
874                max_seq_len: cfg.max_position_embeddings,
875                num_layers: cfg.num_hidden_layers,
876                hidden_size: cfg.hidden_size,
877                num_kv_heads: (cfg.num_attention_heads / mapper.get_comm_for(0)?.world_size())
878                    .max(1),
879                num_attn_heads: (cfg.num_attention_heads / mapper.get_comm_for(0)?.world_size())
880                    .max(1),
881                sliding_window: None,
882                k_head_dim: cfg.q_head_dim(),
883                v_head_dim: if matches!(
884                    attention_mechanism,
885                    AttentionImplementation::PagedAttention
886                ) {
887                    cfg.q_head_dim()
888                } else {
889                    cfg.v_head_dim
890                },
891            },
892            mapper,
893        })
894    }
895
896    #[allow(clippy::too_many_arguments)]
897    pub fn forward(
898        &self,
899        input_ids: &Tensor,
900        seqlen_offsets: &[usize],
901        context_lens: Vec<(usize, usize)>,
902        metadata: Option<(Vec<(Tensor, Tensor)>, &PagedAttentionInputMetadata)>,
903        flash_params: &FlashParams,
904    ) -> Result<Tensor> {
905        let mut xs = self.embed_tokens.forward(input_ids)?;
906        let cache = &mut self.cache.normal().0;
907        let attention_mask = CausalMasker.make_causal_mask_matrix(
908            input_ids,
909            metadata
910                .as_ref()
911                .map(|(_, _)| &seqlen_offsets as &dyn PastKvLenCache)
912                .unwrap_or(cache as &dyn PastKvLenCache),
913            xs.dtype(),
914            self.cfg.num_attn_heads,
915        )?;
916        // PagedAttention prompt chunking
917        let attention_mask = attention_mask.filter(|_| {
918            metadata
919                .as_ref()
920                .map(|(_, meta)| meta.is_first_prompt_chunk)
921                .unwrap_or(true)
922        });
923        for (i, layer) in self.layers.iter().enumerate() {
924            xs = self.mapper.map(xs, i)?;
925            xs = layer.forward(
926                &xs,
927                attention_mask
928                    .as_ref()
929                    .map(|m| m.to_device(xs.device()).unwrap())
930                    .as_ref(),
931                seqlen_offsets,
932                &mut cache[i],
933                metadata
934                    .as_ref()
935                    .map(|(kv_cache, metadata)| (kv_cache[i].clone(), *metadata)),
936                flash_params,
937            )?;
938        }
939        let xs = xs.to_device(&self.device)?;
940        let xs = xs.apply(&self.norm)?;
941        extract_logits(&self.lm_head.forward_autocast(&xs)?, context_lens)
942    }
943}

972                }
973                MoeOrMlp::Moe(moe) => {

981                        tensors.push((&mut mlp.up, Some(i)));
982                        tensors.push((&mut mlp.down, Some(i)));
