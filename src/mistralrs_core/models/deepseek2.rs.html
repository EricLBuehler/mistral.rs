109        let mut softmax_scale = 1.0 / (self.q_head_dim() as f32).sqrt();
110        if let Some(DeepSeekV2RopeScaling::Yarn {

116            let mscale = DeepSeekV2RotaryEmbedding::yarn_get_mscale(factor, mscale_all_dim);

134        match self {
135            Self::Lora { a, norm, b } => {
136                b.forward_autocast(&norm.forward(&a.forward_autocast(xs)?)?)

138            Self::Plain(lin) => lin.forward_autocast(xs),

159    #[allow(clippy::too_many_arguments)]
160    fn new(
161        rotary_emb: Arc<DeepSeekV2RotaryEmbedding>,
162        cfg: &DeepSeekV2Config,
163        vb: ShardedVarBuilder,
164        mapper: &dyn DeviceMapper,
165        layer_idx: usize,
166        loading_isq: bool,
167        paged_attn: Option<PagedAttention>,
168        comm: &Arc<mistralrs_quant::Comm>,
169    ) -> Result<Self> {
170        let q_head_dim = cfg.q_head_dim();
171        let q = match cfg.q_lora_rank {
172            Some(lora_rank) => {
173                let a = ReplicatedLayer::new(
174                    cfg.hidden_size,
175                    lora_rank,
176                    &cfg.quantization_config,
177                    cfg.attention_bias,
178                    mapper.set_device(layer_idx, vb.pp("q_a_proj"), loading_isq),
179                )?;

181                    lora_rank,
182                    cfg.rms_norm_eps,
183                    mapper.set_device(layer_idx, vb.pp("q_a_layernorm"), false),
184                )?;
185                let b = ColumnParallelLayer::new(
186                    lora_rank,
187                    cfg.num_attention_heads * q_head_dim,
188                    &cfg.quantization_config,
189                    false,
190                    comm,
191                    mapper.set_device(layer_idx, vb.pp("q_b_proj"), loading_isq),
192                )?;
193                QProj::Lora { a, norm, b }
194            }
195            None => QProj::Plain(ColumnParallelLayer::new(
196                cfg.hidden_size,
197                cfg.num_attention_heads * q_head_dim,
198                &cfg.quantization_config,
199                false,
200                comm,
201                mapper.set_device(layer_idx, vb.pp("q_proj"), loading_isq),
202            )?),
203        };
204
205        let kv_a_proj_with_mqa = ReplicatedLayer::new(
206            cfg.hidden_size,
207            cfg.kv_lora_rank + cfg.qk_rope_head_dim,
208            &cfg.quantization_config,
209            cfg.attention_bias,
210            mapper.set_device(layer_idx, vb.pp("kv_a_proj_with_mqa"), loading_isq),
211        )?;
212        let kv_a_layernorm = RmsNorm::new(
213            cfg.kv_lora_rank,
214            cfg.rms_norm_eps,
215            mapper.set_device(layer_idx, vb.pp("kv_a_layernorm"), false),
216        )?;
217        let kv_b_proj = ColumnParallelLayer::new(
218            cfg.kv_lora_rank,
219            cfg.num_attention_heads * (q_head_dim - cfg.qk_rope_head_dim + cfg.v_head_dim),
220            &cfg.quantization_config,
221            false,
222            comm,
223            mapper.set_device(layer_idx, vb.pp("kv_b_proj"), loading_isq),
224        )?;
225
226        let o_proj = RowParallelLayer::new(
227            cfg.num_attention_heads * cfg.v_head_dim,
228            cfg.hidden_size,
229            &cfg.quantization_config,
230            cfg.attention_bias,
231            comm,
232            mapper.set_device(layer_idx, vb.pp("o_proj"), loading_isq),
233        )?;
234
235        let mla_weights = MlaWeights::new(
236            paged_attn.is_some(),
237            mapper.device_for(layer_idx, loading_isq),
238        );
239
240        Ok(Self {
241            q,
242            kv_a_proj_with_mqa,
243            kv_a_layernorm,
244            kv_b_proj,
245            o_proj,
246            rotary_emb,
247            cfg: cfg.clone(),
248            q_head_dim,
249            paged_attn,
250            num_attention_heads: cfg.num_attention_heads / comm.world_size(),
251            sdpa_params: SdpaParams {
252                n_kv_groups: 1,
253                softcap: None,
254                softmax_scale: cfg.softmax_scale(),
255                sliding_window: None,
256            },
257            mla_weights,
258        })

261    fn forward(
262        &self,
263        xs: &Tensor,
264        attention_mask: Option<&Tensor>,
265        seqlen_offsets: &[usize],
266        kv_cache: &mut KvCache,
267        metadata: Option<((Tensor, Tensor), &PagedAttentionInputMetadata)>,
268        flash_params: &FlashParams,
269    ) -> Result<Tensor> {
270        let (bs, seq_len, _) = xs.dims3()?;
271
272        let mut q = self.q.forward(xs)?;
273        q = q
274            .reshape((bs, seq_len, self.num_attention_heads, self.q_head_dim))?
275            .transpose(1, 2)?;
276        let q_split = q.split(
277            &[self.cfg.qk_nope_head_dim, self.cfg.qk_rope_head_dim],
278            D::Minus1,
279        )?;
280        let q_nope = q_split[0].clone();
281        let mut q_pe = q_split[1].clone();
282
283        let mut compressed_kv = self.kv_a_proj_with_mqa.forward_autocast(xs)?;
284        let ckv_split = compressed_kv.split(
285            &[self.cfg.kv_lora_rank, self.cfg.qk_rope_head_dim],
286            D::Minus1,
287        )?;
288        compressed_kv = ckv_split[0].clone();
289        let mut k_pe = ckv_split[1].clone();
290        k_pe = k_pe
291            .reshape((bs, seq_len, 1, self.cfg.qk_rope_head_dim))?
292            .transpose(1, 2)?;
293
294        let ckv = self.kv_a_layernorm.forward(&compressed_kv)?;
295
296        (q_pe, k_pe) = self.rotary_emb.forward(&q_pe, &k_pe, seqlen_offsets)?;
297
298        let use_mla_decode = should_use_mla_decode(
299            attention_mask,
300            seq_len,
301            self.paged_attn.is_some(),
302            q_nope.device(),
303            &metadata,
304        );
305
306        let mut attn_out = if use_mla_decode {
307            mla_decode_forward(
308                &q_nope,
309                &q_pe,
310                &ckv,
311                &k_pe,
312                &metadata,
313                &self.mla_weights,
314                self.kv_b_proj.as_ref(),
315                &self.sdpa_params,
316                self.num_attention_heads,
317                self.cfg.kv_lora_rank,
318                self.cfg.qk_rope_head_dim,
319                self.cfg.qk_nope_head_dim,
320                self.cfg.v_head_dim,
321                bs,
322                seq_len,
323            )?
324        } else {
325            let mut kv = self.kv_b_proj.forward_autocast(&ckv)?;
326            kv = kv
327                .reshape((
328                    bs,
329                    seq_len,
330                    self.num_attention_heads,
331                    self.cfg.qk_nope_head_dim + self.cfg.v_head_dim,
332                ))?
333                .transpose(1, 2)?;
334
335            let kv_split =
336                kv.split(&[self.cfg.qk_nope_head_dim, self.cfg.v_head_dim], D::Minus1)?;
337            let k_nope = kv_split[0].clone();
338            let mut v = kv_split[1].clone();
339
340            let q = Tensor::cat(&[&q_nope, &q_pe], D::Minus1)?.contiguous()?;
341            let mut k = Tensor::cat(
342                &[&k_nope, &k_pe.repeat((1, self.num_attention_heads, 1, 1))?],
343                D::Minus1,
344            )?
345            .contiguous()?;
346
347            let use_mla_cache = should_use_mla_cache(self.paged_attn.is_some(), q.device());
348
349            if use_mla_cache {
350                mla_cache_forward(
351                    &q,
352                    &k,
353                    &v,
354                    &ckv,
355                    &k_pe,
356                    attention_mask,
357                    seqlen_offsets,
358                    &metadata,
359                    flash_params,
360                    self.kv_b_proj.as_ref(),
361                    &self.sdpa_params,
362                    self.num_attention_heads,
363                    self.cfg.kv_lora_rank,
364                    self.cfg.qk_rope_head_dim,
365                    self.cfg.qk_nope_head_dim,
366                    self.cfg.v_head_dim,
367                    bs,
368                    seq_len,
369                )?
370            } else {
371                match &self.paged_attn {
372                    Some(paged_attn) => match metadata {
373                        Some(((key_cache, value_cache), input_metadata)) => {
374                            let v = v
375                                .pad_with_zeros(
376                                    D::Minus1,
377                                    0,
378                                    self.q_head_dim - self.cfg.v_head_dim,
379                                )?
380                                .contiguous()?;
381                            paged_attn
382                                .forward(
383                                    &q,
384                                    &k,
385                                    &v,
386                                    attention_mask,
387                                    Some(key_cache),
388                                    Some(value_cache),
389                                    input_metadata,
390                                    &self.sdpa_params,
391                                    Some(flash_params),
392                                )?
393                                .narrow(D::Minus1, 0, self.cfg.v_head_dim)?
394                        }
395                        None => {
396                            // If we don't have metadata, we are most likely generating an imatrix so we don't want to populate that.
397                            // Generating the dummy metadata with the assumption that we are not generating text (only processing prompts).
398                            let input_metadata = PagedAttentionInputMetadata::dummy(q.device())?;
399                            // Sanity check.
400                            assert!(attention_mask.is_some());
401                            let v = v
402                                .pad_with_zeros(
403                                    D::Minus1,
404                                    0,
405                                    self.q_head_dim - self.cfg.v_head_dim,
406                                )?
407                                .contiguous()?;
408                            paged_attn
409                                .forward(
410                                    &q,
411                                    &k,
412                                    &v,
413                                    attention_mask,
414                                    None,
415                                    None,
416                                    &input_metadata,
417                                    &self.sdpa_params,
418                                    Some(flash_params),
419                                )?
420                                .narrow(D::Minus1, 0, self.cfg.v_head_dim)?
421                        }
422                    },
423                    None => {
424                        (k, v) = kv_cache.append(&k, &v)?;
425
426                        Sdpa.run_attention(
427                            &q,
428                            &k,
429                            &v,
430                            attention_mask,
431                            Some(flash_params),
432                            &self.sdpa_params,
433                        )?
434                    }
435                }
436            }
437        };
438
439        attn_out = if attention_mask.is_some() {
440            attn_out.transpose(1, 2)?.reshape((bs, seq_len, ()))?
441        } else {
442            attn_out.reshape((bs, seq_len, ()))?
443        };
444
445        self.o_proj.forward_autocast(&attn_out)
446    }
447}

449struct MoeGate {
450    weight: Tensor,
451    cfg: DeepSeekV2Config,
452    top_k: usize,
453    n_routed_experts: usize,
454}
455
456impl MoeGate {
457    fn new(cfg: &DeepSeekV2Config, vb: ShardedVarBuilder, n_routed_experts: usize) -> Result<Self> {
458        let weight = vb.get((n_routed_experts, cfg.hidden_size), "weight")?;
459        Ok(Self {
460            weight,
461            cfg: cfg.clone(),
462            top_k: cfg.num_experts_per_tok.unwrap(),
463            n_routed_experts,
464        })
465    }
466
467    /// (topk_idx, topk_weight)
468    fn forward(&self, xs: &Tensor) -> Result<(Tensor, Tensor)> {
469        let (bs, seq_len, h) = xs.dims3()?;
470        // Compute gating score
471        let xs = xs.reshape(((), h))?;
472        let logits = xs
473            .to_dtype(DType::F32)?
474            .broadcast_matmul(&self.weight.t()?.to_dtype(DType::F32)?)?;
475        let scores = match self.cfg.scoring_func {
476            ScoringFunc::Softmax => candle_nn::ops::softmax_last_dim(&logits)?,
477        };
478
479        // Select top-k experts
480        let (mut topk_weight, topk_idx) = match self.cfg.topk_method {
481            TopkMethod::Greedy => {
482                let TopKOutput { values, indices } = scores.topk_unsorted(self.top_k)?;
483                (values, indices)
484            }
485            TopkMethod::GroupLimitedGreedy => {
486                // (n, n_group)
487                let group_scores = scores
488                    .reshape((bs * seq_len, self.cfg.n_group, ()))?
489                    .max(D::Minus1)?;
490                // (n, topk_group)
491                let group_idx = group_scores.topk_unsorted(self.cfg.topk_group)?.indices;
492                // (n, n_group)
493                let mut group_mask = group_scores.zeros_like()?;
494                // (n, n_group)
495                group_mask = group_mask.scatter_add(
496                    &group_idx,
497                    &group_idx.ones_like()?.to_dtype(group_mask.dtype())?,
498                    1,
499                )?;
500                // (n, e)
501                let score_mask = group_mask
502                    .unsqueeze(D::Minus1)?
503                    .expand((
504                        bs * seq_len,
505                        self.cfg.n_group,
506                        self.n_routed_experts / self.cfg.n_group,
507                    ))?
508                    .reshape((bs * seq_len, ()))?;
509                // (n, e)
510                // Invert the mask
511                let tmp_scores = masked_fill(&score_mask, &(1. - &score_mask.ne(0.)?)?, 0.)?;
512                let TopKOutput { values, indices } = tmp_scores.topk_unsorted(self.top_k)?;
513                (values, indices)
514            }
515        };
516
517        if self.top_k > 1 && self.cfg.norm_topk_prob {
518            let denmoninator = (topk_weight.sum_keepdim(D::Minus1)? + 1e-20)?;
519            topk_weight = (topk_weight / denmoninator)?;
520        } else {
521            topk_weight = (topk_weight * self.cfg.routed_scaling_factor)?;
522        }
523        Ok((topk_idx, topk_weight))
524    }
525}

527struct Moe {
528    experts: MoEExperts,
529    shared_experts: Option<Mlp>,
530    gate: MoeGate,
531}
532
533impl Moe {
534    #[allow(clippy::too_many_arguments)]
535    fn new(
536        cfg: &DeepSeekV2Config,
537        vb: ShardedVarBuilder,
538        mapper: &dyn DeviceMapper,
539        layer_idx: usize,
540        loading_isq: bool,
541        n_shared_experts: Option<usize>,
542        n_routed_experts: usize,
543        comm: &Arc<mistralrs_quant::Comm>,
544        real_device: Device,
545    ) -> Result<Self> {
546        let layer_device = mapper
547            .device_for(layer_idx, false)
548            .cloned()
549            .unwrap_or(real_device);
550
551        let moe_cfg = MoEExpertsConfig {
552            num_experts: n_routed_experts,
553            num_experts_per_tok: cfg.num_experts_per_tok.unwrap(),
554            hidden_size: cfg.hidden_size,
555            moe_intermediate_size: cfg.moe_intermediate_size,
556        };
557
558        // Use the optimized MoEExperts with automatic backend selection
559        let experts = MoEExperts::new(
560            &moe_cfg,
561            mapper.set_device(layer_idx, vb.clone(), loading_isq),
562            layer_device,
563            comm,
564            loading_isq,
565            &cfg.quantization_config,
566            cfg.hidden_act,
567        )?;
568
569        let shared_experts = if let Some(n_shared_experts) = n_shared_experts {
570            let intermediate_size = cfg.moe_intermediate_size * n_shared_experts;
571            Some(Mlp::new(
572                mapper.set_device(layer_idx, vb.pp("shared_experts"), loading_isq),
573                cfg.hidden_size,
574                intermediate_size,
575                &cfg.quantization_config,
576                cfg.hidden_act,
577                comm,
578            )?)
579        } else {
580            None
581        };
582        let gate = MoeGate::new(
583            cfg,
584            mapper.set_device(layer_idx, vb.pp("gate"), false),
585            n_routed_experts,
586        )?;
587        Ok(Self {
588            experts,
589            shared_experts,
590            gate,
591        })
592    }
593
594    fn forward(&self, xs: &Tensor) -> Result<Tensor> {
595        let identity = xs.clone();
596        let (b_size, seq_len, hidden_dim) = xs.dims3()?;
597
598        // Get routing weights from gate
599        let (topk_idx, topk_weight) = self.gate.forward(xs)?;
600
601        // Forward through routed experts using optimized MoEExperts
602        let mut y = self.experts.forward(xs, topk_weight, &topk_idx)?;
603        y = y.reshape((b_size, seq_len, hidden_dim))?;
604
605        // Add shared expert output
606        if let Some(ref shared_experts) = self.shared_experts {
607            y = (y + shared_experts.forward(&identity)?)?;
608        }
609
610        Ok(y)
611    }
612
613    fn get_isq_layers(&mut self) -> Vec<&mut Arc<dyn QuantMethod>> {
614        let mut layers = self.experts.get_isq_layers();
615        if let Some(ref mut shared) = self.shared_experts {
616            layers.push(&mut shared.gate);
617            layers.push(&mut shared.up);
618            layers.push(&mut shared.down);
619        }
620        layers
621    }
622}

630    fn forward(&self, xs: &Tensor) -> Result<Tensor> {
631        match self {
632            Self::Mlp(mlp) => mlp.forward(xs),
633            Self::Moe(moe) => moe.forward(xs),
634        }
635    }
636}

638struct DecoderLayer {
639    input_layernorm: RmsNorm,
640    post_attention_layernorm: RmsNorm,
641    attn: Attention,
642    moe_or_mlp: MoeOrMlp,
643}
644
645impl DecoderLayer {
646    #[allow(clippy::too_many_arguments)]
647    fn new(
648        rotary_emb: Arc<DeepSeekV2RotaryEmbedding>,
649        cfg: &DeepSeekV2Config,
650        vb: ShardedVarBuilder,
651        mapper: &dyn DeviceMapper,
652        layer_idx: usize,
653        loading_isq: bool,
654        paged_attn: Option<PagedAttention>,
655        comm: &Arc<mistralrs_quant::Comm>,
656        real_device: Device,
657    ) -> Result<Self> {
658        let attn = Attention::new(
659            rotary_emb,
660            cfg,
661            vb.pp("self_attn"),
662            mapper,
663            layer_idx,
664            loading_isq,
665            paged_attn,
666            comm,
667        )?;
668        let input_layernorm = RmsNorm::new(
669            cfg.hidden_size,
670            cfg.rms_norm_eps,
671            mapper.set_device(layer_idx, vb.pp("input_layernorm"), false),
672        )?;
673        let post_attention_layernorm = RmsNorm::new(
674            cfg.hidden_size,
675            cfg.rms_norm_eps,
676            mapper.set_device(layer_idx, vb.pp("post_attention_layernorm"), false),
677        )?;
678        let moe_or_mlp = if let Some(n_routed_experts) = cfg.n_routed_experts.filter(|_| {
679            layer_idx >= cfg.first_k_dense_replace && layer_idx % cfg.moe_layer_freq == 0
680        }) {
681            MoeOrMlp::Moe(Box::new(Moe::new(
682                cfg,
683                vb.pp("mlp"),
684                mapper,
685                layer_idx,
686                loading_isq,
687                cfg.n_shared_experts,
688                n_routed_experts,
689                comm,
690                real_device,
691            )?))
692        } else {
693            MoeOrMlp::Mlp(Mlp::new(
694                mapper.set_device(layer_idx, vb.pp("mlp"), loading_isq),
695                cfg.hidden_size,
696                cfg.intermediate_size,
697                &cfg.quantization_config,
698                cfg.hidden_act,
699                comm,
700            )?)
701        };
702
703        Ok(Self {
704            input_layernorm,
705            post_attention_layernorm,
706            attn,
707            moe_or_mlp,
708        })
709    }
710
711    fn forward(
712        &self,
713        xs: &Tensor,
714        attention_mask: Option<&Tensor>,
715        seqlen_offsets: &[usize],
716        kv_cache: &mut KvCache,
717        metadata: Option<((Tensor, Tensor), &PagedAttentionInputMetadata)>,
718        flash_params: &FlashParams,
719    ) -> Result<Tensor> {
720        let residual = xs;
721        let xs = self.input_layernorm.forward(xs)?;
722        let xs = self.attn.forward(
723            &xs,
724            attention_mask,
725            seqlen_offsets,
726            kv_cache,
727            metadata,
728            flash_params,
729        )?;
730        let xs = (xs + residual)?;
731        let residual = &xs;
732        let xs = self
733            .moe_or_mlp
734            .forward(&xs.apply(&self.post_attention_layernorm)?)?;
735        residual + xs
736    }
737}

739pub struct DeepSeekV2 {
740    lm_head: Arc<dyn QuantMethod>,
741    embed_tokens: Embedding,
742    norm: RmsNorm,
743    layers: Vec<DecoderLayer>,
744    cache: EitherCache,

746    max_seq_len: usize,

751impl DeepSeekV2 {
752    pub fn new(
753        cfg: &DeepSeekV2Config,
754        vb: ShardedVarBuilder,
755        _is_gptx: bool,
756        normal_loading_metadata: NormalLoadingMetadata,
757        attention_mechanism: AttentionImplementation,
758    ) -> Result<Self> {
759        let vb_m = vb.pp("model");
760
761        let mapper = normal_loading_metadata.mapper;
762
763        let embed_tokens = embedding(
764            cfg.vocab_size,
765            cfg.hidden_size,
766            mapper.set_nm_device(vb_m.pp("embed_tokens"), false),
767            &cfg.quantization_config,
768        )?;
769        let lm_head = if !cfg.tie_word_embeddings {
770            ReplicatedLayer::new(
771                cfg.hidden_size,
772                cfg.vocab_size,
773                &cfg.quantization_config,
774                false,
775                mapper.set_nm_device(vb.pp("lm_head"), normal_loading_metadata.loading_isq),
776            )?
777        } else {
778            ReplicatedLayer::from_linear(candle_nn::Linear::new(
779                mapper.cast_nm_device(
780                    embed_tokens.embeddings(),
781                    normal_loading_metadata.loading_isq,
782                )?,
783                None,
784            ))?
785        };
786        let norm = RmsNorm::new(
787            cfg.hidden_size,
788            cfg.rms_norm_eps,
789            mapper.set_nm_device(vb_m.pp("norm"), false),
790        )?;
791
792        let mut ropes = HashMap::new();
793        let rope_cfg = DeepSeekV2RopeConfig {
794            rope_scaling: cfg.rope_scaling.clone(),
795            max_position_embeddings: cfg.max_position_embeddings,
796            rope_theta: cfg.rope_theta,
797            qk_rope_head_dim: cfg.qk_rope_head_dim,
798        };
799        for i in 0..cfg.num_hidden_layers {
800            let device = mapper
801                .device_for(i, false)
802                .unwrap_or(&normal_loading_metadata.real_device);
803            ropes.insert(
804                device.location(),
805                Arc::new(DeepSeekV2RotaryEmbedding::new(
806                    &rope_cfg,
807                    vb.dtype(),
808                    device,
809                )?),
810            );
811        }
812
813        let vb_l = vb_m.pp("layers");
814        let layers: Vec<DecoderLayer> = NiceProgressBar::<_, 'b'>(
815            0..cfg.num_hidden_layers,
816            "Loading repeating layers",
817            &normal_loading_metadata.multi_progress,
818        )
819        .par_iter_if_isq(|layer_idx| {
820            let device = mapper
821                .device_for(layer_idx, false)
822                .unwrap_or(&normal_loading_metadata.real_device);
823            let rotary_emb = ropes
824                .get(&device.location())
825                .expect("No RoPE for device location!")
826                .clone();
827            let paged_attn = match &attention_mechanism {
828                AttentionImplementation::Eager => None,
829                AttentionImplementation::PagedAttention => Some(
830                    PagedAttention::new(cfg.v_head_dim, device, None)
831                        .expect("Failed to create PagedAttention"),
832                ),
833            };
834            let comm = mapper.get_comm_for(layer_idx)?;
835            DecoderLayer::new(
836                rotary_emb.clone(),
837                cfg,
838                vb_l.pp(layer_idx),
839                &*mapper,
840                layer_idx,
841                normal_loading_metadata.loading_isq,
842                paged_attn,
843                &comm,
844                normal_loading_metadata.real_device.clone(),
845            )
846        })?;
847
848        Ok(Self {
849            lm_head,
850            embed_tokens,
851            norm,
852            layers,
853            cache: EitherCache::Normal(NormalCache::new(
854                cfg.num_hidden_layers,
855                cfg.max_position_embeddings,
856            )),
857            device: normal_loading_metadata.real_device.clone(),
858            max_seq_len: cfg.max_position_embeddings,
859            cfg: ModelConfigMetadata {
860                max_seq_len: cfg.max_position_embeddings,
861                num_layers: cfg.num_hidden_layers,
862                hidden_size: cfg.hidden_size,
863                num_kv_heads: (cfg.num_attention_heads / mapper.get_comm_for(0)?.world_size())
864                    .max(1),
865                num_attn_heads: (cfg.num_attention_heads / mapper.get_comm_for(0)?.world_size())
866                    .max(1),
867                sliding_window: None,
868                k_head_dim: cfg.q_head_dim(),
869                v_head_dim: if matches!(
870                    attention_mechanism,
871                    AttentionImplementation::PagedAttention
872                ) {
873                    cfg.q_head_dim()
874                } else {
875                    cfg.v_head_dim
876                },
877                #[cfg(all(feature = "cuda", target_family = "unix"))]
878                kv_cache_layout: if matches!(
879                    attention_mechanism,
880                    AttentionImplementation::PagedAttention
881                ) {
882                    crate::paged_attention::KvCacheLayout::Mla {
883                        kv_lora_rank: cfg.kv_lora_rank,
884                        kpe_head_dim: cfg.qk_rope_head_dim,
885                    }
886                } else {
887                    crate::paged_attention::KvCacheLayout::Standard
888                },
889                #[cfg(not(all(feature = "cuda", target_family = "unix")))]
890                kv_cache_layout: crate::paged_attention::KvCacheLayout::Standard,
891            },
892            mapper,
893        })
894    }
895
896    #[allow(clippy::too_many_arguments)]
897    pub fn forward(
898        &self,
899        input_ids: &Tensor,
900        seqlen_offsets: &[usize],
901        context_lens: Vec<(usize, usize)>,
902        metadata: Option<(Vec<(Tensor, Tensor)>, &PagedAttentionInputMetadata)>,
903        flash_params: &FlashParams,
904    ) -> Result<Tensor> {
905        let mut xs = self.embed_tokens.forward(input_ids)?;
906        let cache = &mut self.cache.normal().0;
907        let attention_mask = CausalMasker.make_causal_mask_matrix(
908            input_ids,
909            metadata
910                .as_ref()
911                .map(|(_, _)| &seqlen_offsets as &dyn PastKvLenCache)
912                .unwrap_or(cache as &dyn PastKvLenCache),
913            xs.dtype(),
914            self.cfg.num_attn_heads,
915        )?;
916        // PagedAttention prompt chunking
917        let attention_mask = attention_mask.filter(|_| {
918            metadata
919                .as_ref()
920                .map(|(_, meta)| meta.is_first_prompt_chunk)
921                .unwrap_or(true)
922        });
923        for (i, layer) in self.layers.iter().enumerate() {
924            xs = self.mapper.map(xs, i)?;
925            xs = layer.forward(
926                &xs,
927                attention_mask
928                    .as_ref()
929                    .map(|m| m.to_device(xs.device()).unwrap())
930                    .as_ref(),
931                seqlen_offsets,
932                &mut cache[i],
933                metadata
934                    .as_ref()
935                    .map(|(kv_cache, metadata)| (kv_cache[i].clone(), *metadata)),
936                flash_params,
937            )?;
938        }
939        let xs = xs.to_device(&self.device)?;
940        let xs = xs.apply(&self.norm)?;
941        extract_logits(&self.lm_head.forward_autocast(&xs)?, context_lens)
942    }
943}

946    fn get_layers(
947        &mut self,
948    ) -> (
949        Vec<(&mut Arc<dyn QuantMethod>, Option<usize>)>,
950        &dyn DeviceMapper,
951    ) {
952        let mut tensors = Vec::new();
953        tensors.push((&mut self.lm_head, None));
954        for (i, layer) in self.layers.iter_mut().enumerate() {
955            match &mut layer.attn.q {
956                QProj::Plain(q) => {
957                    tensors.push((q, Some(i)));
958                }
959                QProj::Lora { a, norm: _, b } => {
960                    tensors.push((a, Some(i)));
961                    tensors.push((b, Some(i)));
962                }
963            }
964            tensors.push((&mut layer.attn.kv_a_proj_with_mqa, Some(i)));
965            tensors.push((&mut layer.attn.kv_b_proj, Some(i)));
966            tensors.push((&mut layer.attn.o_proj, Some(i)));
967            match &mut layer.moe_or_mlp {
968                MoeOrMlp::Mlp(mlp) => {
969                    tensors.push((&mut mlp.gate, Some(i)));
970                    tensors.push((&mut mlp.up, Some(i)));
971                    tensors.push((&mut mlp.down, Some(i)));
972                }
973                MoeOrMlp::Moe(moe) => {
974                    for layer in moe.get_isq_layers() {
975                        tensors.push((layer, Some(i)));
976                    }
977                }
978            }
979        }
980        (tensors, &*self.mapper)
981    }
