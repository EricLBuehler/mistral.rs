30macro_rules! sliding_window {
31    ($layer_idx:expr, $cfg:expr) => {
32        if !($cfg.sliding_window.is_some()
33            && $cfg.use_sliding_window
34            && $layer_idx >= $cfg.max_window_layers)
35        {
36            None
37        } else {
38            $cfg.sliding_window
39        }
40    };
41}

94impl Attention {
95    #[allow(clippy::too_many_arguments)]
96    fn new(
97        rotary_emb: Arc<RotaryEmbedding>,
98        cfg: &Config,
99        vb: ShardedVarBuilder,
100        mapper: &dyn DeviceMapper,
101        layer_idx: usize,
102        loading_isq: bool,
103        paged_attn: Option<PagedAttention>,
104        comm: &Arc<mistralrs_quant::Comm>,
105    ) -> Result<Self> {
106        let hidden_sz = cfg.hidden_size;
107        let num_heads = cfg.num_attention_heads;
108        let num_kv_heads = cfg.num_key_value_heads;
109        let head_dim = cfg.head_dim();
110        let q_proj = ColumnParallelLayer::new(
111            hidden_sz,
112            num_heads * head_dim,
113            &cfg.quantization_config,
114            false,
115            comm,
116            mapper.set_device(layer_idx, vb.pp("q_proj"), loading_isq),
117        )?;
118        let kv_shard = mistralrs_quant::compute_kv_shard(
119            cfg.num_key_value_heads,
120            cfg.hidden_size / cfg.num_attention_heads,
121            comm,
122        );
123        let k_proj = ColumnParallelLayer::new_with_shard(
124            hidden_sz,
125            num_kv_heads * head_dim,
126            &cfg.quantization_config,
127            false,
128            comm,
129            kv_shard,
130            mapper.set_device(layer_idx, vb.pp("k_proj"), loading_isq),
131        )?;
132        let v_proj = ColumnParallelLayer::new_with_shard(
133            hidden_sz,
134            num_kv_heads * head_dim,
135            &cfg.quantization_config,
136            false,
137            comm,
138            kv_shard,
139            mapper.set_device(layer_idx, vb.pp("v_proj"), loading_isq),
140        )?;
141        let o_proj = RowParallelLayer::new(
142            num_heads * head_dim,
143            hidden_sz,
144            &cfg.quantization_config,
145            false,
146            comm,
147            mapper.set_device(layer_idx, vb.pp("o_proj"), loading_isq),
148        )?;
149        let sliding_window = sliding_window!(layer_idx, cfg);
150        let q_norm = RmsNorm::new(
151            cfg.head_dim(),
152            cfg.rms_norm_eps,
153            mapper.set_device(layer_idx, vb.pp("q_norm"), false),
154        )?;
155        let k_norm = RmsNorm::new(
156            cfg.head_dim(),
157            cfg.rms_norm_eps,
158            mapper.set_device(layer_idx, vb.pp("k_norm"), false),
159        )?;
160        Ok(Self {
161            q_proj,
162            k_proj,
163            v_proj,
164            o_proj,
165            q_norm,
166            k_norm,
167            num_heads: num_heads / comm.world_size(),
168            num_kv_heads: (num_kv_heads / comm.world_size()).max(1),
169            head_dim,
170            rotary_emb,
171            paged_attn,
172            sdpa_params: SdpaParams {
173                n_kv_groups: mistralrs_quant::compute_n_kv_groups(
174                    cfg.num_key_value_heads,
175                    cfg.num_attention_heads,
176                    comm,
177                ),
178                softcap: None,
179                softmax_scale: 1.0 / (head_dim as f32).sqrt(),
180                sliding_window,
181            },
182        })
183    }
184
185    #[allow(clippy::too_many_arguments)]
186    fn forward(
187        &self,
188        xs: &Tensor,
189        attention_mask: Option<&Tensor>,
190        seqlen_offsets: &[usize],
191        kv_cache: &mut KvCache,
192        metadata: Option<((Tensor, Tensor), &PagedAttentionInputMetadata)>,
193        flash_params: &FlashParams,
194    ) -> Result<Tensor> {
195        let (b_sz, q_len, _) = xs.dims3()?;
196
197        let original_dtype = xs.dtype();
198        let mut xs = xs.clone();
199        if let Some(t) = self.q_proj.quantized_act_type() {
200            xs = xs.to_dtype(t)?;
201        }
202        let mut q = MatMul.qmethod_matmul(&xs, &*self.q_proj)?;
203        let mut k = MatMul.qmethod_matmul(&xs, &*self.k_proj)?;
204        let mut v = MatMul.qmethod_matmul(&xs, &*self.v_proj)?;
205        if self.q_proj.quantized_act_type().is_some() {
206            q = q.to_dtype(original_dtype)?;
207            k = k.to_dtype(original_dtype)?;
208            v = v.to_dtype(original_dtype)?;
209        }
210
211        (q, k, v) = if q_len != 1 {
212            let q = q
213                .reshape((b_sz, q_len, self.num_heads, self.head_dim))?
214                .transpose(1, 2)?;
215            let k = k
216                .reshape((b_sz, q_len, self.num_kv_heads, self.head_dim))?
217                .transpose(1, 2)?;
218            let v = v
219                .reshape((b_sz, q_len, self.num_kv_heads, self.head_dim))?
220                .transpose(1, 2)?;
221            (q, k, v)
222        } else {
223            let q = q.reshape((b_sz, self.num_heads, q_len, self.head_dim))?;
224            let k = k.reshape((b_sz, self.num_kv_heads, q_len, self.head_dim))?;
225            let v = v.reshape((b_sz, self.num_kv_heads, q_len, self.head_dim))?;
226            (q, k, v)
227        };
228
229        q = q.apply(&self.q_norm)?;
230        k = k.apply(&self.k_norm)?;
231
232        (q, k) = self.rotary_emb.forward(&q, &k, seqlen_offsets)?;
233
234        let mut attn_output = match &self.paged_attn {
235            Some(paged_attn) => match metadata {
236                Some(((key_cache, value_cache), input_metadata)) => paged_attn.forward(
237                    &q,
238                    &k,
239                    &v,
240                    attention_mask,
241                    Some(key_cache),
242                    Some(value_cache),
243                    input_metadata,
244                    &self.sdpa_params,
245                    Some(flash_params),
246                )?,
247                None => {
248                    // If we don't have metadata, we are most likely generating an imatrix so we don't want to populate that.
249                    // Generating the dummy metadata with the assumption that we are not generating text (only processing prompts).
250                    let input_metadata = PagedAttentionInputMetadata::dummy(q.device())?;
251                    // Sanity check.
252                    assert!(attention_mask.is_some());
253                    paged_attn.forward(
254                        &q,
255                        &k,
256                        &v,
257                        attention_mask,
258                        None,
259                        None,
260                        &input_metadata,
261                        &self.sdpa_params,
262                        Some(flash_params),
263                    )?
264                }
265            },
266            None => {
267                let (k, v) = kv_cache.append(&k, &v)?;
268
269                Sdpa.run_attention(
270                    &q,
271                    &k,
272                    &v,
273                    attention_mask,
274                    Some(flash_params),
275                    &self.sdpa_params,
276                )?
277            }
278        };
279
280        if let Some(t) = self.q_proj.quantized_act_type() {
281            attn_output = attn_output.to_dtype(t)?;
282        }
283        attn_output = if attention_mask.is_some() {
284            attn_output.transpose(1, 2)?.reshape((b_sz, q_len, ()))?
285        } else {
286            attn_output.reshape((b_sz, q_len, ()))?
287        };
288        let mut res = MatMul.qmethod_matmul(&attn_output, &*self.o_proj)?;
289        if self.q_proj.quantized_act_type().is_some() {
290            res = res.to_dtype(original_dtype)?;
291        }
292        Ok(res)
293    }
294}

296#[derive(Clone)]
297struct Mlp {
298    gate_proj: Arc<dyn QuantMethod>,
299    up_proj: Arc<dyn QuantMethod>,
300    down_proj: Arc<dyn QuantMethod>,
301    act_fn: Activation,
302}
303
304impl Mlp {
305    fn new(
306        cfg: &Config,
307        vb: ShardedVarBuilder,
308        comm: &Arc<mistralrs_quant::Comm>,
309        i_size: usize,
310    ) -> Result<Self> {
311        let hidden_size = cfg.hidden_size;
312
313        let gate_proj = ColumnParallelLayer::new(
314            hidden_size,
315            i_size,
316            &cfg.quantization_config,
317            false,
318            comm,
319            vb.pp("gate_proj"),
320        )?;
321        let up_proj = RowParallelLayer::new(
322            hidden_size,
323            i_size,
324            &cfg.quantization_config,
325            false,
326            comm,
327            vb.pp("up_proj"),
328        )?;
329        let down_proj = ColumnParallelLayer::new(
330            i_size,
331            hidden_size,
332            &cfg.quantization_config,
333            false,
334            comm,
335            vb.pp("down_proj"),
336        )?;
337
338        Ok(Self {
339            gate_proj,
340            up_proj,
341            down_proj,
342            act_fn: cfg.hidden_act,
343        })
344    }
345
346    fn forward(&self, xs: &Tensor) -> Result<Tensor> {
347        let original_dtype = xs.dtype();
348        let mut xs = xs.clone();
349        if let Some(t) = self.gate_proj.quantized_act_type() {
350            xs = xs.to_dtype(t)?;
351        }
352        let gate_out = MatMul.qmethod_matmul(&xs, &*self.gate_proj)?;
353        let up_out = MatMul.qmethod_matmul(&xs, &*self.up_proj)?;
354        let current_hidden_states = crate::ops::mul_and_act(&gate_out, &up_out, self.act_fn)?;
355        let mut res = MatMul.qmethod_matmul(&current_hidden_states, &*self.down_proj)?;
356        if self.gate_proj.quantized_act_type().is_some() {
357            res = res.to_dtype(original_dtype)?;
358        }
359        Ok(res)
360    }
361}

363/// MoE MLP layer for Qwen3 MoE
364struct MoeMlp {
365    gate: Linear,
366    experts: MoEExperts,
367    num_experts_per_tok: usize,
368    norm_topk_prob: bool,
369}
370
371impl MoeMlp {
372    fn new(
373        cfg: &Config,
374        vb: ShardedVarBuilder,
375        layer_device: Device,
376        comm: &Arc<mistralrs_quant::Comm>,
377        loading_isq: bool,
378    ) -> Result<Self> {
379        // Load gate
380        let gate = layers::linear_no_bias(
381            cfg.hidden_size,
382            cfg.num_experts,
383            vb.pp("gate").set_device(layer_device.clone()),
384        )?;
385
386        let moe_cfg = MoEExpertsConfig {
387            num_experts: cfg.num_experts,
388            num_experts_per_tok: cfg.num_experts_per_tok,
389            hidden_size: cfg.hidden_size,
390            moe_intermediate_size: cfg.moe_intermediate_size,
391        };
392
393        // Load experts with automatic backend selection
394        let experts = MoEExperts::new(
395            &moe_cfg,
396            vb,
397            layer_device,
398            comm,
399            loading_isq,
400            &cfg.quantization_config,
401            cfg.hidden_act,
402        )?;
403
404        Ok(Self {
405            gate,
406            experts,
407            num_experts_per_tok: cfg.num_experts_per_tok,
408            norm_topk_prob: cfg.norm_topk_prob,
409        })
410    }
411
412    fn forward(&self, xs: &Tensor) -> Result<Tensor> {
413        let (b_size, seq_len, hidden_dim) = xs.dims3()?;
414        let xs_flat = xs.reshape(((), hidden_dim))?;
415
416        // Compute routing weights
417        let router_logits = self.gate.forward(&xs_flat)?;
418        let routing_weights =
419            candle_nn::ops::softmax_last_dim(&router_logits.to_dtype(DType::F32)?)?;
420
421        // Get top-k experts
422        let topk_ids = routing_weights
423            .arg_sort_last_dim(false)?
424            .narrow(D::Minus1, 0, self.num_experts_per_tok)?
425            .contiguous()?;
426
427        let mut topk_weights = routing_weights.gather(&topk_ids, D::Minus1)?;
428
429        if self.norm_topk_prob {
430            topk_weights = topk_weights.broadcast_div(&topk_weights.sum_keepdim(D::Minus1)?)?;
431        }
432
433        // Forward through experts (is_prefill determined internally based on seq_len)
434        let ys = self.experts.forward(xs, topk_weights, &topk_ids)?;
435
436        ys.reshape((b_size, seq_len, hidden_dim))
437    }
438
439    fn gate(&self) -> &Linear {
440        &self.gate
441    }
442
443    fn get_isq_layers(&mut self) -> Vec<&mut Arc<dyn QuantMethod>> {
444        self.experts.get_isq_layers()
445    }
446}

462struct DecoderLayer {
463    self_attn: Attention,
464    mlp: MoeOrMlp,
465    input_layernorm: RmsNorm,
466    post_attention_layernorm: RmsNorm,
467}
468
469impl DecoderLayer {
470    #[allow(clippy::too_many_arguments)]
471    fn new(
472        rotary_emb: Arc<RotaryEmbedding>,
473        cfg: &Config,
474        vb: ShardedVarBuilder,
475        mapper: &dyn DeviceMapper,
476        layer_idx: usize,
477        loading_isq: bool,
478        paged_attn: Option<PagedAttention>,
479        real_device: Device,
480        comm: &Arc<mistralrs_quant::Comm>,
481    ) -> Result<Self> {
482        let self_attn = Attention::new(
483            rotary_emb,
484            cfg,
485            mapper.set_device(layer_idx, vb.pp("self_attn"), loading_isq),
486            mapper,
487            layer_idx,
488            loading_isq,
489            paged_attn,
490            comm,
491        )?;
492
493        let mlp = if !cfg.mlp_only_layers.contains(&layer_idx)
494            && (cfg.num_experts > 0 && (layer_idx + 1) % cfg.decoder_sparse_step == 0)
495        {
496            let vb = mapper.set_device(layer_idx, vb.pp("mlp"), loading_isq);
497            let layer_device = mapper
498                .device_for(layer_idx, false)
499                .cloned()
500                .unwrap_or(real_device);
501
502            MoeOrMlp::Moe(MoeMlp::new(cfg, vb, layer_device, comm, loading_isq)?)
503        } else {
504            MoeOrMlp::Mlp(Mlp::new(
505                cfg,
506                mapper.set_device(layer_idx, vb.pp("mlp"), loading_isq),
507                comm,
508                cfg.intermediate_size,
509            )?)
510        };
511        let input_layernorm = RmsNorm::new(
512            cfg.hidden_size,
513            cfg.rms_norm_eps,
514            mapper.set_device(layer_idx, vb.pp("input_layernorm"), false),
515        )?;
516        let post_attention_layernorm = RmsNorm::new(
517            cfg.hidden_size,
518            cfg.rms_norm_eps,
519            mapper.set_device(layer_idx, vb.pp("post_attention_layernorm"), false),
520        )?;
521        Ok(Self {
522            self_attn,
523            mlp,
524            input_layernorm,
525            post_attention_layernorm,
526        })
527    }
528
529    #[allow(clippy::too_many_arguments)]
530    fn forward(
531        &self,
532        xs: &Tensor,
533        attention_mask: Option<&Tensor>,
534        seqlen_offsets: &[usize],
535        kv_cache: &mut KvCache,
536        metadata: Option<((Tensor, Tensor), &PagedAttentionInputMetadata)>,
537        flash_params: &FlashParams,
538    ) -> Result<Tensor> {
539        let residual = xs;
540        let xs = self.input_layernorm.forward(xs)?;
541        let xs = self.self_attn.forward(
542            &xs,
543            attention_mask,
544            seqlen_offsets,
545            kv_cache,
546            metadata,
547            flash_params,
548        )?;
549        let xs = (xs + residual)?;
550        let residual = &xs;
551        let xs = self
552            .mlp
553            .forward(&xs.apply(&self.post_attention_layernorm)?)?;
554        residual + xs
555    }
556}

571impl Model {
572    pub fn new(
573        cfg: &Config,
574        vb: ShardedVarBuilder,
575        is_gptx: bool,
576        normal_loading_metadata: NormalLoadingMetadata,
577        attention_mechanism: AttentionImplementation,
578    ) -> Result<Self> {
579        let vb_m = vb.pp("model");
580        let vb_lm_head = vb.pp("lm_head");
581        Self::new_inner(
582            cfg,
583            vb_m,
584            vb_lm_head,
585            is_gptx,
586            normal_loading_metadata,
587            attention_mechanism,
588        )

590
591    pub fn new_inner(
592        cfg: &Config,
593        vb_m: ShardedVarBuilder,
594        vb_lm_head: ShardedVarBuilder,
595        is_gptx: bool,
596        normal_loading_metadata: NormalLoadingMetadata,
597        attention_mechanism: AttentionImplementation,
598    ) -> Result<Self> {
599        if let Some(ref quant_cfg) = &cfg.quantization_config {
600            tracing::info!(
601                "Using {} quantization: {}.",
602                quant_cfg.name(),
603                quant_cfg.get_bits_name(&vb_m)
604            );
605        }
606        let mapper = normal_loading_metadata.mapper;
607
608        let embed_tokens = embedding(
609            cfg.vocab_size,
610            cfg.hidden_size,
611            mapper.set_nm_device(vb_m.pp("embed_tokens"), false),
612            &cfg.quantization_config,
613        )?;
614
615        let head_dim = cfg.head_dim();
616        let mut ropes = HashMap::new();
617        for layer_idx in 0..cfg.num_hidden_layers {
618            let device = mapper
619                .device_for(layer_idx, false)
620                .unwrap_or(&normal_loading_metadata.real_device);
621            ropes.insert(
622                device.location(),
623                Arc::new(RotaryEmbedding::new(
624                    cfg.rope_theta as f32,
625                    head_dim,
626                    cfg.max_position_embeddings,
627                    device,
628                    is_gptx,
629                    vb_m.dtype(),
630                )?),
631            );
632        }
633
634        let load_in_parallel =
635            !(normal_loading_metadata.real_device.is_metal() && cfg.quantization_config.is_none());
636        let vb_l = vb_m.pp("layers");
637        let layers: Vec<DecoderLayer> = NiceProgressBar::<_, 'b'>(
638            0..cfg.num_hidden_layers,
639            "Loading repeating layers",
640            &normal_loading_metadata.multi_progress,
641        )
642        .run(load_in_parallel, |layer_idx| {
643            let device = mapper
644                .device_for(layer_idx, false)
645                .unwrap_or(&normal_loading_metadata.real_device);
646            let rotary_emb = ropes
647                .get(&device.location())
648                .expect("No RoPE for device location!")
649                .clone();
650            let paged_attn = match &attention_mechanism {
651                AttentionImplementation::Eager => None,
652                AttentionImplementation::PagedAttention => Some(
653                    PagedAttention::new(head_dim, device, None)
654                        .expect("PagedAttention creation failed"),
655                ),
656            };
657            let comm = mapper
658                .get_comm_for(layer_idx)
659                .expect("Failed to get comm for layer");
660            DecoderLayer::new(
661                rotary_emb.clone(),
662                cfg,
663                vb_l.pp(layer_idx),
664                &*mapper,
665                layer_idx,
666                normal_loading_metadata.loading_isq,
667                paged_attn,
668                normal_loading_metadata.real_device.clone(),
669                &comm,
670            )
671        })?;
672        let norm = RmsNorm::new(
673            cfg.hidden_size,
674            cfg.rms_norm_eps,
675            mapper.set_nm_device(vb_m.pp("norm"), false),
676        )?;
677        let lm_head = if !cfg.tie_word_embeddings {
678            ReplicatedLayer::new(
679                cfg.hidden_size,
680                cfg.vocab_size,
681                &cfg.quantization_config,
682                false,
683                mapper.set_nm_device(vb_lm_head, normal_loading_metadata.loading_isq),
684            )?
685        } else {
686            ReplicatedLayer::from_linear(candle_nn::Linear::new(
687                mapper.cast_nm_device(
688                    embed_tokens.embeddings(),
689                    normal_loading_metadata.loading_isq,
690                )?,
691                None,
692            ))?
693        };
694        let cache_types = (0..cfg.num_hidden_layers)
695            .map(|layer_idx| {
696                sliding_window!(layer_idx, cfg)
697                    .map(|window| NormalCacheType::SlidingWindow { window })
698                    .unwrap_or(NormalCacheType::Normal {
699                        max_seq_len: cfg.max_position_embeddings,
700                    })
701            })
702            .collect::<Vec<_>>();
703        Ok(Self {
704            embed_tokens,
705            layers,
706            norm,
707            lm_head,
708            sliding_window: cfg.sliding_window,
709            device: normal_loading_metadata.real_device,
710            cache: EitherCache::Normal(NormalCache::from_types(cache_types)),
711            max_seq_len: cfg.max_position_embeddings,
712            cfg: ModelConfigMetadata {
713                max_seq_len: cfg.max_position_embeddings,
714                num_layers: cfg.num_hidden_layers,
715                hidden_size: cfg.hidden_size,
716                num_kv_heads: (cfg.num_key_value_heads / mapper.get_comm_for(0)?.world_size())
717                    .max(1),
718                num_attn_heads: cfg.num_attention_heads / mapper.get_comm_for(0)?.world_size(),
719                sliding_window: cfg.sliding_window,
720                k_head_dim: cfg.head_dim(),
721                v_head_dim: cfg.head_dim(),
722            },
723            mapper,
724        })
725    }
726
727    pub fn forward(
728        &self,
729        input_ids: &Tensor,
730        seqlen_offsets: &[usize],
731        context_lens: Vec<(usize, usize)>,
732        metadata: Option<(Vec<(Tensor, Tensor)>, &PagedAttentionInputMetadata)>,
733        flash_params: &FlashParams,
734    ) -> Result<Tensor> {
735        self.forward_embeds(
736            input_ids,
737            self.embed_tokens.forward(input_ids)?,
738            seqlen_offsets,
739            context_lens,
740            metadata,
741            flash_params,
742        )
743    }
744
745    #[allow(clippy::too_many_arguments)]
746    pub fn forward_embeds(
747        &self,
748        input_ids: &Tensor,
749        input_embeds: Tensor,
750        seqlen_offsets: &[usize],
751        context_lens: Vec<(usize, usize)>,
752        metadata: Option<(Vec<(Tensor, Tensor)>, &PagedAttentionInputMetadata)>,
753        flash_params: &FlashParams,
754    ) -> Result<Tensor> {
755        let mut xs = input_embeds;
756        let cache = &mut self.cache.normal().0;
757        let attention_mask = CausalMasker.make_sliding_window_causal_mask_matrix(
758            input_ids,
759            metadata
760                .as_ref()
761                .map(|(_, _)| &seqlen_offsets as &dyn PastKvLenCache)
762                .unwrap_or(cache as &dyn PastKvLenCache),
763            self.sliding_window,
764            xs.dtype(),
765            self.cfg.num_attn_heads,
766        )?;
767        // PagedAttention prompt chunking
768        let attention_mask = attention_mask.filter(|_| {
769            metadata
770                .as_ref()
771                .map(|(_, meta)| meta.is_first_prompt_chunk)
772                .unwrap_or(true)
773        });
774        for (i, layer) in self.layers.iter().enumerate() {
775            xs = self.mapper.map(xs, i)?;
776            xs = layer.forward(
777                &xs,
778                attention_mask
779                    .as_ref()
780                    .map(|m| m.to_device(xs.device()).unwrap())
781                    .as_ref(),
782                seqlen_offsets,
783                &mut cache[i],
784                metadata
785                    .as_ref()
786                    .map(|(kv_cache, metadata)| (kv_cache[i].clone(), *metadata)),
787                flash_params,
788            )?;
789            // dbg!(&i);
790        }
791        let xs = xs.to_device(&self.device)?;
792        let mut xs = xs.apply(&self.norm)?;
793        if let Some(t) = self.lm_head.quantized_act_type() {
794            xs = xs.to_dtype(t)?;
795        }
796        extract_logits(&MatMul.qmethod_matmul(&xs, &*self.lm_head)?, context_lens)
797    }
798}

800impl IsqModel for Model {
801    fn get_layers(
802        &mut self,
803    ) -> (
804        Vec<(&mut Arc<dyn QuantMethod>, Option<usize>)>,
805        &dyn DeviceMapper,
806    ) {
807        let mut tensors = Vec::new();
808        tensors.push((&mut self.lm_head, None));
809        for (i, layer) in self.layers.iter_mut().enumerate() {
810            tensors.push((&mut layer.self_attn.q_proj, Some(i)));
811            tensors.push((&mut layer.self_attn.k_proj, Some(i)));
812            tensors.push((&mut layer.self_attn.v_proj, Some(i)));
813            tensors.push((&mut layer.self_attn.o_proj, Some(i)));
814            match &mut layer.mlp {
815                MoeOrMlp::Mlp(mlp) => {
816                    tensors.push((&mut mlp.gate_proj, Some(i)));
817                    tensors.push((&mut mlp.up_proj, Some(i)));
818                    tensors.push((&mut mlp.down_proj, Some(i)));
819                }
820                MoeOrMlp::Moe(moe) => {
821                    for layer in moe.get_isq_layers() {
822                        tensors.push((layer, Some(i)));
823                    }
824                }
825            }
826        }
827        (tensors, &*self.mapper)
828    }
829
830    fn residual_tensors(&self) -> Vec<(String, Tensor)> {
831        let uvb = UnVarBuilder::new();
832
833        let uvb_m = uvb.pp("model");
834        uvb_m.pp("embed_tokens").add(&self.embed_tokens);
835        uvb_m.pp("norm").add(&self.norm);
836
837        for (layer_idx, layer) in self.layers.iter().enumerate() {
838            let uvb_l = uvb_m.pp("layers").pp(layer_idx);
839            uvb_l.pp("input_layernorm").add(&layer.input_layernorm);
840            uvb_l
841                .pp("post_attention_layernorm")
842                .add(&layer.post_attention_layernorm);
843            uvb_l
844                .pp("self_attn")
845                .pp("q_norm")
846                .add(&layer.self_attn.q_norm);
847            uvb_l
848                .pp("self_attn")
849                .pp("k_norm")
850                .add(&layer.self_attn.k_norm);
851            if let MoeOrMlp::Moe(moe) = &layer.mlp {
852                uvb_l.pp("mlp").pp("gate").add(moe.gate());
853            }
854        }
855
856        uvb.to_safetensors()
857    }
858}
