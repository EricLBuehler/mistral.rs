28macro_rules! sliding_window {
29    ($layer_idx:expr, $cfg:expr) => {
30        if !($cfg.sliding_window.is_some()
31            && $cfg.use_sliding_window
32            && $layer_idx > $cfg.max_window_layers)
33        {
34            None
35        } else {
36            $cfg.sliding_window
37        }
38    };
39}

93impl Attention {
94    #[allow(clippy::too_many_arguments)]
95    fn new(
96        rotary_emb: Arc<RotaryEmbedding>,
97        cfg: &Config,
98        vb: ShardedVarBuilder,
99        mapper: &dyn DeviceMapper,
100        layer_idx: usize,
101        loading_isq: bool,
102        paged_attn: Option<PagedAttention>,
103        comm: &Arc<mistralrs_quant::Comm>,
104    ) -> Result<Self> {
105        let hidden_sz = cfg.hidden_size;
106        let num_heads = cfg.num_attention_heads;
107        let num_kv_heads = cfg.num_key_value_heads;
108        let head_dim = cfg.head_dim();
109        let q_proj = ColumnParallelLayer::new(
110            hidden_sz,
111            num_heads * head_dim,
112            &cfg.quantization_config,
113            false,
114            comm,
115            mapper.set_device(layer_idx, vb.pp("q_proj"), loading_isq),
116        )?;
117        let kv_shard = mistralrs_quant::compute_kv_shard(
118            cfg.num_key_value_heads,
119            cfg.hidden_size / cfg.num_attention_heads,
120            comm,
121        );
122        let k_proj = ColumnParallelLayer::new_with_shard(
123            hidden_sz,
124            num_kv_heads * head_dim,
125            &cfg.quantization_config,
126            false,
127            comm,
128            kv_shard,
129            mapper.set_device(layer_idx, vb.pp("k_proj"), loading_isq),
130        )?;
131        let v_proj = ColumnParallelLayer::new_with_shard(
132            hidden_sz,
133            num_kv_heads * head_dim,
134            &cfg.quantization_config,
135            false,
136            comm,
137            kv_shard,
138            mapper.set_device(layer_idx, vb.pp("v_proj"), loading_isq),
139        )?;
140        let o_proj = RowParallelLayer::new(
141            num_heads * head_dim,
142            hidden_sz,
143            &cfg.quantization_config,
144            false,
145            comm,
146            mapper.set_device(layer_idx, vb.pp("o_proj"), loading_isq),
147        )?;
148        let sliding_window = sliding_window!(layer_idx, cfg);
149        let q_norm = RmsNorm::new(
150            cfg.head_dim(),
151            cfg.rms_norm_eps,
152            mapper.set_device(layer_idx, vb.pp("q_norm"), false),
153        )?;
154        let k_norm = RmsNorm::new(
155            cfg.head_dim(),
156            cfg.rms_norm_eps,
157            mapper.set_device(layer_idx, vb.pp("k_norm"), false),
158        )?;
159        Ok(Self {
160            q_proj,
161            k_proj,
162            v_proj,
163            o_proj,
164            q_norm,
165            k_norm,
166            num_heads: num_heads / comm.world_size(),
167            num_kv_heads: (num_kv_heads / comm.world_size()).max(1),
168            head_dim,
169            rotary_emb,
170            paged_attn,
171            sdpa_params: SdpaParams {
172                n_kv_groups: mistralrs_quant::compute_n_kv_groups(
173                    cfg.num_key_value_heads,
174                    cfg.num_attention_heads,
175                    comm,
176                ),
177                softcap: None,
178                softmax_scale: 1.0 / (head_dim as f32).sqrt(),
179                sliding_window,
180            },
181        })
182    }
183
184    #[allow(clippy::too_many_arguments)]
185    fn forward(
186        &self,
187        xs: &Tensor,
188        attention_mask: Option<&Tensor>,
189        seqlen_offsets: &[usize],
190        kv_cache: &mut KvCache,
191        metadata: Option<((Tensor, Tensor), &PagedAttentionInputMetadata)>,
192        flash_params: &FlashParams,
193    ) -> Result<Tensor> {
194        let (b_sz, q_len, _) = xs.dims3()?;
195
196        let original_dtype = xs.dtype();
197        let mut xs = xs.clone();
198        if let Some(t) = self.q_proj.quantized_act_type() {
199            xs = xs.to_dtype(t)?;
200        }
201        let mut q = MatMul.qmethod_matmul(&xs, &*self.q_proj)?;
202        let mut k = MatMul.qmethod_matmul(&xs, &*self.k_proj)?;
203        let mut v = MatMul.qmethod_matmul(&xs, &*self.v_proj)?;
204        if self.q_proj.quantized_act_type().is_some() {
205            q = q.to_dtype(original_dtype)?;
206            k = k.to_dtype(original_dtype)?;
207            v = v.to_dtype(original_dtype)?;
208        }
209
210        (q, k, v) = if q_len != 1 {
211            let q = q
212                .reshape((b_sz, q_len, self.num_heads, self.head_dim))?
213                .transpose(1, 2)?;
214            let k = k
215                .reshape((b_sz, q_len, self.num_kv_heads, self.head_dim))?
216                .transpose(1, 2)?;
217            let v = v
218                .reshape((b_sz, q_len, self.num_kv_heads, self.head_dim))?
219                .transpose(1, 2)?;
220            (q, k, v)
221        } else {
222            let q = q.reshape((b_sz, self.num_heads, q_len, self.head_dim))?;
223            let k = k.reshape((b_sz, self.num_kv_heads, q_len, self.head_dim))?;
224            let v = v.reshape((b_sz, self.num_kv_heads, q_len, self.head_dim))?;
225            (q, k, v)
226        };
227
228        q = q.apply(&self.q_norm)?;
229        k = k.apply(&self.k_norm)?;
230
231        (q, k) = self.rotary_emb.forward(&q, &k, seqlen_offsets)?;
232
233        let mut attn_output = match &self.paged_attn {
234            Some(paged_attn) => match metadata {
235                Some(((key_cache, value_cache), input_metadata)) => paged_attn.forward(
236                    &q,
237                    &k,
238                    &v,
239                    attention_mask,
240                    Some(key_cache),
241                    Some(value_cache),
242                    input_metadata,
243                    &self.sdpa_params,
244                    Some(flash_params),
245                )?,
246                None => {
247                    // If we don't have metadata, we are most likely generating an imatrix so we don't want to populate that.
248                    // Generating the dummy metadata with the assumption that we are not generating text (only processing prompts).
249                    let input_metadata = PagedAttentionInputMetadata::dummy(q.device())?;
250                    // Sanity check.
251                    assert!(attention_mask.is_some());
252                    paged_attn.forward(
253                        &q,
254                        &k,
255                        &v,
256                        attention_mask,
257                        None,
258                        None,
259                        &input_metadata,
260                        &self.sdpa_params,
261                        Some(flash_params),
262                    )?
263                }
264            },
265            None => {
266                let (k, v) = kv_cache.append(&k, &v)?;
267
268                Sdpa.run_attention(
269                    &q,
270                    &k,
271                    &v,
272                    attention_mask,
273                    Some(flash_params),
274                    &self.sdpa_params,
275                )?
276            }
277        };
278
279        if let Some(t) = self.q_proj.quantized_act_type() {
280            attn_output = attn_output.to_dtype(t)?;
281        }
282        attn_output = if attention_mask.is_some() {
283            attn_output.transpose(1, 2)?.reshape((b_sz, q_len, ()))?
284        } else {
285            attn_output.reshape((b_sz, q_len, ()))?
286        };
287        let mut res = MatMul.qmethod_matmul(&attn_output, &*self.o_proj)?;
288        if self.q_proj.quantized_act_type().is_some() {
289            res = res.to_dtype(original_dtype)?;
290        }
291        Ok(res)
292    }
293}

295#[derive(Clone)]
296struct Mlp {
297    gate_proj: Arc<dyn QuantMethod>,
298    up_proj: Arc<dyn QuantMethod>,
299    down_proj: Arc<dyn QuantMethod>,
300    act_fn: Activation,
301}
302
303impl Mlp {
304    fn new(
305        cfg: &Config,
306        vb: ShardedVarBuilder,
307        comm: &Arc<mistralrs_quant::Comm>,
308        i_size: usize,
309    ) -> Result<Self> {
310        let hidden_size = cfg.hidden_size;
311
312        let gate_proj = ColumnParallelLayer::new(
313            hidden_size,
314            i_size,
315            &cfg.quantization_config,
316            false,
317            comm,
318            vb.pp("gate_proj"),
319        )?;
320        let up_proj = RowParallelLayer::new(
321            hidden_size,
322            i_size,
323            &cfg.quantization_config,
324            false,
325            comm,
326            vb.pp("up_proj"),
327        )?;
328        let down_proj = ColumnParallelLayer::new(
329            i_size,
330            hidden_size,
331            &cfg.quantization_config,
332            false,
333            comm,
334            vb.pp("down_proj"),
335        )?;
336
337        Ok(Self {
338            gate_proj,
339            up_proj,
340            down_proj,
341            act_fn: cfg.hidden_act,
342        })
343    }
344
345    fn forward(&self, xs: &Tensor) -> Result<Tensor> {
346        let original_dtype = xs.dtype();
347        let mut xs = xs.clone();
348        if let Some(t) = self.gate_proj.quantized_act_type() {
349            xs = xs.to_dtype(t)?;
350        }
351        let mut current_hidden_states = MatMul
352            .qmethod_matmul(&xs, &*self.gate_proj)?
353            .apply(&self.act_fn)?;
354        let rhs = MatMul.qmethod_matmul(&xs, &*self.up_proj)?;
355        current_hidden_states = current_hidden_states.broadcast_mul(&rhs)?;
356        let mut res = MatMul.qmethod_matmul(&current_hidden_states, &*self.down_proj)?;
357        if self.gate_proj.quantized_act_type().is_some() {
358            res = res.to_dtype(original_dtype)?;
359        }
360        Ok(res)
361    }
362}

364struct MoeMlp {
365    gate: candle_nn::Linear,
366    experts: Vec<Mlp>,
367    norm_topk_prob: bool,
368    num_experts_per_tok: usize,
369}
370
371impl MoeMlp {
372    fn new(
373        cfg: &Config,
374        vb: ShardedVarBuilder,
375        layer_device: Device,
376        comm: &Arc<mistralrs_quant::Comm>,
377    ) -> Result<Self> {
378        let num_experts = cfg.num_experts;
379        let gate = layers::linear_no_bias(
380            cfg.hidden_size,
381            num_experts,
382            vb.pp("gate").set_device(layer_device),
383        )?;
384
385        let experts_vb = vb.pp("experts");
386        let mut experts = Vec::with_capacity(num_experts);
387        for i in 0..num_experts {
388            experts.push(Mlp::new(
389                cfg,
390                experts_vb.pp(i),
391                comm,
392                cfg.moe_intermediate_size,
393            )?);
394        }
395
396        Ok(Self {
397            gate,
398            experts,
399            norm_topk_prob: cfg.norm_topk_prob,
400            num_experts_per_tok: cfg.num_experts_per_tok,
401        })
402    }
403
404    fn forward(&self, xs: &Tensor) -> Result<Tensor> {
405        let (b_size, seq_len, hidden_dim) = xs.dims3()?;
406        let xs = xs.reshape(((), hidden_dim))?;
407        let router_logits = xs.apply(&self.gate)?;
408        let routing_weights = candle_nn::ops::softmax_last_dim(&router_logits)?;
409
410        // In order to extract topk, we extract the data from the tensor and manipulate it
411        // directly. Maybe we will want to use some custom ops instead at some point.
412        let experts_per_tok = routing_weights
413            .arg_sort_last_dim(false)?
414            .narrow(D::Minus1, 0, self.num_experts_per_tok)?
415            .contiguous()?;
416        let routing_weights = routing_weights.gather(&experts_per_tok, D::Minus1)?;
417
418        // routing_weights, selected_experts = torch.topk(routing_weights, self.top_k, dim=-1)
419        // top_x contains the row indexes to evaluate for each expert.
420        let routing_weights = routing_weights.to_dtype(DType::F32)?.to_vec2::<f32>()?;
421        let experts_per_tok = experts_per_tok.to_vec2::<u32>()?;
422        let mut top_x = vec![vec![]; self.experts.len()];
423        let mut selected_experts = vec![vec![]; self.experts.len()];
424        for (row_idx, (rw, expert_idxs)) in routing_weights
425            .iter()
426            .zip(experts_per_tok.iter())
427            .enumerate()
428        {
429            let sum_rw = rw.iter().sum::<f32>();
430            for (&rw, &expert_idx) in rw.iter().zip(expert_idxs.iter()) {
431                top_x[expert_idx as usize].push(row_idx as u32);
432                let rw = if self.norm_topk_prob { rw / sum_rw } else { rw };
433                selected_experts[expert_idx as usize].push(rw)
434            }
435        }
436
437        let mut ys = xs.zeros_like()?;
438        for (expert_idx, expert_layer) in self.experts.iter().enumerate() {
439            let top_x = &top_x[expert_idx];
440            if top_x.is_empty() {
441                continue;
442            }
443            let top_x = Tensor::new(top_x.as_slice(), xs.device())?;
444            let selected_experts =
445                Tensor::new(selected_experts[expert_idx].as_slice(), xs.device())?
446                    .reshape(((), 1))?
447                    .to_dtype(xs.dtype())?;
448            // Index the correct hidden states and compute the expert hidden state for
449            // the current expert. We need to make sure to multiply the output hidden
450            // states by `routing_weights` on the corresponding tokens (top-1 and top-2)
451            let current_state = xs.index_select(&top_x, 0)?.reshape(((), hidden_dim))?;
452            // current_hidden_states = expert_layer(current_state, routing_weights[top_x_list, idx_list, None])
453            let current_hidden_states = expert_layer
454                .forward(&current_state.unsqueeze(0)?)?
455                .squeeze(0)?;
456            let current_hidden_states = current_hidden_states.broadcast_mul(&selected_experts)?;
457            ys = ys.index_add(&top_x, &current_hidden_states, 0)?;
458        }
459        let ys = ys.reshape((b_size, seq_len, hidden_dim))?;
460        Ok(ys)
461    }
462}

478struct DecoderLayer {
479    self_attn: Attention,
480    mlp: MoeOrMlp,
481    input_layernorm: RmsNorm,
482    post_attention_layernorm: RmsNorm,
483}
484
485impl DecoderLayer {
486    #[allow(clippy::too_many_arguments)]
487    fn new(
488        rotary_emb: Arc<RotaryEmbedding>,
489        cfg: &Config,
490        vb: ShardedVarBuilder,
491        mapper: &dyn DeviceMapper,
492        layer_idx: usize,
493        loading_isq: bool,
494        paged_attn: Option<PagedAttention>,
495        real_device: Device,
496        comm: &Arc<mistralrs_quant::Comm>,
497    ) -> Result<Self> {
498        let self_attn = Attention::new(
499            rotary_emb,
500            cfg,
501            mapper.set_device(layer_idx, vb.pp("self_attn"), loading_isq),
502            mapper,
503            layer_idx,
504            loading_isq,
505            paged_attn,
506            comm,
507        )?;
508        let mlp = if !cfg.mlp_only_layers.contains(&layer_idx)
509            && (cfg.num_experts > 0 && (layer_idx + 1) % cfg.decoder_sparse_step == 0)
510        {
511            MoeOrMlp::Moe(MoeMlp::new(
512                cfg,
513                mapper.set_device(layer_idx, vb.pp("mlp"), loading_isq),
514                mapper
515                    .device_for(layer_idx, false)
516                    .cloned()
517                    .unwrap_or(real_device),
518                comm,
519            )?)
520        } else {
521            MoeOrMlp::Mlp(Mlp::new(
522                cfg,
523                mapper.set_device(layer_idx, vb.pp("mlp"), loading_isq),
524                comm,
525                cfg.intermediate_size,
526            )?)
527        };
528        let input_layernorm = RmsNorm::new(
529            cfg.hidden_size,
530            cfg.rms_norm_eps,
531            mapper.set_device(layer_idx, vb.pp("input_layernorm"), false),
532        )?;
533        let post_attention_layernorm = RmsNorm::new(
534            cfg.hidden_size,
535            cfg.rms_norm_eps,
536            mapper.set_device(layer_idx, vb.pp("post_attention_layernorm"), false),
537        )?;
538        Ok(Self {
539            self_attn,
540            mlp,
541            input_layernorm,
542            post_attention_layernorm,
543        })
544    }
545
546    #[allow(clippy::too_many_arguments)]
547    fn forward(
548        &self,
549        xs: &Tensor,
550        attention_mask: Option<&Tensor>,
551        seqlen_offsets: &[usize],
552        kv_cache: &mut KvCache,
553        metadata: Option<((Tensor, Tensor), &PagedAttentionInputMetadata)>,
554        flash_params: &FlashParams,
555    ) -> Result<Tensor> {
556        let residual = xs;
557        let xs = self.input_layernorm.forward(xs)?;
558        let xs = self.self_attn.forward(
559            &xs,
560            attention_mask,
561            seqlen_offsets,
562            kv_cache,
563            metadata,
564            flash_params,
565        )?;
566        let xs = (xs + residual)?;
567        let residual = &xs;
568        let xs = self
569            .mlp
570            .forward(&xs.apply(&self.post_attention_layernorm)?)?;
571        residual + xs
572    }
573}

588impl Model {
589    pub fn new(
590        cfg: &Config,
591        vb: ShardedVarBuilder,
592        is_gptx: bool,
593        normal_loading_metadata: NormalLoadingMetadata,
594        attention_mechanism: AttentionImplementation,
595    ) -> Result<Self> {
596        let vb_m = vb.pp("model");
597        let vb_lm_head = vb.pp("lm_head");
598        Self::new_inner(
599            cfg,
600            vb_m,
601            vb_lm_head,
602            is_gptx,
603            normal_loading_metadata,
604            attention_mechanism,
605        )
606    }
607
608    pub fn new_inner(
609        cfg: &Config,
610        vb_m: ShardedVarBuilder,
611        vb_lm_head: ShardedVarBuilder,
612        is_gptx: bool,
613        normal_loading_metadata: NormalLoadingMetadata,
614        attention_mechanism: AttentionImplementation,
615    ) -> Result<Self> {
616        if let Some(ref quant_cfg) = &cfg.quantization_config {
617            tracing::info!(
618                "Using {} quantization: {}.",
619                quant_cfg.name(),
620                quant_cfg.get_bits_name(&vb_m)
621            );
622        }
623        let mapper = normal_loading_metadata.mapper;
624
625        let embed_tokens = embedding(
626            cfg.vocab_size,
627            cfg.hidden_size,
628            mapper.set_nm_device(vb_m.pp("embed_tokens"), false),
629            &cfg.quantization_config,
630        )?;
631
632        let head_dim = cfg.head_dim();
633        let mut ropes = HashMap::new();
634        for layer_idx in 0..cfg.num_hidden_layers {
635            let device = mapper
636                .device_for(layer_idx, false)
637                .unwrap_or(&normal_loading_metadata.real_device);
638            ropes.insert(
639                device.location(),
640                Arc::new(RotaryEmbedding::new(
641                    cfg.rope_theta as f32,
642                    head_dim,
643                    cfg.max_position_embeddings,
644                    device,
645                    is_gptx,
646                    vb_m.dtype(),
647                )?),
648            );
649        }
650        let vb_l = vb_m.pp("layers");
651        let layers: Vec<DecoderLayer> = NiceProgressBar::<_, 'b'>(
652            0..cfg.num_hidden_layers,
653            "Loading repeating layers",
654            &normal_loading_metadata.multi_progress,
655        )
656        .par_iter_if_isq(|layer_idx| {
657            let device = mapper
658                .device_for(layer_idx, false)
659                .unwrap_or(&normal_loading_metadata.real_device);
660            let rotary_emb = ropes
661                .get(&device.location())
662                .expect("No RoPE for device location!")
663                .clone();
664            let paged_attn = match &attention_mechanism {
665                AttentionImplementation::Eager => None,
666                AttentionImplementation::PagedAttention => Some(
667                    PagedAttention::new(head_dim, device, None)
668                        .expect("PagedAttention creation failed"),
669                ),
670            };
671            let comm = mapper
672                .get_comm_for(layer_idx)
673                .expect("Failed to get comm for layer");
674            DecoderLayer::new(
675                rotary_emb.clone(),
676                cfg,
677                vb_l.pp(layer_idx),
678                &*mapper,
679                layer_idx,
680                normal_loading_metadata.loading_isq,
681                paged_attn,
682                normal_loading_metadata.real_device.clone(),
683                &comm,
684            )
685        })?;
686        let norm = RmsNorm::new(
687            cfg.hidden_size,
688            cfg.rms_norm_eps,
689            mapper.set_nm_device(vb_m.pp("norm"), false),
690        )?;
691        let lm_head = if !cfg.tie_word_embeddings {
692            ReplicatedLayer::new(
693                cfg.hidden_size,
694                cfg.vocab_size,
695                &cfg.quantization_config,
696                false,
697                mapper.set_nm_device(vb_lm_head, normal_loading_metadata.loading_isq),
698            )?
699        } else {
700            ReplicatedLayer::from_linear(candle_nn::Linear::new(
701                mapper.cast_nm_device(
702                    embed_tokens.embeddings(),
703                    normal_loading_metadata.loading_isq,
704                )?,
705                None,
706            ))?
707        };
708        let cache_types = (0..cfg.num_hidden_layers)
709            .map(|layer_idx| {
710                sliding_window!(layer_idx, cfg)
711                    .map(|window| NormalCacheType::SlidingWindow { window })
712                    .unwrap_or(NormalCacheType::Normal {
713                        max_seq_len: cfg.max_position_embeddings,
714                    })
715            })
716            .collect::<Vec<_>>();
717        Ok(Self {
718            embed_tokens,
719            layers,
720            norm,
721            lm_head,
722            sliding_window: cfg.sliding_window,
723            device: normal_loading_metadata.real_device,
724            cache: EitherCache::Normal(NormalCache::from_types(cache_types)),
725            max_seq_len: cfg.max_position_embeddings,
726            cfg: ModelConfigMetadata {
727                max_seq_len: cfg.max_position_embeddings,
728                num_layers: cfg.num_hidden_layers,
729                hidden_size: cfg.hidden_size,
730                num_kv_heads: (cfg.num_key_value_heads / mapper.get_comm_for(0)?.world_size())
731                    .max(1),
732                num_attn_heads: cfg.num_attention_heads / mapper.get_comm_for(0)?.world_size(),
733                sliding_window: cfg.sliding_window,
734                k_head_dim: cfg.head_dim(),
735                v_head_dim: cfg.head_dim(),
736            },
737            mapper,
738        })
739    }
740
741    pub fn forward(
742        &self,
743        input_ids: &Tensor,
744        seqlen_offsets: &[usize],
745        context_lens: Vec<(usize, usize)>,
746        metadata: Option<(Vec<(Tensor, Tensor)>, &PagedAttentionInputMetadata)>,
747        flash_params: &FlashParams,
748    ) -> Result<Tensor> {
749        self.forward_embeds(
750            input_ids,
751            self.embed_tokens.forward(input_ids)?,
752            seqlen_offsets,
753            context_lens,
754            metadata,
755            flash_params,
756        )
757    }
758
759    #[allow(clippy::too_many_arguments)]
760    pub fn forward_embeds(
761        &self,
762        input_ids: &Tensor,
763        input_embeds: Tensor,
764        seqlen_offsets: &[usize],
765        context_lens: Vec<(usize, usize)>,
766        metadata: Option<(Vec<(Tensor, Tensor)>, &PagedAttentionInputMetadata)>,
767        flash_params: &FlashParams,
768    ) -> Result<Tensor> {
769        let mut xs = input_embeds;
770        let cache = &mut self.cache.normal().0;
771        let attention_mask = CausalMasker.make_sliding_window_causal_mask_matrix(
772            input_ids,
773            metadata
774                .as_ref()
775                .map(|(_, _)| &seqlen_offsets as &dyn PastKvLenCache)
776                .unwrap_or(cache as &dyn PastKvLenCache),
777            self.sliding_window,
778            xs.dtype(),
779            self.cfg.num_attn_heads,
780        )?;
781        // PagedAttention prompt chunking
782        let attention_mask = attention_mask.filter(|_| {
783            metadata
784                .as_ref()
785                .map(|(_, meta)| meta.is_first_prompt_chunk)
786                .unwrap_or(true)
787        });
788        for (i, layer) in self.layers.iter().enumerate() {
789            xs = self.mapper.map(xs, i)?;
790            xs = layer.forward(
791                &xs,
792                attention_mask
793                    .as_ref()
794                    .map(|m| m.to_device(xs.device()).unwrap())
795                    .as_ref(),
796                seqlen_offsets,
797                &mut cache[i],
798                metadata
799                    .as_ref()
800                    .map(|(kv_cache, metadata)| (kv_cache[i].clone(), *metadata)),
801                flash_params,
802            )?;
803        }
804        let xs = xs.to_device(&self.device)?;
805        let mut xs = xs.apply(&self.norm)?;
806        if let Some(t) = self.lm_head.quantized_act_type() {
807            xs = xs.to_dtype(t)?;
808        }
809        extract_logits(&MatMul.qmethod_matmul(&xs, &*self.lm_head)?, context_lens)
810    }
811}

813impl IsqModel for Model {
814    fn get_layers(
815        &mut self,
816    ) -> (
817        Vec<(&mut Arc<dyn QuantMethod>, Option<usize>)>,
818        &dyn DeviceMapper,
819    ) {
820        let mut tensors = Vec::new();
821        tensors.push((&mut self.lm_head, None));
822        for (i, layer) in self.layers.iter_mut().enumerate() {
823            tensors.push((&mut layer.self_attn.q_proj, Some(i)));
824            tensors.push((&mut layer.self_attn.k_proj, Some(i)));
825            tensors.push((&mut layer.self_attn.v_proj, Some(i)));
826            tensors.push((&mut layer.self_attn.o_proj, Some(i)));
827            match &mut layer.mlp {
828                MoeOrMlp::Mlp(layer) => {
829                    tensors.push((&mut layer.gate_proj, Some(i)));
830                    tensors.push((&mut layer.up_proj, Some(i)));
831                    tensors.push((&mut layer.down_proj, Some(i)));
832                }
833                MoeOrMlp::Moe(layer) => {
834                    for layer in &mut layer.experts {
835                        tensors.push((&mut layer.gate_proj, Some(i)));
836                        tensors.push((&mut layer.up_proj, Some(i)));
837                        tensors.push((&mut layer.down_proj, Some(i)));
838                    }
839                }
840            }
841        }
842        (tensors, &*self.mapper)
843    }
844
845    fn residual_tensors(&self) -> Vec<(String, Tensor)> {
846        let uvb = UnVarBuilder::new();
847
848        let uvb_m = uvb.pp("model");
849        uvb_m.pp("embed_tokens").add(&self.embed_tokens);
850        uvb_m.pp("norm").add(&self.norm);
851
852        for (layer_idx, layer) in self.layers.iter().enumerate() {
853            let uvb_l = uvb_m.pp("layers").pp(layer_idx);
854            uvb_l.pp("input_layernorm").add(&layer.input_layernorm);
855            uvb_l
856                .pp("post_attention_layernorm")
857                .add(&layer.post_attention_layernorm);
858            uvb_l
859                .pp("self_attn")
860                .pp("q_norm")
861                .add(&layer.self_attn.q_norm);
862            uvb_l
863                .pp("self_attn")
864                .pp("k_norm")
865                .add(&layer.self_attn.k_norm);
866            if let MoeOrMlp::Moe(layer) = &layer.mlp {
867                uvb_l.pp("mlp").pp("gate").add(&layer.gate);
868            }
869        }
870
871        uvb.to_safetensors()
872    }
873}
