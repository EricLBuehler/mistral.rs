128impl ConformerEncoderConfig {
129    pub fn finish_nemo_config(&mut self) {
130        // Override any of the defaults with the incoming, user settings
131        if self.nemo_conv_settings.subsampling_factor == usize::MAX {
132            self.nemo_conv_settings.subsampling_factor = self.time_reduction;
133        }
134        if self.nemo_conv_settings.feat_in == usize::MAX {
135            self.nemo_conv_settings.feat_in = self.input_size;
136        }
137        if self.nemo_conv_settings.feat_out == usize::MAX {
138            self.nemo_conv_settings.feat_out = self.attention_dim;
139        }
140    }
