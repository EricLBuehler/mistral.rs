11#[allow(unused)]
12pub struct AbsolutePositionalEncoding {
13    pe: Tensor,
14    xscale: f64,
15}
16
17impl AbsolutePositionalEncoding {
18    pub fn new(cfg: &ConformerEncoderConfig, device: &Device) -> Result<Self> {
19        let max_len = 5000;
20
21        let mut pe = Tensor::zeros((max_len, cfg.attention_dim), DType::F32, device)?;
22        let position = Tensor::arange(0u32, max_len as u32, device)?.unsqueeze(1)?;
23
24        let div_term = (Tensor::arange_step(0u32, cfg.attention_dim as u32, 2, device)?
25            .to_dtype(DType::F32)?
26            * -((10000f64).ln() / cfg.attention_dim as f64))?
27            .exp()?;
28
29        let sin = position
30            .to_dtype(DType::F32)?
31            .broadcast_mul(&div_term)?
32            .sin()?;
33        let cos = position
34            .to_dtype(DType::F32)?
35            .broadcast_mul(&div_term)?
36            .cos()?;
37
38        // Interleave
39        let sin_indices = Tensor::from_vec(
40            (0..cfg.attention_dim)
41                .step_by(2)
42                .map(|x| x as u32)
43                .collect(),
44            cfg.attention_dim / 2,
45            device,
46        )?;
47        let cos_indices = Tensor::from_vec(
48            (1..cfg.attention_dim)
49                .step_by(2)
50                .map(|x| x as u32)
51                .collect(),
52            cfg.attention_dim / 2,
53            device,
54        )?;
55        pe = pe.index_add(&sin_indices, &sin, D::Minus1)?;
56        pe = pe.index_add(&cos_indices, &cos, D::Minus1)?;
57        pe = pe.unsqueeze(0)?;
58
59        Ok(Self {
60            pe,
61            xscale: (cfg.attention_dim as f64).sqrt(),
62        })
63    }
64
65    #[allow(unused)]
66    pub fn forward(&self, xs: &Tensor) -> Result<Tensor> {
67        if xs.dim(1)? >= self.pe.dim(1)? {
68            candle_core::bail!("Need to recompute positional embeds");
69        }
70
71        (xs * self.xscale)?.broadcast_add(&self.pe.i((.., ..xs.dim(1)?))?.to_dtype(xs.dtype())?)
72    }
73}

75pub struct T5RelativeAttentionLogitBias {
76    bias_values: Embedding,
77    skip_bucketing: bool,
78    max_distance: usize,
79    symmetric: bool,
80}
81
82impl T5RelativeAttentionLogitBias {
83    pub fn new(
84        num_heads: usize,
85        num_buckets: Option<usize>,
86        max_distance: usize,
87        symmetric: bool,
88        vb: ShardedVarBuilder,
89    ) -> Result<Self> {
90        let skip_bucketing = num_buckets.is_none();
91        let mut num_buckets = num_buckets.unwrap_or(max_distance);
92        if !symmetric {
93            num_buckets *= 2;
94        }
95
96        Ok(Self {
97            bias_values: layers::embedding(num_buckets, num_heads, vb.pp("bias_values"), &None)?,
98            skip_bucketing,
99            symmetric,
100            max_distance,
101        })
102    }
103
104    pub fn forward(&self, x: &Tensor) -> Result<Tensor> {
105        let maxpos = x.dim(1)?;
106        let device = x.device();
107
108        // Create position matrices
109        let context_position = Tensor::arange(0f32, maxpos as f32, device)?.unsqueeze(1)?;
110        let memory_position = Tensor::arange(0f32, maxpos as f32, device)?.unsqueeze(0)?;
111
112        // Calculate relative positions
113        let relative_position = memory_position.broadcast_sub(&context_position)?;
114
115        // Clip to max distance (equivalent to Python's masked_fill)
116        let max_dist = self.max_distance as i64;
117        let relative_position = relative_position.clamp(-max_dist, max_dist - 1)?;
118
119        // Map to bias indices
120        let bias_idx = if self.skip_bucketing {
121            relative_position
122        } else {
123            unimplemented!("require skip_bucketing");
124        };
125
126        let bias_idx = if self.symmetric {
127            bias_idx.abs()?
128        } else {
129            let offset = (self.bias_values.embeddings().dim(0)? / 2) as i64;
130            (bias_idx + offset as f64)?
131        };
132
133        // Ensure bias_idx is the right type for embedding lookup
134        let bias_idx = bias_idx.to_dtype(DType::U32)?;
135
136        // Get bias values
137        let t5_rel_att_bias = self.bias_values.forward(&bias_idx)?; // [L, L, H]
138        t5_rel_att_bias.permute((2, 0, 1))?.unsqueeze(0) // [1, H, L, L]
139    }
140}
