13pub struct NemoConvSubsampling {
14    conv: Vec<Arc<dyn Module + Send + Sync>>,
15    out: Linear,
16    subsampling_factor: usize,
17}
18
19impl NemoConvSubsampling {
20    pub fn new(cfg: &NemoConvConfig, vb: ShardedVarBuilder) -> Result<Self> {
21        if cfg.subsampling_factor % 2 != 0 {
22            candle_core::bail!("Sampling factor should be a multiple of 2!");
23        }
24
25        let sampling_num = (cfg.subsampling_factor as f32).log2() as usize;
26
27        let mut in_channels = 1;
28        let mut layers: Vec<Arc<dyn Module + Send + Sync>> = Vec::new();
29
30        let stride = 2;
31        let kernel_size = 3;
32        let ceil_mode = false;
33
34        assert!(!cfg.is_causal);
35        assert_eq!(cfg.subsampling, "dw_striding");
36
37        let left_padding = (kernel_size - 1) / 2;
38        let right_padding = (kernel_size - 1) / 2;
39        // let max_cache_len = if cfg.is_causal {
40        //     cfg.subsampling_factor+1
41        // } else {
42        //     0
43        // };
44
45        {
46            let vb_layers = vb.pp("conv");
47
48            let mut idx = 0;
49            layers.push(Arc::new(layers::conv2d(
50                in_channels,
51                cfg.conv_channels,
52                kernel_size,
53                Conv2dConfig {
54                    padding: left_padding,
55                    stride,
56                    dilation: 1,
57                    groups: 1,
58                    cudnn_fwd_algo: None,
59                },
60                vb_layers.pp(idx),
61            )?));
62
63            in_channels = cfg.conv_channels;
64            idx += 1;
65            layers.push(Arc::new(cfg.activation));
66
67            for _ in 0..(sampling_num - 1) {
68                idx += 1;
69                layers.push(Arc::new(layers::conv2d(
70                    in_channels,
71                    in_channels,
72                    kernel_size,
73                    Conv2dConfig {
74                        padding: left_padding,
75                        stride,
76                        dilation: 1,
77                        groups: in_channels,
78                        cudnn_fwd_algo: None,
79                    },
80                    vb_layers.pp(idx),
81                )?));
82
83                idx += 1;
84                layers.push(Arc::new(layers::conv2d(
85                    in_channels,
86                    cfg.conv_channels,
87                    1,
88                    Conv2dConfig {
89                        padding: 0,
90                        stride: 1,
91                        dilation: 1,
92                        groups: 1,
93                        cudnn_fwd_algo: None,
94                    },
95                    vb_layers.pp(idx),
96                )?));
97
98                idx += 1;
99                layers.push(Arc::new(cfg.activation));
100            }
101        }
102
103        let in_length = cfg.feat_in as f32;
104        let out_length = Self::calc_length(
105            in_length,
106            left_padding + right_padding,
107            kernel_size,
108            stride,
109            ceil_mode,
110            sampling_num,
111        );
112        let out = layers::linear_b(
113            cfg.conv_channels * out_length,
114            cfg.feat_out,
115            true,
116            vb.pp("out"),
117        )?;
118
119        Ok(Self {
120            conv: layers,
121            out,
122            subsampling_factor: cfg.subsampling_factor,
123        })
124    }
125
126    fn calc_length(
127        mut length: f32,
128        all_paddings: usize,
129        kernel_size: usize,
130        stride: usize,
131        ceil_mode: bool,
132        repeat_num: usize,
133    ) -> usize {
134        let add_pad = all_paddings as f32 - kernel_size as f32;
135        let one = 1f32;
136        for _ in 0..repeat_num {
137            length = (length + add_pad) / (stride as f32) + one;
138            if ceil_mode {
139                length = length.ceil();
140            } else {
141                length = length.floor();
142            }
143        }
144        length as usize
145    }
146
147    pub fn forward(&self, x: &Tensor, mask: Option<&Tensor>) -> Result<(Tensor, Option<Tensor>)> {
148        // Unsqueeze channel axis for 2D conv
149        let mut x = x.unsqueeze(1)?;
150
151        // Apply conv layers
152        // Dont do self.subsampling_conv_chunking_factor != -1 and self.conv2d_subsampling
153        for layer in &self.conv {
154            x = layer.forward(&x)?;
155        }
156
157        // Flatten and apply output linear layer
158        let (b, c, t, f) = x.dims4()?;
159        x = x.transpose(1, 2)?.reshape((b, t, c * f))?;
160        x = x.apply(&self.out)?;
161
162        // Handle mask
163        let new_mask = if let Some(mask) = mask {
164            let max_audio_length = x.dim(1)?;
165            let feature_lens = mask.sum_keepdim(1)?;
166            let padding_length = feature_lens.apply(&|t: &Tensor| {
167                (t.to_dtype(candle_core::DType::F32)? / self.subsampling_factor as f64)?.ceil()
168            })?;
169
170            let device = x.device();
171            let arange = Tensor::arange(0u32, max_audio_length as u32, device)?
172                .unsqueeze(0)?
173                .broadcast_as((padding_length.dim(0)?, max_audio_length))?;
174            let pad_mask = arange.lt(&padding_length)?;
175            Some(pad_mask.unsqueeze(1)?)
176        } else {
177            None
178        };
179
180        Ok((x, new_mask))
181    }
