66impl LLaVANextInputProcessor {
67    pub fn get_num_image_tokens(cfg: &LLaVANextConfig, image_size: (u32, u32)) -> usize {
68        let patch_size = cfg.vision_config.patch_size;
69        let image_grid_pinpoints = cfg.image_grid_pinpoints.clone().unwrap();
70        let anyres_grid_shape =
71            get_anyres_image_grid_shape(image_size, &image_grid_pinpoints, patch_size as u32);
72        let patch_per_side = cfg.vision_config.image_size / patch_size;
73        let unpad_shape = calculate_unpad(anyres_grid_shape, image_size);
74        patch_per_side * patch_per_side + (unpad_shape.0 as usize + 1) * (unpad_shape.1 as usize)
75    }
76}

78// Copy from phi3_inputs_processor. different is (1) calculate of num_image_token (2) process_anyres_image (3)image_ids_pad
79impl InputsProcessor for LLaVANextInputProcessor {
80    fn get_type(&self) -> InputsProcessorType {
81        InputsProcessorType::Vision
82    }
83    fn process_inputs(
84        &self,
85        tokenizer: Option<Arc<Tokenizer>>,
86        input_seqs: &mut [&mut Sequence],
87        is_prompt: bool,
88        is_xlora: bool,
89        device: &Device,
90        no_kv_cache: bool,
91        last_n_context_len: Option<(usize, usize)>,
92        return_raw_logits: bool,
93        other_config: Option<Arc<dyn Any>>,
94        mut paged_attn_metadata: Option<PagedAttentionMeta>,
95        mapper: Option<&dyn DeviceMapper>,
96    ) -> anyhow::Result<InputProcessorOutput> {
97        if is_xlora {
98            return Err(anyhow::Error::msg(
99                "Cannot make inputs for X-LoRA vision model.",
100            ));
101        }
102        if no_kv_cache {
103            return Err(anyhow::Error::msg("Vision model must have kv cache."));
104        }
105        let Some(tokenizer) = tokenizer else {
106            return Err(anyhow::Error::msg(
107                "LLaVAInputProcessor requires a specified tokenizer.",
108            ));
109        };
110
111        let config = other_config
112            .clone()
113            .expect("Need a PreProcessorConfig config.");
114        let config: &PreProcessorConfig = config.downcast_ref().expect("Downcast failed.");
115        let crop_size = (
116            *config.crop_size.as_ref().unwrap().get("width").unwrap(),
117            *config.crop_size.as_ref().unwrap().get("height").unwrap(),
118        );
119
120        let has_images = input_seqs.iter().all(|seq| seq.has_images());
121
122        let (pixel_values, image_sizes, num_img_tokens, num_image_samples) = if has_images {
123            let mut pixel_values_accum = Vec::new();
124            let mut image_sizes_accum = Vec::new();
125            let mut num_img_tokens_accum = Vec::new();
126            let mut num_image_samples_accum = Vec::new();
127            for seq in input_seqs.iter_mut() {
128                let imgs = seq
129                    .take_images()
130                    .expect("Need to have images by this point.");
131                let PreprocessedImages {
132                    pixel_values,
133                    pixel_attention_mask: _,
134                    image_sizes,
135                    num_img_tokens,
136                    aspect_ratio_ids: _,
137                    aspect_ratio_mask: _,
138                    num_tiles: _,
139                    image_grid_thw: _,
140                    video_grid_thw: _,
141                    rows: _,
142                    cols: _,
143                    pixel_values_list: _,
144                    tgt_sizes: _,
145                    image_sizes_all: _,
146                    num_crops: _,
147                } = self
148                    .preprocess(
149                        imgs.clone(),
150                        vec![],
151                        config,
152                        device,
153                        (usize::MAX, usize::MAX),
154                    )
155                    .expect("Preprocessor failed");
156                let image_sizes = image_sizes.unwrap();
157                pixel_values_accum.push(pixel_values);
158                image_sizes_accum.push(image_sizes);
159                num_img_tokens_accum.push(num_img_tokens.unwrap());
160                let image_grid_pinpoints = self.model_config.image_grid_pinpoints.clone().unwrap();
161                let num_img_samples = imgs
162                    .iter()
163                    .map(|img| {
164                        let original_size = img.dimensions();
165                        get_num_samples(original_size, &image_grid_pinpoints, crop_size) as usize
166                    })
167                    .collect::<Vec<_>>();
168                num_image_samples_accum.push(num_img_samples);
169            }
170            (
171                Some(Tensor::cat(&pixel_values_accum, 0).unwrap()),
172                Some(image_sizes_accum),
173                Some(num_img_tokens_accum),
174                Some(num_image_samples_accum),
175            )
176        } else {
177            return text_models_inputs_processor::TextInputsProcessor
178                .process_inputs(
179                    Some(tokenizer),
180                    input_seqs,
181                    is_prompt,
182                    is_xlora,
183                    device,
184                    no_kv_cache,
185                    last_n_context_len,
186                    return_raw_logits,
187                    other_config,
188                    paged_attn_metadata,
189                    mapper,
190                )
191                .map(|metadata| {
192                    let InputProcessorOutput {
193                        inputs,
194                        seq_indices,
195                    } = metadata;
196
197                    let text_models_inputs_processor::ModelInputs {
198                        input_ids,
199                        input_ids_full: _,
200                        seqlen_offsets,
201                        seqlen_offsets_full: _,
202                        context_lens,
203                        position_ids,
204                        paged_attn_meta,
205                        flash_meta,
206                        flash_meta_full: _,
207                    } = *inputs
208                        .downcast::<text_models_inputs_processor::ModelInputs>()
209                        .expect("Downcast failed.");
210
211                    let inputs: Box<dyn Any> = Box::new(ModelInputs {
212                        input_ids,
213                        seqlen_offsets,
214                        context_lens,
215                        position_ids,
216                        pixel_values: None,
217                        model_specific_args: Box::new(LLaVANextVisionSpecificArgs {
218                            image_sizes: None,
219                            num_image_tokens: None,
220                            num_image_samples: None,
221                        }),
222                        paged_attn_meta,
223                        flash_meta,
224                    });
225                    InputProcessorOutput {
226                        inputs,
227                        seq_indices,
228                    }
229                });
230        };
231
232        let num_image_tokens_flat = num_img_tokens
233            .clone()
234            .unwrap()
235            .iter()
236            .flatten()
237            .copied()
238            .collect::<Vec<_>>();
239        let num_image_samples = num_image_samples
240            .unwrap()
241            .iter()
242            .flatten()
243            .copied()
244            .collect::<Vec<_>>();
245
246        let mut toks = Vec::new();
247        let detokenized = tokenizer
248            .decode_batch(
249                &input_seqs
250                    .iter()
251                    .map(|seq| seq.get_toks())
252                    .collect::<Vec<_>>(),
253                false,
254            )
255            .expect("Decode failed");
256
257        for (detokenized, (seq, num_img_tokens)) in detokenized.into_iter().zip(
258            input_seqs
259                .iter_mut()
260                .zip(num_img_tokens.unwrap().into_iter()),
261        ) {
262            let splits = self
263                .image_tag_splitter
264                .split(&detokenized)
265                .map(|span| &detokenized[span.range()])
266                .collect::<Vec<_>>();
267            let prompt_chunks = splits
268                .iter()
269                .map(|s| {
270                    // we don't use encode_batch here, because encode_batch will pad 0 to the end of the shor sequences, which will cause the image_ids_pad to be wrong.
271                    tokenizer
272                        .encode_fast(*s, false)
273                        .unwrap()
274                        .get_ids()
275                        .to_vec()
276                        .iter()
277                        .map(|x| *x as i64)
278                        .collect()
279                })
280                .collect::<Vec<Vec<_>>>();
281            let mut image_ids_pad = Vec::new();
282            for (i, num_img_token) in num_img_tokens.iter().enumerate() {
283                let mut image_id_pad = vec![0; *num_img_token];
284                image_id_pad[0] = -(i as i64 + 1);
285                image_ids_pad.push(image_id_pad);
286            }
287            let mut input_ids: Vec<i64> = Vec::new();
288            for item in prompt_chunks
289                .iter()
290                .map(|x| x.to_vec())
291                .interleave(image_ids_pad)
292            {
293                input_ids.extend(item);
294            }
295            let new_ids = input_ids
296                .iter()
297                .map(|x| if *x < 0 { 0u32 } else { *x as u32 })
298                .collect::<Vec<_>>();
299            if !seq.multimodal.has_changed_prompt {
300                let new_prompt = tokenizer.decode(&new_ids, false).unwrap();
301                seq.set_initial_prompt(new_prompt);
302                // NOTE(EricLBuehler): Casting to u32 is fine, we don't care about the other toks
303                seq.set_toks_and_reallocate(new_ids, paged_attn_metadata.as_mut());
304                seq.multimodal.has_changed_prompt = true;
305            }
306
307            toks.push(input_ids);
308        }
309
310        let metadata = if is_prompt {
311            get_prompt_input(
312                toks.iter().map(Vec::as_slice).collect(),
313                input_seqs,
314                device,
315                last_n_context_len,
316                return_raw_logits,
317                paged_attn_metadata.as_mut(),
318                mapper,
319            )
320        } else {
321            get_completion_input(
322                toks.iter().map(Vec::as_slice).collect(),
323                input_seqs,
324                device,
325                no_kv_cache,
326                last_n_context_len,
327                return_raw_logits,
328                paged_attn_metadata.as_mut(),
329                mapper,
330            )
331        };
332
333        metadata.map(|metadata| {
334            let text_models_inputs_processor::InnerInputProcessorOutput {
335                inputs:
336                    text_models_inputs_processor::InputMetadata {
337                        input,
338                        positions,
339                        context_lens,
340                        position_ids,
341                        paged_attn_meta,
342                        flash_meta,
343                    },
344                seq_indices,
345            } = metadata;
346            let inputs: Box<dyn Any> = Box::new(ModelInputs {
347                input_ids: input,
348                seqlen_offsets: positions,
349                context_lens,
350                position_ids,
351                pixel_values: pixel_values.clone(),
352                model_specific_args: Box::new(LLaVANextVisionSpecificArgs {
353                    image_sizes: image_sizes.clone(),
354                    num_image_tokens: Some(num_image_tokens_flat.clone()),
355                    num_image_samples: Some(num_image_samples.clone()),
356                }),
357                paged_attn_meta,
358                flash_meta,
359            });
360            InputProcessorOutput {
361                inputs,
362                seq_indices,
363            }
364        })
365    }
366}

368impl ImagePreProcessor for LLaVANextInputProcessor {
369    #[allow(clippy::excessive_precision)]
370    const DEFAULT_MEAN: [f64; 3] = [0.48145466, 0.4578275, 0.40821073];
371    #[allow(clippy::excessive_precision)]
372    const DEFAULT_STD: [f64; 3] = [0.26862954, 0.26130258, 0.27577711];
373    fn preprocess(
374        &self,
375        images: Vec<image::DynamicImage>,
376        videos: Vec<Vec<image::DynamicImage>>,
377        config: &preprocessor_config::PreProcessorConfig,
378        device: &candle_core::Device,
379        (_, _): (usize, usize),
380    ) -> candle_core::Result<image_processor::PreprocessedImages> {
381        if images.len() > 1 {
382            candle_core::bail!("Can only process one image per batch"); // This is no different from phi3_input_processor
383        };
384        assert!(videos.is_empty());
385
386        let resized_size = *config.size.as_ref().unwrap().get("shortest_edge").unwrap() as usize;
387        let image = images[0].clone();
388        let original_size = image.dimensions();
389        let image_grid_pinpoints = self.model_config.image_grid_pinpoints.clone().unwrap();
390        let best_resolution = select_best_resolution(original_size, &image_grid_pinpoints);
391        // Here I didn't use mistral_vision::Transform, because a lot transformations are before turning the image into a tensor
392        let image_padded = resize_and_pad_image(&image, best_resolution);
393        let filter = config.resampling.to_filter()?;
394        let image_original_resize =
395            image.resize_exact(resized_size as u32, resized_size as u32, filter);
396        let mut samples = vec![image_original_resize];
397        for patch in divide_to_samples(
398            &image_padded,
399            (
400                *config.crop_size.as_ref().unwrap().get("width").unwrap(),
401                *config.crop_size.as_ref().unwrap().get("height").unwrap(),
402            ),
403        ) {
404            samples.push(patch);
405        }
406        let image_mean = config
407            .image_mean
408            .unwrap_or(Self::DEFAULT_MEAN)
409            .map(|x| x as f32);
410        let image_std = config
411            .image_std
412            .unwrap_or(Self::DEFAULT_STD)
413            .map(|x| x as f32);
414        let pixel_values = samples
415            .iter()
416            .map(|x| {
417                LLaVAImageProcessor::process_one_image(
418                    x,
419                    config,
420                    resized_size as u32,
421                    filter,
422                    DType::BF16,
423                    device,
424                    &image_mean,
425                    &image_std,
426                )
427            })
428            .collect::<Result<Vec<Tensor>>>()?;
429        let pixel_values = Tensor::stack(&pixel_values, 0)?;
430
431        Ok(image_processor::PreprocessedImages {
432            pixel_values,
433            pixel_attention_mask: None,
434            image_sizes: Some((original_size.0 as usize, original_size.1 as usize)),
435            num_img_tokens: Some(vec![LLaVANextInputProcessor::get_num_image_tokens(
436                &self.model_config,
437                original_size,
438            )]),
439            aspect_ratio_ids: None,
440            aspect_ratio_mask: None,
441            num_tiles: None,
442            image_grid_thw: None,
443            video_grid_thw: None,
444            rows: None,
445            cols: None,
446            pixel_values_list: None,
447            tgt_sizes: None,
448            image_sizes_all: None,
449            num_crops: None,
450        })
451    }
