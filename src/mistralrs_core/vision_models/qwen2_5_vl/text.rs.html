22struct Mlp {
23    gate_proj: Arc<dyn QuantMethod>,
24    up_proj: Arc<dyn QuantMethod>,
25    down_proj: Arc<dyn QuantMethod>,
26    act_fn: Activation,
27}
28
29impl Mlp {
30    fn new(cfg: &Config, vb: ShardedVarBuilder, comm: &Arc<mistralrs_quant::Comm>) -> Result<Self> {
31        let hidden_sz = cfg.hidden_size;
32        let intermediate_sz = cfg.intermediate_size;
33        let gate_proj = ColumnParallelLayer::new(
34            hidden_sz,
35            intermediate_sz,
36            &cfg.quantization_config,
37            false,
38            comm,
39            vb.pp("gate_proj"),
40        )?;
41        let up_proj = ColumnParallelLayer::new(
42            hidden_sz,
43            intermediate_sz,
44            &cfg.quantization_config,
45            false,
46            comm,
47            vb.pp("up_proj"),
48        )?;
49        let down_proj = RowParallelLayer::new(
50            intermediate_sz,
51            hidden_sz,
52            &cfg.quantization_config,
53            false,
54            comm,
55            vb.pp("down_proj"),
56        )?;
57        Ok(Self {
58            gate_proj,
59            up_proj,
60            down_proj,
61            act_fn: cfg.hidden_act,
62        })
63    }
64
65    fn forward(&self, xs: &Tensor) -> Result<Tensor> {
66        let original_dtype = xs.dtype();
67        let mut xs = xs.clone();
68        if let Some(t) = self.gate_proj.quantized_act_type() {
69            xs = xs.to_dtype(t)?;
70        }
71        let lhs = self.gate_proj.forward(&xs)?.apply(&self.act_fn)?;
72        let rhs = self.up_proj.forward(&xs)?;
73        self.down_proj
74            .forward(&(lhs * rhs)?)?
75            .to_dtype(original_dtype)
76    }
77}

79struct Attention {
80    q_proj: Arc<dyn QuantMethod>,
81    k_proj: Arc<dyn QuantMethod>,
82    v_proj: Arc<dyn QuantMethod>,
83    o_proj: Arc<dyn QuantMethod>,
84    num_heads: usize,
85    num_kv_heads: usize,
86    head_dim: usize,
87    rotary_emb: Arc<Qwen2_5VLRotaryEmbedding>,
88    sdpa_params: SdpaParams,
89}
90
91impl Attention {
92    fn new(
93        rotary_emb: Arc<Qwen2_5VLRotaryEmbedding>,
94        cfg: &Config,
95        vb: ShardedVarBuilder,
96        comm: &Arc<mistralrs_quant::Comm>,
97    ) -> Result<Self> {
98        let hidden_sz = cfg.hidden_size;
99        let num_heads = cfg.num_attention_heads;
100        let num_kv_heads = cfg.num_key_value_heads;
101        let head_dim = hidden_sz / num_heads;
102        let q_proj = ColumnParallelLayer::new(
103            hidden_sz,
104            num_heads * head_dim,
105            &cfg.quantization_config,
106            true,
107            comm,
108            vb.pp("q_proj"),
109        )?;
110        let kv_shard = mistralrs_quant::compute_kv_shard(
111            cfg.num_key_value_heads,
112            cfg.hidden_size / cfg.num_attention_heads,
113            comm,
114        );
115        let k_proj = ColumnParallelLayer::new_with_shard(
116            hidden_sz,
117            num_kv_heads * head_dim,
118            &cfg.quantization_config,
119            true,
120            comm,
121            kv_shard,
122            vb.pp("k_proj"),
123        )?;
124        let v_proj = ColumnParallelLayer::new_with_shard(
125            hidden_sz,
126            num_kv_heads * head_dim,
127            &cfg.quantization_config,
128            true,
129            comm,
130            kv_shard,
131            vb.pp("v_proj"),
132        )?;
133        let o_proj = RowParallelLayer::new(
134            num_heads * head_dim,
135            hidden_sz,
136            &cfg.quantization_config,
137            false,
138            comm,
139            vb.pp("o_proj"),
140        )?;
141        Ok(Self {
142            q_proj,
143            k_proj,
144            v_proj,
145            o_proj,
146            num_heads: num_heads / comm.world_size(),
147            num_kv_heads: (num_kv_heads / comm.world_size()).max(1),
148            head_dim,
149            rotary_emb,
150            sdpa_params: SdpaParams {
151                n_kv_groups: mistralrs_quant::compute_n_kv_groups(
152                    cfg.num_key_value_heads,
153                    cfg.num_attention_heads,
154                    comm,
155                ),
156                softcap: None,
157                softmax_scale: 1.0 / (head_dim as f32).sqrt(),
158                sliding_window: None,
159            },
160        })
161    }
162
163    #[allow(clippy::too_many_arguments)]
164    fn forward(
165        &self,
166        xs: &Tensor,
167        attention_mask: Option<&Tensor>,
168        cos_sin: &(Tensor, Tensor),
169        kv_cache: &mut KvCache,
170        flash_params: &FlashParams,
171    ) -> Result<Tensor> {
172        let (b_sz, q_len, _) = xs.dims3()?;
173
174        let original_dtype = xs.dtype();
175        let mut xs = xs.clone();
176        if let Some(t) = self.q_proj.quantized_act_type() {
177            xs = xs.to_dtype(t)?;
178        }
179        let mut q = self.q_proj.forward(&xs)?;
180        let mut k = self.k_proj.forward(&xs)?;
181        let mut v = self.v_proj.forward(&xs)?;
182        if self.q_proj.quantized_act_type().is_some() {
183            q = q.to_dtype(original_dtype)?;
184            k = k.to_dtype(original_dtype)?;
185            v = v.to_dtype(original_dtype)?;
186        }
187
188        let (mut q, mut k, v) = if q_len != 1 {
189            let q = q
190                .reshape((b_sz, q_len, self.num_heads, self.head_dim))?
191                .transpose(1, 2)?;
192            let k = k
193                .reshape((b_sz, q_len, self.num_kv_heads, self.head_dim))?
194                .transpose(1, 2)?;
195            let v = v
196                .reshape((b_sz, q_len, self.num_kv_heads, self.head_dim))?
197                .transpose(1, 2)?;
198            (q, k, v)
199        } else {
200            let q = q.reshape((b_sz, self.num_heads, q_len, self.head_dim))?;
201            let k = k.reshape((b_sz, self.num_kv_heads, q_len, self.head_dim))?;
202            let v = v.reshape((b_sz, self.num_kv_heads, q_len, self.head_dim))?;
203            (q, k, v)
204        };
205
206        // we share cos_sin amount all layers, but when some layers are in different device from layers[0],
207        // need to move cos_sin to corresponding device
208        let cos_sin_relocated = &(
209            cos_sin.0.to_device(q.device())?,
210            cos_sin.1.to_device(q.device())?,
211        );
212        self.rotary_emb.forward(cos_sin_relocated, &mut q, &mut k)?;
213
214        let mut attn_output = {
215            let (k, v) = kv_cache.append(&k, &v)?;
216
217            Sdpa.run_attention(
218                &q.contiguous()?.to_dtype(DType::F32)?,
219                &k.contiguous()?.to_dtype(DType::F32)?,
220                &v.contiguous()?.to_dtype(DType::F32)?,
221                attention_mask
222                    .map(|mask| mask.to_dtype(DType::F32).unwrap())
223                    .as_ref(),
224                Some(flash_params),
225                &self.sdpa_params,
226            )?
227            .to_dtype(q.dtype())?
228        };
229
230        if let Some(t) = self.q_proj.quantized_act_type() {
231            attn_output = attn_output.to_dtype(t)?;
232        }
233        attn_output = if attention_mask.is_some() {
234            attn_output.transpose(1, 2)?.reshape((b_sz, q_len, ()))?
235        } else {
236            attn_output.reshape((b_sz, q_len, ()))?
237        };
238        let mut res = self.o_proj.forward(&attn_output)?;
239        if self.q_proj.quantized_act_type().is_some() {
240            res = res.to_dtype(original_dtype)?;
241        }
242        Ok(res)
243    }
244}

246pub struct DecoderLayer {
247    self_attn: Attention,
248    mlp: Mlp,
249    input_layernorm: F32RmsNorm,
250    post_attention_layernorm: F32RmsNorm,
251}
252
253impl DecoderLayer {
254    fn new(
255        rotary_emb: Arc<Qwen2_5VLRotaryEmbedding>,
256        cfg: &Config,
257        vb: ShardedVarBuilder,
258        mapper: &dyn DeviceMapper,
259        layer_idx: usize,
260        loading_isq: bool,
261        comm: &Arc<mistralrs_quant::Comm>,
262    ) -> Result<Self> {
263        let self_attn = Attention::new(
264            rotary_emb,
265            cfg,
266            mapper.set_device(layer_idx, vb.pp("self_attn"), loading_isq),
267            comm,
268        )?;
269        let mlp = Mlp::new(
270            cfg,
271            mapper.set_device(layer_idx, vb.pp("mlp"), loading_isq),
272            comm,
273        )?;
274        let input_layernorm = F32RmsNorm::new(
275            cfg.hidden_size,
276            cfg.rms_norm_eps,
277            mapper.set_device(layer_idx, vb.pp("input_layernorm"), false),
278        )?;
279        let post_attention_layernorm = F32RmsNorm::new(
280            cfg.hidden_size,
281            cfg.rms_norm_eps,
282            mapper.set_device(layer_idx, vb.pp("post_attention_layernorm"), false),
283        )?;
284        Ok(Self {
285            self_attn,
286            mlp,
287            input_layernorm,
288            post_attention_layernorm,
289        })
290    }
291
292    #[allow(clippy::too_many_arguments)]
293    fn forward(
294        &self,
295        xs: &Tensor,
296        attention_mask: Option<&Tensor>,
297        cos_sin: &(Tensor, Tensor),
298        kv_cache: &mut KvCache,
299        flash_params: &FlashParams,
300    ) -> Result<Tensor> {
301        let residual = xs;
302        let xs = self.input_layernorm.forward(xs)?;
303        let xs = self
304            .self_attn
305            .forward(&xs, attention_mask, cos_sin, kv_cache, flash_params)?;
306        let xs = (xs + residual)?;
307        let residual = &xs;
308        let xs = self
309            .mlp
310            .forward(&xs.apply(&self.post_attention_layernorm)?)?;
311        residual + xs
312    }
313}

315pub struct Qwen2_5VLTextModel {
316    embed_tokens: Embedding,
317    pub(super) norm: F32RmsNorm,
318    layers: Vec<DecoderLayer>,
319    mapper: Box<dyn DeviceMapper + Send + Sync>,
320    lm_head: Arc<dyn QuantMethod>,
321    pub(super) cache: EitherCache,
322    pub(super) cfg: ModelConfigMetadata,
323    pub(super) device: Device,
324    pub(super) dtype: DType,
325    pub(super) max_seq_len: usize,
326}
327
328impl Qwen2_5VLTextModel {
329    pub fn new(
330        cfg: &Config,
331        vb: ShardedVarBuilder,
332        _is_gptx: bool,
333        normal_loading_metadata: NormalLoadingMetadata,
334        attention_mechanism: AttentionImplementation,
335    ) -> Result<Self> {
336        if !matches!(attention_mechanism, AttentionImplementation::Eager) {
337            candle_core::bail!("Expected eager attention implementation");
338        }
339        let mapper = normal_loading_metadata.mapper;
340        // Support both HuggingFace naming (model.*) and MLX naming (language_model.model.*)
341        let vb_m = if vb.contains_tensor("language_model.model.embed_tokens.weight") {
342            vb.pp("language_model").pp("model")
343        } else {
344            vb.pp("model")
345        };
346
347        let embed_tokens = layers::embedding(
348            cfg.vocab_size,
349            cfg.hidden_size,
350            mapper.set_nm_device(vb_m.pp("embed_tokens"), false),
351            &cfg.quantization_config,
352        )?;
353        let head_dim = cfg.hidden_size / cfg.num_attention_heads;
354
355        let mut ropes = HashMap::new();
356        for layer_idx in 0..cfg.num_hidden_layers {
357            let device = mapper
358                .device_for(layer_idx, false)
359                .unwrap_or(&normal_loading_metadata.real_device);
360            ropes.insert(
361                device.location(),
362                Arc::new(Qwen2_5VLRotaryEmbedding::new(
363                    cfg.rope_theta as f32,
364                    head_dim,
365                    device,
366                    cfg.rope_scaling.mrope_section.clone(),
367                )?),
368            );
369        }
370        let vb_l = vb_m.pp("layers");
371        let layers = NiceProgressBar::<_, 'b'>(
372            0..cfg.num_hidden_layers,
373            "Loading repeating layers",
374            &normal_loading_metadata.multi_progress,
375        )
376        .par_iter_if_isq(|layer_idx| {
377            let device = mapper
378                .device_for(layer_idx, false)
379                .unwrap_or(&normal_loading_metadata.real_device);
380            let rotary_emb = ropes
381                .get(&device.location())
382                .expect("No RoPE for device location!")
383                .clone();
384            let comm = mapper.get_comm_for(layer_idx)?;
385            DecoderLayer::new(
386                rotary_emb.clone(),
387                cfg,
388                vb_l.pp(layer_idx),
389                &*mapper,
390                layer_idx,
391                normal_loading_metadata.loading_isq,
392                &comm,
393            )
394        })?;
395        let norm = F32RmsNorm::new(
396            cfg.hidden_size,
397            cfg.rms_norm_eps,
398            mapper.set_nm_device(vb_m.pp("norm"), false),
399        )?;
400        let lm_head = if !cfg.tie_word_embeddings {
401            ReplicatedLayer::new(
402                cfg.hidden_size,
403                cfg.vocab_size,
404                &cfg.quantization_config,
405                false,
406                mapper.set_nm_device(vb.pp("lm_head"), normal_loading_metadata.loading_isq),
407            )?
408        } else {
409            ReplicatedLayer::from_linear(candle_nn::Linear::new(
410                mapper.cast_nm_device(
411                    embed_tokens.embeddings(),
412                    normal_loading_metadata.loading_isq,
413                )?,
414                None,
415            ))?
416        };
417        Ok(Self {
418            embed_tokens,
419            norm,
420            layers,
421            lm_head,
422            cache: EitherCache::Normal(NormalCache::new(
423                cfg.num_hidden_layers,
424                cfg.max_position_embeddings,
425            )),
426            max_seq_len: cfg.max_position_embeddings,
427            cfg: ModelConfigMetadata {
428                max_seq_len: cfg.max_position_embeddings,
429                num_layers: cfg.num_hidden_layers,
430                hidden_size: cfg.hidden_size,
431                num_attn_heads: cfg.num_attention_heads / mapper.get_comm_for(0)?.world_size(),
432                num_kv_heads: (cfg.num_key_value_heads / mapper.get_comm_for(0)?.world_size())
433                    .max(1),
434                sliding_window: cfg.sliding_window,
435                k_head_dim: cfg.hidden_size / cfg.num_attention_heads,
436                v_head_dim: cfg.hidden_size / cfg.num_attention_heads,
437            },
438            device: normal_loading_metadata.real_device.clone(),
439            dtype: vb.dtype(),
440            mapper,
441        })
442    }
443
444    pub fn embed_tokens(&self, input_ids: &Tensor) -> Result<Tensor> {
445        self.embed_tokens.forward(input_ids)
446    }
447
448    pub fn forward_embeds(
449        &self,
450        mut xs: Tensor,
451        attention_mask: Option<&Tensor>,
452        position_ids: &Tensor,
453        context_lens: Vec<(usize, usize)>,
454        flash_params: &FlashParams,
455    ) -> Result<Tensor> {
456        let cache = &mut self.cache.normal().0;
457        let cos_sin = self.layers[0]
458            .self_attn
459            .rotary_emb
460            .compute_cos_sin(position_ids, xs.dtype())?;
461
462        for (i, layer) in self.layers.iter().enumerate() {
463            xs = self.mapper.map(xs, i)?;
464            xs = layer.forward(
465                &xs,
466                attention_mask
467                    .as_ref()
468                    .map(|m| m.to_device(xs.device()).unwrap())
469                    .as_ref(),
470                &cos_sin,
471                &mut cache[i],
472                flash_params,
473            )?
474        }
475        let xs = xs.to_device(&self.device)?;
476        let mut xs = xs.apply(&self.norm)?;
477        if let Some(t) = self.lm_head.quantized_act_type() {
478            xs = xs.to_dtype(t)?;
479        }
480        extract_logits(&self.lm_head.forward(&xs)?, context_lens)
481    }
482}

484impl IsqModel for Qwen2_5VLTextModel {
485    fn get_layers(
486        &mut self,
487    ) -> (
488        Vec<(&mut Arc<dyn QuantMethod>, Option<usize>)>,
489        &dyn DeviceMapper,
490    ) {
491        let mut tensors = Vec::new();
492        tensors.push((&mut self.lm_head, None));
493        for (i, layer) in self.layers.iter_mut().enumerate() {
494            tensors.push((&mut layer.self_attn.q_proj, Some(i)));
495            tensors.push((&mut layer.self_attn.k_proj, Some(i)));
496            tensors.push((&mut layer.self_attn.v_proj, Some(i)));
497            tensors.push((&mut layer.self_attn.o_proj, Some(i)));
498            tensors.push((&mut layer.mlp.gate_proj, Some(i)));
499            tensors.push((&mut layer.mlp.up_proj, Some(i)));
500            tensors.push((&mut layer.mlp.down_proj, Some(i)));
501        }
502        (tensors, &*self.mapper)
503    }
504
505    fn residual_tensors(&self) -> Vec<(String, Tensor)> {
506        let uvb = UnVarBuilder::new();
507
508        let uvb_m = uvb.pp("model");
509        uvb_m.pp("embed_tokens").add(&self.embed_tokens);
510        uvb_m.pp("norm").add(&self.norm);
511
512        for (layer_idx, layer) in self.layers.iter().enumerate() {
513            let uvb_l = uvb_m.pp("layers").pp(layer_idx);
514            uvb_l.pp("input_layernorm").add(&layer.input_layernorm);
515            uvb_l
516                .pp("post_attention_layernorm")
517                .add(&layer.post_attention_layernorm);
518        }
519
520        uvb.to_safetensors()
521    }
