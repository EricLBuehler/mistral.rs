61impl Processor for Idefics2Processor {
62    fn process(
63        &self,
64        pipeline: &dyn Pipeline,
65        messages: Vec<IndexMap<String, MessageContent>>,
66        add_generation_prompt: bool,
67        add_special_tokens: bool,
68        enable_thinking: Option<bool>,
69        reasoning_effort: Option<ReasoningEffort>,
70        tools: Vec<Tool>,
71    ) -> anyhow::Result<(Vec<u32>, String)> {
72        let mut prompt = apply_chat_template(
73            pipeline,
74            messages,
75            add_generation_prompt,
76            enable_thinking,
77            reasoning_effort,
78            self.template_action(),
79            tools,
80        )?;
81
82        let mut image_str = format!(
83            "{}{}{}",
84            self.fake_image_token,
85            self.image_token.repeat(
86                self.config
87                    .image_seq_len
88                    .expect("Idefics 2 model needs `image_seq_len`")
89            ),
90            self.fake_image_token
91        );
92        if self
93            .preprocessor_config
94            .do_image_splitting
95            .is_some_and(|x| x)
96        {
97            // 4 patches + 1 original
98            image_str = image_str.repeat(5);
99        }
100
101        prompt = prompt.replace(self.image_token, &image_str);
102        // Deal with any adjacent images.
103        prompt = prompt.replace(
104            &format!("{}{}", self.fake_image_token, self.fake_image_token),
105            self.fake_image_token,
106        );
107
108        let Some(tokenizer) = &pipeline.tokenizer() else {
109            anyhow::bail!("Idefics2InputProcessor requires a specified tokenizer.",);
110        };
111        let encoding = tokenizer
112            .encode_fast(prompt.clone(), add_special_tokens)
113            .map_err(anyhow::Error::msg)?;
114        Ok((encoding.get_ids().to_vec(), prompt))
115    }
116
117    fn inputs_processor(&self) -> Arc<dyn InputsProcessor> {
118        Arc::new(Idefics2ImageProcessor {
119            max_edge: self.max_edge,
120        })
121    }
122
123    fn get_special_tokens(&self) -> &[&'static str] {
124        &["<fake_token_around_image>", "<image>", "<end_of_utterance>"]
125    }
126
127    fn template_action(&self) -> MessagesAction {
128        MessagesAction::Keep
129    }
130}

132impl InputsProcessor for Idefics2ImageProcessor {
133    fn get_type(&self) -> InputsProcessorType {
134        InputsProcessorType::Vision
135    }
136    fn process_inputs(
137        &self,
138        _: Option<Arc<Tokenizer>>,
139        input_seqs: &mut [&mut Sequence],
140        is_prompt: bool,
141        is_xlora: bool,
142        device: &Device,
143        no_kv_cache: bool,
144        last_n_context_len: Option<(usize, usize)>,
145        return_raw_logits: bool,
146        other_config: Option<Arc<dyn Any>>,
147        mut paged_attn_metadata: Option<PagedAttentionMeta>,
148        mapper: Option<&dyn DeviceMapper>,
149    ) -> anyhow::Result<InputProcessorOutput> {
150        if is_xlora {
151            return Err(anyhow::Error::msg(
152                "Cannot make inputs for X-LoRA vision model.",
153            ));
154        }
155        if no_kv_cache {
156            return Err(anyhow::Error::msg("Vision model must have kv cache."));
157        }
158
159        let text_models_inputs_processor::InnerInputProcessorOutput {
160            inputs:
161                text_models_inputs_processor::InputMetadata {
162                    input,
163                    positions,
164                    context_lens,
165                    position_ids,
166                    paged_attn_meta,
167                    flash_meta,
168                },
169            seq_indices,
170        } = if is_prompt {
171            get_prompt_input(
172                input_seqs
173                    .iter()
174                    .map(|seq| seq.get_toks())
175                    .collect::<Vec<_>>(),
176                input_seqs,
177                device,
178                last_n_context_len,
179                return_raw_logits,
180                paged_attn_metadata.as_mut(),
181                mapper,
182            )
183            .unwrap()
184        } else {
185            get_completion_input(
186                input_seqs
187                    .iter()
188                    .map(|seq| seq.get_toks())
189                    .collect::<Vec<_>>(),
190                input_seqs,
191                device,
192                no_kv_cache,
193                last_n_context_len,
194                return_raw_logits,
195                paged_attn_metadata.as_mut(),
196                mapper,
197            )
198            .unwrap()
199        };
200        let config = other_config.expect("Need a PreProcessorConfig config.");
201        let config: &PreProcessorConfig = config.downcast_ref().expect("Downcast failed.");
202
203        let has_images = input_seqs.iter().all(|seq| seq.has_images());
204
205        let (pixel_values, pixel_attention_mask) = if has_images {
206            let mut pixel_values_accum = Vec::new();
207            let mut pixel_attention_mask_accum = Vec::new();
208            for seq in input_seqs.iter_mut() {
209                let PreprocessedImages {
210                    pixel_values,
211                    pixel_attention_mask,
212                    image_sizes: _,
213                    num_img_tokens: _,
214                    aspect_ratio_ids: _,
215                    aspect_ratio_mask: _,
216                    num_tiles: _,
217                    image_grid_thw: _,
218                    video_grid_thw: _,
219                    rows: _,
220                    cols: _,
221                    pixel_values_list: _,
222                    tgt_sizes: _,
223                    image_sizes_all: _,
224                    num_crops: _,
225                } = self
226                    .preprocess(
227                        seq.take_images()
228                            .expect("Need to have images by this point."),
229                        vec![],
230                        config,
231                        device,
232                        (usize::MAX, usize::MAX), // Don't use it here...
233                    )
234                    .expect("Preprocessing failed");
235                pixel_values_accum.push(pixel_values.unsqueeze(0).unwrap());
236                pixel_attention_mask_accum
237                    .push(pixel_attention_mask.unwrap().unsqueeze(0).unwrap());
238            }
239            (
240                Some(Tensor::cat(&pixel_values_accum, 0).unwrap()),
241                Some(Tensor::cat(&pixel_attention_mask_accum, 0).unwrap()),
242            )
243        } else {
244            (None, None)
245        };
246
247        let inputs: Box<dyn Any> = Box::new(ModelInputs {
248            input_ids: input,
249            seqlen_offsets: positions,
250            context_lens,
251            position_ids,
252            pixel_values,
253            model_specific_args: Box::new(pixel_attention_mask),
254            paged_attn_meta,
255            flash_meta,
256        });
257        Ok(InputProcessorOutput {
258            inputs,
259            seq_indices,
260        })
261    }
262}

264impl ImagePreProcessor for Idefics2ImageProcessor {
265    #[allow(clippy::excessive_precision)]
266    const DEFAULT_MEAN: [f64; 3] = [0.48145466, 0.4578275, 0.40821073];
267    #[allow(clippy::excessive_precision)]
268    const DEFAULT_STD: [f64; 3] = [0.26862954, 0.26130258, 0.27577711];
269
270    fn preprocess(
271        &self,
272        mut images: Vec<DynamicImage>,
273        videos: Vec<Vec<DynamicImage>>,
274        config: &PreProcessorConfig,
275        device: &Device,
276        (_bs, _max_num_images): (usize, usize),
277    ) -> Result<PreprocessedImages> {
278        assert!(videos.is_empty());
279
280        let mut patch_masks = Vec::new();
281        let mut pixel_values = Vec::new();
282
283        // Image splitting
284        if config.do_image_splitting.is_some_and(|x| x) {
285            let mut new_images = Vec::new();
286            for image in images {
287                let (w, h) = image.dimensions();
288                let mid_w = w / 2;
289                let mid_h = h / 2;
290                new_images.push(image.crop_imm(0, 0, mid_w, mid_h));
291                new_images.push(image.crop_imm(mid_w, 0, w, mid_h));
292                new_images.push(image.crop_imm(0, mid_h, mid_w, h));
293                new_images.push(image.crop_imm(mid_w, mid_h, w, h));
294                new_images.push(image);
295            }
296            images = new_images;
297        }
298
299        for image in images.iter_mut() {
300            // Resize
301            if config.do_resize.is_some_and(|x| x) {
302                let size = config.size.as_ref().unwrap();
303                let (h, w) = if size.contains_key("shortest_edge")
304                    && size.contains_key("longest_edge")
305                {
306                    mistralrs_vision::get_resize_image_size(
307                        (image.dimensions().1 as usize, image.dimensions().0 as usize),
308                        (
309                            size["shortest_edge"] as usize,
310                            size["longest_edge"] as usize,
311                        ),
312                    )
313                } else if size.contains_key("height") && size.contains_key("width") {
314                    (size["height"] as usize, size["width"] as usize)
315                } else {
316                    candle_core::bail!("Size must be a map of `shortest_edge` and `longest_edge` or `height` and `width`.");
317                };
318
319                *image = image.resize_exact(w as u32, h as u32, config.resampling.to_filter()?);
320            }
321        }
322
323        if let Some(max_edge) = self.max_edge {
324            images = mistralrs_vision::pad_to_max_edge(&images, max_edge);
325        }
326
327        let mut max_h = 0;
328        let mut max_w = 0;
329        for image in &images {
330            let (w, h) = image.dimensions();
331            if w > max_w {
332                max_w = w;
333            }
334            if h > max_h {
335                max_h = h;
336            }
337        }
338
339        for image in images.iter_mut() {
340            // Convert to rgb
341            if config.do_convert_rgb.is_some_and(|x| x) {
342                *image = DynamicImage::ImageRgb8(image.to_rgb8());
343            }
344
345            let transforms = Transforms {
346                input: &ToTensorNoNorm,
347                inner_transforms: &[
348                    &config
349                        .do_rescale
350                        .is_some_and(|x| x)
351                        .then_some(())
352                        .map(|_| Rescale {
353                            factor: config.rescale_factor,
354                        }),
355                    &config
356                        .do_normalize
357                        .is_some_and(|x| x)
358                        .then_some(())
359                        .map(|_| Normalize {
360                            mean: config.image_mean.unwrap_or(Self::DEFAULT_MEAN).to_vec(),
361                            std: config.image_std.unwrap_or(Self::DEFAULT_STD).to_vec(),
362                        }),
363                ],
364            };
365
366            let mut image = image.apply(transforms, device)?;
367            // Pad images, calculating attention mask.
368            if config.do_pad.is_some_and(|x| x) {
369                let (_c, h, w) = image.dims3()?;
370                let padded = mistralrs_vision::pad(&image, max_h as usize, max_w as usize)?;
371                let mask = mistralrs_vision::make_pixel_mask(&padded, h, w)?;
372                patch_masks.push(mask.unsqueeze(0)?);
373                image = padded;
374            }
375
376            // Get pixel values
377            pixel_values.push(image.unsqueeze(0)?)
378        }
379
380        Ok(PreprocessedImages {
381            pixel_values: Tensor::cat(&pixel_values, 0)?,
382            pixel_attention_mask: Some(Tensor::cat(&patch_masks, 0)?),
383            image_sizes: None,
384            num_img_tokens: None,
385            aspect_ratio_ids: None,
386            aspect_ratio_mask: None,
387            num_tiles: None,
388            image_grid_thw: None,
389            video_grid_thw: None,
390            rows: None,
391            cols: None,
392            pixel_values_list: None,
393            tgt_sizes: None,
394            image_sizes_all: None,
395            num_crops: None,
396        })
397    }
