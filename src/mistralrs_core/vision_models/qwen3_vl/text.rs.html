23struct Mlp {
24    gate_proj: Arc<dyn QuantMethod>,
25    up_proj: Arc<dyn QuantMethod>,
26    down_proj: Arc<dyn QuantMethod>,
27    act_fn: Activation,
28}
29
30impl Mlp {
31    fn new(
32        cfg: &TextConfig,
33        vb: ShardedVarBuilder,
34        comm: &Arc<mistralrs_quant::Comm>,
35    ) -> Result<Self> {
36        let hidden_sz = cfg.hidden_size;
37        let intermediate_sz = cfg.intermediate_size;
38        let gate_proj = ColumnParallelLayer::new(
39            hidden_sz,
40            intermediate_sz,
41            &cfg.quantization_config,
42            false,
43            comm,
44            vb.pp("gate_proj"),
45        )?;
46        let up_proj = ColumnParallelLayer::new(
47            hidden_sz,
48            intermediate_sz,
49            &cfg.quantization_config,
50            false,
51            comm,
52            vb.pp("up_proj"),
53        )?;
54        let down_proj = RowParallelLayer::new(
55            intermediate_sz,
56            hidden_sz,
57            &cfg.quantization_config,
58            false,
59            comm,
60            vb.pp("down_proj"),
61        )?;
62        Ok(Self {
63            gate_proj,
64            up_proj,
65            down_proj,
66            act_fn: cfg.hidden_act,
67        })
68    }
69
70    fn forward(&self, xs: &Tensor) -> Result<Tensor> {
71        let original_dtype = xs.dtype();
72        let mut xs = xs.clone();
73        if let Some(t) = self.gate_proj.quantized_act_type() {
74            xs = xs.to_dtype(t)?;
75        }
76        let lhs = self.gate_proj.forward(&xs)?.apply(&self.act_fn)?;
77        let rhs = self.up_proj.forward(&xs)?;
78        self.down_proj
79            .forward(&(lhs * rhs)?)?
80            .to_dtype(original_dtype)
81    }
82}

84struct Attention {
85    q_proj: Arc<dyn QuantMethod>,
86    k_proj: Arc<dyn QuantMethod>,
87    v_proj: Arc<dyn QuantMethod>,
88    o_proj: Arc<dyn QuantMethod>,
89    q_norm: RmsNorm,
90    k_norm: RmsNorm,
91    num_heads: usize,
92    num_kv_heads: usize,
93    head_dim: usize,
94    rotary_emb: Arc<RotaryEmbedding>,
95    paged_attn: Option<PagedAttention>,
96    sdpa_params: SdpaParams,
97}

99impl Attention {
100    #[allow(clippy::too_many_arguments)]
101    fn new(
102        rotary_emb: Arc<RotaryEmbedding>,
103        cfg: &TextConfig,
104        vb: ShardedVarBuilder,
105        mapper: &dyn DeviceMapper,
106        layer_idx: usize,
107        loading_isq: bool,
108        paged_attn: Option<PagedAttention>,
109        comm: &Arc<mistralrs_quant::Comm>,
110    ) -> Result<Self> {
111        let hidden_sz = cfg.hidden_size;
112        let num_heads = cfg.num_attention_heads;
113        let num_kv_heads = cfg.num_key_value_heads;
114        let q_proj = ColumnParallelLayer::new(
115            hidden_sz,
116            num_heads * cfg.head_dim,
117            &cfg.quantization_config,
118            false,
119            comm,
120            mapper.set_device(layer_idx, vb.pp("q_proj"), loading_isq),
121        )?;
122        let kv_shard = mistralrs_quant::compute_kv_shard(
123            cfg.num_key_value_heads,
124            cfg.hidden_size / cfg.num_attention_heads,
125            comm,
126        );
127        let k_proj = ColumnParallelLayer::new_with_shard(
128            hidden_sz,
129            num_kv_heads * cfg.head_dim,
130            &cfg.quantization_config,
131            false,
132            comm,
133            kv_shard,
134            mapper.set_device(layer_idx, vb.pp("k_proj"), loading_isq),
135        )?;
136        let v_proj = ColumnParallelLayer::new_with_shard(
137            hidden_sz,
138            num_kv_heads * cfg.head_dim,
139            &cfg.quantization_config,
140            false,
141            comm,
142            kv_shard,
143            mapper.set_device(layer_idx, vb.pp("v_proj"), loading_isq),
144        )?;
145        let o_proj = RowParallelLayer::new(
146            num_heads * cfg.head_dim,
147            hidden_sz,
148            &cfg.quantization_config,
149            false,
150            comm,
151            mapper.set_device(layer_idx, vb.pp("o_proj"), loading_isq),
152        )?;
153        let q_norm = RmsNorm::new(
154            cfg.head_dim,
155            cfg.rms_norm_eps,
156            mapper.set_device(layer_idx, vb.pp("q_norm"), false),
157        )?;
158        let k_norm = RmsNorm::new(
159            cfg.head_dim,
160            cfg.rms_norm_eps,
161            mapper.set_device(layer_idx, vb.pp("k_norm"), false),
162        )?;
163        Ok(Self {
164            q_proj,
165            k_proj,
166            v_proj,
167            o_proj,
168            q_norm,
169            k_norm,
170            num_heads: num_heads / comm.world_size(),
171            num_kv_heads: (num_kv_heads / comm.world_size()).max(1),
172            head_dim: cfg.head_dim,
173            rotary_emb,
174            paged_attn,
175            sdpa_params: SdpaParams {
176                n_kv_groups: mistralrs_quant::compute_n_kv_groups(
177                    cfg.num_key_value_heads,
178                    cfg.num_attention_heads,
179                    comm,
180                ),
181                softcap: None,
182                softmax_scale: 1.0 / (cfg.head_dim as f32).sqrt(),
183                sliding_window: None,
184            },
185        })
186    }
187
188    #[allow(clippy::too_many_arguments)]
189    fn forward(
190        &self,
191        xs: &Tensor,
192        attention_mask: Option<&Tensor>,
193        seqlen_offsets: &[usize],
194        kv_cache: &mut KvCache,
195        metadata: Option<((Tensor, Tensor), &PagedAttentionInputMetadata)>,
196        flash_params: &FlashParams,
197    ) -> Result<Tensor> {
198        let (b_sz, q_len, _) = xs.dims3()?;
199
200        let original_dtype = xs.dtype();
201        let mut xs = xs.clone();
202        if let Some(t) = self.q_proj.quantized_act_type() {
203            xs = xs.to_dtype(t)?;
204        }
205        let mut q = MatMul.qmethod_matmul(&xs, &*self.q_proj)?;
206        let mut k = MatMul.qmethod_matmul(&xs, &*self.k_proj)?;
207        let mut v = MatMul.qmethod_matmul(&xs, &*self.v_proj)?;
208        if self.q_proj.quantized_act_type().is_some() {
209            q = q.to_dtype(original_dtype)?;
210            k = k.to_dtype(original_dtype)?;
211            v = v.to_dtype(original_dtype)?;
212        }
213
214        (q, k, v) = if q_len != 1 {
215            let q = q
216                .reshape((b_sz, q_len, self.num_heads, self.head_dim))?
217                .transpose(1, 2)?;
218            let k = k
219                .reshape((b_sz, q_len, self.num_kv_heads, self.head_dim))?
220                .transpose(1, 2)?;
221            let v = v
222                .reshape((b_sz, q_len, self.num_kv_heads, self.head_dim))?
223                .transpose(1, 2)?;
224            (q, k, v)
225        } else {
226            let q = q.reshape((b_sz, self.num_heads, q_len, self.head_dim))?;
227            let k = k.reshape((b_sz, self.num_kv_heads, q_len, self.head_dim))?;
228            let v = v.reshape((b_sz, self.num_kv_heads, q_len, self.head_dim))?;
229            (q, k, v)
230        };
231
232        q = q.apply(&self.q_norm)?;
233        k = k.apply(&self.k_norm)?;
234
235        (q, k) = self.rotary_emb.forward(&q, &k, seqlen_offsets)?;
236
237        let q = q.contiguous()?;
238        let k = k.contiguous()?;
239        let v = v.contiguous()?;
240
241        let mut attn_output = match &self.paged_attn {
242            Some(paged_attn) => match metadata {
243                Some(((key_cache, value_cache), input_metadata)) => paged_attn.forward(
244                    &q,
245                    &k,
246                    &v,
247                    attention_mask,
248                    Some(key_cache),
249                    Some(value_cache),
250                    input_metadata,
251                    &self.sdpa_params,
252                    Some(flash_params),
253                )?,
254                None => {
255                    // If we don't have metadata, we are most likely generating an imatrix so we don't want to populate that.
256                    // Generating the dummy metadata with the assumption that we are not generating text (only processing prompts).
257                    let input_metadata = PagedAttentionInputMetadata::dummy(q.device())?;
258                    // Sanity check.
259                    assert!(attention_mask.is_some());
260                    paged_attn.forward(
261                        &q,
262                        &k,
263                        &v,
264                        attention_mask,
265                        None,
266                        None,
267                        &input_metadata,
268                        &self.sdpa_params,
269                        Some(flash_params),
270                    )?
271                }
272            },
273            None => {
274                let (cache_k, cache_v) = kv_cache.append(&k, &v)?;
275                Sdpa.run_attention(
276                    &q,
277                    &cache_k.contiguous()?,
278                    &cache_v.contiguous()?,
279                    attention_mask,
280                    Some(flash_params),
281                    &self.sdpa_params,
282                )?
283            }
284        };
285
286        if let Some(t) = self.q_proj.quantized_act_type() {
287            attn_output = attn_output.to_dtype(t)?;
288        }
289        attn_output = if attention_mask.is_some() {
290            attn_output.transpose(1, 2)?.reshape((b_sz, q_len, ()))?
291        } else {
292            attn_output.reshape((b_sz, q_len, ()))?
293        };
294        let mut res = self.o_proj.forward(&attn_output)?;
295        if self.q_proj.quantized_act_type().is_some() {
296            res = res.to_dtype(original_dtype)?;
297        }
298        Ok(res)
299    }
300}

302pub struct DecoderLayer {
303    self_attn: Attention,
304    mlp: Mlp,
305    input_layernorm: F32RmsNorm,
306    post_attention_layernorm: F32RmsNorm,
307}
308
309impl DecoderLayer {
310    #[allow(clippy::too_many_arguments)]
311    fn new(
312        rotary_emb: Arc<RotaryEmbedding>,
313        cfg: &TextConfig,
314        vb: ShardedVarBuilder,
315        mapper: &dyn DeviceMapper,
316        layer_idx: usize,
317        loading_isq: bool,
318        paged_attn: Option<PagedAttention>,
319        comm: &Arc<mistralrs_quant::Comm>,
320    ) -> Result<Self> {
321        let self_attn = Attention::new(
322            rotary_emb,
323            cfg,
324            vb.pp("self_attn"),
325            mapper,
326            layer_idx,
327            loading_isq,
328            paged_attn,
329            comm,
330        )?;
331        let mlp = Mlp::new(
332            cfg,
333            mapper.set_device(layer_idx, vb.pp("mlp"), loading_isq),
334            comm,
335        )?;
336        let input_layernorm = F32RmsNorm::new(
337            cfg.hidden_size,
338            cfg.rms_norm_eps,
339            mapper.set_device(layer_idx, vb.pp("input_layernorm"), false),
340        )?;
341        let post_attention_layernorm = F32RmsNorm::new(
342            cfg.hidden_size,
343            cfg.rms_norm_eps,
344            mapper.set_device(layer_idx, vb.pp("post_attention_layernorm"), false),
345        )?;
346        Ok(Self {
347            self_attn,
348            mlp,
349            input_layernorm,
350            post_attention_layernorm,
351        })
352    }
353
354    #[allow(clippy::too_many_arguments)]
355    fn forward(
356        &self,
357        xs: &Tensor,
358        attention_mask: Option<&Tensor>,
359        seqlen_offsets: &[usize],
360        kv_cache: &mut KvCache,
361        metadata: Option<((Tensor, Tensor), &PagedAttentionInputMetadata)>,
362        flash_params: &FlashParams,
363    ) -> Result<Tensor> {
364        let residual = xs;
365        let xs = self.input_layernorm.forward(xs)?;
366        let xs = self.self_attn.forward(
367            &xs,
368            attention_mask,
369            seqlen_offsets,
370            kv_cache,
371            metadata,
372            flash_params,
373        )?;
374        let xs = (xs + residual)?;
375        let residual = &xs;
376        let xs = self
377            .mlp
378            .forward(&xs.apply(&self.post_attention_layernorm)?)?;
379        residual + xs
380    }
381}

386    layers: Vec<DecoderLayer>,

388    lm_head: Arc<dyn QuantMethod>,

396impl Qwen3VLTextModel {
397    pub fn new(
398        cfg: &TextConfig,
399        vb: ShardedVarBuilder,
400        tie: bool,
401        normal_loading_metadata: NormalLoadingMetadata,
402        attention_mechanism: AttentionImplementation,
403    ) -> Result<Self> {
404        let mapper = normal_loading_metadata.mapper;
405        let vb_m = vb.pp("model").pp("language_model");
406
407        let embed_tokens = layers::embedding(
408            cfg.vocab_size,
409            cfg.hidden_size,
410            mapper.set_nm_device(vb_m.pp("embed_tokens"), false),
411            &cfg.quantization_config,
412        )?;
413
414        let mut ropes = HashMap::new();
415        for layer_idx in 0..cfg.num_hidden_layers {
416            let device = mapper
417                .device_for(layer_idx, false)
418                .unwrap_or(&normal_loading_metadata.real_device);
419            ropes.insert(
420                device.location(),
421                Arc::new(RotaryEmbedding::new(
422                    cfg.rope_theta as f32,
423                    cfg.head_dim,
424                    cfg.max_position_embeddings,
425                    device,
426                    true,
427                    vb_m.dtype(),
428                )?),
429            );
430        }
431        let vb_l = vb_m.pp("layers");
432        let layers = NiceProgressBar::<_, 'b'>(
433            0..cfg.num_hidden_layers,
434            "Loading repeating layers",
435            &normal_loading_metadata.multi_progress,
436        )
437        .par_iter_if_isq(|layer_idx| {
438            let device = mapper
439                .device_for(layer_idx, false)
440                .unwrap_or(&normal_loading_metadata.real_device);
441            let rotary_emb = ropes
442                .get(&device.location())
443                .expect("No RoPE for device location!")
444                .clone();
445            let paged_attn = match &attention_mechanism {
446                AttentionImplementation::Eager => None,
447                AttentionImplementation::PagedAttention => {
448                    Some(PagedAttention::new(cfg.head_dim, device, None)?)
449                }
450            };
451            let comm = mapper.get_comm_for(layer_idx)?;
452            DecoderLayer::new(
453                rotary_emb.clone(),
454                cfg,
455                vb_l.pp(layer_idx),
456                &*mapper,
457                layer_idx,
458                normal_loading_metadata.loading_isq,
459                paged_attn,
460                &comm,
461            )
462        })?;
463        let norm = F32RmsNorm::new(
464            cfg.hidden_size,
465            cfg.rms_norm_eps,
466            mapper.set_nm_device(vb_m.pp("norm"), false),
467        )?;
468        let lm_head = if !tie {
469            ReplicatedLayer::new(
470                cfg.hidden_size,
471                cfg.vocab_size,
472                &cfg.quantization_config,
473                false,
474                mapper.set_nm_device(vb.pp("lm_head"), normal_loading_metadata.loading_isq),
475            )?
476        } else {
477            ReplicatedLayer::from_linear(candle_nn::Linear::new(
478                mapper.cast_nm_device(
479                    embed_tokens.embeddings(),
480                    normal_loading_metadata.loading_isq,
481                )?,
482                None,
483            ))?
484        };
485        Ok(Self {
486            embed_tokens,
487            norm,
488            layers,
489            lm_head,
490            cache: EitherCache::Normal(NormalCache::new(
491                cfg.num_hidden_layers,
492                cfg.max_position_embeddings,
493            )),
494            max_seq_len: cfg.max_position_embeddings,
495            cfg: ModelConfigMetadata {
496                max_seq_len: cfg.max_position_embeddings,
497                num_layers: cfg.num_hidden_layers,
498                hidden_size: cfg.hidden_size,
499                num_attn_heads: cfg.num_attention_heads / mapper.get_comm_for(0)?.world_size(),
500                num_kv_heads: (cfg.num_key_value_heads / mapper.get_comm_for(0)?.world_size())
501                    .max(1),
502                sliding_window: cfg.sliding_window,
503                k_head_dim: cfg.head_dim,
504                v_head_dim: cfg.head_dim,
505            },
506            device: normal_loading_metadata.real_device.clone(),
507            dtype: vb.dtype(),
508            mapper,
509        })
510    }

512    pub fn embed_tokens(&self, input_ids: &Tensor) -> Result<Tensor> {
513        self.embed_tokens.forward(input_ids)
514    }

516    #[allow(clippy::too_many_arguments)]
517    pub fn forward_embeds(
518        &self,
519        mut xs: Tensor,
520        attention_mask: Option<&Tensor>,
521        seqlen_offsets: &[usize],
522        _position_ids: &Tensor,
523        context_lens: Vec<(usize, usize)>,
524        metadata: Option<(Vec<(Tensor, Tensor)>, &PagedAttentionInputMetadata)>,
525        flash_params: &FlashParams,
526        visual_pos_masks: Option<&Tensor>,
527        deepstack_visual_embeds: Option<&[Tensor]>,
528    ) -> Result<Tensor> {
529        let cache = &mut self.cache.normal().0;
530
531        for (i, layer) in self.layers.iter().enumerate() {
532            xs = self.mapper.map(xs, i)?;
533            xs = layer.forward(
534                &xs,
535                attention_mask
536                    .as_ref()
537                    .map(|m| m.to_device(xs.device()).unwrap())
538                    .as_ref(),
539                seqlen_offsets,
540                &mut cache[i],
541                metadata
542                    .as_ref()
543                    .map(|(kv_cache, meta)| (kv_cache[i].clone(), *meta)),
544                flash_params,
545            )?;
546
547            // Integrate DeepStack visual features when provided.
548            if let (Some(visual_pos_masks), Some(deepstack)) =
549                (visual_pos_masks, deepstack_visual_embeds)
550            {
551                if i < deepstack.len() {
552                    xs = self.deepstack_process(xs, visual_pos_masks, &deepstack[i])?;
553                }
554            }
555        }
556        let xs = xs.to_device(&self.device)?;
557        let mut xs = xs.apply(&self.norm)?;
558        if let Some(t) = self.lm_head.quantized_act_type() {
559            xs = xs.to_dtype(t)?;
560        }
561        extract_logits(&self.lm_head.forward(&xs)?, context_lens)
562    }
563
564    fn deepstack_process(
565        &self,
566        hidden_states: Tensor,
567        visual_pos_masks: &Tensor,
568        visual_embeds: &Tensor,
569    ) -> Result<Tensor> {
570        let device = hidden_states.device();
571        let dtype = hidden_states.dtype();
572
573        let mask = visual_pos_masks.to_device(device)?.to_dtype(DType::F32)?;
574        let mask_flat = mask.flatten_all()?;
575
576        let masked_count = mask_flat.sum_all()?.to_scalar::<f32>()? as usize;
577        let visual_embeds = visual_embeds.to_device(device)?.to_dtype(dtype)?;
578
579        if masked_count == 0 {
580            if visual_embeds.dim(0)? != 0 {
581                candle_core::bail!(
582                    "DeepStack visual embeds ({}) provided but mask is empty",
583                    visual_embeds.dim(0)?
584                );
585            }
586            return Ok(hidden_states);
587        }
588
589        if visual_embeds.dim(0)? != masked_count {
590            candle_core::bail!(
591                "Mismatch between DeepStack visual embeds ({}) and mask positions ({})",
592                visual_embeds.dim(0)?,
593                masked_count
594            );
595        }
596
597        let (batch, seq, hidden) = hidden_states.dims3()?;
598        let total_positions = batch * seq;
599        let mut hidden_flat = hidden_states.reshape((total_positions, hidden))?;
600
601        let prefix = mask_flat.cumsum(0)?;
602        let rank = (prefix - &mask_flat)?.mul(&mask_flat)?;
603        let rank_u32 = rank.to_dtype(DType::U32)?;
604
605        let positions = Tensor::arange(0u32, total_positions as u32, device)?;
606        let positions_f32 = positions.to_dtype(DType::F32)?;
607        let masked_positions = positions_f32.mul(&mask_flat)?;
608
609        let mut position_per_rank = Tensor::zeros((masked_count,), DType::F32, device)?;
610        position_per_rank = position_per_rank.scatter_add(&rank_u32, &masked_positions, 0)?;
611        let position_per_rank = position_per_rank.to_dtype(DType::U32)?;
612
613        let linear_index = position_per_rank.unsqueeze(1)?.repeat((1, hidden))?;
614
615        hidden_flat = hidden_flat.scatter_add(&linear_index, &visual_embeds, 0)?;
616        hidden_flat.reshape((batch, seq, hidden))
617    }
618}

620impl IsqModel for Qwen3VLTextModel {
621    fn get_layers(
622        &mut self,
623    ) -> (
624        Vec<(&mut Arc<dyn QuantMethod>, Option<usize>)>,
625        &dyn DeviceMapper,
626    ) {
627        let mut tensors = Vec::new();
628        tensors.push((&mut self.lm_head, None));
629        for (i, layer) in self.layers.iter_mut().enumerate() {
630            tensors.push((&mut layer.self_attn.q_proj, Some(i)));
631            tensors.push((&mut layer.self_attn.k_proj, Some(i)));
632            tensors.push((&mut layer.self_attn.v_proj, Some(i)));
633            tensors.push((&mut layer.self_attn.o_proj, Some(i)));
634            tensors.push((&mut layer.mlp.gate_proj, Some(i)));
635            tensors.push((&mut layer.mlp.up_proj, Some(i)));
636            tensors.push((&mut layer.mlp.down_proj, Some(i)));
637        }
638        (tensors, &*self.mapper)
639    }
640
641    fn residual_tensors(&self) -> Vec<(String, Tensor)> {
642        let uvb = UnVarBuilder::new();
643
644        let uvb_lm = uvb.pp("model").pp("language_model");
645        uvb_lm.pp("embed_tokens").add(&self.embed_tokens);
646        uvb_lm.pp("norm").add(&self.norm);
647
648        for (layer_idx, layer) in self.layers.iter().enumerate() {
649            let uvb_l = uvb_lm.pp("layers").pp(layer_idx);
