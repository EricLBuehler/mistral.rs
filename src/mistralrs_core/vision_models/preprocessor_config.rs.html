83#[allow(dead_code)]
84pub(crate) trait ToFilter {
85    fn to_filter(self) -> Result<FilterType>;
86}
87
88impl ToFilter for Option<usize> {
89    // https://github.com/python-pillow/Pillow/blob/4b68563e8a818fb9c528fa159ddf3f4eaefa35e6/src/PIL/Image.py#L164-L170
90    // Default: https://github.com/huggingface/transformers/blob/0df888ffb72ea370555efdef45985378d3cc7b2b/src/transformers/models/idefics2/image_processing_idefics2.py#L226
91    fn to_filter(self) -> Result<FilterType> {
92        match self {
93            Some(0) => Ok(FilterType::Nearest),
94            Some(1) => Ok(FilterType::Lanczos3),
95            Some(2) | None => Ok(FilterType::Triangle), // BiLinear
96            Some(3) => Ok(FilterType::CatmullRom),      // BiCubic
97            Some(4) => Ok(FilterType::Nearest),
98            Some(x) => candle_core::bail!("Filter number {x} not supported"),
99        }
100    }
