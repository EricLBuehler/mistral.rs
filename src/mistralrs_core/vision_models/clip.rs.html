17pub enum Activation {
18    QuickGelu,
19}
20
21impl Module for Activation {
22    fn forward(&self, xs: &Tensor) -> Result<Tensor> {
23        match self {
24            Activation::QuickGelu => xs * candle_nn::ops::sigmoid(&(xs * 1.702f64)?),
25        }
26    }
27}

60struct ClipVisionEmbeddings {
61    patch_embedding: candle_nn::Conv2d,
62    position_ids: Tensor,
63    class_embedding: Tensor,
64    position_embedding: candle_nn::Embedding,
65}
66
67impl ClipVisionEmbeddings {
68    fn new(vs: ShardedVarBuilder, c: &ClipConfig) -> Result<Self> {
69        // originally nn.Parameter
70        let class_embedding = if vs.contains_tensor("class_embedding") {
71            vs.get(c.hidden_size, "class_embedding")?
72        } else {
73            Tensor::randn(0f32, 1f32, c.hidden_size, vs.device())?
74        };
75
76        let num_patches = (c.image_size / c.patch_size).pow(2);
77        let num_positions = num_patches + 1;
78        let position_ids = Tensor::arange(0, num_positions as i64, vs.device())?;
79
80        let conv2dconfig = Conv2dConfig {
81            stride: c.patch_size,
82            ..Default::default()
83        };
84        let position_embedding = layers::embedding(
85            num_positions,
86            c.hidden_size,
87            vs.pp("position_embedding"),
88            &None,
89        )?;
90        let patch_embedding = layers::conv2d_no_bias(
91            c.num_channels,
92            c.hidden_size,
93            c.patch_size,
94            conv2dconfig,
95            vs.pp("patch_embedding"),
96        )?;
97        Ok(Self {
98            patch_embedding,
99            position_ids,
100            class_embedding,
101            position_embedding,
102        })
103    }
104}
105
106impl Module for ClipVisionEmbeddings {
107    fn forward(&self, pixel_values: &Tensor) -> Result<Tensor> {
108        let batch_size = pixel_values.shape().dims();
109        let patch_embeds = Convolution
110            .forward_2d(&self.patch_embedding, &pixel_values)?
111            .flatten_from(2)?
112            .transpose(1, 2)?;
113        let shape = Shape::from((batch_size[0], 1, self.class_embedding.dim(D::Minus1)?));
114        let class_embeds = self.class_embedding.expand(shape)?;
115        let embeddings = Tensor::cat(&[class_embeds, patch_embeds], 1)?;
116        let position_embedding = self.position_embedding.forward(&self.position_ids)?;
117        embeddings.broadcast_add(&position_embedding)
118    }
119}

121#[derive(Clone, Debug)]
122struct ClipAttention {
123    k_proj: Arc<dyn QuantMethod>,
124    v_proj: Arc<dyn QuantMethod>,
125    q_proj: Arc<dyn QuantMethod>,
126    out_proj: Arc<dyn QuantMethod>,
127    head_dim: usize,
128    scale: f64,
129    num_attention_heads: usize,
130}
131
132impl ClipAttention {
133    fn new(vs: ShardedVarBuilder, c: &ClipConfig) -> Result<Self> {
134        let hidden_size = c.hidden_size;
135        let num_attention_heads = c.num_attention_heads;
136        let k_proj = mistralrs_quant::linear(hidden_size, hidden_size, &None, vs.pp("k_proj"))?;
137        let v_proj = mistralrs_quant::linear(hidden_size, hidden_size, &None, vs.pp("v_proj"))?;
138        let q_proj = mistralrs_quant::linear(hidden_size, hidden_size, &None, vs.pp("q_proj"))?;
139        let out_proj = mistralrs_quant::linear(hidden_size, hidden_size, &None, vs.pp("out_proj"))?;
140        let head_dim = hidden_size / num_attention_heads;
141        let scale = (head_dim as f64).powf(-0.5);
142
143        Ok(ClipAttention {
144            k_proj,
145            v_proj,
146            q_proj,
147            out_proj,
148            head_dim,
149            scale,
150            num_attention_heads,
151        })
152    }
153
154    fn shape(&self, xs: &Tensor, seq_len: usize, bsz: usize) -> Result<Tensor> {
155        xs.reshape((bsz, seq_len, self.num_attention_heads, self.head_dim))?
156            .transpose(1, 2)?
157            .contiguous()
158    }
159
160    fn forward(&self, xs: &Tensor, causal_attention_mask: Option<&Tensor>) -> Result<Tensor> {
161        let (bsz, seq_len, hidden_size) = xs.dims3()?;
162
163        let query_states = (self.q_proj.forward(xs)? * self.scale)?;
164        let proj_shape = (bsz * self.num_attention_heads, seq_len, self.head_dim);
165        let query_states = self
166            .shape(&query_states, seq_len, bsz)?
167            .reshape(proj_shape)?;
168        let key_states = self
169            .shape(&self.k_proj.forward(xs)?, seq_len, bsz)?
170            .reshape(proj_shape)?;
171        let value_states = self
172            .shape(&self.v_proj.forward(xs)?, seq_len, bsz)?
173            .reshape(proj_shape)?;
174        let attn_weights = MatMul.matmul(&query_states, &key_states.transpose(1, 2)?)?;
175
176        let src_len = key_states.dim(1)?;
177
178        let attn_weights = if let Some(causal_attention_mask) = causal_attention_mask {
179            attn_weights
180                .reshape((bsz, self.num_attention_heads, seq_len, src_len))?
181                .broadcast_add(causal_attention_mask)?
182                .reshape((bsz * self.num_attention_heads, seq_len, src_len))?
183        } else {
184            attn_weights
185        };
186
187        let attn_weights = candle_nn::ops::softmax_last_dim(&attn_weights)?;
188
189        let attn_output = MatMul.matmul(&attn_weights, &value_states)?;
190        let attn_output = attn_output
191            .reshape((bsz, self.num_attention_heads, seq_len, self.head_dim))?
192            .transpose(1, 2)?
193            .reshape((bsz, seq_len, hidden_size))?;
194        self.out_proj.forward(&attn_output)
195    }
196}

198#[derive(Clone, Debug)]
199struct ClipMlp {
200    fc1: Arc<dyn QuantMethod>,
201    fc2: Arc<dyn QuantMethod>,
202    activation: Activation,
203}
204
205impl ClipMlp {
206    fn new(vs: ShardedVarBuilder, c: &ClipConfig) -> Result<Self> {
207        let fc1 = mistralrs_quant::linear(c.hidden_size, c.intermediate_size, &None, vs.pp("fc1"))?;
208        let fc2 = mistralrs_quant::linear(c.intermediate_size, c.hidden_size, &None, vs.pp("fc2"))?;
209
210        Ok(ClipMlp {
211            fc1,
212            fc2,
213            activation: c.hidden_act,
214        })
215    }
216}
217
218impl ClipMlp {
219    fn forward(&self, xs: &Tensor) -> Result<Tensor> {
220        let xs = self.fc1.forward(xs)?;
221        self.fc2.forward(&self.activation.forward(&xs)?)
222    }
223}

225#[derive(Clone, Debug)]
226struct ClipEncoderLayer {
227    self_attn: ClipAttention,
228    layer_norm1: candle_nn::LayerNorm,
229    mlp: ClipMlp,
230    layer_norm2: candle_nn::LayerNorm,
231}
232
233impl ClipEncoderLayer {
234    fn new(vs: ShardedVarBuilder, c: &ClipConfig) -> Result<Self> {
235        let self_attn = ClipAttention::new(vs.pp("self_attn"), c)?;
236        let layer_norm1 = layers::layer_norm(c.hidden_size, 1e-5, vs.pp("layer_norm1"))?;
237        let mlp = ClipMlp::new(vs.pp("mlp"), c)?;
238        let layer_norm2 = layers::layer_norm(c.hidden_size, 1e-5, vs.pp("layer_norm2"))?;
239
240        Ok(ClipEncoderLayer {
241            self_attn,
242            layer_norm1,
243            mlp,
244            layer_norm2,
245        })
246    }
247
248    fn forward(&self, xs: &Tensor, causal_attention_mask: Option<&Tensor>) -> Result<Tensor> {
249        let residual = xs;
250        let xs = self.layer_norm1.forward(xs)?;
251        let xs = self.self_attn.forward(&xs, causal_attention_mask)?;
252        let xs = (xs + residual)?;
253
254        let residual = &xs;
255        let xs = self.layer_norm2.forward(&xs)?;
256        let xs = self.mlp.forward(&xs)?;
257        xs + residual
258    }
259}

261#[derive(Clone, Debug)]
262pub struct ClipEncoder {
263    layers: Vec<ClipEncoderLayer>,
264}
265
266impl ClipEncoder {
267    pub fn new(vs: ShardedVarBuilder, c: &ClipConfig) -> Result<Self> {
268        let vs = vs.pp("layers");
269        let mut layers: Vec<ClipEncoderLayer> = Vec::new();
270        for index in 0..c.num_hidden_layers {
271            let layer = ClipEncoderLayer::new(vs.pp(index.to_string()), c)?;
272            layers.push(layer)
273        }
274        Ok(ClipEncoder { layers })
275    }
276
277    pub fn forward_get_hidden_states(
278        &self,
279        xs: &Tensor,
280        causal_attention_mask: Option<&Tensor>,
281    ) -> Result<Vec<Tensor>> {
282        let mut xs = xs.clone();
283        let mut hidden_states = Vec::new();
284        for layer in self.layers.iter() {
285            xs = layer.forward(&xs, causal_attention_mask)?;
286            hidden_states.push(xs.clone());
287        }
288        Ok(hidden_states)
289    }
290}

293#[derive(Clone, Debug)]
294pub struct ClipVisionTransformer {
295    embeddings: ClipVisionEmbeddings,
296    encoder: ClipEncoder,
297    pre_layer_norm: candle_nn::LayerNorm,
298    final_layer_norm: candle_nn::LayerNorm,
299}
300
301impl ClipVisionTransformer {
302    /// Create a CLIP vision transformer model. Expects the vb to point to the root (not model)
303    /// where (for example) `.pp("embeddings")` is valid.
304    pub fn new(vb: ShardedVarBuilder, c: &ClipConfig) -> Result<Self> {
305        let embeddings = ClipVisionEmbeddings::new(vb.pp("embeddings"), c)?;
306        let pre_layer_norm = layers::layer_norm(c.hidden_size, 1e-5, vb.pp("pre_layrnorm"))?;
307        let encoder = ClipEncoder::new(vb.pp("encoder"), c)?;
308        let final_layer_norm = layers::layer_norm(c.hidden_size, 1e-5, vb.pp("post_layernorm"))?;
309        Ok(Self {
310            embeddings,
311            encoder,
312            final_layer_norm,
313            pre_layer_norm,
314        })
315    }
316
317    pub fn forward_get_hidden_states(&self, pixel_values: &Tensor) -> Result<Vec<Tensor>> {
318        let hidden_states = pixel_values
319            .apply(&self.embeddings)?
320            .apply(&self.pre_layer_norm)?;
321        let mut result = self
322            .encoder
323            .forward_get_hidden_states(&hidden_states, None)?;
324        let encoder_outputs = result.last().unwrap();
325        let pooled_output = encoder_outputs.i((.., 0, ..))?;
326        result.push(self.final_layer_norm.forward(&pooled_output)?.clone());
327        Ok(result)
328    }
329
330    pub fn residual_tensors(&self) -> Vec<(String, Tensor)> {
331        let uvb = UnVarBuilder::new();
332
333        uvb.pp("pre_layrnorm").add(&self.pre_layer_norm);
334        uvb.pp("post_layernorm").add(&self.final_layer_norm);
335
336        // vision embeddings
337        {
338            let uvb_emb = uvb.pp("embeddings");
339
340            uvb_emb.add_tensor("class_embedding", self.embeddings.class_embedding.clone());
341            uvb_emb
342                .pp("position_embedding")
343                .add(&self.embeddings.position_embedding);
344            uvb_emb
345                .pp("patch_embedding")
346                .add(&self.embeddings.patch_embedding);
347        }
348
349        // encoder
350        {
351            let uvb_enc = uvb.pp("encoder");
352
353            for (i, layer) in self.encoder.layers.iter().enumerate() {
354                let uvb_l = uvb_enc.pp("layers").pp(i);
355
356                uvb_l.pp("layer_norm1").add(&layer.layer_norm1);
357                uvb_l.pp("layer_norm2").add(&layer.layer_norm2);
358
359                let uvb_mlp = uvb_l.pp("mlp");
360                uvb_mlp.pp("fc1").add(&layer.mlp.fc1);
361                uvb_mlp.pp("fc2").add(&layer.mlp.fc2);
362
363                let uvb_attn = uvb_l.pp("self_attn");
364                uvb_attn.pp("q_proj").add(&layer.self_attn.q_proj);
365                uvb_attn.pp("k_proj").add(&layer.self_attn.k_proj);
366                uvb_attn.pp("v_proj").add(&layer.self_attn.v_proj);
367                uvb_attn.pp("out_proj").add(&layer.self_attn.out_proj);
368            }
369        }
370
371        uvb.to_safetensors()
372    }
