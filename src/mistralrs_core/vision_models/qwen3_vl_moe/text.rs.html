26// Dense MLP for non-MoE layers
27#[derive(Clone)]
28struct Mlp {
29    gate_proj: Arc<dyn QuantMethod>,
30    up_proj: Arc<dyn QuantMethod>,
31    down_proj: Arc<dyn QuantMethod>,
32    act_fn: Activation,
33}
34
35impl Mlp {
36    fn new(
37        cfg: &TextConfig,
38        vb: ShardedVarBuilder,
39        comm: &Arc<mistralrs_quant::Comm>,
40        i_size: usize,
41    ) -> Result<Self> {
42        let hidden_size = cfg.hidden_size;
43
44        let gate_proj = ColumnParallelLayer::new(
45            hidden_size,
46            i_size,
47            &cfg.quantization_config,
48            false,
49            comm,
50            vb.pp("gate_proj"),
51        )?;
52        let up_proj = ColumnParallelLayer::new(
53            hidden_size,
54            i_size,
55            &cfg.quantization_config,
56            false,
57            comm,
58            vb.pp("up_proj"),
59        )?;
60        let down_proj = RowParallelLayer::new(
61            i_size,
62            hidden_size,
63            &cfg.quantization_config,
64            false,
65            comm,
66            vb.pp("down_proj"),
67        )?;
68
69        Ok(Self {
70            gate_proj,
71            up_proj,
72            down_proj,
73            act_fn: cfg.hidden_act,
74        })
75    }
76
77    fn forward(&self, xs: &Tensor) -> Result<Tensor> {
78        let original_dtype = xs.dtype();
79        let mut xs = xs.clone();
80        if let Some(t) = self.gate_proj.quantized_act_type() {
81            xs = xs.to_dtype(t)?;
82        }
83        let lhs = MatMul
84            .qmethod_matmul(&xs, &*self.gate_proj)?
85            .apply(&self.act_fn)?;
86        let rhs = MatMul.qmethod_matmul(&xs, &*self.up_proj)?;
87        let mut res = MatMul.qmethod_matmul(&(lhs * rhs)?, &*self.down_proj)?;
88        if self.gate_proj.quantized_act_type().is_some() {
89            res = res.to_dtype(original_dtype)?;
90        }
91        Ok(res)
92    }
93}

95/// MoE MLP layer for Qwen3 VL MoE
96struct MoeMlp {
97    gate: Linear,
98    experts: MoEExperts,
99    num_experts_per_tok: usize,
100    norm_topk_prob: bool,
101}
102
103impl MoeMlp {
104    fn new(
105        cfg: &TextConfig,
106        vb: ShardedVarBuilder,
107        layer_device: Device,
108        comm: &Arc<mistralrs_quant::Comm>,
109        loading_isq: bool,
110    ) -> Result<Self> {
111        // Load gate
112        let gate = layers::linear_no_bias(
113            cfg.hidden_size,
114            cfg.num_experts,
115            vb.pp("gate").set_device(layer_device.clone()),
116        )?;
117
118        let moe_cfg = MoEExpertsConfig {
119            num_experts: cfg.num_experts,
120            num_experts_per_tok: cfg.num_experts_per_tok,
121            hidden_size: cfg.hidden_size,
122            moe_intermediate_size: cfg.moe_intermediate_size,
123        };
124
125        // Load experts with automatic backend selection
126        let experts = MoEExperts::new(
127            &moe_cfg,
128            vb,
129            layer_device,
130            comm,
131            loading_isq,
132            &cfg.quantization_config,
133            cfg.hidden_act,
134        )?;
135
136        Ok(Self {
137            gate,
138            experts,
139            num_experts_per_tok: cfg.num_experts_per_tok,
140            norm_topk_prob: cfg.norm_topk_prob,
141        })
142    }
143
144    fn forward(&self, xs: &Tensor) -> Result<Tensor> {
145        let (b_size, seq_len, hidden_dim) = xs.dims3()?;
146        let xs_flat = xs.reshape(((), hidden_dim))?;
147
148        // Compute routing weights
149        let router_logits = self.gate.forward(&xs_flat)?;
150        let routing_weights =
151            candle_nn::ops::softmax_last_dim(&router_logits.to_dtype(DType::F32)?)?;
152
153        // Get top-k experts
154        let topk_ids = routing_weights
155            .arg_sort_last_dim(false)?
156            .narrow(D::Minus1, 0, self.num_experts_per_tok)?
157            .contiguous()?;
158
159        let mut topk_weights = routing_weights.gather(&topk_ids, D::Minus1)?;
160
161        if self.norm_topk_prob {
162            topk_weights = topk_weights.broadcast_div(&topk_weights.sum_keepdim(D::Minus1)?)?;
163        }
164
165        // Forward through experts (is_prefill determined internally based on seq_len)
166        let ys = self.experts.forward(xs, topk_weights, &topk_ids)?;
167
168        ys.reshape((b_size, seq_len, hidden_dim))
169    }
170
171    fn gate(&self) -> &Linear {
172        &self.gate
173    }
174
175    fn get_isq_layers(&mut self) -> Vec<&mut Arc<dyn QuantMethod>> {
176        self.experts.get_isq_layers()
177    }
178}

194struct Attention {
195    q_proj: Arc<dyn QuantMethod>,
196    k_proj: Arc<dyn QuantMethod>,
197    v_proj: Arc<dyn QuantMethod>,
198    o_proj: Arc<dyn QuantMethod>,
199    q_norm: RmsNorm,
200    k_norm: RmsNorm,
201    num_heads: usize,
202    num_kv_heads: usize,
203    head_dim: usize,
204    rotary_emb: Arc<RotaryEmbedding>,
205    paged_attn: Option<PagedAttention>,
206    sdpa_params: SdpaParams,
207}
208
209impl Attention {
210    #[allow(clippy::too_many_arguments)]
211    fn new(
212        rotary_emb: Arc<RotaryEmbedding>,
213        cfg: &TextConfig,
214        vb: ShardedVarBuilder,
215        mapper: &dyn DeviceMapper,
216        layer_idx: usize,
217        loading_isq: bool,
218        paged_attn: Option<PagedAttention>,
219        comm: &Arc<mistralrs_quant::Comm>,
220    ) -> Result<Self> {
221        let hidden_sz = cfg.hidden_size;
222        let num_heads = cfg.num_attention_heads;
223        let num_kv_heads = cfg.num_key_value_heads;
224        let q_proj = ColumnParallelLayer::new(
225            hidden_sz,
226            num_heads * cfg.head_dim,
227            &cfg.quantization_config,
228            false,
229            comm,
230            mapper.set_device(layer_idx, vb.pp("q_proj"), loading_isq),
231        )?;
232        let kv_shard = mistralrs_quant::compute_kv_shard(
233            cfg.num_key_value_heads,
234            cfg.hidden_size / cfg.num_attention_heads,
235            comm,
236        );
237        let k_proj = ColumnParallelLayer::new_with_shard(
238            hidden_sz,
239            num_kv_heads * cfg.head_dim,
240            &cfg.quantization_config,
241            false,
242            comm,
243            kv_shard,
244            mapper.set_device(layer_idx, vb.pp("k_proj"), loading_isq),
245        )?;
246        let v_proj = ColumnParallelLayer::new_with_shard(
247            hidden_sz,
248            num_kv_heads * cfg.head_dim,
249            &cfg.quantization_config,
250            false,
251            comm,
252            kv_shard,
253            mapper.set_device(layer_idx, vb.pp("v_proj"), loading_isq),
254        )?;
255        let o_proj = RowParallelLayer::new(
256            num_heads * cfg.head_dim,
257            hidden_sz,
258            &cfg.quantization_config,
259            false,
260            comm,
261            mapper.set_device(layer_idx, vb.pp("o_proj"), loading_isq),
262        )?;
263        let q_norm = RmsNorm::new(
264            cfg.head_dim,
265            cfg.rms_norm_eps,
266            mapper.set_device(layer_idx, vb.pp("q_norm"), false),
267        )?;
268        let k_norm = RmsNorm::new(
269            cfg.head_dim,
270            cfg.rms_norm_eps,
271            mapper.set_device(layer_idx, vb.pp("k_norm"), false),
272        )?;
273        Ok(Self {
274            q_proj,
275            k_proj,
276            v_proj,
277            o_proj,
278            q_norm,
279            k_norm,
280            num_heads: num_heads / comm.world_size(),
281            num_kv_heads: (num_kv_heads / comm.world_size()).max(1),
282            head_dim: cfg.head_dim,
283            rotary_emb,
284            paged_attn,
285            sdpa_params: SdpaParams {
286                n_kv_groups: mistralrs_quant::compute_n_kv_groups(
287                    cfg.num_key_value_heads,
288                    cfg.num_attention_heads,
289                    comm,
290                ),
291                softcap: None,
292                softmax_scale: 1.0 / (cfg.head_dim as f32).sqrt(),
293                sliding_window: None,
294            },
295        })
296    }

298    #[allow(clippy::too_many_arguments)]
299    fn forward(
300        &self,
301        xs: &Tensor,
302        attention_mask: Option<&Tensor>,
303        seqlen_offsets: &[usize],
304        kv_cache: &mut KvCache,
305        metadata: Option<((Tensor, Tensor), &PagedAttentionInputMetadata)>,
306        flash_params: &FlashParams,
307    ) -> Result<Tensor> {
308        let (b_sz, q_len, _) = xs.dims3()?;
309
310        let original_dtype = xs.dtype();
311        let mut xs = xs.clone();
312        if let Some(t) = self.q_proj.quantized_act_type() {
313            xs = xs.to_dtype(t)?;
314        }
315        let mut q = MatMul.qmethod_matmul(&xs, &*self.q_proj)?;
316        let mut k = MatMul.qmethod_matmul(&xs, &*self.k_proj)?;
317        let mut v = MatMul.qmethod_matmul(&xs, &*self.v_proj)?;
318        if self.q_proj.quantized_act_type().is_some() {
319            q = q.to_dtype(original_dtype)?;
320            k = k.to_dtype(original_dtype)?;
321            v = v.to_dtype(original_dtype)?;
322        }
323
324        (q, k, v) = if q_len != 1 {
325            let q = q
326                .reshape((b_sz, q_len, self.num_heads, self.head_dim))?
327                .transpose(1, 2)?;
328            let k = k
329                .reshape((b_sz, q_len, self.num_kv_heads, self.head_dim))?
330                .transpose(1, 2)?;
331            let v = v
332                .reshape((b_sz, q_len, self.num_kv_heads, self.head_dim))?
333                .transpose(1, 2)?;
334            (q, k, v)
335        } else {
336            let q = q.reshape((b_sz, self.num_heads, q_len, self.head_dim))?;
337            let k = k.reshape((b_sz, self.num_kv_heads, q_len, self.head_dim))?;
338            let v = v.reshape((b_sz, self.num_kv_heads, q_len, self.head_dim))?;
339            (q, k, v)
340        };
341
342        q = q.apply(&self.q_norm)?;
343        k = k.apply(&self.k_norm)?;
344
345        (q, k) = self.rotary_emb.forward(&q, &k, seqlen_offsets)?;
346
347        let q = q.contiguous()?;
348        let k = k.contiguous()?;
349        let v = v.contiguous()?;
350
351        let mut attn_output = match &self.paged_attn {
352            Some(paged_attn) => match metadata {
353                Some(((key_cache, value_cache), input_metadata)) => paged_attn.forward(
354                    &q,
355                    &k,
356                    &v,
357                    attention_mask,
358                    Some(key_cache),
359                    Some(value_cache),
360                    input_metadata,
361                    &self.sdpa_params,
362                    Some(flash_params),
363                )?,
364                None => {
365                    let input_metadata = PagedAttentionInputMetadata::dummy(q.device())?;
366                    assert!(attention_mask.is_some());
367                    paged_attn.forward(
368                        &q,
369                        &k,
370                        &v,
371                        attention_mask,
372                        None,
373                        None,
374                        &input_metadata,
375                        &self.sdpa_params,
376                        Some(flash_params),
377                    )?
378                }
379            },
380            None => {
381                let (cache_k, cache_v) = kv_cache.append(&k, &v)?;
382                Sdpa.run_attention(
383                    &q,
384                    &cache_k.contiguous()?,
385                    &cache_v.contiguous()?,
386                    attention_mask,
387                    Some(flash_params),
388                    &self.sdpa_params,
389                )?
390            }
391        };
392
393        if let Some(t) = self.q_proj.quantized_act_type() {
394            attn_output = attn_output.to_dtype(t)?;
395        }
396        attn_output = if attention_mask.is_some() {
397            attn_output.transpose(1, 2)?.reshape((b_sz, q_len, ()))?
398        } else {
399            attn_output.reshape((b_sz, q_len, ()))?
400        };
401        let mut res = self.o_proj.forward(&attn_output)?;
402        if self.q_proj.quantized_act_type().is_some() {
403            res = res.to_dtype(original_dtype)?;
404        }
405        Ok(res)
406    }
407}

409pub struct DecoderLayer {
410    self_attn: Attention,
411    mlp: MoeOrMlp,
412    input_layernorm: RmsNorm,
413    post_attention_layernorm: RmsNorm,
414}
415
416impl DecoderLayer {
417    #[allow(clippy::too_many_arguments)]
418    fn new(
419        rotary_emb: Arc<RotaryEmbedding>,
420        cfg: &TextConfig,
421        vb: ShardedVarBuilder,
422        mapper: &dyn DeviceMapper,
423        layer_idx: usize,
424        loading_isq: bool,
425        paged_attn: Option<PagedAttention>,
426        comm: &Arc<mistralrs_quant::Comm>,
427        real_device: Device,
428    ) -> Result<Self> {
429        let self_attn = Attention::new(
430            rotary_emb,
431            cfg,
432            vb.pp("self_attn"),
433            mapper,
434            layer_idx,
435            loading_isq,
436            paged_attn,
437            comm,
438        )?;
439
440        // Check if this layer should be MoE or dense MLP
441        let is_moe = !cfg.mlp_only_layers.contains(&layer_idx)
442            && (cfg.num_experts > 0 && (layer_idx + 1) % cfg.decoder_sparse_step == 0);
443
444        let mlp = if is_moe {
445            let vb = mapper.set_device(layer_idx, vb.pp("mlp"), loading_isq);
446            let layer_device = mapper
447                .device_for(layer_idx, false)
448                .cloned()
449                .unwrap_or(real_device.clone());
450
451            MoeOrMlp::Moe(MoeMlp::new(cfg, vb, layer_device, comm, loading_isq)?)
452        } else {
453            MoeOrMlp::Mlp(Mlp::new(
454                cfg,
455                mapper.set_device(layer_idx, vb.pp("mlp"), loading_isq),
456                comm,
457                cfg.intermediate_size,
458            )?)
459        };
460
461        let input_layernorm = RmsNorm::new(
462            cfg.hidden_size,
463            cfg.rms_norm_eps,
464            mapper.set_device(layer_idx, vb.pp("input_layernorm"), false),
465        )?;
466        let post_attention_layernorm = RmsNorm::new(
467            cfg.hidden_size,
468            cfg.rms_norm_eps,
469            mapper.set_device(layer_idx, vb.pp("post_attention_layernorm"), false),
470        )?;
471        Ok(Self {
472            self_attn,
473            mlp,
474            input_layernorm,
475            post_attention_layernorm,
476        })
477    }
478
479    #[allow(clippy::too_many_arguments)]
480    fn forward(
481        &self,
482        xs: &Tensor,
483        attention_mask: Option<&Tensor>,
484        seqlen_offsets: &[usize],
485        kv_cache: &mut KvCache,
486        metadata: Option<((Tensor, Tensor), &PagedAttentionInputMetadata)>,
487        flash_params: &FlashParams,
488    ) -> Result<Tensor> {
489        let residual = xs;
490        let xs = self.input_layernorm.forward(xs)?;
491        let xs = self.self_attn.forward(
492            &xs,
493            attention_mask,
494            seqlen_offsets,
495            kv_cache,
496            metadata,
497            flash_params,
498        )?;
499        let xs = (xs + residual)?;
500        let residual = &xs;
501        let xs = self
502            .mlp
503            .forward(&xs.apply(&self.post_attention_layernorm)?)?;
504        residual + xs
505    }
506}

521impl Qwen3VLMoETextModel {
522    pub fn new(
523        cfg: &TextConfig,
524        vb: ShardedVarBuilder,
525        tie: bool,
526        normal_loading_metadata: NormalLoadingMetadata,
527        attention_mechanism: AttentionImplementation,
528    ) -> Result<Self> {
529        let mapper = normal_loading_metadata.mapper;
530        // Support both HuggingFace naming (model.language_model.*) and MLX naming (language_model.model.*)
531        let vb_m = if vb.contains_tensor("language_model.model.embed_tokens.weight") {
532            vb.pp("language_model").pp("model")
533        } else {
534            vb.pp("model").pp("language_model")
535        };
536
537        let embed_tokens = layers::embedding(
538            cfg.vocab_size,
539            cfg.hidden_size,
540            mapper.set_nm_device(vb_m.pp("embed_tokens"), false),
541            &cfg.quantization_config,
542        )?;
543
544        let mut ropes = HashMap::new();
545        for layer_idx in 0..cfg.num_hidden_layers {
546            let device = mapper
547                .device_for(layer_idx, false)
548                .unwrap_or(&normal_loading_metadata.real_device);
549            ropes.insert(
550                device.location(),
551                Arc::new(RotaryEmbedding::new(
552                    cfg.rope_theta as f32,
553                    cfg.head_dim,
554                    cfg.max_position_embeddings,
555                    device,
556                    true,
557                    vb_m.dtype(),
558                )?),
559            );
560        }
561        let vb_l = vb_m.pp("layers");
562        let layers = NiceProgressBar::<_, 'b'>(
563            0..cfg.num_hidden_layers,
564            "Loading repeating layers",
565            &normal_loading_metadata.multi_progress,
566        )
567        .par_iter_if_isq(|layer_idx| {
568            let device = mapper
569                .device_for(layer_idx, false)
570                .unwrap_or(&normal_loading_metadata.real_device);
571            let rotary_emb = ropes
572                .get(&device.location())
573                .expect("No RoPE for device location!")
574                .clone();
575            let paged_attn = match &attention_mechanism {
576                AttentionImplementation::Eager => None,
577                AttentionImplementation::PagedAttention => {
578                    Some(PagedAttention::new(cfg.head_dim, device, None)?)
579                }
580            };
581            let comm = mapper.get_comm_for(layer_idx)?;
582            DecoderLayer::new(
583                rotary_emb.clone(),
584                cfg,
585                vb_l.pp(layer_idx),
586                &*mapper,
587                layer_idx,
588                normal_loading_metadata.loading_isq,
589                paged_attn,
590                &comm,
591                normal_loading_metadata.real_device.clone(),
592            )
593        })?;
594        let norm = RmsNorm::new(
595            cfg.hidden_size,
596            cfg.rms_norm_eps,
597            mapper.set_nm_device(vb_m.pp("norm"), false),
598        )?;
599        let lm_head = if !tie {
600            ReplicatedLayer::new(
601                cfg.hidden_size,
602                cfg.vocab_size,
603                &cfg.quantization_config,
604                false,
605                mapper.set_nm_device(vb.pp("lm_head"), normal_loading_metadata.loading_isq),
606            )?
607        } else {
608            ReplicatedLayer::from_linear(candle_nn::Linear::new(
609                mapper.cast_nm_device(
610                    embed_tokens.embeddings(),
611                    normal_loading_metadata.loading_isq,
612                )?,
613                None,
614            ))?
615        };
616        Ok(Self {
617            embed_tokens,
618            norm,
619            layers,
620            lm_head,
621            cache: EitherCache::Normal(NormalCache::new(
622                cfg.num_hidden_layers,
623                cfg.max_position_embeddings,
624            )),
625            max_seq_len: cfg.max_position_embeddings,
626            cfg: ModelConfigMetadata {
627                max_seq_len: cfg.max_position_embeddings,
628                num_layers: cfg.num_hidden_layers,
629                hidden_size: cfg.hidden_size,
630                num_attn_heads: cfg.num_attention_heads / mapper.get_comm_for(0)?.world_size(),
631                num_kv_heads: (cfg.num_key_value_heads / mapper.get_comm_for(0)?.world_size())
632                    .max(1),
633                sliding_window: cfg.sliding_window,
634                k_head_dim: cfg.head_dim,
635                v_head_dim: cfg.head_dim,
636            },
637            device: normal_loading_metadata.real_device.clone(),
638            dtype: vb.dtype(),
639            mapper,
640        })
641    }

647    #[allow(clippy::too_many_arguments)]
648    pub fn forward_embeds(
649        &self,
650        mut xs: Tensor,
651        attention_mask: Option<&Tensor>,
652        seqlen_offsets: &[usize],
653        _position_ids: &Tensor,
654        context_lens: Vec<(usize, usize)>,
655        metadata: Option<(Vec<(Tensor, Tensor)>, &PagedAttentionInputMetadata)>,
656        flash_params: &FlashParams,
657        visual_pos_masks: Option<&Tensor>,
658        deepstack_visual_embeds: Option<&[Tensor]>,
659    ) -> Result<Tensor> {
660        let cache = &mut self.cache.normal().0;
661
662        for (i, layer) in self.layers.iter().enumerate() {
663            xs = self.mapper.map(xs, i)?;
664            xs = layer.forward(
665                &xs,
666                attention_mask
667                    .as_ref()
668                    .map(|m| m.to_device(xs.device()).unwrap())
669                    .as_ref(),
670                seqlen_offsets,
671                &mut cache[i],
672                metadata
673                    .as_ref()
674                    .map(|(kv_cache, meta)| (kv_cache[i].clone(), *meta)),
675                flash_params,
676            )?;
677
678            // Integrate DeepStack visual features when provided.
679            if let (Some(visual_pos_masks), Some(deepstack)) =
680                (visual_pos_masks, deepstack_visual_embeds)
681            {
682                if i < deepstack.len() {
683                    xs = self.deepstack_process(xs, visual_pos_masks, &deepstack[i])?;
684                }
685            }
686        }
687        let xs = xs.to_device(&self.device)?;
688        let mut xs = xs.apply(&self.norm)?;
689        if let Some(t) = self.lm_head.quantized_act_type() {
690            xs = xs.to_dtype(t)?;
691        }
692        extract_logits(&self.lm_head.forward(&xs)?, context_lens)
693    }
694
695    fn deepstack_process(
696        &self,
697        hidden_states: Tensor,
698        visual_pos_masks: &Tensor,
699        visual_embeds: &Tensor,
700    ) -> Result<Tensor> {
701        let device = hidden_states.device();
702        let dtype = hidden_states.dtype();
703
704        let mask = visual_pos_masks.to_device(device)?.to_dtype(DType::F32)?;
705        let mask_flat = mask.flatten_all()?;
706
707        let masked_count = mask_flat.sum_all()?.to_scalar::<f32>()? as usize;
708        let visual_embeds = visual_embeds.to_device(device)?.to_dtype(dtype)?;
709
710        if masked_count == 0 {
711            if visual_embeds.dim(0)? != 0 {
712                candle_core::bail!(
713                    "DeepStack visual embeds ({}) provided but mask is empty",
714                    visual_embeds.dim(0)?
715                );
716            }
717            return Ok(hidden_states);
718        }
719
720        if visual_embeds.dim(0)? != masked_count {
721            candle_core::bail!(
722                "Mismatch between DeepStack visual embeds ({}) and mask positions ({})",
723                visual_embeds.dim(0)?,
724                masked_count
725            );
726        }
727
728        let (batch, seq, hidden) = hidden_states.dims3()?;
729        let total_positions = batch * seq;
730        let mut hidden_flat = hidden_states.reshape((total_positions, hidden))?;
731
732        let prefix = mask_flat.cumsum(0)?;
733        let rank = (prefix - &mask_flat)?.mul(&mask_flat)?;
734        let rank_u32 = rank.to_dtype(DType::U32)?;
735
736        let positions = Tensor::arange(0u32, total_positions as u32, device)?;
737        let positions_f32 = positions.to_dtype(DType::F32)?;
738        let masked_positions = positions_f32.mul(&mask_flat)?;
739
740        let mut position_per_rank = Tensor::zeros((masked_count,), DType::F32, device)?;
741        position_per_rank = position_per_rank.scatter_add(&rank_u32, &masked_positions, 0)?;
742        let position_per_rank = position_per_rank.to_dtype(DType::U32)?;
743
744        let linear_index = position_per_rank.unsqueeze(1)?.repeat((1, hidden))?;
745
746        hidden_flat = hidden_flat.scatter_add(&linear_index, &visual_embeds, 0)?;
747        hidden_flat.reshape((batch, seq, hidden))
748    }
749}

751impl IsqModel for Qwen3VLMoETextModel {
752    fn get_layers(
753        &mut self,
754    ) -> (
755        Vec<(&mut Arc<dyn QuantMethod>, Option<usize>)>,
756        &dyn DeviceMapper,
757    ) {
758        let mut tensors = Vec::new();
759        tensors.push((&mut self.lm_head, None));
760        for (i, layer) in self.layers.iter_mut().enumerate() {
761            tensors.push((&mut layer.self_attn.q_proj, Some(i)));
762            tensors.push((&mut layer.self_attn.k_proj, Some(i)));
763            tensors.push((&mut layer.self_attn.v_proj, Some(i)));
764            tensors.push((&mut layer.self_attn.o_proj, Some(i)));
765            match &mut layer.mlp {
766                MoeOrMlp::Mlp(mlp) => {
767                    tensors.push((&mut mlp.gate_proj, Some(i)));
768                    tensors.push((&mut mlp.up_proj, Some(i)));
769                    tensors.push((&mut mlp.down_proj, Some(i)));
770                }
771                MoeOrMlp::Moe(moe) => {
772                    for layer in moe.get_isq_layers() {
773                        tensors.push((layer, Some(i)));
774                    }
775                }
776            }
777        }
778        (tensors, &*self.mapper)
779    }
780
781    fn residual_tensors(&self) -> Vec<(String, Tensor)> {
782        let uvb = UnVarBuilder::new();
783
784        let uvb_lm = uvb.pp("model").pp("language_model");
785        uvb_lm.pp("embed_tokens").add(&self.embed_tokens);
786        uvb_lm.pp("norm").add(&self.norm);
787
788        for (layer_idx, layer) in self.layers.iter().enumerate() {
789            let uvb_l = uvb_lm.pp("layers").pp(layer_idx);
790            uvb_l.pp("input_layernorm").add(&layer.input_layernorm);
791            uvb_l
792                .pp("post_attention_layernorm")
793                .add(&layer.post_attention_layernorm);
794
795            uvb_l
796                .pp("self_attn")
797                .pp("q_norm")
798                .add(&layer.self_attn.q_norm);
799            uvb_l
800                .pp("self_attn")
801                .pp("k_norm")
802                .add(&layer.self_attn.k_norm);
803
804            if let MoeOrMlp::Moe(moe) = &layer.mlp {
805                uvb_l.pp("mlp").pp("gate").add(moe.gate());
806            }
807        }
808
809        uvb.to_safetensors()
810    }
