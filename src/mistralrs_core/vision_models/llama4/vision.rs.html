20struct Llama4UnfoldConvolution {
21    linear: Linear,
22    kernel_size: usize,
23    patch_size: usize,
24}

26impl Llama4UnfoldConvolution {
27    fn new(cfg: &VisionConfig, vb: ShardedVarBuilder) -> Result<Self> {
28        let kernel_size = cfg.patch_size;
29        let linear = linear_no_bias(
30            cfg.num_channels * kernel_size * kernel_size,
31            cfg.hidden_size,
32            vb.pp("linear"),
33        )?;
34        Ok(Self {
35            linear,
36            kernel_size,
37            patch_size: cfg.patch_size,
38        })
39    }
40
41    fn unfold(&self, xs: &Tensor) -> Result<Tensor> {
42        // In original code
43        let kernel_size = (self.kernel_size, self.kernel_size);
44        let stride = (self.patch_size, self.patch_size);
45        let padding = (0, 0);
46        let dilation = (1, 1);
47        let (bs, c, h, w) = xs.dims4()?;
48
49        let h_out = (h + 2 * padding.0 - dilation.0 * (kernel_size.0 - 1) - 1) / stride.0 + 1;
50        let w_out = (w + 2 * padding.1 - dilation.1 * (kernel_size.1 - 1) - 1) / stride.1 + 1;
51
52        // Extract blocks
53        let mut blocks = Vec::new();
54        for i in (0..h - kernel_size.0 * dilation.0 + 1).step_by(stride.0) {
55            for j in (0..w - kernel_size.1 * dilation.1 + 1).step_by(stride.1) {
56                let mut block = Vec::new();
57                for di in 0..kernel_size.0 {
58                    for dj in 0..kernel_size.1 {
59                        let h_idx = i + di * dilation.0;
60                        let w_idx = j + dj * dilation.0;
61                        // Get the block for all channels and add to our list
62                        block.push(xs.i((.., .., h_idx, w_idx))?);
63                    }
64                }
65
66                // Stack the channel-blocks
67                // (b, k*k, c)
68                let mut block = Tensor::stack(&block, 1)?;
69                block = block.permute((0, 2, 1))?;
70                blocks.push(block);
71            }
72        }
73
74        // (b, c, k*k, l)
75        let mut result = Tensor::stack(&blocks, D::Minus1)?;
76        // (b, c*k*k, l)
77        result = result.reshape((bs, c * kernel_size.0 * kernel_size.1, h_out * w_out))?;
78        Ok(result)
79    }
80
81    fn forward(&self, hidden_states: &Tensor) -> Result<Tensor> {
82        // let hidden_states = {
83        //     let mut patches = hidden_states
84        //         .unfold(2, self.kernel_size, self.patch_size)?
85        //         .unfold(3, self.kernel_size, self.patch_size)?;
86        //     patches = patches.contiguous()?.permute((0, 2, 3, 1, 4, 5))?;
87        //     let b = patches.dim(0)?;
88        //     let out_h = patches.dim(1)?;
89        //     let out_w = patches.dim(2)?;
90        //     let c = patches.dim(3)?;
91        //     patches.reshape((b, out_h * out_w, c * self.kernel_size * self.kernel_size))?
92        // };
93
94        let mut hidden_states = self.unfold(hidden_states)?;
95        hidden_states = hidden_states.transpose(1, 2)?;
96        self.linear.forward(&hidden_states)
97    }
98}

100struct Llama4VisionAttention {
101    q_proj: Arc<dyn QuantMethod>,
102    k_proj: Arc<dyn QuantMethod>,
103    v_proj: Arc<dyn QuantMethod>,
104    o_proj: Arc<dyn QuantMethod>,
105    sdpa_params: SdpaParams,
106    head_dim: usize,
107    freqs: Llama4VisionRotaryEmbedding,
108}
109
110impl Llama4VisionAttention {
111    fn new(
112        cfg: &VisionConfig,
113        vb: ShardedVarBuilder,
114        freqs: Llama4VisionRotaryEmbedding,
115        comm: &Arc<mistralrs_quant::Comm>,
116    ) -> Result<Self> {
117        let head_dim = cfg.hidden_size / cfg.num_attention_heads;
118        Ok(Self {
119            q_proj: ColumnParallelLayer::new(
120                cfg.hidden_size,
121                cfg.num_attention_heads * head_dim,
122                &None,
123                true,
124                comm,
125                vb.pp("q_proj"),
126            )?,
127            k_proj: ColumnParallelLayer::new(
128                cfg.hidden_size,
129                cfg.num_attention_heads * head_dim,
130                &None,
131                true,
132                comm,
133                vb.pp("k_proj"),
134            )?,
135            v_proj: ColumnParallelLayer::new(
136                cfg.hidden_size,
137                cfg.num_attention_heads * head_dim,
138                &None,
139                true,
140                comm,
141                vb.pp("v_proj"),
142            )?,
143            o_proj: RowParallelLayer::new(
144                cfg.hidden_size,
145                cfg.num_attention_heads * head_dim,
146                &None,
147                true,
148                comm,
149                vb.pp("o_proj"),
150            )?,
151            sdpa_params: SdpaParams {
152                n_kv_groups: 1,
153                softcap: None,
154                softmax_scale: 1.0 / (head_dim as f32).sqrt(),
155                sliding_window: None,
156            },
157            head_dim,
158            freqs,
159        })
160    }
161
162    fn forward(&self, hidden_state: &Tensor, attention_mask: Option<&Tensor>) -> Result<Tensor> {
163        let mut hidden_state = hidden_state.clone();
164        let original_dtype = hidden_state.dtype();
165        if let Some(t) = self.q_proj.quantized_act_type() {
166            hidden_state = hidden_state.to_dtype(t)?;
167        }
168        let mut q = self.q_proj.forward(&hidden_state)?;
169        let mut k = self.k_proj.forward(&hidden_state)?;
170        let mut v = self.v_proj.forward(&hidden_state)?;
171        if self.q_proj.quantized_act_type().is_some() {
172            q = q.to_dtype(original_dtype)?;
173            k = k.to_dtype(original_dtype)?;
174            v = v.to_dtype(original_dtype)?;
175        }
176
177        // Should be same, no caching...
178        let (bs, q_sq, _) = q.dims3()?;
179        let (_, k_sq, _) = k.dims3()?;
180
181        q = q
182            .reshape((bs, q_sq, (), self.head_dim))?
183            .transpose(1, 2)?
184            .contiguous()?;
185        k = k
186            .reshape((bs, k_sq, (), self.head_dim))?
187            .transpose(1, 2)?
188            .contiguous()?;
189        v = v
190            .reshape((bs, k_sq, (), self.head_dim))?
191            .transpose(1, 2)?
192            .contiguous()?;
193
194        // Apply rope
195        {
196            q = candle_nn::rotary_emb::rope_i(&q, &self.freqs.cos, &self.freqs.sin)?;
197            k = candle_nn::rotary_emb::rope_i(&k, &self.freqs.cos, &self.freqs.sin)?;
198        }
199
200        let mut attn_output = Sdpa
201            .run_attention(&q, &k, &v, attention_mask, None, &self.sdpa_params)?
202            .transpose(1, 2)?
203            .contiguous()?
204            .reshape((bs, q_sq, ()))?
205            .to_dtype(q.dtype())?;
206
207        if let Some(t) = self.q_proj.quantized_act_type() {
208            attn_output = attn_output.to_dtype(t)?;
209        }
210        let mut res = self.o_proj.forward(&attn_output)?;
211        if self.q_proj.quantized_act_type().is_some() {
212            res = res.to_dtype(original_dtype)?;
213        }
214        Ok(res)
215    }
216}

267struct Llama4VisionEncoderLayer {
268    self_attn: Llama4VisionAttention,
269    mlp: Llama4Mlp,
270    input_layernorm: LayerNorm,
271    post_attention_layernorm: LayerNorm,
272}
273
274impl Llama4VisionEncoderLayer {
275    fn new(
276        cfg: &VisionConfig,
277        vb: ShardedVarBuilder,
278        freqs: Llama4VisionRotaryEmbedding,
279        real_dev: &Device,
280        comm: &Arc<mistralrs_quant::Comm>,
281    ) -> Result<Self> {
282        let self_attn = Llama4VisionAttention::new(cfg, vb.pp("self_attn"), freqs, comm)?;
283        let mlp = Llama4Mlp::new(cfg, vb.pp("mlp"), comm)?;
284
285        let input_layernorm = layer_norm(
286            cfg.hidden_size,
287            cfg.norm_eps,
288            vb.pp("input_layernorm").set_device(real_dev.clone()),
289        )?;
290        let post_attention_layernorm = layer_norm(
291            cfg.hidden_size,
292            cfg.norm_eps,
293            vb.pp("post_attention_layernorm")
294                .set_device(real_dev.clone()),
295        )?;
296
297        Ok(Self {
298            self_attn,
299            mlp,
300            input_layernorm,
301            post_attention_layernorm,
302        })
303    }
304
305    fn forward(&self, hidden_state: &Tensor, attention_mask: Option<&Tensor>) -> Result<Tensor> {
306        // Self attn
307        let residual = hidden_state;
308        let mut hidden_state = self.input_layernorm.forward(hidden_state)?;
309
310        hidden_state = self.self_attn.forward(&hidden_state, attention_mask)?;
311        hidden_state = (residual + hidden_state)?;
312
313        // FF
314        let residual = hidden_state.clone();
315        hidden_state = self.post_attention_layernorm.forward(&hidden_state)?;
316
317        hidden_state = self.mlp.forward(&hidden_state)?;
318        residual + hidden_state
319    }
320}

322struct Llama4VisionEncoder {
323    layers: Vec<Llama4VisionEncoderLayer>,
324}
325
326impl Llama4VisionEncoder {
327    fn new(
328        cfg: &VisionConfig,
329        num_layers: usize,
330        vb: ShardedVarBuilder,
331        freqs: Llama4VisionRotaryEmbedding,
332        real_dev: &Device,
333        comm: &Arc<mistralrs_quant::Comm>,
334        multi_progress: &Arc<MultiProgress>,
335    ) -> Result<Self> {
336        let layers_vb = vb.pp("layers");
337        let layers = NiceProgressBar::<_, 'b'>(
338            0..num_layers,
339            "Loading vision repeating layers",
340            multi_progress,
341        )
342        .par_iter_if_isq(|i| {
343            Llama4VisionEncoderLayer::new(cfg, layers_vb.pp(i), freqs.clone(), real_dev, comm)
344        })?;
345        Ok(Self { layers })
346    }
347
348    fn forward_with_states(
349        &self,
350        hidden_state: &Tensor,
351        attention_mask: Option<&Tensor>,
352    ) -> Result<Tensor> {
353        let mut hidden_state = hidden_state.clone();
354        for layer in self.layers.iter() {
355            hidden_state = layer.forward(&hidden_state, attention_mask)?;
356        }
357        Ok(hidden_state)
358    }
359
360    fn residual_tensors(&self) -> Vec<(String, Tensor)> {
361        let uvb_t = UnVarBuilder::new();
362
363        for (i, layer) in self.layers.iter().enumerate() {
364            let uvb_l = uvb_t.pp("layers").pp(i);
365            uvb_l.pp("input_layernorm").add(&layer.input_layernorm);
366            uvb_l
367                .pp("post_attention_layernorm")
368                .add(&layer.post_attention_layernorm);
369        }
370
371        uvb_t.to_safetensors()
372    }
373}

426struct Llama4VisionPixelShuffle {
427    mlp: Llama4VisionPixelShuffleMLP,
428    pixel_shuffle_ratio: f32,
429}
430
431impl Llama4VisionPixelShuffle {
432    fn new(
433        cfg: &VisionConfig,
434        vb: ShardedVarBuilder,
435        comm: &Arc<mistralrs_quant::Comm>,
436    ) -> Result<Self> {
437        let mlp = Llama4VisionPixelShuffleMLP::new(cfg, vb.pp("mlp"), comm)?;
438        Ok(Self {
439            mlp,
440            pixel_shuffle_ratio: cfg.pixel_shuffle_ratio,
441        })
442    }
443
444    fn pixel_shuffle(&self, xs: &Tensor) -> Result<Tensor> {
445        let (bs, num_patches, _c) = xs.dims3()?;
446        let patch_size = (num_patches as f32).sqrt() as usize;
447
448        let mut xs = xs.reshape((bs, patch_size, patch_size, ()))?;
449        let (_bs, h, w, c) = xs.dims4()?;
450
451        xs = xs.reshape((
452            bs,
453            h,
454            (w as f32 * self.pixel_shuffle_ratio) as usize,
455            (c as f32 / self.pixel_shuffle_ratio) as usize,
456        ))?;
457        xs = xs.permute((0, 2, 1, 3))?.contiguous()?;
458
459        xs = xs.reshape((
460            bs,
461            (h as f32 * self.pixel_shuffle_ratio) as usize,
462            (w as f32 * self.pixel_shuffle_ratio) as usize,
463            (c as f32 / self.pixel_shuffle_ratio.powi(2)) as usize,
464        ))?;
465        xs = xs.permute((0, 2, 1, 3))?.contiguous()?;
466
467        xs.reshape((bs, (), xs.dim(D::Minus1)?))
468    }
469
470    fn forward(&self, encoded_patches: &Tensor) -> Result<Tensor> {
471        let encoded_patches = self.pixel_shuffle(encoded_patches)?;
472        self.mlp.forward(&encoded_patches)
473    }
474}

477struct Llama4VisionRotaryEmbedding {
478    cos: Tensor,
479    sin: Tensor,
480}
481
482impl Llama4VisionRotaryEmbedding {
483    fn new(cfg: &VisionConfig, device: &Device, dtype: DType) -> Result<Self> {
484        let idx = cfg.image_size / cfg.patch_size;
485        let mut img_idx =
486            Tensor::arange(0f32, idx.pow(2) as f32, device)?.reshape((idx.pow(2), 1))?;
487        img_idx = Tensor::cat(&[&img_idx, &img_idx.narrow(0, 0, 1)?], 0)?;
488        // Insert ID_CLS_TOKEN in the bottom right
489        img_idx = img_idx.slice_assign(
490            &[
491                img_idx.dim(0)? - 1..img_idx.dim(0)?,
492                img_idx.dim(1)? - 1..img_idx.dim(1)?,
493            ],
494            &Tensor::new(-2f32, device)?.reshape((1, 1))?,
495        )?;
496        let img_ids_flat = img_idx.flatten_all()?.to_vec1::<f32>()?;
497        // frequencies_x = img_idx % idx
498        // get the coordinates of the 2d matrix along x
499        let frequencies_x = {
500            let frequencies_x = img_ids_flat
501                .iter()
502                .map(|x| x % idx as f32)
503                .collect::<Vec<_>>();
504            Tensor::from_vec(frequencies_x, img_idx.shape().clone(), device)?
505        };
506        // frequencies_y = img_idx // idx
507        // get the coordinates of the 2d matrix along y
508        let frequencies_y = {
509            let frequencies_y = img_ids_flat
510                .iter()
511                .map(|x| x / idx as f32)
512                .collect::<Vec<_>>();
513            Tensor::from_vec(frequencies_y, img_idx.shape().clone(), device)?
514        };
515        let rope_freq = {
516            let freq_dim = cfg.hidden_size / cfg.num_attention_heads / 2;
517            let freqs: Vec<_> = (0..freq_dim)
518                .step_by(2)
519                .take(freq_dim / 2)
520                .map(|i| 1f32 / cfg.rope_theta.powf(i as f32 / freq_dim as f32))
521                .collect();
522            let freqs_len = freqs.len();
523            Tensor::from_vec(freqs, freqs_len, device)?
524        };
525        let freqs_x = (frequencies_x + 1.)?
526            .unsqueeze(D::Minus1)?
527            .broadcast_mul(&rope_freq.unsqueeze(0)?.unsqueeze(0)?)?
528            .repeat_interleave(2, D::Minus1)?;
529        let freqs_y = (frequencies_y + 1.)?
530            .unsqueeze(D::Minus1)?
531            .broadcast_mul(&rope_freq.unsqueeze(0)?.unsqueeze(0)?)?
532            .repeat_interleave(2, D::Minus1)?;
533        let mut freqs = {
534            let freqs = Tensor::cat(&[freqs_x, freqs_y], D::Minus1)?.contiguous()?;
535            // This implements [..., ::2]
536            let indices_every_two = Tensor::new(
537                (0..freqs.dim(D::Minus1)?)
538                    .step_by(2)
539                    .map(|x| x as u32)
540                    .collect::<Vec<_>>(),
541                device,
542            )?;
543            freqs.index_select(&indices_every_two, D::Minus1)?
544        };
545        freqs = freqs.squeeze(1)?;
546        freqs = freqs.lt(0.)?.where_cond(&freqs.zeros_like()?, &freqs)?;
547
548        Ok(Self {
549            cos: freqs.cos()?.to_dtype(dtype)?,
550            sin: freqs.sin()?.to_dtype(dtype)?,
551        })
552    }
553}

555pub(super) struct Llama4VisionModel {
556    patch_embedding: Llama4UnfoldConvolution,
557    class_embedding: Tensor,
558    positional_embedding_vlm: Tensor,
559    layernorm_pre: LayerNorm,
560    layernorm_post: LayerNorm,
561    model: Llama4VisionEncoder,
562    vision_adapter: Llama4VisionPixelShuffle,
563}
564
565impl Llama4VisionModel {
566    pub(super) fn new(
567        cfg: &VisionConfig,
568        vb: ShardedVarBuilder,
569        real_dev: &Device,
570        comm: &Arc<mistralrs_quant::Comm>,
571        multi_progress: &Arc<MultiProgress>,
572    ) -> Result<Self> {
573        let patch_embedding = Llama4UnfoldConvolution::new(
574            cfg,
575            vb.pp("patch_embedding").set_device(real_dev.clone()),
576        )?;
577
578        let class_embedding = vb
579            .get((cfg.hidden_size,), "class_embedding")?
580            .to_device(real_dev)?;
581        let num_patches = cfg.num_patches();
582        let positional_embedding_vlm = vb
583            .get((num_patches, cfg.hidden_size), "positional_embedding_vlm")?
584            .to_device(real_dev)?;
585
586        // layer norms
587        let layernorm_pre = layer_norm(
588            cfg.hidden_size,
589            LayerNormConfig::default(),
590            vb.pp("layernorm_pre").set_device(real_dev.clone()),
591        )?;
592        let layernorm_post = layer_norm(
593            cfg.hidden_size,
594            LayerNormConfig::default(),
595            vb.pp("layernorm_post").set_device(real_dev.clone()),
596        )?;
597
598        let rotary_embedding = Llama4VisionRotaryEmbedding::new(cfg, real_dev, vb.dtype())?;
599        let model = Llama4VisionEncoder::new(
600            cfg,
601            cfg.num_hidden_layers,
602            vb.pp("model"),
603            rotary_embedding,
604            real_dev,
605            comm,
606            multi_progress,
607        )?;
608
609        let vision_adapter = Llama4VisionPixelShuffle::new(cfg, vb.pp("vision_adapter"), comm)?;
610
611        assert_eq!(cfg.vision_feature_layer, -1);
612
613        Ok(Self {
614            patch_embedding,
615            class_embedding,
616            positional_embedding_vlm,
617            layernorm_post,
618            layernorm_pre,
619            model,
620            vision_adapter,
621        })
622    }
623
624    pub(super) fn forward(&self, pixel_values: &Tensor) -> Result<Tensor> {
625        let pixel_values = pixel_values.to_dtype(self.class_embedding.dtype())?;
626
627        let (bs_times_num_tiles, _num_channels, _height, _width) = pixel_values.dims4()?;
628        let num_concurrent_media = 1;
629
630        // Patch embedding
631        let mut hidden_state = self.patch_embedding.forward(&pixel_values)?;
632        let (_, mut num_patches, hidden_dim) = hidden_state.dims3()?;
633
634        // Add cls token
635        hidden_state = hidden_state.reshape((
636            bs_times_num_tiles * num_concurrent_media,
637            num_patches,
638            hidden_dim,
639        ))?;
640        let class_embedding =
641            self.class_embedding
642                .expand((hidden_state.dim(0)?, 1, hidden_state.dim(D::Minus1)?))?;
643        hidden_state = Tensor::cat(&[hidden_state, class_embedding], 1)?;
644        num_patches += 1;
645
646        // Position embeddings
647        hidden_state = hidden_state.reshape((
648            bs_times_num_tiles * num_concurrent_media,
649            num_patches,
650            hidden_dim,
651        ))?;
652        hidden_state = hidden_state.broadcast_add(&self.positional_embedding_vlm)?;
653
654        hidden_state = self.layernorm_pre.forward(&hidden_state)?;
655
656        hidden_state = hidden_state.reshape((bs_times_num_tiles, (), hidden_dim))?;
657
658        // Apply encoder
659        hidden_state =
660            hidden_state.reshape((bs_times_num_tiles * num_concurrent_media, (), hidden_dim))?;
661        hidden_state = self.model.forward_with_states(&hidden_state, None)?;
662
663        hidden_state = self.layernorm_post.forward(&hidden_state)?;
664
665        hidden_state = hidden_state.narrow(1, 0, hidden_state.dim(1)? - 1)?;
666
667        self.vision_adapter.forward(&hidden_state)
668    }
669
670    pub fn get_isq_layers(&mut self) -> Vec<&mut std::sync::Arc<dyn mistralrs_quant::QuantMethod>> {
671        let mut layers = Vec::new();
672        for layer in &mut self.model.layers {
673            layers.push(&mut layer.self_attn.q_proj);
674            layers.push(&mut layer.self_attn.k_proj);
675            layers.push(&mut layer.self_attn.v_proj);
676            layers.push(&mut layer.self_attn.o_proj);
677
678            layers.push(&mut layer.mlp.fc1);
679            layers.push(&mut layer.mlp.fc2);
680        }
681        layers.push(&mut self.vision_adapter.mlp.fc1);
682        layers.push(&mut self.vision_adapter.mlp.fc2);
683        layers
684    }
685}

700        let uvb = UnVarBuilder::new();
701

703            .pp("linear")
704            .add(&self.patch_embedding.linear);
705        uvb.add_tensor("class_embedding", self.class_embedding.clone());
706        uvb.add_tensor(
707            "positional_embedding_vlm",
708            self.positional_embedding_vlm.clone(),
709        );
710
711        uvb.pp("layernorm_pre").add(&self.layernorm_pre);
712        uvb.pp("layernorm_post").add(&self.layernorm_post);
713
714        uvb.pp("model").extend(self.model.residual_tensors());
715
716        uvb.to_safetensors()
717    }
