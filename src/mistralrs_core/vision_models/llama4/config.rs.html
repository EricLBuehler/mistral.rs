45impl TextConfig {
46    pub fn moe_layers(&self) -> Vec<usize> {
47        self.moe_layers.clone().unwrap_or(
48            (self.interleave_moe_layer_step - 1..self.num_hidden_layers)
49                .step_by(self.interleave_moe_layer_step)
50                .collect(),
51        )
52    }
53}
