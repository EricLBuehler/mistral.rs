37    rotary_emb: Arc<Llama3RotaryEmbedding>,
38    max_seq_len: usize,
39    paged_attn: Option<PagedAttention>,
40    sdpa_params: SdpaParams,
41    norm: Option<RmsNorm>,

44    attn_scale: Option<f32>,
45    attn_temperature_tuning: Option<f32>,

48impl CausalSelfAttention {
49    #[allow(clippy::too_many_arguments)]
50    fn new(
51        vb: ShardedVarBuilder,
52        cfg: &TextConfig,
53        layer_idx: usize,
54        loading_isq: bool,
55        mapper: &dyn DeviceMapper,
56        rope: Arc<Llama3RotaryEmbedding>,
57        paged_attn: Option<PagedAttention>,
58        comm: &Arc<mistralrs_quant::Comm>,
59    ) -> Result<Self> {
60        let size_in = cfg.hidden_size;
61        let size_q = (cfg.hidden_size / cfg.num_attention_heads) * cfg.num_attention_heads;
62        let size_kv = (cfg.hidden_size / cfg.num_attention_heads) * cfg.num_key_value_heads;
63        let q_proj = ColumnParallelLayer::new(
64            size_in,
65            size_q,
66            &cfg.quantization_config,
67            false,
68            comm,
69            mapper.set_device(layer_idx, vb.pp("q_proj"), loading_isq),
70        )?;
71        let kv_shard = mistralrs_quant::compute_kv_shard(
72            cfg.num_key_value_heads,
73            cfg.hidden_size / cfg.num_attention_heads,
74            comm,
75        );
76        let k_proj = ColumnParallelLayer::new_with_shard(
77            size_in,
78            size_kv,
79            &cfg.quantization_config,
80            false,
81            comm,
82            kv_shard,
83            mapper.set_device(layer_idx, vb.pp("k_proj"), loading_isq),
84        )?;
85        let v_proj = ColumnParallelLayer::new_with_shard(
86            size_in,
87            size_kv,
88            &cfg.quantization_config,
89            false,
90            comm,
91            kv_shard,
92            mapper.set_device(layer_idx, vb.pp("v_proj"), loading_isq),
93        )?;
94        let o_proj = RowParallelLayer::new(
95            size_q,
96            size_in,
97            &cfg.quantization_config,
98            false,
99            comm,
100            mapper.set_device(layer_idx, vb.pp("o_proj"), loading_isq),
101        )?;
102        let use_rope = (layer_idx + 1) % 4 != 0;
103        let head_dim = cfg.hidden_size / cfg.num_attention_heads;
104        let norm = if cfg.use_qk_norm && use_rope {
105            let vb = mapper.set_device(layer_idx, vb, false);
106            Some(RmsNorm::from_w(
107                Tensor::ones(head_dim, vb.dtype(), vb.device())?,
108                1e-6,
109            )?)
110        } else {
111            None
112        };
113
114        Ok(Self {
115            q_proj,
116            k_proj,
117            v_proj,
118            o_proj,
119            num_attention_heads: cfg.num_attention_heads / comm.world_size(),
120            num_key_value_heads: (cfg.num_key_value_heads / comm.world_size()).max(1),
121            head_dim,
122            rotary_emb: rope,
123            max_seq_len: cfg.max_position_embeddings,
124            paged_attn,
125            sdpa_params: SdpaParams {
126                n_kv_groups: mistralrs_quant::compute_n_kv_groups(
127                    cfg.num_key_value_heads,
128                    cfg.num_attention_heads,
129                    comm,
130                ),
131                softcap: None,
132                softmax_scale: 1.0 / (head_dim as f32).sqrt(),
133                sliding_window: None,
134            },
135            norm,
136            use_rope,
137            floor_scale: cfg.floor_scale,
138            attn_scale: cfg.attn_scale,
139            attn_temperature_tuning: cfg.attn_temperature_tuning,
140        })
141    }
142
143    #[allow(clippy::too_many_arguments)]
144    fn forward(
145        &self,
146        x: &Tensor,
147        position_ids: &Tensor,
148        attention_mask: &Option<Tensor>,
149        seqlen_offsets: &[usize],
150        kv_cache: &mut KvCache,
151        metadata: Option<((Tensor, Tensor), &PagedAttentionInputMetadata)>,
152        flash_params: &FlashParams,
153    ) -> Result<Tensor> {
154        let (b_sz, seq_len, _) = x.dims3()?;
155
156        let mut q = self.q_proj.forward_autocast(x)?;
157        let mut k = self.k_proj.forward_autocast(x)?;
158        let mut v = self.v_proj.forward_autocast(x)?;
159
160        q = q
161            .reshape((b_sz, seq_len, self.num_attention_heads, self.head_dim))?
162            .transpose(1, 2)?;
163        k = k
164            .reshape((b_sz, seq_len, self.num_key_value_heads, self.head_dim))?
165            .transpose(1, 2)?;
166        v = v
167            .reshape((b_sz, seq_len, self.num_key_value_heads, self.head_dim))?
168            .transpose(1, 2)?;
169
170        if self.use_rope {
171            (q, k) = self.rotary_emb.forward(&q, &k, seqlen_offsets)?;
172        }
173
174        if let Some(qk_norm) = &self.norm {
175            q = qk_norm.forward(&q)?;
176            k = qk_norm.forward(&k)?;
177        }
178
179        if self.attn_temperature_tuning.is_some() && !self.use_rope {
180            let floor_scale = self.floor_scale.unwrap() as f64;
181            let attn_scale = self.attn_scale.unwrap() as f64;
182            let floor = ((position_ids.to_dtype(DType::F32)? + 1.)? / floor_scale)?.floor()?;
183            let attn_scales = (((floor + 1.0)?.log()? * attn_scale)? + 1.0)?;
184
185            q = q
186                .to_dtype(DType::F32)?
187                .broadcast_mul(&attn_scales.unsqueeze(D::Minus1)?)?
188                .to_dtype(q.dtype())?;
189        }
190
191        let mut y = match &self.paged_attn {
192            Some(paged_attn) => match metadata {
193                Some(((key_cache, value_cache), input_metadata)) => paged_attn.forward(
194                    &q,
195                    &k,
196                    &v,
197                    attention_mask.clone().as_ref(),
198                    Some(key_cache),
199                    Some(value_cache),
200                    input_metadata,
201                    &self.sdpa_params,
202                    Some(flash_params),
203                )?,
204                None => {
205                    // If we don't have metadata, we are most likely generating an imatrix so we don't want to populate that.
206                    // Generating the dummy metadata with the assumption that we are not generating text (only processing prompts).
207                    let input_metadata = PagedAttentionInputMetadata::dummy(q.device())?;
208                    // Sanity check.
209                    assert!(attention_mask.is_some());
210                    paged_attn.forward(
211                        &q,
212                        &k,
213                        &v,
214                        attention_mask.clone().as_ref(),
215                        None,
216                        None,
217                        &input_metadata,
218                        &self.sdpa_params,
219                        Some(flash_params),
220                    )?
221                }
222            },
223            None => {
224                let (k, v) = kv_cache.append(&k, &v)?;
225
226                Sdpa.run_attention(
227                    &q.contiguous()?,
228                    &k.contiguous()?,
229                    &v.contiguous()?,
230                    attention_mask.clone().as_ref(),
231                    Some(flash_params),
232                    &self.sdpa_params,
233                )?
234            }
235        };
236
237        y = if attention_mask.is_some() {
238            y.transpose(1, 2)?.reshape((b_sz, seq_len, ()))?
239        } else {
240            y.reshape((b_sz, seq_len, ()))?
241        };
242        self.o_proj.forward_autocast(&y)
243    }
244}

246struct Mlp {
247    gate: Arc<dyn QuantMethod>,
248    up: Arc<dyn QuantMethod>,
249    down: Arc<dyn QuantMethod>,
250    act: Activation,
251}
252
253impl Mlp {
254    fn new(
255        vb: ShardedVarBuilder,
256        hidden_size: usize,
257        intermediate_size: usize,
258        quantization_config: &Option<QuantizedConfig>,
259        hidden_act: Activation,
260        comm: &Arc<mistralrs_quant::Comm>,
261    ) -> Result<Self> {
262        Ok(Self {
263            gate: ColumnParallelLayer::new(
264                hidden_size,
265                intermediate_size,
266                quantization_config,
267                false,
268                comm,
269                vb.pp("gate_proj"),
270            )?,
271            up: ColumnParallelLayer::new(
272                hidden_size,
273                intermediate_size,
274                quantization_config,
275                false,
276                comm,
277                vb.pp("up_proj"),
278            )?,
279            down: RowParallelLayer::new(
280                intermediate_size,
281                hidden_size,
282                quantization_config,
283                false,
284                comm,
285                vb.pp("down_proj"),
286            )?,
287            act: hidden_act,
288        })
289    }
290
291    fn forward(&self, xs: &Tensor) -> Result<Tensor> {
292        let lhs = self.gate.forward_autocast(xs)?;
293        let rhs = self.up.forward_autocast(xs)?;
294
295        self.down
296            .forward_autocast(&crate::ops::mul_and_act(&lhs, &rhs, self.act)?)
297    }
298}

300struct TextExperts {
301    gate_proj: Vec<Arc<dyn QuantMethod>>,
302    up_proj: Vec<Arc<dyn QuantMethod>>,
303    down_proj: Vec<Arc<dyn QuantMethod>>,
304    act: Activation,
305    hidden_size: usize,
306    sum_all_reduce: SumAllReduce,
307}
308
309impl TextExperts {
310    fn new(
311        vb: ShardedVarBuilder,
312        cfg: &TextConfig,
313        quantization_config: &Option<QuantizedConfig>,
314        comm: &Arc<mistralrs_quant::Comm>,
315    ) -> Result<Self> {
316        let PackedExperts {
317            gate_proj,
318            up_proj,
319            down_proj,
320        } = PackedExperts::new(
321            cfg.num_local_experts,
322            cfg.hidden_size,
323            cfg.intermediate_size,
324            quantization_config,
325            false,
326            comm,
327            vb,
328        )?;
329        Ok(Self {
330            gate_proj,
331            up_proj,
332            down_proj,
333            act: cfg.hidden_act,
334            hidden_size: cfg.hidden_size,
335            sum_all_reduce: SumAllReduce::new(comm),
336        })
337    }
338
339    // xs: (bs * seq_len, hidden_size)
340    // expert indices: (bs * seq_len)
341    fn forward(&self, xs: &Tensor, indices: &Tensor) -> Result<Tensor> {
342        let xs = xs.unsqueeze(1)?;
343
344        if self.gate_proj.len() == 1 {
345            let gate = self.gate_proj[0].gather_forward_autocast(&xs, indices)?;
346            let up = self.up_proj[0].gather_forward_autocast(&xs, indices)?;
347            let mut xs = self.down_proj[0]
348                .gather_forward_autocast(&(up * gate.apply(&self.act)?)?, indices)?;
349            xs = self.sum_all_reduce.sum_all_reduce(&xs)?;
350            xs.reshape(((), self.hidden_size))
351        } else {
352            let indices = indices.to_vec1::<u32>()?;
353            let mut results = Vec::new();
354            for (tok, id) in indices.into_iter().enumerate() {
355                let xs = xs.i(tok)?.reshape((1, self.hidden_size))?;
356
357                let res = {
358                    let gate = self.gate_proj[id as usize].forward_autocast(&xs)?;
359                    let up = self.up_proj[id as usize].forward_autocast(&xs)?;
360                    self.down_proj[id as usize].forward_autocast(&(up * gate.apply(&self.act)?)?)?
361                };
362                results.push(res);
363            }
364            let mut xs = Tensor::cat(&results, 0)?;
365            xs = self.sum_all_reduce.sum_all_reduce(&xs)?;
366            xs.reshape(((), self.hidden_size))
367        }
368    }
369}

371struct TextMoe {
372    experts: TextExperts,
373    shared_expert: Mlp,
374    router: Arc<dyn QuantMethod>,
375    topk: usize,
376}
377
378impl TextMoe {
379    fn new(
380        vb: ShardedVarBuilder,
381        cfg: &TextConfig,
382        quantization_config: &Option<QuantizedConfig>,
383        comm: &Arc<mistralrs_quant::Comm>,
384    ) -> Result<Self> {
385        let experts = TextExperts::new(vb.pp("experts"), cfg, quantization_config, comm)?;
386        let router = linear_no_bias(
387            cfg.hidden_size,
388            cfg.num_local_experts,
389            quantization_config,
390            vb.pp("router"),
391        )?;
392        let shared_expert = Mlp::new(
393            vb.pp("shared_expert"),
394            cfg.hidden_size,
395            cfg.intermediate_size,
396            quantization_config,
397            cfg.hidden_act,
398            comm,
399        )?;
400        Ok(Self {
401            experts,
402            shared_expert,
403            router,
404            topk: cfg.num_experts_per_tok,
405        })
406    }
407
408    fn forward(&self, xs: &Tensor) -> Result<Tensor> {
409        let (bs, seq_len, hidden_dim) = xs.dims3()?;
410        let xs = xs.reshape(((), hidden_dim))?;
411        let router_logits = self.router.forward_autocast(&xs)?;
412
413        let TopKOutput {
414            values: router_top_value,
415            indices: router_indices,
416        } = router_logits.topk(self.topk)?;
417
418        let router_scores = candle_nn::ops::sigmoid(&router_top_value.to_dtype(DType::F32)?)?
419            .to_dtype(router_top_value.dtype())?;
420
421        let routed_in = xs.broadcast_mul(&router_scores)?;
422        let routed_out = self
423            .experts
424            .forward(&routed_in, &router_indices.squeeze(D::Minus1)?)?
425            .reshape((bs, seq_len, hidden_dim))?;
426        let out = self
427            .shared_expert
428            .forward(&xs.reshape((bs, seq_len, hidden_dim))?)?;
429
430        out + routed_out
431    }
432}

448struct Block {
449    rms_1: RmsNorm,
450    attn: CausalSelfAttention,
451    rms_2: RmsNorm,
452    ff: MoeOrMlp,
453    use_chunked_attention: bool,
454}
455
456impl Block {
457    #[allow(clippy::too_many_arguments)]
458    fn new(
459        vb: ShardedVarBuilder,
460        cfg: &TextConfig,
461        mapper: &dyn DeviceMapper,
462        layer_idx: usize,
463        loading_isq: bool,
464        rope: Arc<Llama3RotaryEmbedding>,
465        paged_attn: Option<PagedAttention>,
466        comm: &Arc<mistralrs_quant::Comm>,
467    ) -> Result<Self> {
468        let use_chunked_attention = (layer_idx + 1) % 4 != 0;
469        let attn = CausalSelfAttention::new(
470            vb.pp("self_attn"),
471            cfg,
472            layer_idx,
473            loading_isq,
474            mapper,
475            rope,
476            paged_attn,
477            comm,
478        )?;
479        let is_moe_layer = cfg.moe_layers().contains(&layer_idx);
480        let ff = if is_moe_layer {
481            let moe = TextMoe::new(
482                mapper.set_device(layer_idx, vb.pp("feed_forward"), loading_isq),
483                cfg,
484                &cfg.quantization_config,
485                comm,
486            )?;
487            MoeOrMlp::Moe(moe)
488        } else {
489            let mlp = Mlp::new(
490                mapper.set_device(layer_idx, vb.pp("feed_forward"), loading_isq),
491                cfg.hidden_size,
492                cfg.intermediate_size_mlp,
493                &cfg.quantization_config,
494                cfg.hidden_act,
495                comm,
496            )?;
497            MoeOrMlp::Mlp(mlp)
498        };
499        let rms_1 = RmsNorm::new(
500            cfg.hidden_size,
501            cfg.rms_norm_eps,
502            mapper.set_device(layer_idx, vb.pp("input_layernorm"), false),
503        )?;
504        let rms_2 = RmsNorm::new(
505            cfg.hidden_size,
506            cfg.rms_norm_eps,
507            mapper.set_device(layer_idx, vb.pp("post_attention_layernorm"), false),
508        )?;
509        Ok(Self {
510            rms_1,
511            attn,
512            rms_2,
513            ff,
514            use_chunked_attention,
515        })
516    }
517
518    #[allow(clippy::too_many_arguments)]
519    fn forward(
520        &self,
521        x: &Tensor,
522        position_ids: &Tensor,
523        attention_mask: &Option<Tensor>,
524        chunked_mask: &Option<Tensor>,
525        seqlen_offsets: &[usize],
526        kv_cache: &mut KvCache,
527        metadata: Option<((Tensor, Tensor), &PagedAttentionInputMetadata)>,
528        flash_params: &FlashParams,
529    ) -> Result<Tensor> {
530        let residual = x;
531        let x = self.rms_1.forward(x)?;
532        let mask = if self.use_chunked_attention {
533            chunked_mask
534        } else {
535            attention_mask
536        };
537        let x = (self.attn.forward(
538            &x,
539            position_ids,
540            mask,
541            seqlen_offsets,
542            kv_cache,
543            metadata,
544            flash_params,
545        )? + residual)?;
546        let residual = &x;
547        let x = (self.ff.forward(&self.rms_2.forward(&x)?)? + residual)?;
548        Ok(x)
549    }
550}

554    blocks: Vec<Block>,
555    ln_f: RmsNorm,
556    lm_head: Arc<dyn QuantMethod>,
557    kv_cache: crate::pipeline::EitherCache,
558    device: Device,

560    cfg: ModelConfigMetadata,

563
564impl TextModel {
565    pub fn new(
566        cfg: &TextConfig,
567        vb: ShardedVarBuilder,
568        is_gptx: bool,
569        normal_loading_metadata: NormalLoadingMetadata,
570        attention_mechanism: AttentionImplementation,
571    ) -> Result<Self> {
572        let vb_m = vb.pp("model");
573        let vb_lm_head = vb.pp("lm_head");
574        Self::new_inner(
575            cfg,
576            vb_m,
577            vb_lm_head,
578            is_gptx,
579            normal_loading_metadata,
580            attention_mechanism,
581        )
582    }
583
584    pub fn new_inner(
585        cfg: &TextConfig,
586        vb_m: ShardedVarBuilder,
587        vb_lm_head: ShardedVarBuilder,
588        is_gptx: bool,
589        normal_loading_metadata: NormalLoadingMetadata,
590        attention_mechanism: AttentionImplementation,
591    ) -> Result<Self> {
592        if let Some(ref quant_cfg) = &cfg.quantization_config {
593            tracing::info!(
594                "Using {} quantization: {}.",
595                quant_cfg.name(),
596                quant_cfg.get_bits_name(&vb_m)
597            );
598        }
599        let mapper = normal_loading_metadata.mapper;
600
601        let wte = embedding(
602            cfg.vocab_size,
603            cfg.hidden_size,
604            mapper.set_nm_device(vb_m.pp("embed_tokens"), false),
605            &cfg.quantization_config,
606        )?;
607        let lm_head = if !cfg.tie_word_embeddings {
608            ReplicatedLayer::new(
609                cfg.hidden_size,
610                cfg.vocab_size,
611                &cfg.quantization_config,
612                false,
613                mapper.set_nm_device(vb_lm_head, normal_loading_metadata.loading_isq),
614            )?
615        } else {
616            ReplicatedLayer::from_linear(candle_nn::Linear::new(
617                mapper.cast_nm_device(wte.embeddings(), normal_loading_metadata.loading_isq)?,
618                None,
619            ))?
620        };
621        let ln_f = RmsNorm::new(
622            cfg.hidden_size,
623            cfg.rms_norm_eps,
624            mapper.set_nm_device(vb_m.pp("norm"), false),
625        )?;
626        let head_dim = cfg.hidden_size / cfg.num_attention_heads;
627        let mut ropes = HashMap::new();
628        for i in 0..cfg.num_hidden_layers {
629            let device = mapper
630                .device_for(i, false)
631                .unwrap_or(&normal_loading_metadata.real_device);
632            ropes.insert(
633                device.location(),
634                Arc::new(Llama3RotaryEmbedding::new_llama4(
635                    vb_m.dtype(),
636                    cfg,
637                    device,
638                    is_gptx,
639                )?),
640            );
641        }
642        let blocks = NiceProgressBar::<_, 'b'>(
643            0..cfg.num_hidden_layers,
644            "Loading text repeating layers",
645            &normal_loading_metadata.multi_progress,
646        )
647        .par_iter_if_isq(|i| {
648            let device = mapper
649                .device_for(i, false)
650                .unwrap_or(&normal_loading_metadata.real_device);
651            let rotary_emb = ropes
652                .get(&device.location())
653                .expect("No RoPE for device location!")
654                .clone();
655            let paged_attn = match &attention_mechanism {
656                AttentionImplementation::Eager => None,
657                AttentionImplementation::PagedAttention => {
658                    Some(PagedAttention::new(head_dim, device, None)?)
659                }
660            };
661            let comm = mapper.get_comm_for(i)?;
662            Block::new(
663                vb_m.pp(format!("layers.{i}")),
664                cfg,
665                &*mapper,
666                i,
667                normal_loading_metadata.loading_isq,
668                rotary_emb,
669                paged_attn,
670                &comm,
671            )
672        })?;
673
674        Ok(Self {
675            wte,
676            blocks,
677            ln_f,
678            lm_head,
679            kv_cache: EitherCache::Normal(NormalCache::new(
680                cfg.num_hidden_layers,
681                cfg.max_position_embeddings,
682            )),
683            device: normal_loading_metadata.real_device,
684            cfg: ModelConfigMetadata {
685                max_seq_len: cfg.max_position_embeddings,
686                num_layers: cfg.num_hidden_layers,
687                hidden_size: cfg.hidden_size,
688                num_kv_heads: (cfg.num_key_value_heads / mapper.get_comm_for(0)?.world_size())
689                    .max(1),
690                num_attn_heads: cfg.num_attention_heads / mapper.get_comm_for(0)?.world_size(),
691                sliding_window: None,
692                k_head_dim: cfg.hidden_size / cfg.num_attention_heads,
693                v_head_dim: cfg.hidden_size / cfg.num_attention_heads,
694            },
695            mapper,
696            attention_chunk_size: cfg.attention_chunk_size,
697        })
698    }
699
700    pub fn get_input_embeddings(&self, input_ids: &Tensor) -> Result<Tensor> {
701        self.wte.forward(input_ids)
702    }
703
704    #[allow(clippy::too_many_arguments)]
705    pub fn forward_embeds(
706        &self,
707        input_ids: &Tensor,
708        input_embeds: Tensor,
709        seqlen_offsets: &[usize],
710        context_lens: Vec<(usize, usize)>,
711        metadata: Option<(Vec<(Tensor, Tensor)>, &PagedAttentionInputMetadata)>,
712        flash_params: &FlashParams,
713    ) -> Result<Tensor> {
714        let mut x = input_embeds;
715        let cache = &mut self.kv_cache.normal().0;
716        let cache_for_mask = metadata
717            .as_ref()
718            .map(|(_, _)| &seqlen_offsets as &dyn PastKvLenCache)
719            .unwrap_or(cache as &dyn PastKvLenCache);
720
721        let position_ids = Tensor::new(
722            seqlen_offsets.iter().map(|o| *o as i32).collect::<Vec<_>>(),
723            input_ids.device(),
724        )?;
725
726        let mask = CausalMasker.make_causal_mask_matrix(
727            input_ids,
728            cache_for_mask,
729            x.dtype(),
730            self.blocks[0].attn.num_attention_heads,
731        )?;
732        let chunked_mask = CausalMasker.make_chunked_mask_matrix(
733            input_ids,
734            self.attention_chunk_size,
735            cache_for_mask,
736            x.dtype(),
737            self.blocks[0].attn.num_attention_heads,
738        )?;
739        // PagedAttention prompt chunking
740        let mask = mask.filter(|_| {
741            metadata
742                .as_ref()
743                .map(|(_, meta)| meta.is_first_prompt_chunk)
744                .unwrap_or(true)
745        });
746        // PagedAttention prompt chunking
747        let chunked_mask = chunked_mask.filter(|_| {
748            metadata
749                .as_ref()
750                .map(|(_, meta)| meta.is_first_prompt_chunk)
751                .unwrap_or(true)
752        });
753        for (block_idx, block) in self.blocks.iter().enumerate() {
754            x = self.mapper.map(x, block_idx)?;
755            x = block.forward(
756                &x,
757                &position_ids.to_device(x.device())?,
758                &mask.clone().map(|m| m.to_device(x.device()).unwrap()),
759                &chunked_mask
760                    .clone()
761                    .map(|m| m.to_device(x.device()).unwrap()),
762                seqlen_offsets,
763                &mut cache[block_idx],
764                metadata
765                    .as_ref()
766                    .map(|(kv_cache, metadata)| (kv_cache[block_idx].clone(), *metadata)),
767                flash_params,
768            )?;
769        }
770        let mut x = x.to_device(&self.device)?;
771        x = self.ln_f.forward(&x)?;
772        x = self.lm_head.forward_autocast(&x)?;
773        extract_logits(&x, context_lens)
774    }
775
776    pub fn residual_tensors_m(&self, uvb_m: UnVarBuilder) -> Vec<(String, Tensor)> {
777        uvb_m.pp("embed_tokens").add(&self.wte);
778        uvb_m.pp("norm").add(&self.ln_f);
779
780        for (layer_idx, layer) in self.blocks.iter().enumerate() {
781            let uvb_l = uvb_m.pp("layers").pp(layer_idx);
782            uvb_l.pp("input_layernorm").add(&layer.rms_1);
783            uvb_l.pp("post_attention_layernorm").add(&layer.rms_2);
784        }
785
786        uvb_m.to_safetensors()
787    }
788}

790impl IsqModel for TextModel {
791    fn get_layers(
792        &mut self,
793    ) -> (
794        Vec<(&mut Arc<dyn QuantMethod>, Option<usize>)>,
795        &dyn DeviceMapper,
796    ) {
797        let mut tensors = Vec::new();
798        tensors.push((&mut self.lm_head, None));
799        for (i, layer) in self.blocks.iter_mut().enumerate() {
800            tensors.push((&mut layer.attn.q_proj, Some(i)));
801            tensors.push((&mut layer.attn.k_proj, Some(i)));
802            tensors.push((&mut layer.attn.v_proj, Some(i)));
803            tensors.push((&mut layer.attn.o_proj, Some(i)));
804            match &mut layer.ff {
805                MoeOrMlp::Mlp(x) => {
806                    tensors.push((&mut x.gate, Some(i)));
807                    tensors.push((&mut x.up, Some(i)));
808                    tensors.push((&mut x.down, Some(i)));
809                }
810                MoeOrMlp::Moe(x) => {
811                    tensors.push((&mut x.router, Some(i)));
812                    for g in &mut x.experts.gate_proj {
813                        tensors.push((g, Some(i)));
814                    }
815                    for u in &mut x.experts.up_proj {
816                        tensors.push((u, Some(i)));
817                    }
818                    for d in &mut x.experts.down_proj {
819                        tensors.push((d, Some(i)));
820                    }
821                    tensors.push((&mut x.shared_expert.gate, Some(i)));
822                    tensors.push((&mut x.shared_expert.up, Some(i)));
823                    tensors.push((&mut x.shared_expert.down, Some(i)));
824                }
825            }
826        }
827        (tensors, &*self.mapper)
828    }
829
830    fn residual_tensors(&self) -> Vec<(String, Tensor)> {
831        let uvb = UnVarBuilder::new();
832        self.residual_tensors_m(uvb.pp("model"))
833    }
