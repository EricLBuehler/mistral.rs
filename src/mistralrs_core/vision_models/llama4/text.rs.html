42    norm: Option<RmsNorm>,

46    attn_temperature_tuning: Option<f32>,

49impl CausalSelfAttention {
50    #[allow(clippy::too_many_arguments)]
51    fn new(
52        vb: ShardedVarBuilder,
53        cfg: &TextConfig,
54        layer_idx: usize,
55        loading_isq: bool,
56        mapper: &dyn DeviceMapper,
57        rope: Arc<Llama3RotaryEmbedding>,
58        paged_attn: Option<PagedAttention>,
59        comm: &Arc<mistralrs_quant::Comm>,
60    ) -> Result<Self> {
61        let size_in = cfg.hidden_size;
62        let size_q = (cfg.hidden_size / cfg.num_attention_heads) * cfg.num_attention_heads;
63        let size_kv = (cfg.hidden_size / cfg.num_attention_heads) * cfg.num_key_value_heads;
64        let q_proj = ColumnParallelLayer::new(
65            size_in,
66            size_q,
67            &cfg.quantization_config,
68            false,
69            comm,
70            mapper.set_device(layer_idx, vb.pp("q_proj"), loading_isq),
71        )?;
72        let kv_shard = mistralrs_quant::compute_kv_shard(
73            cfg.num_key_value_heads,
74            cfg.hidden_size / cfg.num_attention_heads,
75            comm,
76        );
77        let k_proj = ColumnParallelLayer::new_with_shard(
78            size_in,
79            size_kv,
80            &cfg.quantization_config,
81            false,
82            comm,
83            kv_shard,
84            mapper.set_device(layer_idx, vb.pp("k_proj"), loading_isq),
85        )?;
86        let v_proj = ColumnParallelLayer::new_with_shard(
87            size_in,
88            size_kv,
89            &cfg.quantization_config,
90            false,
91            comm,
92            kv_shard,
93            mapper.set_device(layer_idx, vb.pp("v_proj"), loading_isq),
94        )?;
95        let o_proj = RowParallelLayer::new(
96            size_q,
97            size_in,
98            &cfg.quantization_config,
99            false,
100            comm,
101            mapper.set_device(layer_idx, vb.pp("o_proj"), loading_isq),
102        )?;
103        let use_rope = (layer_idx + 1) % 4 != 0;
104        let head_dim = cfg.hidden_size / cfg.num_attention_heads;
105        let norm = if cfg.use_qk_norm && use_rope {
106            let vb = mapper.set_device(layer_idx, vb, false);
107            Some(RmsNorm::from_w(
108                Tensor::ones(head_dim, vb.dtype(), vb.device())?,
109                1e-6,
110            )?)
111        } else {
112            None
113        };
114
115        Ok(Self {
116            q_proj,
117            k_proj,
118            v_proj,
119            o_proj,
120            num_attention_heads: cfg.num_attention_heads / comm.world_size(),
121            num_key_value_heads: (cfg.num_key_value_heads / comm.world_size()).max(1),
122            head_dim,
123            rotary_emb: rope,
124            max_seq_len: cfg.max_position_embeddings,
125            paged_attn,
126            sdpa_params: SdpaParams {
127                n_kv_groups: mistralrs_quant::compute_n_kv_groups(
128                    cfg.num_key_value_heads,
129                    cfg.num_attention_heads,
130                    comm,
131                ),
132                softcap: None,
133                softmax_scale: 1.0 / (head_dim as f32).sqrt(),
134                sliding_window: None,
135            },
136            norm,
137            use_rope,
138            floor_scale: cfg.floor_scale,
139            attn_scale: cfg.attn_scale,
140            attn_temperature_tuning: cfg.attn_temperature_tuning,
141        })
142    }
143
144    #[allow(clippy::too_many_arguments)]
145    fn forward(
146        &self,
147        x: &Tensor,
148        position_ids: &Tensor,
149        attention_mask: &Option<Tensor>,
150        seqlen_offsets: &[usize],
151        kv_cache: &mut KvCache,
152        metadata: Option<((Tensor, Tensor), &PagedAttentionInputMetadata)>,
153        flash_params: &FlashParams,
154    ) -> Result<Tensor> {
155        let (b_sz, seq_len, _) = x.dims3()?;
156
157        let mut q = self.q_proj.forward_autocast(x)?;
158        let mut k = self.k_proj.forward_autocast(x)?;
159        let mut v = self.v_proj.forward_autocast(x)?;
160
161        q = q
162            .reshape((b_sz, seq_len, self.num_attention_heads, self.head_dim))?
163            .transpose(1, 2)?;
164        k = k
165            .reshape((b_sz, seq_len, self.num_key_value_heads, self.head_dim))?
166            .transpose(1, 2)?;
167        v = v
168            .reshape((b_sz, seq_len, self.num_key_value_heads, self.head_dim))?
169            .transpose(1, 2)?;
170
171        if self.use_rope {
172            (q, k) = self.rotary_emb.forward(&q, &k, seqlen_offsets)?;
173        }
174
175        if let Some(qk_norm) = &self.norm {
176            q = qk_norm.forward(&q)?;
177            k = qk_norm.forward(&k)?;
178        }
179
180        if self.attn_temperature_tuning.is_some() && !self.use_rope {
181            let floor_scale = self.floor_scale.unwrap() as f64;
182            let attn_scale = self.attn_scale.unwrap() as f64;
183            let floor = ((position_ids.to_dtype(DType::F32)? + 1.)? / floor_scale)?.floor()?;
184            let attn_scales = (((floor + 1.0)?.log()? * attn_scale)? + 1.0)?;
185
186            q = q
187                .to_dtype(DType::F32)?
188                .broadcast_mul(&attn_scales.unsqueeze(D::Minus1)?)?
189                .to_dtype(q.dtype())?;
190        }
191
192        let mut y = match &self.paged_attn {
193            Some(paged_attn) => match metadata {
194                Some(((key_cache, value_cache), input_metadata)) => paged_attn.forward(
195                    &q,
196                    &k,
197                    &v,
198                    attention_mask.clone().as_ref(),
199                    Some(key_cache),
200                    Some(value_cache),
201                    input_metadata,
202                    &self.sdpa_params,
203                    Some(flash_params),
204                )?,
205                None => {
206                    // If we don't have metadata, we are most likely generating an imatrix so we don't want to populate that.
207                    // Generating the dummy metadata with the assumption that we are not generating text (only processing prompts).
208                    let input_metadata = PagedAttentionInputMetadata::dummy(q.device())?;
209                    // Sanity check.
210                    assert!(attention_mask.is_some());
211                    paged_attn.forward(
212                        &q,
213                        &k,
214                        &v,
215                        attention_mask.clone().as_ref(),
216                        None,
217                        None,
218                        &input_metadata,
219                        &self.sdpa_params,
220                        Some(flash_params),
221                    )?
222                }
223            },
224            None => {
225                let (k, v) = kv_cache.append(&k, &v)?;
226
227                Sdpa.run_attention(
228                    &q.contiguous()?,
229                    &k.contiguous()?,
230                    &v.contiguous()?,
231                    attention_mask.clone().as_ref(),
232                    Some(flash_params),
233                    &self.sdpa_params,
234                )?
235            }
236        };
237
238        y = if attention_mask.is_some() {
239            y.transpose(1, 2)?.reshape((b_sz, seq_len, ()))?
240        } else {
241            y.reshape((b_sz, seq_len, ()))?
242        };
243        self.o_proj.forward_autocast(&y)
244    }
245}

247struct Mlp {
248    gate: Arc<dyn QuantMethod>,
249    up: Arc<dyn QuantMethod>,
250    down: Arc<dyn QuantMethod>,
251    act: Activation,
252}
253
254impl Mlp {
255    fn new(
256        vb: ShardedVarBuilder,
257        hidden_size: usize,
258        intermediate_size: usize,
259        quantization_config: &Option<QuantizedConfig>,
260        hidden_act: Activation,
261        comm: &Arc<mistralrs_quant::Comm>,
262    ) -> Result<Self> {
263        Ok(Self {
264            gate: ColumnParallelLayer::new(
265                hidden_size,
266                intermediate_size,
267                quantization_config,
268                false,
269                comm,
270                vb.pp("gate_proj"),
271            )?,
272            up: ColumnParallelLayer::new(
273                hidden_size,
274                intermediate_size,
275                quantization_config,
276                false,
277                comm,
278                vb.pp("up_proj"),
279            )?,
280            down: RowParallelLayer::new(
281                intermediate_size,
282                hidden_size,
283                quantization_config,
284                false,
285                comm,
286                vb.pp("down_proj"),
287            )?,
288            act: hidden_act,
289        })
290    }
291
292    fn forward(&self, xs: &Tensor) -> Result<Tensor> {
293        let lhs = self.gate.forward_autocast(xs)?;
294        let rhs = self.up.forward_autocast(xs)?;
295
296        self.down
297            .forward_autocast(&crate::ops::mul_and_act(&lhs, &rhs, self.act)?)
298    }
299}

301/// MoE layer for Llama4 using the unified MoEExperts
302struct TextMoe {
303    experts: MoEExperts,
304    shared_expert: Mlp,
305    router: Arc<dyn QuantMethod>,
306    topk: usize,
307}
308
309impl TextMoe {
310    fn new(
311        vb: ShardedVarBuilder,
312        cfg: &TextConfig,
313        quantization_config: &Option<QuantizedConfig>,
314        comm: &Arc<mistralrs_quant::Comm>,
315        loading_isq: bool,
316        layer_device: Device,
317    ) -> Result<Self> {
318        let moe_cfg = MoEExpertsConfig {
319            num_experts: cfg.num_local_experts,
320            num_experts_per_tok: cfg.num_experts_per_tok,
321            hidden_size: cfg.hidden_size,
322            moe_intermediate_size: cfg.intermediate_size,
323        };
324
325        let experts = MoEExperts::new(
326            &moe_cfg,
327            vb.clone(),
328            layer_device.clone(),
329            comm,
330            loading_isq,
331            quantization_config,
332            cfg.hidden_act,
333        )?;
334
335        let router = linear_no_bias(
336            cfg.hidden_size,
337            cfg.num_local_experts,
338            quantization_config,
339            vb.pp("router").set_device(layer_device),
340        )?;
341
342        let shared_expert = Mlp::new(
343            vb.pp("shared_expert"),
344            cfg.hidden_size,
345            cfg.intermediate_size,
346            quantization_config,
347            cfg.hidden_act,
348            comm,
349        )?;
350
351        Ok(Self {
352            experts,
353            shared_expert,
354            router,
355            topk: cfg.num_experts_per_tok,
356        })
357    }
358
359    fn forward(&self, xs: &Tensor) -> Result<Tensor> {
360        let (bs, seq_len, hidden_dim) = xs.dims3()?;
361        let xs_flat = xs.reshape(((), hidden_dim))?;
362        let router_logits = self.router.forward_autocast(&xs_flat)?;
363
364        let TopKOutput {
365            values: router_top_value,
366            indices: router_indices,
367        } = router_logits.topk(self.topk)?;
368
369        let router_scores = candle_nn::ops::sigmoid(&router_top_value.to_dtype(DType::F32)?)?
370            .to_dtype(router_top_value.dtype())?;
371
372        // Forward through routed experts (is_prefill determined internally)
373        let routed_out = self
374            .experts
375            .forward(xs, router_scores, &router_indices)?
376            .reshape((bs, seq_len, hidden_dim))?;
377
378        // Forward through shared expert and add
379        let out = self.shared_expert.forward(xs)?;
380
381        out + routed_out
382    }
383}

399struct Block {
400    rms_1: RmsNorm,
401    attn: CausalSelfAttention,
402    rms_2: RmsNorm,
403    ff: MoeOrMlp,
404    use_chunked_attention: bool,
405}
406
407impl Block {
408    #[allow(clippy::too_many_arguments)]
409    fn new(
410        vb: ShardedVarBuilder,
411        cfg: &TextConfig,
412        mapper: &dyn DeviceMapper,
413        layer_idx: usize,
414        loading_isq: bool,
415        rope: Arc<Llama3RotaryEmbedding>,
416        paged_attn: Option<PagedAttention>,
417        comm: &Arc<mistralrs_quant::Comm>,
418        real_device: Device,
419    ) -> Result<Self> {
420        let use_chunked_attention = (layer_idx + 1) % 4 != 0;
421        let attn = CausalSelfAttention::new(
422            vb.pp("self_attn"),
423            cfg,
424            layer_idx,
425            loading_isq,
426            mapper,
427            rope,
428            paged_attn,
429            comm,
430        )?;
431        let is_moe_layer = cfg.moe_layers().contains(&layer_idx);
432        let layer_device = mapper
433            .device_for(layer_idx, false)
434            .cloned()
435            .unwrap_or(real_device);
436        let ff = if is_moe_layer {
437            let moe = TextMoe::new(
438                mapper.set_device(layer_idx, vb.pp("feed_forward"), loading_isq),
439                cfg,
440                &cfg.quantization_config,
441                comm,
442                loading_isq,
443                layer_device,
444            )?;
445            MoeOrMlp::Moe(moe)
446        } else {
447            let mlp = Mlp::new(
448                mapper.set_device(layer_idx, vb.pp("feed_forward"), loading_isq),
449                cfg.hidden_size,
450                cfg.intermediate_size_mlp,
451                &cfg.quantization_config,
452                cfg.hidden_act,
453                comm,
454            )?;
455            MoeOrMlp::Mlp(mlp)
456        };
457        let rms_1 = RmsNorm::new(
458            cfg.hidden_size,
459            cfg.rms_norm_eps,
460            mapper.set_device(layer_idx, vb.pp("input_layernorm"), false),
461        )?;
462        let rms_2 = RmsNorm::new(
463            cfg.hidden_size,
464            cfg.rms_norm_eps,
465            mapper.set_device(layer_idx, vb.pp("post_attention_layernorm"), false),
466        )?;
467        Ok(Self {
468            rms_1,
469            attn,
470            rms_2,
471            ff,
472            use_chunked_attention,
473        })
474    }
475
476    #[allow(clippy::too_many_arguments)]
477    fn forward(
478        &self,
479        x: &Tensor,
480        position_ids: &Tensor,
481        attention_mask: &Option<Tensor>,
482        chunked_mask: &Option<Tensor>,
483        seqlen_offsets: &[usize],
484        kv_cache: &mut KvCache,
485        metadata: Option<((Tensor, Tensor), &PagedAttentionInputMetadata)>,
486        flash_params: &FlashParams,
487    ) -> Result<Tensor> {
488        let residual = x;
489        let x = self.rms_1.forward(x)?;
490        let mask = if self.use_chunked_attention {
491            chunked_mask
492        } else {
493            attention_mask
494        };
495        let x = (self.attn.forward(
496            &x,
497            position_ids,
498            mask,
499            seqlen_offsets,
500            kv_cache,
501            metadata,
502            flash_params,
503        )? + residual)?;
504        let residual = &x;
505        let x = (self.ff.forward(&self.rms_2.forward(&x)?)? + residual)?;
506        Ok(x)
507    }
508}

542    pub fn new_inner(
543        cfg: &TextConfig,
544        vb_m: ShardedVarBuilder,
545        vb_lm_head: ShardedVarBuilder,
546        is_gptx: bool,
547        normal_loading_metadata: NormalLoadingMetadata,
548        attention_mechanism: AttentionImplementation,
549    ) -> Result<Self> {
550        if let Some(ref quant_cfg) = &cfg.quantization_config {
551            tracing::info!(
552                "Using {} quantization: {}.",
553                quant_cfg.name(),
554                quant_cfg.get_bits_name(&vb_m)
555            );
556        }
557        let mapper = normal_loading_metadata.mapper;
558
559        let wte = embedding(
560            cfg.vocab_size,
561            cfg.hidden_size,
562            mapper.set_nm_device(vb_m.pp("embed_tokens"), false),
563            &cfg.quantization_config,
564        )?;
565        let lm_head = if !cfg.tie_word_embeddings {
566            ReplicatedLayer::new(
567                cfg.hidden_size,
568                cfg.vocab_size,
569                &cfg.quantization_config,
570                false,
571                mapper.set_nm_device(vb_lm_head, normal_loading_metadata.loading_isq),
572            )?
573        } else {
574            ReplicatedLayer::from_linear(candle_nn::Linear::new(
575                mapper.cast_nm_device(wte.embeddings(), normal_loading_metadata.loading_isq)?,
576                None,
577            ))?
578        };
579        let ln_f = RmsNorm::new(
580            cfg.hidden_size,
581            cfg.rms_norm_eps,
582            mapper.set_nm_device(vb_m.pp("norm"), false),
583        )?;
584        let head_dim = cfg.hidden_size / cfg.num_attention_heads;
585        let mut ropes = HashMap::new();
586        for i in 0..cfg.num_hidden_layers {
587            let device = mapper
588                .device_for(i, false)
589                .unwrap_or(&normal_loading_metadata.real_device);
590            ropes.insert(
591                device.location(),
592                Arc::new(Llama3RotaryEmbedding::new_llama4(
593                    vb_m.dtype(),
594                    cfg,
595                    device,
596                    is_gptx,
597                )?),
598            );
599        }
600        let blocks = NiceProgressBar::<_, 'b'>(
601            0..cfg.num_hidden_layers,
602            "Loading text repeating layers",
603            &normal_loading_metadata.multi_progress,
604        )
605        .par_iter_if_isq(|i| {
606            let device = mapper
607                .device_for(i, false)
608                .unwrap_or(&normal_loading_metadata.real_device);
609            let rotary_emb = ropes
610                .get(&device.location())
611                .expect("No RoPE for device location!")
612                .clone();
613            let paged_attn = match &attention_mechanism {
614                AttentionImplementation::Eager => None,
615                AttentionImplementation::PagedAttention => {
616                    Some(PagedAttention::new(head_dim, device, None)?)
617                }
618            };
619            let comm = mapper.get_comm_for(i)?;
620            Block::new(
621                vb_m.pp(format!("layers.{i}")),
622                cfg,
623                &*mapper,
624                i,
625                normal_loading_metadata.loading_isq,
626                rotary_emb,
627                paged_attn,
628                &comm,
629                normal_loading_metadata.real_device.clone(),
630            )
631        })?;
632
633        Ok(Self {
634            wte,
635            blocks,
636            ln_f,
637            lm_head,
638            kv_cache: EitherCache::Normal(NormalCache::new(
639                cfg.num_hidden_layers,
640                cfg.max_position_embeddings,
641            )),
642            device: normal_loading_metadata.real_device,
643            cfg: ModelConfigMetadata {
644                max_seq_len: cfg.max_position_embeddings,
645                num_layers: cfg.num_hidden_layers,
646                hidden_size: cfg.hidden_size,
647                num_kv_heads: (cfg.num_key_value_heads / mapper.get_comm_for(0)?.world_size())
648                    .max(1),
649                num_attn_heads: cfg.num_attention_heads / mapper.get_comm_for(0)?.world_size(),
650                sliding_window: None,
651                k_head_dim: cfg.hidden_size / cfg.num_attention_heads,
652                v_head_dim: cfg.hidden_size / cfg.num_attention_heads,
653            },
654            mapper,
655            attention_chunk_size: cfg.attention_chunk_size,
656        })
657    }

659    pub fn get_input_embeddings(&self, input_ids: &Tensor) -> Result<Tensor> {
660        self.wte.forward(input_ids)
661    }
662
663    #[allow(clippy::too_many_arguments)]
664    pub fn forward_embeds(
665        &self,
666        input_ids: &Tensor,
667        input_embeds: Tensor,
668        seqlen_offsets: &[usize],
669        context_lens: Vec<(usize, usize)>,
670        metadata: Option<(Vec<(Tensor, Tensor)>, &PagedAttentionInputMetadata)>,
671        flash_params: &FlashParams,
672    ) -> Result<Tensor> {
673        let mut x = input_embeds;
674        let cache = &mut self.kv_cache.normal().0;
675        let cache_for_mask = metadata
676            .as_ref()
677            .map(|(_, _)| &seqlen_offsets as &dyn PastKvLenCache)
678            .unwrap_or(cache as &dyn PastKvLenCache);
679
680        let position_ids = Tensor::new(
681            seqlen_offsets.iter().map(|o| *o as i32).collect::<Vec<_>>(),
682            input_ids.device(),
683        )?;
684
685        let mask = CausalMasker.make_causal_mask_matrix(
686            input_ids,
687            cache_for_mask,
688            x.dtype(),
689            self.blocks[0].attn.num_attention_heads,
690        )?;
691        let chunked_mask = CausalMasker.make_chunked_mask_matrix(
692            input_ids,
693            self.attention_chunk_size,
694            cache_for_mask,
695            x.dtype(),
696            self.blocks[0].attn.num_attention_heads,
697        )?;
698        // PagedAttention prompt chunking
699        let mask = mask.filter(|_| {
700            metadata
701                .as_ref()
702                .map(|(_, meta)| meta.is_first_prompt_chunk)
703                .unwrap_or(true)
704        });
705        // PagedAttention prompt chunking
706        let chunked_mask = chunked_mask.filter(|_| {
707            metadata
708                .as_ref()
709                .map(|(_, meta)| meta.is_first_prompt_chunk)
710                .unwrap_or(true)
711        });
712        for (block_idx, block) in self.blocks.iter().enumerate() {
713            x = self.mapper.map(x, block_idx)?;
714            x = block.forward(
715                &x,
716                &position_ids.to_device(x.device())?,
717                &mask.clone().map(|m| m.to_device(x.device()).unwrap()),
718                &chunked_mask
719                    .clone()
720                    .map(|m| m.to_device(x.device()).unwrap()),
721                seqlen_offsets,
722                &mut cache[block_idx],
723                metadata
724                    .as_ref()
725                    .map(|(kv_cache, metadata)| (kv_cache[block_idx].clone(), *metadata)),
726                flash_params,
727            )?;
728        }
729        let mut x = x.to_device(&self.device)?;
730        x = self.ln_f.forward(&x)?;
731        x = self.lm_head.forward_autocast(&x)?;
732        extract_logits(&x, context_lens)
733    }
734
735    pub fn residual_tensors_m(&self, uvb_m: UnVarBuilder) -> Vec<(String, Tensor)> {
736        uvb_m.pp("embed_tokens").add(&self.wte);
737        uvb_m.pp("norm").add(&self.ln_f);
738
739        for (layer_idx, layer) in self.blocks.iter().enumerate() {
740            let uvb_l = uvb_m.pp("layers").pp(layer_idx);
741            uvb_l.pp("input_layernorm").add(&layer.rms_1);
742            uvb_l.pp("post_attention_layernorm").add(&layer.rms_2);
743        }
744
745        uvb_m.to_safetensors()
746    }
747}

749impl IsqModel for TextModel {
750    fn get_layers(
751        &mut self,
752    ) -> (
753        Vec<(&mut Arc<dyn QuantMethod>, Option<usize>)>,
754        &dyn DeviceMapper,
755    ) {
756        let mut tensors = Vec::new();
757        tensors.push((&mut self.lm_head, None));
758        for (i, layer) in self.blocks.iter_mut().enumerate() {
759            tensors.push((&mut layer.attn.q_proj, Some(i)));
760            tensors.push((&mut layer.attn.k_proj, Some(i)));
761            tensors.push((&mut layer.attn.v_proj, Some(i)));
762            tensors.push((&mut layer.attn.o_proj, Some(i)));
763            match &mut layer.ff {
764                MoeOrMlp::Mlp(x) => {
765                    tensors.push((&mut x.gate, Some(i)));
766                    tensors.push((&mut x.up, Some(i)));
767                    tensors.push((&mut x.down, Some(i)));
768                }
769                MoeOrMlp::Moe(x) => {
770                    tensors.push((&mut x.router, Some(i)));
771                    for layer in x.experts.get_isq_layers() {
772                        tensors.push((layer, Some(i)));
773                    }
774                    tensors.push((&mut x.shared_expert.gate, Some(i)));
775                    tensors.push((&mut x.shared_expert.up, Some(i)));
776                    tensors.push((&mut x.shared_expert.down, Some(i)));
777                }
778            }
779        }
780        (tensors, &*self.mapper)
781    }
782
783    fn residual_tensors(&self) -> Vec<(String, Tensor)> {
784        let uvb = UnVarBuilder::new();
785        self.residual_tensors_m(uvb.pp("model"))
786    }
787}
