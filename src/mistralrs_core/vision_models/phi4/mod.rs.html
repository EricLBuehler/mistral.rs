33struct Attention {
34    qkv_proj: Arc<dyn QuantMethod>,
35    o_proj: Arc<dyn QuantMethod>,
36    num_heads: usize,
37    num_kv_heads: usize,
38    head_dim: usize,
39    rotary_emb: Arc<Phi4MMRotaryEmbedding>,
40    paged_attn: Option<PagedAttention>,
41    sdpa_params: SdpaParams,
42}
43
44impl Attention {
45    fn new(
46        rotary_emb: Arc<Phi4MMRotaryEmbedding>,
47        cfg: &Phi4MMConfig,
48        vb: ShardedVarBuilder,
49        paged_attn: Option<PagedAttention>,
50    ) -> Result<Self> {
51        let num_heads = cfg.num_attention_heads;
52        let num_kv_heads = cfg.num_key_value_heads();
53        let head_dim = cfg.head_dim();
54        let op_size = num_heads * head_dim + 2 * num_kv_heads * head_dim;
55
56        // No TP here.
57        let qkv_proj = mistralrs_quant::linear_no_bias_static_lora(
58            cfg.hidden_size,
59            op_size,
60            cfg.loras(),
61            vb.pp("qkv_proj"),
62        )?;
63
64        let o_proj = mistralrs_quant::linear_no_bias_static_lora(
65            num_heads * head_dim,
66            cfg.hidden_size,
67            cfg.loras(),
68            vb.pp("o_proj"),
69        )?;
70
71        Ok(Self {
72            qkv_proj,
73            o_proj,
74            rotary_emb,
75            num_heads,
76            num_kv_heads,
77            head_dim,
78            paged_attn,
79            sdpa_params: SdpaParams {
80                n_kv_groups: num_heads / num_kv_heads,
81                softcap: None,
82                softmax_scale: 1.0 / (head_dim as f32).sqrt(),
83                sliding_window: cfg.sliding_window,
84            },
85        })
86    }
87
88    #[allow(clippy::too_many_arguments)]
89    fn forward(
90        &self,
91        xs: &Tensor,
92        attention_mask: Option<&Tensor>,
93        seqlen_offsets: &[usize],
94        position_ids: &[usize],
95        kv_cache: &mut KvCache,
96        metadata: Option<((Tensor, Tensor), &PagedAttentionInputMetadata)>,
97        flash_params: &FlashParams,
98    ) -> Result<Tensor> {
99        let (b_sz, q_len, _) = xs.dims3()?;
100
101        let original_dtype = xs.dtype();
102        let mut xs = xs.clone();
103        if let Some(t) = self.qkv_proj.quantized_act_type() {
104            xs = xs.to_dtype(t)?;
105        }
106        let mut qkv = MatMul.qmethod_matmul(&xs, &*self.qkv_proj)?;
107        if self.qkv_proj.quantized_act_type().is_some() {
108            qkv = qkv.to_dtype(original_dtype)?;
109        }
110        let query_pos = self.num_heads * self.head_dim;
111        let q = qkv.narrow(D::Minus1, 0, query_pos)?;
112        let k = qkv.narrow(D::Minus1, query_pos, self.num_kv_heads * self.head_dim)?;
113        let v = qkv.narrow(
114            D::Minus1,
115            query_pos + self.num_kv_heads * self.head_dim,
116            self.num_kv_heads * self.head_dim,
117        )?;
118
119        let (q, k, v) = if q_len != 1 {
120            let q = q
121                .reshape((b_sz, q_len, self.num_heads, self.head_dim))?
122                .transpose(1, 2)?;
123            let k = k
124                .reshape((b_sz, q_len, self.num_kv_heads, self.head_dim))?
125                .transpose(1, 2)?;
126            let v = v
127                .reshape((b_sz, q_len, self.num_kv_heads, self.head_dim))?
128                .transpose(1, 2)?;
129            (q, k, v)
130        } else {
131            let q = q.reshape((b_sz, self.num_heads, q_len, self.head_dim))?;
132            let k = k.reshape((b_sz, self.num_kv_heads, q_len, self.head_dim))?;
133            let v = v.reshape((b_sz, self.num_kv_heads, q_len, self.head_dim))?;
134            (q, k, v)
135        };
136
137        let (q, k) = self
138            .rotary_emb
139            .forward(&q, &k, seqlen_offsets, position_ids)?;
140
141        let mut attn_output = match &self.paged_attn {
142            Some(paged_attn) => match metadata {
143                Some(((key_cache, value_cache), input_metadata)) => paged_attn.forward(
144                    &q,
145                    &k.contiguous()?,
146                    &v.contiguous()?,
147                    attention_mask,
148                    Some(key_cache),
149                    Some(value_cache),
150                    input_metadata,
151                    &self.sdpa_params,
152                    Some(flash_params),
153                )?,
154                None => {
155                    // If we don't have metadata, we are most likely generating an imatrix so we don't want to populate that.
156                    // Generating the dummy metadata with the assumption that we are not generating text (only processing prompts).
157                    let input_metadata = PagedAttentionInputMetadata::dummy(q.device())?;
158                    // Sanity check.
159                    assert!(attention_mask.is_some());
160                    paged_attn.forward(
161                        &q,
162                        &k.contiguous()?,
163                        &v.contiguous()?,
164                        attention_mask,
165                        None,
166                        None,
167                        &input_metadata,
168                        &self.sdpa_params,
169                        Some(flash_params),
170                    )?
171                }
172            },
173            None => {
174                let (k, v) = kv_cache.append(&k, &v)?;
175
176                Sdpa.run_attention(
177                    &q,
178                    &k,
179                    &v,
180                    attention_mask,
181                    Some(flash_params),
182                    &self.sdpa_params,
183                )?
184            }
185        };
186
187        if let Some(t) = self.qkv_proj.quantized_act_type() {
188            attn_output = attn_output.to_dtype(t)?;
189        }
190        attn_output = if attention_mask.is_some() {
191            attn_output.transpose(1, 2)?.reshape((b_sz, q_len, ()))?
192        } else {
193            attn_output.reshape((b_sz, q_len, ()))?
194        };
195        let mut res = MatMul.qmethod_matmul(&attn_output, &*self.o_proj)?;
196        if self.qkv_proj.quantized_act_type().is_some() {
197            res = res.to_dtype(original_dtype)?;
198        }
199        Ok(res)
200    }
201}

203#[derive(Clone)]
204struct Mlp {
205    gate_up_proj: Arc<dyn QuantMethod>,
206    down_proj: Arc<dyn QuantMethod>,
207    act_fn: Activation,
208    i_size: usize,
209}
210
211impl Mlp {
212    fn new(cfg: &Phi4MMConfig, vb: ShardedVarBuilder) -> Result<Self> {
213        let hidden_size = cfg.hidden_size;
214        let i_size = cfg.intermediate_size;
215
216        // No TP here.
217        let gate_up_proj = mistralrs_quant::linear_no_bias_static_lora(
218            hidden_size,
219            2 * i_size,
220            cfg.loras(),
221            vb.pp("gate_up_proj"),
222        )?;
223
224        let down_proj = mistralrs_quant::linear_no_bias_static_lora(
225            i_size,
226            hidden_size,
227            cfg.loras(),
228            vb.pp("down_proj"),
229        )?;
230
231        Ok(Self {
232            gate_up_proj,
233            down_proj,
234            act_fn: cfg.hidden_act,
235            i_size,
236        })
237    }
238
239    fn forward(&self, xs: &Tensor) -> Result<Tensor> {
240        let original_dtype = xs.dtype();
241        let mut xs = xs.clone();
242        if let Some(t) = self.gate_up_proj.quantized_act_type() {
243            xs = xs.to_dtype(t)?;
244        }
245        let up_states = MatMul.qmethod_matmul(&xs, &*self.gate_up_proj)?;
246        let gate = up_states.narrow(D::Minus1, 0, self.i_size)?;
247        let up_states = up_states.narrow(D::Minus1, self.i_size, self.i_size)?;
248        let up_states = (up_states * gate.apply(&self.act_fn))?;
249        let mut res = MatMul.qmethod_matmul(&up_states, &*self.down_proj)?;
250        if self.gate_up_proj.quantized_act_type().is_some() {
251            res = res.to_dtype(original_dtype)?;
252        }
253        Ok(res)
254    }
255}

257struct DecoderLayer {
258    input_layernorm: RmsNorm,
259    post_attention_layernorm: RmsNorm,
260    mlp: Mlp,
261    self_attn: Attention,
262}
263
264impl DecoderLayer {
265    fn new(
266        rotary_emb: Arc<Phi4MMRotaryEmbedding>,
267        cfg: &Phi4MMConfig,
268        vb: ShardedVarBuilder,
269        mapper: &dyn DeviceMapper,
270        layer_idx: usize,
271        loading_isq: bool,
272        paged_attn: Option<PagedAttention>,
273    ) -> Result<Self> {
274        let self_attn = Attention::new(
275            rotary_emb,
276            cfg,
277            mapper.set_device(layer_idx, vb.pp("self_attn"), loading_isq),
278            paged_attn,
279        )?;
280        let mlp = Mlp::new(cfg, mapper.set_device(layer_idx, vb.pp("mlp"), loading_isq))?;
281        let input_layernorm = RmsNorm::new(
282            cfg.hidden_size,
283            cfg.rms_norm_eps,
284            mapper.set_device(layer_idx, vb.pp("input_layernorm"), false),
285        )?;
286        let post_attention_layernorm = RmsNorm::new(
287            cfg.hidden_size,
288            cfg.rms_norm_eps,
289            mapper.set_device(layer_idx, vb.pp("post_attention_layernorm"), false),
290        )?;
291
292        Ok(Self {
293            input_layernorm,
294            post_attention_layernorm,
295            mlp,
296            self_attn,
297        })
298    }
299
300    #[allow(clippy::too_many_arguments)]
301    fn forward(
302        &self,
303        xs: &Tensor,
304        attention_mask: Option<&Tensor>,
305        seqlen_offsets: &[usize],
306        position_ids: &[usize],
307        kv_cache: &mut KvCache,
308        metadata: Option<((Tensor, Tensor), &PagedAttentionInputMetadata)>,
309        flash_params: &FlashParams,
310    ) -> Result<Tensor> {
311        let residual = xs;
312        let xs = self.input_layernorm.forward(xs)?;
313        let xs = self.self_attn.forward(
314            &xs,
315            attention_mask,
316            seqlen_offsets,
317            position_ids,
318            kv_cache,
319            metadata,
320            flash_params,
321        )?;
322        let xs = (xs + residual)?;
323        let residual = &xs;
324        let xs = self
325            .mlp
326            .forward(&xs.apply(&self.post_attention_layernorm)?)?;
327        residual + xs
328    }
329}

331pub struct Phi4MMModel {
332    embed_tokens: candle_nn::Embedding,
333    embed_tokens_extend: Phi4MMImageAudioEmbedding,
334    layers: Vec<DecoderLayer>,
335    norm: RmsNorm,
336    lm_head: Arc<dyn QuantMethod>,
337    device: Device,
338    cache: EitherCache,
339    max_seq_len: usize,
340    mapper: Box<dyn DeviceMapper + Send + Sync>,
341    sliding_window: Option<usize>,
342    cfg: ModelConfigMetadata,
343}
344
345impl Phi4MMModel {
346    pub fn new(
347        cfg: &Phi4MMConfig,
348        vb: ShardedVarBuilder,
349        _is_gptx: bool,
350        normal_loading_metadata: NormalLoadingMetadata,
351        attention_mechanism: AttentionImplementation,
352    ) -> Result<Self> {
353        let mapper = normal_loading_metadata.mapper;
354        let vb_m = vb.pp("model");
355
356        let embed_tokens = layers::embedding(
357            cfg.vocab_size,
358            cfg.hidden_size,
359            mapper.set_nm_device(vb_m.pp("embed_tokens"), false),
360            &cfg.quantization_config,
361        )?;
362
363        let vb_l = vb_m.pp("layers");
364        let mut ropes = HashMap::new();
365        for layer_idx in 0..cfg.num_hidden_layers {
366            let device = mapper
367                .device_for(layer_idx, false)
368                .unwrap_or(&normal_loading_metadata.real_device);
369            ropes.insert(
370                device.location(),
371                Arc::new(Phi4MMRotaryEmbedding::new(vb.dtype(), cfg, device)?),
372            );
373        }
374        let layers = NiceProgressBar::<_, 'b'>(
375            0..cfg.num_hidden_layers,
376            "Loading repeating layers",
377            &normal_loading_metadata.multi_progress,
378        )
379        .par_iter_if_isq(|layer_idx| {
380            let device = mapper
381                .device_for(layer_idx, false)
382                .unwrap_or(&normal_loading_metadata.real_device);
383            let rotary_emb = ropes
384                .get(&device.location())
385                .expect("No RoPE for device location!")
386                .clone();
387            let paged_attn = match &attention_mechanism {
388                AttentionImplementation::Eager => None,
389                AttentionImplementation::PagedAttention => {
390                    Some(PagedAttention::new(cfg.head_dim(), device, None)?)
391                }
392            };
393            DecoderLayer::new(
394                rotary_emb,
395                cfg,
396                vb_l.pp(layer_idx),
397                &*mapper,
398                layer_idx,
399                normal_loading_metadata.loading_isq,
400                paged_attn,
401            )
402        })?;
403        let norm = RmsNorm::new(
404            cfg.hidden_size,
405            cfg.rms_norm_eps,
406            mapper.set_nm_device(vb_m.pp("norm"), false),
407        )?;
408        let lm_head = if !cfg.tie_word_embeddings {
409            ReplicatedLayer::new(
410                cfg.hidden_size,
411                cfg.vocab_size,
412                &None,
413                false,
414                mapper.set_nm_device(vb.pp("lm_head"), normal_loading_metadata.loading_isq),
415            )?
416        } else {
417            ReplicatedLayer::from_linear(candle_nn::Linear::new(
418                mapper.cast_nm_device(
419                    embed_tokens.embeddings(),
420                    normal_loading_metadata.loading_isq,
421                )?,
422                None,
423            ))?
424        };
425
426        let embed_tokens_extend = Phi4MMImageAudioEmbedding::new(
427            cfg,
428            embed_tokens.clone(),
429            mapper.set_nm_device(vb_m.pp("embed_tokens_extend"), false),
430        )?;
431
432        Ok(Self {
433            layers,
434            norm,
435            lm_head,
436            device: normal_loading_metadata.real_device,
437            cache: EitherCache::Normal(NormalCache::new_sliding(
438                cfg.num_hidden_layers,
439                cfg.max_position_embeddings,
440                cfg.sliding_window,
441            )),
442            max_seq_len: cfg.max_position_embeddings,
443            sliding_window: cfg.sliding_window,
444            embed_tokens,
445            cfg: ModelConfigMetadata {
446                max_seq_len: cfg.max_position_embeddings,
447                num_layers: cfg.num_hidden_layers,
448                hidden_size: cfg.hidden_size,
449                num_attn_heads: cfg.num_attention_heads / mapper.get_comm_for(0)?.world_size(),
450                num_kv_heads: (cfg.num_key_value_heads() / mapper.get_comm_for(0)?.world_size())
451                    .max(1),
452                sliding_window: cfg.sliding_window,
453                k_head_dim: cfg.head_dim(),
454                v_head_dim: cfg.head_dim(),
455            },
456            mapper,
457            embed_tokens_extend,
458        })
459    }
460
461    #[allow(clippy::too_many_arguments)]
462    pub fn forward(
463        &self,
464        input_ids: &Tensor,
465        input_image_embeds: Option<Tensor>,
466        image_attention_mask: Option<Tensor>,
467        seqlen_offsets: &[usize],
468        position_ids: &[usize],
469        context_lens: Vec<(usize, usize)>,
470        image_sizes: Option<Vec<(u32, u32)>>,
471        metadata: Option<(Vec<(Tensor, Tensor)>, &PagedAttentionInputMetadata)>,
472        flash_params: &FlashParams,
473    ) -> Result<Tensor> {
474        let mut xs = if let Some(input_image_embeds) = &input_image_embeds {
475            self.embed_tokens_extend.forward(
476                input_ids,
477                input_image_embeds,
478                image_attention_mask.as_ref(),
479                image_sizes,
480            )?
481        } else {
482            self.embed_tokens.forward(input_ids)?
483        };
484        let cache = &mut self.cache.normal().0;
485        let attention_mask = CausalMasker.make_sliding_window_causal_mask_matrix(
486            input_ids,
487            metadata
488                .as_ref()
489                .map(|(_, _)| &seqlen_offsets as &dyn PastKvLenCache)
490                .unwrap_or(&*cache as &dyn PastKvLenCache),
491            self.sliding_window,
492            xs.dtype(),
493            self.cfg.num_attn_heads,
494        )?;
495        let attention_mask = attention_mask.filter(|_| {
496            metadata
497                .as_ref()
498                .map(|(_, meta)| meta.is_first_prompt_chunk)
499                .unwrap_or(true)
500        });
501
502        for (i, layer) in self.layers.iter().enumerate() {
503            xs = self.mapper.map(xs, i)?;
504            xs = layer.forward(
505                &xs,
506                attention_mask
507                    .as_ref()
508                    .map(|m| m.to_device(xs.device()).unwrap())
509                    .as_ref(),
510                seqlen_offsets,
511                position_ids,
512                &mut cache[i],
513                metadata
514                    .as_ref()
515                    .map(|(kv_cache, metadata)| (kv_cache[i].clone(), *metadata)),
516                flash_params,
517            )?
518        }
519        let xs = xs.to_device(&self.device)?;
520        let mut xs = xs.apply(&self.norm)?;
521        if let Some(t) = self.lm_head.quantized_act_type() {
522            xs = xs.to_dtype(t)?;
523        }
524        extract_logits(&MatMul.qmethod_matmul(&xs, &*self.lm_head)?, context_lens)
525    }
526}

586impl IsqModel for Phi4MMModel {
587    fn get_layers(
588        &mut self,
589    ) -> (
590        Vec<(&mut Arc<dyn QuantMethod>, Option<usize>)>,
591        &dyn DeviceMapper,
592    ) {
593        let mut tensors = Vec::new();
594        tensors.push((&mut self.lm_head, None));
595        for (i, layer) in self.layers.iter_mut().enumerate() {
596            tensors.push((&mut layer.self_attn.qkv_proj, Some(i)));
597            tensors.push((&mut layer.self_attn.o_proj, Some(i)));
598            tensors.push((&mut layer.mlp.gate_up_proj, Some(i)));
599            tensors.push((&mut layer.mlp.down_proj, Some(i)));
600        }
601        (tensors, &*self.mapper)
602    }
603
604    fn residual_tensors(&self) -> Vec<(String, Tensor)> {
605        let uvb = UnVarBuilder::new();
606
607        let uvb_m = uvb.pp("model");
608        uvb_m.pp("embed_tokens").add(&self.embed_tokens);
609        uvb_m.pp("norm").add(&self.norm);
610        uvb_m
611            .pp("embed_tokens_extend")
612            .extend(self.embed_tokens_extend.residual_tensors());
613
614        for (layer_idx, layer) in self.layers.iter().enumerate() {
615            let uvb_l = uvb_m.pp("layers").pp(layer_idx);
616            uvb_l.pp("input_layernorm").add(&layer.input_layernorm);
617            uvb_l
618                .pp("post_attention_layernorm")
619                .add(&layer.post_attention_layernorm);
620        }
621
622        uvb.to_safetensors()
623    }
