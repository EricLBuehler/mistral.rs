11const MAX_INPUT_ID: f64 = 1e9;

13#[derive(Eq, Hash, PartialEq, Debug, Clone, Copy)]
14pub enum InputMode {
15    /// If only speech
16    Speech,
17    /// If vision + speech or only vision (not sure why that is necessary though)
18    Vision,

21pub struct Phi4MMImageAudioEmbedding {
22    audio_embed: Option<AudioEmbedding>,
23    image_embed: Option<ImageEmbedding>,
24    image_input_id: f64,
25    wte: candle_nn::Embedding,
26}
27
28impl Phi4MMImageAudioEmbedding {
29    pub fn new(
30        cfg: &Phi4MMConfig,
31        wte: candle_nn::Embedding,
32        vb: ShardedVarBuilder,
33    ) -> Result<Self> {
34        let image_embed = if let Some(img_embd_config) = &cfg.embd_layer.image_embd_layer {
35            Some(ImageEmbedding::new(
36                cfg,
37                img_embd_config,
38                wte.clone(),
39                vb.pp("image_embed"),
40            )?)
41        } else {
42            None
43        };
44        let audio_embed = if let Some(audio_embd_config) = &cfg.embd_layer.audio_embd_layer {
45            Some(AudioEmbedding::new(
46                cfg,
47                wte.clone(),
48                audio_embd_config,
49                vb.pp("audio_embed"),
50            )?)
51        } else {
52            None
53        };
54
55        Ok(Self {
56            image_embed,
57            audio_embed,
58            image_input_id: cfg.image_input_id.unwrap_or(-1.),
59            wte,
60        })
61    }
62
63    #[allow(clippy::too_many_arguments)]
64    pub fn forward(
65        &self,
66        input_ids: &Tensor,
67        input_image_embeds: Option<&Tensor>,
68        image_attention_mask: Option<&Tensor>,
69        image_sizes: Option<Vec<(u32, u32)>>,
70        input_audio_embeds: Option<&Tensor>,
71        audio_embed_sizes: Option<Vec<usize>>,
72        audio_attention_mask: Option<&Tensor>,
73        input_mode: InputMode,
74    ) -> Result<Tensor> {
75        assert!(-MAX_INPUT_ID < self.image_input_id);
76
77        let input_ids = input_ids.reshape(((), input_ids.dim(D::Minus1)?))?;
78
79        let image_hidden_states = match &self.image_embed {
80            Some(image_embed) if input_image_embeds.is_some() => Some(image_embed.forward(
81                &input_ids,
82                input_image_embeds.expect("input_image_embeds"),
83                image_attention_mask,
84                image_sizes,
85            )?),
86            _ => None,
87        };
88
89        let audio_hidden_states = match &self.audio_embed {
90            Some(audio_embed) if input_audio_embeds.is_some() => Some(audio_embed.forward(
91                &input_ids,
92                input_audio_embeds.expect("input_audio_embeds"),
93                audio_embed_sizes.expect("audio_embed_sizes"),
94                audio_attention_mask,
95                &input_mode,
96            )?),
97            _ => None,
98        };
99
100        let image_position_mask = input_ids.eq(IMAGE_SPECIAL_TOKEN_ID)?;
101        let non_image_position_mask = input_ids.ne(IMAGE_SPECIAL_TOKEN_ID)?;
102
103        match (image_hidden_states, audio_hidden_states) {
104            (Some(image_hidden_states), Some(audio_hidden_states)) => {
105                // Merge
106                image_hidden_states.broadcast_mul(
107                    &image_position_mask
108                        .to_dtype(image_hidden_states.dtype())?
109                        .unsqueeze(D::Minus1)?,
110                )? + audio_hidden_states.broadcast_mul(
111                    &non_image_position_mask
112                        .to_dtype(audio_hidden_states.dtype())?
113                        .unsqueeze(D::Minus1)?,
114                )?
115            }
116            (Some(image_hidden_states), None) => Ok(image_hidden_states),
117            (None, Some(audio_hidden_states)) => Ok(audio_hidden_states),
118
119            (None, None) => self.wte.forward(&input_ids),
120        }
121    }
122
123    pub fn residual_tensors(&self) -> Vec<(String, Tensor)> {
124        let uvb = UnVarBuilder::new();
125
126        if let Some(image_embed) = &self.image_embed {
127            uvb.pp("image_embed").extend(image_embed.residual_tensors());
128        }
129
130        uvb.to_safetensors()
131    }
132}
