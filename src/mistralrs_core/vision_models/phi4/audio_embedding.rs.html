22pub struct AudioEmbedding {
23    wte: candle_nn::Embedding,

29impl AudioEmbedding {
30    pub fn new(
31        cfg: &Phi4MMConfig,
32        wte: candle_nn::Embedding,
33        audio_embd_config: &Phi4MMAudioEmbedConfig,
34        vb: ShardedVarBuilder,
35    ) -> Result<Self> {
36        let hidden_size = audio_embd_config.n_embd.unwrap_or(cfg.hidden_size);
37
38        let conformer_config = match &cfg.audio_processor {
39            Some(Phi4MMAudioConfig { config, name }) if name == "cascades" => config,
40            _ => candle_core::bail!("Must have audio processor (`cascades`)"),
41        };
42        let encoder = ConformerEncoder::new(conformer_config.clone(), vb.pp("encoder"))?;
43
44        // let audio_dim_in = conformer_config.input_size;
45        let audio_dim_out = conformer_config.attention_dim;
46
47        let mut proj = HashMap::new();
48        {
49            assert_eq!(audio_embd_config.projection_cls, "mlp");
50
51            let dim_projection = hidden_size;
52            let depth = 2;
53            let linear_downsample_rate = audio_embd_config.downsample_rate;
54
55            let embedding_cls_vb = vb.pp("audio_projection");
56
57            let mut layers_for_speech: Vec<Arc<dyn Module + Send + Sync>> =
58                vec![Arc::new(layers::linear(
59                    audio_dim_out * linear_downsample_rate,
60                    dim_projection,
61                    embedding_cls_vb.pp("speech").pp(0),
62                )?)];
63            for i in 1..depth {
64                layers_for_speech.push(Arc::new(Activation::Gelu));
65                layers_for_speech.push(Arc::new(layers::linear(
66                    dim_projection,
67                    dim_projection,
68                    embedding_cls_vb.pp("speech").pp(i + 1),
69                )?));
70            }
71
72            let mut layers_for_vision: Vec<Arc<dyn Module + Send + Sync>> =
73                vec![Arc::new(layers::linear(
74                    audio_dim_out * linear_downsample_rate,
75                    dim_projection,
76                    embedding_cls_vb.pp("vision").pp(0),
77                )?)];
78            for i in 1..depth {
79                layers_for_vision.push(Arc::new(Activation::Gelu));
80                layers_for_vision.push(Arc::new(layers::linear(
81                    dim_projection,
82                    dim_projection,
83                    embedding_cls_vb.pp("vision").pp(i + 1),
84                )?));
85            }
86
87            proj.insert(InputMode::Speech, layers_for_speech);
88            proj.insert(InputMode::Vision, layers_for_vision);
89        }
90
91        Ok(Self {
92            wte,
93            proj,
94            encoder,
95            target_device_dtype: (vb.device().clone(), vb.dtype()),
96        })
97    }
98
99    fn get_audio_features(
100        &self,
101        input_embeds: &Tensor,
102        audio_attention_mask: Option<&Tensor>,
103        input_mode: &InputMode,
104    ) -> Result<Tensor> {
105        // Get audio features from encoder
106        let (audio_features, _masks) = self.encoder.forward(input_embeds, audio_attention_mask)?;
107
108        // Apply projection based on mode
109        let projection_layers = self.proj.get(input_mode).ok_or_else(|| {
110            candle_core::Error::Msg(format!("Projection mode {input_mode:?} not found"))
111        })?;
112
113        let mut audio_set_tensor = audio_features;
114        for layer in projection_layers {
115            audio_set_tensor = layer.forward(&audio_set_tensor)?;
116        }
117
118        Ok(audio_set_tensor)
119    }
120
121    pub fn forward(
122        &self,
123        input_ids: &Tensor,
124        input_embeds: &Tensor,
125        audio_embed_sizes: Vec<usize>,
126        audio_attention_mask: Option<&Tensor>,
127        input_mode: &InputMode,
128    ) -> Result<Tensor> {
129        // Reshape input_ids to 2D
130        let input_shape = input_ids.shape();
131        let input_ids = if input_shape.rank() > 2 {
132            input_ids.reshape((
133                input_shape.elem_count() / input_shape.dims()[input_shape.rank() - 1],
134                input_shape.dims()[input_shape.rank() - 1],
135            ))?
136        } else {
137            input_ids.clone()
138        };
139
140        let positions = input_ids.eq(AUDIO_SPECIAL_TOKEN_ID)?.nonzero()?;
141
142        // Get target device and dtype from projection layers
143        let (target_device, target_dtype) = self.target_device_dtype.clone();
144
145        let audio_set_tensor = if positions.dim(0)? > 0 {
146            // Convert to target device/dtype if needed
147            let input_embeds = if input_embeds.device().same_device(&target_device)
148                || input_embeds.dtype() != target_dtype
149            {
150                input_embeds
151                    .to_device(&target_device)?
152                    .to_dtype(target_dtype)?
153            } else {
154                input_embeds.clone()
155            };
156
157            self.get_audio_features(&input_embeds, audio_attention_mask, input_mode)?
158        } else {
159            // Return early if no audio tokens and not training
160            return self.wte.forward(&input_ids);
161        };
162
163        // Get initial hidden states from word embeddings
164        let mut hidden_states = self.wte.forward(&input_ids)?;
165
166        // Verify that audio_embed_sizes sum matches positions count
167        let total_audio_tokens = audio_embed_sizes.iter().sum::<usize>();
168        if total_audio_tokens != positions.dim(0)? {
169            return Err(candle_core::Error::Msg(format!(
170                "Audio embed sizes sum ({}) doesn't match positions count ({})",
171                total_audio_tokens,
172                positions.dim(0)?
173            )));
174        }
175
176        let mut audio_sets = Vec::new();
177        for (i, size) in audio_embed_sizes.into_iter().enumerate() {
178            audio_sets.push(audio_set_tensor.i((i, ..size, ..))?);
179        }
180        let merged_audio_set_tensor = Tensor::cat(&audio_sets, 0)?;
181
182        let original_shape = hidden_states.shape().clone();
183        let (hs_b, hs_l, hs_d) = hidden_states.dims3()?;
184        let mut hidden_states_flat = hidden_states.reshape(((), hs_d))?;
185
186        // Get the equiv 0th and 1th rows of the positions_tuple
187        let positions_transposed = positions.to_dtype(DType::F32)?;
188        let positions_transposed_0 = positions_transposed.i((.., 0))?;
189        let positions_transposed_1 = positions_transposed.i((.., 1))?;
190
191        let mut linear_index =
192            ((positions_transposed_0 * (hs_l * hs_b) as f64)? + positions_transposed_1)?;
193        linear_index = linear_index.to_dtype(DType::U32)?;
194        linear_index = linear_index.unsqueeze(1)?.repeat((1, hs_d))?;
195
196        let current_vals = hidden_states_flat.gather(&linear_index, 0)?;
197        let delta = merged_audio_set_tensor.broadcast_sub(&current_vals)?;
198
199        hidden_states_flat = hidden_states_flat.scatter_add(&linear_index, &delta, 0)?;
200
201        hidden_states = hidden_states_flat.reshape(original_shape)?;
202
203        Ok(hidden_states)
204    }
