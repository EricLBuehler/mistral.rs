36impl Idefics3Model {
37    pub fn new(
38        cfg: &Idefics3Config,
39        vb: ShardedVarBuilder,
40        is_gptx: bool,
41        normal_loading_metadata: NormalLoadingMetadata,
42        attention_mechanism: AttentionImplementation,
43    ) -> Result<Self> {
44        let vb_m = vb.pp("model");
45        let connector = Idefics3Connector::new(
46            cfg,
47            vb_m.pp("connector")
48                .set_device(normal_loading_metadata.real_device.clone()),
49        )?;
50        let vision = Idefics3VisionTransformer::new(
51            &cfg.vision_config,
52            vb_m.pp("vision_model")
53                .set_device(normal_loading_metadata.real_device.clone()),
54        )?;
55        let text_model = Llama::new_inner(
56            &cfg.text_config,
57            vb_m.pp("text_model"),
58            vb.pp("lm_head"),
59            is_gptx,
60            normal_loading_metadata,
61            attention_mechanism,
62        )?;
63        Ok(Self {
64            text_model,
65            connector,
66            vision,
67            config: cfg.clone(),
68            dtype: vb.dtype(),
69        })
70    }
71
72    fn inputs_merger(
73        &self,
74        indices: &Tensor,
75        input_embeds: &Tensor,
76        image_hidden_states: &Tensor,
77    ) -> Result<Tensor> {
78        let mut x_flat = input_embeds.flatten_all()?;
79        let src_flat = image_hidden_states.flatten_all()?;
80
81        let current_vals = x_flat.gather(indices, 0)?;
82        let diff = (src_flat - current_vals)?;
83        x_flat = x_flat.scatter_add(indices, &diff, 0)?;
84
85        x_flat.reshape(input_embeds.shape())
86    }
87
88    #[allow(clippy::too_many_arguments)]
89    fn forward_inner(
90        &self,
91        input_ids: &Tensor,
92        pixel_values: Option<Tensor>,
93        seqlen_offsets: &[usize],
94        context_lens: Vec<(usize, usize)>,
95        pixel_attention_mask: Option<Tensor>,
96        metadata: Option<(Vec<(Tensor, Tensor)>, &PagedAttentionInputMetadata)>,
97        flash_params: &FlashParams,
98    ) -> Result<Tensor> {
99        let input_embeds = if let Some(pixel_values) = pixel_values {
100            let input_embeds = self.text_model.get_input_embeddings(input_ids)?;
101            let special_image_mask = input_ids
102                .eq(self.config.image_token_id as f64)?
103                .unsqueeze(D::Minus1)?
104                .broadcast_as(input_embeds.shape())?
105                .to_dtype(DType::U32)?;
106
107            let mask_flat = special_image_mask.flatten_all()?;
108            // Nonzero before vision model to allow async processing all the way through logits.
109            let indices = mask_flat.nonzero()?.squeeze(1)?;
110
111            // == START VISUAL INPUTS INTEGRATION ==
112            let (batch_size, num_images, _, _, _) = pixel_values.dims5()?;
113            let mut s = vec![batch_size * num_images];
114            s.extend(pixel_values.dims()[2..].to_vec());
115            let pixel_values = pixel_values.reshape(s)?;
116
117            // Remove padding images which are full of 0s
118            let nb_values_per_image = pixel_values.dims()[1..].iter().product::<usize>();
119            let real_images_inds = pixel_values
120                .eq(0.0f64)?
121                .sum(vec![
122                    pixel_values.dims().len() - 1,
123                    pixel_values.dims().len() - 2,
124                    pixel_values.dims().len() - 3,
125                ])?
126                .ne(nb_values_per_image as f64)?;
127            let mut batches = Vec::new();
128            for (batch, use_it) in pixel_values
129                .chunk(pixel_values.dim(0)?, 0)?
130                .iter()
131                .zip(real_images_inds.chunk(real_images_inds.dim(0)?, 0)?)
132            {
133                let use_it = use_it.squeeze(0)?.to_scalar::<u8>()? != 0;
134                if use_it {
135                    batches.push(batch.clone());
136                }
137            }
138            let pixel_values = Tensor::cat(&batches, 0)?;
139
140            // Vision attention mask
141            let pixel_attention_mask = if let Some(pixel_attention_mask) = pixel_attention_mask {
142                let pixel_attention_mask = pixel_attention_mask.reshape((
143                    batch_size * num_images,
144                    pixel_attention_mask.dims()[2],
145                    pixel_attention_mask.dims()[3],
146                ))?;
147                let mut batches = Vec::new();
148                for (batch, use_it) in pixel_attention_mask
149                    .chunk(pixel_attention_mask.dim(0)?, 0)?
150                    .iter()
151                    .zip(real_images_inds.chunk(real_images_inds.dim(0)?, 0)?)
152                {
153                    let use_it = use_it.squeeze(0)?.to_scalar::<u8>()? != 0;
154                    if use_it {
155                        batches.push(batch.clone());
156                    }
157                }
158                Tensor::cat(&batches, 0)?
159            } else {
160                Tensor::ones(
161                    (
162                        pixel_values.dims()[0],
163                        pixel_values.dims()[2],
164                        pixel_values.dims()[3],
165                    ),
166                    DType::U8,
167                    pixel_values.device(),
168                )?
169            };
170
171            let patch_size = self.config.vision_config.patch_size;
172            let patches_subgrid = pixel_attention_mask.unfold(1, patch_size, patch_size)?;
173            let patches_subgrid = patches_subgrid.unfold(2, patch_size, patch_size)?;
174
175            let patch_attention_mask = patches_subgrid
176                .sum((D::Minus1, D::Minus2))?
177                .gt(0.0)?
178                .to_dtype(DType::U8)?;
179
180            let pixel_values = pixel_values.to_dtype(self.dtype)?;
181
182            // Get seq from vision encoder
183            let image_hidden_states = self
184                .vision
185                .forward(&pixel_values, Some(&patch_attention_mask))?;
186
187            // Modality proj and perceiver resampling
188            let image_hidden_states = self.connector.forward(&image_hidden_states)?;
189
190            self.inputs_merger(&indices, &input_embeds, &image_hidden_states)?
191                .to_dtype(self.dtype)?
192        } else {
193            self.text_model.get_input_embeddings(input_ids)?
194        };
195
196        self.text_model.forward_embeds(
197            input_ids,
198            input_embeds,
199            seqlen_offsets,
200            context_lens,
201            metadata,
202            flash_params,
203        )
204    }
205}

207impl IsqModel for Idefics3Model {
208    fn get_layers(
209        &mut self,
210    ) -> (
211        Vec<(
212            &mut std::sync::Arc<dyn mistralrs_quant::QuantMethod>,
213            Option<usize>,
214        )>,
215        &dyn DeviceMapper,
216    ) {
217        self.text_model.get_layers()
218    }
219
220    fn residual_tensors(&self) -> Vec<(String, Tensor)> {
221        let uvb = UnVarBuilder::new();
222
223        let uvb_m = uvb.pp("model");
224        uvb_m
225            .pp("connector")
226            .pp("modality_projection")
227            .pp("proj")
228            .add(&self.connector.modality_projection.proj);
229        uvb.extend(self.text_model.residual_tensors_m(uvb_m.pp("text_model")));
230        uvb_m
231            .pp("vision_model")
232            .extend(self.vision.residual_tensors());
233
234        uvb.to_safetensors()
235    }

269impl VisionModel for Idefics3Model {
270    fn forward(
271        &self,
272        input_ids: &Tensor,
273        pixel_values: Option<Tensor>,
274        seqlen_offsets: &[usize],
275        context_lens: Vec<(usize, usize)>,
276        _: Vec<usize>, // Ignore, it is for phi3
277        model_specific_args: Box<dyn Any>,
278        metadata: Option<(Vec<(Tensor, Tensor)>, &PagedAttentionInputMetadata)>,
279        flash_params: &FlashParams,
280    ) -> candle_core::Result<Tensor> {
281        let pixel_attention_mask: Option<Tensor> = *model_specific_args
282            .downcast()
283            .expect("Cannot downcast into `Option<Tensor>`");
284        self.forward_inner(
285            input_ids,
286            pixel_values,
287            seqlen_offsets,
288            context_lens,
289            pixel_attention_mask,
290            metadata,
291            flash_params,
292        )
293    }
294    fn cache(&self) -> &EitherCache {
295        self.text_model.cache()
296    }
297    fn cache_mut(&mut self) -> &mut EitherCache {
298        self.text_model.cache_mut()
299    }
300    fn device(&self) -> &Device {
301        self.text_model.device()
302    }
303    fn max_seq_len(&self) -> usize {
304        self.text_model.max_seq_len()
305    }
306    fn config(&self) -> &ModelConfigMetadata {
307        self.text_model.config()
308    }
309    fn default_model_specific_args(&self, _input_ids: &Tensor) -> Box<dyn Any> {
310        let args: Option<Tensor> = None;
311        Box::new(args)
312    }
