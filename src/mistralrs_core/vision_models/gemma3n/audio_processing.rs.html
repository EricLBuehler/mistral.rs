43            max_frequency: config.max_frequency.unwrap_or(7600.0) as f32,

53    pub fn process_audio(
54        &self,
55        audio_input: &AudioInput,
56        device: &Device,
57    ) -> Result<(Tensor, Tensor)> {
58        // Convert to mono
59        let mono_samples = audio_input.to_mono();
60
61        // Apply input scaling
62        let scaled_samples: Vec<f32> = mono_samples
63            .iter()
64            .map(|&s| s * self.input_scale_factor)
65            .collect();
66
67        // Apply dithering if configured
68        let dithered_samples = if self.dither > 0.0 {
69            use rand_distr::{Distribution, Normal};
70            let mut rng = rand::rng();
71            let normal = Normal::new(0.0, self.dither as f64).unwrap();
72            scaled_samples
73                .iter()
74                .map(|&s| s + normal.sample(&mut rng) as f32)
75                .collect()
76        } else {
77            scaled_samples
78        };
79
80        // Resample if necessary
81        let resampled = if audio_input.sample_rate != self.target_sample_rate {
82            self.resample(
83                &dithered_samples,
84                audio_input.sample_rate,
85                self.target_sample_rate,
86            )?
87        } else {
88            dithered_samples
89        };
90
91        // Use frame parameters from config
92        let frame_length = self.frame_length;
93        let hop_length = self.hop_length;
94
95        // Compute mel-spectrogram
96        let mel_spectrogram = self.compute_mel_spectrogram(
97            &resampled,
98            frame_length,
99            hop_length,
100            self.target_sample_rate,
101        )?;
102
103        // Apply per-bin normalization if configured
104        let normalized_spectrogram = if self.per_bin_mean.is_some() || self.per_bin_stddev.is_some()
105        {
106            self.normalize_mel_spectrogram(mel_spectrogram)?
107        } else {
108            mel_spectrogram
109        };
110
111        // Convert to tensors
112        let num_frames = normalized_spectrogram.len();
113        let mel_data: Vec<f32> = normalized_spectrogram.into_iter().flatten().collect();
114        let mel_tensor = Tensor::from_vec(mel_data, (1, num_frames, self.feature_size), device)?;
115
116        // Create mask (all valid for now)
117        let mask = Tensor::zeros((1, num_frames), candle_core::DType::F32, device)?;
118
119        Ok((mel_tensor, mask))
120    }

122    fn resample(&self, samples: &[f32], from_rate: u32, to_rate: u32) -> Result<Vec<f32>> {
123        if from_rate == to_rate {
124            return Ok(samples.to_vec());
125        }
126
127        let sinc = rubato::SincInterpolationParameters {
128            sinc_len: 256,
129            f_cutoff: 0.95,
130            interpolation: rubato::SincInterpolationType::Linear,
131            oversampling_factor: 256,
132            window: rubato::WindowFunction::BlackmanHarris2,
133        };
134
135        let mut resampler = rubato::SincFixedIn::<f32>::new(
136            to_rate as f64 / from_rate as f64,
137            2.0,
138            sinc,
139            samples.len(),
140            1,
141        )?;
142
143        let samples_vec = vec![samples.to_vec()];
144        let result = resampler.process(&samples_vec, None)?;
145        Ok(result[0].clone())
146    }

148    #[allow(dead_code)]
149    fn apply_preemphasis(&self, samples: &[f32], coeff: f32) -> Vec<f32> {
150        // Retained for potential future use (e.g. non-HTK pre-emphasis) but not
151        // called in the current implementation.
152        if samples.is_empty() {
153            return vec![];
154        }
155
156        let mut out = Vec::with_capacity(samples.len());
157        out.push(samples[0] * (1.0 - coeff));
158        for i in 1..samples.len() {
159            out.push(samples[i] - coeff * samples[i - 1]);
160        }
161        out
162    }

164    fn compute_mel_spectrogram(
165        &self,
166        samples: &[f32],
167        frame_length: usize,
168        hop_length: usize,
169        sample_rate: u32,
170    ) -> Result<Vec<Vec<f32>>> {
171        // FFT size (with overdrive if enabled)
172        let mut n_fft = frame_length.next_power_of_two();
173        if self.fft_overdrive {
174            n_fft *= 2;
175        }
176
177        // Create FFT planner
178        let mut planner = FftPlanner::<f32>::new();
179        let fft = planner.plan_fft_forward(n_fft);
180
181        // === Hann window (same formulation as the reference implementation) ===
182        let window: Vec<f64> = (0..frame_length)
183            .map(|n| {
184                0.5 * (1.0 - (2.0 * std::f64::consts::PI * n as f64 / frame_length as f64).cos())
185            })
186            .collect();
187
188        // Create mel filterbank
189        let mel_filters =
190            self.create_mel_filterbank(self.feature_size, n_fft, sample_rate as f32)?;
191
192        // Process frames
193        // We replicate the exact frame logic used in
194        // `Gemma3nAudioFeatureExtractor` from the Python reference.
195        // frame_length_py == self.frame_length (512) and Python computes
196        // `frame_size_for_unfold = frame_length + 1` to be able to reference
197        // the previous sample when applying pre-emphasis.  Instead of copying
198        // a full additional sample we simply look back one position when
199        // computing each point **inside the same frame**.  The resulting
200        // values are identical to the reference implementation.
201
202        let frame_size_for_pe = frame_length + 1; // 513
203        if samples.len() < frame_size_for_pe {
204            return Ok(Vec::new());
205        }
206
207        let num_frames = (samples.len() - frame_size_for_pe) / hop_length + 1;
208        let mut mel_features = Vec::with_capacity(num_frames);
209
210        for frame_idx in 0..num_frames {
211            let start = frame_idx * hop_length;
212            let raw_frame = &samples[start..start + frame_size_for_pe];
213
214            // === Pre-emphasis (HTK flavour) ===
215            let mut frame: Vec<f32> = Vec::with_capacity(frame_length);
216            // First sample – scaled, no look-back.
217            frame.push(raw_frame[0] * (1.0 - self.preemphasis));
218            // Remaining samples use the previous raw sample within the frame.
219            for i in 1..frame_length {
220                frame.push(raw_frame[i] - self.preemphasis * raw_frame[i - 1]);
221            }
222
223            // === Window ===
224            let mut windowed: Vec<Complex32> = frame
225                .iter()
226                .zip(window.iter())
227                .map(|(s, w)| Complex32::new(s * *w as f32, 0.0))
228                .collect();
229
230            // === FFT ===
231            windowed.resize(n_fft, Complex32::new(0.0, 0.0));
232            fft.process(&mut windowed);
233
234            // Positive frequencies magnitude (length n_fft/2 + 1)
235            let magnitude: Vec<f32> = windowed[0..n_fft / 2 + 1]
236                .iter()
237                .map(|c| c.norm())
238                .collect();
239
240            // === Mel filter-bank projection ===
241            let mut mel_frame = vec![0.0f32; self.feature_size];
242            for (mel_idx, filter) in mel_filters.iter().enumerate() {
243                let mut sum = 0.0f32;
244                for (freq_idx, &coeff) in filter.iter().enumerate() {
245                    if freq_idx < magnitude.len() {
246                        sum += magnitude[freq_idx] * coeff;
247                    }
248                }
249                mel_frame[mel_idx] = (sum.max(self.mel_floor)).ln();
250            }
251
252            mel_features.push(mel_frame);
253        }
254
255        Ok(mel_features)
256    }

258    fn create_mel_filterbank(
259        &self,
260        n_mels: usize,
261        n_fft: usize,
262        sample_rate: f32,
263    ) -> Result<Vec<Vec<f32>>> {
264        let n_freqs = n_fft / 2 + 1;
265
266        // Create frequency bins (actual frequencies for each FFT bin)
267        let all_freqs: Vec<f32> = (0..n_freqs)
268            .map(|i| i as f32 * sample_rate / n_fft as f32)
269            .collect();
270
271        // Mel scale conversion functions
272        let hz_to_mel = |hz: f32| -> f32 { 2595.0 * (1.0 + hz / 700.0).log10() };
273        let mel_to_hz = |mel: f32| -> f32 { 700.0 * (10.0_f32.powf(mel / 2595.0) - 1.0) };
274
275        // Create mel points in Hz (not bin indices!)
276        let min_mel = hz_to_mel(self.min_frequency);
277        let max_mel = hz_to_mel(self.max_frequency);
278
279        // Linear spacing in mel scale
280        let mut f_pts = Vec::with_capacity(n_mels + 2);
281        for i in 0..n_mels + 2 {
282            let mel = min_mel + (max_mel - min_mel) * i as f32 / (n_mels + 1) as f32;
283            f_pts.push(mel_to_hz(mel));
284        }
285
286        // Create triangular filters
287        let mut filterbank = vec![vec![0.0; n_freqs]; n_mels];
288
289        for i in 0..n_mels {
290            let left_freq = f_pts[i];
291            let center_freq = f_pts[i + 1];
292            let right_freq = f_pts[i + 2];
293
294            // Apply triangular filter based on actual frequencies
295            for (j, &freq) in all_freqs.iter().enumerate() {
296                if freq >= left_freq && freq <= center_freq {
297                    // Rising edge
298                    if center_freq > left_freq {
299                        filterbank[i][j] = (freq - left_freq) / (center_freq - left_freq);
300                    }
301                } else if freq > center_freq && freq <= right_freq {
302                    // Falling edge
303                    if right_freq > center_freq {
304                        filterbank[i][j] = (right_freq - freq) / (right_freq - center_freq);
305                    }
306                }
307            }
308        }
309
310        Ok(filterbank)
311    }

313    fn normalize_mel_spectrogram(&self, mel_spectrogram: Vec<Vec<f32>>) -> Result<Vec<Vec<f32>>> {
314        let mut normalized = mel_spectrogram;
315
316        // Apply per-bin mean subtraction
317        if let Some(ref mean) = self.per_bin_mean {
318            for frame in normalized.iter_mut() {
319                for (i, val) in frame.iter_mut().enumerate() {
320                    if i < mean.len() {
321                        *val -= mean[i] as f32;
322                    }
323                }
324            }
325        }
326
327        // Apply per-bin stddev division
328        if let Some(ref stddev) = self.per_bin_stddev {
329            for frame in normalized.iter_mut() {
330                for (i, val) in frame.iter_mut().enumerate() {
331                    if i < stddev.len() && stddev[i] != 0.0 {
332                        *val /= stddev[i] as f32;
333                    }
334                }
335            }
336        }
337
338        Ok(normalized)
339    }
