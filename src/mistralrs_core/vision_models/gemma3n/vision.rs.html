45// Helper function to calculate same padding
46fn pad_same(x: &Tensor, kernel_size: usize, stride: usize, dilation: usize) -> Result<Tensor> {
47    let (_, _, ih, iw) = x.dims4()?;
48    let oh = ih.div_ceil(stride);
49    let ow = iw.div_ceil(stride);
50
51    // Calculate effective kernel size
52    let effective_kernel_h = dilation * (kernel_size - 1) + 1;
53    let effective_kernel_w = dilation * (kernel_size - 1) + 1;
54
55    let pad_h = ((oh - 1) * stride + effective_kernel_h).saturating_sub(ih);
56    let pad_w = ((ow - 1) * stride + effective_kernel_w).saturating_sub(iw);
57
58    let pad_top = pad_h / 2;
59    let pad_bottom = pad_h - pad_top;
60    let pad_left = pad_w / 2;
61    let pad_right = pad_w - pad_left;
62
63    if pad_h > 0 || pad_w > 0 {
64        x.pad_with_zeros(2, pad_top, pad_bottom)?
65            .pad_with_zeros(3, pad_left, pad_right)
66    } else {
67        Ok(x.clone())
68    }
69}

73struct Conv2dSame {
74    conv: Conv2d,
75    kernel_size: usize,
76    stride: usize,
77    dilation: usize,
78}

80impl Conv2dSame {
81    #[allow(clippy::too_many_arguments)]
82    fn new(
83        in_channels: usize,
84        out_channels: usize,
85        kernel_size: usize,
86        stride: usize,
87        dilation: usize,
88        groups: usize,
89        bias: bool,
90        vb: ShardedVarBuilder,
91    ) -> Result<Self> {
92        let cfg = Conv2dConfig {
93            padding: 0, // We'll handle padding manually
94            stride,
95            dilation,
96            groups,
97            cudnn_fwd_algo: None,
98        };
99
100        let conv = if bias {
101            conv2d(in_channels, out_channels, kernel_size, cfg, vb)?
102        } else {
103            conv2d_no_bias(in_channels, out_channels, kernel_size, cfg, vb)?
104        };
105
106        Ok(Self {
107            conv,
108            kernel_size,
109            stride,
110            dilation,
111        })
112    }
113}

116    fn forward(&self, x: &Tensor) -> Result<Tensor> {
117        let x = pad_same(x, self.kernel_size, self.stride, self.dilation)?;
118        Convolution.forward_2d(&self.conv, &x)
119    }

135#[derive(Debug, Clone)]
136struct RMSNormAct2d {
137    norm: RMSNorm,
138    activation: Option<Activation>,
139}
140
141impl RMSNormAct2d {
142    fn new(num_channels: usize, eps: f64, apply_act: bool, vb: ShardedVarBuilder) -> Result<Self> {
143        let norm = RMSNorm::new(num_channels, eps, vb)?;
144        let activation = if apply_act {
145            Some(Activation::Gelu)
146        } else {
147            None
148        };
149        Ok(Self { norm, activation })
150    }
151
152    fn forward(&self, x: &Tensor) -> Result<Tensor> {
153        let mut x = candle_nn::ops::rms_norm(
154            &x.permute((0, 2, 3, 1))?.contiguous()?,
155            &self.norm.weight,
156            self.norm.eps as f32,
157        )?
158        .permute((0, 3, 1, 2))?;
159
160        // Optional activation
161        if let Some(act) = &self.activation {
162            x = x.apply(act)?;
163        }
164
165        Ok(x)
166    }
167}

169#[derive(Debug, Clone)]
170struct LayerScale2d {
171    gamma: Tensor,
172}
173
174impl LayerScale2d {
175    fn new(dim: usize, vb: ShardedVarBuilder) -> Result<Self> {
176        let gamma = vb.get(dim, "gamma")?;
177        Ok(Self { gamma })
178    }
179
180    fn forward(&self, x: &Tensor) -> Result<Tensor> {
181        let c = self.gamma.dims1()?;
182        let gamma = self.gamma.reshape((1, c, 1, 1))?;
183        x.broadcast_mul(&gamma)
184    }
185}

202#[derive(Debug, Clone)]
203struct ConvNormAct {
204    conv: ConvType,
205    norm: Option<RMSNormAct2d>,
206}
207
208impl ConvNormAct {
209    #[allow(clippy::too_many_arguments)]
210    fn new(
211        in_chs: usize,
212        out_chs: usize,
213        kernel_size: usize,
214        stride: usize,
215        padding: usize,
216        groups: usize,
217        apply_act: bool,
218        eps: f64,
219        bias: bool,
220        vb: ShardedVarBuilder,
221    ) -> Result<Self> {
222        // Use Conv2dSame for depthwise convolutions (groups == in_chs or groups == out_chs)
223        // and for convolutions with kernel_size > 1 where padding would be needed
224        let use_same_padding = groups == in_chs || (kernel_size > 1 && padding > 0);
225
226        let conv = if use_same_padding {
227            ConvType::Same(Conv2dSame::new(
228                in_chs,
229                out_chs,
230                kernel_size,
231                stride,
232                1, // dilation
233                groups,
234                bias,
235                vb.pp("conv"),
236            )?)
237        } else {
238            let conv_cfg = Conv2dConfig {
239                stride,
240                padding,
241                groups,
242                ..Default::default()
243            };
244            let conv = if bias {
245                conv2d(in_chs, out_chs, kernel_size, conv_cfg, vb.pp("conv"))?
246            } else {
247                conv2d_no_bias(in_chs, out_chs, kernel_size, conv_cfg, vb.pp("conv"))?
248            };
249            ConvType::Regular(conv)
250        };
251
252        let norm = Some(RMSNormAct2d::new(out_chs, eps, apply_act, vb.pp("bn"))?);
253
254        Ok(Self { conv, norm })
255    }
256
257    fn forward(&self, x: &Tensor) -> Result<Tensor> {
258        let mut x = self.conv.forward(x)?;
259        if let Some(norm) = &self.norm {
260            x = norm.forward(&x)?;
261        }
262        Ok(x)
263    }
264}
265
266impl Module for ConvNormAct {
267    fn forward(&self, x: &Tensor) -> Result<Tensor> {
268        self.forward(x)
269    }
270}

272#[derive(Debug, Clone)]
273struct EdgeResidual {
274    conv_exp: Conv2dSame,
275    bn1: RMSNormAct2d,
276    conv_pwl: Conv2d,
277    bn2: RMSNormAct2d,
278    has_skip: bool,
279}
280
281impl EdgeResidual {
282    fn new(
283        in_chs: usize,
284        out_chs: usize,
285        exp_kernel_size: usize,
286        stride: usize,
287        expand_ratio: f64,
288        vb: ShardedVarBuilder,
289    ) -> Result<Self> {
290        let mid_chs = make_divisible(in_chs as f64 * expand_ratio, 8);
291        let has_skip = in_chs == out_chs && stride == 1;
292
293        let conv_exp = Conv2dSame::new(
294            in_chs,
295            mid_chs,
296            exp_kernel_size,
297            stride,
298            1, // dilation
299            1, // groups
300            false,
301            vb.pp("conv_exp"),
302        )?;
303
304        let bn1 = RMSNormAct2d::new(mid_chs, 1e-5, true, vb.pp("bn1"))?;
305
306        let conv_pwl_cfg = Conv2dConfig {
307            ..Default::default()
308        };
309
310        let conv_pwl = conv2d_no_bias(mid_chs, out_chs, 1, conv_pwl_cfg, vb.pp("conv_pwl"))?;
311
312        let bn2 = RMSNormAct2d::new(out_chs, 1e-5, false, vb.pp("bn2"))?;
313
314        Ok(Self {
315            conv_exp,
316            bn1,
317            conv_pwl,
318            bn2,
319            has_skip,
320        })
321    }
322
323    fn forward(&self, x: &Tensor) -> Result<Tensor> {
324        let shortcut = x.clone();
325        let mut x = self.conv_exp.forward(x)?;
326        x = self.bn1.forward(&x)?;
327        x = Convolution.forward_2d(&self.conv_pwl, &x)?;
328        x = self.bn2.forward(&x)?;
329
330        if self.has_skip {
331            x = (x + shortcut)?;
332        }
333
334        Ok(x)
335    }
336}

339    fn forward(&self, x: &Tensor) -> Result<Tensor> {
340        self.forward(x)
341    }

344#[derive(Debug, Clone)]
345struct UniversalInvertedResidual {
346    dw_start: Option<ConvNormAct>,
347    pw_exp: ConvNormAct,
348    dw_mid: Option<ConvNormAct>,
349    pw_proj: ConvNormAct,
350    layer_scale: Option<LayerScale2d>,
351    has_skip: bool,
352}
353
354impl UniversalInvertedResidual {
355    #[allow(clippy::too_many_arguments)]
356    fn new(
357        in_chs: usize,
358        out_chs: usize,
359        dw_kernel_size_start: usize,
360        dw_kernel_size_mid: usize,
361        stride: usize,
362        exp_ratio: f64,
363        layer_scale_init_value: Option<f64>,
364        vb: ShardedVarBuilder,
365    ) -> Result<Self> {
366        let has_skip = in_chs == out_chs && stride == 1;
367        let mid_chs = make_divisible(in_chs as f64 * exp_ratio, 8);
368
369        // DW start (optional)
370        let dw_start = if dw_kernel_size_start > 0 {
371            let dw_start_stride = if dw_kernel_size_mid > 0 { 1 } else { stride };
372            Some(ConvNormAct::new(
373                in_chs,
374                in_chs,
375                dw_kernel_size_start,
376                dw_start_stride,
377                dw_kernel_size_start / 2,
378                in_chs, // Depthwise
379                false,
380                1e-5,
381                false,
382                vb.pp("dw_start"),
383            )?)
384        } else {
385            None
386        };
387
388        // PW expansion
389        let pw_exp = ConvNormAct::new(
390            in_chs,
391            mid_chs,
392            1,
393            1,
394            0,
395            1,
396            true,
397            1e-5,
398            false,
399            vb.pp("pw_exp"),
400        )?;
401
402        // DW mid (optional)
403        let dw_mid = if dw_kernel_size_mid > 0 {
404            Some(ConvNormAct::new(
405                mid_chs,
406                mid_chs,
407                dw_kernel_size_mid,
408                stride,
409                dw_kernel_size_mid / 2,
410                mid_chs, // Depthwise
411                true,
412                1e-5,
413                false,
414                vb.pp("dw_mid"),
415            )?)
416        } else {
417            None
418        };
419
420        // PW projection
421        let pw_proj = ConvNormAct::new(
422            mid_chs,
423            out_chs,
424            1,
425            1,
426            0,
427            1,
428            false,
429            1e-5,
430            false,
431            vb.pp("pw_proj"),
432        )?;
433
434        // Layer scale
435        let layer_scale = if layer_scale_init_value.is_some() {
436            Some(LayerScale2d::new(out_chs, vb.pp("layer_scale"))?)
437        } else {
438            None
439        };
440
441        Ok(Self {
442            dw_start,
443            pw_exp,
444            dw_mid,
445            pw_proj,
446            layer_scale,
447            has_skip,
448        })
449    }
450
451    fn forward(&self, x: &Tensor) -> Result<Tensor> {
452        let shortcut = x.clone();
453
454        let mut x = x.clone();
455        if let Some(dw) = &self.dw_start {
456            x = dw.forward(&x)?;
457        }
458
459        x = self.pw_exp.forward(&x)?;
460
461        if let Some(dw) = &self.dw_mid {
462            x = dw.forward(&x)?;
463        }
464
465        x = self.pw_proj.forward(&x)?;
466
467        if let Some(ls) = &self.layer_scale {
468            x = ls.forward(&x)?;
469        }
470
471        if self.has_skip {
472            x = (x + shortcut)?;
473        }
474
475        Ok(x)
476    }
477}

481        self.forward(x)
482    }

485#[derive(Debug, Clone)]
486struct MultiQueryAttention2d {
487    num_heads: usize,
488    key_dim: usize,
489    value_dim: usize,
490    scale: f64,
491    query_proj: Conv2d,
492    key_down_conv: Option<Conv2dSame>,
493    key_norm: Option<RMSNormAct2d>,
494    key_proj: Conv2d,
495    value_down_conv: Option<Conv2dSame>,
496    value_norm: Option<RMSNormAct2d>,
497    value_proj: Conv2d,
498    output_proj: Conv2d,
499}
500
501impl MultiQueryAttention2d {
502    #[allow(clippy::too_many_arguments)]
503    fn new(
504        dim: usize,
505        dim_out: usize,
506        num_heads: usize,
507        key_dim: usize,
508        value_dim: usize,
509        kv_stride: usize,
510        dw_kernel_size: usize,
511        vb: ShardedVarBuilder,
512    ) -> Result<Self> {
513        let scale = (key_dim as f64).powf(-0.5);
514
515        // Query projection
516        let query_proj = conv2d_no_bias(
517            dim,
518            num_heads * key_dim,
519            1,
520            Conv2dConfig::default(),
521            vb.pp("query").pp("proj"),
522        )?;
523
524        // Key path
525        let (key_down_conv, key_norm) = if kv_stride > 1 {
526            let down_conv = Conv2dSame::new(
527                dim,
528                dim,
529                dw_kernel_size,
530                kv_stride,
531                1,   // dilation
532                dim, // Depthwise
533                false,
534                vb.pp("key").pp("down_conv"),
535            )?;
536            let norm = RMSNormAct2d::new(dim, 1e-6, false, vb.pp("key").pp("norm"))?;
537            (Some(down_conv), Some(norm))
538        } else {
539            (None, None)
540        };
541
542        let key_proj = conv2d_no_bias(
543            dim,
544            key_dim,
545            1,
546            Conv2dConfig::default(),
547            vb.pp("key").pp("proj"),
548        )?;
549
550        // Value path
551        let (value_down_conv, value_norm) = if kv_stride > 1 {
552            let down_conv = Conv2dSame::new(
553                dim,
554                dim,
555                dw_kernel_size,
556                kv_stride,
557                1,   // dilation
558                dim, // Depthwise
559                false,
560                vb.pp("value").pp("down_conv"),
561            )?;
562            let norm = RMSNormAct2d::new(dim, 1e-6, false, vb.pp("value").pp("norm"))?;
563            (Some(down_conv), Some(norm))
564        } else {
565            (None, None)
566        };
567
568        let value_proj = conv2d_no_bias(
569            dim,
570            value_dim,
571            1,
572            Conv2dConfig::default(),
573            vb.pp("value").pp("proj"),
574        )?;
575
576        // Output projection
577        let output_proj = conv2d_no_bias(
578            value_dim * num_heads,
579            dim_out,
580            1,
581            Conv2dConfig::default(),
582            vb.pp("output").pp("proj"),
583        )?;
584
585        Ok(Self {
586            num_heads,
587            key_dim,
588            value_dim,
589            scale,
590            query_proj,
591            key_down_conv,
592            key_norm,
593            key_proj,
594            value_down_conv,
595            value_norm,
596            value_proj,
597            output_proj,
598        })
599    }
600
601    fn forward(&self, x: &Tensor) -> Result<Tensor> {
602        let (b, _c, h, w) = x.dims4()?;
603
604        // Query projection and reshape
605        // [B, H, W, C] -> [B, H, W, num_heads * key_dim] -> [B, H*W, num_heads, key_dim] -> [B, num_heads, H*W, key_dim]
606        let mut q = Convolution.forward_2d(&self.query_proj, x)?;
607        q = q
608            .permute((0, 2, 3, 1))? // NCHW -> NHWC
609            .reshape((b, h * w, self.num_heads, self.key_dim))?
610            .permute((0, 2, 1, 3))?; // [B, num_heads, H*W, key_dim]
611
612        // Key projection and reshape
613        let mut k = x.clone();
614        if let (Some(down_conv), Some(norm)) = (&self.key_down_conv, &self.key_norm) {
615            k = down_conv.forward(&k)?;
616            k = norm.forward(&k)?;
617        }
618        k = Convolution.forward_2d(&self.key_proj, &k)?;
619        let (_, _, kh, kw) = k.dims4()?;
620        // [B, C, H, W] -> [B, H, W, C] -> [B, H*W, C] -> [B, 1, H*W, C]
621        k = k
622            .permute((0, 2, 3, 1))? // NCHW -> NHWC
623            .reshape((b, kh * kw, self.key_dim))?
624            .unsqueeze(1)?; // [B, 1, kh*kw, key_dim]
625
626        // Value projection and reshape
627        let mut v = x.clone();
628        if let (Some(down_conv), Some(norm)) = (&self.value_down_conv, &self.value_norm) {
629            v = down_conv.forward(&v)?;
630            v = norm.forward(&v)?;
631        }
632        v = Convolution.forward_2d(&self.value_proj, &v)?;
633        let (_, _, vh, vw) = v.dims4()?;
634        // [B, C, H, W] -> [B, H, W, C] -> [B, H*W, C] -> [B, 1, H*W, C]
635        v = v
636            .permute((0, 2, 3, 1))? // NCHW -> NHWC
637            .reshape((b, vh * vw, self.value_dim))?
638            .unsqueeze(1)?; // [B, 1, vh*vw, value_dim]
639
640        let sdpa_params = SdpaParams {
641            n_kv_groups: self.num_heads,
642            softcap: None,
643            softmax_scale: self.scale as f32,
644            sliding_window: None,
645        };
646        let mut o = Sdpa.run_attention_noflash(&q, &k, &v, None, &sdpa_params)?;
647
648        // Reshape output back
649        // [B, num_heads, H*W, value_dim] -> [B, H*W, num_heads, value_dim] -> [B, H, W, num_heads * value_dim]
650        o = o
651            .permute((0, 2, 1, 3))? // [B, H*W, num_heads, value_dim]
652            .reshape((b, h, w, self.num_heads * self.value_dim))?
653            .permute((0, 3, 1, 2))?; // NHWC -> NCHW
654
655        o = Convolution.forward_2d(&self.output_proj, &o)?;
656
657        Ok(o)
658    }
659}

661#[derive(Debug, Clone)]
662struct MobileAttention {
663    norm: RMSNormAct2d,
664    attn: MultiQueryAttention2d,
665    layer_scale: Option<LayerScale2d>,
666    has_skip: bool,
667}
668
669impl MobileAttention {
670    #[allow(clippy::too_many_arguments)]
671    fn new(
672        in_chs: usize,
673        out_chs: usize,
674        stride: usize,
675        num_heads: usize,
676        key_dim: usize,
677        value_dim: usize,
678        kv_stride: usize,
679        dw_kernel_size: usize,
680        layer_scale_init_value: Option<f64>,
681        vb: ShardedVarBuilder,
682    ) -> Result<Self> {
683        let has_skip = stride == 1 && in_chs == out_chs;
684
685        let norm = RMSNormAct2d::new(in_chs, 1e-5, false, vb.pp("norm"))?;
686
687        let attn = MultiQueryAttention2d::new(
688            in_chs,
689            out_chs,
690            num_heads,
691            key_dim,
692            value_dim,
693            kv_stride,
694            dw_kernel_size,
695            vb.pp("attn"),
696        )?;
697
698        let layer_scale = if layer_scale_init_value.is_some() {
699            Some(LayerScale2d::new(out_chs, vb.pp("layer_scale"))?)
700        } else {
701            None
702        };
703
704        Ok(Self {
705            norm,
706            attn,
707            layer_scale,
708            has_skip,
709        })
710    }
711
712    fn forward(&self, x: &Tensor) -> Result<Tensor> {
713        let shortcut = x.clone();
714
715        let mut x = self.norm.forward(x)?;
716        x = self.attn.forward(&x)?;
717
718        if let Some(ls) = &self.layer_scale {
719            x = ls.forward(&x)?;
720        }
721
722        if self.has_skip {
723            x = (x + shortcut)?;
724        }
725
726        Ok(x)
727    }
728}

736// Helper functions
737pub fn make_divisible(v: f64, divisor: usize) -> usize {
738    let divisor = divisor as f64;
739    let min_value = divisor;
740    let new_v = ((v + divisor / 2.0) / divisor).floor() * divisor;
741    let new_v = new_v.max(min_value) as usize;
742    // Make sure that round down does not go down by more than 10%.
743    if (new_v as f64) < 0.9 * v {
744        new_v + divisor as usize
745    } else {
746        new_v
747    }
748}

750// Multi-scale fusion adapter
751#[derive(Debug, Clone)]
752struct MobileNetV5MultiScaleFusionAdapter {
753    output_resolution: (usize, usize),
754    ffn: UniversalInvertedResidual,
755    norm: RMSNormAct2d,
756}
757
758impl MobileNetV5MultiScaleFusionAdapter {
759    fn new(
760        in_chs: Vec<usize>,
761        out_chs: usize,
762        output_resolution: (usize, usize),
763        expansion_ratio: f64,
764        use_layer_scale: bool,
765        layer_scale_init_value: f64,
766        vb: ShardedVarBuilder,
767    ) -> Result<Self> {
768        let in_channels = in_chs.iter().sum();
769
770        let layer_scale = if use_layer_scale {
771            Some(layer_scale_init_value)
772        } else {
773            None
774        };
775
776        let ffn = UniversalInvertedResidual::new(
777            in_channels,
778            out_chs,
779            0, // dw_kernel_size_start
780            0, // dw_kernel_size_mid
781            1, // stride
782            expansion_ratio,
783            layer_scale,
784            vb.pp("ffn"),
785        )?;
786
787        let norm = RMSNormAct2d::new(out_chs, 1e-6, false, vb.pp("norm"))?;
788
789        Ok(Self {
790            output_resolution,
791            ffn,
792            norm,
793        })
794    }
795
796    fn forward(&self, inputs: &[Tensor]) -> Result<Tensor> {
797        // Get the highest resolution from the first input
798        let (_, _, h0, w0) = inputs[0].dims4()?;
799
800        // Resize inputs to match highest resolution
801        let mut resized_inputs = Vec::with_capacity(inputs.len());
802        for img in inputs {
803            let (_, _, h, w) = img.dims4()?;
804            if h < h0 || w < w0 {
805                // Use bilinear interpolation as nearest equivalent
806                let resized = img.upsample_nearest2d(h0, w0)?;
807                resized_inputs.push(resized);
808            } else {
809                resized_inputs.push(img.clone());
810            }
811        }
812
813        // Concatenate along channel dimension
814        let channel_cat_imgs = Tensor::cat(&resized_inputs, 1)?;
815
816        // Apply FFN first
817        let mut img = self.ffn.forward(&channel_cat_imgs)?;
818
819        // Resize / pool to target output resolution *before* final normalisation
820        let (out_h, out_w) = self.output_resolution;
821        if h0 != out_h || w0 != out_w {
822            if h0 % out_h != 0 || w0 % out_w != 0 {
823                // Fallback to bilinear interpolation if input spatial dims are
824                // not integer multiples of the desired output size. We use
825                // Candles nearest-neighbour upsample as the closest available op.
826                img = img.upsample_nearest2d(out_h, out_w)?;
827            } else {
828                let h_stride = h0 / out_h;
829                let w_stride = w0 / out_w;
830                img = img.avg_pool2d((h_stride, w_stride))?;
831            }
832        }
833
834        img = self.norm.forward(&img)?;
835
836        Ok(img)
837    }
838}

1057impl VisionTower {
1058    pub fn new(vb: ShardedVarBuilder) -> Result<Self> {
1059        // Some models have invalid vision tower weights from the old gemma 3n upload
1060        // https://github.com/EricLBuehler/mistral.rs/issues/1592
1061        let old_vision_tower = !vb.contains_tensor("conv_stem.conv.bias");

1065See: https://github.com/EricLBuehler/mistral.rs/issues/1592
1066
1067The vision tower for this model will still be loaded, but you might experience degraded quality."
1068            );
1069        }
1070        let conv_stem_bias = !old_vision_tower;
1071        // Initial stem convolution
1072        let conv_stem = ConvNormAct::new(
1073            3,              // in_chs
1074            64,             // out_chs
1075            3,              // kernel_size
1076            2,              // stride
1077            1,              // padding
1078            1,              // groups
1079            true,           // apply_act
1080            1e-5,           // eps
1081            conv_stem_bias, // bias
1082            vb.pp("conv_stem"),
1083        )?;
1084
1085        // Build blocks according to architecture definition
1086        let block_defs = gemma3n_mobilenet_def();
1087        let mut blocks = Vec::with_capacity(block_defs.len());
1088        let mut in_chs = 64;
1089
1090        for (stage_idx, stage_blocks) in block_defs.iter().enumerate() {
1091            let mut stage = Vec::with_capacity(stage_blocks.len());
1092
1093            for (block_idx, block_type) in stage_blocks.iter().enumerate() {
1094                let block = match block_type {
1095                    BlockType::EdgeResidual {
1096                        out_channels,
1097                        kernel_size,
1098                        stride,
1099                        expand_ratio,
1100                        ..
1101                    } => {
1102                        let edge_res = EdgeResidual::new(
1103                            in_chs,
1104                            *out_channels,
1105                            *kernel_size,
1106                            *stride,
1107                            *expand_ratio,
1108                            vb.pp(format!("blocks.{stage_idx}.{block_idx}")),
1109                        )?;
1110                        in_chs = *out_channels;
1111                        Block::EdgeResidual(edge_res)
1112                    }
1113                    BlockType::UniversalInvertedResidual {
1114                        out_channels,
1115                        start_kernel_size,
1116                        mid_kernel_size,
1117                        stride,
1118                        expand_ratio,
1119                        ..
1120                    } => {
1121                        let uir = UniversalInvertedResidual::new(
1122                            in_chs,
1123                            *out_channels,
1124                            *start_kernel_size,
1125                            *mid_kernel_size,
1126                            *stride,
1127                            *expand_ratio,
1128                            Some(1e-5), // layer_scale_init_value
1129                            vb.pp(format!("blocks.{stage_idx}.{block_idx}")),
1130                        )?;
1131                        in_chs = *out_channels;
1132                        Block::UniversalInvertedResidual(uir)
1133                    }
1134                    BlockType::MultiQueryAttention {
1135                        num_heads,
1136                        kv_dim,
1137                        kv_stride,
1138                        ..
1139                    } => {
1140                        let ma = MobileAttention::new(
1141                            in_chs,
1142                            in_chs, // out_chs same as in_chs
1143                            1,      // stride
1144                            *num_heads,
1145                            *kv_dim,
1146                            *kv_dim, // value_dim same as key_dim
1147                            *kv_stride,
1148                            3,          // dw_kernel_size
1149                            Some(1e-5), // layer_scale_init_value
1150                            vb.pp(format!("blocks.{stage_idx}.{block_idx}")),
1151                        )?;
1152                        Block::MobileAttention(ma)
1153                    }
1154                };
1155                stage.push(block);
1156            }
1157            blocks.push(stage);
1158        }
1159
1160        // Multi-scale fusion adapter
1161        // Collecting from stages 3 and 4 (after 640 and 1280 channel blocks)
1162        let msfa = MobileNetV5MultiScaleFusionAdapter::new(
1163            vec![640, 1280], // in_chs from stages 3 and 4
1164            2048,            // out_chs
1165            (16, 16),        // output_resolution
1166            2.0,             // expansion_ratio
1167            false,           // use_layer_scale
1168            1e-5,            // layer_scale_init_value
1169            vb.pp("msfa"),
1170        )?;
1171
1172        Ok(Self {
1173            conv_stem,
1174            blocks,
1175            msfa,
1176            msfa_indices: vec![3, 4], // Indices for multi-scale features
1177            old_vision_tower,
1178        })
1179    }
1180
1181    pub fn forward(&self, x: &Tensor) -> Result<Tensor> {
1182        let mut x = if self.old_vision_tower {
1183            // Some models have invalid vision tower weights from the old gemma 3n upload
1184            // https://github.com/EricLBuehler/mistral.rs/issues/1592
1185
1186            // This is a hack necessary because the weights for Gemma 3n are broken and require the image to be rotated.
1187            x.t()?
1188        } else {
1189            x.clone()
1190        };
1191
1192        // Apply stem
1193        x = self.conv_stem.forward(&x)?;
1194
1195        let mut intermediates = Vec::new();
1196
1197        // Process blocks stage by stage
1198        for (stage_idx, stage) in self.blocks.iter().enumerate() {
1199            for block in stage {
1200                x = block.forward(&x)?;
1201            }
1202
1203            // Collect intermediate features for multi-scale fusion
1204            if self.msfa_indices.contains(&(stage_idx + 1)) {
1205                intermediates.push(x.clone());
1206            }
1207        }
1208
1209        // Apply multi-scale fusion adapter
1210        let x = self.msfa.forward(&intermediates)?;
1211
1212        Ok(x)
1213    }
1214
1215    pub fn residual_tensors(&self) -> Vec<(String, Tensor)> {
1216        let uvb = UnVarBuilder::new();
1217
1218        // Add conv_stem tensors
1219        add_conv_norm_act(&uvb.pp("conv_stem"), &self.conv_stem);
1220
1221        // Add blocks tensors
1222        for (stage_idx, stage) in self.blocks.iter().enumerate() {
1223            for (block_idx, block) in stage.iter().enumerate() {
1224                let uvb_block = uvb.pp(format!("blocks.{stage_idx}.{block_idx}"));
1225                match block {
1226                    Block::EdgeResidual(edge) => add_edge_residual(&uvb_block, edge),
1227                    Block::UniversalInvertedResidual(uir) => {
1228                        add_universal_inverted_residual(&uvb_block, uir)
1229                    }
1230                    Block::MobileAttention(ma) => add_mobile_attention(&uvb_block, ma),
1231                }
1232            }
1233        }
1234
1235        // Add MSFA tensors
1236        add_msfa(&uvb.pp("msfa"), &self.msfa);
1237
1238        uvb.to_safetensors()
1239    }
1240}

1242// Helper functions for adding residual tensors
1243fn add_conv_norm_act(uvb: &UnVarBuilder, cna: &ConvNormAct) {
1244    // Add conv layer
1245    match &cna.conv {
1246        ConvType::Regular(conv) => uvb.pp("conv").add(conv),
1247        ConvType::Same(conv) => uvb.pp("conv").add(&conv.conv),
1248    }
1249
1250    // Add norm layer
1251    if let Some(norm) = &cna.norm {
1252        uvb.pp("bn").add_tensor("weight", norm.norm.weight.clone());
1253    }
1254}
1255
1256fn add_edge_residual(uvb: &UnVarBuilder, edge: &EdgeResidual) {
1257    uvb.pp("conv_exp").add(&edge.conv_exp.conv);
1258    uvb.pp("bn1")
1259        .add_tensor("weight", edge.bn1.norm.weight.clone());
1260    uvb.pp("conv_pwl").add(&edge.conv_pwl);
1261    uvb.pp("bn2")
1262        .add_tensor("weight", edge.bn2.norm.weight.clone());
1263}
1264
1265fn add_universal_inverted_residual(uvb: &UnVarBuilder, uir: &UniversalInvertedResidual) {
1266    // Add dw_start if present
1267    if let Some(dw_start) = &uir.dw_start {
1268        add_conv_norm_act(&uvb.pp("dw_start"), dw_start);
1269    }
1270
1271    // Add pw_exp
1272    add_conv_norm_act(&uvb.pp("pw_exp"), &uir.pw_exp);
1273
1274    // Add dw_mid if present
1275    if let Some(dw_mid) = &uir.dw_mid {
1276        add_conv_norm_act(&uvb.pp("dw_mid"), dw_mid);
1277    }
1278
1279    // Add pw_proj
1280    add_conv_norm_act(&uvb.pp("pw_proj"), &uir.pw_proj);
1281
1282    // Add layer_scale if present
1283    if let Some(layer_scale) = &uir.layer_scale {
1284        uvb.pp("layer_scale")
1285            .add_tensor("gamma", layer_scale.gamma.clone());
1286    }
1287}
1288
1289fn add_mobile_attention(uvb: &UnVarBuilder, ma: &MobileAttention) {
1290    // Add norm
1291    uvb.pp("norm")
1292        .add_tensor("weight", ma.norm.norm.weight.clone());
1293
1294    // Add attention components
1295    let uvb_attn = uvb.pp("attn");
1296
1297    // Query projection
1298    uvb_attn.pp("query").pp("proj").add(&ma.attn.query_proj);
1299
1300    // Key components
1301    if let Some(key_down_conv) = &ma.attn.key_down_conv {
1302        uvb_attn.pp("key").pp("down_conv").add(&key_down_conv.conv);
1303    }
1304    if let Some(key_norm) = &ma.attn.key_norm {
1305        uvb_attn
1306            .pp("key")
1307            .pp("norm")
1308            .add_tensor("weight", key_norm.norm.weight.clone());
1309    }
1310    uvb_attn.pp("key").pp("proj").add(&ma.attn.key_proj);
1311
1312    // Value components
1313    if let Some(value_down_conv) = &ma.attn.value_down_conv {
1314        uvb_attn
1315            .pp("value")
1316            .pp("down_conv")
1317            .add(&value_down_conv.conv);
1318    }
1319    if let Some(value_norm) = &ma.attn.value_norm {
1320        uvb_attn
1321            .pp("value")
1322            .pp("norm")
1323            .add_tensor("weight", value_norm.norm.weight.clone());
1324    }
1325    uvb_attn.pp("value").pp("proj").add(&ma.attn.value_proj);
1326
1327    // Output projection
1328    uvb_attn.pp("output").pp("proj").add(&ma.attn.output_proj);
1329
1330    // Layer scale if present
1331    if let Some(layer_scale) = &ma.layer_scale {
1332        uvb.pp("layer_scale")
1333            .add_tensor("gamma", layer_scale.gamma.clone());
1334    }
1335}

1337fn add_msfa(uvb: &UnVarBuilder, msfa: &MobileNetV5MultiScaleFusionAdapter) {
1338    // Add FFN (UniversalInvertedResidual)
1339    add_universal_inverted_residual(&uvb.pp("ffn"), &msfa.ffn);
1340
1341    // Add norm
1342    uvb.pp("norm")
1343        .add_tensor("weight", msfa.norm.norm.weight.clone());
