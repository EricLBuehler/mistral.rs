133    fn std_multiplier(p: f64) -> f64 {
134        let normal = Normal::new(0.0, 1.0).unwrap();
135        normal.inverse_cdf(p)
136    }

138    fn gaussian_topk(&self, xs: &Tensor) -> Result<Tensor> {
139        // Cast to float32 for better precision in statistical calculations
140        let xs_f32 = xs.to_dtype(DType::F32)?;
141        let xs_mean = xs_f32.mean_keepdim(D::Minus1)?;
142        let xs_sq_mean = xs_f32.sqr()?.mean_keepdim(D::Minus1)?;
143        let var = (&xs_sq_mean - xs_mean.sqr()?)?;
144        let xs_std = (var + EPS)?.sqrt()?;
145        let cutoff_xs = (xs_mean + (xs_std * self.std_multiplier)?)?;
146        // Convert back to original dtype after computation
147        xs.broadcast_sub(&cutoff_xs.to_dtype(xs.dtype())?)?.relu()
148    }
149
150    fn forward(&self, xs: &Tensor) -> Result<Tensor> {
151        let original_dtype = xs.dtype();
152        let mut xs = xs.clone();
153        if let Some(t) = self.gate.quantized_act_type() {
154            xs = xs.to_dtype(t)?;
155        }
156        let mut gate = self.gate.forward(&xs)?;
157        if self.activation_sparsity > 0. {
158            gate = self.gaussian_topk(&gate)?;
159        }
160        let up = self.up.forward(&xs)?;
161        // let mut res = self.down.forward(&crate::ops::mul_and_act(
162        //     &gate,
163        //     &up,
164        //     self.act.try_into()?,
165        // )?)?;
166        let mut res = self.down.forward(&(&gate.apply(&self.act)? * &up)?)?;
167        if self.gate.quantized_act_type().is_some() {
168            res = res.to_dtype(original_dtype)?;
169        }
170        Ok(res)
171    }

314    fn rotate_half(xs: &Tensor) -> Result<Tensor> {
315        let last_dim = xs.dim(D::Minus1)?;
316        let xs1 = xs.narrow(D::Minus1, 0, last_dim / 2)?;
317        let xs2 = xs.narrow(D::Minus1, last_dim / 2, last_dim - last_dim / 2)?;
318        Tensor::cat(&[&xs2.neg()?, &xs1], D::Minus1)
319    }
320
321    fn apply_rotary_pos_emb(xs: &Tensor, cos: &Tensor, sin: &Tensor) -> Result<Tensor> {
322        // Perform rotary position embedding in float32 for better precision
323        let xs_f32 = xs.to_dtype(DType::F32)?;
324        let cos_f32 = cos.to_dtype(DType::F32)?;
325        let sin_f32 = sin.to_dtype(DType::F32)?;
326        let result_f32 = (xs_f32.broadcast_mul(&cos_f32.unsqueeze(2)?)?
327            + Self::rotate_half(&xs_f32)?.broadcast_mul(&sin_f32.unsqueeze(2)?)?)?;
328        result_f32.to_dtype(xs.dtype())
329    }

331    fn apply_rope(&self, xs: &Tensor, seqlen_offsets: &[usize], seq_len: usize) -> Result<Tensor> {
332        match self.use_sliding_window {
333            true => {
334                let (cos, sin) = self.rotary_emb_local.get_cos_sin()?;
335
336                let mut embeds = Vec::new();
337                for (i, offset) in seqlen_offsets.iter().enumerate() {
338                    let cos = cos
339                        .narrow(0, *offset, seq_len)?
340                        .unsqueeze(0)?
341                        .repeat((1, 1, 2))?;
342                    let sin = sin
343                        .narrow(0, *offset, seq_len)?
344                        .unsqueeze(0)?
345                        .repeat((1, 1, 2))?;
346                    let embed = Self::apply_rotary_pos_emb(
347                        &xs.i(i)?.unsqueeze(0)?.contiguous()?,
348                        &cos,
349                        &sin,
350                    )?;
351                    embeds.push(embed);
352                }
353                Tensor::cat(&embeds, 0)
354            }
355            false => {
356                let (cos, sin) = self.rotary_emb_global.get_cos_sin()?;
357
358                let mut embeds = Vec::new();
359                for (i, offset) in seqlen_offsets.iter().enumerate() {
360                    let cos = cos
361                        .narrow(0, *offset, seq_len)?
362                        .unsqueeze(0)?
363                        .repeat((1, 1, 2))?;
364                    let sin = sin
365                        .narrow(0, *offset, seq_len)?
366                        .unsqueeze(0)?
367                        .repeat((1, 1, 2))?;
368                    let embed = Self::apply_rotary_pos_emb(
369                        &xs.i(i)?.unsqueeze(0)?.contiguous()?,
370                        &cos,
371                        &sin,
372                    )?;
373                    embeds.push(embed);
374                }
375                Tensor::cat(&embeds, 0)
376            }
377        }
378    }

380    #[allow(clippy::too_many_arguments)]
381    fn forward(
382        &self,
383        xs: &Tensor,
384        attention_mask: Option<&Tensor>,
385        sliding_attention_mask: Option<&Tensor>,
386        seqlen_offsets: &[usize],
387        kv_caches: &mut [KvCache],
388        flash_params: &FlashParams,
389    ) -> Result<Tensor> {
390        let (b_sz, q_len, _) = xs.dims3()?;
391
392        let mut q = self.q_proj.forward_autocast(xs)?;
393        q = q.reshape((b_sz, q_len, self.num_heads, self.head_dim))?;
394        q = q.apply(&self.q_norm)?;
395        q = self.apply_rope(&q, seqlen_offsets, q_len)?;
396        q = q.transpose(1, 2)?;
397
398        let ((k, v), is_shared_kv) = if let Some(kv_shared_layer_index) = self.kv_shared_layer_index
399        {
400            let shared_cache = &kv_caches[kv_shared_layer_index];
401            (
402                (shared_cache.k()?.unwrap(), shared_cache.v()?.unwrap()),
403                true,
404            )
405        } else {
406            let mut k = self.k_proj.forward_autocast(xs)?;
407            k = k.reshape((b_sz, q_len, self.num_kv_heads, self.head_dim))?;
408            k = k.apply(&self.k_norm)?;
409            k = self.apply_rope(&k, seqlen_offsets, q_len)?;
410            k = k.transpose(1, 2)?;
411
412            let mut v = self.v_proj.forward_autocast(xs)?;
413            v = v.reshape((b_sz, q_len, self.num_kv_heads, self.head_dim))?;
414            v = v.apply(&self.v_norm)?;
415            v = v.transpose(1, 2)?;
416
417            (kv_caches[self.layer_idx].append(&k, &v)?, false)
418        };
419
420        let mask = if self.use_sliding_window {
421            sliding_attention_mask
422        } else {
423            attention_mask
424        };
425
426        // Adjust mask dimensions if using shared KV cache
427        let mask = if is_shared_kv {
428            if let Some(mask) = mask {
429                let kv_seq_len = k.dims()[2];
430                let mask_dims = mask.dims();
431
432                // Check if we need to adjust the mask dimensions
433                match mask.rank() {
434                    2 => {
435                        // For 2D masks: (q_len, kv_len)
436                        if mask_dims[1] != kv_seq_len {
437                            Some(mask.narrow(1, 0, kv_seq_len)?)
438                        } else {
439                            Some(mask.clone())
440                        }
441                    }
442                    3 => {
443                        // For 3D masks: (batch, q_len, kv_len)
444                        if mask_dims[2] != kv_seq_len {
445                            Some(mask.narrow(2, 0, kv_seq_len)?)
446                        } else {
447                            Some(mask.clone())
448                        }
449                    }
450                    4 => {
451                        // For 4D masks: (batch, heads, q_len, kv_len)
452                        if mask_dims[3] != kv_seq_len {
453                            Some(mask.narrow(3, 0, kv_seq_len)?)
454                        } else {
455                            Some(mask.clone())
456                        }
457                    }
458                    _ => Some(mask.clone()),
459                }
460            } else {
461                None
462            }
463        } else {
464            mask.cloned()
465        };
466
467        let mut attn_output = Sdpa.run_attention(
468            &q,
469            &k,
470            &v,
471            mask.as_ref(),
472            Some(flash_params),
473            &self.sdpa_params,
474        )?;
475
476        attn_output = attn_output.transpose(1, 2)?.reshape((b_sz, q_len, ()))?;
477        self.o_proj.forward_autocast(&attn_output)
478    }
479}

481struct TextAltUp {
482    correct_output_scale: Tensor,
483    correction_coefs: Linear,
484    prediction_coefs: Linear,
485    modality_router: Linear,
486    router_norm: RmsNorm,
487    router_input_scale: f64,
488    altup_active_idx: usize,
489    altup_num_inputs: usize,
490}
491
492impl TextAltUp {
493    fn new(cfg: &Gemma3nTextConfig, vb: ShardedVarBuilder) -> Result<Self> {
494        let correct_output_scale = vb.get(cfg.hidden_size, "correct_output_scale")?;
495        let mut correction_coefs = layers::linear_no_bias(
496            cfg.altup_num_inputs,
497            cfg.altup_num_inputs,
498            vb.pp("correction_coefs"),
499        )?;
500        if let Some(altup_coef_clip) = cfg.altup_coef_clip {
501            correction_coefs = Linear::new(
502                correction_coefs
503                    .weight()
504                    .clamp(-altup_coef_clip, altup_coef_clip)?,
505                None,
506            );
507        }
508        let prediction_coefs = layers::linear_no_bias(
509            cfg.altup_num_inputs,
510            cfg.altup_num_inputs.pow(2),
511            vb.pp("prediction_coefs"),
512        )?;
513        let modality_router = layers::linear_no_bias(
514            cfg.hidden_size,
515            cfg.altup_num_inputs,
516            vb.pp("modality_router"),
517        )?;
518        let router_norm = RmsNorm::new_gemma_3n(
519            cfg.hidden_size,
520            cfg.rms_norm_eps,
521            true,
522            vb.pp("router_norm"),
523        )?;
524
525        Ok(Self {
526            correct_output_scale,
527            correction_coefs,
528            prediction_coefs,
529            modality_router,
530            router_norm,
531            router_input_scale: 1. / (cfg.hidden_size as f64),
532            altup_active_idx: cfg.altup_active_idx,
533            altup_num_inputs: cfg.altup_num_inputs,
534        })
535    }
536
537    fn compute_router_modalities(&self, xs: &Tensor) -> Result<Tensor> {
538        // Perform routing calculations in float32 for better precision
539        let router_inputs_normed = self.router_norm.forward(xs)?;
540        let router_inputs_f32 = router_inputs_normed.to_dtype(DType::F32)?;
541        let router_inputs = (router_inputs_f32 * self.router_input_scale)?.to_dtype(xs.dtype())?;
542        let routed = self.modality_router.forward(&router_inputs)?;
543        routed.to_dtype(DType::F32)?.tanh()?.to_dtype(xs.dtype())
544    }
545
546    fn predict(&self, xs: &Tensor) -> Result<Tensor> {
547        let modalities = self.compute_router_modalities(&xs.i(self.altup_active_idx)?)?;
548
549        // Project and then transpose all 2D matrices contained so that mulmat gives the correct result
550        let shape = [
551            modalities.dims()[..modalities.dims().len() - 1].to_vec(),
552            vec![self.altup_num_inputs, self.altup_num_inputs],
553        ]
554        .concat();
555        let all_coefs = self
556            .prediction_coefs
557            .forward(&modalities)?
558            .reshape(shape)?
559            .permute((0, 1, 3, 2))?;
560
561        // permute hidden_states to [batch_size, num_tokens, hidden_size, altup_num_inputs]
562        let mut predictions = xs
563            .permute((1, 2, 3, 0))?
564            .contiguous()?
565            .matmul(&all_coefs.contiguous()?)?;
566        predictions = predictions.permute((3, 0, 1, 2))?;
567        // Perform addition in float32 for precision
568        let predictions_f32 = predictions.to_dtype(DType::F32)?;
569        let xs_f32 = xs.to_dtype(DType::F32)?;
570        predictions = (predictions_f32 + xs_f32)?.to_dtype(predictions.dtype())?;
571        predictions.contiguous()
572    }
573
574    fn correct(&self, predictions: &Tensor, activated: &Tensor) -> Result<Tensor> {
575        let modalities = self.compute_router_modalities(activated)?;
576        // Compute innovation in float32 for better precision
577        let activated_f32 = activated.to_dtype(DType::F32)?;
578        let pred_active_f32 = predictions.i(self.altup_active_idx)?.to_dtype(DType::F32)?;
579        let innovation_f32 = activated_f32.broadcast_sub(&pred_active_f32)?;
580        let innovation =
581            innovation_f32
582                .to_dtype(activated.dtype())?
583                .repeat((self.altup_num_inputs, 1, 1, 1))?;
584
585        // Perform coefficient computation in float32
586        let coefs = self.correction_coefs.forward(&modalities)?;
587        let coefs_f32 = coefs.to_dtype(DType::F32)?;
588        let all_coefs = (coefs_f32 + 1.)?
589            .to_dtype(coefs.dtype())?
590            .permute((2, 0, 1))?
591            .unsqueeze(D::Minus1)?;
592
593        innovation
594            .broadcast_mul(&all_coefs)?
595            .broadcast_add(predictions)?
596            .contiguous()
597    }
598
599    fn scale_corrected_output(&self, xs: &Tensor) -> Result<Tensor> {
600        xs.to_dtype(self.correct_output_scale.dtype())?
601            .broadcast_mul(&self.correct_output_scale)?
602            .to_dtype(xs.dtype())
603    }

612impl TextLaurelBlock {
613    fn new(cfg: &Gemma3nTextConfig, vb: ShardedVarBuilder) -> Result<Self> {
614        Ok(Self {
615            left: layers::linear_no_bias(cfg.hidden_size, cfg.laurel_rank, vb.pp("linear_left"))?,
616            right: layers::linear_no_bias(cfg.laurel_rank, cfg.hidden_size, vb.pp("linear_right"))?,
617            post_norm: RmsNorm::new_gemma_3n(
618                cfg.hidden_size,
619                cfg.rms_norm_eps,
620                true,
621                vb.pp("post_laurel_norm"),
622            )?,
623        })
624    }
625
626    fn forward(&self, xs: &Tensor) -> Result<Tensor> {
627        let mut laurel_xs = self.left.forward(xs)?;
628        laurel_xs = self.right.forward(&laurel_xs)?;
629        laurel_xs = self.post_norm.forward(&laurel_xs)?;
630        // Perform addition in float32 for precision
631        let xs_f32 = xs.to_dtype(DType::F32)?;
632        let laurel_xs_f32 = laurel_xs.to_dtype(DType::F32)?;
633        (xs_f32 + laurel_xs_f32)?.to_dtype(xs.dtype())
634    }

655    #[allow(clippy::too_many_arguments)]

657        rotary_emb_global: Arc<Gemma3nRotaryEmbedding>,

683            cfg.rms_norm_eps,

690            true,
691            mapper.set_device(layer_idx, vb.pp("post_attention_layernorm"), false),

694            cfg.hidden_size,
695            cfg.rms_norm_eps,
696            true,
697            mapper.set_device(layer_idx, vb.pp("pre_feedforward_layernorm"), false),

704        )?;
705
706        let altup = TextAltUp::new(cfg, mapper.set_device(layer_idx, vb.pp("altup"), false))?;
707        let laurel =
708            TextLaurelBlock::new(cfg, mapper.set_device(layer_idx, vb.pp("laurel"), false))?;
709        let per_layer_input_gate = layers::linear_no_bias(
710            cfg.hidden_size,

727            mlp,
728            input_layernorm,
729            post_attention_layernorm,
730            pre_feedforward_layernorm,
731            post_feedforward_layernorm,
732            altup,
733            laurel,
734            per_layer_input_gate,
735            per_layer_projection,
736            post_per_layer_input_norm,
737            altup_active_idx: cfg.altup_active_idx,
738            altup_correct_scale: cfg.altup_correct_scale,
739            act: cfg.hidden_activation,
740        })
741    }
742
743    #[allow(clippy::too_many_arguments)]
744    fn forward(
745        &self,
746        xs: &Tensor,
747        per_layer_input: &Tensor,
748        attention_mask: Option<&Tensor>,
749        sliding_attention_mask: Option<&Tensor>,
750        seqlen_offsets: &[usize],
751        kv_caches: &mut [KvCache],
752        flash_params: &FlashParams,
753    ) -> Result<Tensor> {
754        let predictions = self.altup.predict(xs)?;
755        let active_prediction = predictions.i(self.altup_active_idx)?;
756
757        let active_prediction_normed = self.input_layernorm.forward(&active_prediction)?;
758        let laurel_output = self.laurel.forward(&active_prediction_normed)?;
759
760        let attn = self
761            .self_attn
762            .forward(
763                &active_prediction_normed,
764                attention_mask,
765                sliding_attention_mask,
766                seqlen_offsets,
767                kv_caches,
768                flash_params,
769            )?
770            .apply(&self.post_attention_layernorm)?;
771
772        // Perform addition and scaling in float32 for precision
773        let active_pred_f32 = active_prediction.to_dtype(DType::F32)?;
774        let attn_f32 = attn.to_dtype(DType::F32)?;
775        let attn_gated_f32 = (active_pred_f32 + attn_f32)?;
776        let attn_gated = attn_gated_f32.to_dtype(active_prediction.dtype())?;
777        let attn_laurel_f32 = (attn_gated_f32 + laurel_output.to_dtype(DType::F32)?)?;
778        let attn_laurel = (attn_laurel_f32 / 2f64.sqrt())?.to_dtype(attn_gated.dtype())?;
779
780        let attn_norm = self.pre_feedforward_layernorm.forward(&attn_laurel)?;
781        let attn_ffw = self.mlp.forward(&attn_norm)?;
782        let attn_ffw_norm = self.post_feedforward_layernorm.forward(&attn_ffw)?;
783        // Perform addition in float32 for precision
784        let attn_laurel_f32 = attn_laurel.to_dtype(DType::F32)?;
785        let attn_ffw_norm_f32 = attn_ffw_norm.to_dtype(DType::F32)?;
786        let attn_ffw_laurel_gated =
787            (attn_laurel_f32 + attn_ffw_norm_f32)?.to_dtype(attn_laurel.dtype())?;
788        let mut corrected_predictions = self.altup.correct(&predictions, &attn_ffw_laurel_gated)?;
789
790        let mut first_prediction = corrected_predictions.i(self.altup_active_idx)?;
791        if self.altup_correct_scale {
792            first_prediction = self.altup.scale_corrected_output(&first_prediction)?;
793        }
794        first_prediction = self.per_layer_input_gate.forward(&first_prediction)?;
795        first_prediction = self.act.forward(&first_prediction)?;
796        // Perform multiplication in float32
797        let first_pred_f32 = first_prediction.to_dtype(DType::F32)?;
798        let per_layer_input_f32 = per_layer_input.to_dtype(DType::F32)?;
799        first_prediction =
800            (first_pred_f32 * per_layer_input_f32)?.to_dtype(first_prediction.dtype())?;
801
802        first_prediction = self.per_layer_projection.forward(&first_prediction)?;
803        first_prediction = self.post_per_layer_input_norm.forward(&first_prediction)?;
804
805        // Perform broadcast add in float32 for precision
806        let slice_f32 = corrected_predictions
807            .i((1.., .., .., ..))?
808            .to_dtype(DType::F32)?;
809        let first_pred_f32 = first_prediction.to_dtype(DType::F32)?;
810        let added = slice_f32
811            .broadcast_add(&first_pred_f32)?
812            .to_dtype(corrected_predictions.dtype())?;
813        let cp_dim0 = corrected_predictions.dim(0)?;
814        let cp_dim1 = corrected_predictions.dim(1)?;
815        let cp_dim2 = corrected_predictions.dim(2)?;
816        let cp_dim3 = corrected_predictions.dim(3)?;
817        corrected_predictions = corrected_predictions
818            .slice_assign(&[1..cp_dim0, 0..cp_dim1, 0..cp_dim2, 0..cp_dim3], &added)?;
819
820        Ok(corrected_predictions)
821    }
822}

824type MatformerSliceResult = (
825    Gemma3nTextConfig,
826    Option<Tensor>,
827    usize,
828    Option<HashMap<usize, usize>>,
829    Option<Vec<usize>>,
830);

832pub(crate) fn handle_matformer_slicing(
833    cfg: &Gemma3nTextConfig,
834    matformer_slicing_config: &Option<MatformerSliceConfig>,
835    mapper: &(dyn DeviceMapper + Send + Sync),
836) -> Result<MatformerSliceResult> {
837    match matformer_slicing_config {
838        Some(slicing_config) => {
839            let matformer_slice = slicing_config.get_slicing().ok_or_else(|| {
840                candle_core::Error::Msg(format!(
841                    "Matformer slice '{}' not found in config",
842                    slicing_config.slice_name
843                ))
844            })?;
845            let mut cfg = cfg.clone();
846
847            let layers_skipped = matformer_slice.layers_skipped.clone().unwrap_or_default();
848
849            let first_kv_shared_layer_idx = cfg.num_hidden_layers - cfg.num_kv_shared_layers;
850            let local_kv_sharing_layer_idx = first_kv_shared_layer_idx - 2;
851            let global_kv_sharing_layer_idx = first_kv_shared_layer_idx - 1;
852
853            if layers_skipped.contains(&local_kv_sharing_layer_idx)
854                || layers_skipped.contains(&global_kv_sharing_layer_idx)
855            {
856                candle_core::bail!(
857                    "Layers {} and {} are reserved.",
858                    local_kv_sharing_layer_idx,
859                    global_kv_sharing_layer_idx
860                );
861            }
862
863            let count_kv_sharing = layers_skipped.iter().filter(|x| **x >= 20).count();
864            cfg.num_kv_shared_layers -= count_kv_sharing;
865
866            let count_activation_sparsity = layers_skipped.iter().filter(|x| **x <= 9).count();
867            let final_num_layers = cfg.num_hidden_layers - layers_skipped.len();
868
869            let kept_layers_indices = (0..cfg.num_hidden_layers)
870                .filter_map(|idx| {
871                    if !layers_skipped.contains(&idx) {
872                        Some(idx as u32)
873                    } else {
874                        None
875                    }
876                })
877                .collect::<Vec<_>>();
878
879            cfg.layer_types = cfg
880                .layer_types
881                .iter()
882                .enumerate()
883                .filter_map(|(idx, layer_type)| {
884                    if !layers_skipped.contains(&idx) {
885                        Some(layer_type.clone())
886                    } else {
887                        None
888                    }
889                })
890                .collect::<Vec<_>>();
891
892            let layer_rename_map = kept_layers_indices
893                .iter()
894                .enumerate()
895                .map(|(new_idx, old_idx)| (*old_idx as usize, new_idx))
896                .collect::<HashMap<_, _>>();
897
898            let kept_layers_indices_tensor =
899                mapper.cast_nm_device(&Tensor::new(kept_layers_indices, &Device::Cpu)?, false)?;
900
901            let orig_num_hidden_layers = cfg.num_hidden_layers;
902
903            let activation_sparsity_list = [
904                vec![0.95f64; 10 - count_activation_sparsity],
905                vec![0f64; final_num_layers - 10 + count_activation_sparsity],
906            ]
907            .concat();
908            cfg.activation_sparsity_pattern = activation_sparsity_list;
909
910            cfg.num_hidden_layers = final_num_layers;
911            let orig_intermediate_size = cfg.intermediate_size.0.unwrap_left();
912            cfg.intermediate_size = IntermediateSize(Either::Right((
913                matformer_slice.ffn_hidden_dimensions.clone(),
914                orig_intermediate_size,
915            )));
916
917            Ok((
918                cfg,
919                Some(kept_layers_indices_tensor),
920                orig_num_hidden_layers,
921                Some(layer_rename_map),
922                Some(layers_skipped),
923            ))
924        }
925        None => Ok((cfg.clone(), None, cfg.num_hidden_layers, None, None)),
926    }
927}

957        attention_mechanism: AttentionImplementation,

962                quant_cfg.name(),
963                quant_cfg.get_bits_name(&vb)

965        }

972        // https://colab.research.google.com/github/google-gemini/gemma-cookbook/blob/main/Gemma/[Gemma_3n]MatFormer_Lab.ipynb
973        let (cfg, kept_layers_indices, orig_num_hidden_layers, layer_rename_map, layers_skipped) =

975                cfg,
976                &normal_loading_metadata.matformer_slicing_config,
977                &*mapper,
978            )?;

980
981        // Use float32 for embedding scale factor
982        let embed_scale = (cfg.hidden_size as f64).sqrt();
983        let embed_tokens = ScaledEmbedding::new(
984            embed_scale,
985            embedding(
986                cfg.vocab_size,
987                cfg.hidden_size,
988                mapper.set_nm_device(vb.pp("embed_tokens"), false),
989                &cfg.quantization_config,
990            )?,
991        );
992        // Use float32 for per-layer embedding scale factor
993        let per_layer_embed_scale = (cfg.hidden_size_per_layer_input as f64).sqrt();
994        // Keep embed_tokens_per_layer on CPU if not using Metal
995        let embed_tokens_per_layer_vb = if normal_loading_metadata.real_device.is_metal() {
996            mapper.set_nm_device(vb.pp("embed_tokens_per_layer"), false)
997        } else {
998            vb.pp("embed_tokens_per_layer").set_device(Device::Cpu)
999        };
1000        let mut embed_tokens_per_layer = ScaledEmbedding::new(
1001            per_layer_embed_scale,
1002            embedding(
1003                cfg.vocab_size_per_layer_input,
1004                cfg.hidden_size_per_layer_input * orig_num_hidden_layers,
1005                embed_tokens_per_layer_vb,
1006                &cfg.quantization_config,
1007            )?,
1008        );
1009        if let Some(kept_layers_indices) = &kept_layers_indices {
1010            let embedding = embed_tokens_per_layer.embedding.clone();
1011            let embedding_reshaped = embedding.reshape((
1012                embedding.dim(0)?,
1013                orig_num_hidden_layers,
1014                embedding.dim(1)? / orig_num_hidden_layers,
1015            ))?;
1016
1017            embed_tokens_per_layer.embedding = embedding_reshaped
1018                .index_select(kept_layers_indices, 1)?
1019                .reshape((embedding_reshaped.dim(0)?, ()))?
1020                .contiguous()?;
1021        }
1022
1023        let mut global_ropes = HashMap::new();
1024        for layer_idx in 0..orig_num_hidden_layers {
1025            let layer_idx = if let Some(layer_rename_map) = &layer_rename_map {
1026                if layers_skipped.as_ref().unwrap().contains(&layer_idx) {
1027                    continue;
1028                }
1029                layer_rename_map[&layer_idx]
1030            } else {
1031                layer_idx
1032            };
1033            let device = mapper
1034                .device_for(layer_idx, false)
1035                .unwrap_or(&normal_loading_metadata.real_device);
1036            global_ropes.insert(
1037                device.location(),
1038                Arc::new(Gemma3nRotaryEmbedding::new(
1039                    is_gptx,
1040                    vb.dtype(),
1041                    cfg,
1042                    device,
1043                )?),
1044            );
1045        }
1046
1047        let mut local_ropes = HashMap::new();

1049            let layer_idx = if let Some(layer_rename_map) = &layer_rename_map {
1050                if layers_skipped.as_ref().unwrap().contains(&layer_idx) {
1051                    continue;
1052                }
1053                layer_rename_map[&layer_idx]
1054            } else {
1055                layer_idx

1058                .device_for(layer_idx, false)
1059                .unwrap_or(&normal_loading_metadata.real_device);

1061                device.location(),
1062                Arc::new(RotaryEmbedding::new(
1063                    cfg.rope_local_base_freq as f32,
1064                    cfg.head_dim,
1065                    cfg.max_position_embeddings,
1066                    device,
1067                    is_gptx,
1068                    vb.dtype(),

1077            "Loading repeating layers",

1082            !layers_skipped
1083                .as_ref()
1084                .is_some_and(|layers_skipped| layers_skipped.contains(layer_idx))
1085        })
1086        .map(|layer_idx| {
1087            let layer_idx_effective = if let Some(layer_rename_map) = &layer_rename_map {
1088                layer_rename_map[&layer_idx]
1089            } else {
1090                layer_idx
1091            };

1093                .device_for(layer_idx, false)
1094                .unwrap_or(&normal_loading_metadata.real_device);
1095            let rotary_emb_global = global_ropes
1096                .get(&device.location())
1097                .expect("No RoPE for device location!")
1098                .clone();
1099            let rotary_emb_local = local_ropes
1100                .get(&device.location())
1101                .expect("No RoPE for device location!")
1102                .clone();
1103            let comm = mapper.get_comm_for(layer_idx)?;
1104            DecoderLayer::new(
1105                rotary_emb_global,
1106                rotary_emb_local,
1107                cfg,
1108                vb_l.pp(layer_idx),
1109                &*mapper,
1110                layer_idx_effective,
1111                normal_loading_metadata.loading_isq,
1112                &comm,
1113            )
1114        })
1115        .collect::<Result<Vec<_>>>()?;
1116
1117        let norm = RmsNorm::new_gemma_3n(
1118            cfg.hidden_size,
1119            cfg.rms_norm_eps,
1120            true,
1121            mapper.set_nm_device(vb.pp("norm"), false),
1122        )?;
1123
1124        let lm_head = if !cfg.tie_word_embeddings {
1125            ReplicatedLayer::new(

1129                false,
1130                mapper.set_nm_device(vb.pp("lm_head"), normal_loading_metadata.loading_isq),

1132        } else {

1134                mapper.cast_nm_device(
1135                    embed_tokens.embeddings(),
1136                    normal_loading_metadata.loading_isq,
1137                )?,
1138                None,
1139            ))?
1140        };

1143            cfg.hidden_size,
1144            orig_num_hidden_layers * cfg.hidden_size_per_layer_input,

1146            orig_num_hidden_layers,
1147            &cfg.quantization_config,
1148            false,

1150                vb.pp("per_layer_model_projection"),
1151                normal_loading_metadata.loading_isq,

1158            mapper.set_nm_device(vb.pp("per_layer_projection_norm"), false),

1180                    vb.pp("altup_unembed_projections").pp(i - 1),
1181                    normal_loading_metadata.loading_isq,
1182                ),
1183            )?);
1184        }
1185
1186        let cache_types = (0..cfg.num_hidden_layers)
1187            .map(|layer_idx| {
1188                is_sliding!(layer_idx, cfg)
1189                    .then(|| NormalCacheType::SlidingWindow {
1190                        window: cfg.sliding_window,
1191                    })
1192                    .unwrap_or(NormalCacheType::Normal {
1193                        max_seq_len: cfg.max_position_embeddings,
1194                    })
1195            })
1196            .collect::<Vec<_>>();
1197        Ok(Self {
1198            embed_tokens,
1199            embed_tokens_per_layer,
1200            layers,
1201            norm,
1202            lm_head,
1203            device: normal_loading_metadata.real_device,
1204            cache: EitherCache::Normal(NormalCache::from_types(cache_types)),
1205            max_seq_len: cfg.max_position_embeddings,
1206            sliding_window: cfg.sliding_window,
1207            cfg: ModelConfigMetadata {
1208                max_seq_len: cfg.max_position_embeddings,
1209                num_layers: cfg.num_hidden_layers,

1235    fn get_per_layer_inputs(&self, input_ids: &Tensor) -> Result<Tensor> {
1236        let shape = [
1237            input_ids.dims().to_vec(),
1238            vec![self.layers.len(), self.hidden_size_per_layer_input],
1239        ]
1240        .concat();
1241        // Cast to float32 for better precision
1242        let input_ids_device = input_ids.device();
1243
1244        // Only cast to CPU if not using Metal
1245        let per_layer_embeds = if input_ids_device.is_metal() {
1246            // On Metal, use input_ids directly
1247            self.embed_tokens_per_layer.forward(input_ids)?
1248        } else {
1249            // On non-Metal devices, cast input to CPU for embedding computation
1250            let input_ids_cpu = input_ids.to_device(&Device::Cpu)?;
1251            self.embed_tokens_per_layer.forward(&input_ids_cpu)?
1252        };
1253
1254        let original_dtype = per_layer_embeds.dtype();
1255        // Move result back to original device after computation (if needed)
1256        let result = per_layer_embeds
1257            .to_dtype(DType::F32)?
1258            .reshape(shape)?
1259            .to_dtype(original_dtype)?;
1260
1261        // Only move back to device if we computed on CPU
1262        if !input_ids_device.is_metal() && input_ids_device.location() != Device::Cpu.location() {
1263            result.to_device(input_ids_device)
1264        } else {
1265            Ok(result)
1266        }
1267    }
1268    fn project_per_layer_inputs(
1269        &self,
1270        xs: &Tensor,
1271        per_layer_inputs: Option<Tensor>,
1272    ) -> Result<Tensor> {
1273        // Cast to float32 for per-layer projection scaling
1274        let mut per_layer_projection = self.per_layer_model_projection.forward_autocast(xs)?;
1275        let original_dtype = per_layer_projection.dtype();
1276        let per_layer_projection_f32 = per_layer_projection.to_dtype(DType::F32)?;
1277        per_layer_projection = (per_layer_projection_f32 * self.per_layer_projection_scale)?;
1278        let shape = [
1279            xs.dims()[..xs.dims().len() - 1].to_vec(),
1280            vec![self.layers.len(), self.hidden_size_per_layer_input],
1281        ]
1282        .concat();
1283        per_layer_projection = per_layer_projection
1284            .reshape(shape)?
1285            .to_dtype(original_dtype)?; // Convert back to model dtype before RMSNorm
1286        per_layer_projection = self
1287            .per_layer_projection_norm
1288            .forward(&per_layer_projection)?;
1289
1290        let Some(mut per_layer_inputs) = per_layer_inputs else {
1291            return Ok(per_layer_projection.clone());
1292        };
1293
1294        if per_layer_projection.shape() != per_layer_inputs.shape() {
1295            // per-layer inputs are sometimes padded with zeros, slice the relevant embeddings.
1296            per_layer_inputs = per_layer_inputs.narrow(D::Minus2, 0, self.layers.len())?;
1297        }
1298        // Perform addition and scaling in float32 for better precision
1299        let per_layer_projection_f32 = per_layer_projection.to_dtype(DType::F32)?;
1300        let per_layer_inputs_f32 = per_layer_inputs.to_dtype(DType::F32)?;
1301        let result_f32 = (per_layer_projection_f32 + per_layer_inputs_f32)?;
1302        (result_f32 * self.per_layer_input_scale)?.to_dtype(per_layer_projection.dtype())
1303    }

1305    pub fn forward_embeds(
1306        &self,
1307        input_ids: &Tensor,
1308        ple_input_ids: &Tensor,
1309        mut xs: Tensor,
1310        seqlen_offsets: &[usize],
1311        context_lens: Vec<(usize, usize)>,
1312        flash_params: &FlashParams,
1313    ) -> Result<Tensor> {
1314        let per_layer_inputs = Some(self.get_per_layer_inputs(ple_input_ids)?);
1315        let per_layer_inputs = self.project_per_layer_inputs(&xs, per_layer_inputs)?;
1316        // Cast per_layer_inputs back to model dtype after float32 operations
1317        let per_layer_inputs = per_layer_inputs.to_dtype(xs.dtype())?;
1318
1319        let cache = &mut self.cache.normal().0;
1320        let attention_mask = CausalMasker.make_causal_mask_matrix(
1321            input_ids,
1322            &*cache,
1323            xs.dtype(),
1324            self.cfg.num_attn_heads,
1325        )?;
1326        let sliding_attention_mask = CausalMasker.make_sliding_window_causal_mask_matrix(
1327            input_ids,
1328            &*cache,
1329            Some(self.sliding_window),
1330            xs.dtype(),
1331            self.cfg.num_attn_heads,
1332        )?;
1333
1334        // Already using float32 for magnitude calculations
1335        let target_magnitude = xs
1336            .to_dtype(DType::F32)?
1337            .sqr()?
1338            .mean_keepdim(D::Minus1)?
1339            .sqrt()?;
1340        let eps = Tensor::new(&[EPS as f32], target_magnitude.device())?;
1341
1342        let mut temp_hidden_states = vec![xs.clone()];
1343        for altup_proj in &self.altup_projections {
1344            let altup_proj = altup_proj.forward_autocast(&xs)?;
1345            let new_magnitude = altup_proj
1346                .to_dtype(DType::F32)?
1347                .sqr()?
1348                .mean_keepdim(D::Minus1)?
1349                .sqrt()?;
1350
1351            let current_hidden_state = altup_proj
1352                .to_dtype(DType::F32)?
1353                .broadcast_mul(
1354                    &target_magnitude.broadcast_div(&new_magnitude.broadcast_maximum(&eps)?)?,
1355                )?
1356                .to_dtype(altup_proj.dtype())?;
1357            temp_hidden_states.push(current_hidden_state);
1358        }
1359        xs = Tensor::stack(&temp_hidden_states, 0)?;
1360
1361        for (i, layer) in self.layers.iter().enumerate() {
1362            let per_layer_input = per_layer_inputs.i((.., .., i, ..))?;
1363            xs = self.mapper.map(xs, i)?;
1364            xs = layer.forward(
1365                &xs,
1366                &per_layer_input.to_device(xs.device())?,
1367                attention_mask
1368                    .as_ref()
1369                    .map(|m| m.to_device(xs.device()).unwrap())
1370                    .as_ref(),
1371                sliding_attention_mask
1372                    .as_ref()
1373                    .map(|m| m.to_device(xs.device()).unwrap())
1374                    .as_ref(),
1375                seqlen_offsets,
1376                &mut *cache,
1377                flash_params,
1378            )?;
1379        }
1380        xs = xs.to_device(&self.device)?;
1381
1382        let target_magnitude = xs
1383            .i(0)?
1384            .to_dtype(DType::F32)?
1385            .sqr()?
1386            .mean_keepdim(D::Minus1)?
1387            .sqrt()?;
1388
1389        let mut temp_hidden_states = vec![xs.i(0)?];
1390        for (i, altup_proj) in self.altup_unembed_projections.iter().enumerate() {
1391            let altup_proj = altup_proj.forward_autocast(&xs.i(i + 1)?)?;
1392            let new_magnitude = altup_proj
1393                .to_dtype(DType::F32)?
1394                .sqr()?
1395                .mean_keepdim(D::Minus1)?
1396                .sqrt()?;
1397
1398            let current_hidden_state = altup_proj
1399                .to_dtype(DType::F32)?
1400                .broadcast_mul(
1401                    &target_magnitude.broadcast_div(&new_magnitude.broadcast_maximum(&eps)?)?,
1402                )?
1403                .to_dtype(altup_proj.dtype())?;
1404            temp_hidden_states.push(current_hidden_state);
1405        }
1406        // Perform mean operation in float32 for better precision
1407        let stacked = Tensor::stack(&temp_hidden_states, 0)?;
1408        let stacked_f32 = stacked.to_dtype(DType::F32)?;
1409        xs = stacked_f32.mean(0)?.to_dtype(stacked.dtype())?;
1410
1411        xs = xs.apply(&self.norm)?;
1412        if let Some(t) = self.lm_head.quantized_act_type() {
1413            xs = xs.to_dtype(t)?;
1414        }
1415
1416        xs = MatMul.qmethod_matmul(&xs, &*self.lm_head)?;
1417
1418        if let Some(final_logit_softcapping) = self.final_logit_softcapping {
1419            // Perform logit softcapping in float32 for precision
1420            let xs_f32 = xs.to_dtype(DType::F32)?;
1421            let capped = (xs_f32 / final_logit_softcapping)?;
1422            let tanh_capped = capped.tanh()?;
1423            xs = (tanh_capped * final_logit_softcapping)?.to_dtype(xs.dtype())?;
1424        }
1425
1426        extract_logits(&xs, context_lens)
1427    }
