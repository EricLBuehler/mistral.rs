46impl PrefixCacheManagerV2 {
47    pub fn new(n_on_device: usize, no_prefix_cache: bool) -> Self {
48        if !no_prefix_cache {
49            info!("PrefixCacherV2 (RadixAttention) enabled - expect higher multi-turn throughput.");
50        }
51        // Pre‑seed our hash map with a small capacity (common branching factor)
52        let root = Arc::new(RwLock::new(TrieNode {
53            children: AHashMap::with_capacity_and_hasher(8, Default::default()),
54            cache: None,
55            prev: None,
56            next: None,
57        }));
58        PrefixCacheManagerV2 {
59            root,
60            n_on_device,
61            no_prefix_cache,
62            lru_head: None,
63            lru_tail: None,
64            current_on_device: 0,
65        }
66    }

68    /// Bump a node to the most‑recently‑used tail
69    fn touch_node(&mut self, node: &NodeRef) {
70        // Take a write lock to unlink & relink
71        let mut nb: RwLockWriteGuard<'_, TrieNode> = node.write();
72        // Unlink from list
73        if let Some(prev_w) = nb.prev.take() {
74            if let Some(prev) = prev_w.upgrade() {
75                let mut prev_nb = prev.write();
76                prev_nb.next = nb.next.clone();
77            }
78        } else {
79            // was head
80            self.lru_head = nb.next.clone();
81        }
82        if let Some(next) = nb.next.take() {
83            let mut next_nb = next.write();
84            next_nb.prev = nb.prev.clone();
85        } else {
86            // was tail
87            self.lru_tail = nb.prev.as_ref().and_then(|w| w.upgrade());
88        }
89        drop(nb);
90
91        // Link at tail
92        if let Some(old_tail) = self.lru_tail.take() {
93            {
94                let mut tail_nb = old_tail.write();
95                tail_nb.next = Some(Arc::clone(node));
96            }
97            let mut me_nb = node.write();
98            me_nb.prev = Some(Arc::downgrade(&old_tail));
99            me_nb.next = None;
100        } else {
101            // empty list
102            self.lru_head = Some(Arc::clone(node));
103            let mut me_nb = node.write();
104            me_nb.prev = None;
105            me_nb.next = None;
106        }
107        self.lru_tail = Some(Arc::clone(node));
108    }

110    /// Insert or update the cache for a full token sequence
111    pub fn add_sequence(&mut self, seq: &mut Sequence) {
112        if self.no_prefix_cache {
113            return;
114        }
115        let data = seq.normal_cache().to_vec();
116        let img_hashes = seq.image_hashes().map(|v| v.to_vec());
117        let mut node = Arc::clone(&self.root);
118
119        // Walk/build trie
120        for &tok in seq.get_toks() {
121            let next = {
122                let mut nb = node.write();
123                nb.children
124                    .entry(tok)
125                    .or_insert_with(|| {
126                        Arc::new(RwLock::new(TrieNode {
127                            children: AHashMap::with_capacity_and_hasher(8, Default::default()),
128                            cache: None,
129                            prev: None,
130                            next: None,
131                        }))
132                    })
133                    .clone()
134            };
135            node = next;
136        }
137
138        {
139            let mut nb = node.write();
140            if nb.cache.is_none() {
141                self.current_on_device += 1;
142            }
143            nb.cache = Some(CacheElement {
144                cache: data,
145                image_hashes: img_hashes,
146            });
147        }
148
149        self.touch_node(&node);
150        // Enforce capacity
151        let _ = self.evict_caches();
152    }

154    /// Evict least‑recently used entries until under capacity
155    pub fn evict_caches(&mut self) -> Result<usize> {
156        if self.no_prefix_cache {
157            return Ok(0);
158        }
159        let mut evicted = 0;
160        while self.current_on_device > self.n_on_device {
161            if let Some(head) = self.lru_head.take() {
162                {
163                    let mut hb = head.write();
164                    if hb.cache.take().is_some() {
165                        evicted += 1;
166                        self.current_on_device -= 1;
167                    }
168                    // unlink head
169                    self.lru_head = hb.next.clone();
170                }
171                if let Some(new_head) = &self.lru_head {
172                    let mut hnb = new_head.write();
173                    hnb.prev = None;
174                } else {
175                    self.lru_tail = None;
176                }
177            } else {
178                break;
179            }
180        }
181        Ok(evicted)
182    }

184    pub fn evict_all_caches(&mut self) -> Result<usize> {
185        let count = self.current_on_device;
186        {
187            let mut root_nb = self.root.write();
188            root_nb.children.clear();
189            root_nb.cache = None;
190        }
191        self.lru_head = None;
192        self.lru_tail = None;
193        self.current_on_device = 0;
194        Ok(count)
195    }

197    /// Lookup the longest‐matching prefix cache for `toks`
198    pub fn search_for_matching_cache(
199        &mut self,
200        toks: &[u32],
201        image_hashes: Option<&[u64]>,
202        _contains_images: bool,
203    ) -> Result<Option<MatchingCache>> {
204        if self.no_prefix_cache || toks.is_empty() {
205            return Ok(None);
206        }
207
208        let mut node = Arc::clone(&self.root);
209        let mut best: Option<NodeRef> = None;
210        let mut match_len = 0;
211        let mut img_match = 0;
212
213        // Fast, shared‐read walk
214        for (i, &tok) in toks.iter().enumerate() {
215            // Acquire *shared* read lock just to access children map
216            let next_opt = {
217                let rb: RwLockReadGuard<'_, TrieNode> = node.read();
218                rb.children.get(&tok).cloned()
219            };
220            if let Some(child) = next_opt {
221                node = child;
222                let rb = node.read();
223                if rb.cache.is_some() {
224                    best = Some(Arc::clone(&node));
225                    match_len = i + 1;
226                    if let (Some(inp), Some(cached)) =
227                        (image_hashes, &rb.cache.as_ref().unwrap().image_hashes)
228                    {
229                        img_match = inp.iter().zip(cached).take_while(|(a, b)| a == b).count();
230                    }
231                }
232            } else {
233                break;
234            }
235        }
236
237        if let Some(matched_node) = best {
238            // Bump in LRU
239            self.touch_node(&matched_node);
240
241            // Prepare result
242            let rb = matched_node.read();
243            let mut elem = rb.cache.clone().unwrap();
244            for layer in elem.cache.iter_mut().flatten() {
245                layer.set_len(match_len)?;
246            }
247            let keep_imgs = image_hashes.map_or(0, |inp| inp.len().saturating_sub(img_match));
248
249            Ok(Some(MatchingCache {
250                normal: elem.cache,
251                images_to_keep: keep_imgs,
252                toks: toks[match_len..].to_vec(),
253                offset: match_len,
254            }))
255        } else {
256            Ok(None)
257        }
258    }
