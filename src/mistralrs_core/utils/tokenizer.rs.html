1use std::{collections::HashMap, path::Path};

3use crate::tokenizer::TokenizerImpl;
4use anyhow::Result;
5use serde::Deserialize;
6use serde_json::Value;
7use tokenizers::{tokenizer, Tokenizer};
8
9#[derive(Deserialize)]
10struct AddedToken {
11    id: usize,
12    content: String,
13}
14
15/// May fix the tokenizer according to: https://gist.github.com/jneuff/682d47b786329f19291d166957b3274a
16pub(crate) fn get_tokenizer<P: AsRef<Path> + Clone>(
17    p: P,
18    processor_added_tokens: Option<&[&str]>,
19) -> Result<TokenizerImpl> {
20    let mut tokenizer = {
21        let raw = std::fs::read(p.clone()).map_err(anyhow::Error::msg)?;
22        let mut tokenizer: Value = serde_json::from_slice(&raw).unwrap();
23        let added_tokens: Vec<AddedToken> =
24            serde_json::from_value(tokenizer["added_tokens"].clone()).unwrap();
25        let vocab: HashMap<String, usize> =
26            serde_json::from_value(tokenizer["model"]["vocab"].clone()).unwrap();
27        for token in added_tokens {
28            if !vocab.contains_key(&token.content) {
29                tokenizer["model"]["vocab"]
30                    .as_object_mut()
31                    .unwrap()
32                    .insert(token.content, token.id.into())
33                    .ok_or(())
34                    .unwrap_err();
35            }
36        }
37        let raw_fixed = serde_json::to_vec_pretty(&tokenizer).unwrap();
38        Tokenizer::from_bytes(&raw_fixed).map_err(anyhow::Error::msg)?
39    };
40    if let Some(added_tokens) = processor_added_tokens {
41        tokenizer.add_special_tokens(
42            &added_tokens
43                .iter()
44                .map(|x| tokenizer::AddedToken::from(x.to_string(), true))
45                .collect::<Vec<_>>(),
46        );
47    }
48    Ok(TokenizerImpl::Tokenizer {
49        tokenizer,
50        bos: None,
51        eos: None,
52        unk: None,
53    })
54}
