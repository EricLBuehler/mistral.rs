28pub enum StopReason {
29    Eos,
30    StopTok(u32),
31    Length(usize),
32    ModelLength(usize),
33    StopString {
34        stop_string_idx: usize,
35        completion_bytes_pos: usize,
36    },

41impl Display for StopReason {
42    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
43        match self {
44            StopReason::Eos => write!(f, "stop"),
45            StopReason::Length(_) | StopReason::ModelLength(_) => write!(f, "length"),
46            StopReason::StopTok(_) | StopReason::StopString { .. } => write!(f, "stop"),
47            StopReason::Canceled => write!(f, "canceled"),
48            StopReason::GeneratedImage => write!(f, "generated-image"),
49        }
50    }

80macro_rules! blocks_to_add_new_tok {
81    ($logical_token_blocks:expr) => {{
82        let last = $logical_token_blocks.last();
83        if !last.is_some_and(|last| last.is_full() || last.is_empty()) {
84            // If we have space
85            0
86        } else {
87            1
88        }
89    }};

92impl SequenceCustomMetadata {
93    fn append_token_to_blocks(&mut self, tok: usize) {
94        match self {
95            Self::PagedAttention {
96                logical_token_blocks,
97                block_size,
98            } => {
99                let last = logical_token_blocks.last_mut();
100                match last {
101                    Some(last) => {
102                        last.append_token_id(tok);
103                    }
104                    None => {
105                        logical_token_blocks.push(LogicalTokenBlock::new(*block_size));
106                        logical_token_blocks
107                            .last_mut()
108                            .unwrap()
109                            .append_token_id(tok);
110                    }
111                }
112                if logical_token_blocks.last().as_ref().unwrap().is_full() {
113                    logical_token_blocks.push(LogicalTokenBlock::new(*block_size));
114                }
115            }
116            Self::None => (),
117        }
118    }
119
120    fn pop_token_from_blocks(&mut self) {
121        match self {
122            Self::PagedAttention {
123                logical_token_blocks,
124                block_size: _,
125            } => {
126                let last = logical_token_blocks.last_mut().unwrap();
127                last.pop_token();
128            }
129            Self::None => (),
130        }
131    }
132
133    fn append_tokens_to_blocks(&mut self, toks: Vec<usize>) {
134        for tok in toks {
135            self.append_token_to_blocks(tok);
136        }
137    }
138
139    fn remove_tokens_from_blocks(&mut self, n: usize) {
140        for _ in 0..n {
141            self.pop_token_from_blocks();
142        }
143    }
144}

400    /// This is the number of tokens. If the KV cache is Some, then it will use that.
401    pub fn len(&self) -> usize {
402        if let Some(toks) = &self.prefill_prompt_toks {
403            return toks.len();
404        }
405        if self.is_tmp {
406            return self.tokens.len();
407        }
408        // Use xlora cache first because of non granular
409        if self.xlora_cache.as_ref().is_some_and(|c| c[0].is_some()) {
410            self.xlora_cache.as_ref().unwrap()[0]
411                .as_ref()
412                .unwrap()
413                .0
414                .dims()[2]
415                + 1
416        } else if let Some((_, x)) = &self.cache[0] {
417            x.dims()[2] + 1
418        } else {
419            self.tokens.len()
420        }

574    pub fn add_token(
575        &mut self,
576        tok: Logprobs,
577        completion_bytes: Vec<u8>,
578        is_done: &Option<StopReason>,
579    ) {
580        let stopped_by_token = matches!(
581            is_done,
582            Some(StopReason::Eos) | Some(StopReason::StopTok(_))
583        );
584        if !stopped_by_token {
585            // Completion bytes is used to check for stop strings, and as the response buffer.
586            // We don't need to add stop tokens to the completion bytes to check for stop strings.
587            // And by not adding it here, we can avoid having to delete these tokens from the output.
588            self.completion_bytes.extend_from_slice(&completion_bytes);
589            self.last_completion_bytes_len = completion_bytes.len();
590        }
591        self.last_logprob = tok.logprob;
592        self.last_is_done = *is_done;
593
594        self.custom_metadata
595            .append_token_to_blocks(tok.token as usize);
596
597        self.cumulative_logprob += tok.logprob;
598        self.tokens.push(tok.token);
599        self.logprobs.push(tok);
600        self.reset_prefill_toks();
601    }

622    pub fn is_done(
623        &self,
624        tok: u32,
625        eos_tok: Option<&[u32]>,
626        max_model_len: usize,
627    ) -> Option<StopReason> {
628        let is_eos = match eos_tok {
629            Some(eos_tok) => eos_tok.iter().any(|t| *t == tok),
630            None => false,
631        };
632        if is_eos {
633            Some(StopReason::Eos)
634        } else if matches!(
635            &*self.state.read().unwrap(),
636            SequenceState::Done(StopReason::Canceled)
637        ) {
638            Some(StopReason::Canceled)
639        } else if self.stop_tokens.contains(&tok) {
640            Some(StopReason::StopTok(tok))
641        } else if self.max_len.is_some()
642            && self.tokens.len().saturating_sub(self.prompt_len) == self.max_len.unwrap()
643        {
644            // add_token was already called
645            Some(StopReason::Length(self.max_len.unwrap()))
646        } else if self.tokens.len().saturating_sub(self.prompt_len) == max_model_len {
647            Some(StopReason::ModelLength(max_model_len))
648        } else {
649            if !self.stop_strings.is_empty() {
650                for (idx, s) in self.stop_strings.iter().enumerate() {
651                    if let Some(pos) = galil_seiferas::gs_find(&self.completion_bytes, s.as_bytes())
652                    {
653                        return Some(StopReason::StopString {
654                            stop_string_idx: idx,
655                            completion_bytes_pos: pos,
656                        });
657                    }
658                }
659            }
660            None
661        }
662    }

680    /// Returns the delta between the last two decoded sequences
681    pub fn get_delta(
682        &mut self,
683    ) -> Result<Option<String>, Box<dyn std::error::Error + Send + Sync>> {
684        let new_decoded = self.peek_delta();
685        if matches!(new_decoded, Ok(Some(_))) {
686            self.stream_idx = self.completion_bytes.len();
687        }
688        new_decoded
689    }

691    /// Peeks at the delta between the last two decoded sequences, but does not advance the stream index.
692    pub fn peek_delta(&self) -> Result<Option<String>, Box<dyn std::error::Error + Send + Sync>> {
693        let is_first = self.stream_idx == 0;
694        let new_decoded = String::from_utf8_lossy(&self.completion_bytes[self.stream_idx..]);
695        // Check if the sequence ends with valid utf8, if not skip it as it probably is a multi token sequence
696        if new_decoded.ends_with('') {
697            return Ok(None);
698        }
699
700        // The first token usually starts with a space. We don't want to add that to the delta.
701        // Since we're using the completion_bytes, we need to take care of that ourselves.
702        // Had we used HF's Tokenizer, it would have taken care of that for us.
703        if is_first {
704            return Ok(Some(new_decoded.trim_start().to_string()));
705        }
706        Ok(Some(new_decoded.to_string()))
707    }

717    fn update_time_info(&self) {
718        let now = SystemTime::now()
719            .duration_since(UNIX_EPOCH)
720            .expect("Time travel has occurred!")
721            .as_millis();
722
723        if let Some(ts) = self.prompt_timestamp {
724            get_mut_group!(self).total_completion_time += now - ts;
725            get_mut_group!(self).total_prompt_time += ts - self.timestamp;
726        }
727
728        get_mut_group!(self).total_time += now - self.timestamp;
729
730        get_mut_group!(self).total_prompt_toks = self.prompt_len;
731        get_mut_group!(self).total_toks = self.len();
732    }

816pub struct SequenceGroup {
817    n_choices: usize, // The target number of choices to return. Can be decreased if an error is thrown.
818    best_of: Option<usize>, // Top n seqs based on cumulative logprobs.
819    pub total_prompt_toks: usize,
820    pub total_toks: usize,
821    pub total_prompt_time: u128,
822    pub total_time: u128,
823    pub total_completion_time: u128,
824    choices: Vec<Choice>,
825    image_choices: Vec<ImageChoice>,
826    raw_choices: Vec<(Vec<Tensor>, Vec<u32>)>,
827    completion_choices: Vec<(f32, CompletionChoice)>,
828    pub chat_streaming_chunks: Vec<ChunkChoice>,
829    pub completion_streaming_chunks: Vec<CompletionChunkChoice>,
830    pub is_streaming: bool,
831    pub is_chat: bool,
832}

840    ) -> Self {

864    /// This may apply the best_of.
865    pub fn get_completion_choices(&self) -> Vec<CompletionChoice> {
866        if let Some(best_of) = self.best_of {
867            let mut choices = self.completion_choices.clone();
868            // Sort by descending logprobs
869            choices.sort_by(|a, b| b.0.partial_cmp(&a.0).expect("No ordering."));
870            choices
871                .into_iter()
872                .take(best_of)
873                .map(|(_, x)| x)
874                .collect::<Vec<_>>()
875        } else {
876            self.completion_choices
877                .clone()
878                .into_iter()
879                .map(|(_, x)| x)
880                .collect::<Vec<_>>()
881        }
882    }

888    pub fn get_usage(&self) -> Usage {
889        #[allow(clippy::cast_precision_loss)]
890        Usage {
891            completion_tokens: self.total_toks - self.total_prompt_toks,
892            prompt_tokens: self.total_prompt_toks,
893            total_tokens: self.total_toks,
894            avg_tok_per_sec: (self.total_toks as f32 / self.total_time as f32) * 1000.,
895            avg_prompt_tok_per_sec: (self.total_prompt_toks as f32 / self.total_prompt_time as f32)
896                * 1000.,
897            avg_compl_tok_per_sec: ((self.total_toks - self.total_prompt_toks) as f32
898                / self.total_completion_time as f32)
899                * 1000.,
900            total_time_sec: self.total_time as f32 / 1000.,
901            total_completion_time_sec: self.total_completion_time as f32 / 1000.,
902            total_prompt_time_sec: self.total_prompt_time as f32 / 1000.,
903        }
904    }

948    pub async fn maybe_send_streaming_response(
949        &mut self,
950        seq: &Sequence,
951        model: String,
952        usage_opt: Option<Usage>,
953    ) -> Result<(), Box<SendError<Response>>> {
954        if self.chat_streaming_chunks.len() == self.n_choices && self.is_streaming {
955            let mut swap_streaming_chunks = vec![];
956
957            std::mem::swap(&mut swap_streaming_chunks, &mut self.chat_streaming_chunks);
958
959            seq.responder()
960                .send(Response::Chunk(ChatCompletionChunkResponse {
961                    id: seq.id.to_string(),
962                    choices: swap_streaming_chunks,
963                    created: seq.timestamp,
964                    model: model.clone(),
965                    system_fingerprint: SYSTEM_FINGERPRINT.to_string(),
966                    object: "chat.completion.chunk".to_string(),
967                    usage: usage_opt,
968                }))
969                .await?;
970        } else if self.completion_streaming_chunks.len() == self.n_choices && self.is_streaming {
971            let mut swap_streaming_chunks = vec![];
972
973            std::mem::swap(
974                &mut swap_streaming_chunks,
975                &mut self.completion_streaming_chunks,
976            );
977
978            seq.responder()
979                .send(Response::CompletionChunk(CompletionChunkResponse {
980                    id: seq.id.to_string(),
981                    choices: swap_streaming_chunks,
982                    created: seq.timestamp,
983                    model: model.clone(),
984                    system_fingerprint: SYSTEM_FINGERPRINT.to_string(),
985                    object: "text_completion".to_string(),
986                }))
987                .await?;
988        }
989        Ok(())
990    }
