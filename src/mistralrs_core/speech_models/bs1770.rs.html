10//! Loudness analysis conforming to [ITU-R BS.1770-4][bs17704].
11//!
12//! This library offers the building blocks to perform BS.1770 loudness
13//! measurements, but you need to put the pieces together yourself.
14//!
15//! [bs17704]: https://www.itu.int/rec/R-REC-BS.1770-4-201510-I/en
16//!
17//! # Stereo integrated loudness example
18//!
19//! ```ignore
20//! # fn load_stereo_audio() -> [Vec<i16>; 2] {
21//! #     [vec![0; 48_000], vec![0; 48_000]]
22//! # }
23//! #
24//! let sample_rate_hz = 44_100;
25//! let bits_per_sample = 16;
26//! let channel_samples: [Vec<i16>; 2] = load_stereo_audio();
27//!
28//! // When converting integer samples to float, note that the maximum amplitude
29//! // is `1 << (bits_per_sample - 1)`, one bit is the sign bit.
30//! let normalizer = 1.0 / (1_u64 << (bits_per_sample - 1)) as f32;
31//!
32//! let channel_power: Vec<_> = channel_samples.iter().map(|samples| {
33//!     let mut meter = bs1770::ChannelLoudnessMeter::new(sample_rate_hz);
34//!     meter.push(samples.iter().map(|&s| s as f32 * normalizer));
35//!     meter.into_100ms_windows()
36//! }).collect();
37//!
38//! let stereo_power = bs1770::reduce_stereo(
39//!     channel_power[0].as_ref(),
40//!     channel_power[1].as_ref(),
41//! );
42//!
43//! let gated_power = bs1770::gated_mean(
44//!     stereo_power.as_ref()
45//! ).unwrap_or(bs1770::Power(0.0));
46//! println!("Integrated loudness: {:.1} LUFS", gated_power.loudness_lkfs());
47//! ```

51/// Coefficients for a 2nd-degree infinite impulse response filter.
52///
53/// Coefficient a0 is implicitly 1.0.
54#[derive(Clone)]
55struct Filter {
56    a1: f32,
57    a2: f32,
58    b0: f32,
59    b1: f32,
60    b2: f32,
61
62    // The past two input and output samples.
63    x1: f32,
64    x2: f32,
65    y1: f32,
66    y2: f32,
67}
68
69impl Filter {
70    /// Stage 1 of th BS.1770-4 pre-filter.
71    pub fn high_shelf(sample_rate_hz: f32) -> Filter {
72        // Coefficients taken from https://github.com/csteinmetz1/pyloudnorm/blob/
73        // 6baa64d59b7794bc812e124438692e7fd2e65c0c/pyloudnorm/meter.py#L135-L136.
74        let gain_db = 3.999_843_8;
75        let q = 0.707_175_25;
76        let center_hz = 1_681.974_5;
77
78        // Formula taken from https://github.com/csteinmetz1/pyloudnorm/blob/
79        // 6baa64d59b7794bc812e124438692e7fd2e65c0c/pyloudnorm/iirfilter.py#L134-L143.
80        let k = (f32::consts::PI * center_hz / sample_rate_hz).tan();
81        let vh = 10.0_f32.powf(gain_db / 20.0);
82        let vb = vh.powf(0.499_666_78);
83        let a0 = 1.0 + k / q + k * k;
84        Filter {
85            b0: (vh + vb * k / q + k * k) / a0,
86            b1: 2.0 * (k * k - vh) / a0,
87            b2: (vh - vb * k / q + k * k) / a0,
88            a1: 2.0 * (k * k - 1.0) / a0,
89            a2: (1.0 - k / q + k * k) / a0,
90
91            x1: 0.0,
92            x2: 0.0,
93            y1: 0.0,
94            y2: 0.0,
95        }
96    }
97
98    /// Stage 2 of th BS.1770-4 pre-filter.
99    pub fn high_pass(sample_rate_hz: f32) -> Filter {
100        // Coefficients taken from https://github.com/csteinmetz1/pyloudnorm/blob/
101        // 6baa64d59b7794bc812e124438692e7fd2e65c0c/pyloudnorm/meter.py#L135-L136.
102        let q = 0.500_327_05;
103        let center_hz = 38.135_47;
104
105        // Formula taken from https://github.com/csteinmetz1/pyloudnorm/blob/
106        // 6baa64d59b7794bc812e124438692e7fd2e65c0c/pyloudnorm/iirfilter.py#L145-L151
107        let k = (f32::consts::PI * center_hz / sample_rate_hz).tan();
108        Filter {
109            a1: 2.0 * (k * k - 1.0) / (1.0 + k / q + k * k),
110            a2: (1.0 - k / q + k * k) / (1.0 + k / q + k * k),
111            b0: 1.0,
112            b1: -2.0,
113            b2: 1.0,
114
115            x1: 0.0,
116            x2: 0.0,
117            y1: 0.0,
118            y2: 0.0,
119        }
120    }
121
122    /// Feed the next input sample, get the next output sample.
123    #[inline(always)]
124    pub fn apply(&mut self, x0: f32) -> f32 {
125        let y0 = 0.0 + self.b0 * x0 + self.b1 * self.x1 + self.b2 * self.x2
126            - self.a1 * self.y1
127            - self.a2 * self.y2;
128
129        self.x2 = self.x1;
130        self.x1 = x0;
131        self.y2 = self.y1;
132        self.y1 = y0;
133
134        y0
135    }
136}

138/// Compensated sum, for summing many values of different orders of magnitude
139/// accurately.
140#[derive(Copy, Clone, PartialEq)]
141struct Sum {
142    sum: f32,
143    residue: f32,
144}
145
146impl Sum {
147    #[inline(always)]
148    fn zero() -> Sum {
149        Sum {
150            sum: 0.0,
151            residue: 0.0,
152        }
153    }
154
155    #[inline(always)]
156    fn add(&mut self, x: f32) {
157        let sum = self.sum + (self.residue + x);
158        self.residue = (self.residue + x) - (sum - self.sum);
159        self.sum = sum;
160    }
161}

163/// The mean of the squares of the K-weighted samples in a window of time.
164///
165/// K-weighted power is equivalent to K-weighted loudness, the only difference
166/// is one of scale: power is quadratic in sample amplitudes, whereas loudness
167/// units are logarithmic. `loudness_lkfs` and `from_lkfs` convert between power,
168/// and K-weighted Loudness Units relative to nominal Full Scale (LKFS).
169///
170/// The term “LKFS” (Loudness Units, K-Weighted, relative to nominal Full Scale)
171/// is used in BS.1770-4 to emphasize K-weighting, but the term is otherwise
172/// interchangeable with the more widespread term “LUFS” (Loudness Units,
173/// relative to Full Scale). Loudness units are related to decibels in the
174/// following sense: boosting a signal that has a loudness of
175/// -<var>L<sub>K</sub></var> LUFS by <var>L<sub>K</sub></var> dB (by
176/// multiplying the amplitude by 10<sup><var>L<sub>K</sub></var>/20</sup>) will
177/// bring the loudness to 0 LUFS.
178///
179/// K-weighting refers to a high-shelf and high-pass filter that model the
180/// effect that humans perceive a certain amount of power in low frequencies to
181/// be less loud than the same amount of power in higher frequencies. In this
182/// library the `Power` type is used exclusively to refer to power after applying K-weighting.
183///
184/// The nominal “full scale” is the range [-1.0, 1.0]. Because the power is the
185/// mean square of the samples, if no input samples exceeded the full scale, the
186/// power will be in the range [0.0, 1.0]. However, the power delivered by
187/// multiple channels, which is a weighted sum over individual channel powers,
188/// can exceed this range, because the weighted sum is not normalized.
189#[derive(Copy, Clone, PartialEq, PartialOrd)]
190pub struct Power(pub f32);
191
192impl Power {
193    /// Convert Loudness Units relative to Full Scale into a squared sample amplitude.
194    ///
195    /// This is the inverse of `loudness_lkfs`.
196    pub fn from_lkfs(lkfs: f32) -> Power {
197        // The inverse of the formula below.
198        Power(10.0_f32.powf((lkfs + 0.691) * 0.1))
199    }
200
201    /// Return the loudness of this window in Loudness Units, K-weighted, relative to Full Scale.
202    ///
203    /// This is the inverse of `from_lkfs`.
204    pub fn loudness_lkfs(&self) -> f32 {
205        // Equation 2 (p.5) of BS.1770-4.
206        -0.691 + 10.0 * self.0.log10()
207    }
208}

210/// A `T` value for non-overlapping windows of audio, 100ms in length.
211///
212/// The `ChannelLoudnessMeter` applies K-weighting and then produces the power
213/// for non-overlapping windows of 100ms duration.
214///
215/// These non-overlapping 100ms windows can later be combined into overlapping
216/// windows of 400ms, spaced 100ms apart, to compute instantaneous loudness or
217/// to perform a gated measurement, or they can be combined into even larger
218/// windows for a momentary loudness measurement.
219#[derive(Copy, Clone, Debug)]
220pub struct Windows100ms<T> {
221    pub inner: T,
222}
223
224impl<T> Windows100ms<T> {
225    /// Wrap a new empty vector.
226    pub fn new() -> Windows100ms<Vec<T>> {
227        Windows100ms { inner: Vec::new() }
228    }
229
230    /// Apply `as_ref` to the inner value.
231    pub fn as_ref(&self) -> Windows100ms<&[Power]>
232    where
233        T: AsRef<[Power]>,
234    {
235        Windows100ms {
236            inner: self.inner.as_ref(),
237        }
238    }
239
240    /// Apply `as_mut` to the inner value.
241    pub fn as_mut(&mut self) -> Windows100ms<&mut [Power]>
242    where
243        T: AsMut<[Power]>,
244    {
245        Windows100ms {
246            inner: self.inner.as_mut(),
247        }
248    }
249
250    #[allow(clippy::len_without_is_empty)]
251    /// Apply `len` to the inner value.
252    pub fn len(&self) -> usize
253    where
254        T: AsRef<[Power]>,
255    {
256        self.inner.as_ref().len()
257    }
258}

260/// Measures K-weighted power of non-overlapping 100ms windows of a single channel of audio.
261///
262/// # Output
263///
264/// The output of the meter is an intermediate result in the form of power for
265/// 100ms non-overlapping windows. The windows need to be processed further to
266/// get one of the instantaneous, momentary, and integrated loudness
267/// measurements defined in BS.1770.
268///
269/// The windows can also be inspected directly; the data is meaningful
270/// on its own (the K-weighted power delivered in that window of time), but it
271/// is not something that BS.1770 defines a term for.
272///
273/// # Multichannel audio
274///
275/// To perform a loudness measurement of multichannel audio, construct a
276/// `ChannelLoudnessMeter` per channel, and later combine the measured power
277/// with e.g. `reduce_stereo`.
278///
279/// # Instantaneous loudness
280///
281/// The instantaneous loudness is the power over a 400ms window, so you can
282/// average four 100ms windows. No special functionality is implemented to help
283/// with that at this time. ([Pull requests would be accepted.][contribute])
284///
285/// # Momentary loudness
286///
287/// The momentary loudness is the power over a 3-second window, so you can
288/// average thirty 100ms windows. No special functionality is implemented to
289/// help with that at this time. ([Pull requests would be accepted.][contribute])
290///
291/// # Integrated loudness
292///
293/// Use `gated_mean` to perform an integrated loudness measurement:
294///
295/// ```ignore
296/// # use std::iter;
297/// # use bs1770::{ChannelLoudnessMeter, gated_mean};
298/// # let sample_rate_hz = 44_100;
299/// # let samples_per_100ms = sample_rate_hz / 10;
300/// # let mut meter = ChannelLoudnessMeter::new(sample_rate_hz);
301/// # meter.push((0..44_100).map(|i| (i as f32 * 0.01).sin()));
302/// let integrated_loudness_lkfs = gated_mean(meter.as_100ms_windows())
303///     .unwrap_or(bs1770::Power(0.0))
304///     .loudness_lkfs();
305/// ```
306///
307/// [contribute]: https://github.com/ruuda/bs1770/blob/master/CONTRIBUTING.md
308#[derive(Clone)]
309pub struct ChannelLoudnessMeter {
310    /// The number of samples that fit in 100ms of audio.
311    samples_per_100ms: u32,
312
313    /// Stage 1 filter (head effects, high shelf).
314    filter_stage1: Filter,
315
316    /// Stage 2 filter (high-pass).
317    filter_stage2: Filter,
318
319    /// Sum of the squares over non-overlapping windows of 100ms.
320    windows: Windows100ms<Vec<Power>>,
321
322    /// The number of samples in the current unfinished window.
323    count: u32,
324
325    /// The sum of the squares of the samples in the current unfinished window.
326    square_sum: Sum,
327}
328
329impl ChannelLoudnessMeter {
330    /// Construct a new loudness meter for the given sample rate.
331    pub fn new(sample_rate_hz: u32) -> ChannelLoudnessMeter {
332        ChannelLoudnessMeter {
333            samples_per_100ms: sample_rate_hz / 10,
334            filter_stage1: Filter::high_shelf(sample_rate_hz as f32),
335            filter_stage2: Filter::high_pass(sample_rate_hz as f32),
336            windows: Windows100ms::new(),
337            count: 0,
338            square_sum: Sum::zero(),
339        }
340    }
341
342    /// Feed input samples for loudness analysis.
343    ///
344    /// # Full scale
345    ///
346    /// Full scale for the input samples is the interval [-1.0, 1.0]. If your
347    /// input consists of signed integer samples, you can convert as follows:
348    ///
349    /// ```ignore
350    /// # let mut meter = bs1770::ChannelLoudnessMeter::new(44_100);
351    /// # let bits_per_sample = 16_usize;
352    /// # let samples = &[0_i16];
353    /// // Note that the maximum amplitude is `1 << (bits_per_sample - 1)`,
354    /// // one bit is the sign bit.
355    /// let normalizer = 1.0 / (1_u64 << (bits_per_sample - 1)) as f32;
356    /// meter.push(samples.iter().map(|&s| s as f32 * normalizer));
357    /// ```
358    ///
359    /// # Repeated calls
360    ///
361    /// You can call `push` multiple times to feed multiple batches of samples.
362    /// This is equivalent to feeding a single chained iterator. The leftover of
363    /// samples that did not fill a full 100ms window is not discarded:
364    ///
365    /// ```ignore
366    /// # use std::iter;
367    /// # use bs1770::ChannelLoudnessMeter;
368    /// let sample_rate_hz = 44_100;
369    /// let samples_per_100ms = sample_rate_hz / 10;
370    /// let mut meter = ChannelLoudnessMeter::new(sample_rate_hz);
371    ///
372    /// meter.push(iter::repeat(0.0).take(samples_per_100ms as usize - 1));
373    /// assert_eq!(meter.as_100ms_windows().len(), 0);
374    ///
375    /// meter.push(iter::once(0.0));
376    /// assert_eq!(meter.as_100ms_windows().len(), 1);
377    /// ```
378    pub fn push<I: Iterator<Item = f32>>(&mut self, samples: I) {
379        let normalizer = 1.0 / self.samples_per_100ms as f32;
380
381        // LLVM, if you could go ahead and inline those apply calls, and then
382        // unroll and vectorize the loop, that'd be terrific.
383        for x in samples {
384            let y = self.filter_stage1.apply(x);
385            let z = self.filter_stage2.apply(y);
386
387            self.square_sum.add(z * z);
388            self.count += 1;
389
390            // TODO: Should this branch be marked cold?
391            if self.count == self.samples_per_100ms {
392                let mean_squares = Power(self.square_sum.sum * normalizer);
393                self.windows.inner.push(mean_squares);
394                // We intentionally do not reset the residue. That way, leftover
395                // energy from this window is not lost, so for the file overall,
396                // the sum remains more accurate.
397                self.square_sum.sum = 0.0;
398                self.count = 0;
399            }
400        }
401    }
402
403    /// Return a reference to the 100ms windows analyzed so far.
404    pub fn as_100ms_windows(&self) -> Windows100ms<&[Power]> {
405        self.windows.as_ref()
406    }
407
408    /// Return all 100ms windows analyzed so far.
409    pub fn into_100ms_windows(self) -> Windows100ms<Vec<Power>> {
410        self.windows
411    }
412}

414/// Combine power for multiple channels by taking a weighted sum.
415///
416/// Note that BS.1770-4 defines power for a multi-channel signal as a weighted
417/// sum over channels which is not normalized. This means that a stereo signal
418/// is inherently louder than a mono signal. For a mono signal played back on
419/// stereo speakers, you should therefore still apply `reduce_stereo`, passing
420/// in the same signal for both channels.
421pub fn reduce_stereo(
422    left: Windows100ms<&[Power]>,
423    right: Windows100ms<&[Power]>,
424) -> Windows100ms<Vec<Power>> {
425    assert_eq!(
426        left.len(),
427        right.len(),
428        "Channels must have the same length."
429    );
430    let mut result = Vec::with_capacity(left.len());
431    for (l, r) in left.inner.iter().zip(right.inner) {
432        result.push(Power(l.0 + r.0));
433    }
434    Windows100ms { inner: result }
435}
436
437/// In-place version of `reduce_stereo` that stores the result in the former left channel.
438pub fn reduce_stereo_in_place(left: Windows100ms<&mut [Power]>, right: Windows100ms<&[Power]>) {
439    assert_eq!(
440        left.len(),
441        right.len(),
442        "Channels must have the same length."
443    );
444    for (l, r) in left.inner.iter_mut().zip(right.inner) {
445        l.0 += r.0;
446    }
447}

449/// Perform gating and averaging for a BS.1770-4 integrated loudness measurement.
450///
451/// The integrated loudness measurement is not just the average power over the
452/// entire signal. BS.1770-4 defines two stages of gating that exclude
453/// parts of the signal, to ensure that silent parts do not contribute to the
454/// loudness measurement. This function performs that gating, and returns the
455/// average power over the windows that were not excluded.
456///
457/// The result of this function is the integrated loudness measurement.
458///
459/// When no signal remains after applying the gate, this function returns
460/// `None`. In particular, this happens when all of the signal is softer than
461/// -70 LKFS, including a signal that consists of pure silence.
462pub fn gated_mean(windows_100ms: Windows100ms<&[Power]>) -> Option<Power> {
463    let mut gating_blocks = Vec::with_capacity(windows_100ms.len());
464
465    // Stage 1: an absolute threshold of -70 LKFS. (Equation 6, p.6.)
466    let absolute_threshold = Power::from_lkfs(-70.0);
467
468    // Iterate over all 400ms windows.
469    for window in windows_100ms.inner.windows(4) {
470        // Note that the sum over channels has already been performed at this point.
471        let gating_block_power = Power(0.25 * window.iter().map(|mean| mean.0).sum::<f32>());
472
473        if gating_block_power > absolute_threshold {
474            gating_blocks.push(gating_block_power);
475        }
476    }
477
478    if gating_blocks.is_empty() {
479        return None;
480    }
481
482    // Compute the loudness after applying the absolute gate, in order to
483    // determine the threshold for the relative gate.
484    let mut sum_power = Sum::zero();
485    for &gating_block_power in &gating_blocks {
486        sum_power.add(gating_block_power.0);
487    }
488    let absolute_gated_power = Power(sum_power.sum / (gating_blocks.len() as f32));
489
490    // Stage 2: Apply the relative gate.
491    let relative_threshold = Power::from_lkfs(absolute_gated_power.loudness_lkfs() - 10.0);
492    let mut sum_power = Sum::zero();
493    let mut n_blocks = 0_usize;
494    for &gating_block_power in &gating_blocks {
495        if gating_block_power > relative_threshold {
496            sum_power.add(gating_block_power.0);
497            n_blocks += 1;
498        }
499    }
500
501    if n_blocks == 0 {
502        return None;
503    }
504
505    let relative_gated_power = Power(sum_power.sum / n_blocks as f32);
506    Some(relative_gated_power)
