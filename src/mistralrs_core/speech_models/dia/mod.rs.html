44fn create_attn_mask(q_padding_mask_1d: &Tensor, k_padding_mask_1d: &Tensor) -> Result<Tensor> {
45    let (b1, _tq) = q_padding_mask_1d.dims2()?;
46    let (b2, _tk) = k_padding_mask_1d.dims2()?;
47    assert_eq!(b1, b2);
48
49    let p_mask_q = q_padding_mask_1d.unsqueeze(2)?;
50    let p_mask_k = k_padding_mask_1d.unsqueeze(1)?;
51
52    // # Condition A: Non-padding query attends to non-padding key
53    // non_pad_attends_non_pad = p_mask_q & p_mask_k  # Shape [B, Tq, Tk]
54
55    // # Condition B: Padding query attends to padding key
56    // pad_attends_pad = (~p_mask_q) & (~p_mask_k)  # Shape [B, Tq, Tk]
57
58    // # Combine: True if padding status is compatible (both non-pad OR both pad)
59    // mask = non_pad_attends_non_pad | pad_attends_pad  # Shape [B, Tq, Tk]
60
61    // let np_att_np = p_mask_q.bitwise_and(&p_mask_k)?;
62    // let p_att_p = p_mask_q.bitwise_not()?.bitwise_and(&p_mask_k.bitwise_not()?)?;
63    // let mask = np_att_np.bitwise_or(&p_att_p)?;
64
65    let mask = p_mask_q.broadcast_eq(&p_mask_k)?;
66
67    mask.unsqueeze(1)
68}

97    fn prepare_audio_prompt(&self) -> Result<Tensor> {
98        let num_channels = self.cfg.data.channels;
99        let audio_pad_value = self.cfg.data.audio_pad_value;
100        let audio_bos_value = self.cfg.data.audio_bos_value;
101        let delay_pattern = &self.cfg.data.delay_pattern;
102        let max_delay_pattern = *delay_pattern.iter().max().unwrap() as usize;
103
104        let prefill =
105            (Tensor::ones((1, num_channels), DType::F32, &self.device)? * audio_bos_value as f64)?;
106
107        let delay_pad_tensor = (Tensor::ones(
108            (max_delay_pattern - 1, num_channels),
109            DType::F32,
110            &self.device,
111        )? * -1f64)?;
112        let prefill = Tensor::cat(&[prefill, delay_pad_tensor], 0)?;
113
114        let delay_precomp = build_delay_indices(
115            1,
116            prefill.dim(0)?,
117            num_channels,
118            &delay_pattern.iter().map(|x| *x as i64).collect::<Vec<_>>(),
119            &self.device,
120        )?;
121
122        let prefill = apply_audio_delay(
123            &prefill.unsqueeze(0)?,
124            audio_pad_value as i64,
125            audio_bos_value as i64,
126            &delay_precomp,
127        )?
128        .squeeze(0)?;
129
130        Ok(prefill)
131    }

133    fn prepare_text_prompt(&self, text: &str) -> Result<Tensor> {
134        let text_pad_value = self.cfg.data.text_pad_value;
135        let max_len = self.cfg.data.text_length;
136
137        let text = text.replace("[S1]", "\x01").replace("[S2]", "\x02");
138        let text_tokens = text.as_bytes();
139
140        let current_len = text_tokens.len();
141        let padding_needed = max_len - current_len;
142        let padded_text_np = if max_len <= current_len {
143            let text_tokens = &text_tokens[..max_len];
144            Tensor::new(text_tokens, &self.device)?
145        } else {
146            let text = Tensor::new(text_tokens, &self.device)?;
147            let pad =
148                (Tensor::ones(padding_needed, DType::U8, &self.device)? * text_pad_value as f64)?;
149            Tensor::cat(&[text, pad], 0)?
150        };
151
152        padded_text_np.to_dtype(DType::U32)?.unsqueeze(0)
153    }

155    /// Returns:
156    /// - generated tokens
157    /// - decoder attn mask
158    /// - encoder out
159    /// - encoder positions
160    /// - cross cache
161    /// - self cache
162    fn prepare_generation(&self, text: &str) -> Result<PrepareGenerationOutput> {
163        let enc_input_cond = self.prepare_text_prompt(text)?;
164        let enc_input_uncond = enc_input_cond.zeros_like()?;
165        let enc_input = Tensor::cat(&[&enc_input_uncond, &enc_input_cond], 0)?;
166
167        let prefill = self.prepare_audio_prompt()?;
168
169        let encoder_positions =
170            Tensor::arange(0f32, self.cfg.data.text_length as f32, &self.device)?
171                .unsqueeze(0)?
172                .repeat((2, 1))?;
173        let encoder_padding_mask = enc_input_cond
174            .ne(self.cfg.data.text_pad_value)?
175            .repeat((2, 1))?;
176        let encoder_attn_mask = create_attn_mask(&encoder_padding_mask, &encoder_padding_mask)?;
177        let encoder_out =
178            self.model
179                .encoder
180                .forward(&enc_input, &encoder_positions, Some(&encoder_attn_mask))?;
181
182        let decoder_cross_attn_cache = self
183            .model
184            .decoder
185            .precompute_cross_attn_cache(&encoder_out, &encoder_positions)?;
186        let decoder_padding_mask = Tensor::ones((2, 1), DType::U8, &self.device)?;
187        let decoder_attn_mask = create_attn_mask(&decoder_padding_mask, &encoder_padding_mask)?;
188
189        let max_audio_length = self.cfg.data.audio_length;
190        let generated_tokens = Tensor::zeros(
191            (max_audio_length, self.cfg.data.channels),
192            DType::F32,
193            &self.device,
194        )?;
195
196        generated_tokens.slice_set(&prefill.to_dtype(DType::F32)?, 0, 0)?;
197
198        let mut decoder_self_attn_cache = Vec::new();
199        for _ in 0..self.cfg.model.decoder.n_layer {
200            decoder_self_attn_cache.push(Some(DiaKvCache::new(
201                (
202                    2,
203                    self.cfg.model.decoder.kv_heads,
204                    max_audio_length,
205                    self.cfg.model.decoder.gqa_head_dim,
206                ),
207                self.dtype,
208                &self.device,
209            )?));
210        }
211
212        Ok(PrepareGenerationOutput {
213            generated_tokens,
214            decoder_attn_mask,
215            encoder_out,
216            encoder_positions,
217            cross_cache: decoder_cross_attn_cache,
218            self_cache: decoder_self_attn_cache,
219        })
220    }

222    fn sample_next_token(
223        &self,
224        logits: &Tensor,
225        temperature: f32,
226        top_p: f32,
227        cfg_filter_top_k: Option<usize>,
228        rng: &mut Isaac64Rng,
229    ) -> Result<Vec<u32>> {
230        if temperature == 0. {
231            return logits.argmax(D::Minus1)?.to_vec1();
232        }
233
234        let logits = candle_nn::ops::softmax_last_dim(
235            &(logits.to_dtype(DType::F32)? / temperature as f64)?,
236        )?;
237        let batch_logits: Vec<Vec<f32>> = logits.to_vec2::<f32>()?;
238
239        let mut sampled = Vec::with_capacity(batch_logits.len());
240        let audio_eos_value = self.cfg.data.audio_eos_value as usize;
241
242        for mut probs in batch_logits {
243            let mut argsort_indices: Vec<usize> = (0..probs.len()).collect();
244            argsort_indices.sort_unstable_by(|&i, &j| probs[j].partial_cmp(&probs[i]).unwrap());
245
246            // ---- Mask out EOS unless it is the highest ----
247            if !argsort_indices.is_empty() && argsort_indices[0] != audio_eos_value {
248                probs[audio_eos_value] = 0.0;
249            }
250
251            // Top-k
252            if let Some(cfg_filter_top_k) = cfg_filter_top_k {
253                // Clamp smaller probabilities to zero.
254                for (index, val) in argsort_indices.iter().enumerate() {
255                    if index >= cfg_filter_top_k {
256                        probs[*val] = 0.0;
257                    }
258                }
259            }
260
261            // Top-p
262            let mut cumsum = 0.;
263            for index in &argsort_indices {
264                if cumsum >= top_p {
265                    probs[*index] = 0.0;
266                } else {
267                    cumsum += probs[*index];
268                }
269            }
270
271            let distr = WeightedIndex::new(&probs).map_err(candle_core::Error::msg)?;
272            sampled.push(distr.sample(rng) as u32);
273        }
274        Ok(sampled)
275    }

277    #[allow(clippy::too_many_arguments)]
278    fn decoder_step(
279        &self,
280        tokens: &Tensor,
281        encoder_out: &Tensor,
282        self_attn_mask: Option<&Tensor>,
283        cross_attn_mask: Option<&Tensor>,
284        encoder_positions: &Tensor,
285        decoder_positions: &Tensor,
286        self_attn_cache: &mut [Option<DiaKvCache>],
287        cross_attn_cache: &mut [Option<DiaKvCache>],
288        cfg_scale: f32,
289        temperature: f32,
290        top_p: f32,
291        cfg_filter_top_k: Option<usize>,
292        rng: &mut Isaac64Rng,
293        current_idx: usize,
294    ) -> Result<Vec<u32>> {
295        let audio_eos_value = self.cfg.data.audio_eos_value as usize;
296
297        // println!("{tokens}");
298        let mut logits = self.model.decoder.decode_step(
299            tokens,
300            encoder_out,
301            self_attn_mask,
302            cross_attn_mask,
303            encoder_positions,
304            decoder_positions,
305            self_attn_cache,
306            cross_attn_cache,
307            current_idx,
308        )?;
309
310        let logits_last = logits.i((.., logits.dim(1)? - 1.., .., ..))?.squeeze(1)?;
311        let uncond_logits = logits_last.i((0, .., ..))?.squeeze(0)?;
312        let cond_logits = logits_last.i((1, .., ..))?.squeeze(0)?;
313
314        logits = (&cond_logits + (cfg_scale as f64 * (&cond_logits - uncond_logits)?)?)?;
315        // logits_CxV[:, audio_eos_value + 1 :] = -torch.inf
316        // logits_CxV[1:, audio_eos_value:] = -torch.inf
317        logits = logits.slice_assign(
318            &[0..logits.dim(0)?, audio_eos_value + 1..logits.dim(1)?],
319            &(Tensor::ones(
320                (logits.dim(0)?, logits.dim(1)? - (audio_eos_value + 1)),
321                logits.dtype(),
322                logits.device(),
323            )? * f64::NEG_INFINITY)?,
324        )?;
325        logits = logits.slice_assign(
326            &[1..logits.dim(0)?, audio_eos_value..logits.dim(1)?],
327            &(Tensor::ones(
328                (logits.dim(0)? - 1, logits.dim(1)? - audio_eos_value),
329                logits.dtype(),
330                logits.device(),
331            )? * f64::NEG_INFINITY)?,
332        )?;
333
334        let next = self.sample_next_token(&logits, temperature, top_p, cfg_filter_top_k, rng)?;
335        Ok(next)
336    }

338    fn generate_output(&self, generated_codes: &Tensor) -> Result<Vec<f32>> {
339        let num_channels = self.cfg.data.channels;
340        let seq_length = generated_codes.dim(0)?;
341        let audio_pad_value = self.cfg.data.audio_pad_value;
342        let delay_pattern = &self.cfg.data.delay_pattern;
343        let max_delay_pattern = *delay_pattern.iter().max().unwrap() as usize;
344
345        let revert_precomp = build_revert_indices(
346            1,
347            seq_length,
348            num_channels,
349            &delay_pattern.iter().map(|x| *x as i64).collect::<Vec<_>>(),
350            &self.device,
351        )?;
352
353        let mut codebook = revert_audio_delay(
354            &generated_codes.unsqueeze(0)?,
355            audio_pad_value as i64,
356            &revert_precomp,
357            seq_length,
358        )?;
359        codebook = codebook.i((.., ..codebook.dim(1)? - max_delay_pattern, ..))?;
360
361        let min_valid_index = 0f64;
362        let max_valid_index = 1023f64;
363
364        // Original code scatters values where the below `invalid_mask` is true to 0.
365        // We do the opposite inverse.
366        let invalid_mask = codebook
367            .lt(min_valid_index)?
368            .bitwise_or(&codebook.gt(max_valid_index)?)?;
369        codebook = invalid_mask.where_cond(&codebook.zeros_like()?, &codebook)?;
370
371        let codes = codebook.transpose(1, 2)?;
372        let pcm = self.dac.decode_codes(&codes.to_dtype(DType::U32)?)?;
373        let pcm = pcm.i((0, 0))?;
374        let pcm = normalize_loudness(&pcm, RATE as u32, true)?;
375        let pcm = pcm.to_vec1::<f32>()?;
376
377        Ok(pcm)
378    }

380    pub fn generate(
381        &self,
382        text: &str,
383        cfg: &SpeechGenerationConfig,
384    ) -> Result<SpeechGenerationOutput> {
385        let SpeechGenerationConfig::Dia {
386            max_tokens,
387            cfg_scale,
388            temperature,
389            top_p,
390            top_k,
391        } = cfg;
392
393        let audio_pad_value = self.cfg.data.audio_pad_value as u32;
394        let audio_eos_value = self.cfg.data.audio_eos_value as u32;
395        let delay_pattern = &self.cfg.data.delay_pattern;
396        let max_tokens = max_tokens.unwrap_or(self.cfg.data.audio_length);
397        let max_delay_pattern = *delay_pattern.iter().max().unwrap() as usize;
398
399        let PrepareGenerationOutput {
400            mut generated_tokens,
401            decoder_attn_mask,
402            encoder_out,
403            encoder_positions,
404            cross_cache: mut decoder_cross_attn_cache,
405            self_cache: mut decoder_self_attn_cache,
406        } = self.prepare_generation(text)?;
407
408        let self_attn_mask = apply_triangular(
409            &Tensor::ones(
410                (self.cfg.data.audio_length, self.cfg.data.audio_length),
411                DType::U8,
412                decoder_attn_mask.device(),
413            )?,
414            0,
415            false,
416        )?;
417
418        let mut dec_step = 0;
419
420        let mut bos_countdown = max_delay_pattern;
421        let mut eos_detected = false;
422        let mut eos_countdown: Option<usize> = None;
423
424        let mut rng = Isaac64Rng::seed_from_u64(0);
425
426        let mut start = Instant::now();
427        while dec_step < max_tokens {
428            let dec_positions = Tensor::full(dec_step as f32, (2, 1), &self.device)?;
429            let current_tokens = generated_tokens
430                .i((dec_step..dec_step + 1, ..))?
431                .unsqueeze(0)?
432                .repeat((2, 1, 1))?;
433
434            let mut pred_c = self.decoder_step(
435                &current_tokens.to_dtype(DType::U32)?,
436                &encoder_out,
437                Some(&self_attn_mask),
438                Some(&decoder_attn_mask),
439                &encoder_positions,
440                &dec_positions,
441                &mut decoder_self_attn_cache,
442                &mut decoder_cross_attn_cache,
443                *cfg_scale,
444                *temperature,
445                *top_p,
446                *top_k,
447                &mut rng,
448                dec_step,
449            )?;
450
451            if (!eos_detected && pred_c[0] == audio_eos_value)
452                || dec_step == max_tokens - max_delay_pattern - 1
453            {
454                eos_detected = true;
455                eos_countdown = Some(max_delay_pattern);
456            }
457
458            if let Some(eos_countdown) = &mut eos_countdown {
459                let step_after_eos = max_delay_pattern - *eos_countdown;
460                for (i, d) in delay_pattern.iter().enumerate() {
461                    match step_after_eos.cmp(&(*d as usize)) {
462                        std::cmp::Ordering::Equal => pred_c[i] = audio_eos_value,
463                        std::cmp::Ordering::Greater => pred_c[i] = audio_pad_value,
464                        std::cmp::Ordering::Less => {}
465                    }
466                }
467                *eos_countdown -= 1;
468            }
469
470            bos_countdown = bos_countdown.saturating_sub(1);
471
472            let apply_mask = bos_countdown > 0;
473            if apply_mask {
474                let len = pred_c.len();
475                let dec_out = Tensor::from_vec(pred_c, len, &self.device)?.to_dtype(DType::F32)?;
476
477                let mask = generated_tokens
478                    .i((dec_step + 1..dec_step + 2, ..))?
479                    .eq(-1.)?;
480                generated_tokens = generated_tokens.slice_assign(
481                    &[dec_step + 1..dec_step + 2, 0..generated_tokens.dim(1)?],
482                    &mask.where_cond(
483                        &dec_out.unsqueeze(0)?,
484                        &generated_tokens.i((dec_step + 1..dec_step + 2, ..))?,
485                    )?,
486                )?;
487            } else {
488                let len = pred_c.len();
489                generated_tokens.slice_set(
490                    &Tensor::from_vec(pred_c, len, &self.device)?
491                        .unsqueeze(0)?
492                        .to_dtype(DType::F32)?,
493                    0,
494                    dec_step + 1,
495                )?;
496            }
497
498            if eos_countdown.is_some_and(|eos_countdown| eos_countdown == 0) {
499                break;
500            }
501
502            dec_step += 1;
503
504            let end = Instant::now();
505            if dec_step % TOKENS_PER_SECOND == 0 {
506                info!(
507                    "Generated {}s of audio, {dec_step} tokens at {:.2} tokens/second.",
508                    dec_step / TOKENS_PER_SECOND,
509                    TOKENS_PER_SECOND as f32 / (end - start).as_secs_f32()
510                );
511                start = end;
512            }
513        }
514
515        let generated_codes = generated_tokens.i((0..dec_step + 1, ..))?;
516        let pcm = self.generate_output(&generated_codes)?;
517        Ok(SpeechGenerationOutput {
518            pcm: Arc::new(pcm),
519            rate: RATE,
520            channels: CHANNELS,
521        })
522    }
