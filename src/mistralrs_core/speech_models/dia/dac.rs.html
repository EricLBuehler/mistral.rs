11// Applies weight norm for inference by recomputing the weight tensor. This
12// does not apply to training.
13// https://pytorch.org/docs/stable/generated/torch.nn.utils.weight_norm.html
14fn conv1d_weight_norm(
15    in_c: usize,
16    out_c: usize,
17    kernel_size: usize,
18    config: candle_nn::Conv1dConfig,
19    vb: VarBuilder,
20) -> Result<Conv1d> {
21    let weight_g = vb.get((out_c, 1, 1), "weight_g")?;
22    let weight_v = vb.get((out_c, in_c, kernel_size), "weight_v")?;
23    let norm_v = weight_v.sqr()?.sum_keepdim((1, 2))?.sqrt()?;
24    let weight = weight_v.broadcast_mul(&weight_g)?.broadcast_div(&norm_v)?;
25    let bias = vb.get(out_c, "bias")?;
26    Ok(Conv1d::new(weight, Some(bias), config))
27}
28
29fn conv_transpose1d_weight_norm(
30    in_c: usize,
31    out_c: usize,
32    kernel_size: usize,
33    bias: bool,
34    config: candle_nn::ConvTranspose1dConfig,
35    vb: VarBuilder,
36) -> Result<ConvTranspose1d> {
37    let weight_g = vb.get((in_c, 1, 1), "weight_g")?;
38    let weight_v = vb.get((in_c, out_c, kernel_size), "weight_v")?;
39    let norm_v = weight_v.sqr()?.sum_keepdim((1, 2))?.sqrt()?;
40    let weight = weight_v.broadcast_mul(&weight_g)?.broadcast_div(&norm_v)?;
41    let bias = if bias {
42        Some(vb.get(out_c, "bias")?)
43    } else {
44        None
45    };
46    Ok(ConvTranspose1d::new(weight, bias, config))
47}

66#[derive(Debug, Clone)]
67pub struct Snake1d {
68    alpha: Tensor,
69}
70
71impl Snake1d {
72    pub fn new(channels: usize, vb: VarBuilder) -> Result<Self> {
73        let alpha = vb.get((1, channels, 1), "alpha")?;
74        Ok(Self { alpha })
75    }
76}
77
78impl candle_core::Module for Snake1d {
79    fn forward(&self, xs: &Tensor) -> Result<Tensor> {
80        let xs_shape = xs.shape();
81        let xs = xs.flatten_from(2)?;
82        let sin = self.alpha.broadcast_mul(&xs)?.sin()?;
83        let sin = (&sin * &sin)?;
84        (xs + (&self.alpha + 1e-9)?.recip()?.broadcast_mul(&sin)?)?.reshape(xs_shape)
85    }
86}

88#[derive(Debug, Clone)]
89pub struct ResidualUnit {
90    snake1: Snake1d,
91    conv1: Conv1d,
92    snake2: Snake1d,
93    conv2: Conv1d,
94}
95
96impl ResidualUnit {
97    pub fn new(dim: usize, dilation: usize, vb: VarBuilder) -> Result<Self> {
98        let pad = ((7 - 1) * dilation) / 2;
99        let vb = vb.pp("block");
100        let snake1 = Snake1d::new(dim, vb.pp(0))?;
101        let cfg1 = Conv1dConfig {
102            dilation,
103            padding: pad,
104            ..Default::default()
105        };
106        let conv1 = conv1d_weight_norm(dim, dim, 7, cfg1, vb.pp(1))?;
107        let snake2 = Snake1d::new(dim, vb.pp(2))?;
108        let conv2 = conv1d_weight_norm(dim, dim, 1, Default::default(), vb.pp(3))?;
109        Ok(Self {
110            snake1,
111            conv1,
112            snake2,
113            conv2,
114        })
115    }
116}
117
118impl candle_core::Module for ResidualUnit {
119    fn forward(&self, xs: &Tensor) -> Result<Tensor> {
120        let mut ys = self.snake1.forward(xs)?;
121        ys = Convolution.forward_1d(&self.conv1, &ys)?;
122        ys = self.snake2.forward(&ys)?;
123        ys = Convolution.forward_1d(&self.conv2, &ys)?;
124        let pad = (xs.dim(D::Minus1)? - ys.dim(D::Minus1)?) / 2;
125        if pad > 0 {
126            &ys + xs.narrow(D::Minus1, pad, ys.dim(D::Minus1)?)
127        } else {
128            ys + xs
129        }
130    }
131}

133#[derive(Debug, Clone)]
134pub struct DecoderBlock {
135    snake1: Snake1d,
136    conv_tr1: ConvTranspose1d,
137    res1: ResidualUnit,
138    res2: ResidualUnit,
139    res3: ResidualUnit,
140}
141
142impl DecoderBlock {
143    pub fn new(in_dim: usize, out_dim: usize, stride: usize, vb: VarBuilder) -> Result<Self> {
144        let vb = vb.pp("block");
145        let snake1 = Snake1d::new(in_dim, vb.pp(0))?;
146        let cfg = ConvTranspose1dConfig {
147            stride,
148            padding: stride.div_ceil(2),
149            ..Default::default()
150        };
151        let conv_tr1 =
152            conv_transpose1d_weight_norm(in_dim, out_dim, 2 * stride, true, cfg, vb.pp(1))?;
153        let res1 = ResidualUnit::new(out_dim, 1, vb.pp(2))?;
154        let res2 = ResidualUnit::new(out_dim, 3, vb.pp(3))?;
155        let res3 = ResidualUnit::new(out_dim, 9, vb.pp(4))?;
156        Ok(Self {
157            snake1,
158            conv_tr1,
159            res1,
160            res2,
161            res3,
162        })
163    }
164}
165
166impl candle_nn::Module for DecoderBlock {
167    fn forward(&self, xs: &Tensor) -> Result<Tensor> {
168        xs.apply(&self.snake1)?
169            .apply(&self.conv_tr1)?
170            .apply(&self.res1)?
171            .apply(&self.res2)?
172            .apply(&self.res3)
173    }
174}

176#[derive(Debug, Clone)]
177pub struct Decoder {
178    conv1: Conv1d,
179    blocks: Vec<DecoderBlock>,
180    snake1: Snake1d,
181    conv2: Conv1d,
182}
183
184impl Decoder {
185    pub fn new(
186        in_c: usize,
187        mut channels: usize,
188        rates: &[usize],
189        d_out: usize,
190        vb: VarBuilder,
191    ) -> Result<Self> {
192        let cfg1 = Conv1dConfig {
193            padding: 3,
194            ..Default::default()
195        };
196        let conv1 = conv1d_weight_norm(in_c, channels, 7, cfg1, vb.pp(0))?;
197        let mut blocks = Vec::with_capacity(rates.len());
198        for (idx, stride) in rates.iter().enumerate() {
199            let block = DecoderBlock::new(channels, channels / 2, *stride, vb.pp(idx + 1))?;
200            channels /= 2;
201            blocks.push(block)
202        }
203        let snake1 = Snake1d::new(channels, vb.pp(rates.len() + 1))?;
204        let conv2 = conv1d_weight_norm(channels, d_out, 7, cfg1, vb.pp(rates.len() + 2))?;
205        Ok(Self {
206            conv1,
207            blocks,
208            snake1,
209            conv2,
210        })
211    }
212}
213
214impl candle_core::Module for Decoder {
215    fn forward(&self, xs: &Tensor) -> Result<Tensor> {
216        let mut xs = Convolution.forward_1d(&self.conv1, xs)?;
217        for block in self.blocks.iter() {
218            xs = xs.apply(block)?
219        }
220        xs = self.snake1.forward(&xs)?;
221        Convolution.forward_1d(&self.conv2, &xs)
222    }
223}

225#[allow(unused)]
226#[derive(Clone, Debug)]
227pub struct VectorQuantizer {
228    in_proj: Conv1d,
229    out_proj: Conv1d,
230    codebook: candle_nn::Embedding,
231}
232
233impl VectorQuantizer {
234    pub fn new(in_dim: usize, cb_size: usize, cb_dim: usize, vb: VarBuilder) -> Result<Self> {
235        let in_proj = conv1d_weight_norm(in_dim, cb_dim, 1, Default::default(), vb.pp("in_proj"))?;
236        let out_proj =
237            conv1d_weight_norm(cb_dim, in_dim, 1, Default::default(), vb.pp("out_proj"))?;
238        let codebook = candle_nn::embedding(cb_size, cb_dim, vb.pp("codebook"))?;
239        Ok(Self {
240            in_proj,
241            out_proj,
242            codebook,
243        })
244    }
245
246    pub fn embed_code(&self, embed_id: &Tensor) -> Result<Tensor> {
247        embed_id.apply(&self.codebook)
248    }
249
250    pub fn decode_code(&self, embed_id: &Tensor) -> Result<Tensor> {
251        self.embed_code(embed_id)?.transpose(1, 2)
252    }
253}

255#[derive(Clone, Debug)]
256pub struct ResidualVectorQuantizer {
257    quantizers: Vec<VectorQuantizer>,
258}
259
260impl ResidualVectorQuantizer {
261    pub fn new(
262        input_dim: usize,
263        n_codebooks: usize,
264        cb_size: usize,
265        cb_dim: usize,
266        vb: VarBuilder,
267    ) -> Result<Self> {
268        let vb = &vb.pp("quantizers");
269        let quantizers = (0..n_codebooks)
270            .map(|i| VectorQuantizer::new(input_dim, cb_size, cb_dim, vb.pp(i)))
271            .collect::<Result<Vec<_>>>()?;
272        Ok(Self { quantizers })
273    }
274
275    #[allow(clippy::wrong_self_convention)]
276    pub fn from_codes(&self, codes: &Tensor) -> Result<Tensor> {
277        let mut sum = None;
278        for (idx, quantizer) in self.quantizers.iter().enumerate() {
279            let z_p_i = quantizer.decode_code(&codes.i((.., idx))?)?;
280            let z_q_i = z_p_i.apply(&quantizer.out_proj)?;
281            let s = match sum {
282                None => z_q_i,
283                Some(s) => (s + z_q_i)?,
284            };
285            sum = Some(s)
286        }
287        match sum {
288            Some(s) => Ok(s),
289            None => candle_core::bail!("empty codebooks"),
290        }
291    }
292}

294#[derive(Debug, Clone)]
295pub struct Model {
296    // pub encoder: Encoder,
297    pub quantizer: ResidualVectorQuantizer,
298    pub decoder: Decoder,
299}
300
301impl Model {
302    pub fn new(cfg: &Config, vb: VarBuilder) -> Result<Self> {
303        // let encoder = Encoder::new(64, &[2, 4, 8, 8], cfg.latent_dim, vb.pp("encoder").pp("model"))?;
304        let quantizer = ResidualVectorQuantizer::new(
305            cfg.latent_dim,
306            cfg.num_codebooks,
307            cfg.codebook_size,
308            8,
309            vb.pp("quantizer"),
310        )?;
311        let decoder = Decoder::new(
312            cfg.latent_dim,
313            1536,
314            &[8, 8, 4, 2],
315            1,
316            vb.pp("decoder").pp("model"),
317        )?;
318        Ok(Self {
319            // encoder,
320            decoder,
321            quantizer,
322        })
323    }
324
325    pub fn decode_codes(&self, audio_codes: &Tensor) -> Result<Tensor> {
326        let audio_values = self.quantizer.from_codes(audio_codes)?;
327        audio_values.apply(&self.decoder)
328    }
