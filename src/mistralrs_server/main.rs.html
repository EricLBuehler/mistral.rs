164/// Load MCP configuration from file path or environment variable
165fn load_mcp_config(mcp_config_path: Option<&str>) -> Result<Option<McpClientConfig>> {
166    let config_path = if let Some(path) = mcp_config_path {
167        Some(path.to_string())
168    } else {
169        // Check environment variable if no CLI arg provided
170        std::env::var("MCP_CONFIG_PATH").ok()
171    };
172
173    if let Some(path) = config_path {
174        match std::fs::read_to_string(&path) {
175            Ok(config_content) => {
176                match serde_json::from_str::<McpClientConfig>(&config_content) {
177                    Ok(config) => {
178                        // Validate configuration
179                        if let Err(e) = validate_mcp_config(&config) {
180                            error!("MCP configuration validation failed: {}", e);
181                            anyhow::bail!("Invalid MCP configuration: {}", e);
182                        }
183
184                        info!("Loaded and validated MCP configuration from {}", path);
185                        info!("Configured {} MCP servers", config.servers.len());
186                        Ok(Some(config))
187                    }
188                    Err(e) => {
189                        error!("Failed to parse MCP configuration: {}", e);
190                        error!("Please check your JSON syntax and ensure it matches the MCP configuration schema");
191                        anyhow::bail!("Invalid MCP configuration format: {}", e);
192                    }
193                }
194            }
195            Err(e) => {
196                error!("Failed to read MCP configuration file {}: {}", path, e);
197                error!("Please ensure the file exists and is readable");
198                anyhow::bail!("Cannot read MCP configuration file: {}", e);
199            }
200        }
201    } else {
202        Ok(None)
203    }
204}

206/// Validate MCP configuration for common issues
207fn validate_mcp_config(config: &McpClientConfig) -> Result<()> {
208    use std::collections::HashSet;
209
210    // Check for duplicate server IDs
211    let mut seen_ids = HashSet::new();
212    for server in &config.servers {
213        if !seen_ids.insert(&server.id) {
214            anyhow::bail!("Duplicate server ID: {}", server.id);
215        }
216
217        // Validate server ID format
218        if !server
219            .id
220            .chars()
221            .all(|c| c.is_alphanumeric() || c == '-' || c == '_')
222        {
223            anyhow::bail!(
224                "Invalid server ID '{}': must contain only alphanumeric, hyphen, underscore",
225                server.id
226            );
227        }
228
229        // Validate URLs for HTTP/WebSocket sources
230        match &server.source {
231            mistralrs_core::McpServerSource::Http { url, .. }
232            | mistralrs_core::McpServerSource::WebSocket { url, .. } => {
233                // Basic URL validation - check for scheme
234                if !url.starts_with("http://")
235                    && !url.starts_with("https://")
236                    && !url.starts_with("ws://")
237                    && !url.starts_with("wss://")
238                {
239                    anyhow::bail!("Invalid URL for server '{}': must start with http://, https://, ws://, or wss://", server.id);
240                }
241                if url.len() < 10 {
242                    anyhow::bail!("Invalid URL for server '{}': URL too short", server.id);
243                }
244            }
245            mistralrs_core::McpServerSource::Process { command, .. } => {
246                if command.is_empty() {
247                    anyhow::bail!("Empty command for server '{}'", server.id);
248                }
249            }
250        }
251    }
252
253    // Validate global settings
254    if let Some(timeout) = config.tool_timeout_secs {
255        if timeout == 0 {
256            anyhow::bail!("tool_timeout_secs must be greater than 0");
257        }
258    }
259
260    if let Some(max_calls) = config.max_concurrent_calls {
261        if max_calls == 0 {
262            anyhow::bail!("max_concurrent_calls must be greater than 0");
263        }
264    }
265
266    Ok(())
267}

269#[tokio::main]
270async fn main() -> Result<()> {
271    let args = Args::parse();
272
273    initialize_logging();
274
275    // Load MCP configuration if provided
276    let mcp_config = load_mcp_config(args.mcp_config.as_deref())?;
277
278    let paged_attn = configure_paged_attn_from_flags(args.paged_attn, args.no_paged_attn)?;
279
280    let mistralrs = MistralRsForServerBuilder::new()
281        .with_truncate_sequence(args.truncate_sequence)
282        .with_model(args.model)
283        .with_max_seqs(args.max_seqs)
284        .with_no_kv_cache(args.no_kv_cache)
285        .with_token_source(args.token_source)
286        .with_interactive_mode(args.interactive_mode)
287        .with_prefix_cache_n(args.prefix_cache_n)
288        .set_paged_attn(paged_attn)
289        .with_cpu(args.cpu)
290        .with_enable_search(args.enable_search)
291        .with_seed_optional(args.seed)
292        .with_log_optional(args.log)
293        .with_chat_template_optional(args.chat_template)
294        .with_jinja_explicit_optional(args.jinja_explicit)
295        .with_num_device_layers_optional(args.num_device_layers)
296        .with_in_situ_quant_optional(args.in_situ_quant)
297        .with_paged_attn_gpu_mem_optional(args.paged_attn_gpu_mem)
298        .with_paged_attn_gpu_mem_usage_optional(args.paged_attn_gpu_mem_usage)
299        .with_paged_ctxt_len_optional(args.paged_ctxt_len)
300        .with_paged_attn_block_size_optional(args.paged_attn_block_size)
301        .with_prompt_chunksize_optional(args.prompt_chunksize)
302        .with_mcp_config_optional(mcp_config)
303        .build()
304        .await?;
305
306    // TODO: refactor this
307    let bert_model = get_bert_model(args.enable_search, args.search_bert_model);
308
309    if args.interactive_mode {
310        interactive_mode(
311            mistralrs,
312            bert_model.is_some(),
313            args.enable_thinking.then_some(true),
314        )
315        .await;
316        return Ok(());
317    }
318
319    if !args.interactive_mode && args.port.is_none() && args.mcp_port.is_none() {
320        anyhow::bail!("Interactive mode was not specified, so expected port to be specified. Perhaps you forgot `-i` or `--port` or `--mcp-port`?")
321    }
322
323    let mcp_port = if let Some(port) = args.mcp_port {
324        let host = args
325            .serve_ip
326            .clone()
327            .unwrap_or_else(|| "0.0.0.0".to_string());
328        info!("MCP server listening on http://{host}:{port}/mcp.");
329        info!("MCP protocol version is {}.", LATEST_PROTOCOL_VERSION);
330        let mcp_server = mcp_server::create_http_mcp_server(mistralrs.clone(), host, port);
331
332        tokio::spawn(async move {
333            if let Err(e) = mcp_server.await {
334                eprintln!("MCP server error: {e}");
335            }
336        })
337    } else {
338        tokio::spawn(async {})
339    };
340
341    let oai_port = if let Some(port) = args.port {
342        let ip = args
343            .serve_ip
344            .clone()
345            .unwrap_or_else(|| "0.0.0.0".to_string());
346
347        // Create listener early to validate address before model loading
348        let listener = tokio::net::TcpListener::bind(format!("{ip}:{port}")).await?;
349
350        let app = MistralRsServerRouterBuilder::new()
351            .with_mistralrs(mistralrs)
352            .build()
353            .await?;
354
355        info!("OpenAI-compatible server listening on http://{ip}:{port}.");
356
357        tokio::spawn(async move {
358            if let Err(e) = axum::serve(listener, app).await {
359                eprintln!("OpenAI server error: {e}");
360            }
361        })
362    } else {
363        tokio::spawn(async {})
364    };
365
366    let (_, _) = join!(oai_port, mcp_port);
367
368    Ok(())
369}
