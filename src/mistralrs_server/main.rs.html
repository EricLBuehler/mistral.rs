17#[derive(Parser)]
18#[command(version, about, long_about = None)]
19struct Args {
20    /// IP to serve on. Defaults to "0.0.0.0"
21    #[arg(long)]
22    serve_ip: Option<String>,
23
24    /// Integer seed to ensure reproducible random number generation.
25    #[arg(short, long)]
26    seed: Option<u64>,
27
28    /// Port to serve on.
29    #[arg(short, long)]
30    port: Option<u16>,
31
32    /// Log all responses and requests to this file
33    #[clap(long, short)]
34    log: Option<String>,
35
36    /// If a sequence is larger than the maximum model length, truncate the number
37    /// of tokens such that the sequence will fit at most the maximum length.
38    /// If `max_tokens` is not specified in the request, space for 10 tokens will be reserved instead.
39    #[clap(long, short, action)]
40    truncate_sequence: bool,
41
42    /// Model selector
43    #[clap(subcommand)]
44    model: ModelSelected,
45
46    /// Maximum running sequences at any time. If the `tgt_non_granular_index` flag is set for X-LoRA models, this will be set to 1.
47    #[arg(long, default_value_t = defaults::MAX_SEQS)]
48    max_seqs: usize,
49
50    /// Use no KV cache.
51    #[arg(long, default_value_t = defaults::NO_KV_CACHE)]
52    no_kv_cache: bool,
53
54    /// Chat template file with a JINJA file with `messages`, `add_generation_prompt`, `bos_token`, `eos_token`, and `unk_token` as inputs.
55    /// Used if the automatic deserialization fails. If this ends with `.json` (ie., it is a file) then that template is loaded.
56    #[arg(short, long)]
57    chat_template: Option<String>,
58
59    /// Explicit JINJA chat template file (.jinja) to be used. If specified, this overrides all other chat templates.
60    #[arg(short, long)]
61    jinja_explicit: Option<String>,
62
63    /// Source of the token for authentication.
64    /// Can be in the formats: `literal:<value>`, `env:<value>`, `path:<value>`, `cache` to use a cached token, or `none` to use no token.
65    /// Defaults to `cache`.
66    #[arg(long, default_value_t = defaults::TOKEN_SOURCE, value_parser = parse_token_source)]
67    token_source: TokenSource,
68
69    /// Enter interactive mode instead of serving a chat server.
70    #[clap(long, short, action)]
71    interactive_mode: bool,
72
73    /// Number of prefix caches to hold on the device. Other caches are evicted to the CPU based on a LRU strategy.
74    #[arg(long, default_value_t = defaults::PREFIX_CACHE_N)]
75    prefix_cache_n: usize,
76
77    /// NOTE: This can be omitted to use automatic device mapping!
78    /// Number of device layers to load and run on GPU(s). All others will be on the CPU.
79    /// If one GPU is used, then this value should be an integer. Otherwise, it follows the following pattern:
80    /// ORD:NUM;... Where ORD is a unique device ordinal and NUM is the number of layers for that device.
81    #[arg(short, long, value_parser, value_delimiter = ';')]
82    num_device_layers: Option<Vec<String>>,
83
84    /// In-situ quantization to apply.
85    #[arg(long = "isq")]
86    in_situ_quant: Option<String>,
87
88    /// GPU memory to allocate for KV cache with PagedAttention in MBs.
89    /// PagedAttention is supported on CUDA and Metal. It is automatically activated on CUDA but not on Metal.
90    /// The priority is as follows: `pa-ctxt-len` > `pa-gpu-mem-usage` > `pa-gpu-mem`.
91    #[arg(long = "pa-gpu-mem")]
92    paged_attn_gpu_mem: Option<usize>,
93
94    /// Percentage of GPU memory to utilize after allocation of KV cache with PagedAttention, from 0 to 1.
95    /// If this is not set and the device is CUDA, it will default to `0.9`.
96    /// PagedAttention is supported on CUDA and Metal. It is automatically activated on CUDA but not on Metal.
97    /// The priority is as follows: `pa-ctxt-len` > `pa-gpu-mem-usage` > `pa-gpu-mem`.
98    #[arg(long = "pa-gpu-mem-usage")]
99    paged_attn_gpu_mem_usage: Option<f32>,
100
101    /// Total context length to allocate the KV cache for (total number of tokens which the KV cache can hold).
102    /// PagedAttention is supported on CUDA and Metal. It is automatically activated on CUDA but not on Metal.
103    /// The priority is as follows: `pa-ctxt-len` > `pa-gpu-mem-usage` > `pa-gpu-mem`.
104    /// This is the default setting, and it defaults to the `max-seq-len` specified in after the model type.
105    #[arg(long = "pa-ctxt-len")]
106    paged_ctxt_len: Option<usize>,
107
108    /// Block size (number of tokens per block) for PagedAttention. If this is not set and the device is CUDA, it will default to 32.
109    /// PagedAttention is supported on CUDA and Metal. It is automatically activated on CUDA but not on Metal.
110    #[arg(long = "pa-blk-size")]
111    paged_attn_block_size: Option<usize>,
112
113    /// Disable PagedAttention on CUDA. Because PagedAttention is already disabled on Metal, this is only applicable on CUDA.
114    #[arg(long = "no-paged-attn", default_value_t = defaults::NO_PAGED_ATTN)]
115    no_paged_attn: bool,
116
117    /// Enable PagedAttention on Metal. Because PagedAttention is already enabled on CUDA, this is only applicable on Metal.
118    #[arg(long = "paged-attn", default_value_t = defaults::PAGED_ATTN)]
119    paged_attn: bool,
120
121    /// Number of tokens to batch the prompt step into. This can help with OOM errors when in the prompt step, but reduces performance.
122    #[arg(long = "prompt-batchsize")]
123    prompt_chunksize: Option<usize>,
124
125    /// Use CPU only
126    #[arg(long)]
127    cpu: bool,
128
129    /// Enable searching compatible with the OpenAI `web_search_options` setting. This uses the BERT model specified below or the default.
130    #[arg(long = "enable-search")]
131    enable_search: bool,
132
133    /// Specify a Hugging Face model ID for a BERT model to assist web searching. Defaults to Snowflake Arctic Embed L.
134    #[arg(long = "search-bert-model")]
135    search_bert_model: Option<String>,
136
137    /// Enable thinking for interactive mode and models that support it.
138    #[arg(long = "enable-thinking")]
139    enable_thinking: bool,
140
141    /// Port to serve MCP protocol on
142    #[arg(long)]
143    mcp_port: Option<u16>,
144}

150#[tokio::main]
151async fn main() -> Result<()> {
152    let args = Args::parse();
153
154    initialize_logging();
155
156    let mistralrs = MistralRsForServerBuilder::new()
157        .with_truncate_sequence(args.truncate_sequence)
158        .with_model(args.model)
159        .with_max_seqs(args.max_seqs)
160        .with_no_kv_cache(args.no_kv_cache)
161        .with_token_source(args.token_source)
162        .with_interactive_mode(args.interactive_mode)
163        .with_prefix_cache_n(args.prefix_cache_n)
164        .with_no_paged_attn(args.no_paged_attn)
165        .with_paged_attn(args.paged_attn)
166        .with_cpu(args.cpu)
167        .with_enable_search(args.enable_search)
168        .with_seed_optional(args.seed)
169        .with_log_optional(args.log)
170        .with_chat_template_optional(args.chat_template)
171        .with_jinja_explicit_optional(args.jinja_explicit)
172        .with_num_device_layers_optional(args.num_device_layers)
173        .with_in_situ_quant_optional(args.in_situ_quant)
174        .with_paged_attn_gpu_mem_optional(args.paged_attn_gpu_mem)
175        .with_paged_attn_gpu_mem_usage_optional(args.paged_attn_gpu_mem_usage)
176        .with_paged_ctxt_len_optional(args.paged_ctxt_len)
177        .with_paged_attn_block_size_optional(args.paged_attn_block_size)
178        .with_prompt_chunksize_optional(args.prompt_chunksize)
179        .build()
180        .await?;
181
182    // TODO: refactor this
183    let bert_model = get_bert_model(args.enable_search, args.search_bert_model);
184
185    if args.interactive_mode {
186        interactive_mode(
187            mistralrs,
188            bert_model.is_some(),
189            args.enable_thinking.then_some(true),
190        )
191        .await;
192        return Ok(());
193    }
194
195    if !args.interactive_mode && args.port.is_none() && args.mcp_port.is_none() {
196        anyhow::bail!("Interactive mode was not specified, so expected port to be specified. Perhaps you forgot `-i` or `--port` or `--mcp-port`?")
197    }
198
199    let mcp_port = if let Some(port) = args.mcp_port {
200        let host = args
201            .serve_ip
202            .clone()
203            .unwrap_or_else(|| "0.0.0.0".to_string());
204        info!("MCP server listening on http://{host}:{port}/mcp.");
205        info!("MCP protocol version is {}.", LATEST_PROTOCOL_VERSION);
206        let mcp_server = mcp_server::create_http_mcp_server(mistralrs.clone(), host, port);
207
208        tokio::spawn(async move {
209            if let Err(e) = mcp_server.await {
210                eprintln!("MCP server error: {e}");
211            }
212        })
213    } else {
214        tokio::spawn(async {})
215    };
216
217    let oai_port = if let Some(port) = args.port {
218        let ip = args
219            .serve_ip
220            .clone()
221            .unwrap_or_else(|| "0.0.0.0".to_string());
222
223        // Create listener early to validate address before model loading
224        let listener = tokio::net::TcpListener::bind(format!("{ip}:{port}")).await?;
225
226        let app = MistralRsServerRouterBuilder::new()
227            .with_mistralrs(mistralrs)
228            .build()
229            .await?;
230
231        info!("OpenAI-compatible server listening on http://{ip}:{port}.");
232
233        tokio::spawn(async move {
234            if let Err(e) = axum::serve(listener, app).await {
235                eprintln!("OpenAI server error: {e}");
236            }
237        })
238    } else {
239        tokio::spawn(async {})
240    };
241
242    let (_, _) = join!(oai_port, mcp_port);
243
244    Ok(())
245}
