168/// Load MCP configuration from file path or environment variable
169fn load_mcp_config(mcp_config_path: Option<&str>) -> Result<Option<McpClientConfig>> {
170    let config_path = if let Some(path) = mcp_config_path {
171        Some(path.to_string())
172    } else {
173        // Check environment variable if no CLI arg provided
174        std::env::var("MCP_CONFIG_PATH").ok()
175    };
176
177    if let Some(path) = config_path {
178        match std::fs::read_to_string(&path) {
179            Ok(config_content) => {
180                match serde_json::from_str::<McpClientConfig>(&config_content) {
181                    Ok(config) => {
182                        // Validate configuration
183                        if let Err(e) = validate_mcp_config(&config) {
184                            error!("MCP configuration validation failed: {}", e);
185                            anyhow::bail!("Invalid MCP configuration: {}", e);
186                        }
187
188                        info!("Loaded and validated MCP configuration from {}", path);
189                        info!("Configured {} MCP servers", config.servers.len());
190                        Ok(Some(config))
191                    }
192                    Err(e) => {
193                        error!("Failed to parse MCP configuration: {}", e);
194                        error!("Please check your JSON syntax and ensure it matches the MCP configuration schema");
195                        anyhow::bail!("Invalid MCP configuration format: {}", e);
196                    }
197                }
198            }
199            Err(e) => {
200                error!("Failed to read MCP configuration file {}: {}", path, e);
201                error!("Please ensure the file exists and is readable");
202                anyhow::bail!("Cannot read MCP configuration file: {}", e);
203            }
204        }
205    } else {
206        Ok(None)
207    }
208}

210/// Validate MCP configuration for common issues
211fn validate_mcp_config(config: &McpClientConfig) -> Result<()> {
212    use std::collections::HashSet;
213
214    // Check for duplicate server IDs
215    let mut seen_ids = HashSet::new();
216    for server in &config.servers {
217        if !seen_ids.insert(&server.id) {
218            anyhow::bail!("Duplicate server ID: {}", server.id);
219        }
220
221        // Validate server ID format
222        if !server
223            .id
224            .chars()
225            .all(|c| c.is_alphanumeric() || c == '-' || c == '_')
226        {
227            anyhow::bail!(
228                "Invalid server ID '{}': must contain only alphanumeric, hyphen, underscore",
229                server.id
230            );
231        }
232
233        // Validate URLs for HTTP/WebSocket sources
234        match &server.source {
235            mistralrs_core::McpServerSource::Http { url, .. }
236            | mistralrs_core::McpServerSource::WebSocket { url, .. } => {
237                // Basic URL validation - check for scheme
238                if !url.starts_with("http://")
239                    && !url.starts_with("https://")
240                    && !url.starts_with("ws://")
241                    && !url.starts_with("wss://")
242                {
243                    anyhow::bail!("Invalid URL for server '{}': must start with http://, https://, ws://, or wss://", server.id);
244                }
245                if url.len() < 10 {
246                    anyhow::bail!("Invalid URL for server '{}': URL too short", server.id);
247                }
248            }
249            mistralrs_core::McpServerSource::Process { command, .. } => {
250                if command.is_empty() {
251                    anyhow::bail!("Empty command for server '{}'", server.id);
252                }
253            }
254        }
255    }
256
257    // Validate global settings
258    if let Some(timeout) = config.tool_timeout_secs {
259        if timeout == 0 {
260            anyhow::bail!("tool_timeout_secs must be greater than 0");
261        }
262    }
263
264    if let Some(max_calls) = config.max_concurrent_calls {
265        if max_calls == 0 {
266            anyhow::bail!("max_concurrent_calls must be greater than 0");
267        }
268    }
269
270    Ok(())
271}

291/// Load multi-model configuration from file
292fn load_multi_model_config(config_path: &str) -> Result<Vec<ModelConfig>> {
293    let config_content = std::fs::read_to_string(config_path).map_err(|e| {
294        anyhow::anyhow!(
295            "Failed to read multi-model config file {}: {}",
296            config_path,
297            e
298        )
299    })?;
300
301    let configs_parsed: HashMap<String, ModelConfigParsed> = serde_json::from_str(&config_content)
302        .map_err(|e| anyhow::anyhow!("Failed to parse multi-model config: {}", e))?;
303
304    if configs_parsed.is_empty() {
305        anyhow::bail!("Multi-model configuration file is empty");
306    }
307
308    let mut configs = Vec::new();
309    for (model_id, parsed_config) in configs_parsed {
310        let config = ModelConfig {
311            model_id,
312            alias: parsed_config.alias,
313            model: parsed_config.model,
314            chat_template: parsed_config.chat_template,
315            jinja_explicit: parsed_config.jinja_explicit,
316            num_device_layers: parsed_config.num_device_layers,
317            in_situ_quant: parsed_config.in_situ_quant,
318        };
319        configs.push(config);
320    }
321
322    info!(
323        "Loaded multi-model configuration with {} models",
324        configs.len()
325    );
326    Ok(configs)
327}

329#[tokio::main]
330async fn main() -> Result<()> {
331    let args = Args::parse();
332
333    initialize_logging();
334
335    // Load MCP configuration if provided
336    let mcp_config = load_mcp_config(args.mcp_config.as_deref())?;
337
338    let paged_attn = configure_paged_attn_from_flags(args.paged_attn, args.no_paged_attn)?;
339
340    let mistralrs = match args.model {
341        ModelSelected::MultiModel {
342            config,
343            default_model_id,
344        } => {
345            // Multi-model mode
346            let model_configs = load_multi_model_config(&config)?;
347
348            let mut builder = MistralRsForServerBuilder::new()
349                .with_max_seqs(args.max_seqs)
350                .with_no_kv_cache(args.no_kv_cache)
351                .with_token_source(args.token_source)
352                .with_interactive_mode(args.interactive_mode)
353                .with_prefix_cache_n(args.prefix_cache_n)
354                .set_paged_attn(paged_attn)
355                .with_cpu(args.cpu)
356                .with_enable_search(args.enable_search)
357                .with_seed_optional(args.seed)
358                .with_log_optional(args.log)
359                .with_mcp_config_optional(mcp_config)
360                .with_paged_attn_cache_type(args.cache_type.unwrap_or_default());
361
362            // Add models to builder
363            for config in model_configs {
364                builder = builder.add_model_config(config);
365            }
366
367            // Set default model if specified
368            if let Some(default_id) = default_model_id {
369                builder = builder.with_default_model_id(default_id);
370            }
371
372            if let Some(model) = args.search_embedding_model {
373                builder = builder.with_search_embedding_model(model);
374            }
375
376            builder.build_multi_model().await?
377        }
378        model => {
379            // Single-model mode
380            let mut builder = MistralRsForServerBuilder::new()
381                .with_model(model)
382                .with_max_seqs(args.max_seqs)
383                .with_no_kv_cache(args.no_kv_cache)
384                .with_token_source(args.token_source)
385                .with_interactive_mode(args.interactive_mode)
386                .with_prefix_cache_n(args.prefix_cache_n)
387                .set_paged_attn(paged_attn)
388                .with_cpu(args.cpu)
389                .with_enable_search(args.enable_search)
390                .with_seed_optional(args.seed)
391                .with_log_optional(args.log)
392                .with_chat_template_optional(args.chat_template)
393                .with_jinja_explicit_optional(args.jinja_explicit)
394                .with_num_device_layers_optional(args.num_device_layers)
395                .with_in_situ_quant_optional(args.in_situ_quant)
396                .with_paged_attn_gpu_mem_optional(args.paged_attn_gpu_mem)
397                .with_paged_attn_gpu_mem_usage_optional(args.paged_attn_gpu_mem_usage)
398                .with_paged_ctxt_len_optional(args.paged_ctxt_len)
399                .with_paged_attn_block_size_optional(args.paged_attn_block_size)
400                .with_mcp_config_optional(mcp_config)
401                .with_paged_attn_cache_type(args.cache_type.unwrap_or_default());
402
403            if let Some(model) = args.search_embedding_model {
404                builder = builder.with_search_embedding_model(model);
405            }
406
407            builder.build().await?
408        }
409    };
410
411    // TODO: refactor this
412    let search_embedding_model =
413        get_search_embedding_model(args.enable_search, args.search_embedding_model);
414
415    if args.interactive_mode {
416        interactive_mode(
417            mistralrs,
418            search_embedding_model.is_some(),
419            args.enable_thinking.then_some(true),
420        )
421        .await;
422        return Ok(());
423    }
424
425    if !args.interactive_mode && args.port.is_none() && args.mcp_port.is_none() {
426        anyhow::bail!("Interactive mode was not specified, so expected port to be specified. Perhaps you forgot `-i` or `--port` or `--mcp-port`?")
427    }
428
429    let mcp_port = if let Some(port) = args.mcp_port {
430        let host = args
431            .serve_ip
432            .clone()
433            .unwrap_or_else(|| "0.0.0.0".to_string());
434        info!("MCP server listening on http://{host}:{port}/mcp.");
435        info!("MCP protocol version is {}.", LATEST_PROTOCOL_VERSION);
436        let mcp_server = mcp_server::create_http_mcp_server(mistralrs.clone(), host, port);
437
438        tokio::spawn(async move {
439            if let Err(e) = mcp_server.await {
440                eprintln!("MCP server error: {e}");
441            }
442        })
443    } else {
444        tokio::spawn(async {})
445    };
446
447    let oai_port = if let Some(port) = args.port {
448        let ip = args
449            .serve_ip
450            .clone()
451            .unwrap_or_else(|| "0.0.0.0".to_string());
452
453        // Create listener early to validate address before model loading
454        let listener = tokio::net::TcpListener::bind(format!("{ip}:{port}")).await?;
455
456        let app = MistralRsServerRouterBuilder::new()
457            .with_mistralrs(mistralrs)
458            .build()
459            .await?;
460
461        info!("OpenAI-compatible server listening on http://{ip}:{port}.");
462
463        tokio::spawn(async move {
464            if let Err(e) = axum::serve(listener, app).await {
465                eprintln!("OpenAI server error: {e}");
466            }
467        })
468    } else {
469        tokio::spawn(async {})
470    };
471
472    let (_, _) = join!(oai_port, mcp_port);
473
474    Ok(())
