207#[derive(Debug, Clone, Deserialize, Serialize, ToSchema)]
208struct ReIsqRequest {
209    #[schema(example = "Q4K")]
210    ggml_type: String,
211}
212
213#[utoipa::path(
214    post,
215    tag = "Mistral.rs",
216    path = "/re_isq",
217    request_body = ReIsqRequest,
218    responses((status = 200, description = "Reapply ISQ to a non GGUF or GGML model."))
219)]
220async fn re_isq(
221    State(state): State<Arc<MistralRs>>,
222    Json(request): Json<ReIsqRequest>,
223) -> Result<String, String> {
224    let repr = format!("Re ISQ: {:?}", request.ggml_type);
225    MistralRs::maybe_log_request(state.clone(), repr.clone());
226    let request = Request::ReIsq(parse_isq_value(&request.ggml_type, None)?);
227    state.get_sender().unwrap().send(request).await.unwrap();
228    Ok(repr)
229}

231fn get_router(state: Arc<MistralRs>) -> Router {
232    #[derive(OpenApi)]
233    #[openapi(
234        paths(models, health, chatcompletions),
235        components(
236            schemas(ModelObjects, ModelObject, ChatCompletionRequest, CompletionRequest, ImageGenerationRequest, StopTokens, Message)),
237        tags(
238            (name = "Mistral.rs", description = "Mistral.rs API")
239        ),
240        info(
241            title = "Mistral.rs",
242            license(
243            name = "MIT",
244        )
245        )
246    )]
247    struct ApiDoc;
248
249    let doc = { ApiDoc::openapi() };
250
251    let allow_origin = AllowOrigin::any();
252    let cors_layer = CorsLayer::new()
253        .allow_methods([Method::GET, Method::POST])
254        .allow_headers([http::header::CONTENT_TYPE, http::header::AUTHORIZATION])
255        .allow_origin(allow_origin);
256
257    Router::new()
258        .merge(SwaggerUi::new("/docs").url("/api-doc/openapi.json", doc))
259        .route("/v1/chat/completions", post(chatcompletions))
260        .route("/v1/completions", post(completions))
261        .route("/v1/models", get(models))
262        .route("/health", get(health))
263        .route("/", get(health))
264        .route("/re_isq", post(re_isq))
265        .route("/v1/images/generations", post(image_generation))
266        .route("/v1/audio/speech", post(speech_generation))
267        .layer(cors_layer)
268        .layer(DefaultBodyLimit::max(N_INPUT_SIZE * MB_TO_B))
269        .with_state(state)
270}

272#[tokio::main]
273async fn main() -> Result<()> {
274    let mut args = Args::parse();
275    initialize_logging();
276
277    let tgt_non_granular_index = get_tgt_non_granular_index(&args.model);
278    let dtype = get_model_dtype(&args.model)?;
279    let auto_device_map_params = get_auto_device_map_params(&args.model)?;
280
281    if tgt_non_granular_index.is_some() {
282        args.max_seqs = 1;
283    }
284
285    let prompt_chunksize = match args.prompt_chunksize {
286        Some(0) => {
287            anyhow::bail!("`prompt_chunksize` must be a strictly positive integer, got 0.",)
288        }
289        Some(x) => Some(NonZeroUsize::new(x).unwrap()),
290        None => None,
291    };
292
293    let max_seq_len = auto_device_map_params.max_seq_len();
294
295    let loader: Box<dyn Loader> = LoaderBuilder::new(args.model)
296        .with_no_kv_cache(args.no_kv_cache)
297        .with_chat_template(args.chat_template)
298        .with_prompt_chunksize(prompt_chunksize)
299        .with_jinja_explicit(args.jinja_explicit)
300        .build()?;
301
302    #[cfg(feature = "metal")]
303    let device = if args.cpu {
304        args.no_paged_attn = true;
305        Device::Cpu
306    } else {
307        Device::new_metal(0)?
308    };
309    #[cfg(not(feature = "metal"))]
310    let device = if args.cpu {
311        args.no_paged_attn = true;
312        Device::Cpu
313    } else if mistralrs_core::distributed::use_nccl() {
314        Device::Cpu
315    } else {
316        Device::cuda_if_available(0)?
317    };
318
319    if let Some(seed) = args.seed {
320        device.set_seed(seed)?;
321    }
322
323    info!(
324        "avx: {}, neon: {}, simd128: {}, f16c: {}",
325        candle_core::utils::with_avx(),
326        candle_core::utils::with_neon(),
327        candle_core::utils::with_simd128(),
328        candle_core::utils::with_f16c()
329    );
330    info!("Sampling method: penalties -> temperature -> topk -> topp -> minp -> multinomial");
331    info!("Model kind is: {}", loader.get_kind().to_string());
332
333    // Parse device mapper
334    let mapper = if let Some(device_layers) = args.num_device_layers {
335        if device_layers.len() == 1 && device_layers[0].parse::<usize>().is_ok() {
336            let layers = device_layers[0].parse::<usize>().unwrap();
337            DeviceMapSetting::Map(DeviceMapMetadata::from_num_device_layers(vec![
338                DeviceLayerMapMetadata { ordinal: 0, layers },
339            ]))
340        } else {
341            let mut mapping = Vec::new();
342            for layer in device_layers {
343                let split = layer.splitn(2, ':').collect::<Vec<_>>();
344                if split.len() < 2 {
345                    panic!("Expected layer to be of format ORD:NUM, got {layer}");
346                }
347                let ord = split[0]
348                    .parse::<usize>()
349                    .unwrap_or_else(|_| panic!("Failed to parse {} as integer.", split[0]));
350                let num = split[1]
351                    .parse::<usize>()
352                    .unwrap_or_else(|_| panic!("Failed to parse {} as integer.", split[1]));
353                for DeviceLayerMapMetadata { ordinal, layers: _ } in &mapping {
354                    if *ordinal == ord {
355                        panic!("Duplicate ordinal {ord}");
356                    }
357                }
358                mapping.push(DeviceLayerMapMetadata {
359                    ordinal: ord,
360                    layers: num,
361                });
362            }
363            DeviceMapSetting::Map(DeviceMapMetadata::from_num_device_layers(mapping))
364        }
365    } else {
366        DeviceMapSetting::Auto(auto_device_map_params)
367    };
368
369    let no_paged_attn = if device.is_cuda() || mistralrs_core::distributed::use_nccl() {
370        args.no_paged_attn
371    } else if device.is_metal() {
372        !args.paged_attn
373    } else {
374        true
375    };
376
377    // Allocate 0.5 GB of CPU memory just as a placeholder.
378    // Nothing happens here as we have no `swap_out`, see `_preempt_by_swap`.
379    let cache_config = match (
380        args.paged_attn_block_size,
381        args.paged_attn_gpu_mem,
382        args.paged_attn_gpu_mem_usage,
383        args.paged_ctxt_len,
384        paged_attn_supported(),
385        no_paged_attn,
386    ) {
387        (block_size, None, None, None, true, false) => Some(PagedAttentionConfig::new(
388            block_size,
389            512,
390            MemoryGpuConfig::ContextSize(max_seq_len),
391        )?),
392        (block_size, None, None, Some(ctxt), true, false) => Some(PagedAttentionConfig::new(
393            block_size,
394            512,
395            MemoryGpuConfig::ContextSize(ctxt),
396        )?),
397        (block_size, None, Some(f), None, true, false) => Some(PagedAttentionConfig::new(
398            block_size,
399            512,
400            MemoryGpuConfig::Utilization(f),
401        )?),
402        (block_size, Some(m), None, None, true, false) => Some(PagedAttentionConfig::new(
403            block_size,
404            512,
405            MemoryGpuConfig::MbAmount(m),
406        )?),
407        (block_size, Some(_m), Some(f), None, true, false) => {
408            info!("Both memory size, and usage were specified, defaulting to the usage value.");
409            Some(PagedAttentionConfig::new(
410                block_size,
411                512,
412                MemoryGpuConfig::Utilization(f),
413            )?)
414        }
415        (block_size, Some(_m), None, Some(ctxt), true, false) => {
416            info!("All memory size and ctxt len, defaulting to the context len value.");
417            Some(PagedAttentionConfig::new(
418                block_size,
419                512,
420                MemoryGpuConfig::ContextSize(ctxt),
421            )?)
422        }
423        (block_size, None, Some(f), Some(_ctxt), true, false) => {
424            info!("Both ctxt len and usage were specified, defaulting to the usage value.");
425            Some(PagedAttentionConfig::new(
426                block_size,
427                512,
428                MemoryGpuConfig::Utilization(f),
429            )?)
430        }
431        (_, _, _, _, _, _) => None,
432    };
433
434    let isq = args
435        .in_situ_quant
436        .as_ref()
437        .and_then(|isq| parse_isq_value(isq, Some(&device)).ok());
438
439    let pipeline = loader.load_model_from_hf(
440        None,
441        args.token_source,
442        &dtype,
443        &device,
444        false,
445        mapper,
446        isq,
447        cache_config,
448    )?;
449    info!("Model loaded.");
450
451    let scheduler_config = if cache_config.is_some() {
452        // Handle case where we may have device mapping
453        if let Some(ref cache_config) = pipeline.lock().await.get_metadata().cache_config {
454            SchedulerConfig::PagedAttentionMeta {
455                max_num_seqs: args.max_seqs,
456                config: cache_config.clone(),
457            }
458        } else {
459            SchedulerConfig::DefaultScheduler {
460                method: DefaultSchedulerMethod::Fixed(args.max_seqs.try_into().unwrap()),
461            }
462        }
463    } else {
464        SchedulerConfig::DefaultScheduler {
465            method: DefaultSchedulerMethod::Fixed(args.max_seqs.try_into().unwrap()),
466        }
467    };
468    let bert_model = if args.enable_search {
469        Some(
470            args.search_bert_model
471                .map(BertEmbeddingModel::Custom)
472                .unwrap_or_default(),
473        )
474    } else {
475        None
476    };
477    // Throughput logging in the server
478    let mistralrs = MistralRsBuilder::new(
479        pipeline,
480        scheduler_config,
481        !args.interactive_mode,
482        bert_model.clone(),
483    )
484    .with_opt_log(args.log)
485    .with_truncate_sequence(args.truncate_sequence)
486    .with_no_kv_cache(args.no_kv_cache)
487    .with_prefix_cache_n(args.prefix_cache_n)
488    .build();
489
490    if args.interactive_mode {
491        interactive_mode(
492            mistralrs,
493            bert_model.is_some(),
494            args.enable_thinking.then_some(true),
495        )
496        .await;
497        return Ok(());
498    }
499
500    // Needs to be after the .build call as that is where the daemon waits.
501    let setting_server = if !args.interactive_mode {
502        let port = args.port.expect("Interactive mode was not specified, so expected port to be specified. Perhaps you forgot `-i` or `--port`?");
503        let ip = args.serve_ip.unwrap_or_else(|| "0.0.0.0".to_string());
504
505        // Create listener early to validate address before model loading
506        let listener = tokio::net::TcpListener::bind(format!("{ip}:{port}")).await?;
507        Some((listener, ip, port))
508    } else {
509        None
510    };
511
512    let app = get_router(mistralrs);
513    if let Some((listener, ip, port)) = setting_server {
514        info!("Serving on http://{ip}:{}.", port);
515        axum::serve(listener, app).await?;
516    };
517
518    Ok(())
519}
