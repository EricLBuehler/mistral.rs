202// Helper function to create a grammar variant schema
203fn create_grammar_variant_schema(type_value: &str, value_schema: Schema) -> Schema {
204    Schema::Object(
205        ObjectBuilder::new()
206            .schema_type(SchemaType::Object)
207            .property(
208                "type",
209                RefOr::T(Schema::Object(
210                    ObjectBuilder::new()
211                        .schema_type(SchemaType::String)
212                        .enum_values(Some(vec![serde_json::Value::String(
213                            type_value.to_string(),
214                        )]))
215                        .build(),
216                )),
217            )
218            .property("value", RefOr::T(value_schema))
219            .required("type")
220            .required("value")
221            .build(),
222    )
223}

225fn llguidance_schema() -> Schema {
226    let grammar_with_lexer_schema = Schema::Object(
227        ObjectBuilder::new()
228            .schema_type(SchemaType::Object)
229            .property(
230                "name",
231                RefOr::T(Schema::Object(
232                    ObjectBuilder::new()
233                        .schema_type(SchemaType::String)
234                        .nullable(true)
235                        .description(Some(
236                            "The name of this grammar, can be used in GenGrammar nodes",
237                        ))
238                        .build(),
239                )),
240            )
241            .property(
242                "json_schema",
243                RefOr::T(Schema::Object(
244                    ObjectBuilder::new()
245                        .schema_type(SchemaType::Object)
246                        .nullable(true)
247                        .description(Some("The JSON schema that the grammar should generate"))
248                        .build(),
249                )),
250            )
251            .property(
252                "lark_grammar",
253                RefOr::T(Schema::Object(
254                    ObjectBuilder::new()
255                        .schema_type(SchemaType::String)
256                        .nullable(true)
257                        .description(Some("The Lark grammar that the grammar should generate"))
258                        .build(),
259                )),
260            )
261            .description(Some("Grammar configuration with lexer settings"))
262            .build(),
263    );
264
265    Schema::Object(
266        ObjectBuilder::new()
267            .schema_type(SchemaType::Object)
268            .property(
269                "grammars",
270                RefOr::T(Schema::Array(
271                    ArrayBuilder::new()
272                        .items(RefOr::T(grammar_with_lexer_schema))
273                        .description(Some("List of grammar configurations"))
274                        .build(),
275                )),
276            )
277            .property(
278                "max_tokens",
279                RefOr::T(Schema::Object(
280                    ObjectBuilder::new()
281                        .schema_type(SchemaType::Integer)
282                        .nullable(true)
283                        .description(Some("Maximum number of tokens to generate"))
284                        .build(),
285                )),
286            )
287            .required("grammars")
288            .description(Some("Top-level grammar configuration for LLGuidance"))
289            .build(),
290    )
291}

490#[derive(Debug, Clone, Copy, PartialEq, Eq, Serialize, Deserialize, Default, ToSchema)]
491#[serde(rename_all = "lowercase")]
492pub enum AudioResponseFormat {
493    #[default]
494    Mp3,
495    Opus,
496    Aac,
497    Flac,
498    Wav,
499    Pcm,
500}
501
502impl AudioResponseFormat {
503    pub fn audio_content_type(
504        &self,
505        pcm_rate: usize,
506        pcm_channels: usize,
507        pcm_format: &'static str,
508    ) -> String {
509        let content_type = match &self {
510            AudioResponseFormat::Mp3 => "audio/mpeg".to_string(),
511            AudioResponseFormat::Opus => "audio/ogg; codecs=opus".to_string(),
512            AudioResponseFormat::Aac => "audio/aac".to_string(),
513            AudioResponseFormat::Flac => "audio/flac".to_string(),
514            AudioResponseFormat::Wav => "audio/wav".to_string(),
515            AudioResponseFormat::Pcm => format!("audio/pcm; codecs=1; format={pcm_format}"),
516        };
517
518        format!("{content_type}; rate={pcm_rate}; channels={pcm_channels}")
519    }
520}
