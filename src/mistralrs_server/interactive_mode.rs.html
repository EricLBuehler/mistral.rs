33pub async fn interactive_mode(
34    mistralrs: Arc<MistralRs>,
35    do_search: bool,
36    enable_thinking: Option<bool>,
37) {
38    match mistralrs.get_model_category() {
39        ModelCategory::Text => text_interactive_mode(mistralrs, do_search, enable_thinking).await,
40        ModelCategory::Vision { .. } => {
41            vision_interactive_mode(mistralrs, do_search, enable_thinking).await
42        }
43        ModelCategory::Diffusion => diffusion_interactive_mode(mistralrs, do_search).await,
44    }
45}

89async fn text_interactive_mode(
90    mistralrs: Arc<MistralRs>,
91    do_search: bool,
92    enable_thinking: Option<bool>,
93) {
94    let sender = mistralrs.get_sender().unwrap();
95    let mut messages: Vec<IndexMap<String, MessageContent>> = Vec::new();
96
97    let sampling_params = SamplingParams {
98        temperature: Some(0.1),
99        top_k: Some(32),
100        top_p: Some(0.1),
101        min_p: Some(0.05),
102        top_n_logprobs: 0,
103        frequency_penalty: Some(0.1),
104        presence_penalty: Some(0.1),
105        max_len: Some(4096),
106        stop_toks: None,
107        logits_bias: None,
108        n_choices: 1,
109        dry_params: Some(DrySamplingParams::default()),
110    };
111
112    info!("Starting interactive loop with sampling params: {sampling_params:?}");
113    println!(
114        "{}{TEXT_INTERACTIVE_HELP}{}",
115        "=".repeat(20),
116        "=".repeat(20)
117    );
118
119    // Set the handler to process exit
120    *CTRLC_HANDLER.lock().unwrap() = &exit_handler;
121
122    ctrlc::set_handler(move || CTRLC_HANDLER.lock().unwrap()())
123        .expect("Failed to set CTRL-C handler for interactive mode");
124
125    'outer: loop {
126        // Set the handler to process exit
127        *CTRLC_HANDLER.lock().unwrap() = &exit_handler;
128
129        let mut prompt = String::new();
130        print!("> ");
131        io::stdout().flush().unwrap();
132        io::stdin()
133            .read_line(&mut prompt)
134            .expect("Failed to get input");
135
136        match prompt.as_str().trim() {
137            "" => continue,
138            HELP_CMD => {
139                println!(
140                    "{}{TEXT_INTERACTIVE_HELP}{}",
141                    "=".repeat(20),
142                    "=".repeat(20)
143                );
144                continue;
145            }
146            EXIT_CMD => {
147                break;
148            }
149            CLEAR_CMD => {
150                messages.clear();
151                info!("Cleared chat history.");
152                continue;
153            }
154            prompt if prompt.trim().starts_with(SYSTEM_CMD) => {
155                let parsed = match &prompt.split(SYSTEM_CMD).collect::<Vec<_>>()[..] {
156                    &["", a] => a.trim(),
157                    _ => {
158                        println!("Error: Setting the system command should be done with this format: `{SYSTEM_CMD} This is a system message.`");
159                        continue;
160                    }
161                };
162                info!("Set system message to `{parsed}`.");
163                let mut user_message: IndexMap<String, MessageContent> = IndexMap::new();
164                user_message.insert("role".to_string(), Either::Left("system".to_string()));
165                user_message.insert("content".to_string(), Either::Left(parsed.to_string()));
166                messages.push(user_message);
167                continue;
168            }
169            message => {
170                let mut user_message: IndexMap<String, MessageContent> = IndexMap::new();
171                user_message.insert("role".to_string(), Either::Left("user".to_string()));
172                user_message.insert("content".to_string(), Either::Left(message.to_string()));
173                messages.push(user_message);
174            }
175        }
176
177        // Set the handler to terminate all seqs, so allowing cancelling running
178        *CTRLC_HANDLER.lock().unwrap() = &terminate_handler;
179
180        let request_messages = RequestMessage::Chat {
181            messages: messages.clone(),
182            enable_thinking,
183        };
184
185        let (tx, mut rx) = channel(10_000);
186        let req = Request::Normal(NormalRequest {
187            id: mistralrs.next_request_id(),
188            messages: request_messages,
189            sampling_params: sampling_params.clone(),
190            response: tx,
191            return_logprobs: false,
192            is_streaming: true,
193            constraint: Constraint::None,
194            suffix: None,
195            tool_choice: None,
196            tools: None,
197            logits_processors: None,
198            return_raw_logits: false,
199            web_search_options: do_search.then(WebSearchOptions::default),
200        });
201        sender.send(req).await.unwrap();
202
203        let mut assistant_output = String::new();
204
205        let mut last_usage = None;
206        while let Some(resp) = rx.recv().await {
207            match resp {
208                Response::Chunk(chunk) => {
209                    last_usage = chunk.usage.clone();
210                    if let ChunkChoice {
211                        delta:
212                            Delta {
213                                content: Some(content),
214                                ..
215                            },
216                        finish_reason,
217                        ..
218                    } = &chunk.choices[0]
219                    {
220                        assistant_output.push_str(content);
221                        print!("{}", content);
222                        io::stdout().flush().unwrap();
223                        if finish_reason.is_some() {
224                            if matches!(finish_reason.as_ref().unwrap().as_str(), "length") {
225                                print!("...");
226                            }
227                            break;
228                        }
229                    }
230                }
231                Response::InternalError(e) => {
232                    error!("Got an internal error: {e:?}");
233                    break 'outer;
234                }
235                Response::ModelError(e, resp) => {
236                    error!("Got a model error: {e:?}, response: {resp:?}");
237                    break 'outer;
238                }
239                Response::ValidationError(e) => {
240                    error!("Got a validation error: {e:?}");
241                    break 'outer;
242                }
243                Response::Done(_) => unreachable!(),
244                Response::CompletionDone(_) => unreachable!(),
245                Response::CompletionModelError(_, _) => unreachable!(),
246                Response::CompletionChunk(_) => unreachable!(),
247                Response::ImageGeneration(_) => unreachable!(),
248                Response::Raw { .. } => unreachable!(),
249            }
250        }
251
252        if let Some(last_usage) = last_usage {
253            println!();
254            println!();
255            println!("Stats:");
256            println!(
257                "Prompt: {} tokens, {:.2} T/s",
258                last_usage.prompt_tokens, last_usage.avg_prompt_tok_per_sec
259            );
260            println!(
261                "Decode: {} tokens, {:.2} T/s",
262                last_usage.completion_tokens, last_usage.avg_compl_tok_per_sec
263            );
264        }
265        let mut assistant_message: IndexMap<String, Either<String, Vec<IndexMap<String, Value>>>> =
266            IndexMap::new();
267        assistant_message.insert("role".to_string(), Either::Left("assistant".to_string()));
268        assistant_message.insert("content".to_string(), Either::Left(assistant_output));
269        messages.push(assistant_message);
270        println!();
271    }
272}

274fn parse_image_urls_and_message(input: &str) -> (Vec<String>, String) {
275    // Capture HTTP/HTTPS URLs and local file paths ending with common image extensions
276    let re = Regex::new(r#"((?:https?://|file://)?\S+\.(?:png|jpe?g|bmp|gif|webp))"#).unwrap();
277    // Collect all URLs
278    let urls: Vec<String> = re
279        .captures_iter(input)
280        .filter_map(|cap| cap.get(1).map(|m| m.as_str().to_string()))
281        .collect();
282    // Remove the URLs from the input to get the message text
283    let text = re.replace_all(input, "").trim().to_string();
284    (urls, text)
285}

287async fn vision_interactive_mode(
288    mistralrs: Arc<MistralRs>,
289    do_search: bool,
290    enable_thinking: Option<bool>,
291) {
292    let sender = mistralrs.get_sender().unwrap();
293    let mut messages: Vec<IndexMap<String, MessageContent>> = Vec::new();
294    let mut images = Vec::new();
295
296    let prefixer = match &mistralrs.config().category {
297        ModelCategory::Text | ModelCategory::Diffusion => {
298            panic!("`add_image_message` expects a vision model.")
299        }
300        ModelCategory::Vision {
301            has_conv2d: _,
302            prefixer,
303        } => prefixer,
304    };
305
306    let sampling_params = SamplingParams {
307        temperature: Some(0.1),
308        top_k: Some(32),
309        top_p: Some(0.1),
310        min_p: Some(0.05),
311        top_n_logprobs: 0,
312        frequency_penalty: Some(0.1),
313        presence_penalty: Some(0.1),
314        max_len: Some(4096),
315        stop_toks: None,
316        logits_bias: None,
317        n_choices: 1,
318        dry_params: Some(DrySamplingParams::default()),
319    };
320
321    info!("Starting interactive loop with sampling params: {sampling_params:?}");
322    println!(
323        "{}{VISION_INTERACTIVE_HELP}{}",
324        "=".repeat(20),
325        "=".repeat(20)
326    );
327
328    // Set the handler to process exit
329    *CTRLC_HANDLER.lock().unwrap() = &exit_handler;
330
331    ctrlc::set_handler(move || CTRLC_HANDLER.lock().unwrap()())
332        .expect("Failed to set CTRL-C handler for interactive mode");
333
334    'outer: loop {
335        // Set the handler to process exit
336        *CTRLC_HANDLER.lock().unwrap() = &exit_handler;
337
338        let mut prompt = String::new();
339        print!("> ");
340        io::stdout().flush().unwrap();
341        io::stdin()
342            .read_line(&mut prompt)
343            .expect("Failed to get input");
344
345        match prompt.as_str().trim() {
346            "" => continue,
347            HELP_CMD => {
348                println!(
349                    "{}{VISION_INTERACTIVE_HELP}{}",
350                    "=".repeat(20),
351                    "=".repeat(20)
352                );
353                continue;
354            }
355            EXIT_CMD => {
356                break;
357            }
358            CLEAR_CMD => {
359                messages.clear();
360                info!("Cleared chat history.");
361                continue;
362            }
363            prompt if prompt.trim().starts_with(SYSTEM_CMD) => {
364                let parsed = match &prompt.split(SYSTEM_CMD).collect::<Vec<_>>()[..] {
365                    &["", a] => a.trim(),
366                    _ => {
367                        println!("Error: Setting the system command should be done with this format: `{SYSTEM_CMD} This is a system message.`");
368                        continue;
369                    }
370                };
371                info!("Set system message to `{parsed}`.");
372                let mut user_message: IndexMap<String, MessageContent> = IndexMap::new();
373                user_message.insert("role".to_string(), Either::Left("system".to_string()));
374                user_message.insert("content".to_string(), Either::Left(parsed.to_string()));
375                messages.push(user_message);
376                continue;
377            }
378            // Extract any image URLs and the remaining text
379            _ => {
380                let (urls, text) = parse_image_urls_and_message(prompt.trim());
381                if !urls.is_empty() {
382                    let mut image_indexes = Vec::new();
383                    // Load all images first
384                    for url in &urls {
385                        match util::parse_image_url(url).await {
386                            Ok(image) => {
387                                image_indexes.push(images.len());
388                                images.push(image);
389                            }
390                            Err(e) => {
391                                error!("Failed to read image from URL/path {}: {}", url, e);
392                                continue 'outer;
393                            }
394                        }
395                    }
396                    // Build a single user message with multiple images and then the text
397                    let mut content_vec: Vec<IndexMap<String, Value>> = Vec::new();
398                    // Add one image part per URL
399                    for _ in &urls {
400                        content_vec.push(IndexMap::from([(
401                            "type".to_string(),
402                            Value::String("image".to_string()),
403                        )]));
404                    }
405                    // Add the text part once
406                    let text = prefixer.prefix_image(image_indexes, &text);
407                    content_vec.push(IndexMap::from([
408                        ("type".to_string(), Value::String("text".to_string())),
409                        ("text".to_string(), Value::String(text.clone())),
410                    ]));
411                    let mut user_message: IndexMap<String, MessageContent> = IndexMap::new();
412                    user_message.insert("role".to_string(), Either::Left("user".to_string()));
413                    user_message.insert("content".to_string(), Either::Right(content_vec));
414                    messages.push(user_message);
415                } else {
416                    // Default: handle as text-only prompt
417                    let mut user_message: IndexMap<String, MessageContent> = IndexMap::new();
418                    user_message.insert("role".to_string(), Either::Left("user".to_string()));
419                    user_message.insert(
420                        "content".to_string(),
421                        Either::Left(prompt.trim().to_string()),
422                    );
423                    messages.push(user_message);
424                }
425            }
426        };
427
428        // Set the handler to terminate all seqs, so allowing cancelling running
429        *CTRLC_HANDLER.lock().unwrap() = &terminate_handler;
430

433            messages: messages.clone(),
434            enable_thinking,
435        };

438        let req = Request::Normal(NormalRequest {
439            id: mistralrs.next_request_id(),
440            messages: request_messages,
441            sampling_params: sampling_params.clone(),
442            response: tx,
443            return_logprobs: false,
444            is_streaming: true,

447            tool_choice: None,
448            tools: None,
449            logits_processors: None,
450            return_raw_logits: false,
451            web_search_options: do_search.then(WebSearchOptions::default),
452        });
453        sender.send(req).await.unwrap();
454
455        let mut assistant_output = String::new();

458        while let Some(resp) = rx.recv().await {

460                Response::Chunk(chunk) => {

526async fn diffusion_interactive_mode(mistralrs: Arc<MistralRs>, do_search: bool) {
527    let sender = mistralrs.get_sender().unwrap();
528
529    let diffusion_params = DiffusionGenerationParams::default();
530
531    info!("Starting interactive loop with generation params: {diffusion_params:?}");
532    println!(
533        "{}{DIFFUSION_INTERACTIVE_HELP}{}",
534        "=".repeat(20),
535        "=".repeat(20)
536    );
537
538    // Set the handler to process exit
539    *CTRLC_HANDLER.lock().unwrap() = &exit_handler;
540
541    ctrlc::set_handler(move || CTRLC_HANDLER.lock().unwrap()())
542        .expect("Failed to set CTRL-C handler for interactive mode");
543
544    loop {
545        // Set the handler to process exit
546        *CTRLC_HANDLER.lock().unwrap() = &exit_handler;
547
548        let mut prompt = String::new();
549        print!("> ");
550        io::stdout().flush().unwrap();
551        io::stdin()
552            .read_line(&mut prompt)
553            .expect("Failed to get input");
554
555        let prompt = match prompt.as_str().trim() {
556            "" => continue,
557            HELP_CMD => {
558                println!(
559                    "{}{DIFFUSION_INTERACTIVE_HELP}{}",
560                    "=".repeat(20),
561                    "=".repeat(20)
562                );
563                continue;
564            }
565            EXIT_CMD => {
566                break;
567            }
568            prompt => prompt.to_string(),
569        };
570
571        // Set the handler to terminate all seqs, so allowing cancelling running
572        *CTRLC_HANDLER.lock().unwrap() = &terminate_handler;
573
574        let (tx, mut rx) = channel(10_000);
575        let req = Request::Normal(NormalRequest {
576            id: 0,
577            messages: RequestMessage::ImageGeneration {
578                prompt: prompt.to_string(),
579                format: ImageGenerationResponseFormat::Url,
580                generation_params: diffusion_params.clone(),
581            },
582            sampling_params: SamplingParams::deterministic(),
583            response: tx,
584            return_logprobs: false,
585            is_streaming: false,
586            suffix: None,
587            constraint: Constraint::None,
588            tool_choice: None,
589            tools: None,
590            logits_processors: None,
591            return_raw_logits: false,
592            web_search_options: do_search.then(WebSearchOptions::default),
593        });
594
595        let start = Instant::now();
596        sender.send(req).await.unwrap();
597
598        let ResponseOk::ImageGeneration(response) = rx.recv().await.unwrap().as_result().unwrap()
599        else {
600            panic!("Got unexpected response type.")
601        };
602        let end = Instant::now();
603
604        let duration = end.duration_since(start).as_secs_f32();
605        let pixels_per_s = (diffusion_params.height * diffusion_params.width) as f32 / duration;
606
607        println!(
608            "Image generated can be found at: image is at `{}`. Took {duration:.2}s ({pixels_per_s:.2} pixels/s).",
609            response.data[0].url.as_ref().unwrap(),
610        );
611
612        println!();
613    }
