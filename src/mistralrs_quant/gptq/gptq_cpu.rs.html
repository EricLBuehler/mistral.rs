69macro_rules! pack_factor {
70    ($bits:expr) => {
71        32 / $bits
72    };
73}
74
75pub fn gptq_linear(
76    in_dim: usize,
77    out_dim: usize,
78    config: &QuantizedConfig,
79    vb: ShardedVarBuilder,
80) -> Result<Arc<dyn QuantMethod>> {
81    // Handle the case where the layer is dummy (no tensors)
82    if !(vb.contains_tensor("qweight")
83        && vb.contains_tensor("qzeros")
84        && vb.contains_tensor("g_idx")
85        && vb.contains_tensor("scales"))
86    {
87        let layer = <DummyLayer as QuantMethod>::new(QuantMethodConfig::Dummy)?;
88        return Ok(Arc::new(layer) as Arc<dyn QuantMethod>);
89    }
90
91    let bits = config.bits.expect("GPTQ requires bits in config");
92    let qweight = vb.get_with_hints_dtype(
93        (in_dim / pack_factor!(bits), out_dim),
94        "qweight",
95        Default::default(),
96        DType::I32,
97    )?;
98    let scale_and_zero_size = in_dim
99        / config
100            .group_size
101            .expect("GPTQ requires group size in config");
102    let qzeros = vb.get_with_hints_dtype(
103        (scale_and_zero_size, out_dim / pack_factor!(bits)),
104        "qzeros",
105        Default::default(),
106        DType::I32,
107    )?;
108    let g_idx = vb.get_with_hints_dtype((in_dim,), "g_idx", Default::default(), DType::I32)?;
109    let scales = vb.get_with_hints_dtype(
110        (scale_and_zero_size, out_dim),
111        "scales",
112        Default::default(),
113        DType::F16,
114    )?;
115    let bias = if vb.contains_tensor("bias") {
116        Some(vb.get_with_hints_dtype((out_dim,), "bias", Default::default(), DType::F16)?)
117    } else {
118        None
119    };
120
121    let config = QuantMethodConfig::Gptq {
122        bits: bits as i32,
123        use_exllama: false,
124        q_weight: qweight,
125        gptq_qzeros: Some(qzeros),
126        gptq_scales: scales,
127        g_idx: Some(g_idx),
128        bias,
129        workspace: None,
130        is_marlin: false,
131    };
132    Ok(Arc::new(GptqLayer::new(config)?))
133}
