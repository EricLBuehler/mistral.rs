1use candle_core::{quantized::GgmlDType, Tensor};
2
3pub enum QuantizationBehaviour {
4    Quantize(GgmlDType),
5    Skip,
6}
7
8/// Return the fallback dtype for the given dtype.
9fn get_fallback(dtype: GgmlDType) -> QuantizationBehaviour {
10    // The normal `Q` quants are a bit more lenient than the `K` quants.
11    // => Try to fallback to a similar `Q` quant.
12    // If that's not possible, skip this tensor.
13    match dtype {
14        GgmlDType::Q2K => QuantizationBehaviour::Quantize(GgmlDType::Q4_0),
15        GgmlDType::Q3K => QuantizationBehaviour::Quantize(GgmlDType::Q4_0),
16        GgmlDType::Q4K => QuantizationBehaviour::Quantize(GgmlDType::Q4_1),
17        GgmlDType::Q5K => QuantizationBehaviour::Quantize(GgmlDType::Q5_0),
18        GgmlDType::Q6K => QuantizationBehaviour::Quantize(GgmlDType::Q5_1),
19        GgmlDType::Q8K => QuantizationBehaviour::Quantize(GgmlDType::Q8_1),
20        _ => QuantizationBehaviour::Skip,
21    }
22}
23
24/// Check if the tensor can be quantized with the given dtype.
25fn can_quantize(tensor: &Tensor, dtype: GgmlDType) -> bool {
26    let dims = tensor.shape().dims();
27    // The tensor must not be empty and the last dimension must be a multiple of the block size.
28    !(dims.is_empty() || (dims[dims.len() - 1] % dtype.block_size() != 0))
29}
30
31/// Check if we should quantize the tensor and if so, with which dtype.
32pub(crate) fn get_quantization_behaviour(
33    tensor: &Tensor,
34    dtype: GgmlDType,
35) -> QuantizationBehaviour {
36    if dtype == GgmlDType::F32 {
37        return QuantizationBehaviour::Skip;
38    }
39
40    if can_quantize(tensor, dtype) {
41        return QuantizationBehaviour::Quantize(dtype);
42    }
43    let fallback = get_fallback(dtype);
44    match fallback {
45        QuantizationBehaviour::Skip => fallback,
46        QuantizationBehaviour::Quantize(new_dtype) => get_quantization_behaviour(tensor, new_dtype),
47    }
48}

50#[macro_export]
51#[doc(hidden)]
52macro_rules! generate_isq {
53    ($tensor:expr, $device:expr, $dtype:expr, $n_quantized:expr, $guard:expr) => {
54        {
55            let quantization_behaviour = $crate::utils::isq::get_quantization_behaviour(&$tensor, $dtype);
56            let dtype = match quantization_behaviour{
57                $crate::utils::isq::QuantizationBehaviour::Skip => {
58                    let shape = $tensor.shape();
59                    tracing::warn!("Skipping quantization of tensor with shape {shape:?} as it is not quantizable.");
60                    GgmlDType::F32
61                },
62                $crate::utils::isq::QuantizationBehaviour::Quantize(dtype) => {
63                    $n_quantized.fetch_add(1, std::sync::atomic::Ordering::Relaxed);
64                    dtype
65                }
66            };
67
68            let initial = candle_core::quantized::QTensor::quantize(&$tensor, dtype)?;
69            let data = initial.data()?;
70
71            let _acquired_quantize_guard = $guard.acquire();
72            let qstorage = candle_core::quantized::QStorage::from_data(data, &$device, dtype)?;
73
74            Arc::new(candle_core::quantized::QTensor::new(qstorage, $tensor.shape())?)
75        }
76    };
77}
