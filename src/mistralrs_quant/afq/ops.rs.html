19/// Returns (w_q, scales, biases)
20pub(crate) fn afq_quantize_op(
21    w: &Tensor,
22    group_size: AfqGroupSize,
23    bits: AfqBits,
24) -> Result<(Tensor, Tensor, Tensor)> {
25    let group_size = group_size as usize;
26    let bits = bits as usize;
27
28    if w.rank() < 2 {
29        candle_core::bail!("AFQ quantize expects weight matrix of at least rank 2");
30    }
31    if w.dim(D::Minus1)? % group_size != 0 {
32        candle_core::bail!(
33            "Last dim of weight matrix ({:?}) must be divisible by group size {group_size}.",
34            w.dims()
35        );
36    }
37
38    #[cfg(feature = "metal")]
39    {
40        let w_s = w.storage_and_layout().0;
41        let Storage::Metal(w_s) = &*w_s else {
42            candle_core::bail!("expected metal")
43        };
44        let device = w_s.device();
45
46        let encoder = device.command_encoder()?;
47        encoder.set_label("afq-quantize");
48
49        let mut wq_shape = w.dims().to_vec();
50        *wq_shape.last_mut().unwrap() = w.dim(D::Minus1)? * bits / 32;
51        let mut s_shape = w.dims().to_vec();
52        *s_shape.last_mut().unwrap() = w.dim(D::Minus1)? / group_size;
53
54        let output =
55            device.new_buffer(wq_shape.iter().product(), DType::U32, "afq-quantize-output")?;
56        let scales =
57            device.new_buffer(s_shape.iter().product(), w.dtype(), "afq-quantize-scales")?;
58        let biases =
59            device.new_buffer(s_shape.iter().product(), w.dtype(), "afq-quantize-biases")?;
60
61        crate::metal_kernels::call_affine_quantize(
62            device.device(),
63            &encoder,
64            &crate::metal_kernels::Kernels::new(),
65            w.dtype(),
66            w_s.buffer(),
67            w.layout().start_offset() * w_s.dtype().size_in_bytes(),
68            w.dims(),
69            w.stride(),
70            &output,
71            &wq_shape,
72            &scales,
73            &biases,
74            false,
75            group_size,
76            bits,
77        )
78        .map_err(candle_core::Error::wrap)?;
79
80        let output = Tensor::from((
81            Storage::Metal(MetalStorage::new(
82                output,
83                device.clone(),
84                wq_shape.iter().product(),
85                DType::U32,
86            )),
87            Shape::from(wq_shape),
88        ));
89        let scales = Tensor::from((
90            Storage::Metal(MetalStorage::new(
91                scales,
92                device.clone(),
93                s_shape.iter().product(),
94                w.dtype(),
95            )),
96            Shape::from(s_shape.clone()),
97        ));
98        let biases = Tensor::from((
99            Storage::Metal(MetalStorage::new(
100                biases,
101                device.clone(),
102                s_shape.iter().product(),
103                w.dtype(),
104            )),
105            Shape::from(s_shape),
106        ));
107
108        return Ok((output, scales, biases));
109    }
110    #[cfg(feature = "cuda")]
111    if w.device().is_cuda() {
112        return cuda_backend::afq_quantize_op(w, group_size, bits);
113    }
114
115    // CPU fallback for non-accelerated devices
116    cpu_backend::afq_quantize_op(w, group_size, bits)
117}

119pub(crate) fn afq_dequantize_op(
120    w_q: &Tensor,
121    scales: &Tensor,
122    biases: &Tensor,
123    group_size: AfqGroupSize,
124    bits: AfqBits,
125) -> Result<Tensor> {
126    let group_size = group_size as usize;
127    let bits = bits as usize;
128
129    if w_q.rank() < 2 || scales.rank() < 2 || biases.rank() < 2 {
130        candle_core::bail!("AFQ dequantize expects all matrices of at least rank 2");
131    }
132
133    #[cfg(feature = "metal")]
134    {
135        let wq_s = w_q.storage_and_layout().0;
136        let Storage::Metal(wq_s) = &*wq_s else {
137            candle_core::bail!("expected metal")
138        };
139        let s_s = scales.storage_and_layout().0;
140        let Storage::Metal(s_s) = &*s_s else {
141            candle_core::bail!("expected metal")
142        };
143        let b_s = biases.storage_and_layout().0;
144        let Storage::Metal(b_s) = &*b_s else {
145            candle_core::bail!("expected metal")
146        };
147
148        let device = wq_s.device();
149
150        let encoder = device.command_encoder()?;
151        encoder.set_label("afq-dequantize");
152
153        let out_size = w_q.dim(D::Minus1)? * 32 / bits;
154        let mut w_shape = w_q.dims().to_vec();
155        *w_shape.last_mut().unwrap() = out_size;
156
157        if out_size != scales.dim(D::Minus1)? * group_size
158            || out_size != biases.dim(D::Minus1)? * group_size
159        {
160            candle_core::bail!(
161                "Scales and biases do not match the matrix given dequantization parameters."
162            );
163        }
164
165        let output = device.new_buffer(
166            w_shape.iter().product(),
167            scales.dtype(),
168            "afq-dequantize-output",
169        )?;
170
171        assert_eq!(w_q.layout().start_offset(), 0);
172        assert_eq!(scales.layout().start_offset(), 0);
173        assert_eq!(biases.layout().start_offset(), 0);
174        crate::metal_kernels::call_affine_quantize(
175            device.device(),
176            &encoder,
177            &crate::metal_kernels::Kernels::new(),
178            scales.dtype(),
179            wq_s.buffer(),
180            w_q.layout().start_offset() * wq_s.dtype().size_in_bytes(),
181            w_q.dims(),
182            w_q.stride(),
183            &output,
184            &w_shape,
185            s_s.buffer(),
186            b_s.buffer(),
187            true,
188            group_size,
189            bits,
190        )
191        .map_err(candle_core::Error::wrap)?;
192
193        let output = Tensor::from((
194            Storage::Metal(MetalStorage::new(
195                output,
196                device.clone(),
197                w_shape.iter().product(),
198                scales.dtype(),
199            )),
200            Shape::from(w_shape),
201        ));
202
203        return Ok(output);
204    }
205    #[cfg(feature = "cuda")]
206    if w_q.device().is_cuda() {
207        return cuda_backend::afq_dequantize_op(w_q, scales, biases, group_size, bits);
208    }
209
210    // CPU fallback for non-accelerated devices
211    cpu_backend::afq_dequantize_op(w_q, scales, biases, group_size, bits)
212}

231/// The indices lhs_indices and rhs_indices contain flat indices along the batch dimensions (i.e. all but the last two dimensions) of a and b respectively.
232#[allow(clippy::too_many_arguments)]
233pub(crate) fn afq_mm_op(
234    x: &Tensor,
235    w: &Tensor,
236    scales: &Tensor,
237    biases: &Tensor,
238    lhs_indices: Option<&Tensor>,
239    rhs_indices: Option<&Tensor>,
240    group_size: AfqGroupSize,
241    bits: AfqBits,
242    transpose: bool,
243) -> Result<Tensor> {
244    let group_size = group_size as usize;
245    let bits = bits as usize;
246
247    let w_outer_dims = {
248        if w.dtype() != DType::U32 {
249            candle_core::bail!("AFQ weight matrix must be u32");
250        }
251        if scales.dims() != biases.dims() {
252            candle_core::bail!("Scales and biases should have the same shapes");
253        }
254        if w.dim(D::Minus1)? * 32 / bits != scales.dim(D::Minus1)? * group_size {
255            candle_core::bail!("Last dims of w and scales must be compatible.");
256        }
257
258        let x_inner_dims = x.dim(D::Minus1)?;
259
260        // Calculate transpose w dims
261        let w_inner_dims = if transpose {
262            w.dim(D::Minus1)? * 32 / bits
263        } else {
264            w.dim(D::Minus2)?
265        };
266        let w_outer_dims = if transpose {
267            w.dim(D::Minus2)?
268        } else {
269            w.dim(D::Minus1)? * 32 / bits
270        };
271
272        if w_inner_dims != x_inner_dims {
273            candle_core::bail!(
274                "w inner dims ({:?}) must match x inner dims ({:?}). transpose={transpose}",
275                w.dims(),
276                x.dims()
277            );
278        }
279
280        w_outer_dims
281    };
282
283    #[cfg(feature = "metal")]
284    {
285        let x_s = x.storage_and_layout().0;
286        let Storage::Metal(x_s) = &*x_s else {
287            candle_core::bail!("expected metal")
288        };
289        let w_s = w.storage_and_layout().0;
290        let Storage::Metal(w_s) = &*w_s else {
291            candle_core::bail!("expected metal")
292        };
293        let s_s = scales.storage_and_layout().0;
294        let Storage::Metal(s_s) = &*s_s else {
295            candle_core::bail!("expected metal")
296        };
297        let b_s = biases.storage_and_layout().0;
298        let Storage::Metal(b_s) = &*b_s else {
299            candle_core::bail!("expected metal")
300        };
301
302        let device = w_s.device();
303
304        assert_eq!(w.layout().start_offset(), 0);
305        assert_eq!(scales.layout().start_offset(), 0);
306        assert_eq!(biases.layout().start_offset(), 0);
307
308        let (output, out_shape) = if lhs_indices.is_some() || rhs_indices.is_some() {
309            let mut lhs_indices = match lhs_indices {
310                Some(lhs_indices) => lhs_indices.clone(),
311                None => make_dummy_indices(x)?,
312            };
313            let mut rhs_indices = match rhs_indices {
314                Some(rhs_indices) => rhs_indices.clone(),
315                None => make_dummy_indices(w)?,
316            };
317            assert_eq!(lhs_indices.layout().start_offset(), 0);
318            assert_eq!(rhs_indices.layout().start_offset(), 0);
319            if lhs_indices.dtype() != DType::U32 || rhs_indices.dtype() != DType::U32 {
320                candle_core::bail!("lhs and rhs indices must be u32.")
321            }
322            // Broadcast the indices if applicable.
323            {
324                let mut shape = lhs_indices.shape().clone();
325                shape = rhs_indices
326                    .shape()
327                    .broadcast_shape_binary_op(rhs_indices.shape(), "afq-qmm")?;
328                lhs_indices = lhs_indices.broadcast_as(shape.clone())?;
329                rhs_indices = rhs_indices.broadcast_as(shape)?;
330            }
331
332            let li_s = lhs_indices.storage_and_layout().0;
333            let Storage::Metal(li_s) = &*li_s else {
334                candle_core::bail!("expected metal")
335            };
336            let ri_s = rhs_indices.storage_and_layout().0;
337            let Storage::Metal(ri_s) = &*ri_s else {
338                candle_core::bail!("expected metal")
339            };
340
341            let mut out_shape = lhs_indices.dims().to_vec();
342            out_shape.push(x.dim(D::Minus2)?);
343            out_shape.push(w_outer_dims);
344
345            let encoder = device.command_encoder()?;
346            encoder.set_label("afq-qmm");
347
348            let output =
349                device.new_buffer(out_shape.iter().product(), scales.dtype(), "afq-qmm-output")?;
350
351            crate::metal_kernels::call_afq_qmm(
352                device.device(),
353                &encoder,
354                &crate::metal_kernels::Kernels::new(),
355                scales.dtype(),
356                x_s.buffer(),
357                x.layout().start_offset() * x.dtype().size_in_bytes(),
358                x.dims(),
359                x.stride(),
360                w_s.buffer(),
361                w.dims(),
362                w.stride(),
363                s_s.buffer(),
364                scales.stride(),
365                b_s.buffer(),
366                biases.stride(),
367                &output,
368                &out_shape,
369                Some((li_s.buffer(), ri_s.buffer())),
370                Some(lhs_indices.dims()),
371                Some((lhs_indices.stride(), rhs_indices.stride())),
372                transpose,
373                bits,
374                group_size,
375            )
376            .map_err(candle_core::Error::wrap)?;
377
378            (output, out_shape)
379        } else {
380            let mut out_shape = x.dims().to_vec();
381            *out_shape.last_mut().unwrap() = w_outer_dims;
382
383            let encoder = device.command_encoder()?;
384            encoder.set_label("afq-qmm");
385
386            let output =
387                device.new_buffer(out_shape.iter().product(), scales.dtype(), "afq-qmm-output")?;
388
389            crate::metal_kernels::call_afq_qmm(
390                device.device(),
391                &encoder,
392                &crate::metal_kernels::Kernels::new(),
393                scales.dtype(),
394                x_s.buffer(),
395                x.layout().start_offset() * x.dtype().size_in_bytes(),
396                x.dims(),
397                x.stride(),
398                w_s.buffer(),
399                w.dims(),
400                w.stride(),
401                s_s.buffer(),
402                scales.stride(),
403                b_s.buffer(),
404                biases.stride(),
405                &output,
406                &out_shape,
407                None,
408                None,
409                None,
410                transpose,
411                bits,
412                group_size,
413            )
414            .map_err(candle_core::Error::wrap)?;
415
416            (output, out_shape)
417        };
418
419        let output = Tensor::from((
420            Storage::Metal(MetalStorage::new(
421                output,
422                device.clone(),
423                out_shape.iter().product(),
424                scales.dtype(),
425            )),
426            Shape::from(out_shape),
427        ));
428
429        return Ok(output);
430    }
431    #[cfg(feature = "cuda")]
432    if x.device().is_cuda() {
433        return cuda_backend::afq_mm_op(
434            x,
435            w,
436            scales,
437            biases,
438            lhs_indices,
439            rhs_indices,
440            group_size,
441            bits,
442            transpose,
443        );
444    }
445
446    // CPU fallback for non-accelerated devices
447    cpu_backend::afq_mm_op(
448        x,
449        w,
450        scales,
451        biases,
452        lhs_indices,
453        rhs_indices,
454        group_size,
455        bits,
456        transpose,
457    )
458}

598    /// Simple scalar (reference) quantiser: per`group_size` affine.
599    pub(crate) fn afq_quantize_op(
600        w: &Tensor,
601        group_size: usize,
602        bits: usize,
603    ) -> Result<(Tensor, Tensor, Tensor)> {
604        if bits == 40 {
605            // mxfp4 is not supported in CPU backend
606            candle_core::bail!("mxfp4 quantization is only supported on Metal backend");
607        }
608        let device = w.device().clone();
609        let levels = ((1u32 << bits) - 1) as f32;
610
611        // Flatten everything except the last dim.
612        let w_vec = w.flatten_all()?.to_dtype(DType::F32)?.to_vec1::<f32>()?;
613        let outer: usize = w_vec.len() / w.dim(D::Minus1)?;
614        let inner = w.dim(D::Minus1)?;
615
616        let packed_row = inner * bits / 32;
617        let groups_per_row = inner / group_size;
618
619        let mut q_codes = vec![0u32; outer * packed_row];
620        let mut scales = vec![0f32; outer * groups_per_row];
621        let mut biases = vec![0f32; outer * groups_per_row];
622
623        for row in 0..outer {
624            for g in 0..groups_per_row {
625                let base = row * inner + g * group_size;
626                let slice = &w_vec[base..base + group_size];
627                let (min_v, max_v) = slice
628                    .iter()
629                    .fold((f32::MAX, f32::MIN), |(a, b), &v| (a.min(v), b.max(v)));
630                let scale = if (max_v - min_v).abs() < 1e-12 {
631                    1.0
632                } else {
633                    (max_v - min_v) / levels
634                };
635                let bias = min_v;
636                scales[row * groups_per_row + g] = scale;
637                biases[row * groups_per_row + g] = bias;
638
639                for i in 0..group_size {
640                    let j = g * group_size + i; // position in this row
641                    let bit_off = j * bits; // overall bit offset
642                    let word_id = bit_off / 32; // u32 index
643                    let shift = bit_off % 32; // intraword shift
644
645                    let q_mask = (1u32 << bits) - 1;
646                    let q_val = ((w_vec[base + i] - bias) / scale)
647                        .round()
648                        .clamp(0.0, levels) as u32
649                        & q_mask;
650
651                    let row_base = row * packed_row;
652                    q_codes[row_base + word_id] |= q_val << shift;
653                    if shift + bits > 32 {
654                        q_codes[row_base + word_id + 1] |= q_val >> (32 - shift);
655                    }
656                }
657            }
658        }
659
660        let w_q = Tensor::from_vec(
661            q_codes,
662            {
663                let mut d = w.dims().to_vec();
664                *d.last_mut().unwrap() = packed_row;
665                d
666            },
667            &device,
668        )?
669        .to_dtype(DType::U32)?;
670        let sc = Tensor::from_vec(
671            scales,
672            {
673                let mut d = w.dims().to_vec();
674                *d.last_mut().unwrap() = groups_per_row;
675                d
676            },
677            &device,
678        )?
679        .to_dtype(w.dtype())?;
680        let bs = Tensor::from_vec(
681            biases,
682            {
683                let mut d = w.dims().to_vec();
684                *d.last_mut().unwrap() = groups_per_row;
685                d
686            },
687            &device,
688        )?
689        .to_dtype(w.dtype())?;
690        Ok((w_q, sc, bs))
691    }

693    /// Scalar dequantiser (inverse of the above).
694    pub(crate) fn afq_dequantize_op(
695        w_q: &Tensor,
696        scales: &Tensor,
697        biases: &Tensor,
698        group_size: usize,
699        _bits: usize,
700    ) -> Result<Tensor> {
701        if _bits == 40 {
702            // mxfp4 is not supported in CPU backend
703            candle_core::bail!("mxfp4 dequantization is only supported on Metal backend");
704        }
705        let device = w_q.device().clone();
706        let codes = w_q.flatten_all()?.to_vec1::<u32>()?;
707        let sc = scales
708            .flatten_all()?
709            .to_dtype(DType::F32)?
710            .to_vec1::<f32>()?;
711        let bs = biases
712            .flatten_all()?
713            .to_dtype(DType::F32)?
714            .to_vec1::<f32>()?;
715
716        let packed_row = w_q.dim(D::Minus1)?;
717        let outer = codes.len() / packed_row;
718        let inner = packed_row * 32 / _bits;
719        let groups_per_row = inner / group_size;
720
721        let mut out = vec![0f32; outer * inner];
722        for row in 0..outer {
723            for g in 0..groups_per_row {
724                let scale = sc[row * groups_per_row + g];
725                let bias = bs[row * groups_per_row + g];
726                for i in 0..group_size {
727                    let j = g * group_size + i;
728                    let bit_off = j * _bits;
729                    let word_id = bit_off / 32;
730                    let shift = bit_off % 32;
731
732                    let row_base = row * packed_row;
733                    let mut q = (codes[row_base + word_id] >> shift) & ((1u32 << _bits) - 1);
734                    if shift + _bits > 32 {
735                        q |=
736                            (codes[row_base + word_id + 1] << (32 - shift)) & ((1u32 << _bits) - 1);
737                    }
738
739                    let idx = row * inner + j;
740                    out[idx] = bias + q as f32 * scale;
741                }
742            }
743        }
744
745        Tensor::from_vec(
746            out,
747            {
748                let mut d = w_q.dims().to_vec();
749                *d.last_mut().unwrap() = inner;
750                d
751            },
752            &device,
753        )?
754        .to_dtype(scales.dtype())
755    }

757    /// Very simple (and slow) matmul after full dequantisation.  Handles 2D tensors.
758    #[allow(clippy::too_many_arguments)]
759    pub(crate) fn afq_mm_op(
760        x: &Tensor,
761        w: &Tensor,
762        scales: &Tensor,
763        biases: &Tensor,
764        _lhs_indices: Option<&Tensor>,
765        _rhs_indices: Option<&Tensor>,
766        group_size: usize,
767        bits: usize,
768        transpose: bool,
769    ) -> Result<Tensor> {
770        if bits == 40 {
771            // mxfp4 is not supported in CPU backend
772            candle_core::bail!("mxfp4 matmul is only supported on Metal backend");
773        }
774        let w_f32 = afq_dequantize_op(w, scales, biases, group_size, bits)?.to_dtype(x.dtype())?;
775        if transpose {
776            x.broadcast_matmul(&w_f32.t()?)
777        } else {
778            x.broadcast_matmul(&w_f32)
779        }
780    }

1682    /// CUDA-accelerated AFQ fused matmul
1683    #[allow(clippy::too_many_arguments)]
1684    pub(crate) fn afq_mm_op(
1685        x: &Tensor,
1686        w: &Tensor,
1687        scales: &Tensor,
1688        biases: &Tensor,
1689        _lhs_indices: Option<&Tensor>,
1690        _rhs_indices: Option<&Tensor>,
1691        group_size: usize,
1692        bits: usize,
1693        transpose: bool,
1694    ) -> Result<Tensor> {
1695        if bits == 40 {
1696            candle_core::bail!("mxfp4 matmul is not supported on CUDA backend");
1697        }
1698
1699        // For indexed matmul, fall back to dequantize + matmul for now
1700        if _lhs_indices.is_some() || _rhs_indices.is_some() {
1701            let w_dequant =
1702                afq_dequantize_op(w, scales, biases, group_size, bits)?.to_dtype(x.dtype())?;
1703            return if transpose {
1704                x.broadcast_matmul(&w_dequant.t()?)
1705            } else {
1706                x.broadcast_matmul(&w_dequant)
1707            };
1708        }
1709
1710        if !transpose {
1711            // Non-transposed matmul: fall back to dequantize + matmul
1712            let w_dequant =
1713                afq_dequantize_op(w, scales, biases, group_size, bits)?.to_dtype(x.dtype())?;
1714            return x.broadcast_matmul(&w_dequant);
1715        }
1716
1717        // Transposed case: y = x @ W^T
1718        // x: [M, K], W: [N, K], y: [M, N]
1719        let dev = crate::utils::get_cuda_device(x)?;
1720
1721        let x_rank = x.rank();
1722        let (m, k) = (
1723            x.dims().iter().take(x_rank - 1).product::<usize>(),
1724            x.dim(D::Minus1)?,
1725        );
1726        let n = w.dim(D::Minus2)?;
1727        // Calculate actual_k from scales tensor (works for all bit widths)
1728        let groups_per_row = scales.dim(D::Minus1)?;
1729        let actual_k = groups_per_row * group_size;
1730
1731        if k != actual_k {
1732            candle_core::bail!(
1733                "x inner dim ({k}) does not match w inner dim ({actual_k}) for transposed matmul"
1734            );
1735        }
1736
1737        let out_shape: Vec<usize> = {
1738            let mut s = x.dims().to_vec();
1739            *s.last_mut().unwrap() = n;
1740            s
1741        };
1742
1743        let (x_s, _) = x.storage_and_layout();
1744        let Storage::Cuda(x_s) = &*x_s else {
1745            candle_core::bail!("Expected CUDA storage");
1746        };
1747        let (w_s, _) = w.storage_and_layout();
1748        let Storage::Cuda(w_s) = &*w_s else {
1749            candle_core::bail!("Expected CUDA storage");
1750        };
1751        let (s_s, _) = scales.storage_and_layout();
1752        let Storage::Cuda(s_s) = &*s_s else {
1753            candle_core::bail!("Expected CUDA storage");
1754        };
1755        let (b_s, _) = biases.storage_and_layout();
1756        let Storage::Cuda(b_s) = &*b_s else {
1757            candle_core::bail!("Expected CUDA storage");
1758        };
1759
1760        let (wq_ptr, _wq_guard) =
1761            crate::utils::slice_ptr(w_s.as_cuda_slice::<u32>()?, w.layout().start_offset());
1762
1763        match x.dtype() {
1764            DType::F16 => {
1765                let output_buf = unsafe { dev.alloc::<f16>(m * n)? };
1766                let (x_ptr, _x_guard) =
1767                    crate::utils::slice_ptr(x_s.as_cuda_slice::<f16>()?, x.layout().start_offset());
1768                let (s_ptr, _s_guard) = crate::utils::slice_ptr(
1769                    s_s.as_cuda_slice::<f16>()?,
1770                    scales.layout().start_offset(),
1771                );
1772                let (b_ptr, _b_guard) = crate::utils::slice_ptr(
1773                    b_s.as_cuda_slice::<f16>()?,
1774                    biases.layout().start_offset(),
1775                );
1776                let (out_ptr, out_guard) = output_buf.device_ptr(output_buf.stream());
1777
1778                // Use QMV kernel for fused quantized matmul
1779                unsafe {
1780                    match (bits, group_size) {
1781                        (2, 32) => ffi::afq_qmv_2bit_gs32_f16(
1782                            x_ptr as *const f16,
1783                            wq_ptr as *const u32,
1784                            s_ptr as *const f16,
1785                            b_ptr as *const f16,
1786                            out_ptr as *mut f16,
1787                            m as i32,
1788                            n as i32,
1789                            k as i32,
1790                        ),
1791                        (2, 64) => ffi::afq_qmv_2bit_gs64_f16(
1792                            x_ptr as *const f16,
1793                            wq_ptr as *const u32,
1794                            s_ptr as *const f16,
1795                            b_ptr as *const f16,
1796                            out_ptr as *mut f16,
1797                            m as i32,
1798                            n as i32,
