48#[cfg(feature = "cuda")]
49macro_rules! dequant_for_dtype {
50    ($this:expr, w=$wq_t:ty, sz=$scale_t:ty, $dtype:ident, pack=$pack:expr, $dev:expr, $bit_thing:ident, $postfix:tt) => {{
51        paste::paste! {
52            let (wq, _) = $this.w_q.storage_and_layout();
53            let wq = match &*wq {
54                candle_core::Storage::Cuda(s) => s,
55                _ => candle_core::bail!("wq must be a cuda tensor"),
56            };
57            let (w_slice, _w_guard) = crate::utils::slice_ptr(wq.as_cuda_slice::<$wq_t>()?, $this.w_q.layout().start_offset());
58
59            let (scale, _) = $this.scales.storage_and_layout();
60            let scale = match &*scale {
61                candle_core::Storage::Cuda(s) => s,
62                _ => candle_core::bail!("scale must be a cuda tensor"),
63            };
64            let (scale_slice, _scale_guard) = crate::utils::slice_ptr(scale.as_cuda_slice::<$scale_t>()?, $this.scales.layout().start_offset());
65
66            let (zero, _) = $this.zeros.storage_and_layout();
67            let zero = match &*zero {
68                candle_core::Storage::Cuda(s) => s,
69                _ => candle_core::bail!("zero must be a cuda tensor"),
70            };
71            let (zero_slice, _zero_guard) = crate::utils::slice_ptr(zero.as_cuda_slice::<$scale_t>()?, $this.zeros.layout().start_offset());
72
73            let (h, w) = $this.w_q.dims2()?;
74            let num_packed_elems = $pack;
75            let out_shape = Shape::from_dims(&[num_packed_elems * h, w]);
76
77            let out = unsafe { $dev.alloc::<$scale_t>(out_shape.elem_count())? };
78            let (out_ptr, out_guard) = out.device_ptr(out.stream());
79            unsafe {
80                $bit_thing::[< dequantize_ $postfix >](
81                    w_slice as *const $wq_t,
82                    scale_slice as *const $scale_t,
83                    zero_slice as *const $scale_t,
84                    out_ptr as *mut $scale_t,
85                    h as i32,
86                    w as i32,
87                );
88            }
89            drop(out_guard);
90
91            let storage = CudaStorage {
92                slice: CudaStorageSlice::$dtype(out),
93                device: $dev.clone(),
94            };
95            let storage = Storage::Cuda(storage);
96
97            from_storage_no_op(storage, out_shape, false)
98        }
99    }};

143    // https://github.com/mobiusml/hqq/blob/306e30d9400629523c8e0af70101d8d7073cb3d5/hqq/core/bitpack.py#L10
144    pub(crate) fn bitpack_type(&self) -> impl Fn(Tensor) -> Result<Tensor> {
145        match self {
146            Self::Eight => |wq: Tensor| wq.to_dtype(DType::U8),
147            Self::Four => |wq: Tensor| {
148                let wq = wq.to_dtype(DType::U8)?;
149                let step = (wq.dims()[0] as f64 / 2.) as usize;
150
151                let a = wq.narrow(0, 0, step)?;
152                let b = wq.narrow(0, step, step)?;
153                a.leftshift(4)?.bitwise_or(&b)
154            },
155            Self::Two => |wq: Tensor| {
156                let wq = wq.to_dtype(DType::U8)?;
157                let step = (wq.dims()[0] as f64 / 4.) as usize;
158
159                let a = wq.narrow(0, 0, step)?;
160                let b = wq.narrow(0, step, step)?;
161                let c = wq.narrow(0, step * 2, step)?;
162                let d = wq.narrow(0, step * 3, step)?;
163
164                a.leftshift(6)?
165                    .bitwise_or(&b.leftshift(4)?)?
166                    .bitwise_or(&c.leftshift(2)?)?
167                    .bitwise_or(&d)
168            },
169            Self::Three => |wq_in: Tensor| {
170                let wq = Tensor::zeros(
171                    (
172                        (10. * (wq_in.dims()[0] as f64 / 10.).ceil()) as usize,
173                        wq_in.dims()[1],
174                    ),
175                    DType::U32,
176                    wq_in.device(),
177                )?;
178                let wq = wq
179                    .slice_assign(
180                        &[0..wq_in.dims()[0], 0..wq.dims()[1]],
181                        &wq_in.to_dtype(DType::U32)?,
182                    )?
183                    .to_dtype(DType::I32)?;
184                let step = (wq.dims()[0] as f64 / 10.) as usize;
185
186                let a = wq.narrow(0, 0, step)?;
187                let b = wq.narrow(0, step, step)?;
188                let c = wq.narrow(0, step * 2, step)?;
189                let d = wq.narrow(0, step * 3, step)?;
190                let e = wq.narrow(0, step * 4, step)?;
191                let f = wq.narrow(0, step * 5, step)?;
192                let g = wq.narrow(0, step * 6, step)?;
193                let h = wq.narrow(0, step * 7, step)?;
194                let i = wq.narrow(0, step * 8, step)?;
195                let j = wq.narrow(0, step * 9, step)?;
196
197                a.leftshift(27)
198                    .unwrap()
199                    .bitwise_or(&b.leftshift(24).unwrap())
200                    .unwrap()
201                    .bitwise_or(&c.leftshift(21)?)?
202                    .bitwise_or(&d.leftshift(18)?)?
203                    .bitwise_or(&e.leftshift(15)?)?
204                    .bitwise_or(&f.leftshift(12)?)?
205                    .bitwise_or(&g.leftshift(9)?)?
206                    .bitwise_or(&h.leftshift(6)?)?
207                    .bitwise_or(&i.leftshift(3)?)?
208                    .bitwise_or(&j)
209            },
210            Self::One => |wq: Tensor| {
211                let wq = wq.to_dtype(DType::U8)?;
212                let step = (wq.dims()[0] as f64 / 8.) as usize;
213
214                let a = wq.narrow(0, 0, step)?;
215                let b = wq.narrow(0, step, step)?;
216                let c = wq.narrow(0, step * 2, step)?;
217                let d = wq.narrow(0, step * 3, step)?;
218                let e = wq.narrow(0, step * 4, step)?;
219                let f = wq.narrow(0, step * 5, step)?;
220                let g = wq.narrow(0, step * 6, step)?;
221                let h = wq.narrow(0, step * 7, step)?;
222
223                a.leftshift(7)?
224                    .bitwise_or(&b.leftshift(6)?)?
225                    .bitwise_or(&c.leftshift(5)?)?
226                    .bitwise_or(&d.leftshift(4)?)?
227                    .bitwise_or(&e.leftshift(3)?)?
228                    .bitwise_or(&f.leftshift(2)?)?
229                    .bitwise_or(&g.leftshift(1)?)?
230                    .bitwise_or(&h)
231            },
232        }
233    }

257    /// Dequantize `self` into a tensor of shape `scales` or `zeros`.
258    #[cfg(not(feature = "cuda"))]
259    fn dequantize(&self) -> Result<Tensor> {
260        use crate::hqq::hqq_op::{Dequant1Bit, Dequant2Bit, Dequant3Bit, Dequant4Bit, Dequant8Bit};
261
262        match (self.scales.dtype(), self.zeros.dtype()) {
263            (DType::F16, DType::F16) | (DType::BF16, DType::BF16) | (DType::F32, DType::F32) => (),
264            (a, b) => {
265                candle_core::bail!("Expected all dtypes to be the same, got ({a:?}, {b:?}).")
266            }
267        }
268        if !(self.w_q.is_contiguous() && self.scales.is_contiguous() && self.zeros.is_contiguous())
269        {
270            candle_core::bail!("All tensors must be contiguous!");
271        }
272        if self.cfg.axis as usize != 0 {
273            candle_core::bail!(
274                "CPU HQQ dequantization requires axis == 0, got {}.",
275                self.cfg.axis as usize
276            );
277        }
278        let (h, w) = self.w_q.dims2()?;
279
280        match self.cfg.bits as usize {
281            8 => self
282                .w_q
283                .apply_op3_no_bwd(&self.scales, &self.zeros, &Dequant8Bit { h, w })?
284                .reshape(&self.w_shape),
285            4 => self
286                .w_q
287                .apply_op3_no_bwd(&self.scales, &self.zeros, &Dequant4Bit { h, w })?
288                .reshape(&self.w_shape),
289            3 => self
290                .w_q
291                .apply_op3_no_bwd(&self.scales, &self.zeros, &Dequant3Bit { h, w })?
292                .reshape(&self.w_shape),
293            2 => self
294                .w_q
295                .apply_op3_no_bwd(&self.scales, &self.zeros, &Dequant2Bit { h, w })?
296                .reshape(&self.w_shape),
297            1 => self
298                .w_q
299                .apply_op3_no_bwd(&self.scales, &self.zeros, &Dequant1Bit { h, w })?
300                .reshape(&self.w_shape),
301            b => candle_core::bail!("Unreachable bits {b}"),
302        }
303    }

305    /// Dequantize `self` into a tensor of shape `scales` or `zeros`.
306    #[cfg(feature = "cuda")]
307    fn dequantize(&self) -> Result<Tensor> {
308        match (self.scales.dtype(), self.zeros.dtype()) {
309            (DType::F16, DType::F16) | (DType::BF16, DType::BF16) | (DType::F32, DType::F32) => (),
310            (a, b) => {
311                candle_core::bail!("Expected all dtypes to be the same, got ({a:?}, {b:?}).")
312            }
313        }
314        if !(self.w_q.is_contiguous() && self.scales.is_contiguous() && self.zeros.is_contiguous())
315        {
316            candle_core::bail!("All tensors must be contiguous!");
317        }
318        if self.cfg.axis as usize != 0 {
319            candle_core::bail!(
320                "CUDA HQQ dequantization requires axis == 0, got {}.",
321                self.cfg.axis as usize
322            );
323        }
324        let dev = get_cuda_device(&self.w_q)?;
325
326        let inner = match (self.cfg.bits as usize, self.scales.dtype()) {
327            // 8 bits
328            (8, DType::F32) => {
329                dequant_for_dtype!(
330                    self,
331                    w = u8,
332                    sz = f32,
333                    F32,
334                    pack = 1,
335                    dev,
336                    eight_bit,
337                    8bit_u8_kernel_f32
338                )
339            }
340            (8, DType::F16) => {
341                dequant_for_dtype!(
342                    self,
343                    w = u8,
344                    sz = f16,
345                    F16,
346                    pack = 1,
347                    dev,
348                    eight_bit,
349                    8bit_u8_kernel_f16
350                )
351            }
352            (8, DType::BF16) => {
353                dequant_for_dtype!(
354                    self,
355                    w = u8,
356                    sz = bf16,
357                    BF16,
358                    pack = 1,
359                    dev,
360                    eight_bit,
361                    8bit_u8_kernel_bf16
362                )
363            }
364
365            // 4 bits
366            (4, DType::F32) => {
367                dequant_for_dtype!(
368                    self,
369                    w = u8,
370                    sz = f32,
371                    F32,
372                    pack = 2,
373                    dev,
374                    four_bit,
375                    4bit_u8_kernel_f32
376                )
377            }
378            (4, DType::F16) => {
379                dequant_for_dtype!(
380                    self,
381                    w = u8,
382                    sz = f16,
383                    F16,
384                    pack = 2,
385                    dev,
386                    four_bit,
387                    4bit_u8_kernel_f16
388                )
389            }
390            (4, DType::BF16) => {
391                dequant_for_dtype!(
392                    self,
393                    w = u8,
394                    sz = bf16,
395                    BF16,
396                    pack = 2,
397                    dev,
398                    four_bit,
399                    4bit_u8_kernel_bf16
400                )
401            }
402
403            // 3 bits
404            // https://github.com/mobiusml/hqq/blob/306e30d9400629523c8e0af70101d8d7073cb3d5/hqq/kernels/hqq_aten_cuda.cpp#L42-L45
405            (3, DType::F32) => {
406                let res = dequant_for_dtype!(
407                    self,
408                    w = i32,
409                    sz = f32,
410                    F32,
411                    pack = 10,
412                    dev,
413                    three_bit,
414                    3bit_32_kernel_f32
415                );
416                res.narrow(self.cfg.axis as usize, 0, self.cfg.group_size.into())?
417            }
418            (3, DType::F16) => {
419                let res = dequant_for_dtype!(
420                    self,
421                    w = i32,
422                    sz = f16,
423                    F16,
424                    pack = 10,
425                    dev,
426                    three_bit,
427                    3bit_32_kernel_f16
428                );
429                res.narrow(self.cfg.axis as usize, 0, self.cfg.group_size.into())?
430            }
431            (3, DType::BF16) => {
432                let res = dequant_for_dtype!(
433                    self,
434                    w = i32,
435                    sz = bf16,
436                    BF16,
437                    pack = 10,
438                    dev,
439                    three_bit,
440                    3bit_32_kernel_bf16
441                );
442                res.narrow(self.cfg.axis as usize, 0, self.cfg.group_size.into())?
443            }
444
445            // 2 bits
446            (2, DType::F32) => {
447                dequant_for_dtype!(
448                    self,
449                    w = u8,
450                    sz = f32,
451                    F32,
452                    pack = 4,
453                    dev,
454                    two_bit,
455                    2bit_u8_kernel_f32
456                )
457            }
458            (2, DType::F16) => {
459                dequant_for_dtype!(
460                    self,
461                    w = u8,
462                    sz = f16,
463                    F16,
464                    pack = 4,
465                    dev,
466                    two_bit,
467                    2bit_u8_kernel_f16
468                )
469            }
470            (2, DType::BF16) => {
471                dequant_for_dtype!(
472                    self,
473                    w = u8,
474                    sz = bf16,
475                    BF16,
476                    pack = 4,
477                    dev,
478                    two_bit,
479                    2bit_u8_kernel_bf16
480                )
481            }
482
483            // 1 bit
484            (1, DType::F32) => {
485                dequant_for_dtype!(
486                    self,
487                    w = u8,
488                    sz = f32,
489                    F32,
490                    pack = 8,
491                    dev,
492                    one_bit,
493                    1bit_u8_kernel_f32

495            }
496            (1, DType::F16) => {
497                dequant_for_dtype!(
498                    self,
499                    w = u8,
500                    sz = f16,
501                    F16,
502                    pack = 8,
503                    dev,
504                    one_bit,

506                )
507            }
508            (1, DType::BF16) => {
509                dequant_for_dtype!(
510                    self,
511                    w = u8,
512                    sz = bf16,
513                    BF16,
514                    pack = 8,
515                    dev,
516                    one_bit,
517                    1bit_u8_kernel_bf16
518                )
519            }
520            (bits, dtype) => candle_core::bail!("Unsupported bit width {bits} and dtype {dtype:?}"),
521        };
522        inner.reshape(&self.w_shape)
523    }

525    fn dequantize_matmul(&self, xs: &Tensor) -> Result<Tensor> {
526        let w = self.dequantize()?;
527        // Dispatch to unquant. This uses some cublaslt for bias & on cuda always, so it is better
528        let unquant = UnquantLinear::new(QuantMethodConfig::Unquantized(Linear::new(
529            w,
530            self.bias.clone(),
531        )))?;
532        unquant.forward(xs)
533    }

541impl QuantMethod for HqqLayer {
542    fn new(method: QuantMethodConfig) -> Result<Self>
543    where
544        Self: Sized,
545    {
546        match method {
547            QuantMethodConfig::Gguf { .. }
548            | QuantMethodConfig::Unquantized(_)
549            | QuantMethodConfig::GptqAwq { .. }
550            | QuantMethodConfig::Dummy
551            | QuantMethodConfig::FP8 { .. }
552            | QuantMethodConfig::Bnb { .. }
553            | QuantMethodConfig::BlockwiseFP8 { .. }
554            | QuantMethodConfig::Afq { .. } => {
555                unreachable!()
556            }
557            QuantMethodConfig::Hqq {
558                tensor,
559                bits,
560                group_size,
561                axis,
562                optimization_steps,
563                round_zeros,
564                channel_wise,
565                bias,
566            } => {
567                let cfg = HqqConfig {
568                    bits,
569                    group_size,
570                    axis,
571                    optimization_steps,
572                    round_zeros: round_zeros.unwrap_or(false),
573                    channel_wise: channel_wise.unwrap_or(true),
574                };
575
576                let this = Self::quantize(&tensor, tensor.device(), cfg)?;
577                if let Some(bias) = bias {
578                    Ok(this.with_bias(bias))
579                } else {
580                    Ok(this)
581                }
582            }
583        }
584    }

612    fn apply_isq(
613        self: Arc<Self>,
614        dtype: Option<IsqType>,
615        device: Device,
616        n_quantized: &AtomicUsize,
617        imatrix_weight: Option<Vec<f32>>,
618        guard: QuantizeOntoGuard,
619    ) -> Result<Arc<dyn QuantMethod>> {
620        let _acquired_quantize_guard = guard.acquire(&device);
621        if imatrix_weight.is_some() {
622            // TODO just warn?
623            candle_core::bail!("HQQ does not support imatrix.");
624        }
625
626        n_quantized.fetch_add(1, std::sync::atomic::Ordering::Relaxed);
627        let bits = match dtype {
628            Some(IsqType::HQQ8) => HqqBits::Eight,
629            Some(IsqType::HQQ4) => HqqBits::Four,
630            // Some(IsqType::HQQ3) => HqqBits::Three,
631            // Some(IsqType::HQQ2) => HqqBits::Two,
632            // Some(IsqType::HQQ1) => HqqBits::One,
633            _ => candle_core::bail!("Expected a HQQ ISQ type."),
634        };
635        let cfg = HqqConfig {
636            bits,
637            group_size: ISQ_HQQ_GROUP_SIZE.try_into()?,
638            axis: HqqAxis::Zero,
639            optimization_steps: ISQ_HQQ_DEFAULT_OPT_STEPS,
640            round_zeros: false,
641            channel_wise: true,
642        };
643        let dequant = self.dequantize()?;
644        let res = Self::quantize(&dequant, &device, cfg)?;
645        if let Some(ref bias) = self.bias {
646            let bias = bias
647                .to_device(&device)?
648                .to_dtype(res.dtype_and_device().0)?;
649            Ok(Arc::new(res.with_bias(bias)))
650        } else {
651            Ok(Arc::new(res))
652        }
653    }

656// Serialization structure:
657//
658// -----------------------
659// UQFF version, u32, little endian
660// -----------------------
661// ISQ type (2 for hqq), u8, little endian
662// -----------------------
663// Whether bias data is included, u8 boolean
664// -----------------------

692impl QuantizedSerde for HqqLayer {
693    fn isq_serde_supported(&self) -> bool {
694        true
695    }
696    fn name(&self) -> &'static str {
697        "hqq"
698    }
699    fn serialize(&self) -> Result<Cow<[u8]>> {
700        self.serialize_with_bias(self.bias.clone())
701    }
702    fn serialize_with_bias(&self, bias: Option<Tensor>) -> Result<Cow<[u8]>> {
703        let mut buffer = Vec::new();
704
705        // Version is always first!
706        buffer.extend(&UQFF_VERSION.to_le_bytes());
707
708        // ISQ type for hqq is 2
709        buffer.push(QuantizedSerdeType::Hqq as u8);
710
711        // Has bias
712        buffer.push(bias.is_some() as u8);
713
714        serialize_tensor(&mut buffer, &self.w_q)?;
715        serialize_tensor(&mut buffer, &self.scales)?;
716        serialize_tensor(&mut buffer, &self.zeros)?;
717
718        let w_shape = self.w_shape.dims();
719        let shape_len = w_shape.len();
720        if shape_len > u32::MAX as usize {
721            candle_core::bail!(
722                "Weight tensor has too many dimensions for UQFF format: {} exceeds u32::MAX",
723                shape_len
724            );
725        }
726        buffer.extend((shape_len as u32).to_le_bytes());
727        for dim in w_shape {
728            if *dim > u32::MAX as usize {
729                candle_core::bail!(
730                    "Weight tensor dimension too large for UQFF format: {} exceeds u32::MAX",
731                    dim
732                );
733            }
734            buffer.extend((*dim as u32).to_le_bytes());
735        }
736
737        // Config
738        buffer.push(self.cfg.bits as u8);
739        let group_size = <NonZeroUsize as Into<usize>>::into(self.cfg.group_size);
740        if group_size > u32::MAX as usize {
741            candle_core::bail!(
742                "HQQ group size too large for UQFF format: {} exceeds u32::MAX",
743                group_size
744            );
745        }
746        buffer.extend(&(group_size as u32).to_le_bytes());
747        buffer.push(self.cfg.axis as u8);
748        // NOTE: using 0 as a sentinel for None. This means legitimate 0 values cannot be distinguished from None.
749        // This is acceptable because 0 optimization steps would be functionally equivalent to None.
750        let opt_steps = self.cfg.optimization_steps.unwrap_or(0);
751        if opt_steps > u32::MAX as usize {
752            candle_core::bail!(
753                "HQQ optimization steps too large for UQFF format: {} exceeds u32::MAX",
754                opt_steps
755            );
756        }
757        buffer.extend(&(opt_steps as u32).to_le_bytes());
758        buffer.push(self.cfg.round_zeros as u8);
759        buffer.push(self.cfg.channel_wise as u8);
760
761        if let Some(bias) = &bias {
762            // Bias
763            serialize_tensor(&mut buffer, bias)?;
764        }
765
766        Ok(Cow::from(buffer))
767    }
768
769    fn deserialize(
770        data: Cow<[u8]>,
771        device: &Device,
772        _comm: &Arc<crate::Comm>,
773        guard: QuantizeOntoGuard,
774    ) -> Result<Arc<dyn QuantMethod>>
775    where
776        Self: Sized,
777    {
778        let mut buffer = Cursor::new(data);
779
780        let version = buffer.read_u32::<LittleEndian>()?;
781        if let Err(e) = version_is_compatible(version) {
782            return Err(candle_core::Error::wrap(e));
783        }
784
785        let isq_type = buffer.read_u8()? as usize;
786        if isq_type != QuantizedSerdeType::Hqq as usize {
787            candle_core::bail!(
788                "ISQ type ({isq_type}) doesn't match expected type {}",
789                QuantizedSerdeType::Hqq as usize
790            );
791        }
792
793        let has_bias = buffer.read_u8()? != 0;
794
795        let _acquired_load_guard = guard.acquire(device);
796        let w_q = deserialize_tensor(&mut buffer, device)?;
797        let scales = deserialize_tensor(&mut buffer, device)?;
798        let zeros = deserialize_tensor(&mut buffer, device)?;
799
800        let n_dims = buffer.read_u32::<LittleEndian>()? as usize;
801
802        let mut dims = Vec::with_capacity(n_dims);
803        for _ in 0..n_dims {
804            dims.push(buffer.read_u32::<LittleEndian>()? as usize)
805        }
806        let w_shape = Shape::from_dims(&dims);
807
808        // TODO: keep this in sync with get_isq_type_from_uqff!
809        let bits = HqqBits::try_from(buffer.read_u8()? as usize)?;
810        let group_size = NonZeroUsize::try_from(buffer.read_u32::<LittleEndian>()? as usize)?;
811        let axis = HqqAxis::try_from(buffer.read_u8()? as usize)?;
812        let optimization_steps = match buffer.read_u32::<LittleEndian>()? as usize {
813            0 => None,
814            other => Some(other),
815        };
816        let round_zeros = buffer.read_u8()? != 0;
817        let channel_wise = buffer.read_u8()? != 0;
818
819        let cfg = HqqConfig {
820            bits,
821            group_size,
822            axis,
823            optimization_steps,
824            round_zeros,
825            channel_wise,
826        };
827
828        let b = if has_bias {
829            Some(deserialize_tensor(&mut buffer, device)?)
830        } else {
831            None
832        };
833
834        Ok(Arc::new(Self {
835            w_q,
836            zeros,
837            scales,
838            bias: b,
839            w_shape,
840            cfg,
841        }))
842    }

921impl HqqLayer {
922    pub fn get_isq_type_from_uqff(data: Cow<[u8]>) -> Result<IsqType> {
923        let mut buffer = Cursor::new(data);
924
925        let version = buffer.read_u32::<LittleEndian>()?;
926        if let Err(e) = version_is_compatible(version) {
927            return Err(candle_core::Error::wrap(e));
928        }
929
930        let isq_type = buffer.read_u8()? as usize;
931        if isq_type != QuantizedSerdeType::Hqq as usize {
932            candle_core::bail!(
933                "ISQ type ({isq_type}) doesn't match expected type {}",
934                QuantizedSerdeType::Hqq as usize
935            );
936        }
937
938        let _has_bias = buffer.read_u8()? != 0;
939
940        fake_deserialize_tensor(&mut buffer)?;
941        fake_deserialize_tensor(&mut buffer)?;
942        fake_deserialize_tensor(&mut buffer)?;
943
944        let n_dims = buffer.read_u32::<LittleEndian>()? as usize;
945
946        let mut dims = Vec::with_capacity(n_dims);
947        for _ in 0..n_dims {
948            dims.push(buffer.read_u32::<LittleEndian>()? as usize)
949        }
950        let _w_shape = Shape::from_dims(&dims);
951
952        // TODO: keep this in sync with get_isq_type_from_uqff!
953        let bits = HqqBits::try_from(buffer.read_u8()? as usize)?;
954
955        match bits {
956            HqqBits::Eight => Ok(IsqType::HQQ8),
957            HqqBits::Four => Ok(IsqType::HQQ4),
958            HqqBits::One | HqqBits::Two | HqqBits::Three => {
959                candle_core::bail!("cannot convert hqq bits to isq type")
960            }
961        }
962    }
