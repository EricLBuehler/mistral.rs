25/// Represents different types of speech generation responses.
26pub enum SpeechGenerationResponder {
27    InternalError(Box<dyn Error>),
28    ValidationError(Box<dyn Error>),
29    RawResponse(axum::response::Response),
30}
31
32impl IntoResponse for SpeechGenerationResponder {
33    /// Converts the speech generation responder into an HTTP response.
34    fn into_response(self) -> axum::response::Response {
35        match self {
36            SpeechGenerationResponder::InternalError(e) => {
37                JsonError::new(e.to_string()).to_response(http::StatusCode::INTERNAL_SERVER_ERROR)
38            }
39            SpeechGenerationResponder::ValidationError(e) => {
40                JsonError::new(e.to_string()).to_response(http::StatusCode::UNPROCESSABLE_ENTITY)
41            }
42            SpeechGenerationResponder::RawResponse(resp) => resp,
43        }
44    }
45}

47/// Parses and validates a speech generation request.
48///
49/// This function transforms a speech generation request into the
50/// request format used by mistral.rs.
51pub fn parse_request(
52    oairequest: SpeechGenerationRequest,
53    state: Arc<MistralRs>,
54    tx: Sender<Response>,
55) -> Result<(Request, AudioResponseFormat)> {
56    let repr = serde_json::to_string(&oairequest).expect("Serialization of request failed.");
57    MistralRs::maybe_log_request(state.clone(), repr);
58
59    // Validate that the requested model matches the loaded model
60    validate_model_name(&oairequest.model, state.clone())?;
61
62    let request = Request::Normal(Box::new(NormalRequest {
63        id: state.next_request_id(),
64        messages: RequestMessage::SpeechGeneration {
65            prompt: oairequest.input,
66        },
67        sampling_params: SamplingParams::deterministic(),
68        response: tx,
69        return_logprobs: false,
70        is_streaming: false,
71        suffix: None,
72        constraint: Constraint::None,
73        tool_choice: None,
74        tools: None,
75        logits_processors: None,
76        return_raw_logits: false,
77        web_search_options: None,
78    }));
79
80    Ok((request, oairequest.response_format))
81}

83/// Speech generation endpoint handler.
84#[utoipa::path(
85    post,
86    tag = "Mistral.rs",
87    path = "/v1/audio/speech",
88    request_body = SpeechGenerationRequest,
89    responses((status = 200, description = "Speech generation"))
90)]
91pub async fn speech_generation(
92    State(state): State<Arc<MistralRs>>,
93    Json(oairequest): Json<SpeechGenerationRequest>,
94) -> SpeechGenerationResponder {
95    let (tx, mut rx) = create_response_channel(None);
96
97    let (request, response_format) = match parse_request(oairequest, state.clone(), tx) {
98        Ok(x) => x,
99        Err(e) => return handle_error(state, e.into()),
100    };
101
102    // Validate response format here
103    if !matches!(
104        response_format,
105        AudioResponseFormat::Wav | AudioResponseFormat::Pcm
106    ) {
107        return SpeechGenerationResponder::ValidationError(Box::new(JsonError::new(
108            "Only support wav/pcm response format.".to_string(),
109        )));
110    }
111
112    if let Err(e) = send_request(&state, request).await {
113        return handle_error(state, e.into());
114    }
115
116    process_non_streaming_response(&mut rx, state, response_format).await
117}

119/// Helper function to handle speech generation errors and logging them.
120pub fn handle_error(
121    state: SharedMistralRsState,
122    e: Box<dyn std::error::Error + Send + Sync + 'static>,
123) -> SpeechGenerationResponder {
124    let e = anyhow::Error::msg(e.to_string());
125    MistralRs::maybe_log_error(state, &*e);
126    SpeechGenerationResponder::InternalError(e.into())
127}

129/// Process non-streaming speech generation responses.
130pub async fn process_non_streaming_response(
131    rx: &mut Receiver<Response>,
132    state: SharedMistralRsState,
133    response_format: AudioResponseFormat,
134) -> SpeechGenerationResponder {
135    let response = match rx.recv().await {
136        Some(response) => response,
137        None => {
138            let e = anyhow::Error::msg("No response received from the model.");
139            return handle_error(state, e.into());
140        }
141    };
142
143    match_responses(state, response, response_format)
144}

146/// Matches and processes different types of model responses into appropriate speech generation responses.
147pub fn match_responses(
148    state: SharedMistralRsState,
149    response: Response,
150    response_format: AudioResponseFormat,
151) -> SpeechGenerationResponder {
152    match response {
153        Response::InternalError(e) => {
154            MistralRs::maybe_log_error(state, &*e);
155            SpeechGenerationResponder::InternalError(e)
156        }
157        Response::ValidationError(e) => SpeechGenerationResponder::ValidationError(e),
158        Response::ImageGeneration(_) => unreachable!(),
159        Response::CompletionModelError(m, _) => {
160            let e = anyhow::Error::msg(m.to_string());
161            MistralRs::maybe_log_error(state, &*e);
162            SpeechGenerationResponder::InternalError(e.into())
163        }
164        Response::CompletionDone(_) => unreachable!(),
165        Response::CompletionChunk(_) => unreachable!(),
166        Response::Chunk(_) => unreachable!(),
167        Response::Done(_) => unreachable!(),
168        Response::ModelError(_, _) => unreachable!(),
169        Response::Speech {
170            pcm,
171            rate,
172            channels,
173        } => {
174            let pcm_endianness = "s16le";
175
176            let content_type = response_format.audio_content_type(rate, channels, pcm_endianness);
177            let mut headers = HeaderMap::new();
178            headers.insert(
179                http::header::CONTENT_TYPE,
180                HeaderValue::from_str(&content_type).unwrap(),
181            );
182
183            let encoded = match response_format {
184                AudioResponseFormat::Pcm => {
185                    let samples: &[f32] = &pcm;
186                    let mut buf = Vec::with_capacity(samples.len() * std::mem::size_of::<i64>());
187                    for &sample in samples {
188                        buf.extend_from_slice(&sample.to_i16().to_le_bytes());
189                    }
190                    buf
191                }
192                AudioResponseFormat::Wav => {
193                    // Write WAV data into an in-memory buffer
194                    let mut buf = Vec::new();
195                    speech_utils::write_pcm_as_wav(&mut buf, &pcm, rate as u32, channels as u16)
196                        .unwrap();
197                    buf
198                }
199                _ => unreachable!("Should be validated above."),
200            };
201
202            let bytes = Bytes::from(encoded);
203
204            SpeechGenerationResponder::RawResponse((StatusCode::OK, headers, bytes).into_response())
205        }
206        Response::Raw { .. } => unreachable!(),
207    }
