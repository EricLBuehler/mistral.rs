16/// A callback function that processes streaming response chunks before they are sent to the client.
17///
18/// This hook allows modification of each chunk in the streaming response, enabling features like
19/// content filtering, transformation, or logging. The callback receives a chunk and must return
20/// a (potentially modified) chunk.
21///
22/// ### Examples
23///
24/// ```no_run
25/// use mistralrs_core::ChatCompletionChunkResponse;
26/// use mistralrs_server_core::types::OnChunkCallback;
27///
28/// let on_chunk: OnChunkCallback<ChatCompletionChunkResponse> = Box::new(|mut chunk| {
29///     // Log the chunk or modify its content
30///     println!("Processing chunk: {:?}", chunk);
31///     chunk
32/// });
33/// ```
34pub type OnChunkCallback<R> = Box<dyn Fn(R) -> R + Send + Sync>;
35
36/// A callback function that is executed when the streaming response completes.
37///
38/// This hook receives all chunks that were streamed during the response, allowing for
39/// post-processing, analytics, or cleanup operations after the stream finishes.
40///
41/// ### Examples
42///
43/// ```no_run
44/// use mistralrs_core::ChatCompletionChunkResponse;
45/// use mistralrs_server_core::types::OnDoneCallback;
46///
47/// let on_done: OnDoneCallback<ChatCompletionChunkResponse> = Box::new(|chunks| {
48///     println!("Stream completed with {} chunks", chunks.len());
49///     // Process all chunks for analytics
50/// });
51/// ```
52pub type OnDoneCallback<R> = Box<dyn Fn(&[R]) + Send + Sync>;
