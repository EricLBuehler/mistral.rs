16/// Usage information for a response
17#[derive(Debug, Clone, Serialize, Deserialize, ToSchema, Default)]
18pub struct ResponseUsage {
19    /// Number of input tokens
20    pub input_tokens: usize,
21    /// Number of output tokens
22    pub output_tokens: usize,
23    /// Total number of tokens
24    pub total_tokens: usize,
25    /// Detailed input token breakdown
26    #[serde(skip_serializing_if = "Option::is_none")]
27    pub input_tokens_details: Option<InputTokensDetails>,
28    /// Detailed output token breakdown
29    #[serde(skip_serializing_if = "Option::is_none")]
30    pub output_tokens_details: Option<OutputTokensDetails>,
31}

152    /// Current status of the response
153    pub status: ResponseStatus,
154    /// Output items generated by the model

159    /// Reasoning/chain-of-thought content from the model
160    #[serde(skip_serializing_if = "Option::is_none")]
161    pub reasoning: Option<String>,
162    /// Token usage information
163    #[serde(skip_serializing_if = "Option::is_none")]
164    pub usage: Option<ResponseUsage>,
165    /// Error information (if status is failed)
166    #[serde(skip_serializing_if = "Option::is_none")]
167    pub error: Option<ResponseError>,
168    /// Details about why response is incomplete
169    #[serde(skip_serializing_if = "Option::is_none")]
170    pub incomplete_details: Option<IncompleteDetails>,
171    /// User-provided metadata
172    #[serde(skip_serializing_if = "Option::is_none")]
173    pub metadata: Option<Value>,
174    /// Truncation strategy used
175    #[serde(skip_serializing_if = "Option::is_none")]
176    pub truncation: Option<TruncationStrategy>,
177    /// System instructions (if provided)

206    /// Frequency penalty from the request

421    /// Collect output text from all output items
422    pub fn collect_output_text(&mut self) {
423        let mut texts = Vec::new();
424        for item in &self.output {
425            if let OutputItem::Message { content, .. } = item {
426                for c in content {
427                    if let Some(text) = c.get_text() {
428                        texts.push(text.to_string());
429                    }
430                }
431            }
432        }
433        if !texts.is_empty() {
434            self.output_text = Some(texts.join(""));
435        }
436    }
