99/// Input content types for the OpenResponses API.
100///
101/// Supports both OpenResponses format (input_text, input_image) and
102/// OpenAI Chat format (text, image_url) for compatibility.
103#[derive(Debug, Clone, Serialize, Deserialize)]
104#[serde(untagged)]
105pub enum InputContent {
106    /// OpenResponses format content
107    OpenResponses(OpenResponsesInputContent),
108    /// OpenAI Chat format content
109    OpenAI(OpenAIInputContent),
110}
111
112impl InputContent {
113    /// Normalize to the internal format for processing
114    pub fn into_normalized(self) -> NormalizedInputContent {
115        match self {
116            InputContent::OpenResponses(c) => match c {
117                OpenResponsesInputContent::InputText { text } => {
118                    NormalizedInputContent::Text { text }
119                }
120                OpenResponsesInputContent::InputImage {
121                    image_url,
122                    image_data,
123                    detail,
124                } => NormalizedInputContent::Image {
125                    image_url,
126                    image_data,
127                    detail,
128                },
129                OpenResponsesInputContent::InputAudio { data, format } => {
130                    NormalizedInputContent::Audio { data, format }
131                }
132                OpenResponsesInputContent::InputFile {
133                    file_id,
134                    file_data,
135                    filename,
136                } => NormalizedInputContent::File {
137                    file_id,
138                    file_data,
139                    filename,
140                },
141            },
142            InputContent::OpenAI(c) => match c {
143                OpenAIInputContent::Text { text } => NormalizedInputContent::Text { text },
144                OpenAIInputContent::ImageUrl { image_url } => NormalizedInputContent::Image {
145                    image_url: Some(image_url.url),
146                    image_data: None,
147                    detail: image_url.detail,
148                },
149                OpenAIInputContent::InputAudio { input_audio } => NormalizedInputContent::Audio {
150                    data: input_audio.data,
151                    format: input_audio.format,
152                },
153            },
154        }
155    }
156}

494impl OutputContent {
495    /// Create a new text output content
496    pub fn text(text: String) -> Self {
497        OutputContent::OutputText {
498            text,
499            annotations: None,
500        }
501    }
502
503    /// Create a new text output content with annotations
504    pub fn text_with_annotations(text: String, annotations: Vec<Annotation>) -> Self {
505        OutputContent::OutputText {
506            text,
507            annotations: Some(annotations),
508        }
509    }
510
511    /// Create a new refusal output content
512    pub fn refusal(refusal: String) -> Self {
513        OutputContent::Refusal { refusal }
514    }
515
516    /// Get the text content if this is a text output
517    pub fn get_text(&self) -> Option<&str> {
518        match self {
519            OutputContent::OutputText { text, .. } => Some(text),
520            OutputContent::Refusal { refusal } => Some(refusal),
521        }
522    }
