56/// A builder for creating a mistral.rs instance with configured options for the mistral.rs server.
57///
58/// ### Examples
59///
60/// Basic usage:
61/// ```ignore
62/// use mistralrs_server_core::mistralrs_for_server_builder::MistralRsForServerBuilder;
63///
64/// let args = Args::parse();
65///
66/// let mistralrs = MistralRsForServerBuilder::new()
67///        .with_truncate_sequence(args.truncate_sequence)
68///        .with_model(args.model)
69///        .with_max_seqs(args.max_seqs)
70///        .with_no_kv_cache(args.no_kv_cache)
71///        .with_token_source(args.token_source)
72///        .with_interactive_mode(args.interactive_mode)
73///        .with_prefix_cache_n(args.prefix_cache_n)
74///        .with_paged_attn(args.paged_attn)
75///        .with_cpu(args.cpu)
76///        .with_enable_search(args.enable_search)
77///        .with_seed_optional(args.seed)
78///        .with_log_optional(args.log)
79///        .with_chat_template_optional(args.chat_template)
80///        .with_jinja_explicit_optional(args.jinja_explicit)
81///        .with_num_device_layers_optional(args.num_device_layers)
82///        .with_in_situ_quant_optional(args.in_situ_quant)
83///        .with_paged_attn_gpu_mem_optional(args.paged_attn_gpu_mem)
84///        .with_paged_attn_gpu_mem_usage_optional(args.paged_attn_gpu_mem_usage)
85///        .with_paged_ctxt_len_optional(args.paged_ctxt_len)
86///        .with_paged_attn_block_size_optional(args.paged_attn_block_size)
87///        .with_prompt_chunksize_optional(args.prompt_chunksize)
88///        .build()
89///        .await?;
90/// ```
91pub struct MistralRsForServerBuilder {
92    /// The Candle device to use for model execution (CPU, CUDA, Metal, etc.).
93    device: Option<Device>,
94
95    /// Integer seed to ensure reproducible random number generation.
96    seed: Option<u64>,
97
98    /// Log all responses and requests to this file
99    log: Option<String>,
100
101    /// If a sequence is larger than the maximum model length, truncate the number
102    /// of tokens such that the sequence will fit at most the maximum length.
103    /// If `max_tokens` is not specified in the request, space for 10 tokens will be reserved instead.
104    truncate_sequence: bool,
105
106    /// Model selector
107    model: Option<ModelSelected>,
108
109    /// Maximum running sequences at any time. If the `tgt_non_granular_index` flag is set for X-LoRA models, this will be set to 1.
110    max_seqs: usize,
111
112    /// Use no KV cache.
113    no_kv_cache: bool,
114
115    /// Chat template file with a JINJA file with `messages`, `add_generation_prompt`, `bos_token`, `eos_token`, and `unk_token` as inputs.
116    /// Used if the automatic deserialization fails. If this ends with `.json` (ie., it is a file) then that template is loaded.
117    chat_template: Option<String>,
118
119    /// Explicit JINJA chat template file (.jinja) to be used. If specified, this overrides all other chat templates.
120    jinja_explicit: Option<String>,
121
122    /// Source of the token for authentication.
123    /// Can be in the formats: `literal:<value>`, `env:<value>`, `path:<value>`, `cache` to use a cached token, or `none` to use no token.
124    /// Defaults to `cache`.
125    token_source: TokenSource,
126
127    /// Enter interactive mode instead of serving a chat server.
128    interactive_mode: bool,
129
130    /// Number of prefix caches to hold on the device. Other caches are evicted to the CPU based on a LRU strategy.
131    prefix_cache_n: usize,
132
133    /// NOTE: This can be omitted to use automatic device mapping!
134    /// Number of device layers to load and run on GPU(s). All others will be on the CPU.
135    /// If one GPU is used, then this value should be an integer. Otherwise, it follows the following pattern:
136    /// ORD:NUM;... Where ORD is a unique device ordinal and NUM is the number of layers for that device.
137    num_device_layers: Option<Vec<String>>,
138
139    /// In-situ quantization to apply.
140    in_situ_quant: Option<String>,
141
142    /// GPU memory to allocate for KV cache with PagedAttention in MBs.
143    /// PagedAttention is supported on CUDA and Metal. It is automatically activated on CUDA but not on Metal.
144    /// The priority is as follows: `pa-ctxt-len` > `pa-gpu-mem-usage` > `pa-gpu-mem`.
145    paged_attn_gpu_mem: Option<usize>,
146
147    /// Percentage of GPU memory to utilize after allocation of KV cache with PagedAttention, from 0 to 1.
148    /// If this is not set and the device is CUDA, it will default to `0.9`.
149    /// PagedAttention is supported on CUDA and Metal. It is automatically activated on CUDA but not on Metal.
150    /// The priority is as follows: `pa-ctxt-len` > `pa-gpu-mem-usage` > `pa-gpu-mem`.
151    paged_attn_gpu_mem_usage: Option<f32>,
152
153    /// Total context length to allocate the KV cache for (total number of tokens which the KV cache can hold).
154    /// PagedAttention is supported on CUDA and Metal. It is automatically activated on CUDA but not on Metal.
155    /// The priority is as follows: `pa-ctxt-len` > `pa-gpu-mem-usage` > `pa-gpu-mem`.
156    /// This is the default setting, and it defaults to the `max-seq-len` specified in after the model type.
157    paged_ctxt_len: Option<usize>,
158
159    /// Block size (number of tokens per block) for PagedAttention. If this is not set and the device is CUDA, it will default to 32.
160    /// PagedAttention is supported on CUDA and Metal. It is automatically activated on CUDA but not on Metal.
161    paged_attn_block_size: Option<usize>,
162
163    /// Enables or disables PagedAttention. By default, PagedAttention will be enabled for CUDA and disabled for Metal (and is not supported for CPU). Use this to override the default behavior.
164    paged_attn: Option<bool>,
165
166    /// Number of tokens to batch the prompt step into. This can help with OOM errors when in the prompt step, but reduces performance.
167    prompt_chunksize: Option<usize>,
168
169    /// Use CPU only
170    cpu: bool,
171
172    /// Enable searching compatible with the OpenAI `web_search_options` setting. This uses the BERT model specified below or the default.
173    enable_search: bool,
174
175    /// Specify a Hugging Face model ID for a BERT model to assist web searching. Defaults to Snowflake Arctic Embed L.
176    search_bert_model: Option<String>,
177
178    /// Optional override search callback
179    search_callback: Option<Arc<SearchCallback>>,
180
181    /// Optional MCP client configuration
182    mcp_client_config: Option<McpClientConfig>,
183
184    /// PagedAttention KV cache type
185    paged_cache_type: PagedCacheType,
186}

507    /// Builds the configured mistral.rs instance.
508    ///
509    /// ### Examples
510    ///
511    /// ```ignore
512    /// use mistralrs_server_core::mistralrs_for_server_builder::MistralRsForServerBuilder;
513    ///
514    /// let shared_mistralrs = MistralRsForServerBuilder::new()
515    ///     .with_model(model)
516    ///     .with_in_situ_quant("8".to_string())
517    ///     .set_paged_attn(Some(true))
518    ///     .build()
519    ///     .await?;
520    /// ```
521    pub async fn build(mut self) -> Result<SharedMistralRsState> {
522        let model = self.model.context("Model was None")?;
523
524        let tgt_non_granular_index = get_tgt_non_granular_index(&model);
525        let dtype = get_model_dtype(&model)?;
526        let auto_device_map_params = get_auto_device_map_params(&model)?;
527
528        if tgt_non_granular_index.is_some() {
529            self.max_seqs = 1;
530        }
531
532        let prompt_chunksize = match self.prompt_chunksize {
533            Some(0) => {
534                anyhow::bail!("`prompt_chunksize` must be a strictly positive integer, got 0.",)
535            }
536            Some(x) => Some(NonZeroUsize::new(x).unwrap()),
537            None => None,
538        };
539
540        let max_seq_len = auto_device_map_params.max_seq_len();
541
542        let device = if let Some(device) = self.device {
543            device
544        } else {
545            init_device(self.cpu, self.seed)?
546        };
547
548        let mapper = init_mapper(&self.num_device_layers, &auto_device_map_params);
549        let paged_attn = configure_paged_attn(&device, self.paged_attn);
550
551        // Allocate 0.5 GB of CPU memory just as a placeholder.
552        // Nothing happens here as we have no `swap_out`, see `_preempt_by_swap`.
553        let cache_config = init_cache_config(
554            self.paged_attn_block_size,
555            self.paged_attn_gpu_mem,
556            self.paged_attn_gpu_mem_usage,
557            self.paged_ctxt_len,
558            self.paged_cache_type,
559            !paged_attn,
560            max_seq_len,
561        )?;
562
563        // Configure this last to prevent arg moves
564        let loader: Box<dyn Loader> = LoaderBuilder::new(model)
565            .with_no_kv_cache(self.no_kv_cache)
566            .with_chat_template(self.chat_template)
567            .with_prompt_chunksize(prompt_chunksize)
568            .with_jinja_explicit(self.jinja_explicit)
569            .build()?;
570
571        mistralrs_instance_info(&*loader);
572
573        let isq = self
574            .in_situ_quant
575            .as_ref()
576            .and_then(|isq| parse_isq_value(isq, Some(&device)).ok());
577
578        let pipeline: LoadedPipeline = loader.load_model_from_hf(
579            None,
580            self.token_source,
581            &dtype,
582            &device,
583            false,
584            mapper,
585            isq,
586            cache_config,
587        )?;
588        info!("Model loaded.");
589
590        let scheduler_config = init_scheduler_config(&cache_config, &pipeline, self.max_seqs).await;
591
592        let bert_model = get_bert_model(self.enable_search, self.search_bert_model);
593
594        let mut builder = MistralRsBuilder::new(
595            pipeline,
596            scheduler_config,
597            !self.interactive_mode,
598            bert_model,
599        )
600        .with_opt_log(self.log)
601        .with_truncate_sequence(self.truncate_sequence)
602        .with_no_kv_cache(self.no_kv_cache)
603        .with_prefix_cache_n(self.prefix_cache_n);
604
605        // Add MCP client configuration if provided
606        if let Some(mcp_config) = self.mcp_client_config {
607            builder = builder.with_mcp_client(mcp_config);
608        }
609
610        let mistralrs = builder.build().await;
611
612        Ok(mistralrs)
613    }
614}

642/// Initializes the device mapping configuration for distributing model layers.
643fn init_mapper(
644    num_device_layers: &Option<Vec<String>>,
645    auto_device_map_params: &AutoDeviceMapParams,
646) -> DeviceMapSetting {
647    // Parse device mapper
648    if let Some(device_layers) = num_device_layers {
649        if device_layers.len() == 1 && device_layers[0].parse::<usize>().is_ok() {
650            let layers = device_layers[0].parse::<usize>().unwrap();
651            DeviceMapSetting::Map(DeviceMapMetadata::from_num_device_layers(vec![
652                DeviceLayerMapMetadata { ordinal: 0, layers },
653            ]))
654        } else {
655            let mut mapping = Vec::new();
656            for layer in device_layers {
657                let split = layer.splitn(2, ':').collect::<Vec<_>>();
658                if split.len() < 2 {
659                    panic!("Expected layer to be of format ORD:NUM, got {layer}");
660                }
661                let ord = split[0]
662                    .parse::<usize>()
663                    .unwrap_or_else(|_| panic!("Failed to parse {} as integer.", split[0]));
664                let num = split[1]
665                    .parse::<usize>()
666                    .unwrap_or_else(|_| panic!("Failed to parse {} as integer.", split[1]));
667                for DeviceLayerMapMetadata { ordinal, layers: _ } in &mapping {
668                    if *ordinal == ord {
669                        panic!("Duplicate ordinal {ord}");
670                    }
671                }
672                mapping.push(DeviceLayerMapMetadata {
673                    ordinal: ord,
674                    layers: num,
675                });
676            }
677            DeviceMapSetting::Map(DeviceMapMetadata::from_num_device_layers(mapping))
678        }
679    } else {
680        DeviceMapSetting::Auto(auto_device_map_params.clone())
681    }
682}

698/// Determines whether paged attention should be enabled based on device type and preferences.
699fn configure_paged_attn(device: &Device, paged_attn: Option<bool>) -> bool {
700    if device.is_cpu() {
701        if paged_attn == Some(true) {
702            warn!("Paged attention is not supported on CPU.");
703        }
704
705        defaults::PAGED_ATTN_CPU
706    } else if device.is_cuda() || mistralrs_core::distributed::use_nccl() {
707        paged_attn.unwrap_or(defaults::PAGED_ATTN_CUDA)
708    } else if device.is_metal() {
709        paged_attn.unwrap_or(defaults::PAGED_ATTN_METAL)
710    } else {
711        false
712    }
713}

715/// Initializes the cache configuration for paged attention based on provided parameters.
716fn init_cache_config(
717    paged_attn_block_size: Option<usize>,
718    paged_attn_gpu_mem: Option<usize>,
719    paged_attn_gpu_mem_usage: Option<f32>,
720    paged_ctxt_len: Option<usize>,
721    cache_type: PagedCacheType,
722    no_paged_attn: bool,
723    max_seq_len: usize,
724) -> Result<Option<PagedAttentionConfig>> {
725    match (
726        paged_attn_block_size,
727        paged_attn_gpu_mem,
728        paged_attn_gpu_mem_usage,
729        paged_ctxt_len,
730        paged_attn_supported(),
731        no_paged_attn,
732    ) {
733        (block_size, None, None, None, true, false) => Ok(Some(PagedAttentionConfig::new(
734            block_size,
735            512,
736            MemoryGpuConfig::ContextSize(max_seq_len),
737            cache_type,
738        )?)),
739        (block_size, None, None, Some(ctxt), true, false) => Ok(Some(PagedAttentionConfig::new(
740            block_size,
741            512,
742            MemoryGpuConfig::ContextSize(ctxt),
743            cache_type,
744        )?)),
745        (block_size, None, Some(f), None, true, false) => Ok(Some(PagedAttentionConfig::new(
746            block_size,
747            512,
748            MemoryGpuConfig::Utilization(f),
749            cache_type,
750        )?)),
751        (block_size, Some(m), None, None, true, false) => Ok(Some(PagedAttentionConfig::new(
752            block_size,
753            512,
754            MemoryGpuConfig::MbAmount(m),
755            cache_type,
756        )?)),
757        (block_size, Some(_m), Some(f), None, true, false) => {
758            info!("Both memory size, and usage were specified, defaulting to the usage value.");
759            Ok(Some(PagedAttentionConfig::new(
760                block_size,
761                512,
762                MemoryGpuConfig::Utilization(f),
763                cache_type,
764            )?))
765        }
766        (block_size, Some(_m), None, Some(ctxt), true, false) => {
767            info!("All memory size and ctxt len, defaulting to the context len value.");
768            Ok(Some(PagedAttentionConfig::new(
769                block_size,
770                512,
771                MemoryGpuConfig::ContextSize(ctxt),
772                cache_type,
773            )?))
774        }
775        (block_size, None, Some(f), Some(_ctxt), true, false) => {
776            info!("Both ctxt len and usage were specified, defaulting to the usage value.");
777            Ok(Some(PagedAttentionConfig::new(
778                block_size,
779                512,
780                MemoryGpuConfig::Utilization(f),
781                cache_type,
782            )?))
783        }
784        (_, _, _, _, _, _) => Ok(None),
785    }
786}

788/// Initializes the scheduler configuration based on cache settings and pipeline metadata.
789async fn init_scheduler_config(
790    cache_config: &Option<PagedAttentionConfig>,
791    pipeline: &LoadedPipeline,
792    args_max_seqs: usize,
793) -> SchedulerConfig {
794    if cache_config.is_some() {
795        // Handle case where we may have device mapping
796        if let Some(ref cache_config) = pipeline.lock().await.get_metadata().cache_config {
797            SchedulerConfig::PagedAttentionMeta {
798                max_num_seqs: args_max_seqs,
799                config: cache_config.clone(),
800            }
801        } else {
802            SchedulerConfig::DefaultScheduler {
803                method: DefaultSchedulerMethod::Fixed(args_max_seqs.try_into().unwrap()),
804            }
805        }
806    } else {
807        SchedulerConfig::DefaultScheduler {
808            method: DefaultSchedulerMethod::Fixed(args_max_seqs.try_into().unwrap()),
809        }
810    }
811}

813/// Configures PagedAttention based on two flags.
814///
815/// This function resolves the tri-state PagedAttention configuration from
816/// the mutually exclusive `paged_attn` and `no_paged_attn` flags.
817pub fn configure_paged_attn_from_flags(
818    paged_attn: bool,
819    no_paged_attn: bool,
820) -> Result<Option<bool>> {
821    match (paged_attn, no_paged_attn) {
822        (true, true) => {
823            anyhow::bail!("Error: `--paged-attn` and `--no-paged-attn` cannot be used together.");
824        }
825        (true, false) => Ok(Some(true)),
826        (false, true) => Ok(Some(false)),
827        (false, false) => Ok(None),
828    }
829}
