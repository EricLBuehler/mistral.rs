48/// Distinguish at runtime which kind of model we have loaded.
49#[derive(Clone)]
50enum LoadedModel {

52    Vision(Arc<Model>),
53}

60#[tokio::main]
61async fn main() -> Result<()> {
62    let cli = Cli::parse();
63    if cli.text_models.is_empty() && cli.vision_models.is_empty() {
64        eprintln!("At least one --text-model or --vision-model is required");
65        std::process::exit(1);
66    }
67
68    let default_isq = if cfg!(feature = "metal") {
69        IsqType::AFQ6
70    } else {
71        IsqType::Q6K
72    };
73
74    let device = best_device(false)?;
75
76    let isq = cli
77        .isq
78        .as_ref()
79        .and_then(|isq| parse_isq_value(isq, Some(&device)).ok());
80
81    let mut models: IndexMap<String, LoadedModel> = IndexMap::new();
82
83    // Insert text models first
84    for path in cli.text_models {
85        let name = std::path::Path::new(&path)
86            .file_name()
87            .and_then(|p| p.to_str())
88            .unwrap_or("text-model")
89            .to_string();
90        println!("📝 Loading text model: {name}");
91        let m = TextModelBuilder::new(path)
92            .with_isq(isq.unwrap_or(default_isq))
93            .with_logging()
94            .with_throughput_logging()
95            .build()
96            .await?;
97        models.insert(name, LoadedModel::Text(Arc::new(m)));
98    }
99
100    // Then insert vision models (preserving order)
101    for path in cli.vision_models {
102        let name = std::path::Path::new(&path)
103            .file_name()
104            .and_then(|p| p.to_str())
105            .unwrap_or("vision-model")
106            .to_string();
107        println!("🖼️  Loading vision model: {name}");
108        let m = VisionModelBuilder::new(path)
109            .with_isq(isq.unwrap_or(default_isq))
110            .with_logging()
111            .with_throughput_logging()
112            .build()
113            .await?;
114        models.insert(name, LoadedModel::Vision(Arc::new(m)));
115    }
116
117    let app_state = Arc::new(AppState {
118        models,
119        current: RwLock::new(None),
120    });
121
122    let app = Router::new()
123        .route("/ws", get(ws_handler))
124        .route("/api/upload_image", post(upload_image))
125        .route("/api/list_models", get(list_models))
126        .route("/api/select_model", post(select_model))
127        .nest_service(
128            "/",
129            get_service(ServeDir::new(concat!(
130                env!("CARGO_MANIFEST_DIR"),
131                "/static"
132            ))),
133        )
134        .layer(DefaultBodyLimit::max(50 * 1024 * 1024))
135        .with_state(app_state.clone());
136
137    let addr: SocketAddr = ([0, 0, 0, 0], cli.port.unwrap_or(8080)).into();
138    let listener = TcpListener::bind(addr).await?;
139    println!("🔌 listening on http://{}", addr);
140    axum::serve(listener, app).await?;
141    Ok(())
142}

144/// Accepts multipart image upload, stores it under `static/uploads/`, and returns its URL.
145async fn upload_image(
146    State(_app): State<Arc<AppState>>,
147    mut multipart: Multipart,
148) -> impl IntoResponse {
149    // Expect single "image" part
150    if let Ok(Some(field)) = multipart.next_field().await {
151        let ext = field
152            .file_name()
153            .and_then(|n| n.rsplit('.').next())
154            .unwrap_or("bin")
155            .to_string();
156
157        let data = match field.bytes().await {
158            Ok(b) => b,
159            Err(e) => {
160                error!("multipart bytes error: {}", e);
161                let msg = if e.to_string().contains("exceeded") {
162                    "image too large (limit 50 MB)"
163                } else {
164                    "failed to read upload"
165                };
166                return (StatusCode::BAD_REQUEST, msg).into_response();
167            }
168        };
169
170        // Ensure dir exists
171        let upload_dir = format!("{}/static/uploads", env!("CARGO_MANIFEST_DIR"));
172        if let Err(e) = tokio::fs::create_dir_all(&upload_dir).await {
173            error!("mkdir error: {}", e);
174            return (StatusCode::INTERNAL_SERVER_ERROR, "server error").into_response();
175        }
176
177        // Build unique filename
178        let filename = format!("{}.{}", Uuid::new_v4(), ext);
179        let path = format!("{}/{}", upload_dir, filename);
180
181        if let Err(e) = tokio::fs::write(&path, &data).await {
182            error!("write error: {}", e);
183            return (StatusCode::INTERNAL_SERVER_ERROR, "server error").into_response();
184        }
185
186        return (StatusCode::OK, Json(json!({ "url": path }))).into_response();
187    }
188
189    (StatusCode::BAD_REQUEST, "no image field").into_response()
190}

197async fn list_models(State(app): State<Arc<AppState>>) -> impl IntoResponse {
198    let items: Vec<_> = app
199        .models
200        .iter()
201        .map(|(n, m)| {
202            let kind = match m {
203                LoadedModel::Text(_) => "text",
204                LoadedModel::Vision(_) => "vision",
205            };
206            json!({ "name": n, "kind": kind })
207        })
208        .collect();
209    Json(json!({ "models": items }))
210}

212async fn select_model(
213    State(app): State<Arc<AppState>>,
214    Json(req): Json<SelectRequest>,
215) -> impl IntoResponse {
216    if app.models.contains_key(&req.name) {
217        let mut cur = app.current.write().await;
218        *cur = Some(req.name.clone());
219        (StatusCode::OK, "Model selected").into_response()
220    } else {
221        (StatusCode::NOT_FOUND, "Model not found").into_response()
222    }
223}

225/// Upgrades an HTTP request to a WebSocket connection.
226async fn ws_handler(ws: WebSocketUpgrade, State(app): State<Arc<AppState>>) -> impl IntoResponse {
227    ws.on_upgrade(move |socket| handle_socket(socket, app))

230/// Generic helper to stream tokens and forward them to the websocket.
231async fn stream_and_forward<Msgs, F, E>(
232    model: &Arc<Model>,
233    msgs: Msgs,
234    socket: &mut WebSocket,
235    mut on_token: F,
236    mut on_end: E,
237) -> Result<(), anyhow::Error>
238where
239    Msgs: mistralrs::RequestLike + Send + 'static,
240    F: FnMut(&str),
241    E: FnMut(),
242{
243    match model.stream_chat_request(msgs).await {
244        Ok(mut stream) => {
245            let mut assistant_reply = String::new();
246            while let Some(chunk) = stream.next().await {
247                if let mistralrs::Response::Chunk(resp) = chunk {
248                    if let Some(choice) = resp.choices.first() {
249                        if let Some(token) = &choice.delta.content {
250                            if socket.send(Message::Text(token.clone())).await.is_err() {
251                                break;
252                            }
253                            assistant_reply.push_str(token);
254                        }
255                    }
256                }
257            }
258            on_token(&assistant_reply);
259            on_end();
260            Ok(())
261        }
262        Err(e) => {
263            let _ = socket.send(Message::Text(format!("Error: {}", e))).await;
264            Err(e)
265        }
266    }
267}

269/// Per-connection task.
270async fn handle_socket(mut socket: WebSocket, app: Arc<AppState>) {
271    let mut text_msgs = TextMessages::new();
272    let mut vision_msgs = VisionMessages::new();
273    let mut image_buffer: Vec<image::DynamicImage> = Vec::new();
274    // `true` while we are streaming a reply back to the client.
275    let mut streaming = false;
276
277    while let Some(Ok(Message::Text(user_msg))) = socket.next().await {
278        // Allow client to request a context reset without closing the socket
279        if user_msg == CLEAR_CMD {
280            if streaming {
281                let _ = socket
282                    .send(Message::Text(
283                        "Cannot clear while assistant is replying.".into(),
284                    ))
285                    .await;
286            } else {
287                text_msgs = TextMessages::new();
288                vision_msgs = VisionMessages::new();
289                image_buffer.clear();
290                let _ = socket.send(Message::Text("[Context cleared]".into())).await;
291            }
292            continue;
293        }
294        let model_name_opt = { app.current.read().await.clone() };
295        let Some(model_name) = model_name_opt else {
296            let _ = socket
297                .send(Message::Text(
298                    "No model selected. Choose one in the sidebar.".into(),
299                ))
300                .await;
301            continue;
302        };
303        let Some(model_loaded) = app.models.get(&model_name).cloned() else {
304            let _ = socket
305                .send(Message::Text("Selected model not found.".into()))
306                .await;
307            continue;
308        };
309
310        match model_loaded {
311            LoadedModel::Text(model) => {
312                text_msgs = text_msgs.add_message(TextMessageRole::User, &user_msg);
313                let msgs_snapshot = text_msgs.clone();
314
315                streaming = true;
316                let stream_res = stream_and_forward(
317                    &model,
318                    msgs_snapshot,
319                    &mut socket,
320                    |tok| {
321                        let cur = mem::take(&mut text_msgs);
322                        text_msgs = cur.add_message(TextMessageRole::Assistant, tok);
323                    },
324                    || streaming = false,
325                )
326                .await;
327                if let Err(e) = stream_res {
328                    error!("stream error: {}", e);
329                }
330            }
331            LoadedModel::Vision(model) => {
332                // --- Vision input routing ---
333                if let Ok(val) = serde_json::from_str::<Value>(&user_msg) {
334                    // Case 1: pure image payload => buffer it and wait for a prompt
335                    if let Some(url) = val.get("image").and_then(|v| v.as_str()) {
336                        // load & decode
337                        match tokio::fs::read(&url).await {
338                            Ok(bytes) => match image::load_from_memory(&bytes) {
339                                Ok(img) => {
340                                    image_buffer.push(img);
341                                }
342                                Err(e) => {
343                                    error!("image decode error: {}", e);
344                                    let _ =
345                                        socket.send(Message::Text(format!("Error: {}", e))).await;
346                                }
347                            },
348                            Err(e) => {
349                                error!("image read error: {}", e);
350                                let _ = socket.send(Message::Text(format!("Error: {}", e))).await;
351                            }
352                        }
353                        // Skip sending to model until we get a prompt
354                        continue;
355                    } else {
356                        // Fallback: treat whole JSON as text
357                        vision_msgs = vision_msgs.add_message(TextMessageRole::User, &user_msg);
358                    }
359                } else {
360                    // Plain-text prompt arrives here
361                    if image_buffer.is_empty() {
362                        vision_msgs = vision_msgs.add_message(TextMessageRole::User, &user_msg);
363                    } else {
364                        match vision_msgs.clone().add_image_message(
365                            TextMessageRole::User,
366                            &user_msg,
367                            image_buffer.clone(),
368                            &model,
369                        ) {
370                            Ok(updated) => {
371                                vision_msgs = updated;
372                                image_buffer.clear();
373                            }
374                            Err(e) => {
375                                error!("image prompt error: {}", e);
376                                let _ = socket.send(Message::Text(format!("Error: {}", e))).await;
377                            }
378                        }
379                    }
380                }
381
382                let msgs_snapshot = vision_msgs.clone();
383                streaming = true;
384                let stream_res = stream_and_forward(
385                    &model,
386                    msgs_snapshot,
387                    &mut socket,
388                    |tok| {
389                        let cur = mem::take(&mut vision_msgs);
390                        vision_msgs = cur.add_message(TextMessageRole::Assistant, tok);
391                    },
392                    || streaming = false,
393                )
394                .await;
395                if let Err(e) = stream_res {
396                    error!("stream error: {}", e);
397                }
398            }
399        }
400    }
