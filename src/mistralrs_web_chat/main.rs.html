27#[tokio::main]
28async fn main() -> Result<()> {
29    let cli = Cli::parse();
30    if cli.text_models.is_empty() && cli.vision_models.is_empty() && cli.speech_models.is_empty() {
31        eprintln!("At least one --text-model, --vision-model, or --speech-model is required");
32        std::process::exit(1);
33    }
34
35    let default_isq = if cfg!(feature = "metal") {
36        IsqType::AFQ6
37    } else {
38        IsqType::Q6K
39    };
40
41    let device = best_device(false)?;
42
43    let isq = cli
44        .isq
45        .as_ref()
46        .and_then(|isq| parse_isq_value(isq, Some(&device)).ok());
47
48    let mut models: IndexMap<String, LoadedModel> = IndexMap::new();
49
50    // Insert text models first
51    for path in cli.text_models {
52        let name = std::path::Path::new(&path)
53            .file_name()
54            .and_then(|p| p.to_str())
55            .unwrap_or("text-model")
56            .to_string();
57        println!("üìù Loading text model: {name}");
58        let m = TextModelBuilder::new(path)
59            .with_isq(isq.unwrap_or(default_isq))
60            .with_logging()
61            .with_throughput_logging()
62            .build()
63            .await?;
64        models.insert(name, LoadedModel::Text(Arc::new(m)));
65    }
66
67    // Then insert vision models (preserving order)
68    for path in cli.vision_models {
69        let name = std::path::Path::new(&path)
70            .file_name()
71            .and_then(|p| p.to_str())
72            .unwrap_or("vision-model")
73            .to_string();
74        println!("üñºÔ∏è  Loading vision model: {name}");
75        let m = VisionModelBuilder::new(path)
76            .with_isq(isq.unwrap_or(default_isq))
77            .with_logging()
78            .with_throughput_logging()
79            .build()
80            .await?;
81        models.insert(name, LoadedModel::Vision(Arc::new(m)));
82    }
83
84    // Then insert speech models (text-to-speech)
85    for path in cli.speech_models {
86        let name = std::path::Path::new(&path)
87            .file_name()
88            .and_then(|p| p.to_str())
89            .unwrap_or("speech-model")
90            .to_string();
91        println!("üîä Loading speech model: {name}");
92        let m = SpeechModelBuilder::new(path.clone(), SpeechLoaderType::Dia)
93            .with_logging()
94            .build()
95            .await?;
96        models.insert(name, LoadedModel::Speech(Arc::new(m)));
97    }
98
99    // Initialize cache directory and chats subdirectory
100    let base_cache = utils::get_cache_dir();
101    println!("üîß Using cache directory: {}", base_cache.display());
102    let chats_dir = base_cache.join("chats").to_string_lossy().to_string();
103    tokio::fs::create_dir_all(&chats_dir).await?;
104    // Initialize speech output directory for generated wav files
105    let speech_dir = base_cache.join("speech").to_string_lossy().to_string();
106    tokio::fs::create_dir_all(&speech_dir).await?;
107    let mut next_id = 1u32;
108    if let Ok(mut dir) = fs::read_dir(&chats_dir).await {
109        while let Ok(Some(entry)) = dir.next_entry().await {
110            if let Some(name) = entry.file_name().to_str() {
111                if let Some(num) = name
112                    .strip_prefix("chat_")
113                    .and_then(|s| s.strip_suffix(".json"))
114                {
115                    if let Ok(n) = num.parse::<u32>() {
116                        next_id = next_id.max(n + 1);
117                    }
118                }
119            }
120        }
121    }
122
123    let app_state = Arc::new(AppState {
124        models,
125        current: tokio::sync::RwLock::new(None),
126        chats_dir,
127        speech_dir: speech_dir.clone(),
128        current_chat: tokio::sync::RwLock::new(None),
129        next_chat_id: tokio::sync::RwLock::new(next_id),
130    });
131
132    let app = Router::new()
133        .route("/ws", get(ws_handler))
134        .route("/api/upload_image", post(upload_image))
135        .route("/api/upload_text", post(upload_text))
136        .route("/api/list_models", get(list_models))
137        .route("/api/select_model", post(select_model))
138        .route("/api/list_chats", get(list_chats))
139        .route("/api/new_chat", post(new_chat))
140        .route("/api/delete_chat", post(delete_chat))
141        .route("/api/load_chat", post(load_chat))
142        .route("/api/rename_chat", post(rename_chat))
143        .route("/api/append_message", post(append_message))
144        // Text-to-speech generation endpoint
145        .route("/api/generate_speech", post(generate_speech))
146        // Serve generated speech files
147        .nest_service("/speech", get_service(ServeDir::new(speech_dir.clone())))
148        // Serve static assets
149        .nest_service(
150            "/",
151            get_service(ServeDir::new(concat!(
152                env!("CARGO_MANIFEST_DIR"),
153                "/static"
154            ))),
155        )
156        .layer(DefaultBodyLimit::max(50 * 1024 * 1024))
157        .with_state(app_state.clone());
158
159    let addr: SocketAddr = ([0, 0, 0, 0], cli.port.unwrap_or(8080)).into();
160    let listener = TcpListener::bind(addr).await?;
161    println!("üîå listening on http://{}", addr);
162    axum::serve(listener, app).await?;
163    Ok(())
164}
