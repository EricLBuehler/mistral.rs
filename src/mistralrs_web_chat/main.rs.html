34async fn static_handler(uri: Uri) -> impl IntoResponse {
35    let path = uri.path().trim_start_matches('/');
36    // Serve index.html when requesting the root path
37    let path = if path.is_empty() { "index.html" } else { path };
38    if let Some(file) = STATIC_DIR.get_file(path) {
39        let mime = mime_guess::from_path(path).first_or_octet_stream();
40        Response::builder()
41            .status(StatusCode::OK)
42            .header(http::header::CONTENT_TYPE, mime.as_ref())
43            .body(Body::from(file.contents()))
44            .unwrap()
45    } else {
46        Response::builder()
47            .status(StatusCode::NOT_FOUND)
48            .body(Body::from("Not Found"))
49            .unwrap()
50    }
51}

53#[tokio::main]
54async fn main() -> Result<()> {
55    let cli = Cli::parse();
56    if cli.text_models.is_empty() && cli.vision_models.is_empty() && cli.speech_models.is_empty() {
57        eprintln!("At least one --text-model, --vision-model, or --speech-model is required");
58        std::process::exit(1);
59    }
60
61    let default_isq = if cfg!(feature = "metal") {
62        IsqType::AFQ6
63    } else {
64        IsqType::Q6K
65    };
66
67    let device = best_device(false)?;
68
69    let isq = cli
70        .isq
71        .as_ref()
72        .and_then(|isq| parse_isq_value(isq, Some(&device)).ok());
73
74    // Determine embedding model for web search if enabled
75    let search_embedding_model: Option<SearchEmbeddingModel> = if cli.enable_search {
76        Some(cli.search_embedding_model.unwrap_or_default())
77    } else {
78        None
79    };
80    let mut models: IndexMap<String, LoadedModel> = IndexMap::new();
81
82    // Insert text models first
83    for path in cli.text_models {
84        let name = std::path::Path::new(&path)
85            .file_name()
86            .and_then(|p| p.to_str())
87            .unwrap_or("text-model")
88            .to_string();
89        println!("üìù Loading text model: {name}");
90        let mut builder = TextModelBuilder::new(path)
91            .with_isq(isq.unwrap_or(default_isq))
92            .with_logging()
93            .with_throughput_logging();
94        if let Some(ref search_embedding_model) = search_embedding_model {
95            builder = builder.with_search(*search_embedding_model);
96        }
97        let m = builder.build().await?;
98        models.insert(name, LoadedModel::Text(Arc::new(m)));
99    }
100
101    // Then insert vision models (preserving order)
102    for path in cli.vision_models {
103        let name = std::path::Path::new(&path)
104            .file_name()
105            .and_then(|p| p.to_str())
106            .unwrap_or("vision-model")
107            .to_string();
108        println!("üñºÔ∏è  Loading vision model: {name}");
109        let mut builder = VisionModelBuilder::new(path)
110            .with_isq(isq.unwrap_or(default_isq))
111            .with_logging()
112            .with_throughput_logging();
113        if let Some(ref search_embedding_model) = search_embedding_model {
114            builder = builder.with_search(*search_embedding_model);
115        }
116        let m = builder.build().await?;
117        models.insert(name, LoadedModel::Vision(Arc::new(m)));
118    }
119
120    // Then insert speech models (text-to-speech)
121    for path in cli.speech_models {
122        let name = std::path::Path::new(&path)
123            .file_name()
124            .and_then(|p| p.to_str())
125            .unwrap_or("speech-model")
126            .to_string();
127        println!("üîä Loading speech model: {name}");
128        let m = SpeechModelBuilder::new(path.clone(), SpeechLoaderType::Dia)
129            .with_logging()
130            .build()
131            .await?;
132        models.insert(name, LoadedModel::Speech(Arc::new(m)));
133    }
134
135    // Initialize cache directory and chats subdirectory
136    let base_cache = utils::get_cache_dir();
137    println!("üîß Using cache directory: {}", base_cache.display());
138    let chats_dir = base_cache.join("chats").to_string_lossy().to_string();
139    tokio::fs::create_dir_all(&chats_dir).await?;
140    // Initialize speech output directory for generated wav files
141    let speech_dir = base_cache.join("speech").to_string_lossy().to_string();
142    tokio::fs::create_dir_all(&speech_dir).await?;
143    let mut next_id = 1u32;
144    if let Ok(mut dir) = fs::read_dir(&chats_dir).await {
145        while let Ok(Some(entry)) = dir.next_entry().await {
146            if let Some(name) = entry.file_name().to_str() {
147                if let Some(num) = name
148                    .strip_prefix("chat_")
149                    .and_then(|s| s.strip_suffix(".json"))
150                {
151                    if let Ok(n) = num.parse::<u32>() {
152                        next_id = next_id.max(n + 1);
153                    }
154                }
155            }
156        }
157    }
158
159    // Build default generation parameters from CLI
160    let default_params = GenerationParams {
161        temperature: cli.temperature,
162        top_p: cli.top_p,
163        top_k: cli.top_k,
164        max_tokens: cli.max_tokens,
165        repetition_penalty: cli.repetition_penalty,
166        system_prompt: cli.system_prompt.clone(),
167    };
168
169    let app_state = Arc::new(AppState {
170        models,
171        current: tokio::sync::RwLock::new(None),
172        chats_dir,
173        speech_dir: speech_dir.clone(),
174        current_chat: tokio::sync::RwLock::new(None),
175        next_chat_id: tokio::sync::RwLock::new(next_id),
176        default_params,
177        search_enabled: cli.enable_search,
178    });
179
180    let app = Router::new()
181        .route("/ws", get(ws_handler))
182        .route("/api/upload_image", post(upload_image))
183        .route("/api/upload_text", post(upload_text))
184        .route("/api/upload_audio", post(upload_audio))
185        .route("/api/list_models", get(list_models))
186        .route("/api/select_model", post(select_model))
187        .route("/api/list_chats", get(list_chats))
188        .route("/api/new_chat", post(new_chat))
189        .route("/api/delete_chat", post(delete_chat))
190        .route("/api/load_chat", post(load_chat))
191        .route("/api/rename_chat", post(rename_chat))
192        .route("/api/append_message", post(append_message))
193        // Generation settings endpoints
194        .route("/api/settings", get(get_settings))
195        // Text-to-speech generation endpoint
196        .route("/api/generate_speech", post(generate_speech))
197        // Serve generated speech files
198        .nest_service("/speech", get_service(ServeDir::new(speech_dir.clone())))
199        // Serve embedded static assets for the root path
200        .route("/", get(static_handler))
201        .route("/{*path}", get(static_handler))
202        .layer(DefaultBodyLimit::max(50 * 1024 * 1024))
203        .with_state(app_state.clone());
204
205    let host = cli.host.as_deref().unwrap_or("0.0.0.0");
206    let port = cli.port.unwrap_or(8080);
207    let bind_addr = format!("{}:{}", host, port);
208    let listener = TcpListener::bind(&bind_addr).await?;
209    println!("üîå Listening on http://{}", bind_addr);
210    axum::serve(listener, app).await?;
211    Ok(())
212}
