83#[tokio::main]
84async fn main() -> Result<()> {
85    let cli = Cli::parse();
86    if cli.text_models.is_empty() && cli.vision_models.is_empty() {
87        eprintln!("At least one --text-model or --vision-model is required");
88        std::process::exit(1);
89    }
90
91    let default_isq = if cfg!(feature = "metal") {
92        IsqType::AFQ6
93    } else {
94        IsqType::Q6K
95    };
96
97    let device = best_device(false)?;
98
99    let isq = cli
100        .isq
101        .as_ref()
102        .and_then(|isq| parse_isq_value(isq, Some(&device)).ok());
103
104    let mut models: IndexMap<String, LoadedModel> = IndexMap::new();
105
106    // Insert text models first
107    for path in cli.text_models {
108        let name = std::path::Path::new(&path)
109            .file_name()
110            .and_then(|p| p.to_str())
111            .unwrap_or("text-model")
112            .to_string();
113        println!("📝 Loading text model: {name}");
114        let m = TextModelBuilder::new(path)
115            .with_isq(isq.unwrap_or(default_isq))
116            .with_logging()
117            .with_throughput_logging()
118            .build()
119            .await?;
120        models.insert(name, LoadedModel::Text(Arc::new(m)));
121    }
122
123    // Then insert vision models (preserving order)
124    for path in cli.vision_models {
125        let name = std::path::Path::new(&path)
126            .file_name()
127            .and_then(|p| p.to_str())
128            .unwrap_or("vision-model")
129            .to_string();
130        println!("🖼️  Loading vision model: {name}");
131        let m = VisionModelBuilder::new(path)
132            .with_isq(isq.unwrap_or(default_isq))
133            .with_logging()
134            .with_throughput_logging()
135            .build()
136            .await?;
137        models.insert(name, LoadedModel::Vision(Arc::new(m)));
138    }
139
140    let chats_dir = format!("{}/cache/chats", env!("CARGO_MANIFEST_DIR"));
141    tokio::fs::create_dir_all(&chats_dir).await?;
142    let mut next_id = 1u32;
143    if let Ok(mut dir) = fs::read_dir(&chats_dir).await {
144        while let Ok(Some(entry)) = dir.next_entry().await {
145            if let Some(name) = entry.file_name().to_str() {
146                if let Some(num) = name
147                    .strip_prefix("chat_")
148                    .and_then(|s| s.strip_suffix(".json"))
149                {
150                    if let Ok(n) = num.parse::<u32>() {
151                        next_id = next_id.max(n + 1);
152                    }
153                }
154            }
155        }
156    }
157
158    let app_state = Arc::new(AppState {
159        models,
160        current: RwLock::new(None),
161        chats_dir,
162        current_chat: RwLock::new(None),
163        next_chat_id: RwLock::new(next_id),
164    });
165
166    let app = Router::new()
167        .route("/ws", get(ws_handler))
168        .route("/api/upload_image", post(upload_image))
169        .route("/api/list_models", get(list_models))
170        .route("/api/select_model", post(select_model))
171        .route("/api/list_chats", get(list_chats))
172        .route("/api/new_chat", post(new_chat))
173        .route("/api/delete_chat", post(delete_chat))
174        .route("/api/load_chat", post(load_chat))
175        .route("/api/rename_chat", post(rename_chat))
176        .nest_service(
177            "/",
178            get_service(ServeDir::new(concat!(
179                env!("CARGO_MANIFEST_DIR"),
180                "/static"
181            ))),
182        )
183        .layer(DefaultBodyLimit::max(50 * 1024 * 1024))
184        .with_state(app_state.clone());
185
186    let addr: SocketAddr = ([0, 0, 0, 0], cli.port.unwrap_or(8080)).into();
187    let listener = TcpListener::bind(addr).await?;
188    println!("🔌 listening on http://{}", addr);
189    axum::serve(listener, app).await?;
190    Ok(())
191}

193/// Accepts multipart image upload, stores it under `cache/uploads/`, and returns its URL.
194async fn upload_image(
195    State(_app): State<Arc<AppState>>,
196    mut multipart: Multipart,
197) -> impl IntoResponse {
198    // Expect single "image" part
199    if let Ok(Some(field)) = multipart.next_field().await {
200        let ext = field
201            .file_name()
202            .and_then(|n| n.rsplit('.').next())
203            .unwrap_or("bin")
204            .to_string();
205
206        let data = match field.bytes().await {
207            Ok(b) => b,
208            Err(e) => {
209                error!("multipart bytes error: {}", e);
210                let msg = if e.to_string().contains("exceeded") {
211                    "image too large (limit 50 MB)"
212                } else {
213                    "failed to read upload"
214                };
215                return (StatusCode::BAD_REQUEST, msg).into_response();
216            }
217        };
218
219        // Ensure dir exists
220        let upload_dir = format!("{}/cache/uploads", env!("CARGO_MANIFEST_DIR"));
221        if let Err(e) = tokio::fs::create_dir_all(&upload_dir).await {
222            error!("mkdir error: {}", e);
223            return (StatusCode::INTERNAL_SERVER_ERROR, "server error").into_response();
224        }
225
226        // Build unique filename
227        let filename = format!("{}.{}", Uuid::new_v4(), ext);
228        let path = format!("{}/{}", upload_dir, filename);
229
230        if let Err(e) = tokio::fs::write(&path, &data).await {
231            error!("write error: {}", e);
232            return (StatusCode::INTERNAL_SERVER_ERROR, "server error").into_response();
233        }
234
235        return (StatusCode::OK, Json(json!({ "url": path }))).into_response();
236    }
237
238    (StatusCode::BAD_REQUEST, "no image field").into_response()
239}

261async fn select_model(
262    State(app): State<Arc<AppState>>,
263    Json(req): Json<SelectRequest>,
264) -> impl IntoResponse {
265    if let Some(model_loaded) = app.models.get(&req.name) {
266        {
267            let mut cur = app.current.write().await;
268            *cur = Some(req.name.clone());
269        }
270        // --- sync the active chat file so future loads use the correct model ---
271        if let Some(chat_id) = app.current_chat.read().await.clone() {
272            let path = format!("{}/{}.json", app.chats_dir, chat_id);
273            if let Ok(data) = fs::read(&path).await {
274                if let Ok(mut chat) = serde_json::from_slice::<ChatFile>(&data) {
275                    chat.model = req.name.clone();
276                    chat.kind = match model_loaded {
277                        LoadedModel::Text(_) => "text".into(),
278                        LoadedModel::Vision(_) => "vision".into(),
279                    };
280                    // ignore write errors; not fatal for select_model
281                    if let Ok(bytes) = serde_json::to_vec_pretty(&chat) {
282                        let _ = tokio::fs::write(&path, bytes).await;
283                    }
284                }
285            }
286        }
287        (StatusCode::OK, "Model selected").into_response()
288    } else {
289        (StatusCode::NOT_FOUND, "Model not found").into_response()
290    }
291}

293/// ---------- Chat‑history helpers & endpoints ----------
294async fn list_chats(State(app): State<Arc<AppState>>) -> impl IntoResponse {
295    let mut chats = Vec::new();
296    if let Ok(mut dir) = fs::read_dir(&app.chats_dir).await {
297        while let Ok(Some(entry)) = dir.next_entry().await {
298            if let Some(name) = entry.file_name().to_str() {
299                if name.ends_with(".json") {
300                    let id = name.trim_end_matches(".json");
301                    let data = fs::read(format!("{}/{}", app.chats_dir, name)).await.ok();
302                    let (title, created) = data
303                        .and_then(|bytes| serde_json::from_slice::<ChatFile>(&bytes).ok())
304                        .map(|c| (c.title, c.created_at))
305                        .map(|(title, created)| (title.unwrap_or_default(), created))
306                        .unwrap_or_else(|| (String::new(), String::new()));
307                    chats.push(json!({ "id": id, "title": title, "created_at": created }));
308                }
309            }
310        }
311    }
312    Json(json!({ "chats": chats }))
313}

320async fn new_chat(
321    State(app): State<Arc<AppState>>,
322    Json(req): Json<NewChatRequest>,
323) -> impl IntoResponse {
324    let mut id_guard = app.next_chat_id.write().await;
325    let id = *id_guard;
326    *id_guard += 1;
327    drop(id_guard);
328
329    let chat_id = format!("chat_{}", id);
330    let path = format!("{}/{}.json", app.chats_dir, chat_id);
331
332    let kind = if let Some(m) = app.models.get(&req.model) {
333        match m {
334            LoadedModel::Text(_) => "text",
335            LoadedModel::Vision(_) => "vision",
336        }
337    } else {
338        "text"
339    }
340    .to_string();
341
342    let chat = ChatFile {
343        title: None,
344        model: req.model.clone(),
345        kind,
346        created_at: Utc::now().to_rfc3339(),
347        messages: Vec::new(),
348    };
349    let _ = fs::write(&path, serde_json::to_vec_pretty(&chat).unwrap()).await;
350
351    {
352        let mut cur_chat = app.current_chat.write().await;
353        *cur_chat = Some(chat_id.clone());
354        let mut cur_model = app.current.write().await;
355        *cur_model = Some(req.model.clone());
356    }
357
358    Json(json!({ "id": chat_id }))
359}

390async fn load_chat(
391    State(app): State<Arc<AppState>>,
392    Json(req): Json<LoadChatRequest>,
393) -> impl IntoResponse {

397            Ok(chat) => {
398                {

400                    *cur_chat = Some(req.id.clone());
401                    if app.models.contains_key(&chat.model) {
402                        let mut cur_model = app.current.write().await;
403                        *cur_model = Some(chat.model.clone());
404                    }
405                }
406                Json(json!({
407                    "id": req.id,
408                    "title": chat.title.clone().unwrap_or_default(),
409                    "model": chat.model,
410                    "kind": chat.kind,
411                    "created_at": chat.created_at.clone(),
412                    "messages": chat.messages
413                }))
414                .into_response()
415            }
416            Err(_) => (StatusCode::INTERNAL_SERVER_ERROR, "corrupt chat").into_response(),
417        },
418        Err(_) => (StatusCode::NOT_FOUND, "chat not found").into_response(),
419    }

445async fn append_chat_message(app: &Arc<AppState>, role: &str, content: &str) -> Result<()> {
446    // Ignore replay helpers sent from the front‑end,
447    // which look like: {"restore":{...}}
448    if content.trim_start().starts_with("{\"restore\":") {
449        return Ok(());
450    }
451    let chat_opt = app.current_chat.read().await.clone();
452    let Some(chat_id) = chat_opt else {
453        return Ok(());
454    };
455    let path = format!("{}/{}.json", app.chats_dir, chat_id);
456
457    let mut chat: ChatFile = if let Ok(data) = fs::read(&path).await {
458        serde_json::from_slice(&data).unwrap_or(ChatFile {
459            title: None,
460            model: app.current.read().await.clone().unwrap_or_default(),
461            kind: String::new(),
462            created_at: Utc::now().to_rfc3339(),
463            messages: Vec::new(),
464        })
465    } else {
466        ChatFile {
467            title: None,
468            model: app.current.read().await.clone().unwrap_or_default(),
469            kind: String::new(),
470            created_at: Utc::now().to_rfc3339(),
471            messages: Vec::new(),
472        }
473    };
474
475    chat.messages.push(ChatMessage {
476        role: role.into(),
477        content: content.into(),
478    });
479    fs::write(&path, serde_json::to_vec_pretty(&chat)?).await?;
480    Ok(())
481}

488/// Generic helper to stream tokens and forward them to the websocket.
489async fn stream_and_forward<Msgs, F, E>(
490    model: &Arc<Model>,
491    msgs: Msgs,
492    socket: &mut WebSocket,
493    mut on_token: F,
494    mut on_end: E,
495) -> Result<(), anyhow::Error>
496where
497    Msgs: mistralrs::RequestLike + Send + 'static,
498    F: FnMut(&str),
499    E: FnMut(),
500{
501    match model.stream_chat_request(msgs).await {
502        Ok(mut stream) => {
503            let mut assistant_reply = String::new();
504            while let Some(chunk) = stream.next().await {
505                if let mistralrs::Response::Chunk(resp) = chunk {
506                    if let Some(choice) = resp.choices.first() {
507                        if let Some(token) = &choice.delta.content {
508                            if socket.send(Message::Text(token.clone())).await.is_err() {
509                                break;
510                            }
511                            assistant_reply.push_str(token);
512                        }
513                    }
514                }
515            }
516            on_token(&assistant_reply);
517            on_end();
518            Ok(())
519        }
520        Err(e) => {
521            let _ = socket.send(Message::Text(format!("Error: {}", e))).await;
522            Err(e)
523        }
524    }
525}

527/// Per-connection task.
528async fn handle_socket(mut socket: WebSocket, app: Arc<AppState>) {
529    let mut text_msgs = TextMessages::new();
530    let mut vision_msgs = VisionMessages::new();
531    let mut image_buffer: Vec<image::DynamicImage> = Vec::new();
532    // `true` while we are streaming a reply back to the client.
533    let mut streaming = false;
534
535    while let Some(Ok(Message::Text(user_msg))) = socket.next().await {
536        // Allow client to request a context reset without closing the socket
537        if user_msg == CLEAR_CMD {
538            if streaming {
539                let _ = socket
540                    .send(Message::Text(
541                        "Cannot clear while assistant is replying.".into(),
542                    ))
543                    .await;
544            } else {
545                text_msgs = TextMessages::new();
546                vision_msgs = VisionMessages::new();
547                image_buffer.clear();
548                let _ = socket.send(Message::Text("[Context cleared]".into())).await;
549            }
550            continue;
551        }
552        // Handle front‑end replay helper messages without triggering inference
553        if user_msg.trim_start().starts_with("{\"restore\":") {
554            if let Ok(val) = serde_json::from_str::<Value>(&user_msg) {
555                if let Some(obj) = val.get("restore") {
556                    if let (Some(role), Some(content)) = (
557                        obj.get("role").and_then(|v| v.as_str()),
558                        obj.get("content").and_then(|v| v.as_str()),
559                    ) {
560                        // Push into the local context so the model sees the history
561                        match app.current.read().await.as_deref() {
562                            Some(model_name) if app.models.get(model_name).is_some() => {
563                                match app.models.get(model_name).unwrap() {
564                                    LoadedModel::Text(_) => {
565                                        let role_enum = if role == "assistant" {
566                                            TextMessageRole::Assistant
567                                        } else {
568                                            TextMessageRole::User
569                                        };
570                                        text_msgs = text_msgs.add_message(role_enum, content);
571                                    }
572                                    LoadedModel::Vision(_) => {
573                                        let role_enum = if role == "assistant" {
574                                            TextMessageRole::Assistant
575                                        } else {
576                                            TextMessageRole::User
577                                        };
578                                        vision_msgs = vision_msgs.add_message(role_enum, content);
579                                    }
580                                }
581                            }
582                            _ => {}
583                        }
584                    }
585                }
586            }
587            // Skip normal handling (no saving, no inference)
588            continue;
589        }
590        let model_name_opt = { app.current.read().await.clone() };
591        let Some(model_name) = model_name_opt else {
592            let _ = socket
593                .send(Message::Text(
594                    "No model selected. Choose one in the sidebar.".into(),
595                ))
596                .await;
597            continue;
598        };
599        let Some(model_loaded) = app.models.get(&model_name).cloned() else {
600            let _ = socket
601                .send(Message::Text("Selected model not found.".into()))
602                .await;
603            continue;
604        };
605
606        match model_loaded {
607            LoadedModel::Text(model) => {
608                text_msgs = text_msgs.add_message(TextMessageRole::User, &user_msg);
609                if let Err(e) = append_chat_message(&app, "user", &user_msg).await {
610                    error!("chat save error: {}", e);
611                }
612                let mut assistant_content = String::new();
613                let msgs_snapshot = text_msgs.clone();
614
615                streaming = true;
616                let stream_res = stream_and_forward(
617                    &model,
618                    msgs_snapshot,
619                    &mut socket,
620                    |tok| {
621                        assistant_content = tok.to_string();
622                        let cur = mem::take(&mut text_msgs);
623                        text_msgs = cur.add_message(TextMessageRole::Assistant, tok);
624                    },
625                    || streaming = false,
626                )
627                .await;
628                if !assistant_content.is_empty() {
629                    let _ = append_chat_message(&app, "assistant", &assistant_content).await;
630                }
631                if let Err(e) = stream_res {
632                    error!("stream error: {}", e);
633                }
634            }
635            LoadedModel::Vision(model) => {
636                // --- Vision input routing ---
637                if let Ok(val) = serde_json::from_str::<Value>(&user_msg) {
638                    // Case 1: pure image payload => buffer it and wait for a prompt
639                    if let Some(url) = val.get("image").and_then(|v| v.as_str()) {
640                        // load & decode
641                        match tokio::fs::read(&url).await {
642                            Ok(bytes) => match image::load_from_memory(&bytes) {
643                                Ok(img) => {
644                                    image_buffer.push(img);
645                                }
646                                Err(e) => {
647                                    error!("image decode error: {}", e);
648                                    let _ =
649                                        socket.send(Message::Text(format!("Error: {}", e))).await;
650                                }
651                            },
652                            Err(e) => {
653                                error!("image read error: {}", e);
654                                let _ = socket.send(Message::Text(format!("Error: {}", e))).await;
655                            }
656                        }
657                        // Skip sending to model until we get a prompt
658                        continue;
659                    } else {
660                        // Fallback: treat whole JSON as text
661                        vision_msgs = vision_msgs.add_message(TextMessageRole::User, &user_msg);
662                        if let Err(e) = append_chat_message(&app, "user", &user_msg).await {
663                            error!("chat save error: {}", e);
664                        }
665                    }
666                } else {
667                    // Plain-text prompt arrives here
668                    if image_buffer.is_empty() {
669                        vision_msgs = vision_msgs.add_message(TextMessageRole::User, &user_msg);
670                        if let Err(e) = append_chat_message(&app, "user", &user_msg).await {
671                            error!("chat save error: {}", e);
672                        }
673                    } else {
674                        match vision_msgs.clone().add_image_message(
675                            TextMessageRole::User,
676                            &user_msg,
677                            image_buffer.clone(),
678                            &model,
679                        ) {
680                            Ok(updated) => {
681                                vision_msgs = updated;
682                                image_buffer.clear();
683                            }
684                            Err(e) => {
685                                error!("image prompt error: {}", e);
686                                let _ = socket.send(Message::Text(format!("Error: {}", e))).await;
687                            }
688                        }
689                    }
690                }
691
692                let msgs_snapshot = vision_msgs.clone();
693                streaming = true;
694                let mut assistant_content = String::new();
695                let stream_res = stream_and_forward(
696                    &model,
697                    msgs_snapshot,
698                    &mut socket,
699                    |tok| {
700                        assistant_content = tok.to_string();
701                        let cur = mem::take(&mut vision_msgs);
702                        vision_msgs = cur.add_message(TextMessageRole::Assistant, tok);
703                    },
704                    || streaming = false,
705                )
706                .await;
707                if !assistant_content.is_empty() {
708                    let _ = append_chat_message(&app, "assistant", &assistant_content).await;
709                }
710                if let Err(e) = stream_res {
711                    error!("stream error: {}", e);
712                }
713            }
714        }
715    }
