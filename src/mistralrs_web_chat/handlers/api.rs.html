33fn validate_image_upload(
34    filename: Option<&str>,
35    content_type: Option<&str>,
36) -> Result<String, &'static str> {
37    // Check MIME type first
38    if let Some(mime) = content_type {
39        if !mime.starts_with("image/") {
40            return Err("File must be an image");
41        }
42    }
43
44    // Validate file extension
45    let ext = if let Some(name) = filename {
46        name.rsplit('.').next().unwrap_or("").to_lowercase()
47    } else {
48        return Err("No filename provided");
49    };
50
51    match ext.as_str() {
52        "jpg" | "jpeg" | "png" | "gif" | "webp" | "bmp" | "svg" => Ok(ext),
53        "" => Err("No file extension"),
54        _ => Err("Unsupported image format"),
55    }
56}

82/// Accepts multipart audio upload, stores under `cache/uploads/`, returns its URL.
83pub async fn upload_audio(
84    State(_app): State<Arc<AppState>>,
85    mut multipart: Multipart,
86) -> impl IntoResponse {
87    if let Ok(Some(field)) = multipart.next_field().await {
88        let orig_filename = field.file_name().map(|s| s.to_string());
89        let content_type_opt = field.content_type().map(|s| s.to_string());
90
91        let ext = match validate_audio_upload(orig_filename.as_deref(), content_type_opt.as_deref())
92        {
93            Ok(ext) => ext,
94            Err(msg) => return (StatusCode::BAD_REQUEST, msg).into_response(),
95        };
96
97        // Read bytes (limit 50MB like images)
98        let data = match field.bytes().await {
99            Ok(b) => b,
100            Err(e) => {
101                error!("multipart bytes error: {}", e);
102                let msg = if e.to_string().contains("exceeded") {
103                    "audio too large (limit 50 MB)"
104                } else {
105                    "failed to read upload"
106                };
107                return (StatusCode::BAD_REQUEST, msg).into_response();
108            }
109        };
110
111        // Ensure upload directory exists
112        let uploads_dir = get_cache_dir().join("uploads");
113        if let Err(e) = tokio::fs::create_dir_all(&uploads_dir).await {
114            error!("create uploads dir error: {}", e);
115            return (
116                StatusCode::INTERNAL_SERVER_ERROR,
117                "failed to create uploads directory",
118            )
119                .into_response();
120        }
121
122        let filename = format!("{}.{}", Uuid::new_v4(), ext);
123        let filepath = uploads_dir.join(&filename);
124        if let Err(e) = tokio::fs::write(&filepath, &data).await {
125            error!("write upload error: {}", e);
126            return (StatusCode::INTERNAL_SERVER_ERROR, "failed to save audio").into_response();
127        }
128
129        let url = filepath.to_string_lossy();
130        (StatusCode::OK, Json(json!({ "url": url }))).into_response()
131    } else {
132        (StatusCode::BAD_REQUEST, "missing audio part").into_response()
133    }
134}

136fn validate_text_upload(
137    filename: Option<&str>,
138    content_type: Option<&str>,
139) -> Result<String, &'static str> {
140    // Check MIME type first (allow text/* and application/json)
141    if let Some(mime) = content_type {
142        if !mime.starts_with("text/")
143            && mime != "application/json"
144            && mime != "application/javascript"
145        {
146            // Also allow common binary MIME types that are actually text
147            if !matches!(
148                mime,
149                "application/octet-stream"
150                    | "application/x-python"
151                    | "application/x-rust"
152                    | "application/x-sh"
153            ) {
154                return Err("File must be a text file");
155            }
156        }
157    }
158
159    // Validate file extension
160    let ext = if let Some(name) = filename {
161        name.rsplit('.').next().unwrap_or("").to_lowercase()
162    } else {
163        return Err("No filename provided");
164    };
165
166    match ext.as_str() {
167        // Text files
168        "txt" | "md" | "markdown" | "log" | "csv" | "tsv" | "json" | "xml" | "yaml" | "yml"
169        | "toml" | "ini" | "cfg" | "conf" => Ok(ext),
170        // Code files
171        "rs" | "py" | "js" | "ts" | "jsx" | "tsx" | "html" | "htm" | "css" | "scss" | "sass"
172        | "less" => Ok(ext),
173        "c" | "cpp" | "cc" | "cxx" | "h" | "hpp" | "hxx" | "java" | "kt" | "swift" | "go"
174        | "rb" | "php" => Ok(ext),
175        // GPU and shader languages
176        "cu" | "cuh" | "cl" | "ptx" | "glsl" | "vert" | "frag" | "geom" | "comp" | "tesc"
177        | "tese" | "hlsl" | "metal" | "wgsl" => Ok(ext),
178        // Shell and other scripts
179        "sh" | "bash" | "zsh" | "fish" | "ps1" | "bat" | "cmd" | "sql" | "dockerfile"
180        | "makefile" => Ok(ext),
181        "r" | "R" | "scala" | "clj" | "cljs" | "hs" | "elm" | "ex" | "exs" | "erl" | "fs"
182        | "fsx" | "ml" | "mli" => Ok(ext),
183        "vue" | "svelte" | "astro" | "lua" | "nim" | "zig" | "d" | "dart" | "jl" | "pl" | "pm"
184        | "tcl" => Ok(ext),
185        // Config and other text-like files
186        "gitignore" | "dockerignore" | "editorconfig" | "env" | "htaccess" => Ok(ext),
187        "" => {
188            // Files without extension - check the filename
189            if let Some(name) = filename {
190                let name_lower = name.to_lowercase();
191                if matches!(
192                    name_lower.as_str(),
193                    "readme"
194                        | "license"
195                        | "changelog"
196                        | "makefile"
197                        | "dockerfile"
198                        | "vagrantfile"
199                        | "gemfile"
200                        | "rakefile"
201                ) {
202                    return Ok("txt".to_string());
203                }
204            }
205            Err("No file extension")
206        }
207        _ => Err("Unsupported text file format"),
208    }
209}

211/// Accepts multipart image upload, stores it under `cache/uploads/`, and returns its URL.
212pub async fn upload_image(
213    State(_app): State<Arc<AppState>>,
214    mut multipart: Multipart,
215) -> impl IntoResponse {
216    // Expect single "image" part
217    if let Ok(Some(field)) = multipart.next_field().await {
218        // Clone filename and content type to avoid borrowing `field`
219        let orig_filename = field.file_name().map(|s| s.to_string());
220        let content_type_opt = field.content_type().map(|s| s.to_string());
221
222        let ext = match validate_image_upload(orig_filename.as_deref(), content_type_opt.as_deref())
223        {
224            Ok(extension) => extension,
225            Err(msg) => {
226                return (StatusCode::BAD_REQUEST, msg).into_response();
227            }
228        };
229
230        let data = match field.bytes().await {
231            Ok(b) => b,
232            Err(e) => {
233                error!("multipart bytes error: {}", e);
234                let msg = if e.to_string().contains("exceeded") {
235                    "image too large (limit 50 MB)"
236                } else {
237                    "failed to read upload"
238                };
239                return (StatusCode::BAD_REQUEST, msg).into_response();
240            }
241        };
242
243        const MAX_SIZE: usize = 50 * 1024 * 1024; // 50MB
244        if data.len() > MAX_SIZE {
245            return (StatusCode::BAD_REQUEST, "image too large (limit 50 MB)").into_response();
246        }
247
248        if image::load_from_memory(&data).is_err() {
249            return (StatusCode::BAD_REQUEST, "invalid image file").into_response();
250        }
251
252        // Determine upload directory under cache
253        let base_cache = get_cache_dir();
254        let upload_dir = base_cache.join("uploads");
255        if let Err(e) = tokio::fs::create_dir_all(&upload_dir).await {
256            error!("mkdir error: {}", e);
257            return (StatusCode::INTERNAL_SERVER_ERROR, "server error").into_response();
258        }
259        // Build unique filename and write
260        let filename = format!("{}.{}", Uuid::new_v4(), ext);
261        let path = upload_dir.join(&filename);
262        if let Err(e) = tokio::fs::write(&path, &data).await {
263            error!("write error: {}", e);
264            return (StatusCode::INTERNAL_SERVER_ERROR, "server error").into_response();
265        }
266        return (
267            StatusCode::OK,
268            Json(json!({ "url": path.to_string_lossy() })),
269        )
270            .into_response();
271    }
272
273    (StatusCode::BAD_REQUEST, "no image field").into_response()
274}

276/// Accepts multipart text file upload and returns the file content
277pub async fn upload_text(
278    State(_app): State<Arc<AppState>>,
279    mut multipart: Multipart,
280) -> impl IntoResponse {
281    // Expect single "file" part
282    if let Ok(Some(field)) = multipart.next_field().await {
283        // Clone filename and content type to avoid borrowing `field`
284        let orig_filename = field.file_name().map(|s| s.to_string());
285        let content_type_opt = field.content_type().map(|s| s.to_string());
286
287        let _ext = match validate_text_upload(orig_filename.as_deref(), content_type_opt.as_deref())
288        {
289            Ok(extension) => extension,
290            Err(msg) => {
291                return (StatusCode::BAD_REQUEST, msg).into_response();
292            }
293        };
294
295        let data = match field.bytes().await {
296            Ok(b) => b,
297            Err(e) => {
298                error!("multipart bytes error: {}", e);
299                let msg = if e.to_string().contains("exceeded") {
300                    "file too large (limit 10 MB)"
301                } else {
302                    "failed to read upload"
303                };
304                return (StatusCode::BAD_REQUEST, msg).into_response();
305            }
306        };
307
308        const MAX_SIZE: usize = 10 * 1024 * 1024; // 10MB for text files
309        if data.len() > MAX_SIZE {
310            return (StatusCode::BAD_REQUEST, "file too large (limit 10 MB)").into_response();
311        }
312
313        // Try to decode as UTF-8
314        let content = match String::from_utf8(data.to_vec()) {
315            Ok(text) => text,
316            Err(_) => {
317                return (StatusCode::BAD_REQUEST, "file is not valid UTF-8 text").into_response();
318            }
319        };
320
321        // Limit content length for safety
322        const MAX_CHARS: usize = 1_000_000; // 1 million characters
323        if content.len() > MAX_CHARS {
324            return (
325                StatusCode::BAD_REQUEST,
326                "file content too large (limit 1M characters)",
327            )
328                .into_response();
329        }
330
331        return (
332            StatusCode::OK,
333            Json(json!({
334                "content": content,
335                "filename": orig_filename.unwrap_or_else(|| "untitled".to_string()),
336                "size": data.len()
337            })),
338        )
339            .into_response();
340    }
341
342    (StatusCode::BAD_REQUEST, "no file field").into_response()
343}

361pub async fn select_model(
362    State(app): State<Arc<AppState>>,
363    Json(req): Json<SelectRequest>,
364) -> impl IntoResponse {
365    if let Some(model_loaded) = app.models.get(&req.name) {
366        {
367            let mut cur = app.current.write().await;
368            *cur = Some(req.name.clone());
369        }
370        // --- sync the active chat file so future loads use the correct model ---
371        if let Some(chat_id) = app.current_chat.read().await.clone() {
372            let path = format!("{}/{}.json", app.chats_dir, chat_id);
373            if let Ok(data) = fs::read(&path).await {
374                if let Ok(mut chat) = serde_json::from_slice::<ChatFile>(&data) {
375                    chat.model = req.name.clone();
376                    chat.kind = match model_loaded {
377                        LoadedModel::Text(_) => "text".into(),
378                        LoadedModel::Vision(_) => "vision".into(),
379                        LoadedModel::Speech(_) => "speech".into(),
380                    };
381                    // ignore write errors; not fatal for select_model
382                    if let Ok(bytes) = serde_json::to_vec_pretty(&chat) {
383                        let _ = tokio::fs::write(&path, bytes).await;
384                    }
385                }
386            }
387        }
388        (StatusCode::OK, "Model selected").into_response()
389    } else {
390        (StatusCode::NOT_FOUND, "Model not found").into_response()
391    }
392}

394pub async fn list_chats(State(app): State<Arc<AppState>>) -> impl IntoResponse {
395    let mut chats = Vec::new();
396    if let Ok(mut dir) = fs::read_dir(&app.chats_dir).await {
397        while let Ok(Some(entry)) = dir.next_entry().await {
398            if let Some(name) = entry.file_name().to_str() {
399                if name.ends_with(".json") {
400                    let id = name.trim_end_matches(".json");
401                    let data = fs::read(format!("{}/{}", app.chats_dir, name)).await.ok();
402                    let (title, created) = data
403                        .and_then(|bytes| serde_json::from_slice::<ChatFile>(&bytes).ok())
404                        .map(|c| (c.title, c.created_at))
405                        .map(|(title, created)| (title.unwrap_or_default(), created))
406                        .unwrap_or_else(|| (String::new(), String::new()));
407                    chats.push(json!({ "id": id, "title": title, "created_at": created }));
408                }
409            }
410        }
411    }
412    Json(json!({ "chats": chats }))
413}

415pub async fn new_chat(
416    State(app): State<Arc<AppState>>,
417    Json(req): Json<NewChatRequest>,
418) -> impl IntoResponse {
419    let mut id_guard = app.next_chat_id.write().await;
420    let id = *id_guard;
421    *id_guard += 1;
422    drop(id_guard);
423
424    let chat_id = format!("chat_{id}");
425    let path = format!("{}/{}.json", app.chats_dir, chat_id);
426
427    let kind = if let Some(m) = app.models.get(&req.model) {
428        match m {
429            LoadedModel::Text(_) => "text",
430            LoadedModel::Vision(_) => "vision",
431            LoadedModel::Speech(_) => "speech",
432        }
433    } else {
434        "text"
435    }
436    .to_string();
437
438    let chat = ChatFile {
439        title: None,
440        model: req.model.clone(),
441        kind,
442        created_at: Utc::now().to_rfc3339(),
443        messages: Vec::new(),
444    };
445    let _ = fs::write(&path, serde_json::to_vec_pretty(&chat).unwrap()).await;
446
447    {
448        let mut cur_chat = app.current_chat.write().await;
449        *cur_chat = Some(chat_id.clone());
450        let mut cur_model = app.current.write().await;
451        *cur_model = Some(req.model.clone());
452    }
453
454    Json(json!({ "id": chat_id }))
455}

457pub async fn delete_chat(
458    State(app): State<Arc<AppState>>,
459    Json(req): Json<DeleteChatRequest>,
460) -> impl IntoResponse {
461    let path = format!("{}/{}.json", app.chats_dir, req.id);
462    if let Err(e) = tokio::fs::remove_file(&path).await {
463        error!("delete chat error: {}", e);
464        return (StatusCode::NOT_FOUND, "chat not found").into_response();
465    }
466    {
467        let mut cur_chat = app.current_chat.write().await;
468        if cur_chat.as_ref() == Some(&req.id) {
469            *cur_chat = None;
470            let mut cur_model = app.current.write().await;
471            *cur_model = None;
472        }
473    }
474    (StatusCode::OK, "Deleted").into_response()
475}

477pub async fn load_chat(
478    State(app): State<Arc<AppState>>,
479    Json(req): Json<LoadChatRequest>,
480) -> impl IntoResponse {
481    let path = format!("{}/{}.json", app.chats_dir, req.id);
482    match fs::read(&path).await {
483        Ok(data) => match serde_json::from_slice::<ChatFile>(&data) {
484            Ok(chat) => {
485                {
486                    let mut cur_chat = app.current_chat.write().await;
487                    *cur_chat = Some(req.id.clone());
488                    if app.models.contains_key(&chat.model) {
489                        let mut cur_model = app.current.write().await;
490                        *cur_model = Some(chat.model.clone());
491                    }
492                }
493                Json(json!({
494                    "id": req.id,
495                    "title": chat.title.clone().unwrap_or_default(),
496                    "model": chat.model,
497                    "kind": chat.kind,
498                    "created_at": chat.created_at.clone(),
499                    "messages": chat.messages
500                }))
501                .into_response()
502            }
503            Err(_) => (StatusCode::INTERNAL_SERVER_ERROR, "corrupt chat").into_response(),
504        },
505        Err(_) => (StatusCode::NOT_FOUND, "chat not found").into_response(),
506    }
507}

570/// Endpoint to generate speech (.wav) for a given prompt using a speech model
571pub async fn generate_speech(
572    State(app): State<Arc<AppState>>,
573    Json(req): Json<GenerateSpeechRequest>,
574) -> impl IntoResponse {
575    // Determine selected model
576    let model_name = {
577        let cur = app.current.read().await;
578        if let Some(name) = &*cur {
579            name.clone()
580        } else {
581            return (StatusCode::BAD_REQUEST, "No model selected").into_response();
582        }
583    };
584    // Ensure model exists and is a speech model
585    if let Some(LoadedModel::Speech(m)) = app.models.get(&model_name) {
586        // Generate speech
587        let (pcm, rate, channels) = match m.generate_speech(req.text).await {
588            Ok(res) => res,
589            Err(e) => {
590                error!("speech generation error: {}", e);
591                return (
592                    StatusCode::INTERNAL_SERVER_ERROR,
593                    "speech generation failed",
594                )
595                    .into_response();
596            }
597        };
598        // Write WAV file
599        let filename = format!("{}.wav", Uuid::new_v4());
600        let filepath = PathBuf::from(&app.speech_dir).join(&filename);
601        if let Err(e) = File::create(&filepath).and_then(|mut f| {
602            speech_utils::write_pcm_as_wav(&mut f, &pcm, rate as u32, channels as u16)
603        }) {
604            error!("failed to write wav file: {}", e);
605            return (
606                StatusCode::INTERNAL_SERVER_ERROR,
607                "failed to write wav file",
608            )
609                .into_response();
610        }
611        // Return URL for client download
612        let url = format!("/speech/{filename}");
613        (StatusCode::OK, Json(json!({ "url": url }))).into_response()
614    } else {
615        (
616            StatusCode::BAD_REQUEST,
617            "Selected model is not a speech model",
618        )
619            .into_response()
620    }
