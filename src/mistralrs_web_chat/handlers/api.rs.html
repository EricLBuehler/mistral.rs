29fn validate_image_upload(
30    filename: Option<&str>,
31    content_type: Option<&str>,
32) -> Result<String, &'static str> {
33    // Check MIME type first
34    if let Some(mime) = content_type {
35        if !mime.starts_with("image/") {
36            return Err("File must be an image");
37        }
38    }
39
40    // Validate file extension
41    let ext = if let Some(name) = filename {
42        name.rsplit('.').next().unwrap_or("").to_lowercase()
43    } else {
44        return Err("No filename provided");
45    };
46
47    match ext.as_str() {
48        "jpg" | "jpeg" | "png" | "gif" | "webp" | "bmp" | "svg" => Ok(ext),
49        "" => Err("No file extension"),
50        _ => Err("Unsupported image format"),
51    }
52}

54fn validate_text_upload(
55    filename: Option<&str>,
56    content_type: Option<&str>,
57) -> Result<String, &'static str> {
58    // Check MIME type first (allow text/* and application/json)
59    if let Some(mime) = content_type {
60        if !mime.starts_with("text/")
61            && mime != "application/json"
62            && mime != "application/javascript"
63        {
64            // Also allow common binary MIME types that are actually text
65            if !matches!(
66                mime,
67                "application/octet-stream"
68                    | "application/x-python"
69                    | "application/x-rust"
70                    | "application/x-sh"
71            ) {
72                return Err("File must be a text file");
73            }
74        }
75    }
76
77    // Validate file extension
78    let ext = if let Some(name) = filename {
79        name.rsplit('.').next().unwrap_or("").to_lowercase()
80    } else {
81        return Err("No filename provided");
82    };
83
84    match ext.as_str() {
85        // Text files
86        "txt" | "md" | "markdown" | "log" | "csv" | "tsv" | "json" | "xml" | "yaml" | "yml"
87        | "toml" | "ini" | "cfg" | "conf" => Ok(ext),
88        // Code files
89        "rs" | "py" | "js" | "ts" | "jsx" | "tsx" | "html" | "htm" | "css" | "scss" | "sass"
90        | "less" => Ok(ext),
91        "c" | "cpp" | "cc" | "cxx" | "h" | "hpp" | "hxx" | "java" | "kt" | "swift" | "go"
92        | "rb" | "php" => Ok(ext),
93        // GPU and shader languages
94        "cu" | "cuh" | "cl" | "ptx" | "glsl" | "vert" | "frag" | "geom" | "comp" | "tesc"
95        | "tese" | "hlsl" | "metal" | "wgsl" => Ok(ext),
96        // Shell and other scripts
97        "sh" | "bash" | "zsh" | "fish" | "ps1" | "bat" | "cmd" | "sql" | "dockerfile"
98        | "makefile" => Ok(ext),
99        "r" | "R" | "scala" | "clj" | "cljs" | "hs" | "elm" | "ex" | "exs" | "erl" | "fs"
100        | "fsx" | "ml" | "mli" => Ok(ext),
101        "vue" | "svelte" | "astro" | "lua" | "nim" | "zig" | "d" | "dart" | "jl" | "pl" | "pm"
102        | "tcl" => Ok(ext),
103        // Config and other text-like files
104        "gitignore" | "dockerignore" | "editorconfig" | "env" | "htaccess" => Ok(ext),
105        "" => {
106            // Files without extension - check the filename
107            if let Some(name) = filename {
108                let name_lower = name.to_lowercase();
109                if matches!(
110                    name_lower.as_str(),
111                    "readme"
112                        | "license"
113                        | "changelog"
114                        | "makefile"
115                        | "dockerfile"
116                        | "vagrantfile"
117                        | "gemfile"
118                        | "rakefile"
119                ) {
120                    return Ok("txt".to_string());
121                }
122            }
123            Err("No file extension")
124        }
125        _ => Err("Unsupported text file format"),
126    }
127}

129/// Accepts multipart image upload, stores it under `cache/uploads/`, and returns its URL.
130pub async fn upload_image(
131    State(_app): State<Arc<AppState>>,
132    mut multipart: Multipart,
133) -> impl IntoResponse {
134    // Expect single "image" part
135    if let Ok(Some(field)) = multipart.next_field().await {
136        // Clone filename and content type to avoid borrowing `field`
137        let orig_filename = field.file_name().map(|s| s.to_string());
138        let content_type_opt = field.content_type().map(|s| s.to_string());
139
140        let ext = match validate_image_upload(orig_filename.as_deref(), content_type_opt.as_deref())
141        {
142            Ok(extension) => extension,
143            Err(msg) => {
144                return (StatusCode::BAD_REQUEST, msg).into_response();
145            }
146        };
147
148        let data = match field.bytes().await {
149            Ok(b) => b,
150            Err(e) => {
151                error!("multipart bytes error: {}", e);
152                let msg = if e.to_string().contains("exceeded") {
153                    "image too large (limit 50 MB)"
154                } else {
155                    "failed to read upload"
156                };
157                return (StatusCode::BAD_REQUEST, msg).into_response();
158            }
159        };
160
161        const MAX_SIZE: usize = 50 * 1024 * 1024; // 50MB
162        if data.len() > MAX_SIZE {
163            return (StatusCode::BAD_REQUEST, "image too large (limit 50 MB)").into_response();
164        }
165
166        if image::load_from_memory(&data).is_err() {
167            return (StatusCode::BAD_REQUEST, "invalid image file").into_response();
168        }
169
170        // Determine upload directory under cache
171        let base_cache = get_cache_dir();
172        let upload_dir = base_cache.join("uploads");
173        if let Err(e) = tokio::fs::create_dir_all(&upload_dir).await {
174            error!("mkdir error: {}", e);
175            return (StatusCode::INTERNAL_SERVER_ERROR, "server error").into_response();
176        }
177        // Build unique filename and write
178        let filename = format!("{}.{}", Uuid::new_v4(), ext);
179        let path = upload_dir.join(&filename);
180        if let Err(e) = tokio::fs::write(&path, &data).await {
181            error!("write error: {}", e);
182            return (StatusCode::INTERNAL_SERVER_ERROR, "server error").into_response();
183        }
184        return (
185            StatusCode::OK,
186            Json(json!({ "url": path.to_string_lossy() })),
187        )
188            .into_response();
189    }
190
191    (StatusCode::BAD_REQUEST, "no image field").into_response()
192}

194/// Accepts multipart text file upload and returns the file content
195pub async fn upload_text(
196    State(_app): State<Arc<AppState>>,
197    mut multipart: Multipart,
198) -> impl IntoResponse {
199    // Expect single "file" part
200    if let Ok(Some(field)) = multipart.next_field().await {
201        // Clone filename and content type to avoid borrowing `field`
202        let orig_filename = field.file_name().map(|s| s.to_string());
203        let content_type_opt = field.content_type().map(|s| s.to_string());
204
205        let _ext = match validate_text_upload(orig_filename.as_deref(), content_type_opt.as_deref())
206        {
207            Ok(extension) => extension,
208            Err(msg) => {
209                return (StatusCode::BAD_REQUEST, msg).into_response();
210            }
211        };
212
213        let data = match field.bytes().await {
214            Ok(b) => b,
215            Err(e) => {
216                error!("multipart bytes error: {}", e);
217                let msg = if e.to_string().contains("exceeded") {
218                    "file too large (limit 10 MB)"
219                } else {
220                    "failed to read upload"
221                };
222                return (StatusCode::BAD_REQUEST, msg).into_response();
223            }
224        };
225
226        const MAX_SIZE: usize = 10 * 1024 * 1024; // 10MB for text files
227        if data.len() > MAX_SIZE {
228            return (StatusCode::BAD_REQUEST, "file too large (limit 10 MB)").into_response();
229        }
230
231        // Try to decode as UTF-8
232        let content = match String::from_utf8(data.to_vec()) {
233            Ok(text) => text,
234            Err(_) => {
235                return (StatusCode::BAD_REQUEST, "file is not valid UTF-8 text").into_response();
236            }
237        };
238
239        // Limit content length for safety
240        const MAX_CHARS: usize = 1_000_000; // 1 million characters
241        if content.len() > MAX_CHARS {
242            return (
243                StatusCode::BAD_REQUEST,
244                "file content too large (limit 1M characters)",
245            )
246                .into_response();
247        }
248
249        return (
250            StatusCode::OK,
251            Json(json!({
252                "content": content,
253                "filename": orig_filename.unwrap_or_else(|| "untitled".to_string()),
254                "size": data.len()
255            })),
256        )
257            .into_response();
258    }
259
260    (StatusCode::BAD_REQUEST, "no file field").into_response()
261}

263pub async fn list_models(State(app): State<Arc<AppState>>) -> impl IntoResponse {
264    let items: Vec<_> = app
265        .models
266        .iter()
267        .map(|(n, m)| {
268            let kind = match m {
269                LoadedModel::Text(_) => "text",
270                LoadedModel::Vision(_) => "vision",
271            };
272            json!({ "name": n, "kind": kind })
273        })
274        .collect();
275    Json(json!({ "models": items }))
276}

278pub async fn select_model(
279    State(app): State<Arc<AppState>>,
280    Json(req): Json<SelectRequest>,
281) -> impl IntoResponse {
282    if let Some(model_loaded) = app.models.get(&req.name) {
283        {
284            let mut cur = app.current.write().await;
285            *cur = Some(req.name.clone());
286        }
287        // --- sync the active chat file so future loads use the correct model ---
288        if let Some(chat_id) = app.current_chat.read().await.clone() {
289            let path = format!("{}/{}.json", app.chats_dir, chat_id);
290            if let Ok(data) = fs::read(&path).await {
291                if let Ok(mut chat) = serde_json::from_slice::<ChatFile>(&data) {
292                    chat.model = req.name.clone();
293                    chat.kind = match model_loaded {
294                        LoadedModel::Text(_) => "text".into(),
295                        LoadedModel::Vision(_) => "vision".into(),
296                    };
297                    // ignore write errors; not fatal for select_model
298                    if let Ok(bytes) = serde_json::to_vec_pretty(&chat) {
299                        let _ = tokio::fs::write(&path, bytes).await;
300                    }
301                }
302            }
303        }
304        (StatusCode::OK, "Model selected").into_response()
305    } else {
306        (StatusCode::NOT_FOUND, "Model not found").into_response()
307    }
308}

310pub async fn list_chats(State(app): State<Arc<AppState>>) -> impl IntoResponse {
311    let mut chats = Vec::new();
312    if let Ok(mut dir) = fs::read_dir(&app.chats_dir).await {
313        while let Ok(Some(entry)) = dir.next_entry().await {
314            if let Some(name) = entry.file_name().to_str() {
315                if name.ends_with(".json") {
316                    let id = name.trim_end_matches(".json");
317                    let data = fs::read(format!("{}/{}", app.chats_dir, name)).await.ok();
318                    let (title, created) = data
319                        .and_then(|bytes| serde_json::from_slice::<ChatFile>(&bytes).ok())
320                        .map(|c| (c.title, c.created_at))
321                        .map(|(title, created)| (title.unwrap_or_default(), created))
322                        .unwrap_or_else(|| (String::new(), String::new()));
323                    chats.push(json!({ "id": id, "title": title, "created_at": created }));
324                }
325            }
326        }
327    }
328    Json(json!({ "chats": chats }))
329}

331pub async fn new_chat(
332    State(app): State<Arc<AppState>>,
333    Json(req): Json<NewChatRequest>,
334) -> impl IntoResponse {
335    let mut id_guard = app.next_chat_id.write().await;
336    let id = *id_guard;
337    *id_guard += 1;
338    drop(id_guard);
339
340    let chat_id = format!("chat_{}", id);
341    let path = format!("{}/{}.json", app.chats_dir, chat_id);
342
343    let kind = if let Some(m) = app.models.get(&req.model) {
344        match m {
345            LoadedModel::Text(_) => "text",
346            LoadedModel::Vision(_) => "vision",
347        }
348    } else {
349        "text"
350    }
351    .to_string();
352
353    let chat = ChatFile {
354        title: None,
355        model: req.model.clone(),
356        kind,
357        created_at: Utc::now().to_rfc3339(),
358        messages: Vec::new(),
359    };
360    let _ = fs::write(&path, serde_json::to_vec_pretty(&chat).unwrap()).await;
361
362    {
363        let mut cur_chat = app.current_chat.write().await;
364        *cur_chat = Some(chat_id.clone());
365        let mut cur_model = app.current.write().await;
366        *cur_model = Some(req.model.clone());
367    }
368
369    Json(json!({ "id": chat_id }))
370}

372pub async fn delete_chat(
373    State(app): State<Arc<AppState>>,
374    Json(req): Json<DeleteChatRequest>,
375) -> impl IntoResponse {
376    let path = format!("{}/{}.json", app.chats_dir, req.id);
377    if let Err(e) = tokio::fs::remove_file(&path).await {
378        error!("delete chat error: {}", e);
379        return (StatusCode::NOT_FOUND, "chat not found").into_response();
380    }
381    {
382        let mut cur_chat = app.current_chat.write().await;
383        if cur_chat.as_ref() == Some(&req.id) {
384            *cur_chat = None;
385            let mut cur_model = app.current.write().await;
386            *cur_model = None;
387        }
388    }
389    (StatusCode::OK, "Deleted").into_response()
390}

392pub async fn load_chat(
393    State(app): State<Arc<AppState>>,
394    Json(req): Json<LoadChatRequest>,
395) -> impl IntoResponse {
396    let path = format!("{}/{}.json", app.chats_dir, req.id);
397    match fs::read(&path).await {
398        Ok(data) => match serde_json::from_slice::<ChatFile>(&data) {
399            Ok(chat) => {
400                {
401                    let mut cur_chat = app.current_chat.write().await;
402                    *cur_chat = Some(req.id.clone());
403                    if app.models.contains_key(&chat.model) {
404                        let mut cur_model = app.current.write().await;
405                        *cur_model = Some(chat.model.clone());
406                    }
407                }
408                Json(json!({
409                    "id": req.id,
410                    "title": chat.title.clone().unwrap_or_default(),
411                    "model": chat.model,
412                    "kind": chat.kind,
413                    "created_at": chat.created_at.clone(),
414                    "messages": chat.messages
415                }))
416                .into_response()
417            }
418            Err(_) => (StatusCode::INTERNAL_SERVER_ERROR, "corrupt chat").into_response(),
419        },
420        Err(_) => (StatusCode::NOT_FOUND, "chat not found").into_response(),
421    }
422}

424pub async fn rename_chat(
425    State(app): State<Arc<AppState>>,
426    Json(req): Json<RenameChatRequest>,
427) -> impl IntoResponse {
428    let path = format!("{}/{}.json", app.chats_dir, req.id);
429    if let Ok(data) = fs::read(&path).await {
430        if let Ok(mut chat) = serde_json::from_slice::<ChatFile>(&data) {
431            chat.title = Some(req.title.clone());
432            if let Ok(bytes) = serde_json::to_vec_pretty(&chat) {
433                let _ = tokio::fs::write(&path, bytes).await;
434                return (StatusCode::OK, "Renamed").into_response();
435            }
436        }
437    }
438    (StatusCode::INTERNAL_SERVER_ERROR, "rename failed").into_response()
439}
