20/// Accepts multipart image upload, stores it under `cache/uploads/`, and returns its URL.
21pub async fn upload_image(
22    State(_app): State<Arc<AppState>>,
23    mut multipart: Multipart,
24) -> impl IntoResponse {
25    // Expect single "image" part
26    if let Ok(Some(field)) = multipart.next_field().await {
27        let ext = field
28            .file_name()
29            .and_then(|n| n.rsplit('.').next())
30            .unwrap_or("bin")
31            .to_string();
32
33        let data = match field.bytes().await {
34            Ok(b) => b,
35            Err(e) => {
36                error!("multipart bytes error: {}", e);
37                let msg = if e.to_string().contains("exceeded") {
38                    "image too large (limit 50 MB)"
39                } else {
40                    "failed to read upload"
41                };
42                return (StatusCode::BAD_REQUEST, msg).into_response();
43            }
44        };
45
46        // Ensure dir exists
47        let upload_dir = format!("{}/cache/uploads", env!("CARGO_MANIFEST_DIR"));
48        if let Err(e) = tokio::fs::create_dir_all(&upload_dir).await {
49            error!("mkdir error: {}", e);
50            return (StatusCode::INTERNAL_SERVER_ERROR, "server error").into_response();
51        }
52
53        // Build unique filename
54        let filename = format!("{}.{}", Uuid::new_v4(), ext);
55        let path = format!("{}/{}", upload_dir, filename);
56
57        if let Err(e) = tokio::fs::write(&path, &data).await {
58            error!("write error: {}", e);
59            return (StatusCode::INTERNAL_SERVER_ERROR, "server error").into_response();
60        }
61
62        return (StatusCode::OK, Json(json!({ "url": path }))).into_response();
63    }
64
65    (StatusCode::BAD_REQUEST, "no image field").into_response()
66}

68pub async fn list_models(State(app): State<Arc<AppState>>) -> impl IntoResponse {
69    let items: Vec<_> = app
70        .models
71        .iter()
72        .map(|(n, m)| {
73            let kind = match m {
74                LoadedModel::Text(_) => "text",
75                LoadedModel::Vision(_) => "vision",
76            };
77            json!({ "name": n, "kind": kind })
78        })
79        .collect();
80    Json(json!({ "models": items }))
81}

83pub async fn select_model(
84    State(app): State<Arc<AppState>>,
85    Json(req): Json<SelectRequest>,
86) -> impl IntoResponse {
87    if let Some(model_loaded) = app.models.get(&req.name) {
88        {
89            let mut cur = app.current.write().await;
90            *cur = Some(req.name.clone());
91        }
92        // --- sync the active chat file so future loads use the correct model ---
93        if let Some(chat_id) = app.current_chat.read().await.clone() {
94            let path = format!("{}/{}.json", app.chats_dir, chat_id);
95            if let Ok(data) = fs::read(&path).await {
96                if let Ok(mut chat) = serde_json::from_slice::<ChatFile>(&data) {
97                    chat.model = req.name.clone();
98                    chat.kind = match model_loaded {
99                        LoadedModel::Text(_) => "text".into(),
100                        LoadedModel::Vision(_) => "vision".into(),
101                    };
102                    // ignore write errors; not fatal for select_model
103                    if let Ok(bytes) = serde_json::to_vec_pretty(&chat) {
104                        let _ = tokio::fs::write(&path, bytes).await;
105                    }
106                }
107            }
108        }
109        (StatusCode::OK, "Model selected").into_response()
110    } else {
111        (StatusCode::NOT_FOUND, "Model not found").into_response()
112    }
113}

115pub async fn list_chats(State(app): State<Arc<AppState>>) -> impl IntoResponse {
116    let mut chats = Vec::new();
117    if let Ok(mut dir) = fs::read_dir(&app.chats_dir).await {
118        while let Ok(Some(entry)) = dir.next_entry().await {
119            if let Some(name) = entry.file_name().to_str() {
120                if name.ends_with(".json") {
121                    let id = name.trim_end_matches(".json");
122                    let data = fs::read(format!("{}/{}", app.chats_dir, name)).await.ok();
123                    let (title, created) = data
124                        .and_then(|bytes| serde_json::from_slice::<ChatFile>(&bytes).ok())
125                        .map(|c| (c.title, c.created_at))
126                        .map(|(title, created)| (title.unwrap_or_default(), created))
127                        .unwrap_or_else(|| (String::new(), String::new()));
128                    chats.push(json!({ "id": id, "title": title, "created_at": created }));
129                }
130            }
131        }
132    }
133    Json(json!({ "chats": chats }))
134}

136pub async fn new_chat(
137    State(app): State<Arc<AppState>>,
138    Json(req): Json<NewChatRequest>,
139) -> impl IntoResponse {
140    let mut id_guard = app.next_chat_id.write().await;
141    let id = *id_guard;
142    *id_guard += 1;
143    drop(id_guard);
144
145    let chat_id = format!("chat_{}", id);
146    let path = format!("{}/{}.json", app.chats_dir, chat_id);
147
148    let kind = if let Some(m) = app.models.get(&req.model) {
149        match m {
150            LoadedModel::Text(_) => "text",
151            LoadedModel::Vision(_) => "vision",
152        }
153    } else {
154        "text"
155    }
156    .to_string();
157
158    let chat = ChatFile {
159        title: None,
160        model: req.model.clone(),
161        kind,
162        created_at: Utc::now().to_rfc3339(),
163        messages: Vec::new(),
164    };
165    let _ = fs::write(&path, serde_json::to_vec_pretty(&chat).unwrap()).await;
166
167    {
168        let mut cur_chat = app.current_chat.write().await;
169        *cur_chat = Some(chat_id.clone());
170        let mut cur_model = app.current.write().await;
171        *cur_model = Some(req.model.clone());
172    }
173
174    Json(json!({ "id": chat_id }))
175}

177pub async fn delete_chat(
178    State(app): State<Arc<AppState>>,
179    Json(req): Json<DeleteChatRequest>,
180) -> impl IntoResponse {
181    let path = format!("{}/{}.json", app.chats_dir, req.id);
182    if let Err(e) = tokio::fs::remove_file(&path).await {
183        error!("delete chat error: {}", e);
184        return (StatusCode::NOT_FOUND, "chat not found").into_response();
185    }
186    {
187        let mut cur_chat = app.current_chat.write().await;
188        if cur_chat.as_ref() == Some(&req.id) {
189            *cur_chat = None;
190            let mut cur_model = app.current.write().await;
191            *cur_model = None;
192        }
193    }
194    (StatusCode::OK, "Deleted").into_response()
195}

197pub async fn load_chat(
198    State(app): State<Arc<AppState>>,
199    Json(req): Json<LoadChatRequest>,
200) -> impl IntoResponse {
201    let path = format!("{}/{}.json", app.chats_dir, req.id);
202    match fs::read(&path).await {
203        Ok(data) => match serde_json::from_slice::<ChatFile>(&data) {
204            Ok(chat) => {
205                {
206                    let mut cur_chat = app.current_chat.write().await;
207                    *cur_chat = Some(req.id.clone());
208                    if app.models.contains_key(&chat.model) {
209                        let mut cur_model = app.current.write().await;
210                        *cur_model = Some(chat.model.clone());
211                    }
212                }
213                Json(json!({
214                    "id": req.id,
215                    "title": chat.title.clone().unwrap_or_default(),
216                    "model": chat.model,
217                    "kind": chat.kind,
218                    "created_at": chat.created_at.clone(),
219                    "messages": chat.messages
220                }))
221                .into_response()
222            }
223            Err(_) => (StatusCode::INTERNAL_SERVER_ERROR, "corrupt chat").into_response(),
224        },
225        Err(_) => (StatusCode::NOT_FOUND, "chat not found").into_response(),
226    }
227}

229pub async fn rename_chat(
230    State(app): State<Arc<AppState>>,
231    Json(req): Json<RenameChatRequest>,
232) -> impl IntoResponse {
233    let path = format!("{}/{}.json", app.chats_dir, req.id);
234    if let Ok(data) = fs::read(&path).await {
235        if let Ok(mut chat) = serde_json::from_slice::<ChatFile>(&data) {
236            chat.title = Some(req.title.clone());
237            if let Ok(bytes) = serde_json::to_vec_pretty(&chat) {
238                let _ = tokio::fs::write(&path, bytes).await;
239                return (StatusCode::OK, "Renamed").into_response();
240            }
241        }
242    }
243    (StatusCode::INTERNAL_SERVER_ERROR, "rename failed").into_response()
244}
