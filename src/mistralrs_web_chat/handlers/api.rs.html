33fn validate_image_upload(
34    filename: Option<&str>,
35    content_type: Option<&str>,
36) -> Result<String, &'static str> {
37    // Check MIME type first
38    if let Some(mime) = content_type {
39        if !mime.starts_with("image/") {
40            return Err("File must be an image");
41        }
42    }
43
44    // Validate file extension
45    let ext = if let Some(name) = filename {
46        name.rsplit('.').next().unwrap_or("").to_lowercase()
47    } else {
48        return Err("No filename provided");
49    };
50
51    match ext.as_str() {
52        "jpg" | "jpeg" | "png" | "gif" | "webp" | "bmp" | "svg" => Ok(ext),
53        "" => Err("No file extension"),
54        _ => Err("Unsupported image format"),
55    }
56}

58fn validate_text_upload(
59    filename: Option<&str>,
60    content_type: Option<&str>,
61) -> Result<String, &'static str> {
62    // Check MIME type first (allow text/* and application/json)
63    if let Some(mime) = content_type {
64        if !mime.starts_with("text/")
65            && mime != "application/json"
66            && mime != "application/javascript"
67        {
68            // Also allow common binary MIME types that are actually text
69            if !matches!(
70                mime,
71                "application/octet-stream"
72                    | "application/x-python"
73                    | "application/x-rust"
74                    | "application/x-sh"
75            ) {
76                return Err("File must be a text file");
77            }
78        }
79    }
80
81    // Validate file extension
82    let ext = if let Some(name) = filename {
83        name.rsplit('.').next().unwrap_or("").to_lowercase()
84    } else {
85        return Err("No filename provided");
86    };
87
88    match ext.as_str() {
89        // Text files
90        "txt" | "md" | "markdown" | "log" | "csv" | "tsv" | "json" | "xml" | "yaml" | "yml"
91        | "toml" | "ini" | "cfg" | "conf" => Ok(ext),
92        // Code files
93        "rs" | "py" | "js" | "ts" | "jsx" | "tsx" | "html" | "htm" | "css" | "scss" | "sass"
94        | "less" => Ok(ext),
95        "c" | "cpp" | "cc" | "cxx" | "h" | "hpp" | "hxx" | "java" | "kt" | "swift" | "go"
96        | "rb" | "php" => Ok(ext),
97        // GPU and shader languages
98        "cu" | "cuh" | "cl" | "ptx" | "glsl" | "vert" | "frag" | "geom" | "comp" | "tesc"
99        | "tese" | "hlsl" | "metal" | "wgsl" => Ok(ext),
100        // Shell and other scripts
101        "sh" | "bash" | "zsh" | "fish" | "ps1" | "bat" | "cmd" | "sql" | "dockerfile"
102        | "makefile" => Ok(ext),
103        "r" | "R" | "scala" | "clj" | "cljs" | "hs" | "elm" | "ex" | "exs" | "erl" | "fs"
104        | "fsx" | "ml" | "mli" => Ok(ext),
105        "vue" | "svelte" | "astro" | "lua" | "nim" | "zig" | "d" | "dart" | "jl" | "pl" | "pm"
106        | "tcl" => Ok(ext),
107        // Config and other text-like files
108        "gitignore" | "dockerignore" | "editorconfig" | "env" | "htaccess" => Ok(ext),
109        "" => {
110            // Files without extension - check the filename
111            if let Some(name) = filename {
112                let name_lower = name.to_lowercase();
113                if matches!(
114                    name_lower.as_str(),
115                    "readme"
116                        | "license"
117                        | "changelog"
118                        | "makefile"
119                        | "dockerfile"
120                        | "vagrantfile"
121                        | "gemfile"
122                        | "rakefile"
123                ) {
124                    return Ok("txt".to_string());
125                }
126            }
127            Err("No file extension")
128        }
129        _ => Err("Unsupported text file format"),
130    }
131}

133/// Accepts multipart image upload, stores it under `cache/uploads/`, and returns its URL.
134pub async fn upload_image(
135    State(_app): State<Arc<AppState>>,
136    mut multipart: Multipart,
137) -> impl IntoResponse {
138    // Expect single "image" part
139    if let Ok(Some(field)) = multipart.next_field().await {
140        // Clone filename and content type to avoid borrowing `field`
141        let orig_filename = field.file_name().map(|s| s.to_string());
142        let content_type_opt = field.content_type().map(|s| s.to_string());
143
144        let ext = match validate_image_upload(orig_filename.as_deref(), content_type_opt.as_deref())
145        {
146            Ok(extension) => extension,
147            Err(msg) => {
148                return (StatusCode::BAD_REQUEST, msg).into_response();
149            }
150        };
151
152        let data = match field.bytes().await {
153            Ok(b) => b,
154            Err(e) => {
155                error!("multipart bytes error: {}", e);
156                let msg = if e.to_string().contains("exceeded") {
157                    "image too large (limit 50 MB)"
158                } else {
159                    "failed to read upload"
160                };
161                return (StatusCode::BAD_REQUEST, msg).into_response();
162            }
163        };
164
165        const MAX_SIZE: usize = 50 * 1024 * 1024; // 50MB
166        if data.len() > MAX_SIZE {
167            return (StatusCode::BAD_REQUEST, "image too large (limit 50 MB)").into_response();
168        }
169
170        if image::load_from_memory(&data).is_err() {
171            return (StatusCode::BAD_REQUEST, "invalid image file").into_response();
172        }
173
174        // Determine upload directory under cache
175        let base_cache = get_cache_dir();
176        let upload_dir = base_cache.join("uploads");
177        if let Err(e) = tokio::fs::create_dir_all(&upload_dir).await {
178            error!("mkdir error: {}", e);
179            return (StatusCode::INTERNAL_SERVER_ERROR, "server error").into_response();
180        }
181        // Build unique filename and write
182        let filename = format!("{}.{}", Uuid::new_v4(), ext);
183        let path = upload_dir.join(&filename);
184        if let Err(e) = tokio::fs::write(&path, &data).await {
185            error!("write error: {}", e);
186            return (StatusCode::INTERNAL_SERVER_ERROR, "server error").into_response();
187        }
188        return (
189            StatusCode::OK,
190            Json(json!({ "url": path.to_string_lossy() })),
191        )
192            .into_response();
193    }
194
195    (StatusCode::BAD_REQUEST, "no image field").into_response()
196}

198/// Accepts multipart text file upload and returns the file content
199pub async fn upload_text(
200    State(_app): State<Arc<AppState>>,
201    mut multipart: Multipart,
202) -> impl IntoResponse {
203    // Expect single "file" part
204    if let Ok(Some(field)) = multipart.next_field().await {
205        // Clone filename and content type to avoid borrowing `field`
206        let orig_filename = field.file_name().map(|s| s.to_string());
207        let content_type_opt = field.content_type().map(|s| s.to_string());
208
209        let _ext = match validate_text_upload(orig_filename.as_deref(), content_type_opt.as_deref())
210        {
211            Ok(extension) => extension,
212            Err(msg) => {
213                return (StatusCode::BAD_REQUEST, msg).into_response();
214            }
215        };
216
217        let data = match field.bytes().await {
218            Ok(b) => b,
219            Err(e) => {
220                error!("multipart bytes error: {}", e);
221                let msg = if e.to_string().contains("exceeded") {
222                    "file too large (limit 10 MB)"
223                } else {
224                    "failed to read upload"
225                };
226                return (StatusCode::BAD_REQUEST, msg).into_response();
227            }
228        };
229
230        const MAX_SIZE: usize = 10 * 1024 * 1024; // 10MB for text files
231        if data.len() > MAX_SIZE {
232            return (StatusCode::BAD_REQUEST, "file too large (limit 10 MB)").into_response();
233        }
234
235        // Try to decode as UTF-8
236        let content = match String::from_utf8(data.to_vec()) {
237            Ok(text) => text,
238            Err(_) => {
239                return (StatusCode::BAD_REQUEST, "file is not valid UTF-8 text").into_response();
240            }
241        };
242
243        // Limit content length for safety
244        const MAX_CHARS: usize = 1_000_000; // 1 million characters
245        if content.len() > MAX_CHARS {
246            return (
247                StatusCode::BAD_REQUEST,
248                "file content too large (limit 1M characters)",
249            )
250                .into_response();
251        }
252
253        return (
254            StatusCode::OK,
255            Json(json!({
256                "content": content,
257                "filename": orig_filename.unwrap_or_else(|| "untitled".to_string()),
258                "size": data.len()
259            })),
260        )
261            .into_response();
262    }
263
264    (StatusCode::BAD_REQUEST, "no file field").into_response()
265}

267pub async fn list_models(State(app): State<Arc<AppState>>) -> impl IntoResponse {
268    let items: Vec<_> = app

283pub async fn select_model(
284    State(app): State<Arc<AppState>>,
285    Json(req): Json<SelectRequest>,
286) -> impl IntoResponse {
287    if let Some(model_loaded) = app.models.get(&req.name) {
288        {
289            let mut cur = app.current.write().await;
290            *cur = Some(req.name.clone());
291        }
292        // --- sync the active chat file so future loads use the correct model ---
293        if let Some(chat_id) = app.current_chat.read().await.clone() {
294            let path = format!("{}/{}.json", app.chats_dir, chat_id);
295            if let Ok(data) = fs::read(&path).await {
296                if let Ok(mut chat) = serde_json::from_slice::<ChatFile>(&data) {
297                    chat.model = req.name.clone();
298                    chat.kind = match model_loaded {
299                        LoadedModel::Text(_) => "text".into(),
300                        LoadedModel::Vision(_) => "vision".into(),
301                        LoadedModel::Speech(_) => "speech".into(),
302                    };
303                    // ignore write errors; not fatal for select_model
304                    if let Ok(bytes) = serde_json::to_vec_pretty(&chat) {
305                        let _ = tokio::fs::write(&path, bytes).await;
306                    }
307                }
308            }
309        }
310        (StatusCode::OK, "Model selected").into_response()
311    } else {
312        (StatusCode::NOT_FOUND, "Model not found").into_response()
313    }
314}

316pub async fn list_chats(State(app): State<Arc<AppState>>) -> impl IntoResponse {
317    let mut chats = Vec::new();
318    if let Ok(mut dir) = fs::read_dir(&app.chats_dir).await {
319        while let Ok(Some(entry)) = dir.next_entry().await {
320            if let Some(name) = entry.file_name().to_str() {
321                if name.ends_with(".json") {
322                    let id = name.trim_end_matches(".json");
323                    let data = fs::read(format!("{}/{}", app.chats_dir, name)).await.ok();
324                    let (title, created) = data
325                        .and_then(|bytes| serde_json::from_slice::<ChatFile>(&bytes).ok())
326                        .map(|c| (c.title, c.created_at))
327                        .map(|(title, created)| (title.unwrap_or_default(), created))
328                        .unwrap_or_else(|| (String::new(), String::new()));
329                    chats.push(json!({ "id": id, "title": title, "created_at": created }));
330                }
331            }
332        }
333    }
334    Json(json!({ "chats": chats }))
335}

337pub async fn new_chat(
338    State(app): State<Arc<AppState>>,
339    Json(req): Json<NewChatRequest>,
340) -> impl IntoResponse {
341    let mut id_guard = app.next_chat_id.write().await;
342    let id = *id_guard;
343    *id_guard += 1;
344    drop(id_guard);
345
346    let chat_id = format!("chat_{}", id);
347    let path = format!("{}/{}.json", app.chats_dir, chat_id);
348
349    let kind = if let Some(m) = app.models.get(&req.model) {
350        match m {
351            LoadedModel::Text(_) => "text",
352            LoadedModel::Vision(_) => "vision",
353            LoadedModel::Speech(_) => "speech",
354        }
355    } else {
356        "text"
357    }
358    .to_string();
359
360    let chat = ChatFile {
361        title: None,
362        model: req.model.clone(),
363        kind,
364        created_at: Utc::now().to_rfc3339(),
365        messages: Vec::new(),
366    };
367    let _ = fs::write(&path, serde_json::to_vec_pretty(&chat).unwrap()).await;
368
369    {
370        let mut cur_chat = app.current_chat.write().await;
371        *cur_chat = Some(chat_id.clone());
372        let mut cur_model = app.current.write().await;
373        *cur_model = Some(req.model.clone());
374    }
375
376    Json(json!({ "id": chat_id }))
377}

379pub async fn delete_chat(
380    State(app): State<Arc<AppState>>,
381    Json(req): Json<DeleteChatRequest>,
382) -> impl IntoResponse {
383    let path = format!("{}/{}.json", app.chats_dir, req.id);
384    if let Err(e) = tokio::fs::remove_file(&path).await {
385        error!("delete chat error: {}", e);
386        return (StatusCode::NOT_FOUND, "chat not found").into_response();
387    }
388    {
389        let mut cur_chat = app.current_chat.write().await;
390        if cur_chat.as_ref() == Some(&req.id) {
391            *cur_chat = None;
392            let mut cur_model = app.current.write().await;
393            *cur_model = None;
394        }
395    }
396    (StatusCode::OK, "Deleted").into_response()
397}

399pub async fn load_chat(
400    State(app): State<Arc<AppState>>,
401    Json(req): Json<LoadChatRequest>,
402) -> impl IntoResponse {
403    let path = format!("{}/{}.json", app.chats_dir, req.id);
404    match fs::read(&path).await {
405        Ok(data) => match serde_json::from_slice::<ChatFile>(&data) {
406            Ok(chat) => {
407                {
408                    let mut cur_chat = app.current_chat.write().await;
409                    *cur_chat = Some(req.id.clone());
410                    if app.models.contains_key(&chat.model) {
411                        let mut cur_model = app.current.write().await;
412                        *cur_model = Some(chat.model.clone());
413                    }
414                }
415                Json(json!({
416                    "id": req.id,
417                    "title": chat.title.clone().unwrap_or_default(),
418                    "model": chat.model,
419                    "kind": chat.kind,
420                    "created_at": chat.created_at.clone(),
421                    "messages": chat.messages
422                }))
423                .into_response()
424            }
425            Err(_) => (StatusCode::INTERNAL_SERVER_ERROR, "corrupt chat").into_response(),
426        },
427        Err(_) => (StatusCode::NOT_FOUND, "chat not found").into_response(),
428    }
429}

440                let _ = tokio::fs::write(&path, bytes).await;
441                return (StatusCode::OK, "Renamed").into_response();
442            }
443        }
444    }
445    (StatusCode::INTERNAL_SERVER_ERROR, "rename failed").into_response()
446}

477/// Endpoint to generate speech (.wav) for a given prompt using a speech model
478pub async fn generate_speech(
479    State(app): State<Arc<AppState>>,
480    Json(req): Json<GenerateSpeechRequest>,
481) -> impl IntoResponse {
482    // Determine selected model
483    let model_name = {
484        let cur = app.current.read().await;
485        if let Some(name) = &*cur {
486            name.clone()
487        } else {
488            return (StatusCode::BAD_REQUEST, "No model selected").into_response();
489        }
490    };
491    // Ensure model exists and is a speech model
492    if let Some(LoadedModel::Speech(m)) = app.models.get(&model_name) {
493        // Generate speech
494        let (pcm, rate, channels) = match m.generate_speech(req.text).await {
495            Ok(res) => res,
496            Err(e) => {
497                error!("speech generation error: {}", e);
498                return (
499                    StatusCode::INTERNAL_SERVER_ERROR,
500                    "speech generation failed",
501                )
502                    .into_response();
503            }
504        };
505        // Write WAV file
506        let filename = format!("{}.wav", Uuid::new_v4());
507        let filepath = PathBuf::from(&app.speech_dir).join(&filename);
508        if let Err(e) = File::create(&filepath).and_then(|mut f| {
509            speech_utils::write_pcm_as_wav(&mut f, &pcm, rate as u32, channels as u16)
510        }) {
511            error!("failed to write wav file: {}", e);
512            return (
513                StatusCode::INTERNAL_SERVER_ERROR,
514                "failed to write wav file",
515            )
516                .into_response();
517        }
518        // Return URL for client download
519        let url = format!("/speech/{}", filename);
520        (StatusCode::OK, Json(json!({ "url": url }))).into_response()
521    } else {
522        (
523            StatusCode::BAD_REQUEST,
524            "Selected model is not a speech model",
525        )
526            .into_response()
527    }
