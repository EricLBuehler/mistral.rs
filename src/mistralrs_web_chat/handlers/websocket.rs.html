21/// Upgrades an HTTP request to a WebSocket connection.
22pub async fn ws_handler(
23    ws: WebSocketUpgrade,
24    State(app): State<Arc<AppState>>,
25) -> impl IntoResponse {
26    ws.on_upgrade(move |socket| handle_socket(socket, app))
27}

29/// Generic helper to stream tokens and forward them to the websocket.
30async fn stream_and_forward<Msgs, F, E>(
31    model: &Arc<Model>,
32    msgs: Msgs,
33    socket: &mut WebSocket,
34    mut on_token: F,
35    mut on_end: E,
36) -> Result<(), anyhow::Error>
37where
38    Msgs: mistralrs::RequestLike + Send + 'static,
39    F: FnMut(&str),
40    E: FnMut(),
41{
42    match model.stream_chat_request(msgs).await {
43        Ok(mut stream) => {
44            let mut assistant_reply = String::new();
45            while let Some(chunk) = stream.next().await {
46                if let mistralrs::Response::Chunk(resp) = chunk {
47                    if let Some(choice) = resp.choices.first() {
48                        if let Some(token) = &choice.delta.content {
49                            if socket.send(Message::Text(token.clone())).await.is_err() {
50                                break;
51                            }
52                            assistant_reply.push_str(token);
53                        }
54                    }
55                }
56            }
57            on_token(&assistant_reply);
58            on_end();
59            Ok(())
60        }
61        Err(e) => {
62            let _ = socket.send(Message::Text(format!("Error: {}", e))).await;
63            Err(e)
64        }
65    }
66}

68/// Per-connection task.
69pub async fn handle_socket(mut socket: WebSocket, app: Arc<AppState>) {
70    let mut text_msgs = TextMessages::new();
71    let mut vision_msgs = VisionMessages::new();
72    let mut image_buffer: Vec<image::DynamicImage> = Vec::new();
73    // `true` while we are streaming a reply back to the client.
74    let mut streaming = false;
75    // Track which chat file this websocket has prepared context for.
76    // When the user switches chats via /api/load_chat, app.current_chat changes;
77    // we must reset local state so conversations don't leak across chats.
78    let mut active_chat_id: Option<String> = {
79        // Whatever chat (if any) was active when the socket was opened.
80        app.current_chat.read().await.clone()
81    };
82
83    while let Some(Ok(Message::Text(user_msg))) = socket.next().await {
84        // ----- Detect chat switch -----
85        let cur_chat_id = app.current_chat.read().await.clone();
86        if cur_chat_id != active_chat_id {
87            // A new chat has been selected: wipe the message buffers so
88            // previous conversation state does not bleed into this chat.
89            text_msgs = TextMessages::new();
90            vision_msgs = VisionMessages::new();
91            image_buffer.clear();
92            active_chat_id = cur_chat_id;
93        }
94        // Allow client to request a context reset without closing the socket
95        if user_msg == CLEAR_CMD {
96            if streaming {
97                let _ = socket
98                    .send(Message::Text(
99                        "Cannot clear while assistant is replying.".into(),
100                    ))
101                    .await;
102            } else {
103                text_msgs = TextMessages::new();
104                vision_msgs = VisionMessages::new();
105                image_buffer.clear();
106                let _ = socket.send(Message::Text("[Context cleared]".into())).await;
107            }
108            continue;
109        }
110        // Handle front‑end replay helper messages without triggering inference
111        if user_msg.trim_start().starts_with("{\"restore\":") {
112            handle_restore_message(
113                &user_msg,
114                &app,
115                &mut text_msgs,
116                &mut vision_msgs,
117                &mut image_buffer,
118            )
119            .await;
120            continue;
121        }
122
123        let model_name_opt = { app.current.read().await.clone() };
124        let Some(model_name) = model_name_opt else {
125            let _ = socket
126                .send(Message::Text(
127                    "No model selected. Choose one in the sidebar.".into(),
128                ))
129                .await;
130            continue;
131        };
132        let Some(model_loaded) = app.models.get(&model_name).cloned() else {
133            let _ = socket
134                .send(Message::Text("Selected model not found.".into()))
135                .await;
136            continue;
137        };
138
139        match model_loaded {
140            LoadedModel::Text(model) => {
141                handle_text_model(
142                    &model,
143                    &user_msg,
144                    &mut text_msgs,
145                    &mut socket,
146                    &app,
147                    &mut streaming,
148                )
149                .await;
150            }
151            LoadedModel::Vision(model) => {
152                handle_vision_model(
153                    &model,
154                    &user_msg,
155                    &mut vision_msgs,
156                    &mut image_buffer,
157                    &mut socket,
158                    &app,
159                    &mut streaming,
160                )
161                .await;
162            }
163        }
164    }
165}

167async fn handle_restore_message(
168    user_msg: &str,
169    app: &Arc<AppState>,
170    text_msgs: &mut TextMessages,
171    vision_msgs: &mut VisionMessages,
172    image_buffer: &mut Vec<image::DynamicImage>,
173) {
174    if let Ok(val) = serde_json::from_str::<Value>(user_msg) {
175        if let Some(obj) = val.get("restore") {
176            // Handle restoring saved messages (with optional images)
177            if let (Some(role), Some(content)) = (
178                obj.get("role").and_then(|v| v.as_str()),
179                obj.get("content").and_then(|v| v.as_str()),
180            ) {
181                let has_images = obj
182                    .get("images")
183                    .and_then(|v| v.as_array())
184                    .is_some_and(|arr| !arr.is_empty());
185                match app.current.read().await.as_deref() {
186                    Some(model_name) if app.models.get(model_name).is_some() => {
187                        match app.models.get(model_name).unwrap() {
188                            LoadedModel::Text(_) => {
189                                // Text-only context
190                                *text_msgs = text_msgs.clone().add_message(
191                                    if role == "assistant" {
192                                        TextMessageRole::Assistant
193                                    } else {
194                                        TextMessageRole::User
195                                    },
196                                    content,
197                                );
198                            }
199                            LoadedModel::Vision(model) => {
200                                let role_enum = if role == "assistant" {
201                                    TextMessageRole::Assistant
202                                } else {
203                                    TextMessageRole::User
204                                };
205                                if has_images {
206                                    // Collect restored images
207                                    let mut imgs_b64 = Vec::new();
208                                    if let Some(arr) = obj.get("images").and_then(|v| v.as_array())
209                                    {
210                                        for img_val in arr {
211                                            if let Some(src) = img_val.as_str() {
212                                                if let Some(idx) = src.find(',') {
213                                                    let b64_data = &src[idx + 1..];
214                                                    imgs_b64.push(format!(
215                                                        "data:image/png;base64,{}",
216                                                        b64_data
217                                                    ));
218                                                    if let Ok(img_bytes) =
219                                                        BASE64.decode(b64_data.as_bytes())
220                                                    {
221                                                        if let Ok(img) =
222                                                            image::load_from_memory(&img_bytes)
223                                                        {
224                                                            image_buffer.push(img);
225                                                        }
226                                                    }
227                                                }
228                                            }
229                                        }
230                                    }
231                                    // Restore as an image message
232                                    if let Ok(updated) = vision_msgs.clone().add_image_message(
233                                        role_enum,
234                                        content,
235                                        image_buffer.clone(),
236                                        model,
237                                    ) {
238                                        *vision_msgs = updated;
239                                    }
240                                    // Clear buffer after use
241                                    image_buffer.clear();
242                                } else {
243                                    *vision_msgs =
244                                        vision_msgs.clone().add_message(role_enum, content);
245                                }
246                            }
247                        }
248                    }
249                    _ => {}
250                }
251            }
252        }
253    }
254}

256async fn handle_text_model(
257    model: &Arc<Model>,
258    user_msg: &str,
259    text_msgs: &mut TextMessages,
260    socket: &mut WebSocket,
261    app: &Arc<AppState>,
262    streaming: &mut bool,
263) {
264    *text_msgs = text_msgs
265        .clone()
266        .add_message(TextMessageRole::User, user_msg);
267    if let Err(e) = append_chat_message(app, "user", user_msg, None).await {
268        error!("chat save error: {}", e);
269    }
270    let mut assistant_content = String::new();
271    let msgs_snapshot = text_msgs.clone();
272
273    *streaming = true;
274    let stream_res = stream_and_forward(
275        model,
276        msgs_snapshot,
277        socket,
278        |tok| {
279            assistant_content = tok.to_string();
280            let cur = mem::take(text_msgs);
281            *text_msgs = cur.add_message(TextMessageRole::Assistant, tok);
282        },
283        || *streaming = false,
284    )
285    .await;
286    if !assistant_content.is_empty() {
287        let _ = append_chat_message(app, "assistant", &assistant_content, None).await;
288    }
289    if let Err(e) = stream_res {
290        error!("stream error: {}", e);
291    }
292}

294async fn handle_vision_model(
295    model: &Arc<Model>,
296    user_msg: &str,
297    vision_msgs: &mut VisionMessages,
298    image_buffer: &mut Vec<image::DynamicImage>,
299    socket: &mut WebSocket,
300    app: &Arc<AppState>,
301    streaming: &mut bool,
302) {
303    // Track the exact set of messages that will be sent *this* turn.
304    let mut msgs_for_stream: Option<VisionMessages> = None;
305    // --- Vision input routing ---
306    if let Ok(val) = serde_json::from_str::<Value>(user_msg) {
307        // Case 1: pure image payload => buffer it and wait for a prompt
308        if let Some(url) = val.get("image").and_then(|v| v.as_str()) {
309            // load & decode
310            match tokio::fs::read(url).await {
311                Ok(bytes) => match image::load_from_memory(&bytes) {
312                    Ok(img) => {
313                        image_buffer.push(img);
314                    }
315                    Err(e) => {
316                        error!("image decode error: {}", e);
317                        let _ = socket.send(Message::Text(format!("Error: {}", e))).await;
318                    }
319                },
320                Err(e) => {
321                    error!("image read error: {}", e);
322                    let _ = socket.send(Message::Text(format!("Error: {}", e))).await;
323                }
324            }
325            // Skip sending to model until we get a prompt
326            return;
327        } else {
328            // Fallback: treat whole JSON as text
329            *vision_msgs = vision_msgs
330                .clone()
331                .add_message(TextMessageRole::User, user_msg);
332            msgs_for_stream = Some(vision_msgs.clone());
333            if let Err(e) = append_chat_message(app, "user", user_msg, None).await {
334                error!("chat save error: {}", e);
335            }
336        }
337    } else {
338        // Plain-text prompt arrives here
339        if image_buffer.is_empty() {
340            *vision_msgs = vision_msgs
341                .clone()
342                .add_message(TextMessageRole::User, user_msg);
343            // Send the text‑only context to the model
344            msgs_for_stream = Some(vision_msgs.clone());
345            if let Err(e) = append_chat_message(app, "user", user_msg, None).await {
346                error!("chat save error: {}", e);
347            }
348        } else {
349            match vision_msgs.clone().add_image_message(
350                TextMessageRole::User,
351                user_msg,
352                image_buffer.clone(),
353                model,
354            ) {
355                Ok(updated) => {
356                    // Keep the *text‑only* conversation in our long‑term state,
357                    // but build a one‑off request that includes images.
358                    let temp_msgs = updated;
359                    *vision_msgs = vision_msgs
360                        .clone()
361                        .add_message(TextMessageRole::User, user_msg);
362                    msgs_for_stream = Some(temp_msgs.clone());
363                    // ---- persist user message with images ----
364                    let mut imgs_b64 = Vec::new();
365                    for img in image_buffer.iter() {
366                        let mut buf = Vec::new();
367                        if img
368                            .write_to(&mut Cursor::new(&mut buf), image::ImageFormat::Png)
369                            .is_ok()
370                        {
371                            imgs_b64.push(format!("data:image/png;base64,{}", BASE64.encode(&buf)));
372                        }
373                    }
374                    if let Err(e) = append_chat_message(app, "user", user_msg, Some(imgs_b64)).await
375                    {
376                        error!("chat save error: {}", e);
377                    }
378                    image_buffer.clear();
379                }
380                Err(e) => {
381                    error!("image prompt error: {}", e);
382                    let _ = socket.send(Message::Text(format!("Error: {}", e))).await;
383                }
384            }
385        }
386    }
387
388    *streaming = true;
389    let mut assistant_content = String::new();
390    let stream_res = stream_and_forward(
391        model,
392        msgs_for_stream.expect("msgs_for_stream must be set"),
393        socket,
394        |tok| {
395            assistant_content = tok.to_string();
396            let cur = mem::take(vision_msgs);
397            *vision_msgs = cur.add_message(TextMessageRole::Assistant, tok);
398        },
399        || *streaming = false,
400    )
401    .await;
402    if !assistant_content.is_empty() {
403        let _ = append_chat_message(app, "assistant", &assistant_content, None).await;
404    }
405    if let Err(e) = stream_res {
406        error!("stream error: {}", e);
407    }
