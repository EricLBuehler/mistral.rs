28/// Upgrades an HTTP request to a WebSocket connection.
29pub async fn ws_handler(
30    ws: WebSocketUpgrade,
31    State(app): State<Arc<AppState>>,
32) -> impl IntoResponse {
33    ws.on_upgrade(move |socket| handle_socket(socket, app))

36/// Generic helper to stream tokens and forward them to the websocket.
37async fn stream_and_forward<Msgs, F, E>(
38    model: &Arc<Model>,
39    msgs: Msgs,
40    socket: &mut WebSocket,
41    mut on_token: F,
42    mut on_end: E,
43) -> Result<(), anyhow::Error>
44where
45    Msgs: mistralrs::RequestLike + Send + 'static,
46    F: FnMut(&str),
47    E: FnMut(),
48{
49    match model.stream_chat_request(msgs).await {
50        Ok(mut stream) => {
51            let mut assistant_reply = String::new();
52            while let Some(chunk) = stream.next().await {
53                if let mistralrs::Response::Chunk(resp) = chunk {
54                    if let Some(choice) = resp.choices.first() {
55                        if let Some(token) = &choice.delta.content {
56                            if socket.send(Message::Text(token.clone())).await.is_err() {
57                                break;
58                            }
59                            assistant_reply.push_str(token);
60                        }
61                    }
62                }
63            }
64            on_token(&assistant_reply);
65            on_end();
66            Ok(())
67        }
68        Err(e) => {
69            let _ = socket.send(Message::Text(format!("Error: {}", e))).await;
70            Err(e)
71        }
72    }
73}

75/// Validate that a file path is safe and within the uploads directory
76fn validate_image_path(path: &str) -> Result<String, &'static str> {
77    // Determine upload directory under user cache
78    let uploads_dir = get_cache_dir().join("uploads");
79    let uploads_path = uploads_dir.as_path();
80
81    // Resolve the full path
82    let file_path = Path::new(path);
83
84    // Check if it's an absolute path and starts with our uploads directory
85    if !file_path.is_absolute() {
86        return Err("Path must be absolute");
87    }
88
89    // Normalize and check if the path is within uploads directory
90    match file_path.canonicalize() {
91        Ok(canonical_path) => match uploads_path.canonicalize() {
92            Ok(canonical_uploads) => {
93                if canonical_path.starts_with(canonical_uploads) {
94                    Ok(path.to_string())
95                } else {
96                    Err("Path is outside uploads directory")
97                }
98            }
99            Err(_) => Err("Uploads directory not found"),
100        },
101        Err(_) => Err("Invalid file path"),
102    }
103}

105/// Per-connection task.
106pub async fn handle_socket(mut socket: WebSocket, app: Arc<AppState>) {
107    let mut text_msgs = TextMessages::new();
108    let mut vision_msgs = VisionMessages::new();
109    let mut image_buffer: Vec<image::DynamicImage> = Vec::new();
110    // `true` while we are streaming a reply back to the client.
111    let mut streaming = false;
112    // Track per-connection chat ID; set by client via WebSocket control
113    let mut active_chat_id: Option<String> = None;
114
115    while let Some(Ok(Message::Text(user_msg))) = socket.next().await {
116        // Handle per-connection chat ID setting
117        if let Ok(val) = serde_json::from_str::<Value>(&user_msg) {
118            if let Some(id) = val.get("chat_id").and_then(|v| v.as_str()) {
119                if active_chat_id.as_deref() != Some(id) {
120                    active_chat_id = Some(id.to_string());
121                    text_msgs = TextMessages::new();
122                    vision_msgs = VisionMessages::new();
123                    image_buffer.clear();
124                }
125                continue;
126            }
127        }
128        // Allow client to request a context reset without closing the socket
129        if user_msg == CLEAR_CMD {
130            if streaming {
131                let _ = socket
132                    .send(Message::Text(
133                        "Cannot clear while assistant is replying.".into(),
134                    ))
135                    .await;
136            } else {
137                text_msgs = TextMessages::new();
138                vision_msgs = VisionMessages::new();
139                image_buffer.clear();
140                let _ = socket.send(Message::Text("[Context cleared]".into())).await;
141            }
142            continue;
143        }
144        // Handle front‑end replay helper messages without triggering inference
145        if user_msg.trim_start().starts_with("{\"restore\":") {
146            handle_restore_message(
147                &user_msg,
148                &app,
149                &mut text_msgs,
150                &mut vision_msgs,
151                &mut image_buffer,
152            )
153            .await;
154            continue;
155        }
156
157        let model_name_opt = { app.current.read().await.clone() };
158        let Some(model_name) = model_name_opt else {
159            let _ = socket
160                .send(Message::Text(
161                    "No model selected. Choose one in the sidebar.".into(),
162                ))
163                .await;
164            continue;
165        };
166        let Some(model_loaded) = app.models.get(&model_name).cloned() else {
167            let _ = socket
168                .send(Message::Text("Selected model not found.".into()))
169                .await;
170            continue;
171        };
172
173        match model_loaded {
174            LoadedModel::Text(model) => {
175                handle_text_model(
176                    &model,
177                    &user_msg,
178                    &mut text_msgs,
179                    &mut socket,
180                    &app,
181                    &mut streaming,
182                    &active_chat_id,
183                )
184                .await;
185            }
186            LoadedModel::Vision(model) => {
187                let mut vision_ctx = VisionContext {
188                    msgs: &mut vision_msgs,
189                    image_buffer: &mut image_buffer,
190                };
191                handle_vision_model(
192                    &model,
193                    &user_msg,
194                    &mut vision_ctx,
195                    &mut socket,
196                    &app,
197                    &mut streaming,
198                    &active_chat_id,
199                )
200                .await;
201            }
202        }
203    }
204}

206async fn handle_restore_message(
207    user_msg: &str,
208    app: &Arc<AppState>,
209    text_msgs: &mut TextMessages,
210    vision_msgs: &mut VisionMessages,
211    image_buffer: &mut Vec<image::DynamicImage>,
212) {
213    if let Ok(val) = serde_json::from_str::<Value>(user_msg) {
214        if let Some(obj) = val.get("restore") {
215            // Handle restoring saved messages (with optional images)
216            if let (Some(role), Some(content)) = (
217                obj.get("role").and_then(|v| v.as_str()),
218                obj.get("content").and_then(|v| v.as_str()),
219            ) {
220                let has_images = obj
221                    .get("images")
222                    .and_then(|v| v.as_array())
223                    .is_some_and(|arr| !arr.is_empty());
224                match app.current.read().await.as_deref() {
225                    Some(model_name) if app.models.get(model_name).is_some() => {
226                        match app.models.get(model_name).unwrap() {
227                            LoadedModel::Text(_) => {
228                                // Text-only context
229                                *text_msgs = text_msgs.clone().add_message(
230                                    if role == "assistant" {
231                                        TextMessageRole::Assistant
232                                    } else {
233                                        TextMessageRole::User
234                                    },
235                                    content,
236                                );
237                            }
238                            LoadedModel::Vision(model) => {
239                                let role_enum = if role == "assistant" {
240                                    TextMessageRole::Assistant
241                                } else {
242                                    TextMessageRole::User
243                                };
244                                if has_images {
245                                    // Collect restored images
246                                    let mut imgs_b64 = Vec::new();
247                                    if let Some(arr) = obj.get("images").and_then(|v| v.as_array())
248                                    {
249                                        for img_val in arr {
250                                            if let Some(src) = img_val.as_str() {
251                                                if let Some(idx) = src.find(',') {
252                                                    let b64_data = &src[idx + 1..];
253                                                    imgs_b64.push(format!(
254                                                        "data:image/png;base64,{}",
255                                                        b64_data
256                                                    ));
257                                                    if let Ok(img_bytes) =
258                                                        BASE64.decode(b64_data.as_bytes())
259                                                    {
260                                                        if let Ok(img) =
261                                                            image::load_from_memory(&img_bytes)
262                                                        {
263                                                            image_buffer.push(img);
264                                                        }
265                                                    }
266                                                }
267                                            }
268                                        }
269                                    }
270                                    // Restore as an image message
271                                    if let Ok(updated) = vision_msgs.clone().add_image_message(
272                                        role_enum,
273                                        content,
274                                        image_buffer.clone(),
275                                        model,
276                                    ) {
277                                        *vision_msgs = updated;
278                                    }
279                                    // Clear buffer after use
280                                    image_buffer.clear();
281                                } else {
282                                    *vision_msgs =
283                                        vision_msgs.clone().add_message(role_enum, content);
284                                }
285                            }
286                        }
287                    }
288                    _ => {}
289                }
290            }
291        }
292    }
293}

295async fn handle_text_model(
296    model: &Arc<Model>,
297    user_msg: &str,
298    text_msgs: &mut TextMessages,
299    socket: &mut WebSocket,
300    app: &Arc<AppState>,
301    streaming: &mut bool,
302    active_chat_id: &Option<String>,
303) {
304    *text_msgs = text_msgs
305        .clone()
306        .add_message(TextMessageRole::User, user_msg);
307    if let Some(chat_id) = active_chat_id {
308        if let Err(e) = append_chat_message(app, chat_id, "user", user_msg, None).await {
309            error!("chat save error: {}", e);
310        }
311    }
312    let mut assistant_content = String::new();
313    let msgs_snapshot = text_msgs.clone();
314
315    *streaming = true;
316    let stream_res = stream_and_forward(
317        model,
318        msgs_snapshot,
319        socket,
320        |tok| {
321            assistant_content = tok.to_string();
322            let cur = mem::take(text_msgs);
323            *text_msgs = cur.add_message(TextMessageRole::Assistant, tok);
324        },
325        || *streaming = false,
326    )
327    .await;
328    if !assistant_content.is_empty() {
329        if let Some(chat_id) = active_chat_id {
330            let _ = append_chat_message(app, chat_id, "assistant", &assistant_content, None).await;
331        }
332    }
333    if let Err(e) = stream_res {
334        error!("stream error: {}", e);
335    }
336}

338async fn handle_vision_model(
339    model: &Arc<Model>,
340    user_msg: &str,
341    vision_ctx: &mut VisionContext<'_>,
342    socket: &mut WebSocket,
343    app: &Arc<AppState>,
344    streaming: &mut bool,
345    active_chat_id: &Option<String>,
346) {
347    // Track the exact set of messages that will be sent *this* turn.
348    let mut msgs_for_stream: Option<VisionMessages> = None;
349    // --- Vision input routing ---
350    if let Ok(val) = serde_json::from_str::<Value>(user_msg) {
351        // Case 1: pure image payload => buffer it and wait for a prompt
352        if let Some(url) = val.get("image").and_then(|v| v.as_str()) {
353            match validate_image_path(url) {
354                Ok(safe_path) => {
355                    // load & decode
356                    match tokio::fs::read(&safe_path).await {
357                        Ok(bytes) => match image::load_from_memory(&bytes) {
358                            Ok(img) => {
359                                vision_ctx.image_buffer.push(img);
360                            }
361                            Err(e) => {
362                                error!("image decode error: {}", e);
363                                let _ = socket.send(Message::Text(format!("Error: {}", e))).await;
364                            }
365                        },
366                        Err(e) => {
367                            error!("image read error: {}", e);
368                            let _ = socket.send(Message::Text(format!("Error: {}", e))).await;
369                        }
370                    }
371                }
372                Err(e) => {
373                    error!("Invalid image path: {}", e);
374                    let _ = socket
375                        .send(Message::Text(format!("Error: Invalid image path - {}", e)))
376                        .await;
377                }
378            }
379            // Skip sending to model until we get a prompt
380            return;
381        } else {
382            // Fallback: treat whole JSON as text
383            *vision_ctx.msgs = vision_ctx
384                .msgs
385                .clone()
386                .add_message(TextMessageRole::User, user_msg);
387            msgs_for_stream = Some(vision_ctx.msgs.clone());
388            if let Some(chat_id) = active_chat_id {
389                if let Err(e) = append_chat_message(app, chat_id, "user", user_msg, None).await {
390                    error!("chat save error: {}", e);
391                }
392            }
393        }
394    } else {
395        // Plain-text prompt arrives here
396        if vision_ctx.image_buffer.is_empty() {
397            *vision_ctx.msgs = vision_ctx
398                .msgs
399                .clone()
400                .add_message(TextMessageRole::User, user_msg);
401            // Send the text‑only context to the model
402            msgs_for_stream = Some(vision_ctx.msgs.clone());
403            if let Some(chat_id) = active_chat_id {
404                if let Err(e) = append_chat_message(app, chat_id, "user", user_msg, None).await {
405                    error!("chat save error: {}", e);
406                }
407            }
408        } else {
409            match vision_ctx.msgs.clone().add_image_message(
410                TextMessageRole::User,
411                user_msg,
412                vision_ctx.image_buffer.clone(),
413                model,
414            ) {
415                Ok(updated) => {
416                    // Keep the *text‑only* conversation in our long‑term state,
417                    // but build a one‑off request that includes images.
418                    let temp_msgs = updated;
419                    *vision_ctx.msgs = vision_ctx
420                        .msgs
421                        .clone()
422                        .add_message(TextMessageRole::User, user_msg);
423                    msgs_for_stream = Some(temp_msgs.clone());
424                    // ---- persist user message with images ----
425                    let mut imgs_b64 = Vec::new();
426                    for img in vision_ctx.image_buffer.iter() {
427                        let mut buf = Vec::new();
428                        if img
429                            .write_to(&mut Cursor::new(&mut buf), image::ImageFormat::Png)
430                            .is_ok()
431                        {
432                            imgs_b64.push(format!("data:image/png;base64,{}", BASE64.encode(&buf)));
433                        }
434                    }
435                    if let Some(chat_id) = active_chat_id {
436                        if let Err(e) =
437                            append_chat_message(app, chat_id, "user", user_msg, Some(imgs_b64))
438                                .await
439                        {
440                            error!("chat save error: {}", e);
441                        }
442                    }
443                    vision_ctx.image_buffer.clear();
444                }
445                Err(e) => {
446                    error!("image prompt error: {}", e);
447                    let _ = socket.send(Message::Text(format!("Error: {}", e))).await;
448                }
449            }
450        }
451    }
452
453    *streaming = true;
454    let mut assistant_content = String::new();
455    let stream_res = stream_and_forward(
456        model,
457        msgs_for_stream.expect("msgs_for_stream must be set"),
458        socket,
459        |tok| {
460            assistant_content = tok.to_string();
461            let cur = mem::take(vision_ctx.msgs);
462            *vision_ctx.msgs = cur.add_message(TextMessageRole::Assistant, tok);
463        },
464        || *streaming = false,
465    )
466    .await;
467    if !assistant_content.is_empty() {
468        if let Some(chat_id) = active_chat_id {
469            let _ = append_chat_message(app, chat_id, "assistant", &assistant_content, None).await;
470        }
471    }
472    if let Err(e) = stream_res {
473        error!("stream error: {}", e);
474    }
