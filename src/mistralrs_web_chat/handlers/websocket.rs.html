59/// Apply generation parameters to a RequestBuilder
60fn apply_gen_params(
61    mut builder: RequestBuilder,
62    params: &Option<MessageGenerationParams>,
63    defaults: &GenerationParams,
64) -> RequestBuilder {
65    // Use provided params or fall back to defaults
66    let temp = params
67        .as_ref()
68        .and_then(|p| p.temperature)
69        .unwrap_or(defaults.temperature);
70    let top_p = params
71        .as_ref()
72        .and_then(|p| p.top_p)
73        .unwrap_or(defaults.top_p);
74    let top_k = params
75        .as_ref()
76        .and_then(|p| p.top_k)
77        .unwrap_or(defaults.top_k);
78    let max_tokens = params
79        .as_ref()
80        .and_then(|p| p.max_tokens)
81        .unwrap_or(defaults.max_tokens);
82    let rep_penalty = params
83        .as_ref()
84        .and_then(|p| p.repetition_penalty)
85        .unwrap_or(defaults.repetition_penalty);
86
87    builder = builder
88        .set_sampler_temperature(temp)
89        .set_sampler_topp(top_p)
90        .set_sampler_topk(top_k)
91        .set_sampler_max_len(max_tokens)
92        .set_sampler_frequency_penalty(rep_penalty);
93
94    builder
95}

105/// Generic helper to stream tokens and forward them to the websocket.
106async fn stream_and_forward<Msgs, F, E>(
107    model: &Arc<Model>,
108    msgs: Msgs,
109    socket: &mut WebSocket,
110    mut on_token: F,
111    mut on_end: E,
112) -> Result<(), anyhow::Error>
113where
114    Msgs: mistralrs::RequestLike + Send + 'static,
115    F: FnMut(&str),
116    E: FnMut(),
117{
118    match model.stream_chat_request(msgs).await {
119        Ok(mut stream) => {
120            let mut assistant_reply = String::new();
121            while let Some(chunk) = stream.next().await {
122                if let mistralrs::Response::Chunk(resp) = chunk {
123                    if let Some(choice) = resp.choices.first() {
124                        if let Some(token) = &choice.delta.content {
125                            if socket
126                                .send(Message::Text(token.clone().into()))
127                                .await
128                                .is_err()
129                            {
130                                break;
131                            }
132                            assistant_reply.push_str(token);
133                        }
134                    }
135                }
136            }
137            on_token(&assistant_reply);
138            on_end();
139            Ok(())
140        }
141        Err(e) => {
142            let _ = socket
143                .send(Message::Text(format!("Error: {e}").into()))
144                .await;
145            Err(e)
146        }
147    }
148}

150/// Validate that a file path is safe and within the uploads directory
151fn validate_image_path(path: &str) -> Result<String, &'static str> {
152    // Determine upload directory under user cache
153    let uploads_dir = get_cache_dir().join("uploads");
154    let uploads_path = uploads_dir.as_path();
155
156    // Resolve the full path
157    let file_path = Path::new(path);
158
159    // Check if it's an absolute path and starts with our uploads directory
160    if !file_path.is_absolute() {
161        return Err("Path must be absolute");
162    }
163
164    // Normalize and check if the path is within uploads directory
165    match file_path.canonicalize() {
166        Ok(canonical_path) => match uploads_path.canonicalize() {
167            Ok(canonical_uploads) => {
168                if canonical_path.starts_with(canonical_uploads) {
169                    Ok(path.to_string())
170                } else {
171                    Err("Path is outside uploads directory")
172                }
173            }
174            Err(_) => Err("Uploads directory not found"),
175        },
176        Err(_) => Err("Invalid file path"),
177    }
178}

205/// Per-connection task.
206pub async fn handle_socket(mut socket: WebSocket, app: Arc<AppState>) {
207    let mut text_msgs = TextMessages::new();
208    let mut vision_msgs = VisionMessages::new();
209    let mut image_buffer: Vec<image::DynamicImage> = Vec::new();
210    let mut audio_buffer: Vec<AudioInput> = Vec::new();
211    // `true` while we are streaming a reply back to the client.
212    let mut streaming = false;
213    // Track per-connection chat ID; set by client via WebSocket control
214    let mut active_chat_id: Option<String> = None;
215    // Per-connection system prompt (can be set via WebSocket message)
216    let mut system_prompt: Option<String> = app.default_params.system_prompt.clone();
217    // Track whether we've added the system prompt to the context
218    let mut system_prompt_added = false;
219
220    while let Some(Ok(Message::Text(user_msg))) = socket.next().await {
221        // Handle per-connection chat ID setting
222        if let Ok(val) = serde_json::from_str::<Value>(&user_msg) {
223            if let Some(id) = val.get("chat_id").and_then(|v| v.as_str()) {
224                if active_chat_id.as_deref() != Some(id) {
225                    active_chat_id = Some(id.to_string());
226                    text_msgs = TextMessages::new();
227                    vision_msgs = VisionMessages::new();
228                    image_buffer.clear();
229                    system_prompt_added = false;
230                }
231                continue;
232            }
233            // Handle system prompt updates
234            if let Some(prompt) = val.get("set_system_prompt") {
235                system_prompt = if prompt.is_null() {
236                    None
237                } else {
238                    prompt.as_str().map(|s| s.to_string())
239                };
240                system_prompt_added = false;
241                let _ = socket
242                    .send(Message::Text("[System prompt updated]".into()))
243                    .await;
244                continue;
245            }
246        }
247        // Allow client to request a context reset without closing the socket
248        if user_msg == CLEAR_CMD {
249            if streaming {
250                let _ = socket
251                    .send(Message::Text(
252                        "Cannot clear while assistant is replying.".into(),
253                    ))
254                    .await;
255            } else {
256                text_msgs = TextMessages::new();
257                vision_msgs = VisionMessages::new();
258                image_buffer.clear();
259                system_prompt_added = false;
260                let _ = socket.send(Message::Text("[Context cleared]".into())).await;
261            }
262            continue;
263        }
264        // Handle front-end replay helper messages without triggering inference
265        if user_msg.trim_start().starts_with("{\"restore\":") {
266            handle_restore_message(
267                &user_msg,
268                &app,
269                &mut text_msgs,
270                &mut vision_msgs,
271                &mut image_buffer,
272            )
273            .await;
274            continue;
275        }
276        // Handle chat messages with optional web search options and generation params provided as JSON
277        if let Ok(val) = serde_json::from_str::<Value>(&user_msg) {
278            if let Some(content) = val.get("content").and_then(|v| v.as_str()) {
279                // Extract web search options if provided
280                let web_search_opts = if let Some(opts_val) = val.get("web_search_options") {
281                    match serde_json::from_value::<WebSearchOptions>(opts_val.clone()) {
282                        Ok(opts) => Some(opts),
283                        Err(e) => {
284                            let _ = socket
285                                .send(Message::Text(
286                                    format!("Error parsing web_search_options: {e}").into(),
287                                ))
288                                .await;
289                            None
290                        }
291                    }
292                } else {
293                    None
294                };
295                // Extract generation parameters if provided
296                let gen_params = val.get("generation_params").and_then(|v| {
297                    serde_json::from_value::<MessageGenerationParams>(v.clone()).ok()
298                });
299
300                // Determine selected model
301                let model_name_opt = { app.current.read().await.clone() };
302                let Some(model_name) = model_name_opt else {
303                    let _ = socket
304                        .send(Message::Text(
305                            "No model selected. Choose one in the sidebar.".into(),
306                        ))
307                        .await;
308                    continue;
309                };
310                let Some(model_loaded) = app.models.get(&model_name).cloned() else {
311                    let _ = socket
312                        .send(Message::Text("Selected model not found.".into()))
313                        .await;
314                    continue;
315                };
316
317                // Add system prompt if not already added
318                if !system_prompt_added {
319                    if let Some(ref prompt) = system_prompt {
320                        text_msgs = text_msgs.add_message(TextMessageRole::System, prompt);
321                        vision_msgs = vision_msgs.add_message(TextMessageRole::System, prompt);
322                    }
323                    system_prompt_added = true;
324                }
325
326                match model_loaded {
327                    LoadedModel::Text(model) => {
328                        let mut params = HandlerParams {
329                            socket: &mut socket,
330                            app: &app,
331                            streaming: &mut streaming,
332                            active_chat_id: &active_chat_id,
333                            gen_params: gen_params.clone(),
334                            system_prompt: &system_prompt,
335                        };
336                        handle_text_model(
337                            &model,
338                            content,
339                            web_search_opts.clone(),
340                            &mut text_msgs,
341                            &mut params,
342                        )
343                        .await;
344                    }
345                    LoadedModel::Vision(model) => {
346                        let mut vision_ctx = VisionContext {
347                            msgs: &mut vision_msgs,
348                            image_buffer: &mut image_buffer,
349                            audio_buffer: &mut audio_buffer,
350                        };
351                        let mut params = HandlerParams {
352                            socket: &mut socket,
353                            app: &app,
354                            streaming: &mut streaming,
355                            active_chat_id: &active_chat_id,
356                            gen_params: gen_params.clone(),
357                            system_prompt: &system_prompt,
358                        };
359                        handle_vision_model(
360                            &model,
361                            content,
362                            web_search_opts.clone(),
363                            &mut vision_ctx,
364                            &mut params,
365                        )
366                        .await;
367                    }
368                    LoadedModel::Speech(_) => {
369                        let _ = socket
370                            .send(Message::Text(
371                                "Speech models are not supported over WebSocket".into(),
372                            ))
373                            .await;
374                    }
375                }
376                continue;
377            }
378        }
379
380        let model_name_opt = { app.current.read().await.clone() };
381        let Some(model_name) = model_name_opt else {
382            let _ = socket
383                .send(Message::Text(
384                    "No model selected. Choose one in the sidebar.".into(),
385                ))
386                .await;
387            continue;
388        };
389        let Some(model_loaded) = app.models.get(&model_name).cloned() else {
390            let _ = socket
391                .send(Message::Text("Selected model not found.".into()))
392                .await;
393            continue;
394        };
395
396        // Add system prompt if not already added
397        if !system_prompt_added {
398            if let Some(ref prompt) = system_prompt {
399                text_msgs = text_msgs.add_message(TextMessageRole::System, prompt);
400                vision_msgs = vision_msgs.add_message(TextMessageRole::System, prompt);
401            }
402            system_prompt_added = true;
403        }
404
405        match model_loaded {
406            LoadedModel::Text(model) => {
407                let mut params = HandlerParams {
408                    socket: &mut socket,
409                    app: &app,
410                    streaming: &mut streaming,
411                    active_chat_id: &active_chat_id,
412                    gen_params: None,
413                    system_prompt: &system_prompt,
414                };
415                handle_text_model(&model, &user_msg, None, &mut text_msgs, &mut params).await;
416            }
417            LoadedModel::Vision(model) => {
418                let mut vision_ctx = VisionContext {
419                    msgs: &mut vision_msgs,
420                    image_buffer: &mut image_buffer,
421                    audio_buffer: &mut audio_buffer,
422                };
423                let mut params = HandlerParams {
424                    socket: &mut socket,
425                    app: &app,
426                    streaming: &mut streaming,
427                    active_chat_id: &active_chat_id,
428                    gen_params: None,
429                    system_prompt: &system_prompt,
430                };
431                handle_vision_model(&model, &user_msg, None, &mut vision_ctx, &mut params).await;
432            }
433            // Speech models should use HTTP endpoint; not handled here
434            LoadedModel::Speech(_) => {
435                let _ = socket
436                    .send(Message::Text(
437                        "Speech models are not supported over WebSocket".into(),
438                    ))
439                    .await;
440            }
441        }
442    }
443}

445async fn handle_restore_message(
446    user_msg: &str,
447    app: &Arc<AppState>,
448    text_msgs: &mut TextMessages,
449    vision_msgs: &mut VisionMessages,
450    image_buffer: &mut Vec<image::DynamicImage>,
451) {
452    if let Ok(val) = serde_json::from_str::<Value>(user_msg) {
453        if let Some(obj) = val.get("restore") {
454            // Handle restoring saved messages (with optional images)
455            if let (Some(role), Some(content)) = (
456                obj.get("role").and_then(|v| v.as_str()),
457                obj.get("content").and_then(|v| v.as_str()),
458            ) {
459                let has_images = obj
460                    .get("images")
461                    .and_then(|v| v.as_array())
462                    .is_some_and(|arr| !arr.is_empty());
463                match app.current.read().await.as_deref() {
464                    Some(model_name) if app.models.get(model_name).is_some() => {
465                        match app.models.get(model_name).unwrap() {
466                            LoadedModel::Text(_) => {
467                                // Text-only context
468                                *text_msgs = text_msgs.clone().add_message(
469                                    if role == "assistant" {
470                                        TextMessageRole::Assistant
471                                    } else {
472                                        TextMessageRole::User
473                                    },
474                                    content,
475                                );
476                            }
477                            LoadedModel::Vision(model) => {
478                                let role_enum = if role == "assistant" {
479                                    TextMessageRole::Assistant
480                                } else {
481                                    TextMessageRole::User
482                                };
483                                if has_images {
484                                    // Collect restored images
485                                    let mut imgs_b64 = Vec::new();
486                                    if let Some(arr) = obj.get("images").and_then(|v| v.as_array())
487                                    {
488                                        for img_val in arr {
489                                            if let Some(src) = img_val.as_str() {
490                                                if let Some(idx) = src.find(',') {
491                                                    let b64_data = &src[idx + 1..];
492                                                    imgs_b64.push(format!(
493                                                        "data:image/png;base64,{b64_data}"
494                                                    ));
495                                                    if let Ok(img_bytes) =
496                                                        BASE64.decode(b64_data.as_bytes())
497                                                    {
498                                                        if let Ok(img) =
499                                                            image::load_from_memory(&img_bytes)
500                                                        {
501                                                            image_buffer.push(img);
502                                                        }
503                                                    }
504                                                }
505                                            }
506                                        }
507                                    }
508                                    // Restore as an image message
509                                    if let Ok(updated) = vision_msgs.clone().add_image_message(
510                                        role_enum,
511                                        content,
512                                        image_buffer.clone(),
513                                        model,
514                                    ) {
515                                        *vision_msgs = updated;
516                                    }
517                                    // Clear buffer after use
518                                    image_buffer.clear();
519                                } else {
520                                    *vision_msgs =
521                                        vision_msgs.clone().add_message(role_enum, content);
522                                }
523                            }
524                            // Speech models do not support context restore over WebSocket
525                            LoadedModel::Speech(_) => {}
526                        }
527                    }
528                    _ => {}
529                }
530            }
531        }
532    }
533}

535async fn handle_text_model(
536    model: &Arc<Model>,
537    user_msg: &str,
538    web_search_opts: Option<WebSearchOptions>,
539    text_msgs: &mut TextMessages,
540    params: &mut HandlerParams<'_>,
541) {
542    // Local aliases keep the original body unchanged.
543    let socket = &mut *params.socket;
544    let app = params.app;
545    let streaming = &mut *params.streaming;
546    let active_chat_id = params.active_chat_id;
547    *text_msgs = text_msgs
548        .clone()
549        .add_message(TextMessageRole::User, user_msg);
550    if let Some(chat_id) = active_chat_id {
551        if let Err(e) = append_chat_message(app, chat_id, "user", user_msg, None).await {
552            error!("chat save error: {}", e);
553        }
554    }
555    let mut assistant_content = String::new();
556    let msgs_snapshot = text_msgs.clone();
557    let mut request_builder = RequestBuilder::from(msgs_snapshot);
558
559    // Apply generation parameters
560    request_builder = apply_gen_params(request_builder, &params.gen_params, &app.default_params);
561
562    if let Some(opts) = web_search_opts {
563        request_builder = request_builder.with_web_search_options(opts);
564    }
565
566    *streaming = true;
567    let stream_res = stream_and_forward(
568        model,
569        request_builder,
570        socket,
571        |tok| {
572            assistant_content = tok.to_string();
573            let cur = mem::take(text_msgs);
574            *text_msgs = cur.add_message(TextMessageRole::Assistant, tok);
575        },
576        || *streaming = false,
577    )
578    .await;
579    if !assistant_content.is_empty() {
580        if let Some(chat_id) = active_chat_id {
581            let _ = append_chat_message(app, chat_id, "assistant", &assistant_content, None).await;
582        }
583    }
584    if let Err(e) = stream_res {
585        error!("stream error: {}", e);
586    }
587}

589async fn handle_vision_model(
590    model: &Arc<Model>,
591    user_msg: &str,
592    web_search_opts: Option<WebSearchOptions>,
593    vision_ctx: &mut VisionContext<'_>,
594    params: &mut HandlerParams<'_>,
595) {
596    let socket = &mut *params.socket;
597    let app = params.app;
598    let streaming = &mut *params.streaming;
599    let active_chat_id = params.active_chat_id;
600    // Track the exact set of messages that will be sent *this* turn.
601    let mut msgs_for_stream: Option<VisionMessages> = None;
602    // --- Vision input routing ---
603    if let Ok(val) = serde_json::from_str::<Value>(user_msg) {
604        // Case 1a: pure image payload => buffer it and wait for a prompt
605        if let Some(url) = val.get("image").and_then(|v| v.as_str()) {
606            match validate_image_path(url) {
607                Ok(safe_path) => {
608                    // load & decode
609                    match tokio::fs::read(&safe_path).await {
610                        Ok(bytes) => match image::load_from_memory(&bytes) {
611                            Ok(img) => {
612                                vision_ctx.image_buffer.push(img);
613                            }
614                            Err(e) => {
615                                error!("image decode error: {}", e);
616                                let _ = socket
617                                    .send(Message::Text(format!("Error: {e}").into()))
618                                    .await;
619                            }
620                        },
621                        Err(e) => {
622                            error!("image read error: {}", e);
623                            let _ = socket
624                                .send(Message::Text(format!("Error: {e}").into()))
625                                .await;
626                        }
627                    }
628                }
629                Err(e) => {
630                    error!("Invalid image path: {}", e);
631                    let _ = socket
632                        .send(Message::Text(
633                            format!("Error: Invalid image path - {e}").into(),
634                        ))
635                        .await;
636                }
637            }
638            // Skip sending to model until we get a prompt
639            return;
640        // Case 1b: pure audio payload => buffer and wait for prompt
641        } else if let Some(url) = val.get("audio").and_then(|v| v.as_str()) {
642            match validate_audio_path(url) {
643                Ok(safe_path) => match tokio::fs::read(&safe_path).await {
644                    Ok(bytes) => match AudioInput::from_bytes(&bytes) {
645                        Ok(audio) => {
646                            vision_ctx.audio_buffer.push(audio);
647                        }
648                        Err(e) => {
649                            error!("audio decode error: {}", e);
650                            let _ = socket
651                                .send(Message::Text(format!("Error: {e}").into()))
652                                .await;
653                        }
654                    },
655                    Err(e) => {
656                        error!("audio read error: {}", e);
657                        let _ = socket
658                            .send(Message::Text(format!("Error: {e}").into()))
659                            .await;
660                    }
661                },
662                Err(e) => {
663                    error!("Invalid audio path: {}", e);
664                    let _ = socket
665                        .send(Message::Text(
666                            format!("Error: Invalid audio path - {e}").into(),
667                        ))
668                        .await;
669                }
670            }
671            return;
672        } else {
673            // Fallback: treat whole JSON as text
674            *vision_ctx.msgs = vision_ctx
675                .msgs
676                .clone()
677                .add_message(TextMessageRole::User, user_msg);
678            msgs_for_stream = Some(vision_ctx.msgs.clone());
679            if let Some(chat_id) = active_chat_id {
680                if let Err(e) = append_chat_message(app, chat_id, "user", user_msg, None).await {
681                    error!("chat save error: {}", e);
682                }
683            }
684        }
685    } else {
686        // Plain-text prompt arrives here
687        if vision_ctx.image_buffer.is_empty() && vision_ctx.audio_buffer.is_empty() {
688            *vision_ctx.msgs = vision_ctx
689                .msgs
690                .clone()
691                .add_message(TextMessageRole::User, user_msg);
692            // Send the textonly context to the model
693            msgs_for_stream = Some(vision_ctx.msgs.clone());
694            if let Some(chat_id) = active_chat_id {
695                if let Err(e) = append_chat_message(app, chat_id, "user", user_msg, None).await {
696                    error!("chat save error: {}", e);
697                }
698            }
699        } else {
700            // Prepare multimodal message with images and/or audios
701            match vision_ctx.msgs.clone().add_multimodal_message(
702                TextMessageRole::User,
703                user_msg,
704                vision_ctx.image_buffer.clone(),
705                vision_ctx.audio_buffer.clone(),
706                model,
707            ) {
708                Ok(updated) => {
709                    // Keep the *textonly* conversation in our longterm state,
710                    // but build a oneoff request that includes images.
711                    let temp_msgs = updated;
712                    *vision_ctx.msgs = vision_ctx
713                        .msgs
714                        .clone()
715                        .add_message(TextMessageRole::User, user_msg);
716                    msgs_for_stream = Some(temp_msgs.clone());
717                    // ---- persist user message with images ----
718                    let mut imgs_b64 = Vec::new();
719                    for img in vision_ctx.image_buffer.iter() {
720                        let mut buf = Vec::new();
721                        if img
722                            .write_to(&mut Cursor::new(&mut buf), image::ImageFormat::Png)
723                            .is_ok()
724                        {
725                            imgs_b64.push(format!("data:image/png;base64,{}", BASE64.encode(&buf)));
726                        }
727                    }
728                    if let Some(chat_id) = active_chat_id {
729                        if let Err(e) = append_chat_message(
730                            app,
731                            chat_id,
732                            "user",
733                            user_msg,
734                            if imgs_b64.is_empty() {
735                                None
736                            } else {
737                                Some(imgs_b64)
738                            },
739                        )
740                        .await
741                        {
742                            error!("chat save error: {}", e);
743                        }
744                    }
745                    vision_ctx.image_buffer.clear();
746                    vision_ctx.audio_buffer.clear();
747                }
748                Err(e) => {
749                    error!("image prompt error: {}", e);
750                    let _ = socket
751                        .send(Message::Text(format!("Error: {e}").into()))
752                        .await;
753                }
754            }
755        }
756    }
757
758    let msgs = msgs_for_stream.expect("msgs_for_stream must be set");
759    let mut request_builder = RequestBuilder::from(msgs);
760
761    // Apply generation parameters
762    request_builder = apply_gen_params(request_builder, &params.gen_params, &app.default_params);
763
764    if let Some(opts) = web_search_opts {
765        request_builder = request_builder.with_web_search_options(opts);
766    }
767    *streaming = true;
768    let mut assistant_content = String::new();
769    let stream_res = stream_and_forward(
770        model,
771        request_builder,
772        socket,
773        |tok| {
774            assistant_content = tok.to_string();
775            let cur = mem::take(vision_ctx.msgs);
776            *vision_ctx.msgs = cur.add_message(TextMessageRole::Assistant, tok);
777        },
778        || *streaming = false,
779    )
780    .await;
781    if !assistant_content.is_empty() {
782        if let Some(chat_id) = active_chat_id {
783            let _ = append_chat_message(app, chat_id, "assistant", &assistant_content, None).await;
784        }
785    }
786    if let Err(e) = stream_res {
787        error!("stream error: {}", e);
788    }
789}
