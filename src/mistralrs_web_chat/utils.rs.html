1use std::path::PathBuf;
2
3/// Determine the base cache directory for the application.
4/// Uses XDG_CACHE_HOME or falls back to ~/.cache/mistralrs-web-chat.
5pub fn get_cache_dir() -> PathBuf {
6    // XDG_CACHE_HOME or default to ~/.cache
7    let cache_home = std::env::var("XDG_CACHE_HOME")
8        .map(PathBuf::from)
9        .unwrap_or_else(|_| {
10            std::env::var("HOME")
11                .map(|h| PathBuf::from(h).join(".cache"))
12                .unwrap_or_else(|_| PathBuf::from(".cache"))
13        });
14    cache_home.join("mistralrs-web-chat")
15}
