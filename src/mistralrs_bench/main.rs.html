48fn run_bench(
49    mistralrs: Arc<MistralRs>,
50    prompt: RequestMessage,
51    n_gen: usize,
52    concurrency: usize,
53    repetitions: usize,
54    test_name: TestName,
55) -> anyhow::Result<BenchResult> {
56    let sampling_params = SamplingParams {
57        temperature: Some(0.1),
58        top_k: Some(32),
59        top_p: Some(0.1),
60        min_p: Some(0.05),
61        top_n_logprobs: 0,
62        frequency_penalty: Some(0.1),
63        presence_penalty: Some(0.1),
64        max_len: Some(n_gen),
65        stop_toks: None,
66        logits_bias: None,
67        n_choices: 1,
68        dry_params: Some(DrySamplingParams::default()),
69    };
70    let sender = mistralrs.get_sender().unwrap();
71    let (tx, mut rx) = channel(10_000);
72
73    let req = Request::Normal(NormalRequest {
74        id: mistralrs.next_request_id(),
75        messages: prompt,
76        sampling_params: sampling_params.clone(),
77        response: tx,
78        return_logprobs: false,
79        is_streaming: false,
80        constraint: Constraint::None,
81        suffix: None,
82        tools: None,
83        tool_choice: None,
84        logits_processors: None,
85        return_raw_logits: false,
86        web_search_options: None,
87    });
88
89    let mut usages = Vec::new();
90
91    for _ in 0..repetitions {
92        for _ in 0..concurrency {
93            sender
94                .blocking_send(req.clone())
95                .expect("Expected receiver.");
96        }
97        for _ in 0..concurrency {
98            match rx.blocking_recv() {
99                Some(r) => match r {
100                    Response::InternalError(e) => {
101                        unreachable!("Got an internal error: {e:?}");
102                    }
103                    Response::ModelError(e, resp) => {
104                        unreachable!("Got a model error: {e:?}, response: {resp:?}");
105                    }
106                    Response::ValidationError(e) => {
107                        unreachable!("Got a validation error: {e:?}");
108                    }
109                    Response::Done(res) => {
110                        usages.push(res.usage);
111                    }
112                    Response::Chunk(_) => unreachable!(),
113                    Response::CompletionModelError(_, _) => unreachable!(),
114                    Response::CompletionDone(res) => {
115                        usages.push(res.usage);
116                    }
117                    Response::CompletionChunk(_) => unreachable!(),
118                    Response::ImageGeneration(_) => unreachable!(),
119                    Response::Raw { .. } => unreachable!(),
120                },
121                None => unreachable!("Expected a Done response, got None",),
122            }
123        }
124    }
125
126    Ok(BenchResult {
127        usages,
128        concurrency,
129        test_name,
130    })
131}

133fn get_tok_s(result: &BenchResult) -> UncertainTokSec {
134    let ts_measurements = match result.test_name {
135        TestName::Prompt(_) => result
136            .usages
137            .iter()
138            .map(|u| u.avg_prompt_tok_per_sec)
139            .collect::<Vec<_>>(),
140        TestName::Gen(_) => result
141            .usages
142            .iter()
143            .map(|u| u.avg_compl_tok_per_sec)
144            .collect::<Vec<_>>(),
145    };
146    // Calculate uncertainty
147    let mean = ts_measurements.iter().sum::<f32>() / ts_measurements.len() as f32;
148    let variance = ts_measurements
149        .iter()
150        .map(|e| (mean - e).powf(2.))
151        .sum::<f32>()
152        / ts_measurements.len() as f32;
153    let std_dev = variance.sqrt();
154    UncertainTokSec { mean, std_dev }
155}

157fn get_ms_tok(result: &BenchResult) -> UncertainTokSec {
158    let ms_tok_measurements = match result.test_name {
159        TestName::Prompt(_) => result
160            .usages
161            .iter()
162            .map(|u| 1000. / u.avg_prompt_tok_per_sec)
163            .collect::<Vec<_>>(),
164        TestName::Gen(_) => result
165            .usages
166            .iter()
167            .map(|u| 1000. / u.avg_compl_tok_per_sec)
168            .collect::<Vec<_>>(),
169    };
170    // Calculate uncertainty
171    let mean = ms_tok_measurements.iter().sum::<f32>() / ms_tok_measurements.len() as f32;
172    let variance = ms_tok_measurements
173        .iter()
174        .map(|e| (mean - e).powf(2.))
175        .sum::<f32>()
176        / ms_tok_measurements.len() as f32;
177    let std_dev = variance.sqrt();
178    UncertainTokSec { mean, std_dev }
179}

181fn print_usage(model: &str, device: &Device, results: Vec<BenchResult>) {
182    let backend = match device {
183        Device::Cpu => "CPU",
184        Device::Cuda(_) => "CUDA",
185        Device::Metal(_) => "Metal",
186    };
187    let results: Vec<Vec<CellStruct>> = results
188        .into_iter()
189        .map(|r| {
190            vec![
191                model.cell(),
192                backend.cell(),
193                r.test_name.to_string().cell(),
194                get_tok_s(&r).cell().justify(Justify::Right),
195                get_ms_tok(&r).cell().justify(Justify::Right),
196                r.concurrency.cell().justify(Justify::Right),
197                (get_tok_s(&r).mean * r.concurrency as f32)
198                    .cell()
199                    .justify(Justify::Right),
200            ]
201        })
202        .collect();
203
204    let table = results
205        .table()
206        .title(vec![
207            "model".cell().bold(true),
208            // "size".cell().bold(true),
209            // "params".cell().bold(true),
210            "backend".cell().bold(true),
211            // "ngl".cell().bold(true),
212            "test".cell().bold(true),
213            "t/s".cell().bold(true),
214            "ms/t".cell().bold(true),
215            "concurrency".cell().bold(true),
216            "throughput/s".cell().bold(true),
217        ])
218        .bold(true);
219    print_stdout(table).expect("print table");
220}

291    #[arg(long, short, default_value_t = 5)]
292    repetitions: usize,
293
294    /// NOTE: This can be omitted to use automatic device mapping!
295    /// Number of device layers to load and run on GPU(s). All others will be on the CPU.
296    /// If one GPU is used, then this value should be an integer. Otherwise, it follows the following pattern:
297    /// ORD:NUM;... Where ORD is a unique device ordinal and NUM is the number of layers for that device.
298    #[arg(short, long, value_parser, value_delimiter = ';')]

308    #[arg(long = "pa-gpu-mem")]
309    paged_attn_gpu_mem: Option<usize>,
310
311    /// Percentage of GPU memory to utilize after allocation of KV cache with PagedAttention, from 0 to 1.
312    /// If this is not set and the device is CUDA, it will default to `0.9`.
313    /// PagedAttention is supported on CUDA and Metal. It is automatically activated on CUDA but not on Metal.
314    /// The priority is as follows: `pa-ctxt-len` > `pa-gpu-mem-usage` > `pa-gpu-mem`.
315    #[arg(long = "pa-gpu-mem-usage")]
316    paged_attn_gpu_mem_usage: Option<f32>,
317
318    /// Total context length to allocate the KV cache for (total number of tokens which the KV cache can hold).
319    /// PagedAttention is supported on CUDA and Metal. It is automatically activated on CUDA but not on Metal.

328    paged_attn_block_size: Option<usize>,

334    /// Enable PagedAttention on Metal. Because PagedAttention is already enabled on CUDA, this is only applicable on Metal.

337
338    /// Number of tokens to batch the prompt step into. This can help with OOM errors when in the prompt step, but reduces performance.
339    #[arg(long = "prompt-batchsize")]

343fn main() -> anyhow::Result<()> {
344    let mut args = Args::parse();
345    initialize_logging();
346
347    args.concurrency = Some(args.concurrency.unwrap_or(vec![1]));
348
349    let prompt_chunksize = match args.prompt_chunksize {
350        Some(0) => {
351            anyhow::bail!("`prompt_chunksize` must be a strictly positive integer, got 0.",)
352        }
353        Some(x) => Some(NonZeroUsize::new(x).unwrap()),
354        None => None,
355    };
356
357    let dtype = get_model_dtype(&args.model)?;
358    let auto_device_map_params = get_auto_device_map_params(&args.model)?;
359
360    let max_seq_len = auto_device_map_params.max_seq_len();
361
362    let loader: Box<dyn Loader> = LoaderBuilder::new(args.model)
363        .with_prompt_chunksize(prompt_chunksize)
364        .build()?;
365    let model_name = loader.get_id();
366
367    #[cfg(feature = "metal")]
368    let device = Device::new_metal(0)?;
369    #[cfg(not(feature = "metal"))]
370    let device = if mistralrs_core::distributed::use_nccl() {
371        Device::Cpu
372    } else {
373        Device::cuda_if_available(0)?
374    };
375
376    if let Some(seed) = args.seed {
377        device.set_seed(seed)?;
378    }
379
380    let token_source = TokenSource::CacheToken;
381    info!(
382        "avx: {}, neon: {}, simd128: {}, f16c: {}",
383        candle_core::utils::with_avx(),
384        candle_core::utils::with_neon(),
385        candle_core::utils::with_simd128(),
386        candle_core::utils::with_f16c()
387    );
388    info!("Sampling method: penalties -> temperature -> topk -> topp -> minp -> multinomial");
389    info!("Model kind is: {}", loader.get_kind().to_string());
390
391    // Parse device mapper
392    let mapper = if let Some(device_layers) = args.num_device_layers {
393        if device_layers.len() == 1 && device_layers[0].parse::<usize>().is_ok() {
394            let layers = device_layers[0].parse::<usize>().unwrap();
395            DeviceMapSetting::Map(DeviceMapMetadata::from_num_device_layers(vec![
396                DeviceLayerMapMetadata { ordinal: 0, layers },
397            ]))
398        } else {
399            let mut mapping = Vec::new();
400            for layer in device_layers {
401                let split = layer.splitn(2, ':').collect::<Vec<_>>();
402                if split.len() < 2 {
403                    panic!("Expected layer to be of format ORD:NUM, got {layer}");
404                }
405                let ord = split[0]
406                    .parse::<usize>()
407                    .unwrap_or_else(|_| panic!("Failed to parse {} as integer.", split[0]));
408                let num = split[1]
409                    .parse::<usize>()
410                    .unwrap_or_else(|_| panic!("Failed to parse {} as integer.", split[1]));
411                for DeviceLayerMapMetadata { ordinal, layers: _ } in &mapping {
412                    if *ordinal == ord {
413                        panic!("Duplicate ordinal {ord}");
414                    }
415                }
416                mapping.push(DeviceLayerMapMetadata {
417                    ordinal: ord,
418                    layers: num,
419                });
420            }
421            DeviceMapSetting::Map(DeviceMapMetadata::from_num_device_layers(mapping))
422        }
423    } else {
424        DeviceMapSetting::Auto(auto_device_map_params)
425    };
426
427    let no_paged_attn = if device.is_cuda() || mistralrs_core::distributed::use_nccl() {
428        args.no_paged_attn
429    } else if device.is_metal() {
430        !args.paged_attn
431    } else {
432        true
433    };
434
435    // Allocate 0.5 GB of CPU memory just as a placeholder.
436    // Nothing happens here as we have no `swap_out`, see `_preempt_by_swap`.
437    let cache_config = match (
438        args.paged_attn_block_size,
439        args.paged_attn_gpu_mem,
440        args.paged_attn_gpu_mem_usage,
441        args.paged_ctxt_len,
442        paged_attn_supported(),
443        no_paged_attn,
444    ) {
445        (block_size, None, None, None, true, false) => Some(PagedAttentionConfig::new(
446            block_size,
447            512,
448            MemoryGpuConfig::ContextSize(max_seq_len),
449        )?),
450        (block_size, None, None, Some(ctxt), true, false) => Some(PagedAttentionConfig::new(
451            block_size,
452            512,
453            MemoryGpuConfig::ContextSize(ctxt),
454        )?),
455        (block_size, None, Some(f), None, true, false) => Some(PagedAttentionConfig::new(
456            block_size,
457            512,
458            MemoryGpuConfig::Utilization(f),
459        )?),
460        (block_size, Some(m), None, None, true, false) => Some(PagedAttentionConfig::new(
461            block_size,
462            512,
463            MemoryGpuConfig::MbAmount(m),
464        )?),
465        (block_size, Some(_m), Some(f), None, true, false) => {
466            info!("Both memory size, and usage were specified, defaulting to the usage value.");
467            Some(PagedAttentionConfig::new(
468                block_size,
469                512,
470                MemoryGpuConfig::Utilization(f),
471            )?)
472        }
473        (block_size, Some(_m), None, Some(ctxt), true, false) => {
474            info!("All memory size and ctxt len, defaulting to the context len value.");
475            Some(PagedAttentionConfig::new(
476                block_size,
477                512,
478                MemoryGpuConfig::ContextSize(ctxt),
479            )?)
480        }
481        (block_size, None, Some(f), Some(_ctxt), true, false) => {
482            info!("Both ctxt len and usage were specified, defaulting to the usage value.");
483            Some(PagedAttentionConfig::new(
484                block_size,
485                512,
486                MemoryGpuConfig::Utilization(f),
487            )?)
488        }
489        (_, _, _, _, _, _) => None,
490    };
491
492    let pipeline = loader.load_model_from_hf(
493        None,
494        token_source,
495        &dtype,
496        &device,
497        false,
498        mapper,
499        args.in_situ_quant,
500        cache_config,
501    )?;
502    info!("Model loaded.");
503
504    let scheduler_config = if cache_config.is_some() {
505        // Handle case where we may have device mapping
506        if let Some(ref cache_config) = pipeline.blocking_lock().get_metadata().cache_config {
507            SchedulerConfig::PagedAttentionMeta {
508                max_num_seqs: *args.concurrency.as_ref().unwrap().iter().max().unwrap(),
509                config: cache_config.clone(),
510            }
511        } else {
512            SchedulerConfig::DefaultScheduler {
513                method: DefaultSchedulerMethod::Fixed(
514                    (*args.concurrency.as_ref().unwrap().iter().max().unwrap())
515                        .try_into()
516                        .unwrap(),
517                ),
518            }
519        }
520    } else {
521        SchedulerConfig::DefaultScheduler {
522            method: DefaultSchedulerMethod::Fixed(
523                (*args.concurrency.as_ref().unwrap().iter().max().unwrap())
524                    .try_into()
525                    .unwrap(),
526            ),
527        }
528    };
529    let mistralrs = MistralRsBuilder::new(pipeline, scheduler_config, false, None)
530        .with_no_prefix_cache(true)
531        .with_disable_eos_stop(true)
532        .build();
533
534    info!("Starting warmup run.");
535    warmup_run(mistralrs.clone());
536    info!("Finished warmup run.");
537    info!("Starting benchmarks.");
538
539    for concurrency in args.concurrency.as_ref().unwrap() {
540        let mut results = vec![];
541        if args.n_gen > 0 {
542            let r = run_bench(
543                mistralrs.clone(),
544                RequestMessage::Completion {
545                    text: "Rust".to_string(),
546                    echo_prompt: false,
547                    best_of: None,
548                },
549                args.n_gen - 1,
550                *concurrency,
551                args.repetitions,
552                TestName::Gen(args.n_gen),
553            )?;
554            results.push(r);
555        }
556
557        if args.n_prompt > 0 {
558            let tks = (1000..1000 + args.n_prompt as u32).collect();
559            let r = run_bench(
560                mistralrs.clone(),
561                RequestMessage::CompletionTokens(tks),
562                1,
563                *concurrency,
564                args.repetitions,
565                TestName::Prompt(args.n_prompt),
566            )?;
567
568            results.push(r);
569        }
570
571        print_usage(&model_name, &device, results);
572    }
573
574    Ok(())
575}
